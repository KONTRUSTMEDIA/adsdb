    <script>
        function copyLink() {
            navigator.clipboard.writeText(window.location.href);
            alert("Link copied to clipboard!");
        }

        function cleanAndFormat(content) {
            if (!content) return "";
            let cleaned = content.replace(/watch video|online link/gi, '');
            const ytRegex = /(?:https?:\/\/)?(?:www\.)?(?:youtube\.com\/watch\?v=|youtu\.be\/|youtube\.com\/embed\/)([a-zA-Z0-9_-]{11})/g;
            
            return cleaned.replace(ytRegex, (match, videoId) => {
                const thumb = `https://img.youtube.com/vi/${videoId}/hqdefault.jpg`;
                return `
                    <div class="video-container-wrapper">
                        <div class="video-link-card" onclick="openVideo('${videoId}')" style="cursor:pointer;">
                            <div class="video-thumbnail" style="background-image: url('${thumb}')">
                                <div class="play-overlay"><div class="play-button-icon">▶</div><span>WATCH TUTORIAL</span></div>
                            </div>
                        </div>
                    </div>`;
            });
        }

        function openVideo(videoId) {
            const overlay = document.createElement('div');
            overlay.id = 'video-lightbox';
            overlay.innerHTML = `<div class="lightbox-content"><span class="close-video" onclick="closeVideo()">&times;</span><iframe width="100%" height="100%" src="https://www.youtube.com/embed/${videoId}?autoplay=1" frameborder="0" allow="autoplay; encrypted-media" allowfullscreen></iframe></div>`;
            document.body.appendChild(overlay);
        }

        function closeVideo() { document.getElementById('video-lightbox')?.remove(); }

        async function loadPost() {
            try {
                const params = new URLSearchParams(window.location.search);
                let slug = params.get('slug')?.trim();
                
                if (!slug) {
                    const pathParts = window.location.pathname.split('/').filter(p => p);
                    slug = pathParts[pathParts.length - 1]; 
                }

                const response = await fetch('/all_posts.json'); 
                const posts = await response.json();
                const post = posts.find(p => p.slug === slug) || posts.find(p => p.slug?.toLowerCase() === slug?.toLowerCase());

                if (post) {
                    document.title = post.title + " | Cheptiony";
                    
                    // SEO: SET CANONICAL LINK
                    let canonical = document.querySelector('link[rel="canonical"]');
                    if (!canonical) {
                        canonical = document.createElement('link');
                        canonical.rel = 'canonical';
                        document.head.appendChild(canonical);
                    }
                    canonical.href = `https://cheptiony.com/${post.slug}`;

                    const displayArea = document.getElementById('post-display');
                    displayArea.innerHTML = `
                        <h1 class="post-title">${post.title}</h1>
                        <div class="blog-body-text">${cleanAndFormat(post.content)}</div>
                        <div class="techtube-cta">
                            <h3>Scale Your Brand with Professional Video</h3>
                            <a href="https://techtubestudio.com" target="_blank" class="btn-cta">Visit Techtube Studio</a>
                        </div>`;

                    // Generate TOC
                    const headings = displayArea.querySelectorAll('h2, h3');
                    if (headings.length > 2) {
                        let toc = '<div class="toc-box"><strong>In this article:</strong><ul>';
                        headings.forEach((h, i) => { h.id = 'section-' + i; toc += `<li><a href="#section-${i}">${h.innerText}</a></li>`; });
                        toc += '</ul></div>';
                        document.getElementById('toc-container').innerHTML = toc;
                    }

                    // Related Posts
                    const related = posts.filter(p => p.slug !== slug).sort(() => 0.5 - Math.random()).slice(0, 2);
                    document.getElementById('related-posts').innerHTML = related.map(p => `
                        <a href="/${p.slug}" class="grid-card">
                            <div class="grid-image" style="background-image: url('${p.featured_image}')"></div>
                            <div class="grid-body"><small>${p.date}</small><h3>${p.title}</h3><span class="grid-link">Read More →</span></div>
                        </a>`).join('');

                    // Trigger AdSense for dynamic content if units are present
                    (window.adsbygoogle = window.adsbygoogle || []).push({});

                } else {
                    window.location.href = '/blog';
                }
            } catch (e) { console.error(e); }
        }
        loadPost();
    </script>
