<script>
  /* Navbar */
  let recalculateNavbarHeight = null;

  const computeVisibleHeightFromRect = (r) => {
    const visible = Math.min(r.bottom, window.innerHeight) - Math.max(r.top, 0);
    return Math.max(0, Math.min(r.height, visible));
  };

  const getBannerHeight = () => {
    const banner = document.querySelector(".nav--banner-wrapper");
    if (
      !banner ||
      document.documentElement.classList.contains("hide-nav-banner")
    )
      return 0;

    const r = banner.getBoundingClientRect();
    return computeVisibleHeightFromRect(r);
  };

  const getUtilityHeight = () => {
    const utility = document.querySelector(".nav--utility-wrapper");
    if (!utility) return 0;

    const r = utility.getBoundingClientRect();
    return computeVisibleHeightFromRect(r);
  };

  const getNavWrapperHeight = () => {
    const navWrapper = document.querySelector(".nav--wrapper");
    if (!navWrapper) return 0;
    const r = navWrapper.getBoundingClientRect();
    return computeVisibleHeightFromRect(r);
  };

  const getNavbarHeight = () => {
    // Sum the visible portions of each section
    return (
      (getBannerHeight() || 0) +
      (getUtilityHeight() || 0) +
      (getNavWrapperHeight() || 0)
    );
  };

  const isMobile = () => window.innerWidth < 992;

  const computeMenuOffset = () => {
    // On mobile, when menu is open:
    // - past navbar: only the wrapper is visible at the top
    // - near top: banner + utility + wrapper are visible
    if (isMobile() && isMenuOpen()) {
      if (isPastNavbar()) {
        return getNavWrapperHeight() || 0;
      }
      return getBannerHeight() + getUtilityHeight() + getNavWrapperHeight();
    }
    // Default
    return getNavbarHeight();
  };

  const setMenuOffset = () => {
    // Keeps the mobile overlay positioned below the visible navbar stack
    const offset = Math.round(computeMenuOffset());
    document.body.style.setProperty("--menu-offset", offset + "px");
  };

  const getScrollTop = () =>
    window.pageYOffset || document.documentElement.scrollTop;

  let lastKnownScrollTop = 0;
  let __navScrollingUp = false;

  const isPastNavbar = () => {
    const navbar = document.querySelector(".navbar_component");
    if (!navbar) return getScrollTop() > 0;
    const rect = navbar.getBoundingClientRect();
    return rect.bottom <= 10;
  };

  const isMenuOpen = () => {
    const overlay = document.querySelector(".w-nav-overlay");
    const menu = document.querySelector(".w-nav-menu");
    const button = document.querySelector(".w-nav-button, .navbar_menu-button");
    const overlayVisible =
      !!overlay &&
      getComputedStyle(overlay).display !== "none" &&
      (overlay.clientHeight > 0 || overlay.clientWidth > 0);
    const menuOpen = !!document.querySelector(".w-nav-menu.w--open");
    const buttonOpen = !!document.querySelector(
      ".w-nav-button.w--open, .navbar_menu-button.w--open"
    );
    return overlayVisible || menuOpen || buttonOpen;
  };

  const setMobileNavTop = (navWrapper, scrollTop) => {
    if (!isMobile()) {
      navWrapper.style.removeProperty("top");
      return;
    }
    navWrapper.style.top = "0px";
  };

  const updateNavOverlayOffset = () => {
    const overlay = document.querySelector(".w-nav-overlay");
    if (!overlay) return;
    const atTop = getScrollTop() <= 10;
    if (isMobile() && isMenuOpen() && atTop) {
      overlay.style.transform = "translateY(-2px)";
    } else {
      overlay.style.removeProperty("transform");
    }
  };

  const clearNavInlinePositionalStyles = (navWrapper) => {
    if (!navWrapper) return;
    ["transform", "opacity", "visibility", "display", "top", "z-index"].forEach(
      (prop) => navWrapper.style.removeProperty(prop)
    );
  };

  const showNav = (navWrapper) => {
    navWrapper.classList.add("nav-fixed", "nav-visible");
    navWrapper.classList.remove("nav-hidden");
    document.body.classList.add("nav-is-fixed");
    const navHeight = getNavWrapperHeight();

    document.body.style.setProperty("--nav-offset", navHeight + extra + "px");

    // Ensure inline overrides are cleared so CSS transitions can run

    function makeItemActive() {
      // Pause all other videos
      Object.keys(players).forEach(key => {
        if (players[key] && key != activeIndex) {
          players[key].pause().catch(() => {});
        }
      });

      popupItems.removeClass("is-active");
      let activePopup = popupItems.eq(activeIndex).addClass("is-active");

      // If player already exists for this index, just play it
      if (players[activeIndex]) {
        players[activeIndex].setCurrentTime(0).then(() => {
          return players[activeIndex].play();
        }).catch(() => {
          console.warn('Vimeo autoplay blocked - user interaction required');
        });
        
        let announcement = `Video ${activeIndex + 1} of ${cardItems.length}`;
        section.find(".sr-announcer").text(announcement);
        return;
      }

      // Otherwise, create new player
      let iframe = activePopup.find("iframe")[0];
      if (!iframe) {
        console.warn('No iframe found in popup item', activeIndex);
        return;
      }

      // Initialise Vimeo player
      players[activeIndex] = new Vimeo.Player(iframe);
      players[activeIndex].ready().then(() => {
        players[activeIndex].setCurrentTime(0);
        players[activeIndex].play().catch(() => {
          console.warn('Vimeo autoplay blocked - user interaction required');
        });

        // Announce to screen readers
        let announcement = `Video ${activeIndex + 1} of ${cardItems.length}`;
        section.find(".sr-announcer").text(announcement);
      }).catch((err) => {
        console.error('Vimeo player initialisation failed:', err);
      });
    }

    function trapFocus(e) {
      if (!modalOpen || e.key !== 'Tab') return;

      let focusable = popup.find('a, button, textarea, input, select, [tabindex]:not([tabindex="-1"])').filter(':visible');
      if (focusable.length === 0) return;

      let first = focusable.first()[0];
      let last = focusable.last()[0];

      if (e.shiftKey && document.activeElement === first) {
        e.preventDefault();
        last.focus();
      } else if (!e.shiftKey && document.activeElement === last) {
        e.preventDefault();
        first.focus();
      }
    }

    function closeModal() {
      modalOpen = false;
      
      // Just pause the active player - DON'T destroy
      if (players[activeIndex]) {
        players[activeIndex].pause().catch(() => {});
      }
      
      tl.reverse();
      $("body").css("overflow", "");
      $("body > *:not(.popup_wrap)").removeAttr("aria-hidden");
      
      setTimeout(() => {
        cardItems.eq(activeIndex).focus();
      }, 350);
    }

    function handleKeydown(e) {
      if (!modalOpen) return;

      trapFocus(e);

      if (e.key === "Escape" || e.key === "Esc") {
        e.preventDefault();
        closeModal();
      } else if (e.key === "ArrowLeft") {
        e.preventDefault();
        prev.trigger("click");
      } else if (e.key === "ArrowRight") {
        e.preventDefault();
        next.trigger("click");
      }
    }

    // Open modal on card click
    cardItems.each(function (index) {
      $(this).on("click", function () {
        activeIndex = index;
        tl.play();
        makeItemActive();
        modalOpen = true;
        $("body").css("overflow", "hidden");
        $("body > *:not(.popup_wrap)").attr("aria-hidden", "true");

        setTimeout(() => {
          closeBtn.first().focus();
        }, 350);
      });
    });

    // Previous button
    prev.on("click", function () {
      activeIndex = activeIndex > 0 ? activeIndex - 1 : cardItems.length - 1;
      makeItemActive();
    });

    // Next button
    next.on("click", function () {
      activeIndex = activeIndex < cardItems.length - 1 ? activeIndex + 1 : 0;
      makeItemActive();
    });

    // Close button and background
    closeBtn.on("click", closeModal);
    closeBg.on("click", closeModal);

    // Keyboard navigation
    window.addEventListener("keydown", handleKeydown);

    // Cleanup on unload
    window.addEventListener("beforeunload", () => {
      window.removeEventListener("keydown", handleKeydown);
      Object.keys(players).forEach(key => {
        if (players[key]) players[key].destroy().catch(() => {});
      });
    });
  });
});
</script>

<script>
  const NavConflictConfig = {
    tabletBreakpoint: 991,
    debounceTime: 100,
    // Targeting your specific wrapper from the screenshot
    wrapperSelector: '.nav--locales-wrapper' 
  };
  
  let navResizeTimeout;
  
  function updateNavStopPropagation() {
    document.querySelectorAll(NavConflictConfig.wrapperSelector).forEach(wrapper => {
      wrapper.removeEventListener('click', handleNavClick, false);
      if (window.innerWidth <= NavConflictConfig.tabletBreakpoint) {
        // We use 'false' to allow the click to reach the toggle first
        wrapper.addEventListener('click', handleNavClick, false);
      }
    });
  }
  
  function handleNavClick(event) {
    // 1. Check if click is on the toggle (the flag button)
    const localeToggle = event.target.closest('.nav--locale-dropdown-toggle');
    const menuButton = document.querySelector('.navbar_menu-button');
    
    if (localeToggle && menuButton && menuButton.classList.contains('w--open')) {
      // Let Webflow handle the dropdown opening first
      setTimeout(() => {
        // Then stop the signal that tells the Hamburger menu to close
        event.stopPropagation();
      }, 0);
      
      // Stop immediate bubbling to the Navbar component
      event.stopPropagation();
      return;
    }
    
    // 2. Prevent clicks inside the open list from closing the menu
    if (event.target.closest('.nav--locale-dropdown-list')) {
      // If they clicked a link to change language, don't stop it
      if (event.target.tagName === 'A' || event.target.closest('a')) {
        return;
      }
      event.stopPropagation();
    }
  }
  
  // Initialize
  document.addEventListener('DOMContentLoaded', updateNavStopPropagation);
  
  // Re-check on resize
  window.addEventListener('resize', () => {
    clearTimeout(navResizeTimeout);
    navResizeTimeout = setTimeout(updateNavStopPropagation, NavConflictConfig.debounceTime);
  });
</script></body></html>
