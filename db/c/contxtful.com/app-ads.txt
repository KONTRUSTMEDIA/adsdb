  // ---- Helpers: header offset + smooth scroll + close Elementor menu ----
  function headerOffset() {
    let h = 0;
    document.querySelectorAll('header, .elementor-location-header, .elementor-sticky--active').forEach(el => {
      const cs = getComputedStyle(el);
      if ((cs.position === 'fixed' || cs.position === 'sticky') && parseInt(cs.top) === 0 && el.offsetHeight > 0) {
        h = Math.max(h, el.offsetHeight);
      }
    });
    return h;
  }
  function smoothScrollTo(el) {
    if (!el) return;
    const y = el.getBoundingClientRect().top + window.pageYOffset - (headerOffset() + 12);
    window.scrollTo({ top: y, behavior: 'smooth' });
  }
  function closeElementorMenuIfOpen(triggerInside) {
    const wrapper = triggerInside?.closest('.e-n-menu-wrapper');
    if (!wrapper) return;
    const toggleId = wrapper.getAttribute('aria-labelledby');
    const toggle = toggleId ? document.getElementById(toggleId) : null;
    if (toggle && toggle.getAttribute('aria-expanded') === 'true') toggle.click();

        panels: Array.from(root.querySelectorAll('.rec-panel')),
        itemByTarget: new Map(),

        if (!t) {

          if (ns && ns.classList.contains('rec-mobile-panel') && ns.id) {
            t = ns.id.replace(/-mobile$/, '');

            t = state.panels[i].id;

            const word = item.querySelector('.rec-word')?.textContent?.trim().toLowerCase();
            if (word) t = word.replace(/\s+/g,'-');

        const target = root.querySelector('#' + CSS.escape(id)) || document.getElementById(id);
        if (!target) return;

        const visible = root.querySelector('.rec-panel.is-active');
        if (visible && visible !== target) {
          visible.classList.remove('is-active','animating','animate-from-right','animate-from-left');
          visible.hidden = true;
          visible.setAttribute('aria-hidden','true');
        }

        target.hidden = false;
        target.setAttribute('aria-hidden','false');
        target.classList.remove('animating','animate-from-right','animate-from-left');
        if (direction === 'right') target.classList.add('animate-from-right');
        if (direction === 'left')  target.classList.add('animate-from-left');
        target.classList.add('is-active');

        // trigger transition
        void target.offsetWidth;
        target.classList.add('animating');
        setTimeout(() => target.classList.remove('animating','animate-from-right','animate-from-left'), 400);
      }

      function toggleMobilePanel(item) {
        if (!window.matchMedia('(max-width: 767px)').matches) return;
        const panel = item.nextElementSibling?.classList.contains('rec-mobile-panel') ? item.nextElementSibling : null;
        if (!panel) return;

        // Close others
        root.querySelectorAll('.rec-mobile-panel.open').forEach(p => {
          if (p !== panel) {
            p.style.height = '0px';
            p.classList.remove('open');
            p.setAttribute('aria-hidden','true');
          }
        });
        root.querySelectorAll('.rec-for-item.is-open').forEach(i => {
          if (i !== item) {

      root.__recFor = { activateByKey };
    });

      ensureRecApi();
      handleHashTarget(location.hash);
    });

})();
