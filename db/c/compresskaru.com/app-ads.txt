	async function compressMBPDF(pdfData, targetSizeMB) {
      try {
        var pdf = await pdfjsLib.getDocument({ data: pdfData }).promise;

        // Initial compression ratio
        var compressionRatio = 5;

        // Maximum number of iterations to avoid infinite loop
        var maxIterations = 100;
        // Iterate until the target size is reached or the maximum iterations are reached
        for (var i = 0; i < maxIterations; i++) {
          var compressedPdf = new window.jspdf.jsPDF();

          for (var pageNum = 1; pageNum <= pdf.numPages; pageNum++) {
            var page = await pdf.getPage(pageNum);

            var scale = 2.5;
            var viewport = page.getViewport({ scale });
            var canvas = document.createElement('canvas');
            var context = canvas.getContext('2d');
            canvas.height = viewport.height;
            canvas.width = viewport.width;
			
            var renderContext = {
              canvasContext: context,
              viewport: viewport
            };

            await page.render(renderContext).promise;

            var imageData = canvas.toDataURL("image/jpeg", compressionRatio);
            compressedPdf.addImage(imageData, 'JPEG', 0, 0, 210, 297);

            if (pageNum !== pdf.numPages) {
              compressedPdf.addPage();
            }
          }

          // Calculate the size of the compressed PDF
          var compressedSizeMB = compressedPdf.output('blob').size / 1024; // Convert bytes to kilobytes

          // Check if the compressed size is within the target range
          if (compressedSizeMB <= targetSizeMB) {
            compressedPdf.save('compresskaru.pdf');
            displayResult(targetSizeMB, compressedSizeMB);
            break;
          }

          // Adjust the compression ratio for the next iteration
          compressionRatio *= (targetSizeMB / compressedSizeMB);

          if (i === maxIterations - 1) {
            console.log('Maximum iterations reached. Compression may not have achieved the target size.');
            displayResult(targetSizeMB, compressedSizeMB);
          }
        }
      } catch (error) {
        console.error('Error compressing PDF:', error);
      }
    }

	async function compressPDF(pdfData, targetSizeKB) {
      try {
        var pdf = await pdfjsLib.getDocument({ data: pdfData }).promise;

        // Initial compression ratio
        var compressionRatio = 5;

        // Maximum number of iterations to avoid infinite loop
        var maxIterations = 100;
        // Iterate until the target size is reached or the maximum iterations are reached
        for (var i = 0; i < maxIterations; i++) {
          var compressedPdf = new window.jspdf.jsPDF();

          for (var pageNum = 1; pageNum <= pdf.numPages; pageNum++) {
            var page = await pdf.getPage(pageNum);

            var scale = 1.5;
            var viewport = page.getViewport({ scale });
            var canvas = document.createElement('canvas');
            var context = canvas.getContext('2d');
            canvas.height = viewport.height;
            canvas.width = viewport.width;
			
            var renderContext = {
              canvasContext: context,
              viewport: viewport
            };

            await page.render(renderContext).promise;

            var imageData = canvas.toDataURL("image/jpeg", compressionRatio);
            compressedPdf.addImage(imageData, 'JPEG', 0, 0, 210, 297);

            if (pageNum !== pdf.numPages) {
              compressedPdf.addPage();
            }
          }

          // Calculate the size of the compressed PDF
          var compressedSizeKB = compressedPdf.output('blob').size / 1024; // Convert bytes to kilobytes

          // Check if the compressed size is within the target range
          if (compressedSizeKB <= targetSizeKB) {
            compressedPdf.save('2ndpdf.pdf');
            displayResult(targetSizeKB, compressedSizeKB);
            break;
          }

          // Adjust the compression ratio for the next iteration
          compressionRatio *= (targetSizeKB / compressedSizeKB);

          if (i === maxIterations - 1) {
            console.log('Maximum iterations reached. Compression may not have achieved the target size.');
            displayResult(targetSizeKB, compressedSizeKB);
          }
        }
      } catch (error) {
        console.error('Error compressing PDF:', error);
      }
    }
