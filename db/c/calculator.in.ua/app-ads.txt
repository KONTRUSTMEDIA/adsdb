function device() {
	if ($(window).width() >= '992') {
		
		$('[data-toggle=tooltip-lg]').tooltip({
			html: true
		});


		$('#navbarCollapse .dropdown').hover(
			function () {
				$(this).addClass('show').find('.dropdown-menu').addClass('show');
			},
			function () {
				$(this).removeClass('show').find('.dropdown-menu').removeClass('show');
			}
		);


		$('#navbarCollapse .nav-item.d-none').removeClass('d-none');
		while ($('#navbarCollapse .navbar-nav').offset().left - $('#navbarCollapse').offset().left < 100) {
			$('#navbarCollapse .nav-item').first().remove();
		}


		$('.menu-calculators .dropdown-item').addClass('dropdown-item-inline').removeClass('beta soon version');
		$('.menu-calculators .dropdown-header').addClass('cursor-pointer')
		.each(function() {
			let count = $(this).siblings('.dropdown-item').wrapAll('<div class="dropdown-item-list" />').length;
			$(this).append(' <small>(' + count + ')</small>')
		})
		.click(function(e) {
			e.preventDefault();
			$(this).parent().toggleClass('show');
		})

	} else {

		var menu = false;
		$(window).on('scroll', function() {
			if(menu) return;
			menu = true;
			$('#mobile-menu').css('display', '')
			.delay(100).queue(function(){
				$(this).addClass('show').dequeue();
			});
		})
		
	}
}

function toastShow(message, autohide = true, delay = 1500) {
	if (!message) return;
	$('#toast').toast({autohide: autohide, delay: delay}).toast('show').find('.toast-body').html(message);
}
function modalShow(data, modalClass = 'bg-nav') {
	if (!data.body) return;
	$('#alert .modal-content').removeClass().addClass('modal-content ' + modalClass);
	$('#alert .modal-title').html(data.title);
	$('#alert .modal-body').html(data.body);
	$('#alert .modal-footer').html(data.footer);
	$('#alert').modal('show').removeClass('changed').find('[data-toggle=tooltip]').tooltip({html: true});
}

function uniqid(prefix = '') {
	return prefix + Math.floor((1 + Math.random()) * 0x100000000).toString(16).substring(1);
}
function setLocation(url) {
	try {
		if (!/^https?:\/\//.test(url)) {
			url = location.protocol + '//' + location.host + location.pathname + (url ? ('?' + url) : '');
		}
		history.pushState(null, null, url);
		return;
	} catch(e) {}
	location.hash = '#' + url;
}
