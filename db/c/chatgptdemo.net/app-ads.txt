  async function handleAction(commentId, actionType) {
    var now = Date.now();
    if (!lastActionTime[commentId]) lastActionTime[commentId] = 0;
    if (now - lastActionTime[commentId] < ACTION_COOLDOWN) {
      alert("You are acting too fast! Please wait 2 seconds.");
      return;
    }
    lastActionTime[commentId] = now;

    var cmt = list_coment.find(x => x._id === commentId);
    var errorMsg = document.querySelector(`.comment-box[data-id="${commentId}"] .error-msg`);
    errorMsg.style.display = 'none';

    if (actionType === 'vote_up') {
      cmt.list_vote[user_._id] = 'up';
    } else if (actionType === 'vote_down') {
      cmt.list_vote[user_._id] = 'down';
    } else if (actionType === 'delvote_coment') {
      delete cmt.list_vote[user_._id];
    } else if (actionType === 'toggle_follow') {
      if (cmt.list_follow.includes(user_._id))
        cmt.list_follow = cmt.list_follow.filter(u => u !== user_._id);
      else cmt.list_follow.push(user_._id);
    }

    renderComments();

    let typApi = '';
    if (actionType === 'vote_up') typApi = 'upvote_coment';
    else if (actionType === 'vote_down') typApi = 'downvote_coment';
    else if (actionType === 'delvote_coment') typApi = 'delvote_coment';
    else if (actionType === 'toggle_follow') typApi = 'toggle_follow';
    else return;

    try {
      var res = await fetch('/api_data', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ _id: commentId, type_call: typApi })
      });
      if (!res.ok) throw new Error("fail");
    } catch (e) {
      errorMsg.textContent = '❌ Interaction failed';
      errorMsg.style.display = 'block';
    }
  }

  function addEventListeners() {
    document.querySelectorAll('.vote-up').forEach(btn => {
      btn.onclick = () => {
        var cmtId = btn.closest('.comment-box').dataset.id;
        var cmt = list_coment.find(x => x._id === cmtId);
        const current = cmt.list_vote[user_._id];
        if (current === 'up') handleAction(cmtId, 'delvote_coment');
        else handleAction(cmtId, 'vote_up');
      };
    });

    document.querySelectorAll('.vote-down').forEach(btn => {
      btn.onclick = () => {
        var cmtId = btn.closest('.comment-box').dataset.id;
        var cmt = list_coment.find(x => x._id === cmtId);
        const current = cmt.list_vote[user_._id];
        if (current === 'down') handleAction(cmtId, 'delvote_coment');
        else handleAction(cmtId, 'vote_down');
      };
    });

    document.querySelectorAll('.follow-btn').forEach(btn => {
      btn.onclick = () => {
        var cmtId = btn.closest('.comment-box').dataset.id;
        handleAction(cmtId, 'toggle_follow');
      };
    });
  }

            async function toggleVote(slug, itemType, voteType, btn) {
                const currentVote = btn.parentElement.dataset.userVote;
                const newVote = currentVote === voteType ? 'none' : voteType;

                try {
                    const res = await fetch('/api_data', {
                        method: 'POST',
                        headers: { 'Content-Type': 'application/json' },
                        body: JSON.stringify({ slug: slug, item_type: itemType, vote: newVote, type_call: 'user_vote' })
                    });
                    const result = await res.json();

                    if (res.status === 200) {
                        // Update UI
                        const container = btn.parentElement;
                        const oldVote = container.dataset.userVote;
                        container.dataset.userVote = newVote;

                        const upBtn = container.querySelector('.upvote-btn');
                        const downBtn = container.querySelector('.downvote-btn');
                        const upCountSpan = container.querySelector('.up-count');
                        const downCountSpan = container.querySelector('.down-count');

                        let upCount = parseInt(upCountSpan.innerText);
                        let downCount = parseInt(downCountSpan.innerText);

                        // Adjust counts
                        if (oldVote === 'up') upCount--;
                        if (oldVote === 'down') downCount--;
                        if (newVote === 'up') upCount++;
                        if (newVote === 'down') downCount++;

                        upCountSpan.innerText = upCount;
                        downCountSpan.innerText = downCount;

                        // Toggle classes
                        if (newVote === 'up') {
                            upBtn.classList.add('text-blue-500');
                            downBtn.classList.remove('text-red-500');
                        } else if (newVote === 'down') {
                            downBtn.classList.add('text-red-500');
                            upBtn.classList.remove('text-blue-500');
                        } else {
                            upBtn.classList.remove('text-blue-500');
                            downBtn.classList.remove('text-red-500');
                        }
                    } else if (res.status === 401 || res.status === 303) {
                        if (typeof Fancybox !== "undefined") {
                            Fancybox.show([{ src: "#modal-login", type: "inline" }]);
                        } else {
                            alert("Please login to vote.");
                        }
                    } else {
                        alert(result.message || "Action failed");
                    }
                } catch (e) {
                    console.error(e);
                    alert("Network error. Please try again.");
                }
            }
