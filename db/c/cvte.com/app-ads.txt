  // swiper API
  var swiper = new Swiper('.banner-swiper', {
    autoplay: {
      delay: 8000 // 自动轮播时间
    },
    speed: 1000,
    pagination: {
      el: '.swiper-pagination',
      clickable: true,
      renderBullet: function(index, className) {
        return `<div class='${className}'><i></i></div>`
      }
    },
    on: {
      init: function() {
        const activeBullet = document.querySelector('.swiper-pagination-bullet-active i')
        activeBullet.style.animation = 'progress 8s linear forwards'
      },
      slideChangeTransitionStart: function() {
        const bullets = document.querySelectorAll('.swiper-pagination-bullet i')
        bullets.forEach(bullet => {
          bullet.style.animation = 'none'
          bullet.style.width = '0%'
        })
      },
      slideChangeTransitionEnd: function() {
        const activeBullet = document.querySelector('.swiper-pagination-bullet-active i')
        activeBullet.style.animation = 'progress 8s linear forwards'
      }
    }
  });
    // 含视频的轮播：进入即重置并自动播放，播完后才允许切换；离开时暂停
    (function setupVideoSlideGuard() {
      const setLocked = function (locked) {
        swiper.allowTouchMove = !locked; // 禁止/允许手势滑动
        try {
          if (swiper.autoplay) {
            locked ? swiper.autoplay.stop() : swiper.autoplay.start();
          }
        } catch (e) {}
        // 仅禁用左右按钮；分页圆点保持可点击
        const nextBtn = document.querySelector('.section-1 .swiper-button .swiper-button-next') || document.querySelector('.section-1 .swiper-button-next');
        const prevBtn = document.querySelector('.section-1 .swiper-button .swiper-button-prev') || document.querySelector('.section-1 .swiper-button-prev');
        [nextBtn, prevBtn].forEach(function (el) {
          if (el) el.style.pointerEvents = locked ? 'none' : '';
        });
      };

      const getVideoInSlideByIndex = function (index) {
        const slide = swiper.slides && swiper.slides[index];
        return slide ? slide.querySelector('video') : null;
      };

      const attachGuard = function (videoEl) {
        if (!videoEl) return;
        setLocked(true);
        try { videoEl.currentTime = 0; } catch (e) {}
        try { videoEl.play && videoEl.play(); } catch (e) {}
        if (videoEl._onEndedHandler) {
          videoEl.removeEventListener('ended', videoEl._onEndedHandler);
        }
        const onEnded = function () {
          setLocked(false);
          const nextIndex = Math.min(swiper.slides.length - 1, swiper.activeIndex + 1);
          if (nextIndex !== swiper.activeIndex) {
            swiper.slideTo(nextIndex, 1000);
          }
          videoEl.removeEventListener('ended', onEnded);
          delete videoEl._onEndedHandler;
        };
        videoEl._onEndedHandler = onEnded;
        videoEl.addEventListener('ended', onEnded);
      };

      // 初次进入（若是视频页则直接播放并加锁）
      attachGuard(getVideoInSlideByIndex(swiper.activeIndex));

      // 离开含视频的 slide 时暂停
      swiper.on('slideChangeTransitionStart', function () {
        const prevVideo = getVideoInSlideByIndex(swiper.previousIndex);
        if (prevVideo) {
          try { prevVideo.pause(); } catch (e) {}
        }
      });

      // 进入含视频的 slide 时重置并播放，同时加锁；若不是视频则解锁
      swiper.on('slideChangeTransitionEnd', function () {
        const currVideo = getVideoInSlideByIndex(swiper.activeIndex);
        if (currVideo) {
          attachGuard(currVideo);
        } else {
          setLocked(false);
        }
      });
    })();
  // 产品中心

  const Productswiper = new Swiper('.productswiper', {
    speed: 800,
    spaceBetween: 15,
    slidesPerView: 1,
    pagination: {

      },
    navigation: {
      nextEl: '.swiper-button01 .swiper-button-next',
      prevEl: '.swiper-button01 .swiper-button-prev'
    }
  })

    pagination: {

        clickable: true
      },
    breakpoints: {
      1024: {
        slidesPerView: 3

    }
