// Função para corrigir domínios de e-mail
function correctSubscribeEmail() {
    const emailInput = document.getElementById('subscribe_email');
    let email = emailInput.value.trim().toLowerCase(); // Converte todo o e-mail para minúsculas

    // Verifica se há um '@' no e-mail
    if (email.includes('@')) {
        // Separa o nome do e-mail do domínio
        let [username, domain] = email.split('@');

        // Corrige domínios semelhantes a gmail
        if (domain.startsWith('gma')) {
            domain = 'gmail.com';
        }    
            
   // Corrige domínios semelhantes a hotmail
        else if (domain.startsWith('hotma')) {
            if (!domain.includes('.com')) {
                domain = 'hotmail.com';
            } 
            
          }     
          
        // Corrige domínios semelhantes a hotmail
        else if (domain.startsWith('outlo')) {
            if (!domain.includes('.com')) {
                domain = 'outlook.com';
            } 
            
          } 
           // Corrige domínios semelhantes a hotmail
        else if (domain.startsWith('yaho')) {
            if (!domain.includes('.com')) {
                domain = 'yahoo.com';
            } 
            
          } 
        // Permite adicionar '.br' ao final do domínio
        if (email.endsWith('.br') && !domain.endsWith('.br')) {
            domain += '.br';
        }

        emailInput.value = `${username}@${domain}`;
    }
}

// Adiciona um evento de input ao campo de e-mail
document.getElementById('subscribe_email').addEventListener('input', correctSubscribeEmail);

    $(element).parent(".view-more-parent").addClass("expanded");
    $(element).remove();
  } else if ($(element).hasClass("view-more")) {
    $(element).parent(".view-more-parent").addClass("expanded has-hide");

  }
}
