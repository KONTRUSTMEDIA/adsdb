                    const containerId = $container.attr("id");

                    const debugMode = $container.data("debug") === "true";
                    
                    this.log("Initializing container: " + containerId + " for event: " + eventId);
                    
                    // Store container reference
                    this.containers[containerId] = {

                    if (!container || container.initialized) {

                    container.initialized = true;
                    this.log("Starting widget initialization for: " + containerId);

                    this.checkMembershipStatus(containerId);

                // Check membership status via AJAX
                checkMembershipStatus: function(containerId) {
                    const container = this.containers[containerId];
                    const self = this;
                    
                    this.log("Checking membership status for: " + containerId);
                    
                    // Check for auth token in URL hash
                    let authToken = null;
                    const urlHash = window.location.hash;
                    if (urlHash.includes("hsm_token=")) {
                        const tokenFromUrl = urlHash.split("hsm_token=")[1];
                        if (tokenFromUrl) {
                            authToken = tokenFromUrl;
                            this.log("Auth token found in URL. Storing in sessionStorage.");
                            sessionStorage.setItem("hsm_auth_token", authToken);
                            // Clean the hash from the URL for a cleaner user experience
                            history.pushState("", document.title, window.location.pathname + window.location.search);
                        }
                    }
                    
                    // Check for token in sessionStorage
                    if (!authToken) {
                        const tokenFromStorage = sessionStorage.getItem("hsm_auth_token");
                        if (tokenFromStorage) {
                            authToken = tokenFromStorage;
                            this.log("Auth token found in sessionStorage.");
                        }
                    }
                    
                    const ajaxData = {
                        action: "hsm_get_display_status",
                        nonce: hsm_widget_config.nonce,
                        current_url: window.location.href,
                        event_id: container.eventId
                    };
                    
                    // Add auth token if available
                    if (authToken) {
                        ajaxData.hsm_auth_token = authToken;
                        this.log("Sending auth token with verification request.");
                    }
                    
                    $.ajax({
                        url: hsm_widget_config.ajax_url,
                        type: "POST",
                        timeout: 15000,
                        data: ajaxData,
                        success: function(response) {
                            self.log("Membership response received", response);
                            self.handleMembershipResponse(containerId, response);
                        },
                        error: function(xhr, status, error) {
                            self.log("Membership check failed", {xhr, status, error});
                            self.handleMembershipError(containerId, xhr, status, error);
                        }
                    });
                },
                
                // Handle membership verification response

                    const container = this.containers[containerId];
                    
                    if (response.success) {
                        // Clean up auth token from sessionStorage after successful verification
                        if (sessionStorage.getItem("hsm_auth_token")) {
                            sessionStorage.removeItem("hsm_auth_token");
                            this.log("Cleaned up auth token from sessionStorage after successful verification.");
                        }

                            this.showError(containerId, hsm_widget_config.strings.error_config);
                            return;
                        }
                        
                        if (!response.data.is_logged_in) {
                            this.showLoginPrompt(containerId, response.data.login_url || "#");
                            return;
                        }
                        
                        // User is logged in, determine membership status
                        const membershipStatus = response.data.is_member ? "member" : "non_member";
                        this.log("Membership status determined: " + membershipStatus);
                        
                        // Fetch event widget
                        this.fetchEventWidget(containerId, membershipStatus);
                        
                    } else {
                        const errorMsg = response.data && response.data.message ? 
                            response.data.message : 
                            hsm_widget_config.strings.error_generic;
                        this.showError(containerId, errorMsg);
                    }
                },
                
                // Handle membership verification errors
                handleMembershipError: function(containerId, xhr, status, error) {
                    const container = this.containers[containerId];
                    
                    // Retry logic for network errors
                    if (container.retryCount < container.maxRetries && (xhr.status === 0 || xhr.status >= 500)) {
                        container.retryCount++;
