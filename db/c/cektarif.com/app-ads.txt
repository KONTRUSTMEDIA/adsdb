function validasi_formjne() {
    var oke = true;
    $("#form_searchjne input").each(function() {
        if ($(this).val().length==0 || $(this).val()==0) {
            alert("Kota asal atau kota tujuan tidak valid!\nKetik kota, tunggu sampai dropdown muncul, lalu pilih kota.\nJika dropdown tetap tidak muncul, silahkan refresh dan ulangi kembali.");
            $(this).focus();
            oke = false;
            return false;
        }
    });
    if (oke) return 'oke';
}

function cek_tarifjne() {
  //alert("test");
  //return false;
    if (validasi_formjne()!="oke") {
        return false;
    }

    var ser_data = $("#form_searchjne input,#form_searchjne textarea").serialize();
    
    $.ajax({
        type: "POST",
        url: "https://cektarif.com/exp/jne/jne.tarif.php",
        data: ser_data,
        success: function(msg){
            $("#btnCheckjne").show();
            $("#btnSearchjne").addClass("hidden");
            $("#form_searchjne input").removeAttr("disabled");
                
            var err = msg.search(/error/i);
            if (err >= 0) {
                //alert(msg);
                alert("Ada kesalahan, silahkan refresh browser dan ulangi kembali.");
            } else {
                $("#tarifresult").html(msg);
                $("html, body").animate({
                        //scrollTop: $("#tarifresult").offset().top
                        scrollTop: $("#form_resultjne").offset().top
                    }, 1000);

                //window.location.hash="#tarifresult";
            }
        }, beforeSend : function(a) {
            $("#btnCheckjne").hide();
            $("#btnSearchjne").removeClass("hidden");
            $("#form_searchjne input").attr("disabled","disabled");
        }, complete : function(a) {
            //completed
            //grecaptcha.reset(widgetJne);
        }, error: function (xhr, ajaxOptions, thrownError) {
            $("#btnCheckjne").show();
            $("#btnSearchjne").addClass("hidden");
            $("#form_searchjne input").removeAttr("disabled");
                alert("Ada kesalahan, silahkan refresh browser dan ulangi kembali.");
            //alert(xhr.responseText);
        }
    });
}
