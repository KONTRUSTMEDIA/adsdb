<script type="text/javascript">
function addFav(){
  if (document.all){ window.external.AddFavorite(location.href, document.title); }
  else { alert('Vous pouvez faire CTRL + D pour ajouter cette page dans vos signets, ou favoris.'); }
}

var calc_op = '';
var calc_buf = 0;
var calc_new = false;
var calc_buf2 = 0;
var expr_str = '0';

function setEcran(v){ document.getElementById('ecran').innerHTML = v; }
function getEcran(){ return document.getElementById('ecran').innerHTML; }

function pushTape(line, value){
  var ul=document.getElementById('tape'); if(!ul) return;
  var li=document.createElement('li');
  li.innerHTML=line; li.setAttribute('data-value', value);
  li.onclick=function(){ setEcran(this.getAttribute('data-value')); };
  ul.insertBefore(li, ul.firstChild);
  var items=ul.querySelectorAll('li'); if(items.length>12) ul.removeChild(ul.lastChild);
}

function executeOperation(op,cur){
  if(calc_new) cur = calc_buf2; else calc_buf2 = cur;
  calc_new = true;
  if(calc_op == '+')      calc_buf += parseFloat(cur);
  else if(calc_op == '-') calc_buf -= parseFloat(cur);
  else if(calc_op == '×') calc_buf *= parseFloat(cur);
  else if(calc_op == '÷') calc_buf /= parseFloat(cur);
  else                    calc_buf  = parseFloat(cur);
  setEcran(calc_buf);
  if(op != '=') calc_op = op;

  if(op==='='){
    pushTape(expr_str + ' = <strong>'+calc_buf+'</strong>', calc_buf);
    expr_str = ''+calc_buf;
  }
}

function btPress(obj){
  var cur = getEcran();
  var parts = obj.name.split('_');

  if(parts[0] == 'digit'){
    if(cur.length > 20) return;
    if(calc_new){ calc_new = false; setEcran(parts[1]); expr_str = parts[1]; }
    else if(cur == '0'){ setEcran(parts[1]); expr_str = (expr_str==='0')? parts[1] : expr_str+parts[1]; }
    else { setEcran(cur + parts[1]); expr_str += parts[1]; }
  }
  else if (parts[0] == 'op'){
    if(calc_new && calc_op != '=') calc_op = obj.value;
    else executeOperation(obj.value,cur);
    if(parts[1]!=='equal'){ expr_str += ' '+obj.value+' '; }
  }
  else if (parts[0] == 'special'){
    if(parts[1] == 'plusMinus'){
      if(cur.length > 20) return;
      setEcran(parseFloat(cur) * -1);
    }
    else if(parts[1] == 'oneOn'){
      if(parseFloat(cur) == 0){ alert('Division par 0 non autorisée'); return; }
      setEcran(1/parseFloat(cur)); expr_str='1/('+cur+')';
    }
    else if(parts[1] == 'dot'){
      if(cur.length > 20) return;
      if(calc_new){ setEcran('0.'); calc_new = false; expr_str='0.'; }
      else if(cur.indexOf('.') == -1){ setEcran(cur + '.'); expr_str+='.'; }
    }
    else if(parts[1] == 'pourcent'){ executeOperation('=', parseFloat(cur) / 100); expr_str='('+cur+')%'; }
    else if(parts[1] == 'square'){
      if(parseFloat(cur)<0){ alert('Racine pour une valeur négative non autorisée'); return; }
      setEcran(Math.sqrt(parseFloat(cur))); expr_str='√('+cur+')';
    }
    else if(parts[1] == 'C'){
      calc_buf = 0; calc_buf2 = 0; calc_op = '';
      setEcran('0'); calc_new = false; expr_str='0';
    }
    else if(parts[1] == 'CE'){
      setEcran('0'); calc_new = true; expr_str='0';
    }
    else if(parts[1] == 'equal'){ executeOperation('=', cur); }
    else if(parts[1] == 'back'){
      if(cur.length == 1) setEcran('0'); else setEcran(cur.substr(0, cur.length-1));
      expr_str = getEcran();
    }
    else if(parts[1] == 'copy'){
      var txt = (document.getElementById('ecran').innerText || document.getElementById('ecran').textContent);
      if (navigator.clipboard && navigator.clipboard.writeText) {
        navigator.clipboard.writeText(txt).then(function(){
          var live = document.getElementById('calc-live');
          if(live){ live.innerHTML = 'Résultat copié'; }
        });
      }
    }
    else if(parts[1]==='share'){
      var txt='Résultat: '+getEcran();
      if(navigator.share){ navigator.share({title:'Calculatrice', text:txt, url:location.href}); }
      else{ if(navigator.clipboard){ navigator.clipboard.writeText(txt); alert('Texte copié : '+txt); } }
    }
    else if(parts[1]==='theme'){ document.body.classList.toggle('dark'); localStorage.setItem('theme', document.body.classList.contains('dark')?'dark':''); }
    else if(parts[1]==='voice'){ startVoice(); }
    else if(parts[1]==='tapeToggle'){ document.getElementById('tape-wrap').classList.toggle('open'); }
  }
  obj.blur();
}

var keyPress = function (e){
  e = e || window.event;
  var k = e.keyCode || e.which;
  switch(k){
    case 13: btPress(document.fCalc.special_equal); break;
    case 27: btPress(document.fCalc.special_CE); break;
    case 46: btPress(document.fCalc.special_back); break;
    case 48: case 96: btPress(document.fCalc.digit_0); break;
    case 49: case 97: btPress(document.fCalc.digit_1); break;
    case 50: case 98: btPress(document.fCalc.digit_2); break;
    case 51: case 99: btPress(document.fCalc.digit_3); break;
    case 52: case 100: btPress(document.fCalc.digit_4); break;
    case 53: case 101: btPress(document.fCalc.digit_5); break;
    case 54: case 102: btPress(document.fCalc.digit_6); break;
    case 55: case 103: btPress(document.fCalc.digit_7); break;
    case 56: case 104: btPress(document.fCalc.digit_8); break;
    case 57: case 105: btPress(document.fCalc.digit_9); break;
    case 106: btPress(document.fCalc.op_times); break;
    case 107: btPress(document.fCalc.op_plus); break;
    case 109: btPress(document.fCalc.op_minus); break;
    case 110: btPress(document.fCalc.special_dot); break;
    case 111: btPress(document.fCalc.op_divide); break;
    default: break;
  }
}

function keyPressBackspace(e){
  e = e || window.event;
  var k = e.keyCode || e.which;
  if(k == 8){
    if(e.type=='keydown') btPress(document.fCalc.special_back);
    if(e.preventDefault){
      e.preventDefault(); e.stopPropagation(); e.cancelBubble = true;
    } else { e.cancelBubble = true; }
    return false;
  }
  return true;
}

function startVoice(){
  var SR = window.SpeechRecognition || window.webkitSpeechRecognition;
  if(!SR){ alert('Dictée non prise en charge par ce navigateur.'); return; }
  var r = new SR(); r.lang='fr-FR'; r.interimResults=false; r.maxAlternatives=1;
  r.onresult=function(ev){
    var t = ev.results[0][0].transcript.toLowerCase();
    var map = {'plus':'+','moins':'-','fois':'×','multiplié par':'×','divisé par':'÷','sur':'÷','virgule':'.','point':'.','égale':'='};
    Object.keys(map).forEach(function(k){ t = t.replace(new RegExp('\\b'+k+'\\b','g'), map[k]); });
    var dmap={'zéro':'0','zero':'0','un':'1','deux':'2','trois':'3','quatre':'4','cinq':'5','six':'6','sept':'7','huit':'8','neuf':'9'};
    Object.keys(dmap).forEach(function(k){ t = t.replace(new RegExp('\\b'+k+'\\b','g'), dmap[k]); });
    t = t.replace(/\s+/g,'');
    for(var i=0;i<t.length;i++){
      var ch=t[i];
      if(/\d/.test(ch))      btPress({name:'digit_'+ch, blur:function(){}});
      else if(ch=='.')       btPress({name:'special_dot', blur:function(){}});
      else if(ch=='+')       btPress({name:'op_plus', value:'+', blur:function(){}});
      else if(ch=='-')       btPress({name:'op_minus', value:'-', blur:function(){}});
      else if(ch=='×' || ch=='*') btPress({name:'op_times', value:'×', blur:function(){}});
      else if(ch=='÷' || ch=='/') btPress({name:'op_divide', value:'÷', blur:function(){}});
      else if(ch=='=')       btPress({name:'special_equal', blur:function(){}});
    }
  };
  r.start();
}

function loadKeyboard(){
  document.onkeydown = keyPressBackspace;
  document.onkeypress = keyPressBackspace;
  document.onkeyup = keyPress;
  if(localStorage.getItem('theme')==='dark') document.body.classList.add('dark');

  // mini calc show/hide + sync
  window.addEventListener('scroll', function(){
    var m=document.getElementById('mini-calc'); if(!m) return;
    if(window.scrollY>350) m.classList.add('show'); else m.classList.remove('show');
  });
  document.addEventListener('keyup', function(){ var m=document.getElementById('mini-ecran'); if(m) m.innerHTML=getEcran(); });
  document.addEventListener('click', function(){ var m=document.getElementById('mini-ecran'); if(m) m.innerHTML=getEcran(); }, true);
}
