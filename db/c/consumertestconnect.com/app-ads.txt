            console.log(e);
        }
    })();

    window.addEventListener("load", (event) => {
        const domain = window.location.hostname;
        console.log(domain);

        try {
            const termsLinks = document.querySelectorAll(".terms-and-conditions-link");
            const privacyLinks = document.querySelectorAll(".privacy-policy-link");

            // Define site configurations
            const siteConfigs = {

                    domains: ['consumertestconnect1', 'consumertestconnect2', 'consumertestconnect3','consumertestconnect.com','consumertestconnection.com','consumetestconnect.com'],

                    termsPath: '/terms-conditions',
                    privacyPath: '/privacy-policy',

                },
                rewardspark: {
                    domains: ['rewardspark'],
                    name: 'RewardSpark',
                    termsPath: '/rewards-terms-conditions',
                    privacyPath: '/rewards-privacy-policy',
support: 'support@rewardspark.co'
                }
            };

            // Find matching site configuration
            const activeSite = Object.values(siteConfigs).find(site =>
                site.domains.some(d => domain.includes(d))
            );

            if (activeSite) {
                // Update Twyne links
                termsLinks.forEach(link => {
                    link.href = `https://${domain}${activeSite.termsPath}`;
                });
                privacyLinks.forEach(link => {
                    link.href = `https://${domain}${activeSite.privacyPath}`;
                });

                // Update specific elements if they exist
                document.querySelectorAll('#gc_domainname').forEach(el => el.textContent = activeSite.name);
                const privacyElement = document.getElementById('gc_privacy');
                const termsElement = document.getElementById('gc_terms');
document.querySelectorAll('#gc_support').forEach(el => el.textContent = activeSite.support);

                if (privacyElement) {
                    privacyElement.setAttribute('href', activeSite.privacyPath);
                }
                if (termsElement) {
                    termsElement.setAttribute('href', activeSite.termsPath);
                }

                // Update PATH page links
                updateLink('a[href="/privacy-policy"]', activeSite.privacyPath);
                updateLink('a[href="/terms-conditions"]', activeSite.termsPath);

console.log('active site: ' +activeSite.name);
if(activeSite.name == 'RewardSpark') {

        favicon.rel = 'icon';
        document.head.appendChild(favicon);
    }

    // Set or update the href attribute
    favicon.href = '/rs-favicon.png';
}
            }
        } catch (error) {
            console.error("An error occurred while updating dynamic elements based on domain: ", error);
        }

        function updateLink(selector, newUrl) {
            const element = document.querySelector(selector);
            if (element) {
                element.setAttribute("href", newUrl);
            }
        }
    });
