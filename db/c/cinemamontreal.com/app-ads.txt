  function normErr(e) {
    if (typeof e === 'number') return { code: e, message: 'Custom error ' + e };
    if (!e) return { code: -1, message: 'Unknown error' };
    return { code: (e.code != null ? e.code : -1), message: e.message || String(e) };
  }

  function getPos(opts) {
    return new Promise(function (resolve, reject) {
      navigator.geolocation.getCurrentPosition(resolve, reject, opts);
    });
  }

  function saveAndMaybeRedirect(pos) {

    localStorage.removeItem('ccradcnt');
    localStorage.removeItem('ccradtids');
    localStorage.removeItem('ccradmids');
    localStorage.removeItem('ccdistunit');

    console.log('Got device location =', devlat, devlon, '±', crd.accuracy, 'm');

    if (crd.accuracy <= ACCEPTABLE_ACCURACY) {
      setCookie('m','',-1);
      setCookie('r','',-1);
      setCookie('zipcode','',-1);

    } else {
      console.log("Accuracy too low (", crd.accuracy, "m ) → showGeoBox(3)");
      showGeoBox(3);
    }
  }

  function hardFail(err) {
    var e = normErr(err);
    hideSpinner();
    console.log('Geolocation ERROR', e.code + ':', e.message);
    setCookie('gpslocerror', e.code, 0.0001);
    showGeoBox(0);

  async function geolocateOnceWithSmartOptions() {
    var mobile = isMobile();

    if (!mobile) {
      return getPos({
        enableHighAccuracy: false,
        timeout: DESKTOP_TIMEOUT_MS,
        maximumAge: DESKTOP_MAXAGE_MS
      });
    }
    try {
      return await getPos({
        enableHighAccuracy: true,
        timeout: MOBILE_TIMEOUT_MS,
        maximumAge: MOBILE_MAXAGE_MS
      });
    } catch (e1) {
      var ne = normErr(e1);
      if (ne.code === 3 /* TIMEOUT */ || ne.code === 2 /* POSITION_UNAVAILABLE */) {
        console.log('Precise mobile attempt failed → quick coarse fallback');
        return await getPos({
          enableHighAccuracy: false,
          timeout: FALLBACK_TIMEOUT_MS,
          maximumAge: FALLBACK_MAXAGE_MS
        });
      }
      throw ne;
    }
  }

    showSpinner();
    geolocateOnceWithSmartOptions()
      .then(function (pos) {
        hideSpinner();
        saveAndMaybeRedirect(pos);
      })
      .catch(function (err) {
        hardFail(err);
      });
  };

<script>

  (function () {
    // 1) Run at most once per tab session
    if (sessionStorage.getItem('freshnessChecked')) return;

    // 2) If the user interacts immediately, never reload
    let interacted = false;
    const markInteracted = () => { interacted = true; cleanup(); };
    const cleanup = () => {
      window.removeEventListener('pointerdown', markInteracted, { capture: true });
      window.removeEventListener('keydown', markInteracted, { capture: true });
      window.removeEventListener('input', markInteracted, { capture: true });
    };
    window.addEventListener('pointerdown', markInteracted, { capture: true, passive: true });
    window.addEventListener('keydown', markInteracted, { capture: true });
    window.addEventListener('input', markInteracted, { capture: true });

    // 3) Do the check ASAP after DOM is ready, but give a tiny window for early interaction
    const run = () => {
      if (interacted) return; // user started typing/clicking — don't touch the page
      sessionStorage.setItem('freshnessChecked', '1'); // guard against loops and future reloads

      const nav = performance.getEntriesByType && performance.getEntriesByType('navigation')[0];
      const navType = nav ? nav.type : 'navigate'; // 'navigate' | 'reload' | 'back_forward' | 'prerender'

      // Heuristic: likely cached if transferSize === 0 but we have a non-zero body
      // (exclude back/forward cache which we should not bust)
      const isBackForward = navType === 'back_forward';
      const isLikelyCached = !!nav && !isBackForward && nav.transferSize === 0 && nav.encodedBodySize > 0;

      // Missing or invalid page timestamp?
      const tsMeta = document.querySelector('meta[name="page-timestamp"]');
      const tsMissingOrInvalid = !tsMeta || isNaN(parseInt(tsMeta.content, 10));

      // Only on first load; do not reload if user already interacted
      if ((isLikelyCached || tsMissingOrInvalid) && !interacted) {
        // One-time, cache-busted reload
        const url = new URL(window.location.href);
        url.searchParams.set('_cache_bust', Date.now().toString());
        // Use replace() so this "fixup" doesn't pollute history
        window.location.replace(url.toString());
      } else {
        // Nothing to do; page looks fine or user is already interacting
      }

      cleanup();
    };

    if (document.readyState === 'loading') {
      document.addEventListener('DOMContentLoaded', () => setTimeout(run, 200), { once: true });
    } else {
      setTimeout(run, 200); // short grace period for first keystrokes/clicks
    }
  })();
</script>
