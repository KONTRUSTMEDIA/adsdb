<script>
    $(window).scroll(function() {
        if ($("#mainNav").offset().top > 100)
        {
            $('.logo img').attr('src','svg/cwd_logo_red_black.svg');
            $('.navTrigger i').addClass("scrollBlack");
            $('a.nav-link').addClass("scrollBlack");
        }
        else
        {
            $('.logo img').attr('src','svg/cwd_logo_red_white.svg');
            $('.navTrigger i').removeClass("scrollBlack");
            $('a.nav-link').removeClass("scrollBlack");
        }
    });
</script>
