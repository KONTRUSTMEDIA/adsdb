<script type="text/javascript">
    jQuery(document).ready(function($) {
        function initCustomQuantityButtons() {
            if (window.innerWidth <= 1023) {
                $('input.qty').each(function() {
                    if (!$(this).parent().hasClass('custom-quantity-wrapper')) {
                        $(this).wrap('<div class="custom-quantity-wrapper"></div>');
                        $(this).before('<button type="button" class="decrement">-</button>');
                        $(this).after('<button type="button" class="increment">+</button>');
                    }
                });
            }
        }

        function setupEventListeners() {
            $(document).off('click', '.increment').on('click', '.increment', function() {
                var input = $(this).prev('input.qty');
                var currentValue = parseFloat(input.val()) || 0;
                var max = parseFloat(input.attr('max')) || 9999;
                if (currentValue < max) {
                    input.val(currentValue + 1).trigger('change');
                    triggerCartUpdate(input);
                }
            });

            $(document).off('click', '.decrement').on('click', '.decrement', function() {
                var input = $(this).next('input.qty');
                var currentValue = parseFloat(input.val()) || 0;
                var min = parseFloat(input.attr('min')) || 0;
                if (currentValue > min) {
                    input.val(currentValue - 1).trigger('change');
                    triggerCartUpdate(input);
                }
            });

            $(document).off('keydown', 'input.qty').on('keydown', 'input.qty', function(e) {
                e.preventDefault();
            });
        }

        function triggerCartUpdate(input) {
            var $form = input.closest('form.woocommerce-cart-form');
            if ($form.length) {
                $form.find('[name="update_cart"]').prop('disabled', false).trigger('click');
            }
            refreshCartFragments();
        }

        function refreshCartFragments() {
            $.ajax({
                url: '/?wc-ajax=get_refreshed_fragments', // Direct AJAX endpoint for WooCommerce
                type: 'POST',
                success: function(response) {
                    if (response && response.fragments) {
                        $.each(response.fragments, function(key, value) {
                            $(key).replaceWith(value);
                        });
                    }
                }
            });
        }

        $(document.body).on('updated_wc_div', function() {
            $('.custom-quantity-wrapper').replaceWith(function() {
                return $(this).find('input.qty');
            });
            initCustomQuantityButtons();
            setupEventListeners();
            refreshCartFragments();
        });

        initCustomQuantityButtons();
        setupEventListeners();

        $(window).on('resize', function() {
            $('.custom-quantity-wrapper').replaceWith(function() {
                return $(this).find('input.qty');
            });
            initCustomQuantityButtons();
            setupEventListeners();
        });
    });
