        } else {
            if(naverPayIconType == 0) {
                add_icon = "";
            } else if(naverPayIconType == 1) {
                add_icon = "/assets/img/npay@2x.png";
            } else if(naverPayIconType == 2) {
                add_icon = "/assets/img/npay_plus@2x.png";

                add_icon = "";
            }
        }
        return add_icon;
