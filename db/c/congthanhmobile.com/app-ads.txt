        function changeToggle(e) {
            if ($(e).hasClass('fa-plus-square')) {
                $(e).removeClass('fa-plus-square');
                $(e).addClass('fa-minus-square');
            } else {
                $(e).addClass('fa-plus-square');
                $(e).removeClass('fa-minus-square');
            }

        }

    function createContact(e) {

    window.momaChat = {
        initialize: function() {
            try {
                const MOMA_CHAT_CSRF_TOKEN = 'OZQI1xKJ2i1NKlbJN1oiM8rtmvDWxQSTy3ENOjV3';
                const openButton = document.getElementById('chat-open-button');
                const closeButton = document.querySelector('#chat-window .close-btn');
                const chatWindow = document.getElementById('chat-window');

                if (!openButton || !closeButton || !chatWindow) {
                    return;
                }

                const messageContainer = document.getElementById('chat-messages');
                const chatForm = document.getElementById('chat-form');
                const bodyInput = document.getElementById('chat-input-body');

                let conversationId = sessionStorage.getItem('chat_conversation_id');
                let pollingInterval;

                this.loadChatState(chatWindow);
                setTimeout(() => {
                    chatWindow.classList.add('js-ready');
                    console.log('Chat widget is now ready');
                }, 200);

                if (chatWindow.classList.contains('active')) {
                    if (!conversationId) {
                        messageContainer.innerHTML = '<div class="message incoming">Chào bạn, tôi có thể giúp gì được cho bạn?</div>';
                    }
                    this.fetchMessages();
                    if (conversationId) {
                        this.startPolling();
                    }
                }

                const toggleChat = function(e) {
                    e.preventDefault();
                    chatWindow.classList.toggle('active');
                    this.saveChatState(chatWindow.classList.contains('active'));
                    if (chatWindow.classList.contains('active')) {
                        this.fetchMessages();
                        const currentConversationId = sessionStorage.getItem('chat_conversation_id');
                        if (currentConversationId) {
                            this.startPolling();
                        }
                    } else {
                        this.stopPolling();
                    }
                }.bind(this);

                openButton.addEventListener('click', toggleChat);
                closeButton.addEventListener('click', toggleChat);

                chatForm.addEventListener('submit', (e) => {
                    e.preventDefault();
                    const body = bodyInput.value.trim();
                    if (!body) return;
                    const formData = new FormData();
                    formData.append('__token', MOMA_CHAT_CSRF_TOKEN);
                    formData.append('body', body);
                    if (conversationId) {
                        formData.append('conversation_id', conversationId);
                    }

                    fetch('https://congthanhmobile.com/chat/send', {
                        method: 'POST',
                        body: formData,
                        headers: { 'X-Requested-With': 'XMLHttpRequest' }
                    })
                    .then(response => response.ok ? response.json() : Promise.reject('Network error'))
                    .then(data => {
                        if (data.success && data.conversation_id) {
                            if (!conversationId) {
                                conversationId = data.conversation_id;
                                sessionStorage.setItem('chat_conversation_id', conversationId);
                                this.startPolling();
                            }
                            bodyInput.value = '';
                            this.fetchMessages();
                        }
                    })
                    .catch(error => console.error('Error sending message:', error));
                });

            } catch (e) {
                console.error("A critical error occurred in the chat script:", e);
            }
        },

        renderMessages: function(messages) {
            const messageContainer = document.getElementById('chat-messages');
            if (!messageContainer) return;
            messageContainer.innerHTML = '';
            messages.forEach(function(msg) {
                const msgDiv = document.createElement('div');
                msgDiv.className = 'message ' + (msg.sender_type === 'user' ? 'outgoing' : 'incoming');
                msgDiv.textContent = msg.body;
                messageContainer.appendChild(msgDiv);
            });
            messageContainer.scrollTop = messageContainer.scrollHeight;
        },

        fetchMessages: function() {
            fetch('https://congthanhmobile.com/chat/messages', {
                method: 'GET',
                headers: {
                    'Accept': 'application/json',
                    'X-Requested-With': 'XMLHttpRequest'
                }
            })
                .then(response => response.ok ? response.json() : Promise.reject('Network error'))
                .then(data => this.renderMessages(data.messages))
                .catch(error => console.error('Error fetching messages:', error));
        },

        startPolling: function() {
            this.stopPolling();
            cosole.log("this.pollingInterval = setInterval(() => this.fetchMessages(), 5000);"); 
        },

        stopPolling: function() {
            if (this.pollingInterval) {
                clearInterval(this.pollingInterval);
            }
        },

        saveChatState: function(isActive) {
            const state = isActive ? 'open' : 'closed';
            localStorage.setItem('moma_chat_state', state);
        },

        loadChatState: function(chatWindow) {
            const savedState = localStorage.getItem('moma_chat_state');
            
            if (savedState === 'closed') {
                chatWindow.classList.remove('active');
            } else {
                chatWindow.classList.add('active');
            }
        }
    };
