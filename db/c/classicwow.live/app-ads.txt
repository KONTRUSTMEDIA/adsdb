(function($) {

    $menu.find('.menu-item-has-children').each(function () {
      var $item = $(this);
      var $link = $item.find('> .fl-has-submenu-container > a');

      var $submenu = $item.find('> .sub-menu');

      // Remove all default click handlers
      $item.off('click');
      $link.off('click');
      $toggle.off('click');

      // Let links act normally
      $link.on('click', function(e) {
        e.stopPropagation(); // allow the link, just prevent bubbling
      });

      // Toggle submenu only on arrow click
      $toggle.on('click', function(e) {
        e.preventDefault();
        e.stopPropagation();

        const isOpen = $item.hasClass('fl-active');

        // Toggle current
        if (isOpen) {
          $item.removeClass('fl-active');
          $submenu.slideUp(200);
        } else {
          $item.addClass('fl-active');
          $submenu.slideDown(200);
        }
      });
    });
