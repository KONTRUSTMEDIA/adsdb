          submenuBlock.classList.remove("is-open");
          trigger.classList.remove("is-active");

              closeSubmenu();

            if (isOpen) {
              closeSubmenu();
            } else {
              openSubmenu();
              const firstLink = submenuBlock.querySelector("a, button");
              firstLink?.focus();
            }
          }

          // ↓ ArrowDown → open submenu and focus first item
          if (e.key === "ArrowDown") {
            e.preventDefault();
            openSubmenu();
            const firstItem = submenuBlock.querySelector("a, button");
            firstItem?.focus();
          }
        });

          if (e.key === "Escape") {
            closeSubmenu();
            trigger.focus();
          }
        });

          if (
            !submenuBlock.contains(e.relatedTarget) &&
            e.relatedTarget !== trigger
          ) {
            closeSubmenu();
          }
        });

        trigger.addEventListener("focusout", (e) => {
          if (!submenuBlock.contains(e.relatedTarget)) {
            closeSubmenu();
          }
        });

    function initStackingCards() {
      // Kill only this timeline + its ScrollTrigger
      if (tl) {

      const headerCards = document.querySelector(".header_stacking-cards_root");
      if (!wrapCards || !headerCards) return;

      const heightHeaderCards = headerCards.offsetHeight + 64;
      const cardHeights = cards.map((card) => card?.offsetHeight || 0);
      const stackGap = 48;

      const heightScreen = window.innerHeight;
      const widthScreen = window.innerWidth;

      const isSmallDevices = heightScreen >= 880 && widthScreen < 1920;
      const isLargeDevices = heightScreen >= 1200 && widthScreen >= 1920;

      let shiftUp = 0;

      // Set wrapper height
      if (isSmallDevices || isLargeDevices) {
        wrapCards.style.height = `${lastCardHeight}px`;
      } else {
        shiftUp = heightHeaderCards;
        wrapCards.style.height = `${lastCardHeight - heightHeaderCards}px`;
      }

      tl = gsap.timeline({
        scrollTrigger: {
          trigger: ".stacking-cards_section",
          start: "top top",
          end: "+=250%",
          scrub: 0.5,
          pin: true,
          anticipatePin: 1,
        },
      });

      cards.forEach((card, index) => {
        if (index === cards.length - 1) return;

        if (index === 0) {
          tl.to(card, { y: -shiftUp, duration: 0.4, ease: "power2.out" });

          if (shiftUp === heightHeaderCards) {
            tl.to(
              headerCards,
              {
                opacity: 0,
                y: -20,
                duration: 0.4,
                ease: "power1.inOut",

              "<",

          scale: 0.85,
          opacity: 0,
          duration: 0.4,
          ease: "power1.inOut",
        });

        const stackMove = (cardHeights[index] || 0) + stackGap;
        const moveAmount = index === 0 ? stackMove + shiftUp : stackMove;

          tl.to(
            c,
            {
              y: `-=${moveAmount}px`,
              duration: 0.6,
              ease: "power2.out",
            },
            "<",

    function addAriaLabels(swiper) {
      if (!swiper.pagination || !swiper.pagination.bullets) return;

    function lockAccordionLinks(container) {
      if (!container) return;
      container.querySelectorAll("a[href]").forEach((link) => {
        link.setAttribute("tabindex", "-1");
      });
    }

    function unlockAccordionLinks(container) {
      if (!container) return;
      container.querySelectorAll("a[href]").forEach((link) => {
        link.removeAttribute("tabindex");
      });
    }

    const accordions = document.querySelectorAll(".ai-solutions-accordion");

    function closeAccordion(accordion) {
      const content = accordion.querySelector(
        ".wrap-content_ai-solutions-accordion"
      );
      const toggle = accordion.querySelector(
        ".toggle_ai-solutions-accordion"
      );

      if (!accordion.classList.contains("active")) return;

      accordion.classList.remove("active");

      const startHeight = content.scrollHeight;

      const animation = content.animate(
        [
          { height: startHeight + "px", opacity: 1 },
          { height: "0px", opacity: 0 },
        ],
        { duration: 400, easing: "ease" }
      );

      animation.onfinish = () => {
        content.style.height = "0px";
        content.style.opacity = 0;
        lockAccordionLinks(content);

        window.lenis?.resize();
        ScrollTrigger?.refresh();
      };

      toggle.setAttribute("aria-expanded", "false");
    }

    function openAccordion(accordion) {
      const content = accordion.querySelector(
        ".wrap-content_ai-solutions-accordion"
      );
      const toggle = accordion.querySelector(
        ".toggle_ai-solutions-accordion"
      );

      accordion.classList.add("active");

      const endHeight = content.scrollHeight;

      const animation = content.animate(
        [
          { height: "0px", opacity: 0 },
          { height: endHeight + "px", opacity: 1 },
        ],
        { duration: 400, easing: "ease" }
      );

      animation.onfinish = () => {
        content.style.height = "auto";
        content.style.opacity = 1;
        unlockAccordionLinks(content);

        window.lenis?.resize();
        ScrollTrigger?.refresh();
      };

      toggle.setAttribute("aria-expanded", "true");
    }

    accordions.forEach((accordion) => {
      const toggle = accordion.querySelector(
        ".toggle_ai-solutions-accordion"
      );
      const content = accordion.querySelector(
        ".wrap-content_ai-solutions-accordion"
      );

      if (!toggle || !content) return;

      content.style.height = "0px";
      content.style.opacity = 0;
      lockAccordionLinks(content);

      toggle.addEventListener("click", () => {
        const isOpen = accordion.classList.contains("active");

        accordions.forEach((other) => {
          if (other !== accordion) {
            closeAccordion(other);
          }
        });

        if (isOpen) {
          closeAccordion(accordion);
        } else {
          openAccordion(accordion);
        }
      });
    });

    initTabs("[link-form]", "[container-form]");

    function initTabs(triggerSelector, containerSelector) {
      const triggers = document.querySelectorAll(triggerSelector);
      const containers = document.querySelectorAll(containerSelector);

      if (!triggers.length || !containers.length) return;

      const triggerAttr = triggerSelector.replace(/[\[\]]/g, "");
      const containerAttr = containerSelector.replace(/[\[\]]/g, "");

      const urlParams = new URLSearchParams(window.location.search);
      const triggerAttrQuery = triggerAttr.replace("link-", "");
      const paramValue = urlParams.get(triggerAttrQuery);

      triggers.forEach((trigger) => {
        console.log(trigger);
        trigger.addEventListener("click", () => {
          const target = trigger.getAttribute(triggerAttr);

          containers.forEach((container) =>
            container.classList.remove("is-open"),
          );

          const activeContainer = document.querySelector(
            `[${containerAttr}="${target}"]`,
          );
          if (activeContainer) activeContainer.classList.add("is-open");

          triggers.forEach((t) => t.classList.remove("is-active"));
          trigger.classList.add("is-active");

          setTimeout(() => {
            window.lenis?.resize?.();
            ScrollTrigger?.refresh?.();
          }, 0);
        });
      });

      if (paramValue) {
        setTimeout(() => {
          const triggerFromUrl = Array.from(triggers).find(
            (t) => t.getAttribute(triggerAttr) === paramValue,
          );

          if (triggerFromUrl) {
            triggerFromUrl.click();
          }
        }, 100);
      }
    }

    function moveBlock() {
      const mobContainer = document.querySelector(
        ".mob-container_right-block_solutions-listing",

      const rightBlock = document.querySelector(
        ".right-block_solutions-listing",
      );
      if (!rightBlock) return;
      const originalParent = rightBlock.parentElement;
      if (isMobileDevice()) {
        if (mobContainer && mobContainer.contains(rightBlock) === false) {
          mobContainer.appendChild(rightBlock);
          setTimeout(() => {
            window.lenis.resize();
            ScrollTrigger.refresh();
          }, 0);
        }
      } else {
        if (originalParent && originalParent.contains(rightBlock) === false) {
          originalParent.appendChild(rightBlock);
          setTimeout(() => {
            window.lenis.resize();
            ScrollTrigger.refresh();
          }, 0);
        }
      }
    }
