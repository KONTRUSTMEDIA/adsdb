(function() {
  const ITEMS_PER_PAGE = 36;
  const STORAGE_KEY = 'chewaplay_home_count';
  let allGamesData = []; 
  let currentChunkIndex = 0;
  let displayedCount = 0;
  try {
     displayedCount = document.querySelectorAll('#homeGamesGrid .game-card').length;
  } catch(e) { displayedCount = 0; }
  let filteredGames = [];
  const grid = document.getElementById('homeGamesGrid');
  const loadMoreBtn = document.getElementById('loadMoreBtn');
  let isSearchActive = false;
  let isLoading = false;
  const savedCount = sessionStorage.getItem(STORAGE_KEY);
  let targetCount = savedCount ? parseInt(savedCount, 10) : ITEMS_PER_PAGE;
  if (targetCount < ITEMS_PER_PAGE) targetCount = ITEMS_PER_PAGE;

  function wireHoverAnimations(root = document) {
    root.querySelectorAll('img[data-animated-src]').forEach(img => {
        if (img.dataset.wired) return;
        const animated = img.dataset.animatedSrc;
        const statik = img.dataset.staticSrc || img.getAttribute('src');
        const card = img.closest('.game-card');
        if (card) {
            card.addEventListener('mouseenter', () => { img.src = animated; });
            card.addEventListener('mouseleave', () => { img.src = statik; });
            img.dataset.wired = "true";
        }
    });
  }

  async function loadInitialData() {
    try {
      const firstChunk = await window.gameChunksLoader.loadChunk(0);
      allGamesData = firstChunk;
      filteredGames = allGamesData;
      currentChunkIndex = 1;
      console.log('[Homepage] Loaded first chunk:', firstChunk.length, 'games');
      wireHoverAnimations(grid);
      if (displayedCount < targetCount && filteredGames.length > 0) {
          if (displayedCount === 0) grid.innerHTML = '';
          renderBatch(displayedCount, targetCount);
      }
      updateButtonState();
    } catch (err) {
        console.error('[Homepage] Error loading initial data:', err);
        loadMoreBtn.textContent = "Error loading games. Refresh to try again.";
        loadMoreBtn.style.display = 'inline-block';
        loadMoreBtn.disabled = true;
    }
  }

  async function loadMoreChunks() {
    if (isLoading) return;
    isLoading = true;
    try {
      const nextChunk = await window.gameChunksLoader.loadChunk(currentChunkIndex);
      if (nextChunk && nextChunk.length > 0) {
        allGamesData.push(...nextChunk);
        if (!isSearchActive) { filteredGames = allGamesData; }
        currentChunkIndex++;
        console.log('[Homepage] Loaded chunk', currentChunkIndex - 1, '- Total games:', allGamesData.length);
      }
    } catch (err) {
      console.error('[Homepage] Error loading more chunks:', err);
    } finally {
      isLoading = false;
    }
  }

  function renderBatch(startIndex, endIndex) {
     const batch = filteredGames.slice(startIndex, endIndex);
     if (batch.length === 0) return;
     const fragment = document.createDocumentFragment();
     const currentLang = localStorage.getItem('chewaplay_lang') || 'en';
     batch.forEach(g => {
      const a = document.createElement('a');
      a.href = `/games/${g.s}/`;
      a.className = 'game-card';
      a.dataset.title = g.t.toLowerCase();
      a.dataset.mobile = g.m;
      a.dataset.titleEn = g.t;
      if (g.t_am) a.dataset.titleAm = g.t_am;
      if (g.t_om) a.dataset.titleOm = g.t_om;
      const img = document.createElement('img');
      img.src = g.img;
      img.alt = g.t;
      img.loading = 'lazy';
      if (g.a) {
        img.classList.add('game-thumb', 'animated');
        img.dataset.staticSrc = g.img;
        img.dataset.animatedSrc = g.a;
      } else {
        img.classList.add('game-thumb');
      }
      const h3 = document.createElement('h3');
      let title = g.t;
      if (currentLang === 'am' && g.t_am) { title = g.t_am; }
      else if (currentLang === 'om' && g.t_om) { title = g.t_om; }
      h3.textContent = title;
      a.appendChild(img);
      a.appendChild(h3);
      if (g.m) {
        const span = document.createElement('span');
        span.className = 'card-mobile-badge';
        span.title = window.i18n ? window.i18n.translate('filters.mobileOnly') : 'Mobile Friendly';
        span.setAttribute('data-i18n-title', 'filters.mobileOnly');
        span.innerHTML = '<svg width="12" height="12" viewBox="0 0 24 24" fill="currentColor"><path d="M17 1H7c-1.1 0-2 .9-2 2v18c0 1.1.9 2 2 2h10c1.1 0 2-.9 2-2V3c0-1.1-.9-2-2-2zm0 18H7V5h10v14z"/></svg>';
        a.appendChild(span);
      }
      fragment.appendChild(a);
    });
    grid.appendChild(fragment);
    wireHoverAnimations(grid);
    displayedCount = grid.children.length;
    if (!isSearchActive) { sessionStorage.setItem(STORAGE_KEY, displayedCount); }
  }

  loadMoreBtn.addEventListener('click', async function() {
    if (displayedCount + ITEMS_PER_PAGE >= filteredGames.length && !isSearchActive) {
      await loadMoreChunks();
    }
    renderBatch(displayedCount, displayedCount + ITEMS_PER_PAGE);
    updateButtonState();
  });

  function updateButtonState() {
    const remaining = filteredGames.length - displayedCount;
    if (remaining > 0) {
      loadMoreBtn.style.display = 'inline-block';
      loadMoreBtn.disabled = false;
    } else {
      loadMoreBtn.style.display = 'none';
    }
  }

  window.updateGridVisibility = async function() {
    const searchInput = document.getElementById('searchInput');
    if(!searchInput) return; 
    const mobileFilter = document.getElementById('mobileFilterHome');
    const query = searchInput.value.trim();
    const showMobileOnly = mobileFilter ? mobileFilter.checked : false;
    isSearchActive = (query.length > 0);
    if (isSearchActive) {
      console.log('[Homepage] Search active, loading all games...');
      allGamesData = await window.gameChunksLoader.searchGames(query, { mobileOnly: showMobileOnly });
      filteredGames = allGamesData;
    } else {
      console.log('[Homepage] No search query - showing all games');
      filteredGames = allGamesData;
    }
    grid.innerHTML = '';
    displayedCount = 0;
    if (filteredGames.length > 0) {
      renderBatch(0, ITEMS_PER_PAGE);
    } else {
      const noResultsText = window.i18n ? window.i18n.translate('search.noResults') : 'No games found';
      grid.innerHTML = `<p style="color:#ccc; padding:2rem;" data-i18n="search.noResults">${noResultsText}</p>`;
      displayedCount = 0;
    }
    updateButtonState();
    const stats = document.getElementById('searchStats');
    if (stats) {
      if (isSearchActive) {
        stats.textContent = `${filteredGames.length} result${filteredGames.length !== 1 ? 's' : ''}`;
        stats.style.display = 'block';
      } else {
        stats.style.display = 'none';
      }
    }
  };

  function updateGameTitles() {
    const currentLang = localStorage.getItem('chewaplay_lang') || 'en';
    const gameCards = document.querySelectorAll('#homeGamesGrid .game-card');
    gameCards.forEach(card => {
      const h3 = card.querySelector('h3');
      if (!h3) return;
      const titleEn = card.dataset.titleEn;
      const titleAm = card.dataset.titleAm;
      const titleOm = card.dataset.titleOm;
      if (currentLang === 'am' && titleAm) { h3.textContent = titleAm; }
      else if (currentLang === 'om' && titleOm) { h3.textContent = titleOm; }
      else if (titleEn) { h3.textContent = titleEn; }
    });
  }

  window.addEventListener('storage', function(e) {
    if (e.key === 'chewaplay_lang') { updateGameTitles(); }
  });
  window.addEventListener('languageChanged', updateGameTitles);
  loadInitialData();
})();

        (function() {
            const langToggle = document.getElementById('langToggle');

            const langOptions = document.querySelectorAll('.lang-option');
            
            // Get current language from localStorage or default to 'en'
            let currentLang = localStorage.getItem('chewaplay_lang') || 'en';
            
            // Language display names
            const langNames = {
                'en': 'EN',
                'am': 'አማ',
                'om': 'OM'
            };
            
            // Update current language display
            function updateLangDisplay() {
                langCurrent.textContent = langNames[currentLang] || 'EN';
                document.documentElement.lang = currentLang;
            }
            
            // Toggle dropdown
            function toggleDropdown() {
                langDropdown.classList.toggle('active');
            }
            
            // Close dropdown
            function closeDropdown() {
                langDropdown.classList.remove('active');
            }
            
            // Change language - always reload
            function changeLanguage(lang) {
                if (lang === currentLang) {
                    closeDropdown();
                    return;
                }
                
                // Save to localStorage
                localStorage.setItem('chewaplay_lang', lang);
                
                // Force immediate reload
                window.location.reload();
            }
            
            // Event listeners
            if (langToggle) {
                langToggle.addEventListener('click', (e) => {
                    e.stopPropagation();
                    toggleDropdown();
                });
            }
            
            langOptions.forEach(option => {
                option.addEventListener('click', (e) => {
                    e.preventDefault();
                    const lang = option.dataset.lang;
                    changeLanguage(lang);
                });
            });
            
            // Close dropdown when clicking outside
            document.addEventListener('click', (e) => {
                if (langToggle && !langToggle.contains(e.target) && !langDropdown.contains(e.target)) {
                    closeDropdown();
                }
            });
            
            // Initialize display
            updateLangDisplay();

                    // Lazy load Fuse.js when needed
                    window.loadFuseJS().then(() => {
                        if (window.Fuse) {
                            fuse = new Fuse(allGames, {
                                keys: [
                                    { name: 't', weight: 0.7 }, // Title
                                    { name: 'c', weight: 0.3 }  // Categories
                                ],
                                threshold: 0.3, // Fuzzy match threshold
                                ignoreLocation: true
                            });
                        }
                    }).catch(e => console.error('Failed to load Fuse.js', e));
                    
                    // Expose search function to window for index.njk to use
                    window.performSearch = function(query, showMobileOnly) {
                         if (!query && !showMobileOnly) return allGames;
                         
                         let results = allGames;
                         
                         if (query) {
                             if (fuse) {
                                 results = fuse.search(query).map(result => result.item);
                             } else {
                                 // Fallback if Fuse fails to load
                                 const lowerQ = query.toLowerCase();
                                 results = allGames.filter(g => 
                                    g.t.toLowerCase().includes(lowerQ) || 
                                    (g.c && g.c.some(cat => cat.toLowerCase().includes(lowerQ)))
                                 );
                             }
                         }
                         
                         if (showMobileOnly) {
                             results = results.filter(g => g.m);
                         }
                         
                         return results;
                    };
                })
                .catch(e => console.error("Failed to load search index", e));

            function showDropdown(results) {
                if (!dropdown) return;
                
                dropdown.innerHTML = '';
                
                if (results.length === 0) {
                    const noResultsText = window.i18n ? window.i18n.translate('search.noResults') : 'No games found';
                    dropdown.innerHTML = `<div class="search-no-results">${noResultsText}</div>`;
                    dropdown.classList.add('active');
                    return;
                }
                
                const maxResults = 8;
                const limitedResults = results.slice(0, maxResults);
                
                limitedResults.forEach(game => {
                    const item = document.createElement('a');
                    item.href = `/games/${game.s}/`;
                    item.className = 'search-result-item';
                    
                    const img = document.createElement('img');
                    img.src = game.img;
                    img.alt = game.t;
                    img.loading = 'lazy';
                    img.width = 50;
                    img.height = 50;
                    
                    const title = document.createElement('span');
                    title.textContent = game.t;
                    
                    item.appendChild(img);
                    item.appendChild(title);
                    
                    if (game.m) {
                        const badge = document.createElement('span');
                        badge.className = 'search-mobile-badge';
                        badge.setAttribute('aria-label', 'Mobile friendly');
                        badge.innerHTML = '📱';
                        item.appendChild(badge);
                    }
                    
                    dropdown.appendChild(item);
                });
                
                if (results.length > maxResults) {
                    const more = document.createElement('div');
                    more.className = 'search-more-results';
                    const moreText = window.i18n ? window.i18n.translate('search.moreResults') : 'more results';
                    more.textContent = `+${results.length - maxResults} ${moreText}`;
                    dropdown.appendChild(more);
                }
                
                dropdown.classList.add('active');
            }
            
            function hideDropdown() {
                if (dropdown) {
                    dropdown.classList.remove('active');
                }
            }

            function handleInput() {
                const query = searchInput.value.trim();
                const showMobileOnly = mobileFilter ? mobileFilter.checked : false;

                // If on homepage, trigger grid update
                if (window.updateGridVisibility) {
                    window.updateGridVisibility();
                    hideDropdown(); // Hide dropdown on homepage as grid updates
                } else {
                    // On other pages, show dropdown with results
                    if (query.length > 0) {
                        const results = window.performSearch ? window.performSearch(query, showMobileOnly) : [];
                        showDropdown(results);
                    } else {
                        hideDropdown();
                    }
                }
            }

            if (searchInput) {
                searchInput.addEventListener('input', () => {
                    clearTimeout(debounceTimer);
                    debounceTimer = setTimeout(handleInput, 300);
                });
                
                // Hide dropdown when clicking outside
                document.addEventListener('click', (e) => {
                    if (!searchInput.contains(e.target) && !dropdown.contains(e.target)) {
                        hideDropdown();
                    }
                });
                
                // Show dropdown again when focusing search
                searchInput.addEventListener('focus', () => {
                    if (searchInput.value.trim().length > 0 && !window.updateGridVisibility) {
                        handleInput();
                    }
                });
            }

            if (mobileFilter) {
                mobileFilter.addEventListener('change', handleInput);
            }
        })();
    </script>
