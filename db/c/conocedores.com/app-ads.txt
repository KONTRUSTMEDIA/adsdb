    // Variables
    const popup = document.getElementById('popup-conocedores');

    // querySelector selecciona al primer elemento que encuentra con dicha clase o id. En este caso estamos seleccionando al btn close del popup
    const popup_btn_close = document.querySelectorAll('#popup-conocedores .js-popup__btn-close');

    // Valido popup y btn close
    if (true) {

        const popup_transition = 600;
        const popup_time_open = 5000;
        const popup_time_close = 12000;
        var $is_popup_close = true;

        /*
        sessionStorage es un objeto de pares (key, value). Que mantiene sus datos durante la sesion del sitio.
        Ver:
        https://developer.mozilla.org/es/docs/Web/API/Window/sessionStorage
        https://stackoverflow.com/questions/30269341/modal-popup-only-once-per-session/30270625
        */

        // Por defecto 2 sessionStorage["PopupShown"] = 'yes';
        // sessionStorage["PopupShown"] = 'no';
        // Si se cerró el popup
        if (sessionStorage["PopupShown"] === 'no') {
            //Agrego clase popup-d-none
            popup.classList.add('popup-d-none');
        }

        if (popup_btn_close[0]) {
          popup_btn_close[0].addEventListener("click", function( event ) {
              event.preventDefault();
              sessionStorage["PopupShown"] = 'no';
              popup.classList.add('popup-d-none');
          });
        }

        if (popup_btn_close[1]) {
          popup_btn_close[1].addEventListener("click", function( event ) {
              event.preventDefault();
              sessionStorage["PopupShown"] = 'no';
              popup.classList.add('popup-d-none');
          });
        }
        if (popup_btn_close[2]) {
          popup_btn_close[2].addEventListener("click", function( event ) {
              event.preventDefault();
              sessionStorage["PopupShown"] = 'no';
              popup.classList.add('popup-d-none');
          });
        }
        if (popup_btn_close[3]) {
          popup_btn_close[3].addEventListener("click", function( event ) {
              event.preventDefault();
              sessionStorage["PopupShown"] = 'no';
              popup.classList.add('popup-d-none');
          });
        }
        if (popup_btn_close[4]) {
          popup_btn_close[4].addEventListener("click", function( event ) {
              event.preventDefault();
              sessionStorage["PopupShown"] = 'no';
              popup.classList.add('popup-d-none');
          });
        }

        // Si se abre la sesión
        if (sessionStorage["PopupShown"] != 'no') {
            // Si NO se scrolea, NO se abre el popup. Si se scrolea, se abre solo el popup a los popup_time_open milisegundos y se borra a los popup_time_close milisegundos
            window.addEventListener('scroll', function(e) {
                if ($is_popup_close) {
                    $is_popup_close = false;

                    setTimeout(function(){
                        popup.classList.remove('popup-d-none');
                    }, popup_time_open);
                    setTimeout(function(){
                        popup.classList.add('popup-d-none');
                        sessionStorage["PopupShown"] = 'no';
                    }, popup_time_close);
                }
            });
        }


    }
