function resetBtn(){ 
	document.getElementById('label').innerHTML='';

	changeTextDirect('');

function clearResults()
{
	ul = document.getElementById("myUL");
	li = ul.getElementsByTagName('p');
	// Loop through all list items and hide them:
	var match=false;
	for (i = 0; i < li.length; i++) {
		li[i].style.display = "none";
	}	
}

function changeTextDirect(str){		
	clearResults();
	// Add a delay before showing the results:
	setTimeout(function() {
		changeTextDirect2(str);
	}, 150);
}

function changeTextDirect2(str){		
		
	document.getElementById('label').style = "margin-top: 1em";
	
	/////next is filter from w3schools.com/howto/howto_js_filter_lists.asp
	// Declare variables
	var input, filter, ul, li, a, i;
	input = document.getElementById('myInput');
	filter = input.value.toUpperCase();
	
	// Special case of Ampere, use Amp instead:
	if(filter == "AMPERE" || filter == "AMPERES")
	{
		filter = "AMP"; 
		console.log("using amp instead of ampere ");
	}
	
	console.log(filter);
	ul = document.getElementById("myUL");
	li = ul.getElementsByTagName('p');
	// Loop through all list items, and hide those who don't match the search query
	var match=false;;
	for (i = 0; i < li.length; i++) {
		a = li[i].getElementsByTagName("a")[0];
		if (a.innerHTML.toUpperCase().indexOf(filter) > -1) {
			li[i].style.display = "";
			match=true;
			console.log(match);
		} else {
			li[i].style.display = "none";
			console.log("not a match");
		}
	}	
	if (match!=true){
		
		document.getElementById("ans").innerHTML="Sorry we don't have any <strong>"+str+"</strong> conversion yet, we will take note and you can ask us when we will have it at <a target=_top href='mailto:suggestionsfeed@gmail.com?Subject=Convert-formula' target=_top>suggestionsfeed@gmail.com</a>";
	}
	else{
		document.getElementById("ans").innerHTML="";
		
	}
	
	if (resetting){
		resetting = false;
		document.getElementById('label').innerHTML='';
		document.getElementById("ans").innerHTML="";
		document.getElementById('label').style = "margin-top: 0em";
	}
	else
	{
		document.getElementById('label').innerHTML='Your query matches:';
	}
	
	/////
	ga('send', 'event', 'button', 'textbox', 'Text:'+str);
}
