/*** Auto complete function for Places ***/
function locautocomplete()
{
	if(i>0)return;
	var jud=$("#jud").val();
	var loc=$("#loc").val();
	$('#str').val('');
	$('#islike').val('0');
	//$("#loc").unautocomplete();	
	$("#loc").autocomplete({
		source: siteURL+"modules/views/pages/auto_city.php?jud="+jud,
		/*
		width: 170,
		selectFirst: false,
		matchSubset: false,
		*/
		select: function(event, ui) {
			var locat = ui.item.value;
			locat = locat.split(" (")[0];
			if (ui.item && ui.item.value){
                titleinput = ui.item.value;
				$("#loc").val(locat);
                ui.item.value= $.trim(locat);
            } 
			if($("#loc").val()!=''){
				$('#str').val('');
				LoadZipcodesNew(0);
			}
			
		}
	});
	i++;			
}

				<h1>Coduri postale</h1>
Codurile poștale din România pot fi găsite cel mai ușor cu ajutorul motorului de căutare <em>coduripostale</em>. Codurile poștale pentru localități sunt afișate într-un tabel cu patru coloane, respectiv: Judet, Localitate, Strada (unde există) și Cod poștal, ordonate alfabetic. În cazul localităților dintr-un județ care au denumiri identice, se afișează între paranteze și denumirea comunei, în dreptul localității respective.<br><br>

<h3>Cod postal</h3>
Codul poștal este o serie de litere sau cifre (ori o combinația acestora) adăugate adresei poștale pentru scopul sortării corespondenței. Prima țară care a folosit cod poștal era Germania, încă de la începutul anilor 1960. Majoritatea serviciilor poștale din lume au un sistem de cod poștal, excepție fiind Irlanda, Hong Kong și Noua Zeelandă, ultimul având un sistem de cod poștal parțial. În general aceste coduri de poștă reprezintă anumite zone geografice dar există și excepții, când codurile poștale reprezintă instituții, agenții guvernamentale ori companii comerciale mari.<br><br>

<h3>Cod postal, Romania</h3>
Poșta Română a introdus codificarea poștală din România în 1974, <em>vechile coduri poștale</em> fiind formate din patru cifre. în 2003 (1 Mai 2003) au intrat în vigoare <em>noile coduri poștale</em>, formate din șase cifre, cu o codificarea detaliată până la nivel de stradă. La realizarea codului poștal au fost luate în considerare localitățile din România cuprinse în Legea nr. 2/1968. Scopul noului sistem era scurtarea timpului de operare în serviciile poștale Românești și eliminarea confuziilor între localitățile omonime din același județ. Iată codurile pentru localitățile cu numele <a href="http://www.coduripostale.ro/index.php?loc=Barasti&list=Cauta" rel="nofollow" title="Localități cu nume identice ori similare din România. Exemplu: Bărăști">Bărăști</a>. Lista noilor <em>coduri poștale</em> include toate localitatile si strazile (cu cod poștal) din Romania. Baza de date coduripostale.ro (<strong>cautare coduri postale romanesti</strong>) este prima bază de date de acest gen din România, lansat in 2003, la numai câteva luni după introducerea codurilor poștale noi. Situl a fost complet revizuit și îmbunătățit în 2005, 2011 și 2014.<br><br>

<h3>Coduri postale Romanesti</h3>
Semnificația codurilor poștale românești (noul cod poștal, numerele din codul de șase cifre) sunt: Prima cifră (0-9), reprezintă cele zece regiuni poștale în care a fost împărțită România. A doua cifră (0-5) codifică județele componente ale unei regiuni poștale, București având o situație specială: cifrele 1 și 6 reprezintă cele șase sectoare ale municipiului București (ca la județe), iar cifrele 7 și 8 reprezintă județele Giurgiu și Ilfov. Ultimele patru cifre precizează, din punct de vedere poștal, reședințe de județ, orașe, sate (cu precizarea comunei), străzi, porțiuni de străzi și imobile. <br><br>

<h4>Coduri poștale pe orașe</h4>
Codurile poștale ale localităților din cele 41 județe (plus București) pot fi găsite în tabelul de pe prima pagină. La accesarea unui judet, se afișează toate localitățile din acest judet, în ordine alfabetică.<br>Lista celor 48 de localități cu cod poștal pe străzi se găsesc tot pe prima pagină, sub tabelul codurilor poștale pe județe.<br><br>

<h4>Coduri poștale pe străzi</h4>
Codurile poștale ale străzilor cuprind denumirea străzii în ordine alfabetică, numerele imobilelor cu- și fără soț, în ordine crescătoare. Când se afișează la o adresa 1-T, acesta înseamnă: de la numărul 1 la toate numerele impare, când se afișează 4-T, înseamnă: de la numărul 4 la toate numerele pare.<br><br>

<h4>Lista localitatilor din Romania</h4>
Coduripostale.ro cuprinde și lista completă a localităților din România (localitati pe judete). Se pot face căutări pentru localități pe județe, după nume și dupa cod. Prin accesarea unui județ, se găsesc toate localitățile din România, în ordine alfabetică. Motorul de căutare listează și localitățile omonime (cu același nume).<br><br>			</div>

/*** Global varible for site url  ***/
var siteURL='https://www.coduripostale.ro/';
/*** Function to get places for county select ***/
function GetPlaces()
{
	var jud=$('#submitCounty').val();
	if(jud==''){return;}
	$.ajax({
		type: "POST",
		url: siteURL+"./modules/submit_zipcode.php?",
		data: "mode=GetPL&jud="+jud,
		success: function(d){
			$('#submitPlace').html('<option value="">Alege...</option>'+d);
		}
		,error: function() {$("#error_response").html("<div class='lead text-danger'>Ceva nu e în regulă :-(");}
	});	
}
/*** Function for Submit insert suggesstion ***/
function InsertSuggestions()
{
	var jud=$('#submitCounty :selected').val();
	var place=$('#submitPlace :selected').val();
	var str=$('#submitStreet').val();
	var zip=$('#submitZip').val();
	if(jud==''){
		$("#error_response").html("<div class='lead text-danger'>Selectați județul");
		return false;
	}else if (place =='') {
		$("#error_response").html("<div class='lead text-danger'>Selectați localitatea");
		return false;	
	}else if(zip ==''){
		$("#error_response").html("<div class='lead text-danger'>Introduceți codul poștal");
		return false;
	}else if(zip!='' && zip.length!=6){
		$("#error_response").html("<div class='lead text-danger'>Format incorect");
		return false;
	}

	var data1="mode=InsertSug&jud="+jud+"&place="+place+"&str="+str+"&zip="+zip;
	SendData(data1, 3);
	
}
/*** Submit Data to submit_zipcode.php  ***/
function SendData(data1, tip)
{
	$.ajax({
		url: siteURL+"./modules/submit_zipcode.php?",
		data: data1,
		success: function(d){
			if(d=="1"){
				$("#error_response").html("<div class='lead text-success'>Am trimis mesajul. Mersi! ;)");
				$('#submitButton').addClass('hidden');
        		$('#submitClose').removeClass('hidden');
				$('#submitCounty').val('');
				$('#submitPlace').html('<option value="">Alege...</option>');
				$('#submitStreet').val('');
				$('#submitZip').val('');
			}else if(d=="0"){
				$("#error_response").html("<div class='lead text-danger'>Introduceți datele necesare");
			}else{
				$("#error_response").html("<div class='lead text-danger'>Ceva nu e în regulă :-(");
			}		
		}
		,error: function() {$("#error_response").html("<div class='lead text-danger'>Ceva nu e în regulă :-(");}
	});	
}
