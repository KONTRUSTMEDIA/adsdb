<script>
// Hover + click dropdown with longer delay
document.querySelectorAll('.cu-dropdown').forEach(function(drop){
  let timeout;
  const menu = drop.querySelector('.cu-dropdown-menu');
  const btn = drop.querySelector('button');

  // Hover
  drop.addEventListener('mouseenter', function(){
    clearTimeout(timeout);
    menu.style.display = 'block';
    menu.style.opacity = '1';
  });

  drop.addEventListener('mouseleave', function(){
    timeout = setTimeout(function(){
      menu.style.opacity = '0';
      setTimeout(()=>{ menu.style.display = 'none'; }, 250);
    }, 600); // longer delay
  });

  // Click toggle
  btn.addEventListener('click', function(e){
    e.stopPropagation();
    if(menu.style.display === 'block'){
      menu.style.opacity = '0';
      setTimeout(()=>{ menu.style.display = 'none'; }, 250);
    } else {
      menu.style.display = 'block';
      menu.style.opacity = '1';
    }
  });
});

// Click outside closes all
document.addEventListener('click', function(){
  document.querySelectorAll('.cu-dropdown-menu').forEach(function(menu){
    menu.style.opacity = '0';
    setTimeout(()=>{ menu.style.display = 'none'; }, 250);
  });
});
</script>
