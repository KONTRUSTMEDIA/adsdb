    <script type="module">
      // Theme Toggle
      const root = document.documentElement;
      const storageKey = 'theme';
      const preferred = localStorage.getItem(storageKey) || (window.matchMedia('(prefers-color-scheme: dark)').matches ? 'dark' : 'light');
      setTheme(preferred);
      
      document.getElementById('theme-toggle')?.addEventListener('click', () => {
        const next = root.dataset.theme === 'dark' ? 'light' : 'dark';
        setTheme(next);
      });
      
      function setTheme(theme) {
        root.dataset.theme = theme;
        localStorage.setItem(storageKey, theme);
      }

      // Language Dropdown
      const languageToggle = document.querySelector('.language-toggle');
      const languageMenu = document.querySelector('.language-menu');
      
      if (languageToggle && languageMenu) {
        languageToggle.addEventListener('click', (e) => {
          e.stopPropagation();
          const isOpen = languageToggle.getAttribute('aria-expanded') === 'true';
          languageToggle.setAttribute('aria-expanded', !isOpen);
          languageMenu.classList.toggle('show', !isOpen);
        });

        // Close dropdown when clicking outside
        document.addEventListener('click', (e) => {
          if (!languageToggle.contains(e.target) && !languageMenu.contains(e.target)) {
            languageToggle.setAttribute('aria-expanded', 'false');
            languageMenu.classList.remove('show');
          }
        });

        // Close dropdown on escape key
        document.addEventListener('keydown', (e) => {
          if (e.key === 'Escape') {
            languageToggle.setAttribute('aria-expanded', 'false');
            languageMenu.classList.remove('show');
            languageToggle.focus();
          }
        });
      }
    </script>
