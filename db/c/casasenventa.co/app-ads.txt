      document.addEventListener('DOMContentLoaded', function() {
        const playButton = document.getElementById('play-video');
        const video = document.getElementById('video');
        const backgroundImage = document.getElementById('img-background');
        const videoContainer = document.getElementById('video-background');
        const videoIcon = document.getElementById('video-icon');
        let isPlaying = false; // To track the current state

        playButton.addEventListener('click', function(event) {
            event.preventDefault();

            if (!isPlaying) {
                // Play the video
                backgroundImage.style.display = 'none';
                videoContainer.style.display = 'block';
                video.play();

                // Change the icon to "Pause"
                videoIcon.classList.remove('fa-play');
                videoIcon.classList.add('fa-pause');

                isPlaying = true;
            } else {
                // Pause the video and show the background image again
                video.pause();
                videoContainer.style.display = 'none';
                backgroundImage.style.display = 'block';

                // Change the icon back to "Play"
                videoIcon.classList.remove('fa-pause');
                videoIcon.classList.add('fa-play');

                isPlaying = false;
            }
        });
    });
