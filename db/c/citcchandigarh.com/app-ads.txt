<script>
(function() {
    // Configuration: 30s, 180s (3m), 720s (12m)
    // "Show the popup first after 30 seconds, then after 180 seconds (from initial display), then after 720 seconds (from second display)"
    // This implies intervals of 30s, 180s, 720s.
    const INTERVALS = [30 * 1000, 180 * 1000, 720 * 1000]; 
    const MAX_STAGES = INTERVALS.length;
    const STORAGE_KEY = 'citc_popup_state';

    // Helper to get state
    function getState() {
        try {
            const s = localStorage.getItem(STORAGE_KEY);
            return s ? JSON.parse(s) : null;
        } catch (e) {
            return null;
        }
    }

    // Helper to set state
    function setState(state) {
        localStorage.setItem(STORAGE_KEY, JSON.stringify(state));
    }

    // Initialize or check state
    function init() {
        let state = getState();
        const now = Date.now();

        // If no state, initialize
        if (!state) {
            state = {
                stage: 0, // 0 means waiting for 1st popup (30s)
                nextShowTime: now + INTERVALS[0],
                isCompleted: false
            };
            setState(state);
        }
        
        // Auto-fill the concern based on page title if empty
        const pageTitle = document.title.split('|')[0].trim();
        const concernInput = document.getElementById('global_other_concern');
        if (concernInput && !concernInput.value) {
            concernInput.value = pageTitle;
        }

        // If completed, do nothing
        if (state.isCompleted) {
            return;
        }

        checkAndSchedule(state);
    }

    function checkAndSchedule(state) {
        const now = Date.now();
        
        if (state.isCompleted) return;

        // If it's time to show (or past time)
        if (now >= state.nextShowTime) {
            showPopup();
            
            // Advance state immediately upon showing
            const nextStageIndex = state.stage + 1;
            
            if (nextStageIndex < MAX_STAGES) {
                // Schedule next
                // The prompt says "then after 180 seconds (from initial display)"
                // So the next target is NOW + INTERVALS[nextStageIndex]
                
                state.stage = nextStageIndex;
                state.nextShowTime = now + INTERVALS[nextStageIndex];
                setState(state);
                
                // Schedule the check for the next time
                setTimeout(() => checkAndSchedule(getState()), INTERVALS[nextStageIndex]);
                
            } else {
                // Done
                state.isCompleted = true;
                setState(state);
            }
        } else {
            // Not time yet, schedule a check
            const delay = state.nextShowTime - now;
            if (delay > 0) {
                setTimeout(() => checkAndSchedule(getState()), delay);
            } else {
                 // Should have fired, but maybe very slight overlap. Fire now.
                 checkAndSchedule(state); 
            }
        }
    }

    function showPopup() {
        $('#enquiryPopup').css('display', 'flex').hide().fadeIn();
    }
    
    // Run on load
    $(document).ready(function() {
        init();
        
        // Close button
        $('#closePopup').off('click').on('click', function() {
            $('#enquiryPopup').fadeOut();
        });
        
        // Chat icon
        $('#floatingChatIcon').off('click').on('click', function () {
             $('#enquiryPopup').css('display', 'flex').hide().fadeIn();
        });
        
        // Form Submission
        $('#popupEnquiryForm').off('submit').on('submit', function (e) {
            e.preventDefault();
        
            let $btn = $('#popupSubmitBtn');
        
            $.ajax({
                url: "https://citcchandigarh.com/studentEnqury",
                method: "POST",
                data: $(this).serialize(),
                headers: {
                    'X-CSRF-TOKEN': $('input[name="_token"]').val()
                },
                beforeSend: function () {
                    $btn.prop('disabled', true); 
                    $btn.html('<span><i class="fa fa-spinner fa-spin"></i> Submitting...</span>'); 
                },
                success: function (response) {
                    $btn.prop('disabled', false);
                    $btn.html('<span>Submit</span>');
        
                    if (response === 'success') {
                        Swal.fire({
                            icon: 'success',
                            title: 'Thank you!',
                            text: 'Your enquiry has been submitted.',
                            confirmButtonText: 'OK',
                            confirmButtonColor: '#3085d6',
                            backdrop: 'rgba(0, 0, 0, 0.6)',
                            allowOutsideClick: false,
                            customClass: {
                                popup: 'popup-custom',
                                title: 'swal-title-custom',
                                content: 'swal-content-custom'
                            },
                            didClose: () => {
                                $('#popupEnquiryForm')[0].reset();
                                $('#enquiryPopup').fadeOut();
                            }
                        });
                    } else {
                        Swal.fire({
                            icon: 'error',
                            title: 'Oops...',
                            text: 'Something went wrong. Please try again.',
                            confirmButtonText: 'OK',
                            confirmButtonColor: '#d33',
                            backdrop: 'rgba(0, 0, 0, 0.6)',
                            allowOutsideClick: false,
                            customClass: {
                                popup: 'popup-custom',
                                title: 'swal-title-custom',
                                content: 'swal-content-custom'
                            }
                        });
                    }
                },
                error: function (xhr) {
                    $btn.prop('disabled', false); 
                    $btn.html('<span>Submit</span>');
        
                    console.error(xhr.responseText);
                    Swal.fire({
                        icon: 'error',
                        title: 'Error',
                        text: xhr.responseJSON?.error || 'Internal Server Error.',
                        confirmButtonText: 'OK',
                        confirmButtonColor: '#d33',
                        backdrop: 'rgba(0, 0, 0, 0.6)',
                        allowOutsideClick: false,
                        customClass: {
                            popup: 'popup-custom',
                            title: 'swal-title-custom',
                            content: 'swal-content-custom'
                        }
                    });
                }
            });
        });
    });

})();
</script>
