        <script>
            // Application data object
            var data = {"response_type":"code","client_id":"cityads","state":"2004fe1199eca08aa38309b93dedecf0","redirect_uri":"https:\/\/cityads.com\/app-ads.txt"};

            // Messages
            var slowMessage  = "Application is loading... Please wait a second.",
                serverUnavailableMessage  = "The authorization server is not available.<br><br>The next attempt after :seconds seconds.",
                cookieFailMessage = "<b>Cookies Required</b><br><br>Cookies are not enabled on your browser.<br>Please enable cookies in your browser preferences to continue.";

            var authURL = "https://auth2.cityads.com/oauth/authorize?" + serialize(data);


            var cookieCheckTimeout;
            var countdownInterval;

            var getCookie = function (cname) {
                var name = cname + "=";
                var ca = document.cookie.split(';');
                for(var i = 0; i <ca.length; i++) {
                    var c = ca[i];
                    while (c.charAt(0)==' ') {
                        c = c.substring(1);
                    }
                    if (c.indexOf(name) == 0) {
                        return c.substring(name.length,c.length);
                    }
                }
                return "";
            }

            var failedAttemptsCount = getCookie('auth_client_failed_attempts') || 0;

            // Serialize object to query string
            function serialize(obj) {
                var str = [];

                for (var p in obj) {
                    if (obj.hasOwnProperty(p)) {
                        str.push(encodeURIComponent(p) + "=" + encodeURIComponent(obj[p]));
                    }
                }

                return str.join("&");
            }


            var showMessage = function (message) {
                document.getElementById('text').innerHTML = message;
                document.getElementById('message').removeAttribute('class');
            };
            var requestAuthCode = function () {
                showMessage(slowMessage);
                document.location.href = authURL;
            };

            var serverUnavailableError = function (seconds) {
                showMessage(serverUnavailableMessage.replace(':seconds', seconds));
                var countdown = function () {
                    seconds--;
                    showMessage(serverUnavailableMessage.replace(':seconds', seconds));
                };

                countdownInterval = setInterval(countdown, 1000);

                setTimeout(function () {
                    clearInterval(countdownInterval);
                    requestAuthCode();
                }, seconds * 1000);
            };

            var makeCookieCheck = function () {
                clearTimeout(cookieCheckTimeout);

                if (navigator.cookieEnabled) {
                    if (failedAttemptsCount > 1) {
                        var seconds = 5 * failedAttemptsCount;
                        if (seconds > 20) {
                            seconds = 20;
                        }
                        serverUnavailableError(seconds);
                    } else {
                        requestAuthCode();
                    }
                } else {
                    showMessage(cookieFailMessage);
                    cookieCheckTimeout = setTimeout(makeCookieCheck, 3000);
                }
            };

            makeCookieCheck();
        </script>
