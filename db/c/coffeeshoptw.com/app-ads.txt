    <!-- 認證與 UI 初始化 -->
    <script>
        document.addEventListener('DOMContentLoaded', async function() {
            // 檢查登入狀態
            try {
                const response = await fetch(`${window.API_BASE_URL}/api/auth/status`, {
                    credentials: 'include'
                });
                
                if (response.ok) {
                    const data = await response.json();
                    if (data.authenticated && data.user) {
                        // 更新 UI 為已登入狀態
                        updateAuthUI(data.user);
                    }
                }
            } catch (error) {
                console.log('未登入或無法取得登入狀態');
            }
            
            // 綁定登入按鈕 - 跳轉到登入頁面讓用戶選擇登入方式
            document.getElementById('login-btn')?.addEventListener('click', (e) => {
                e.preventDefault();
                window.location.href = '/login';
            });
            
            document.getElementById('mobile-login-btn')?.addEventListener('click', (e) => {
                e.preventDefault();
                window.location.href = '/login';
            });
            
            // 綁定登出按鈕
            document.getElementById('logout-btn')?.addEventListener('click', async (e) => {
                e.preventDefault();
                try {
                    await fetch(`${window.API_BASE_URL}/api/auth/logout`, {
                        method: 'POST',
                        credentials: 'include'
                    });
                    window.location.reload();
                } catch (error) {
                    console.error('登出失敗', error);
                }
            });
            
            document.getElementById('mobile-logout-btn')?.addEventListener('click', async (e) => {
                e.preventDefault();
                try {
                    await fetch(`${window.API_BASE_URL}/api/auth/logout`, {
                        method: 'POST',
                        credentials: 'include'
                    });
                    window.location.reload();
                } catch (error) {
                    console.error('登出失敗', error);
                }
            });
            
            // 載入統計資料
            loadStats();
        });
        
        // 更新認證 UI
        function updateAuthUI(user) {
            document.documentElement.setAttribute('data-authenticated', '1');
            
            // 隱藏訪客選單，顯示已登入選單
            document.querySelectorAll('.nav-guest-only').forEach(el => el.style.display = 'none');
            document.querySelectorAll('.nav-auth-required').forEach(el => el.style.display = '');
            
            // 更新使用者資訊
            const userNameEl = document.getElementById('user-name');
            const userAvatarEl = document.getElementById('user-avatar');
            const userIconEl = document.getElementById('user-icon');
            
            if (userNameEl) userNameEl.textContent = user.username || '使用者';
            
            if (user.avatar && userAvatarEl) {
                const avatarUrl = user.avatar.startsWith('http') 
                    ? user.avatar 
                    : `https://cdn.discordapp.com/avatars/${user.id}/${user.avatar}.png`;
                userAvatarEl.src = avatarUrl;
                userAvatarEl.alt = user.username;
                userAvatarEl.style.display = '';
                if (userIconEl) userIconEl.style.display = 'none';
            }
        }
        
        // 載入統計資料
        async function loadStats() {
            try {
                const response = await fetch(`${window.API_BASE_URL}/api/health`);
                if (response.ok) {
                    const data = await response.json();
                    const botStatus = data.bot_status;
                    
                    const totalServers = document.getElementById('total-servers');
                    const totalUsers = document.getElementById('total-users');
                    
                    if (totalServers && botStatus.guilds_count !== undefined) {
                        totalServers.textContent = formatNum(botStatus.guilds_count);
                    }
                    if (totalUsers && botStatus.users_count !== undefined) {
                        totalUsers.textContent = formatNum(botStatus.users_count);
                    }
                }
            } catch (error) {
                console.log('無法載入統計資料');
            }
        }
        
        // 格式化數字
        function formatNum(num) {
            if (num >= 1000000) return (num / 1000000).toFixed(1) + 'M';
            if (num >= 1000) return (num / 1000).toFixed(1) + 'K';
            return num.toString();
        }
    </script>
