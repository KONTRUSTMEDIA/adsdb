      class Loader {
        init() {
          this.setGlobals()
          this.clearAuthToken()
        }
        setGlobals() {
          window.accessToken = this.getAccessToken()
          window.sessionToken = this.getSessionToken()
          window.isRemoteApi = window.location.host === window.$API_HOST
          window.isRemoteOrigin = (window.location.host === window.$HOST)
          window.isDemoOrigin = window.location.host === window.$COZYGIANT_DEMO_HOST && window.$ENVIRONMENT !== 'local'
          window.isAdminOrigin = window.location.host === window.$COZYGIANT_ADMIN_HOST && window.$ENVIRONMENT !== 'local'
        }
        getAccessToken() {
          let url = new URL(window.location.href)
          let searchParams = new URLSearchParams(url.search)
          return searchParams.get('jwt') || null
        }
        getSessionToken() {
          let jwt = localStorage.getItem('jwt') ? localStorage.getItem('jwt') : null
          let decoded = jwt ? JSON.parse(atob(jwt.split('.')[1])) : null
          return decoded && (decoded.exp > new Date() / 1000) ? jwt : null
        }
        clearAuthToken() {
          let url = new URL(window.location.href)
          let searchParams = new URLSearchParams(url.search)
          url.searchParams.delete('jwt')
          history.replaceState({}, null, url.href)
        }
      }
      window.$loader = new Loader()
      $loader.init()
