    <!-- UI Logic for Custom Dropdowns -->
    <script>

        document.addEventListener('DOMContentLoaded', function () {

            // Gérer l'ouverture/fermeture des dropdowns
            const wrappers = document.querySelectorAll('.custom-select-wrapper');

            wrappers.forEach(wrapper => {
                const selectBox = wrapper.querySelector('.custom-select');
                const trigger = wrapper.querySelector('.custom-select-trigger');
                const options = wrapper.querySelectorAll('.custom-option');
                const hiddenSelectId = trigger.dataset.for;
                const hiddenSelect = document.getElementById(hiddenSelectId);

                trigger.addEventListener('click', function (e) {
                    // Fermer les autres ouverts
                    document.querySelectorAll('.custom-select').forEach(s => {
                        if (s !== selectBox) s.classList.remove('open');
                    });
                    selectBox.classList.toggle('open');
                    e.stopPropagation();
                });

                options.forEach(option => {
                    option.addEventListener('click', function () {
                        // Mettre à jour la sélection visuelle
                        wrapper.querySelector('.custom-option.selected').classList.remove('selected');
                        this.classList.add('selected');

                        // Mettre à jour le trigger text
                        trigger.querySelector('span').textContent = this.textContent;

                        // Mettre à jour le select caché et déclencher l'événement change
                        const value = this.dataset.value;
                        hiddenSelect.value = value;
                        hiddenSelect.dispatchEvent(new Event('change'));

                        // Fermer
                        selectBox.classList.remove('open');
                    });
                });
            });

            // Fermer si on clique ailleurs
            document.addEventListener('click', function () {
                document.querySelectorAll('.custom-select').forEach(s => s.classList.remove('open'));
            });
        });
    </script>
