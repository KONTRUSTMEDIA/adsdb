      <script>
        new Vue({
          el: "#app",
          data: {
            orShowNav: false,
            orShowSearch: false,
            inputValue: "",
            allList: [],
            list: [],
            newgame: [],
            hotgame: [],
            date: "",
            searchSection: true, //展示分类
            languageShow: false,
          },
          mounted() {
            this.getData();
            this.date = new Date().getFullYear();
          },
          methods: {
            showLanguage() {
              this.languageShow = !this.languageShow;
            },
            getData() {
              this.allList = embedgames;
              this.list = this.allList;
              this.hotgame = this.list.sort(() => 0.5 - Math.random()).slice(0, 60);
              this.newgame = this.list.sort(() => 0.5 - Math.random()).slice(0, 60);
            },
            toList() {
              window.location.href = "./allList.html";
            },
            goToDetail(item) {
              var tmp = item.title.replace(/^\d+-/, "-").replace(/ /g, "-");
              window.location.href = "./detail.html?id=" + item.id + "&title=" + tmp;
            },
            handleInput(event) {
              // 获取输入框的内容
              const inputValue = event.target.value;
              this.inputValue = inputValue;
            },
            handleEnter(event) {
              if (event.key === "Enter") {
                this.searchFn();
              }
            },
            searchFn() {
              const query = this.inputValue.trim().toLowerCase();
              if (this.inputValue !== "" || !this.inputValue) {
                console.log("Search for: ", this.inputValue);

                this.list = this.allList.filter((item) => {
                  const title = item.title ? String(item.title) : "";
                  return title.toLowerCase().includes(query) || (item.tags && item.tags.some((tag) => tag.name.includes(query)));
                });
                this.searchSection = false;
              } else {
                this.list = this.allList;
              }
              if (this.inputValue == "" || !this.inputValue) {
                this.searchSection = true;
              }
            },
          },
        });
      </script>
