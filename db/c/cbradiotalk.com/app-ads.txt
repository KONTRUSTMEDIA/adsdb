<script type="text/javascript">
/**
 * This script ensures that:
 * - Internal links (links to the same domain) open in the current Chrome app window.
 * - External links (links to other domains) open in a new browser tab.
 * - JavaScript links (e.g., "javascript:..." or "#") are not affected and behave as intended.
 * 
 * Author: James Wadsworth
 * Website: https://www.cbradiotalk.com
 * Date: January 24, 2025
 */

document.addEventListener('DOMContentLoaded', () => {
    // Intercept all clicks on the page
    document.body.addEventListener('click', (event) => {
        let target = event.target;

        // Traverse up the DOM to ensure the target is an anchor (<a>) element
        while (target && target.tagName !== 'A') {
            target = target.parentElement;
        }

        // If no anchor (<a>) element was found, exit
        if (!target || !target.href) return;

        // Bypass JavaScript links and "#" links
        if (
            target.href.startsWith('javascript:') || 
            target.getAttribute('href') === '#'
        ) {
            return; // Let these links behave as they normally would
        }

        // Get the current domain and account for both "www" and non-"www"
        const currentDomain = window.location.hostname.replace('www.', '');
        const targetDomain = new URL(target.href).hostname.replace('www.', '');

        if (currentDomain === targetDomain) {
            // Internal link: Allow navigation in the same window
            event.preventDefault();
            window.location.href = target.href; // Navigate normally
        }
    });
});

/**
 * Capitalize Words in Input Fields
 * 
 * 1. capitalizeWords(inputField):
 *    - Capitalizes the first letter of each word in the input field's value.
 *    - Converts the rest of the letters in each word to lowercase.
 *    - Updates the input field with the modified value.
 * 
 * 2. Event Listener on DOMContentLoaded:
 *    - Targets specific input fields by their IDs: "pf_name", "pf_txtcbhandle", "pf_antenna", "pf_radio".
 *    - Adds a "change" event listener to each field.
 *    - Automatically capitalizes words when the input field's value changes.
 */

function capitalizeWords(inputField) {
    var inputValue = inputField.value;

    // Split the input value into an array of words
    var words = inputValue.split(" ");

    // Capitalize the first letter of each word
    for (var j = 0; j < words.length; j++) {
        words[j] = words[j].charAt(0).toUpperCase() + words[j].slice(1).toLowerCase();
    }

    // Join the capitalized words back into a single string
    var capitalizedValue = words.join(" ");

    // Set the capitalized value back into the input field
    inputField.value = capitalizedValue;
}

document.addEventListener("DOMContentLoaded", function() {
    var fieldIds = ["pf_name", "pf_txtcbhandle", "pf_antenna", "pf_radio"];

    // Add change event listeners to each field
    fieldIds.forEach(function(fieldId) {
        var inputField = document.getElementById(fieldId);
        if (inputField) {
            inputField.addEventListener("change", function() {
                capitalizeWords(inputField);
            });
        }
    });
});

/**
 * Smooth Anchor Scrolling Script
 * - Enables smooth scrolling to anchor links with a fixed offset.
 * - Updates the browser URL without reloading the page.
 * - Handles URL hash changes and anchor link clicks.
 */
(function(document, history, location) {
  // Check if browser history and pushState are supported
  var HISTORY_SUPPORT = !!(history && history.pushState);

  // Create an object for handling anchor scrolling
  var anchorScrolls = {
    // Regular expression to match anchor links (e.g., "#example")
    ANCHOR_REGEX: /^#[^ ]+/,
    // Offset height in pixels to adjust the scroll position when jumping to anchors
    OFFSET_HEIGHT_PX: 100,

    // Initialization function
    init: function() {
      // Scroll to the current anchor on page load
      this.scrollToCurrent();
      // Listen for changes in the URL hash (e.g., when clicking on an anchor link)
      window.addEventListener('hashchange', this.scrollToCurrent.bind(this));
      // Listen for clicks on anchor links within the document body
      document.body.addEventListener('click', this.delegateAnchors.bind(this));
    },

    // Get the fixed offset for scrolling
    getFixedOffset: function() {
      return this.OFFSET_HEIGHT_PX;
    },

    // Scroll to the anchor if the URL hash matches an existing anchor
    scrollIfAnchor: function(href, pushToHistory) {
      var match, rect, anchorOffset;

      // Check if the href matches the anchor regex
      if (!this.ANCHOR_REGEX.test(href)) {
        return false;
      }

      // Find the element with the matching ID
      match = document.getElementById(href.slice(1));

      if (match) {
        // Get the position and dimensions of the element
        rect = match.getBoundingClientRect();
        // Calculate the new scroll position
        anchorOffset = window.pageYOffset + rect.top - this.getFixedOffset();
        // Scroll to the new position
        window.scrollTo(window.pageXOffset, anchorOffset);

        // If history support is available and pushToHistory is true, update the URL
        if (HISTORY_SUPPORT && pushToHistory) {
          history.pushState({}, document.title, location.pathname + href);
        }
      }

      // Return true if a match was found and scrolled to, otherwise false
      return !!match;
    },

    // Scroll to the current anchor when the page loads
    scrollToCurrent: function() {
      this.scrollIfAnchor(window.location.hash);
    },

    // Delegate anchor link clicks to scroll to the target anchor
    delegateAnchors: function(e) {
      var elem = e.target;

      if (
        elem.nodeName === 'A' &&
        this.scrollIfAnchor(elem.getAttribute('href'), true)
      ) {
        e.preventDefault();
      }
    }
  };

<script>
/**
 * Progressive Web App (PWA) Install Prompt Script
 * 
 * - Handles the "Add to Home Screen" (A2HS) prompt for devices with a screen width of 500px or smaller.
 * 
 * Features:
 * 1. **Install Prompt Display**:
 *    - Shows the PWA install menu (`pwaandroidinsidemenu`) if the user hasn't declined the prompt and the app is not in standalone mode.
 * 
 * 2. **Install Button**:
 *    - Triggers the A2HS prompt when clicked and tracks user choice (accept or dismiss).
 *    - Saves dismissal in `localStorage` to prevent repeated prompts.
 * 
 * 3. **"Not Now" Button**:
 *    - Hides the prompt and sets a cooldown period of 3 days using `localStorage`.
 * 
 * 4. **Scroll Trigger**:
 *    - Automatically shows the prompt after the user scrolls 2200px, provided conditions are met.
 * 
 * 5. **Event Listeners**:
 *    - `beforeinstallprompt`: Captures the install event and initializes the deferred prompt.
 *    - `scroll`: Checks for auto-prompting on scroll.
 *    - `appinstalled`: Marks the app as installed in `localStorage`.
 *    - `load`: Ensures the prompt is shown on page load if applicable.
 */

if (window.innerWidth <= 500) {
    let deferredPrompt;
    let autoPromptShown = false;
    const pwaAndroidInsideMenu = document.querySelector('.pwaandroidinsidemenu');
    const installButton = document.querySelector('.pwainstall');
    const notNowButton = document.querySelector('.pwanotnow');
    const pwaAndroidDot = document.querySelector('.pwaandroiddot');

    function checkInstallPrompt() {
        if (!localStorage.getItem('pwaDeclineClicked') && !window.matchMedia('(display-mode: standalone)').matches) {
            if (deferredPrompt) {
                pwaAndroidInsideMenu.classList.add('show');
                pwaAndroidDot.style.display = 'block';
            }
        }
    }

    function promptInstall() {
        pwaAndroidInsideMenu.classList.remove('show');
        deferredPrompt.prompt();
        deferredPrompt.userChoice.then((choiceResult) => {
            if (choiceResult.outcome === 'accepted') {
                console.log('User accepted the A2HS prompt');
            } else {
                console.log('User dismissed the A2HS prompt');
                localStorage.setItem('pwaDeclineClicked', 'true');
            }
            deferredPrompt = null;
        });
    }

    window.addEventListener('beforeinstallprompt', (e) => {
        if (!localStorage.getItem('pwaDeclineClicked')) {
            deferredPrompt = e;
            checkInstallPrompt();
        }

        installButton.addEventListener('click', (e) => {
            promptInstall();
        });
    });

    notNowButton.addEventListener('click', () => {
        const date = new Date();
        date.setDate(date.getDate() + 3);
        localStorage.setItem('pwaDeclineClicked', 'true');
        localStorage.setItem('pwaDeclineTime', date.getTime());

        pwaAndroidInsideMenu.classList.remove('show');
        pwaAndroidDot.style.display = 'none';
    });

    window.addEventListener('scroll', () => {
        if (!autoPromptShown && window.scrollY > 2200 && deferredPrompt && !localStorage.getItem('pwaDeclineClicked')) {
            autoPromptShown = true;
            promptInstall();
        }
    });

    window.addEventListener('load', checkInstallPrompt);

    window.addEventListener('appinstalled', () => {
        localStorage.setItem('pwaDeclineClicked', 'true');
    });
}
