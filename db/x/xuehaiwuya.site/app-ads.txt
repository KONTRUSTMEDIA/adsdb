<script>
    // Canvas初始化
    const canvas = document.getElementById('canvas');
    const ctx = canvas.getContext('2d');
    let width = canvas.width = window.innerWidth;
    let height = canvas.height = window.innerHeight;
    window.addEventListener('resize', ()=>{
        width = canvas.width = window.innerWidth;
        height = canvas.height = window.innerHeight;
    });

    // 控件
    const countRange = document.getElementById('countRange');
    const countValue = document.getElementById('countValue');
    const speedRange = document.getElementById('speedRange');
    const speedValue = document.getElementById('speedValue');
    const windRange = document.getElementById('windRange');
    const windValue = document.getElementById('windValue');

    let snowCount = parseInt(countRange.value);
    let speedMultiplier = parseFloat(speedRange.value);
    let wind = parseFloat(windRange.value);

    countRange.addEventListener('input', ()=>{ snowCount=parseInt(countRange.value); countValue.textContent=snowCount; initParticles(); });
    speedRange.addEventListener('input', ()=>{ speedMultiplier=parseFloat(speedRange.value); speedValue.textContent=speedMultiplier; });
    windRange.addEventListener('input', ()=>{ wind=parseFloat(windRange.value); windValue.textContent=wind; });

    // FPS
    let lastTime = performance.now(), frames = 0;
    function updateFPS() {
        const now = performance.now();
        frames++;
        if(now-lastTime>=1000){
            document.getElementById('fps').textContent = "FPS: "+frames;
            frames=0;
            lastTime=now;
        }
    }

    // 生成雪花纹理canvas
    function createSnowflakeTexture(size){
        const c=document.createElement('canvas');
        c.width=c.height=size;
        const ctx2=c.getContext('2d');
        const mid=size/2;
        ctx2.strokeStyle='rgba(255,255,255,0.9)';
        ctx2.lineWidth=1;
        for(let i=0;i<6;i++){
            ctx2.beginPath();
            ctx2.moveTo(mid, mid);
            ctx2.lineTo(mid, 0);
            // 随机分叉
            ctx2.moveTo(mid, mid*0.5);
            ctx2.lineTo(mid*0.8, mid*0.25);
            ctx2.moveTo(mid, mid*0.5);
            ctx2.lineTo(mid*1.2, mid*0.25);
            ctx2.stroke();
            ctx2.translate(mid, mid);
            ctx2.rotate(Math.PI/3);
            ctx2.translate(-mid, -mid);
        }
        return c;
    }

    // 离屏纹理缓存
    const textures = [];
    function generateTextures(){
        textures.length=0;
        for(let s=4;s<=12;s+=2){ // 不同尺寸雪花纹理
            textures.push(createSnowflakeTexture(s));
        }
    }
    generateTextures();

    // 雪花粒子
    let particles=[];
    function initParticles(){
        particles=[];
        for(let i=0;i<snowCount;i++){
            const layer=Math.random()<0.33?0:(Math.random()<0.5?1:2);
            const texture = textures[Math.floor(Math.random()*textures.length)];
            const speed=0.5+Math.random()*1.5;
            particles.push({
                x: Math.random()*width,
                y: Math.random()*height,
                size: texture.width,
                texture: texture,
                angle: Math.random()*Math.PI*2,
                rotationSpeed: (Math.random()*0.02+0.005)*(Math.random()<0.5?-1:1),
                speed: speed*speedMultiplier,
                layer: layer,
                sway: Math.random()*0.5,
            });
        }
    }
    initParticles();

    // 渲染循环
    function drawFrame(){
        requestAnimationFrame(drawFrame);
        updateFPS();

        ctx.fillStyle="#000";
        ctx.fillRect(0,0,width,height);

        for(let p of particles){
            p.y += p.speed*speedMultiplier;
            p.x += Math.sin(p.y*0.01)*p.sway + wind;

            if(p.y>height) p.y=0;
            if(p.x>width) p.x=0;
            if(p.x<0) p.x=width;

            ctx.save();
            ctx.translate(p.x,p.y);
            ctx.rotate(p.angle);
            ctx.drawImage(p.texture, -p.size/2, -p.size/2, p.size, p.size);
            ctx.restore();

            p.angle += p.rotationSpeed;
        }
    }

    drawFrame();
</script>
