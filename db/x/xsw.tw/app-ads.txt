<script>
    let countdown = 3;
    let timerId = null;

    // 启动倒计时
    function startCountdown() {
        timerId = setInterval(() => {
            countdown--;
            document.getElementById('timer').textContent =
                `${countdown}秒後自動跳轉`;

            if (countdown <= 0) {
                clearInterval(timerId);
                redirectHome();
            }
        }, 1000);
    }

    // 重定向函数
    function redirectHome() {
        clearInterval(timerId); // 清除定时器
        window.location.href = "/";
    }

    // 页面加载时启动倒计时
    window.onload = startCountdown;
</script>
