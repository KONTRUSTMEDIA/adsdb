<script>
(function(){
  try{
    var u = new URL(window.location.href);
    var st  = u.searchParams.get('_st');
    var sig = u.searchParams.get('_sig');
    var lid = parseInt(u.searchParams.get('lid') || '0', 10);
    if(!st || !sig){ return; } // nothing to report

    var base = 'https://botcops.us/landed.php';
    var qs   = '_st=' + encodeURIComponent(st) + '&_sig=' + encodeURIComponent(sig) + '&lid=' + (isFinite(lid)?lid:0);
    var url  = base + '?' + qs;

    // Optional lightweight nav hints (will be ignored by server if not useful)
    var navEntry = (performance.getEntriesByType && performance.getEntriesByType('navigation')[0]) || null;
    var conn = navigator.connection || navigator.mozConnection || navigator.webkitConnection || null;
    var body = JSON.stringify({
      nav: {
        nav: {
          nav_type: navEntry ? navEntry.type : null,
          next_hop: navEntry ? navEntry.nextHopProtocol : null,
          dcl_ms: navEntry ? Math.round(navEntry.domContentLoadedEventEnd - navEntry.startTime) : null
        },
        net: conn ? { et: conn.effectiveType || null, rtt: conn.rtt || null, dl: conn.downlink || null } : null
      }
    });

    // 1) sendBeacon with QUERY PARAMS (critical) + optional body
    if (navigator.sendBeacon) {
      var blob = new Blob([body], {type:'application/json'}); // ok if UA downgrades to text/plain; URL carries the IDs
      if (navigator.sendBeacon(url, blob)) return;
    }

    // 2) fetch keepalive (still keeps IDs in URL)
    try{
      fetch(url, {method:'POST', mode:'no-cors', keepalive:true, headers:{'Content-Type':'application/json'}, body: body})
        .catch(function(){});
    }catch(_){}

    // 3) Pixel fallback — least likely to be blocked
    var img = new Image();
    img.src = base + '?gif=1&' + qs + '&t=' + Date.now();
  }catch(_){}
})();
</script>
