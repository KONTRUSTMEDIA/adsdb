	function userPollAnswer(answer, ques_id) {
		var user_id = user_info.id;
		var answer_id = answer;
		var ques_id = ques_id;
		$.ajax({
			method: 'POST',
			url: SITEURL + '/api/v1/post-poll-answer',
			dataType: 'json',
			data: {
				'user_id': user_id,
				"ans_option": answer_id,
				"poll_question_id": ques_id
			},
			headers: {
				'Authorization': token,
				'Accept': 'application/json'
			},
		}).done(function(response) {
			if (response.success == true) {
				$.notify({
					message: response.message
				}, {
					type: "success",
					z_index: 2147483647
				});
			}
		});
	}
	/*======================= End Poll Answer Submit ================================*/

	/*======================= For Check User Voted Or Not ================================*/

	function userAlreadyPolled(answer, ques_id) {
		var user_id = user_info.id;
		var answer_id = answer;
		var ques_id = ques_id;

		$.ajax({
			method: 'GET',
			url: SITEURL + '/api/v1/check-polled?user_id=' + user_id + '&poll_question_id=' + ques_id,
			dataType: 'json',
			headers: {
				'Authorization': token,
				'Accept': 'application/json'
			},
		}).done(function(response) {
			if (response.success == true) {
				$.notify({
					message: response.message
				}, {
					type: "danger",
					z_index: 2147483647
				});
			} else {
				userPollAnswer(answer_id, ques_id);
			}
		});
	}
