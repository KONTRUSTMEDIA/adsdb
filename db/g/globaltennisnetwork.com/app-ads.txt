		function renderPlayerNotifications(markAllRead) {
			if( !markAllRead && jQuery("#unreadNotificationsDiv").hasClass("populated") ) {
				return;
			}

			var dataString = '';

			if( markAllRead ) {
				dataString = 'markAllRead=1';

				jQuery("#unreadNotificationsDiv").html('<div style="padding-top:10px; width:100%;"><center><img src="/images/ajax-loader.gif"></center></div>');
			}

			jQuery.ajax({
				url: "/index.php?option=com_notifications&controller=notifications&task=printUnreadNotifications&format=raw",
				type: "GET",
				data: dataString,

				success: function (html) {
					jQuery("#unreadNotificationsDiv").html(html);

					jQuery("#unreadNotificationsDiv").addClass("populated");

					updateNotificationCount();
				},
				error: function () {
					console.log("Menu could not be generated.");
				}
			});
		}

		function updateNotificationCount()
		{
			var url  = "/index.php?option=com_notifications&format=raw&controller=notifications&task=unreadNotificationsCount";

			jQuery.ajax({
				url: url,
				cache: false,
				type: "GET",
				success: function(count) {
					var paCount = 0;
					var naCount = Number(count);
					var totalCount = paCount + naCount;

					if( Number(totalCount) > 0 ) {
						jQuery('#pan-count').show();

						jQuery('#pan-count').html(totalCount);
					}else{
						jQuery('#pan-count').hide();
					}

					if( Number(naCount) > 0 ) {
						jQuery('#n-badge').show();

						jQuery('#n-badge-count').html(naCount);
					}else{
						jQuery('#n-badge').hide();
					}
				}
			});
		}

		function renderCreateMenu() {
			if( jQuery("#createMenu").hasClass("populated") ) {
				return;
			}

			jQuery.ajax({

				type: "GET",

				success: function (html) {
					jQuery("#createMenu").html(html);

					jQuery("#createMenu").addClass("populated");
				},
				error: function () {
					console.log("Menu could not be generated.");
				}
			});
		}

		function renderPlayerMenu(isRefresh) {
			if( !isRefresh && jQuery("#playerMenu").hasClass("populated") ) {
				return;
			}

			if( document.getElementById('menuActivityType') && document.getElementById('menuActivityRole') ) {
				dataString = 'activityType=' + document.getElementById('menuActivityType').value + '&activityRole=' + document.getElementById('menuActivityRole').value;
			}
	
			jQuery.ajax({
				url: "/index.php?option=com_players&controller=menus&task=printPlayerMenu&format=raw",

		<script>
			function showToast(level, message) {
				// Levels are: success, error, warning

				var title = 'Success';
				var icon = 'fa-check-circle text-success';
				var delay = 4000;

				if (level == 'error') {
					title = 'Error';
					var icon = 'fa-times-circle text-danger';
					var delay = 7000;
				} else if (level == 'warning') {
					title = 'Warning';
					var icon = 'fas fa-exclamation-circle text-warning';
				} else if (level == 'alert') {
					var delay = 10000;
				} else if (level == 'notice') {
					var delay = 2000;
				}

				var toastElement = document.createElement('div');
				toastElement.className = 'toast shadow-lg mb-3 mt-3 me-3';
				toastElement.setAttribute('role', 'alert');
				toastElement.setAttribute('aria-live', 'assertive');
				toastElement.setAttribute('aria-atomic', 'true');

				if( level == 'alert' ) {
					toastElement.innerHTML = `
						<div class="align-items-center" style="font-size:16px; background-color: #fff;">
							<div class="d-flex toast-body justify-content-between">
								<div>
									${message}
								</div>
								<button type="button" class="btn-close" data-bs-dismiss="toast" aria-label="Close"></button>
							</div>
						</div>
					`;
				} else if( level == 'notice' ) {
					toastElement.innerHTML = `
						<div class="align-items-center" style="font-size:16px; background-color: #fff;">
							<div class="toast-body">
								<div>
									${message}
								</div>
							</div>
						</div>
					`;
				} else {
					toastElement.innerHTML = `
						<div class="toast-header" style="font-size:16px; background-color: #fff; border-bottom: 1px solid #d8d8d8;">
							<span class="fas ${icon} me-2"></span>
							<strong class="me-auto">${title}</strong>
							<small class="text-muted">just now</small>
							<button type="button" class="btn-close" data-bs-dismiss="toast" aria-label="Close"></button>
						</div>
						<div class="toast-body" style="font-size:16px; background-color: #fff;">
							${message}
						</div>
					`;
				}

				// Append the toast element to the container
				document.getElementById('toast-messages').appendChild(toastElement);

				// Initialize and show the toast
				var toast = new bootstrap.Toast(toastElement, {
					autohide: true,
					delay: delay // Adjust delay as needed
				});

				toast.show();
			}
		</script>
