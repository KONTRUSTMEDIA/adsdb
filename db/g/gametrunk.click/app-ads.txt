      <script>
        new Vue({
          el: "#app",
          data: {
            allList: [],
            list: [],
            game: [],
            longimageList: [],
            date: "",
            menuModalVisible: false,
            searchModalVisible: false,
            searchQuery: "",
            searchResults: [],
            startX: 0,
            scrollLeft: 0,
            isDragging: false,
            recommendList: [
              { title: "🔥 Trending Games", icon: "./assets/TowerDefense.svg" },
              { title: "⭐ Featured Games", icon: "./assets/Stickman.svg" },
              { title: "💡 DIscover Games", icon: "./assets/Flash.svg" },
            ],
            categoryList: ["Home", "Mobile", "1 Player", "Girls", "Puzzle"],
            videoList: [
              {
                title: "City Stunts",
                image: "https://www.onlinegames.io/media/posts/629/responsive/City-Stunts-Game-xs.jpg",
                video: "./assets/City Stunts-611x343_30fps.mp4",
              },
              {
                title: "Survival Craft",
                image: "https://www.onlinegames.io/media/posts/593/responsive/Survival-Craft-Game-xs.jpg",
                video: "./assets/Survival Craft-611x343_30fps.mp4",
              },
              {
                title: "The Farmer",
                image: "https://www.onlinegames.io/media/posts/533/responsive/The-Farmer-xs.jpg",
                video: "./assets/The Farmer-611x343_30fps.mp4",
              },

              {
                title: "Four Colors",
                image: "https://www.onlinegames.io/media/posts/535/responsive/Four-Colors-xs.jpg",
                video: "./assets/Four Colors-611x343_30fps.mp4",
              },
            ],
          },
          mounted() {
            // 监听滚动事件，控制按钮显示与隐藏
            window.addEventListener("scroll", this.handleScroll);
            this.getData();
            this.date = new Date().getFullYear();
            this.videoList.forEach((video) => {
              const matchedGame = embedgames.find((game) => game.title == video.title);
              if (matchedGame) {
                video.id = matchedGame.id;
              }
            });
            this.longimageList.forEach((long) => {
              const matchedGame = embedgames.find((game) => long.title == game.title);
              if (matchedGame) {
                long.id = matchedGame.id;
              }
            });
          },

          beforeDestroy() {
            // 组件销毁前移除滚动事件监听
            window.removeEventListener("scroll", this.handleScroll);
          },
          methods: {
            showSearchModal() {
              window.location.href = "./search.html";
            },
            hideSearchModal() {
              this.searchModalVisible = false;
              document.body.style.overflow = "";
            },
            showMenuModal() {
              this.menuModalVisible = true;
              document.body.style.overflow = "hidden";
            },
            hideMenuModal() {
              this.menuModalVisible = false;
              document.body.style.overflow = "";
            },
            performSearch() {
              if (this.searchQuery.trim() === "") {
                this.searchResults = this.allList;
                return;
              }
              const query = this.searchQuery.toLowerCase();
              this.searchResults = this.allList.filter(
                (game) => game.title.toLowerCase().includes(query) || (game.tags && game.tags.some((tag) => tag.toLowerCase().includes(query)))
              );
            },
            goToGameDetail(game) {
              this.hideSearchModal();
              this.goToDetail(game);
            },
            handleTouchStart(e, index) {
              this.startX = e.touches[0].clientX;
              this.scrollLeft = this.$refs["scrollContainer" + index][0].scrollLeft;
              this.isDragging = true;
            },
            handleTouchMove(e, index) {
              if (!this.isDragging) return;
              const x = e.touches[0].clientX;
              const walk = (x - this.startX) * 2;
              this.$refs["scrollContainer" + index][0].scrollLeft = this.scrollLeft - walk;
            },
            handleTouchEnd(index) {
              this.isDragging = false;
            },
            toIndex() {
              window.location.href = "./index.html";
            },
            // 处理滚动事件
            handleScroll() {
              const backToTopButton = document.getElementById("back-to-top");
              if (document.body.scrollTop > 20 || document.documentElement.scrollTop > 20) {
                backToTopButton.style.display = "flex";
              } else {
                backToTopButton.style.display = "none";
              }
            },
            // 回到顶部
            scrollToTop() {
              window.scrollTo({
                top: 0,
                behavior: "smooth",
              });
            },
            getData() {
              this.allList = embedgames.sort(() => 0.5 - Math.random());
              this.list = this.allList;
              this.game = games;
              this.longimageList = longgames.sort(() => 0.5 - Math.random());
            },
            toList(name) {
              if (name) {
                if (name == "Home") {
                  return (window.location.href = "./index.html");
                }
                window.location.href = "./allList.html?query=" + name;
              } else {
                window.location.href = "./allList.html";
              }
            },
            goToDetail(item) {
              var tmp = item.title.replace(/^\d+-/, "-").replace(/ /g, "-");
              window.location.href = "./detail.html?id=" + item.id + "&title=" + tmp;
            },
          },
        });
      </script>
