function showFavoritesModal() {
    const favorites = getFavorites();
    
    if (favorites.length === 0) {
        showNotification('Нямате любими канали', 'info');
        return;
    }
    
    // Create modal
    const modal = document.createElement('div');
    modal.className = 'favorites-modal';
    modal.style.cssText = `
        position: fixed;
        top: 0;
        left: 0;
        width: 100%;
        height: 100%;
        background: rgba(0, 0, 0, 0.8);
        backdrop-filter: blur(5px);
        display: flex;
        justify-content: center;
        align-items: center;
        z-index: 9999;
        animation: fadeIn 0.3s ease;
    `;
    
    const content = document.createElement('div');
    content.style.cssText = `
        background: var(--card-bg);
        border-radius: var(--radius);
        padding: 2rem;
        max-width: 800px;
        width: 90%;
        max-height: 80vh;
        overflow-y: auto;
        box-shadow: var(--shadow-lg);
    `;
    
    content.innerHTML = `
        <div style="display: flex; justify-content: space-between; align-items: center; margin-bottom: 1.5rem;">
            <h2 style="color: var(--primary-color); margin: 0;">
                <i class="fas fa-heart" style="margin-right: 10px;"></i>
                Любими Канали (${favorites.length})
            </h2>
            <button id="close-modal" style="
                background: none;
                border: none;
                font-size: 1.5rem;
                cursor: pointer;
                color: var(--text-light);
                padding: 5px;
            ">&times;</button>
        </div>
        <div id="favorites-list" style="display: grid; grid-template-columns: repeat(auto-fill, minmax(200px, 1fr)); gap: 1rem;"></div>
        <div style="margin-top: 2rem; text-align: center;">
            <button id="clear-favorites" style="
                padding: 8px 16px;
                background: var(--primary-transparent);
                color: var(--primary-color);
                border: 1px solid var(--primary-light);
                border-radius: 8px;
                cursor: pointer;
                transition: var(--transition);
                font-size: 14px;
            ">
                <i class="fas fa-trash-alt" style="margin-right: 5px;"></i>
                Изчисти всички любими
            </button>
        </div>
    `;
    
    modal.appendChild(content);
    document.body.appendChild(modal);
    
    // Populate favorites list
    const favoritesList = document.getElementById('favorites-list');
    
    // Function to render favorites list
    function renderFavoritesList(currentFavorites) {
        favoritesList.innerHTML = '';
        
        currentFavorites.forEach(fav => {
            const channelCard = document.createElement('div');
            channelCard.className = 'favorite-channel-item';
            channelCard.style.cssText = `
                background: var(--primary-transparent);
                border-radius: var(--radius);
                padding: 1rem;
                border: 1px solid var(--border-color);
                transition: var(--transition);
                cursor: pointer;
            `;
            
            // Use the correct URL from your aurl() function
            const imageUrl = fav.image ? 
                'https://www.gledaitv.live/assets/images/channels' + fav.image : 
                'https://www.gledaitv.live/assets/images/channels/default.png';
            
            channelCard.innerHTML = `
                <div style="display: flex; align-items: center; gap: 1rem;">
                    <img src="${imageUrl}" 
                         alt="${fav.title}" 
                         style="width: 50px; height: 50px; border-radius: 8px; object-fit: cover;"
                         onerror="this.src='https://www.gledaitv.live/assets/images/channels/default.png'">
                    <div style="flex: 1;">
                        <h4 style="margin: 0 0 5px 0; font-size: 14px; color: var(--text-dark);">${fav.title}</h4>
                        <span style="font-size: 12px; color: var(--primary-color); background: rgba(150, 34, 193, 0.1); padding: 2px 8px; border-radius: 12px;">${fav.category}</span>
                    </div>
                    <button class="remove-favorite" data-id="${fav.id}" style="
                        background: none;
                        border: none;
                        color: var(--text-light);
                        cursor: pointer;
                        padding: 5px;
                        border-radius: 50%;
                        transition: var(--transition);
                        z-index: 2;
                        position: relative;
                    ">
                        <i class="fas fa-times"></i>
                    </button>
                </div>
            `;
            
            // Add click to watch
            channelCard.addEventListener('click', function(e) {
                // Don't trigger if clicking the remove button
                if (e.target.closest('.remove-favorite')) {
                    return;
                }
                
                // Generate URL for watch page
                const slug = fav.title.toLowerCase()
                    .replace(/[^a-z0-9а-я]+/g, '-')
                    .replace(/^-+|-+$/g, '');
                
                window.location.href = 'https://www.gledaitv.live/watch-tv' + fav.id + '/' + slug;
            });
            
            // Add remove button click handler
            const removeBtn = channelCard.querySelector('.remove-favorite');
            removeBtn.addEventListener('click', function(e) {
                e.stopPropagation();
                
                const channelId = parseInt(this.getAttribute('data-id'));
                
                let updatedFavorites = getFavorites();
                updatedFavorites = updatedFavorites.filter(fav => fav.id !== channelId);
                saveFavorites(updatedFavorites);
                
                // Re-render the list
                renderFavoritesList(updatedFavorites);
                
                // Update title counter
                const title = content.querySelector('h2');
                if (title) {
                    title.innerHTML = `<i class="fas fa-heart" style="margin-right: 10px;"></i>
                        Любими Канали (${updatedFavorites.length})`;
                }
                
                // Update all favorite buttons on the page
                initializeFavoriteButtons();
                
                // Dispatch update event
                window.dispatchEvent(new CustomEvent('favoritesUpdated', {
                    detail: { favorites: updatedFavorites }
                }));
                
                // If no favorites left, close modal
                if (updatedFavorites.length === 0) {
                    setTimeout(() => {
                        modal.remove();
                        showNotification('Нямате любими канали', 'info');
                    }, 500);
                } else {
                    showNotification('Каналът е премахнат от любими', 'info');
                }
            });
            
            favoritesList.appendChild(channelCard);
        });
    }
    
    // Initial render
    renderFavoritesList(favorites);
    
    // Close modal
    document.getElementById('close-modal').addEventListener('click', function() {
        modal.remove();
    });
    
    // Clear all favorites
    document.getElementById('clear-favorites').addEventListener('click', function() {
        if (confirm('Сигурни ли сте, че искате да изтриете всички любими канали?')) {
            deleteCookie('tv_favorites');
            showNotification('Всички любими канали са изтрити', 'info');
            modal.remove();
            initializeFavoriteButtons(); // Update all buttons
            window.dispatchEvent(new CustomEvent('favoritesUpdated', {
                detail: { favorites: [] }
            }));
        }
    });
    
    // Close on ESC
    document.addEventListener('keydown', function closeOnEsc(e) {
        if (e.key === 'Escape') {
            modal.remove();
            document.removeEventListener('keydown', closeOnEsc);
        }
    });
    
    // Close on background click
    modal.addEventListener('click', function(e) {
        if (e.target === modal) {
            modal.remove();
        }
    });
}
