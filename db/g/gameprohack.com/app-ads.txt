                announcementBoard.addClass('minimized');
                announcementBody.slideUp(300);
                expandBtn.show();
                localStorage.setItem('announcementBoardState', 'minimized');
            }
        });
        
        // Genişlet butonu (alt kısımdan)
        expandBtn.on('click', function() {
            announcementBoard.removeClass('minimized');
            announcementBody.slideDown(300);
            expandBtn.hide();
            localStorage.setItem('announcementBoardState', 'expanded');
        });
        });
    });
</script>
