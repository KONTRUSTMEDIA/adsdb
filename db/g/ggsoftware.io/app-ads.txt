    function populateInsight(prefix, total, percent) {
            const insightEl = document.getElementById(`${prefix}-insight`);
            if (insightEl) {
                const revEl = document.getElementById(`${prefix}-revenue`);
                if (revEl) {
                    revEl.innerHTML = Number(total).toFixed(2);
                }

                const deltaEl = document.getElementById(`${prefix}-delta`);
                const percentEl = document.getElementById(`${prefix}-percentage`);

                if (percentEl && deltaEl) {
                    percentEl.innerHTML = Number(percent).toFixed(2);

                    const upEl = document.getElementById(`${prefix}-rev-up`);
                    const downEl = document.getElementById(`${prefix}-rev-down`);

                    if (percent === 0) {
                        upEl && upEl.remove();
                        downEl && downEl.remove();
                    } else if (percent > 0) {
                        deltaEl.className = deltaEl.className + ' pos';
                        downEl && downEl.remove();
                        upEl && upEl.classList.remove('hidden');
                    } else {
                        deltaEl.className = deltaEl.className + ' neg';
                        upEl && upEl.remove();
                        downEl && downEl.classList.remove('hidden');
                    }
                }
            }
        }

        const currentMonth = [];

        /**
         * Weekend insights
         */
        let sunday = null;
        let saturday = null;
        for (let item of data) {
            const d = new Date(item.date);
            const day = d.getDay();

            if (!sunday && day === 0) {
                sunday = item;
            }
            
            if (!saturday && sunday && day === 6) {
                saturday = item;

            if (latestD.getMonth() === d.getMonth()) {

                prevMonth.push(item);

                break;
