   // Hàm để mở iframe ở chế độ toàn màn hình
    function openFullscreen() {
        const iframe = document.querySelector('#loader iframe'); // Tìm iframe trong thẻ #loader
        if (iframe) {
            if (iframe.requestFullscreen) {
                iframe.requestFullscreen();
            } else if (iframe.mozRequestFullScreen) { // Firefox
                iframe.mozRequestFullScreen();
            } else if (iframe.webkitRequestFullscreen) { // Chrome, Safari và Opera

            } else if (iframe.msRequestFullscreen) { // IE/Edge
                iframe.msRequestFullscreen();
            } else {
                alert("Trình duyệt của bạn không hỗ trợ chức năng toàn màn hình.");
            }
        } else {
            console.error("Không tìm thấy thẻ iframe!");
        }
    }

    document.getElementById('fullScreenBtn').addEventListener('click', openFullscreen);

  }

  matched.forEach(game => {
    const a = document.createElement('a');
    a.href = '../../' + game.link;
    a.className = 'search-item';
    a.innerHTML = `
      <img src="../../${game.img}" alt="${game.title}">
      <div class="search-title">${game.title}</div>
    `;
    searchResults.appendChild(a);
  });

  searchResults.style.display = 'block';
});

document.addEventListener('click', function (e) {
  if (!document.querySelector('.game-search-wrap').contains(e.target)) {
    searchResults.style.display = 'none';
  }
});
