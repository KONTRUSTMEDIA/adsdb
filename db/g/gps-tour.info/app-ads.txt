      if (!isSearchActive) {
        isResized = true;
      }
    });
  
  
    document.addEventListener('xtx.search.setOpen', function (e) {
      isSearchActive = e.detail.setOpen;
      if(isResized) {
        setTimeout(function() {
          expandExploder();
          exploder.getMC().getGMap().invalidateSize();
          isResized = false;
        }, 200);
      } else {
        expandExploder();
      }
    });

        if(!isSearchOpen) {
          //document.dispatchEvent(new CustomEvent('xtx.search.setOpen', {detail: {setOpen: true}, bubbles: true}));
          xtxFunctions.dispatchCustomEvent(document, 'xtx.search.setOpen', true, true, {setOpen: true});
        }
        setBrowserPos();
      });
  
      document.querySelector(".filterSearch .xtxInput.location .xtxicon-crosshairs").addEventListener('click', function () {
        setBrowserPos();
      });
    }

    var keydownhandler = function(e) {
      e.target.setAttribute('placeholder', "Ort Suchen");

        input.removeEventListener("keydown", keydownhandler, true);
      }.bind(this));
    };
  
    function setBrowserPos() {
      exploder.setToBroserPos();
  
      Array.prototype.forEach.call (document.querySelectorAll('.xtxInput.location input'), function (input) {
        input.setAttribute('placeholder', "Mein Standort");
        input.value = "";
  
        input.addEventListener('keydown', keydownhandler, true);
      }.bind(this));
    }

                lng: 35
              }
