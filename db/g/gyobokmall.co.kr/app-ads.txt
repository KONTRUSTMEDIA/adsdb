<script>
const TIMEOUT = 0;
const REDIRECT_PREFIX = "https://gyobokmall.co.kr/.well-known/cc-challenge";

function getQueryParams() {
    const {search} = window.location;
    const urlParams = new URLSearchParams(search);
    let url = urlParams.get('url');
    const auth = urlParams.get('auth');
    return { url, auth };
}

function getErrorHTML() {
    return `
        <div id="content">
            <div class="imgWarn img403"><span></span></div>
            <div class="mWarn">
                <div class="inner">
                    <h2>Access Temporarily Restricted</h2>
                    <p class="txtDesc">
                        To ensure a secure and reliable service, your access has been temporarily restricted.<br class="RTB">
                        If you reached this page unexpectedly, please contact our customer support.<br class="RMB">
                        We apologize for any inconvenience.
                    </p>
                    <div class="mButton">
                        <a href="javascript:history.back()" class="btnSubmit large"><span>Go Back to Previous Page</span></a>
                    </div>
                </div>
            </div>
        </div>
    `;
}

function javascriptCallback(challenge_token) {
    let url, auth;
    try {
        ({ url, auth } = getQueryParams());
    } catch (e) {
        document.getElementById('container').innerHTML = getErrorHTML();
        return;
    }

    fetch('/validate', {
        method: 'POST',
        headers: {
            'Content-Type': 'application/json'
        },
        body: JSON.stringify({
            challenge_token, url, auth
        })
    })
    .then(response => {
        if (!response.ok) {
            throw new Error('Request failed');
        }
        return response.json();
    })
    .then(data => {
        if (data.token) {
            setTimeout(() => {
                const redirectUrl = new URL(REDIRECT_PREFIX);
                redirectUrl.searchParams.set('token', data.token);
                redirectUrl.searchParams.set('url', url);
                window.location.href = redirectUrl.toString();
            }, TIMEOUT);
        } else {
            document.getElementById('container').innerHTML = getErrorHTML();
        }
    })
    .catch(error => {
        console.error('Error:', error);
        document.getElementById('container').innerHTML = getErrorHTML();
    });
}

function renderCaptcha() {
    const TURNSTILE_BYPASS_ENABLED = "False"
    if (TURNSTILE_BYPASS_ENABLED === "True") {
        javascriptCallback('')
        return;
    }
    const SITE_KEY = "0x4AAAAAABg1Yv76EgOVwBci"
    const urlParams = new URLSearchParams(window.location.search);
    const size = urlParams.get('size') || 'normal'; // normal, compact, flexible
    const appearance = urlParams.get('appearance') || 'always'; // always, execute, interaction-only

    turnstile.render('#myWidget', {
        sitekey: SITE_KEY,
        theme: 'light',
        callback: javascriptCallback,
        size: size,
        appearance: appearance
    });
}

window.onload = renderCaptcha;
