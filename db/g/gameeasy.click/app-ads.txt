      <script>
        new Vue({
          el: "#app",
          data: {
            orShowNav: false,
            orShowSearch: false,
            inputValue: "",
            allList: [],
            list: [],
            newgame: [],
            hotgame: [],
            date: "",
            searchSection: true, //展示分类
            languageShow: false,
            end1: 10,
          },
          mounted() {
            // 挂载完成后隐藏骨架屏
            document.getElementById("skeleton").style.display = "none";
            // 监听滚动事件，控制按钮显示与隐藏
            window.addEventListener("scroll", this.handleScroll);
            window.addEventListener("resize", this.handleEnd1);
            this.handleEnd1();
            this.getData();
            this.date = new Date().getFullYear();
          },
          beforeDestroy() {
            // 组件销毁前移除滚动事件监听
            window.removeEventListener("scroll", this.handleScroll);
          },
          methods: {
            handleEnd1() {
              if (window.innerWidth > 768) {
                this.end1 = 10;
              } else {
                this.end1 = 6;
              }
            },
            // 处理滚动事件
            handleScroll() {
              const backToTopButton = document.getElementById("back-to-top");
              if (document.body.scrollTop > 20 || document.documentElement.scrollTop > 20) {
                backToTopButton.style.display = "flex";
              } else {
                backToTopButton.style.display = "none";
              }
            },
            // 回到顶部
            scrollToTop() {
              window.scrollTo({
                top: 0,
                behavior: "smooth",
              });
            },
            showLanguage() {
              this.languageShow = !this.languageShow;
            },
            getData() {
              this.allList = games.list.sort(() => 0.5 - Math.random());
              this.list = [...this.allList];
              this.hotgame = this.allList.slice(0, 20);
              this.newgame = this.allList.slice(30, 50);
            },
            toList() {
              window.location.href = "./allList.html";
            },
            goToDetail(item) {
              var tmp = item.title.replace(/^\d+-/, "-").replace(/ /g, "-");
              window.location.href = "./detail.html?id=" + item.id + "&title=" + item.title;
            },
            handleInput(event) {
              // 获取输入框的内容
              const inputValue = event.target.value;
              this.inputValue = inputValue;
            },
            handleEnter(event) {
              if (event.key === "Enter") {
                this.searchFn();
              }
            },
            searchFn() {
              const query = this.inputValue.trim().toLowerCase();
              if (this.inputValue !== "" || !this.inputValue) {
                console.log("Search for: ", this.inputValue);

                this.list = this.allList.filter((item) => {
                  const title = item.title ? String(item.title) : "";
                  return title.toLowerCase().includes(query) || (item.tags && item.tags.some((tag) => tag.includes(query)));
                });
                this.searchSection = false;
              } else {
                this.list = this.allList;
              }
              if (this.inputValue == "" || !this.inputValue) {
                this.searchSection = true;
              }
            },
          },
        });
      </script>
