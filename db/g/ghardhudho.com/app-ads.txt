<script>
    let resendInterval;
    document.addEventListener("DOMContentLoaded", function() {
        window.targetOtpForm = null;
        window.otpPhoneNumber = null;
        let otpRequestId = null;
        const otpInputs = document.querySelectorAll('.otp-digit');
        const resendBtn = document.getElementById('resend-otp-btn');
        const timerDisplay = document.getElementById('resend-timer');

        const msg91Configuration = {
            widgetId: "346578746744373932353738",
            tokenAuth: "371952TpKWsMg066524dcfP1",
            identifier: "",
            exposeMethods: true,
            captchaRenderId: 'captchaBox', // id(must be unique) of html element where to render captcha, only works if there is exposedMethod is true,.
            success: (data) => {
                // console.log('success response', data);
            },
            failure: (error) => {
                // console.log('error response', error);
            },

        };

        msg91Initialize = setInterval(() => {
            if (typeof window.initSendOTP === 'function') {
                clearInterval(msg91Initialize);
                window.initSendOTP(msg91Configuration);
            }
        }, 2000);

        // initSendOTP(msg91Configuration);

        function startResendTimer(duration = 30) {
            let timeLeft = duration;
            resendBtn.disabled = true;
            resendBtn.textContent = 'Resend OTP in ';
            timerDisplay.textContent = `00:${String(timeLeft).padStart(2, '0')}`;
            resendBtn.appendChild(timerDisplay);

            resendInterval = setInterval(() => {
                timeLeft--;
                timerDisplay.textContent = `00:${String(timeLeft).padStart(2, '0')}`;

                if (timeLeft <= 0) {
                    clearInterval(resendInterval);
                    resendBtn.disabled = false;
                    resendBtn.textContent = 'Send OTP';
                }
            }, 1000);
        }

        function getOtpFromInputs() {
            return Array.from(otpInputs).map(input => input.value).join('');
        }

        document.getElementById('send-otp-btn').addEventListener('click', function() {
            if (window.otpPhoneNumber) {
                // Remove country code from phone number and keep only 10 digits, pad with leading zeros if needed
                let phoneNumber = window.otpPhoneNumber.replace(/\D/g, '');
                console.log(phoneNumber);

                window.sendOtp(
                    phoneNumber,
                    (data) => {
                        document.getElementById("otpStep-1").style.display = 'none';
                        document.getElementById("otpStep-2").style.display = 'block';
                        startResendTimer(); // Start countdown on first send
                        if (data && data.message) {
                            otpRequestId = data.message;
                        }
                    },
                    (error) => {
                        console.log('Error occurred while sending OTP');
                    }
                );
            } else {
                alert("Invalid phone number");
                console.log("Invalid phone number", window.otpPhoneNumber);
            }
        });

        resendBtn.addEventListener('click', function() {
            window.retryOtp("11", function(success) {
                if (success) {
                    startResendTimer(); // Restart countdown on resend
                    alert("OTP resent successfully.");
                } else {
                    alert("Failed to resend OTP.");
                }
            });

            window.retryOtp(
                "11",
                (data) => {
                    startResendTimer();
                    $(".resend-btn").html('Resend in <span class="resend-time">30s</span>'); // Reset button text
                },
                (error) => {
                    console.log(error);
                },
                otpRequestId
            );
        });

        document.getElementById('verify-otp-btn').addEventListener('click', function() {
            const otp = getOtpFromInputs();

            window.verifyOtp(
                otp,
                (data) => {
                    if (data.type == 'success') {
                        document.getElementById("otp-popup").style.display = 'none';
                        // ✅ If access_token input exists, set value. Else create it.
                        let tokenField = window.targetOtpForm.querySelector('input[name="access_token"]');
                        if (!tokenField) {
                            tokenField = document.createElement('input');
                            tokenField.type = 'hidden';
                            tokenField.name = 'access_token';
                            tokenField.classList.add('access_token');
                            window.targetOtpForm.appendChild(tokenField);
                        }

                        tokenField.value = data.message;

                        // Mark form as verified
                        window.targetOtpForm.dataset.otpVerified = "true";

                        // Hide popup
                        document.getElementById("otp-popup").style.display = 'none';

                        // Trigger submission again
                        window.targetOtpForm.dispatchEvent(new Event('submit', {
                            cancelable: true,
                            bubbles: true
                        }));
                    } else {
                        alert("Invalid OTP. Please try again.");
                        // Mark form as verified
                        window.targetOtpForm.dataset.otpVerified = "false";
                        const tokenField = window.targetOtpForm?.querySelector('input[name="access_token"]');
                        if (tokenField) {
                            tokenField.remove();
                        }
                    }
                },
                (error) => {
                    console.error('Error verifying OTP:', error);
                    alert("An error occurred while verifying OTP. Please try again.");
                    // Mark form as verified
                    window.targetOtpForm.dataset.otpVerified = "false";
                    const tokenField = window.targetOtpForm?.querySelector('input[name="access_token"]');
                    if (tokenField) {
                        tokenField.remove();
                    }
                },
                otpRequestId
            );
        });

        otpInputs.forEach((input, index) => {
            input.addEventListener('input', function() {
                if (this.value && index < otpInputs.length - 1) {
                    otpInputs[index + 1].focus();
                }
            });

            input.addEventListener('keydown', function(e) {
                if (e.key === 'Backspace' && !this.value && index > 0) {
                    otpInputs[index - 1].focus();
                }
            });

            // ✅ Handle paste across boxes
            input.addEventListener('paste', function(e) {
                e.preventDefault();
                const pasteData = (e.clipboardData || window.clipboardData).getData('text');
                const digits = pasteData.replace(/\D/g, '').split('');

                digits.forEach((digit, i) => {
                    if (i < otpInputs.length) {
                        otpInputs[i].value = digit;
                    }
                });

                // Focus last filled box
                const nextIndex = Math.min(digits.length, otpInputs.length - 1);
                otpInputs[nextIndex].focus();
            });
        });

        // Close popup on clicking the close button
        document.querySelector('.closePopup').addEventListener('click', function() {
            document.getElementById("otp-popup").style.display = 'none';
            // Reset form state
            window.targetOtpForm.dataset.otpVerified = "false";
            window.targetOtpForm = null;
            window.otpPhoneNumber = null;
            const tokenField = window.targetOtpForm.querySelector('input[name="access_token"]');
            if (tokenField) {
                tokenField.remove();
            }

            enableBootstrapFocusTrap(); // Re-enable Bootstrap's focus trap if using modals
        });

    });

    function openOtpPopup(form, phoneNumber) {
        disableBootstrapFocusTrap(); // Disable Bootstrap's focus trap if using modals

        window.targetOtpForm = form;
        window.otpPhoneNumber = phoneNumber;

        // Set mobile preview
        document.getElementById('mobile-preview').textContent = phoneNumber;

        // Show the popup
        document.getElementById("otp-popup").style.display = 'flex';
        document.getElementById("otpStep-1").style.display = 'block';
        document.getElementById("otpStep-2").style.display = 'none';

        if (typeof hcaptcha === 'object') {
            hcaptcha.reset();
        }

        // Reset resend timer
        clearInterval(resendInterval);
    }

    function disableBootstrapFocusTrap() {
        $(document).off('focusin.bs.modal');
    }

    function enableBootstrapFocusTrap() {
        $(document).on('focusin.bs.modal', function(e) {
            var $modal = $('.modal.in');
            if ($modal.length && $modal[0] !== e.target && !$modal.has(e.target).length) {
                $modal.trigger('focus');
            }
        });
    }
