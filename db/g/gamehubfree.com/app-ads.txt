    // 内容切换函数
    function showContent(contentId) {
        // 隐藏所有内容
        document.querySelectorAll('.cookie-panel-content').forEach(content => {
            content.classList.remove('cookie-content-active');
        });

        // 显示目标内容

    // 保存设置
    function savePreferences() {
        const settings = {
            analytics: document.querySelector('input[name="analytics"]').checked,
            advertising: document.querySelector('input[name="advertising"]').checked,
            necessary: true
        };

        // 更新Cookie
        document.cookie = `cookie_consent=${JSON.stringify(settings)}; path=/; max-age=${365*24*60*60}; SameSite=Lax`;

        // 更新GTM
        updateGTMConsent(settings);

        // 关闭面板
        document.getElementById('mainPanel').classList.remove('active');
        setTimeout(() => showContent('mainContent'), 300); // 重置内容状态
    }

    // GTM更新函数（保持不变）
    function updateGTMConsent(settings) {
        window.dataLayer = window.dataLayer || [];
        window.dataLayer.push({
            'event': 'cookie_consent_update',
            'consentSettings': {
                'analytics_storage': settings.analytics ? 'granted' : 'denied',
                'ad_storage': settings.advertising ? 'granted' : 'denied'
            }
        });
    }

    // 一键接受处理
    function handleConsent(action) {
        const settings = {
            analytics: action === 'acceptAll',
            advertising: action === 'acceptAll',
            necessary: true
        };

        document.cookie = `cookie_consent=${JSON.stringify(settings)}; path=/; max-age=${365*24*60*60}; SameSite=Lax`;
        updateGTMConsent(settings);
        document.getElementById('mainPanel').classList.remove('active');
