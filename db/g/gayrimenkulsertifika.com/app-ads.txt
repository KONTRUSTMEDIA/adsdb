        // Intersection Observer for animations
        const observerOptions = {
            threshold: 0.1,
            rootMargin: '0px 0px -50px 0px'
        };

        const observer = new IntersectionObserver((entries) => {
            entries.forEach(entry => {
                if (entry.isIntersecting) {
                    entry.target.style.animationPlayState = 'running';
                }
            });
        }, observerOptions);

        // Observe timeline items
        document.querySelectorAll('.timeline-item').forEach(item => {
            observer.observe(item);
        });

        // Add smooth hover effects
        document.querySelectorAll('.timeline-content').forEach(content => {
            content.addEventListener('mouseenter', function () {
                this.style.transform = 'translateY(-8px) scale(1.02)';
            });

            content.addEventListener('mouseleave', function () {
                this.style.transform = 'translateY(0) scale(1)';
            });
        });

        // Scroll indicators functionality
        const timelineContainer = document.getElementById('timelineContainer');
        const scrollDots = document.querySelectorAll('.scroll-dot');

        scrollDots.forEach((dot, index) => {
            dot.addEventListener('click', () => {
                const scrollAmount = (timelineContainer.scrollWidth / scrollDots.length) * index;
                timelineContainer.scrollTo({
                    left: scrollAmount,
                    behavior: 'smooth'
                });
            });
        });

        // Update active scroll indicator
        timelineContainer.addEventListener('scroll', () => {
            const scrollPercent = timelineContainer.scrollLeft / (timelineContainer.scrollWidth - timelineContainer.clientWidth);
            const activeIndex = Math.round(scrollPercent * (scrollDots.length - 1));

            scrollDots.forEach((dot, index) => {
                dot.classList.toggle('active', index === activeIndex);
            });
        });

        // Auto-scroll demo (optional)
        let autoScrollInterval;

        function startAutoScroll() {
            autoScrollInterval = setInterval(() => {
                const currentScroll = timelineContainer.scrollLeft;
                const maxScroll = timelineContainer.scrollWidth - timelineContainer.clientWidth;

                if (currentScroll >= maxScroll) {
                    timelineContainer.scrollTo({ left: 0, behavior: 'smooth' });
                } else {
                    timelineContainer.scrollBy({ left: 300, behavior: 'smooth' });
                }
            }, 3000);
        }

        function stopAutoScroll() {
            clearInterval(autoScrollInterval);
        }

        // Start auto-scroll after page load, stop on user interaction
        setTimeout(startAutoScroll, 2000);

        timelineContainer.addEventListener('mouseenter', stopAutoScroll);
        timelineContainer.addEventListener('touchstart', stopAutoScroll);

<script type="58f962129cadc0754343032c-text/javascript">
// Proje daire verileri (static data)
const projectUnits = [
    { type: "Min 1+1", brutArea: 61.82, certificates: 631516, offerPrice: 4793206 },
    { type: "Min 2+1", brutArea: 88.26, certificates: 863276, offerPrice: 6552265 },
    { type: "Min 3+1", brutArea: 143.61, certificates: 1384916, offerPrice: 10511512 },
    { type: "Min 4+1", brutArea: 194.14, certificates: 1833345, offerPrice: 13915089 }
];

class ProjectTable {
    constructor() {
        this.symbol = 'DMLKT.G';
        this.currentPrice = 0;
        this.updateInterval = 900000; // 15 dakika
        this.init();
    }

    async init() {
        await this.updatePrice();
        this.renderTable();
        
        // Periyodik güncelleme
        setInterval(async () => {
            await this.updatePrice();
            this.renderTable();
        }, this.updateInterval);
    }

    async updatePrice() {
        try {
            const now = Math.floor(Date.now() / 1000); // Şu anki Unix timestamp
            const from = now - (24 * 60 * 60); // 24 saat önce
            
            const response = await fetch(
                `https://gate.fintables.com/barbar/udf/history?symbol=${this.symbol}&resolution=5&from=${from}&to=${now}`
            );
            
            const data = await response.json();
            
            console.log('API Response:', data);
            
            if (data.c && data.c.length > 0) {
                this.currentPrice = data.c[data.c.length - 1];
                const priceElement = document.getElementById('currentCertificatePrice');
            if (priceElement) {
                priceElement.textContent = this.currentPrice;
            }
            }
        } catch (error) {
            console.error('Fiyat güncellenemedi:', error);
        }
    }

    calculateCurrentPrice(certificates) {
        return certificates * this.currentPrice;
    }

    formatNumber(num) {
        return new Intl.NumberFormat('tr-TR', {
            minimumFractionDigits: 0,
            maximumFractionDigits: 0
        }).format(num);
    }

    formatCurrency(num) {
        return new Intl.NumberFormat('tr-TR', {
            style: 'currency',
            currency: 'TRY',
            minimumFractionDigits: 0,
            maximumFractionDigits: 0
        }).format(num);
    }

    renderTable() {
        const tbody = document.getElementById('unitsTableBody');
        
        if (this.currentPrice === 0) {
            tbody.innerHTML = '<tr><td colspan="6" class="text-center">Fiyat verisi bekleniyor...</td></tr>';
            return;
        }

        let html = '';
        
        projectUnits.forEach(unit => {
            const guncelFiyat = this.calculateCurrentPrice(unit.certificates);
            
            html += `
                <tr>
                    <td><strong>${unit.type}</strong></td>
                    <td>${unit.brutArea} m²</td>
                    <td>${this.formatNumber(unit.certificates)}</td>
                    <td><strong>${this.formatCurrency(guncelFiyat)}</strong></td>
                </tr>
            `;
        });
        
        tbody.innerHTML = html;
    }
}

// Sayfa yüklendiğinde başlat
document.addEventListener('DOMContentLoaded', () => {
    new ProjectTable();
});
