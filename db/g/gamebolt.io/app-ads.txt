      (function() {
        var path = window.location.pathname;
        var manifestLink = document.getElementById('pwa-manifest');
        if (!manifestLink) return;
        
        // Détection des pages de jeu
        var gamePatterns = ['/game/', '/jeu/', '/juego/', '/spiel/', '/jogo/'];
        var isGamePage = gamePatterns.some(function(p) { return path.includes(p); });
        
        if (isGamePage) {
          // TRUC VITAL : On pointe vers le dossier actuel
          // Au lieu de aller chercher /manifest/slug.json
          // On dit au navigateur : "Le manifeste est ici, avec moi"
          // Enlève le slash final si présent pour éviter les doubles slashes
          var cleanPath = path.replace(/\/$/, '');
          // Résultat : /game/mon-jeu/manifest.json
          manifestLink.href = cleanPath + '/manifest.json';
        }
      })();
    </script>
