<script>
function openDrawer(){
  document.getElementById('drawer').classList.add('open');
  document.getElementById('drawerOverlay').classList.add('open');
  document.getElementById('burgerBtn').classList.add('open');
  document.body.style.overflow='hidden';
}
function closeDrawer(){
  document.getElementById('drawer').classList.remove('open');
  document.getElementById('drawerOverlay').classList.remove('open');
  document.getElementById('burgerBtn').classList.remove('open');
  document.body.style.overflow='';
}
document.addEventListener('keydown',function(e){if(e.key==='Escape')closeDrawer();});

/* ── cat-nav overflow burger ── */
(function(){
  var nav      = document.getElementById('catNav');
  var more     = document.getElementById('catMore');
  var moreBtn  = document.getElementById('catMoreBtn');
  var moreDrop = document.getElementById('catDrop');
  if (!nav) return;
  var resizeTimer;

  function calcNav() {
    if (window.innerWidth <= 860) return;   /* mobile uses drawer */
    var items = Array.from(nav.querySelectorAll('a'));

    /* reset: show all items, clear dropdown */
    items.forEach(function(a){ a.style.display = ''; });
    moreDrop.innerHTML = '';

    /* show burger (flex sibling) so nav shrinks to give it space */
    more.style.display = 'flex';
    var navW = nav.offsetWidth; /* nav has flex:1 → already excludes burger width */

    /* inline-block links: offsetLeft is relative to offset parent (nav).
       Find first link whose right edge exceeds navW. */
    var cutoff = -1;
    for (var i = 0; i < items.length; i++) {
      if (items[i].offsetLeft + items[i].offsetWidth > navW + 1) { cutoff = i; break; }
    }

    /* all fit — hide burger */
    if (cutoff === -1) { more.style.display = 'none'; return; }

    /* move overflowing items into dropdown */
    for (var j = cutoff; j < items.length; j++) {
      items[j].style.display = 'none';
      var a = document.createElement('a');
      a.href = items[j].href;
      a.textContent = items[j].textContent.trim();
      if (items[j].classList.contains('active')) a.classList.add('active');
      moreDrop.appendChild(a);
    }
  }

  requestAnimationFrame(function(){ requestAnimationFrame(calcNav); });
  window.addEventListener('resize', function(){
    clearTimeout(resizeTimer);
    resizeTimer = setTimeout(calcNav, 50);
  });
  moreBtn.addEventListener('click', function(e){
    e.stopPropagation();
    moreDrop.classList.toggle('open');
  });
  document.addEventListener('click', function(){ moreDrop.classList.remove('open'); });
})();
</script>
