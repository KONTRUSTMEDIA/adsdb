<script>
const marquee = document.getElementById('marquee');
const leftToggle = document.getElementById('leftToggle');
const rightToggle = document.getElementById('rightToggle');

let isPaused = false;

function updateIcons() {
  const icon = isPaused ? '▶' : '⏸';
  leftToggle.textContent = icon;
  rightToggle.textContent = icon;
}

function toggleMarquee() {
  isPaused = !isPaused;
  marquee.style.animationPlayState = isPaused ? 'paused' : 'running';
  updateIcons();
}

leftToggle.addEventListener('click', toggleMarquee);
rightToggle.addEventListener('click', toggleMarquee);

// Pause marquee on hover of any link
const links = marquee.querySelectorAll('a');

links.forEach(link => {
  link.addEventListener('mouseenter', () => {
    marquee.style.animationPlayState = 'paused';
  });
  link.addEventListener('mouseleave', () => {
    if (!isPaused) marquee.style.animationPlayState = 'running';
  });
});
</script>

    const button = container.querySelector('.fancy-more-btn');
    const searchInput = document.querySelector('.link-search');

    let currentIndex = 0;
    const batchSize = 10;

    function showNextBatch() {
      for (let i = currentIndex; i < currentIndex + batchSize && i < items.length; i++) {
        items[i].classList.add('visible');
        items[i].style.display = 'list-item';
      }
      currentIndex += batchSize;
      if (currentIndex >= items.length) {

          item.classList.remove('visible');
          item.style.display = 'none';
        });
        showNextBatch();
        button.style.display = '';
      } else {
        items.forEach(item => {
          const text = item.textContent.toLowerCase();
          if (text.includes(query)) {
            item.style.display = 'list-item';
          } else {
            item.style.display = 'none';
          }
        });
        button.style.display = 'none';
      }
    });
  });
