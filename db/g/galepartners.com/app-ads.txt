// dark light mode toggle button functionality
$(function() {
	let inLightMode = localStorage.getItem('light-mode');

  	let toggleBtn = $('.dark-mode-toggle-checkbox');
    if (inLightMode == true || inLightMode == "true") {
      toggleBtn.prop("checked", true );
      $('body').addClass('light-mode').removeClass('dark-mode');
      $('.dark-mode-toggle-label').addClass('light-mode');
      $('.dark-mode-toggle-label').text('Light mode');
    } else {
      $('body').addClass('dark-mode').removeClass('light-mode');
      $('.dark-mode-toggle-label').removeClass('light-mode');
      $('.dark-mode-toggle-label').text('Dark mode');
    }
	toggleBtn.on("change", (e)=>{
  	let checked = toggleBtn.prop('checked');
    if (!checked) {
    	localStorage.setItem('light-mode', false);
	    $('body').addClass('dark-mode').removeClass('light-mode');
      	$('.dark-mode-toggle-label').removeClass('light-mode');
      	$('.dark-mode-toggle-label').text('Dark mode');
      	$(window).trigger('dark-mode-toggle', [true]);
    } else {
      	localStorage.setItem('light-mode', true);
    	$('body').addClass('light-mode').removeClass('dark-mode');
      	$('.dark-mode-toggle-label').addClass('light-mode');
      	$('.dark-mode-toggle-label').text('Light mode');
      	$(window).trigger('dark-mode-toggle', [false]);
    }
  });
	let navToggle = $('#mobile-nav-toggle');
    let mobileNavBg = $('#mobile-nav-bg');
    let navOverlay = $('.w-nav-overlay');
    let navMenu = $('.nav-menu');
// when expand or collapse the nav by click on mobile nav toggle or w-nav-overlay animate in or out the right element, also make body not scrollable when it's expanded, so if nav items can't display within view port, it will be able to scroll without scrolling body content (also avoid sticky nav issue)
  function expandMobileToggle() {
  	if (navToggle.attr('aria-expanded') == 'false'){
	    mobileNavBg.animate({height:'100vh', opacity:1});
      $('body').addClass('mobile-nav-open');
      $('body').css({height:'100vh', overflow:'hidden'});
    } else {
    	mobileNavBg.animate({height:'0vh', opacity:0});
      $('body').removeClass('mobile-nav-open');
      $('body').css({height:'fit-content', overflow:'auto'});
    }
  }
  
  navToggle.on('click', (e)=>{
	  e.preventDefault();
    expandMobileToggle();
  });
  
  $('.w-nav-overlay').on('click', (e)=>{
  	if(e.target.className != 'w-nav-overlay') {
    	return;
    }
  	e.preventDefault();
    expandMobileToggle();
  });
  
  function resize(){
  	if (window.innerWidth <= 991) {
    	// this is to resize the nav menu height and make sure if window height is too small to see all nav items, users can scroll to it.
      let newHeight = window.innerHeight - $('.nav-background').height();
      if (navMenu){
        navMenu.css({height:newHeight, overflow:'scroll'});
      }
    } else {
    	 // once window resize to desktop we should hide the mobile nav bg
   		 if (navToggle.attr('aria-expanded') == 'false'){
        mobileNavBg.css({height:'0vh', opacity:0});
        $('body').css({height:'fit-content', overflow:'auto'});
      }
       navMenu.css({height:'fit-content', overflow:'visible'});
    }
  }
  
  $(window).on('resize', ()=>{
  	resize();
  });
  
  resize();
});

<script>
  var Webflow = Webflow || [];
  function registerPlayPauseBtn(){
  	const btn = $('.global-play-pause-btn');
    btn.each((index)=> {
      let elem = $(btn[index]);
      let container = elem.closest('.global-play-pause-btn-container');
      const themeColor = elem.data('theme-color');
      const verticalAlign = elem.data('vertical-alignment');
      if (themeColor == 'dark') {
      	elem.addClass('dark');
      }
      if (themeColor == 'toggle') {
      	elem.addClass('toggle');
      }
      if (verticalAlign == 'top'){
      	container.css({top:'16px'});
      }
    });
  }
  function registerLoopingVideoControl(){
	  const autoplayVid = $('.global-looping-video-container');
    autoplayVid.each((index)=>{
    	let elem = $(autoplayVid[index]);
      let vid = elem.find('.looping-video');
      let playVideo = vid.get(0);
      let vidId = vid.attr('id');
      let ctrlId = '#'+vidId+'-pause-btn';
      let controlBtn = elem.find(ctrlId);
      controlBtn.on('click', ()=>{
        if (playVideo.paused) {
          vid.removeClass('paused');
          playVideo.play();
          controlBtn.removeClass('paused');
          controlBtn.attr('aria-pressed', false);
        } else {
          vid.addClass('paused');
          playVideo.pause();
          controlBtn.addClass('paused');
          controlBtn.attr('aria-pressed', true);
        }
      });
    });
  }

	function registerGifAnimationEvent(){
  	const gifContainer = $('.gif-animation-container');
    gifContainer.each((index)=>{
	  let elem = $(gifContainer[index]);
      console.log(elem.attr('id'));
      let gifId = '#'+ elem.attr('id');
      let pauseBtn = elem.find(gifId+'-pause-btn');
      pauseBtn.removeClass('paused');
    	pauseBtn.attr('aria-pressed', true);
      const gifAnimation = new Freezeframe(gifId + ' .gif-animation', {
        trigger:false
      });
      // start animation if it's scrolled in view
      let timer = null;
      let observer = new IntersectionObserver((entries)=>{
        const pausedState = pauseBtn.attr('aria-pressed');
        if (pausedState == 'true'){
          setTimeout(()=>{
              gifAnimation.start();
            clearTimeout(timer);
          },500);
        }
      });
		
      let target = document.querySelector(gifId);
      
      observer.observe(target);
      // play pause btn binding for gif animation control
      pauseBtn.on('click', ()=>{
        const pausedState = pauseBtn.attr('aria-pressed');
        if (pausedState == 'false') {
          gifAnimation.start();
          pauseBtn.removeClass('paused');
          pauseBtn.attr('aria-pressed', true);
        } else {
          gifAnimation.stop();
          pauseBtn.addClass('paused');
          pauseBtn.attr('aria-pressed', false);
        }
      });
    });
  }
  
  Webflow.push(registerPlayPauseBtn);
  Webflow.push(registerLoopingVideoControl);
  Webflow.push(registerGifAnimationEvent);
</script>
