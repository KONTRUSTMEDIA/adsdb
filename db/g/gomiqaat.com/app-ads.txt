                    let platform = 'agoda';

                    /* ---------- Clean location ---------- */
                    // Decode HTML entities (&#8211; = en-dash, etc.)
                    const decodeHtml = (html) => {
                        const txt = document.createElement('textarea');
                        txt.innerHTML = html;
                        return txt.value;
                    };
                    resolvedLocation = decodeHtml(resolvedLocation);

                    // Remove common category suffixes FIRST (before pattern matching)
                    resolvedLocation = resolvedLocation
                        .replace(/\s*[–—-]\s*(Holiday and Destinations?|Events?|Special Offers?).*$/i, '')
                        .replace(/\s*[–—-]\s*(Sovereign|goMiqaat).*$/i, '')
                        .trim();

                    const patterns = [
                        // Format: Hotels Near Landmark, City, Country
                        // Extract "City, Country" for better external search accuracy
                        /Hotels Near .+?,\s*([^–—|.]+)/i,
                        // Format: Hotels and Home stays near/neat Landmark, City...
                        /Hotels? and Home\s?stays?\s+nea[rt]\s+(.+?)(?:\.|$|[–—])/i,
                        // Fallbacks
                        /nea[rt]\s+(?!(Event:|.+,))([^–—|.]+)/i,
                        /in\s+([^–—|.]+)/i,
                        /at\s+([^–—|.]+)/i,
                        /^(.+?)\s+Hotels?/i
                    ];

                    for (let p of patterns) {
                        const m = resolvedLocation.match(p);
                        if (m && (m[1] || m[2])) {
                            resolvedLocation = (m[1] || m[2]).trim();
                            break;
                        }
                    }

                    // Remove any remaining suffixes/garbage
                    if (resolvedLocation.includes('–')) resolvedLocation = resolvedLocation.split('–')[0].trim();
                    if (resolvedLocation.includes('—')) resolvedLocation = resolvedLocation.split('—')[0].trim();
                    if (resolvedLocation.includes('|')) resolvedLocation = resolvedLocation.split('|')[0].trim();

                    const cachedGeo = localStorage.getItem('gomiqaat_geo_data');
