        $('btnLogin').addEventListener('click', async () => {
            var isValid = formvalidater.validate($('login'));
            if(isValid) {
                show('loadInfo');
                var json = {
                    email: $('email').value,
                    password: $('password').value,
                    passwordType: 4,
                    authType: 0
                }
                const resp = await Login(json);
                if(resp.success) {
                    toggleClasses($('contentInfo'), 'text-muted', 'text-success');
                    $('contentInfo').textContent = "Success..."
                    setTimeout(() => {
                        window.location.href = resp.redirectUrl;
                    }, 1000);
                }
                else {
                    hide('loadInfo');
                    removeClass($('errorSummary'), 'd-none');
                    $('errorMessage').textContent = resp.message;
                }
            }
        });
    </script>
