<p>На відміну від звичайної реклами, тут важливий не охоплення, а результат — <strong>конкретні звернення від цільової аудиторії</strong>.</p>
<h3><strong>Як ми працюємо</strong></h3>
<p>Ми не продаємо кліки. Ми створюємо системи, які приносять бізнесу клієнтів.</p>

<li>Більше дзвінків із сайту</li>
<li>Більше повідомлень у WhatsApp і Telegram</li>
<li>Більше підписників і залученості</li>
</ul>
<p>Серед наших проєктів — <strong>адвокати, клініки, рієлтори, логістичні компанії, B2B та B2C бізнеси</strong>.</p>

        // Safety: if Pixel isn't loaded, do nothing (no errors).
        function hasFbq() { return (typeof window.fbq === 'function'); }

        // --- 1) Contact Form 7 success => Lead ---
        document.addEventListener('wpcf7mailsent', function (event) {
            if (!hasFbq()) return;

            // If you ever want to count only specific forms, uncomment and set IDs:
            // var allowed = [123, 456];
            // if (!event.detail || allowed.indexOf(parseInt(event.detail.contactFormId, 10)) === -1) return;

            window.fbq('track', 'Lead', { source: 'cf7' });
        }, false);

        // --- helpers to avoid double fires on the same element ---
        function markOnce(el, key) {
            try {
                if (el.dataset && el.dataset[key] === '1') return false;
                if (el.dataset) el.dataset[key] = '1';
            } catch (e) {}
            return true;
        }

        // Find closest <a> for clicks on icons/spans inside a button/link.
        function closestAnchor(node) {
            while (node && node !== document && node !== window) {
                if (node.tagName && node.tagName.toLowerCase() === 'a') return node;
                node = node.parentNode;
            }
            return null;
        }

        // --- 2) Click tracking: WhatsApp + Phone ---
        document.addEventListener('click', function (e) {
            var a = closestAnchor(e.target);
            if (!a) return;

            var href = (a.getAttribute('href') || '').trim();

            // Phone: tel:
            if (href.toLowerCase().indexOf('tel:') === 0) {
                if (!hasFbq()) return;
                if (!markOnce(a, 'fbqPhoneFired')) return;

                window.fbq('track', 'Contact', { method: 'phone' });
                return;
            }

            // WhatsApp: wa.me, whatsapp://, api.whatsapp.com, whatsapp.com
            var h = href.toLowerCase();
            var isWhatsApp =
                h.indexOf('wa.me/') !== -1 ||
                h.indexOf('whatsapp://') === 0 ||
                h.indexOf('api.whatsapp.com') !== -1 ||
                (h.indexOf('whatsapp.com') !== -1 && h.indexOf('send') !== -1);

            if (isWhatsApp) {
                if (!hasFbq()) return;
                if (!markOnce(a, 'fbqWaFired')) return;

                window.fbq('track', 'Contact', { method: 'whatsapp' });
                return;
            }
        }, true);

    })();
    </script>
