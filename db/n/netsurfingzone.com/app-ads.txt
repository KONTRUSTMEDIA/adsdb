<script>
    document.addEventListener('click', function (e) {
        const langToggle    = document.getElementById('langToggle');
        const langDropdown  = document.getElementById('langDropdown');
        const userToggle    = document.getElementById('userToggle');
        const userDropdown  = document.getElementById('userDropdown');

        if (langToggle && langToggle.contains(e.target)) {
            e.preventDefault();
            langDropdown && langDropdown.classList.toggle('show');
            userDropdown && userDropdown.classList.remove('show');
            return;
        }

        if (userToggle && userToggle.contains(e.target)) {
            e.preventDefault();
            userDropdown && userDropdown.classList.toggle('show');
            langDropdown && langDropdown.classList.remove('show');
            return;
        }

        if (
            (langDropdown && langDropdown.contains(e.target)) ||
            (userDropdown && userDropdown.contains(e.target))
        ) {
            return;
        }

        langDropdown && langDropdown.classList.remove('show');
        userDropdown && userDropdown.classList.remove('show');
    });
</script>

        const computeTop = () => {
            if (!stickyElement) return;
            const y = window.scrollY; 
            let val;
            if (y <= TRIGGER_SCROLL_Y) {
                val = INITIAL_TOP_PX;
            } else {
                const calculatedPos = INITIAL_TOP_PX - (y - TRIGGER_SCROLL_Y);
                val = Math.max(calculatedPos, MAX_SCROLL_UP_PX);
            }
            // stickyElement.style.setProperty('--banner-top', val + 'px');
        };
        window.addEventListener('scroll', computeTop);
        computeTop();
    });

    (function () {
        const second = 1000,
            minute = second * 60,
            hour = minute * 60,
            day = hour * 24;

        // helper untuk angka 2 digit
        function pad(num) {
            return num.toString().padStart(2, "0");
        }

        const countDown = new Date("June 11, 2026 00:00:00 GMT+0700").getTime();


        const x = setInterval(function () {

            const now = new Date().getTime(),
                distance = countDown - now;

            const d = Math.floor(distance / day);
            const h = Math.floor((distance % day) / hour);
            const m = Math.floor((distance % hour) / minute);
            const s = Math.floor((distance % minute) / second);

            // tampilkan ke HTML (dengan zero padding)
            document.getElementById("days").innerText = d;
            document.getElementById("hours").innerText = pad(h);
            document.getElementById("minutes").innerText = pad(m);
            document.getElementById("seconds").innerText = pad(s);

            if (distance < 0) {
                clearInterval(x);
                document.getElementById("countdown").innerHTML = "<h2>Event has started!</h2>";
            }

        }, 1000);
    }());


</script>
