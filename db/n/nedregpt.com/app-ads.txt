  let deferredPrompt; 
  const installBtn = document.getElementById('btn-install');

  // Tarayıcı "Bu site yüklenebilir" sinyali verince çalışır
  window.addEventListener('beforeinstallprompt', (e) => {
    e.preventDefault(); // Otomatik barı gizle
    deferredPrompt = e; // İsteği sakla
    installBtn.style.display = 'block'; // Bizim butonu göster
  });

  // Butona tıklayınca
  installBtn.addEventListener('click', async () => {
    if (deferredPrompt) {
      deferredPrompt.prompt(); // Yükleme ekranını aç
      const { outcome } = await deferredPrompt.userChoice;
      console.log(`Kullanıcı tercihi: ${outcome}`);
      deferredPrompt = null;
      installBtn.style.display = 'none';
    }
  });

  // Zaten yüklüyse butonu gizle
  window.addEventListener('appinstalled', () => {
    installBtn.style.display = 'none';
  });
