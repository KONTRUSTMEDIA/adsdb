  <script>
    const siteUrl = 'https://www.nilsonsergiolima.com.br/';
    const quantidadeDePosts = 4;
    const categoriaID = '';

    let posts = [];
    let indexAtual = 1;
    let intervalo;
    let larguraSlide = 50;

    async function carregarPosts(qtd = quantidadeDePosts, categoriaId = categoriaID) {
      let url = `${siteUrl}/wp-json/wp/v2/posts?per_page=${qtd}&_embed`;
      if (categoriaId) {
        url += `&categories=${categoriaId}`;
      }

      const res = await fetch(url);
      const dados = await res.json();
      posts = dados;
      montarCarrossel();
    }

    function montarCarrossel() {
      const carrossel = document.getElementById('carrossel');
      carrossel.innerHTML = '';

      const slides = [];

      const cloneFinal = criarSlide(posts[posts.length - 1]);
      slides.push(cloneFinal);

      posts.forEach(post => {
        slides.push(criarSlide(post));
      });

      const cloneInicial = criarSlide(posts[0]);
      slides.push(cloneInicial);

      slides.forEach(slide => carrossel.appendChild(slide));

      indexAtual = 1;
      atualizarVisual(true);
      iniciarLoop();
      ativarPausaAoHover();
    }

    function criarSlide(post) {
      const imagem = post._embedded?.['wp:featuredmedia']?.[0]?.source_url || 'https://via.placeholder.com/600x300?text=Sem+Imagem';
      const div = document.createElement('div');
      div.className = 'slide';
      div.innerHTML = `<a href="${post.link}" target="_blank"><img decoding="async" src="${imagem}" alt="Imagem do post"></a>`;
      return div;
    }

    function atualizarVisual(imediato = false) {
      const carrossel = document.getElementById('carrossel');
      larguraSlide = window.innerWidth < 768 ? 100 : 50;
      carrossel.style.transition = imediato ? 'none' : 'transform 0.6s ease-in-out';
      carrossel.style.transform = `translateX(-${indexAtual * larguraSlide}%)`;
    }

    function avancar() {
      if (indexAtual >= posts.length) {
        indexAtual++;
        atualizarVisual();
        setTimeout(() => {
          indexAtual = 1;
          atualizarVisual(true);
        }, 600);
      } else {
        indexAtual++;
        atualizarVisual();
      }
    }

    function voltar() {
      if (indexAtual <= 0) {
        indexAtual--;
        atualizarVisual();
        setTimeout(() => {
          indexAtual = posts.length;
          atualizarVisual(true);
        }, 600);
      } else {
        indexAtual--;
        atualizarVisual();
      }
    }

    function iniciarLoop() {
      clearInterval(intervalo);
      intervalo = setInterval(avancar, 5000);
    }

    function ativarPausaAoHover() {
      const carrossel = document.getElementById('carrossel');
      carrossel.addEventListener('mouseenter', () => clearInterval(intervalo));
      carrossel.addEventListener('mouseleave', iniciarLoop);
    }

    window.addEventListener('resize', () => atualizarVisual(true));
    carregarPosts();
  </script>
