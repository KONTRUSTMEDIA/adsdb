    <script>
        let names = [];

        const searchInput = document.getElementById('searchInput');
        const suggestionList = document.getElementById('suggestionList');
        const searchButton = document.getElementById('searchButton');

        searchInput.addEventListener('input', function() {
            const inputValue = this.value.toLowerCase();
            suggestionList.innerHTML = '';

            if (inputValue.length > 0) {
                const filteredNames = names.filter(name => 
                    name.toLowerCase().startsWith(inputValue)
                );

                filteredNames.forEach(name => {
                    const li = document.createElement('li');
                    li.classList.add('list-group-item', 'list-group-item-action');
                    li.textContent = name;
                    li.addEventListener('click', () => selectName(name));
                    suggestionList.appendChild(li);
                });
            }
        });

        searchButton.addEventListener('click', function() {
            const selectedName = searchInput.value;
            if (selectedName) {
                selectName(selectedName);
            }
        });

        function selectName(name) {
            const url = `https://nevnapma.hu/nevnap/${encodeURIComponent(name)}`;
            window.location.href = url;
        }

        // Nevek betöltése a PHP szkriptből
        fetch('https://nevnapma.hu/search/names.php')
            .then(response => response.json())
            .then(data => {
                names = data;
                console.log('Nevek betöltve:', names);
            })
            .catch(error => console.error('Hiba a nevek betöltése során:', error));
	
	function acceptCookies() {

}

			document.getElementById('cookieBanner').style.display = 'none';
		}
	});
