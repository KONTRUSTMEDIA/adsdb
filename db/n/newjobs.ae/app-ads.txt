        function splitRoute() {
          // Returns { culture, location, categoryInPath }
          const parts = window.location.pathname.split("/").filter(Boolean);

          let culture = null;
          // Heuristic: culture like "en" or "en-US"
          if (parts[0] && /^[a-z]{2}(?:-[A-Z]{2})?$/.test(parts[0])) {
            culture = parts.shift();
          }

          const location = parts[0] || null;
          const categoryInPath = parts[1] || null;

          return { culture, location, categoryInPath };
        }

        function buildPath(culture, location, categoryPath) {
          const segs = [];
          if (culture) segs.push(culture);
          if (location) segs.push(location);
          if (categoryPath) segs.push(categoryPath);
          return "/" + segs.join("/");
        }

        function buildQuery(paramsObj) {
          const p = new URLSearchParams();
          Object.entries(paramsObj).forEach(([k, v]) => {
            if (v !== undefined && v !== null && String(v).trim() !== "") p.set(k, v);
          });
          const s = p.toString();
          return s ? "?" + s : "";
        }

        function getCultureFromDomOrPath() {
            const hiddenCulture = document.getElementById("culture")?.value;
            if (hiddenCulture && hiddenCulture.trim() !== "") {
                return hiddenCulture;
            }

            const parts = window.location.pathname.split("/").filter(Boolean);
            if (parts[0] && /^[a-z]{2}(?:-[A-Z]{2})?$/i.test(parts[0])) {
                return parts[0];
            }

            return ""; // still nothing
        }
