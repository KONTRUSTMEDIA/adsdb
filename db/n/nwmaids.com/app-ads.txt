		timerPopupSection.addEventListener('click', function (e) {
			// Check if the clicked element is the closePopupIcon
			if (e.target === closePopupIcon) {
				// If the popup is not minified, minify it
				if (!timerPopupSection.classList.contains('minified')) {
					timerPopupSection.classList.add('minified');
				} else {
					// Otherwise, remove the visible class
					timerPopupSection.classList.remove('visible');
				}

			}  else {
				// If the target is not the closePopupIcon, enlarge the popup
				if (timerPopupSection.classList.contains('minified')) {
					timerPopupSection.classList.remove('minified');
				}
			}
		});

		// Wait 5 seconds after the page fully loads
		setTimeout(function () {

			// Add the 'visible' class to the timer-popup-section
			if (timerPopupSection) {
				timerPopupSection.classList.add('visible');
			}

			/**
             * Timer logic:
             * Counts down from 5 minutes (300 seconds) to 0
             * Updates the DOM with remaining time each second
             */
			const timerDuration = 5 * 60; // 5 minutes in seconds
			let remainingTime = timerDuration;

			// Get the DOM elements for each digit
			const minuteTens = document.getElementById('minute-tens');
			const minuteOnes = document.getElementById('minute-ones');
			const secondTens = document.getElementById('second-tens');
			const secondOnes = document.getElementById('second-ones');

			function updateTimer() {
				const minutes = Math.floor(remainingTime / 60);
				const seconds = remainingTime % 60;

				// Update each digit
				minuteTens.textContent = Math.floor(minutes / 10);
				minuteOnes.textContent = minutes % 10;
				secondTens.textContent = Math.floor(seconds / 10);
				secondOnes.textContent = seconds % 10;

				if (remainingTime > 0) {
					remainingTime--;
				} else {
					clearInterval(timerInterval); // Stop the timer
				}
			}

			// Start the timer
			const timerInterval = setInterval(updateTimer, 1000);
			updateTimer();
		}, 5000); // Wait 5000ms (5 seconds) before showing the popup
