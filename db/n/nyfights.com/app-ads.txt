<script id="mlit-foxiz-published-date-js-after">
(function(){
  const CFG = window.MLIT_FOXIZ_DATE || {};
  const REST = CFG.restPosts;
  const LABEL = CFG.label || 'Published';
  if (!REST) return;

  const cache = new Map(); // id -> {iso,human}

  function formatHuman(isoString){
    try{
      const d = new Date(isoString);
      if (isNaN(d.getTime())) return '';
      return new Intl.DateTimeFormat(undefined, { year:'numeric', month:'long', day:'2-digit' }).format(d);
    }catch(e){
      return '';
    }
  }

  function chunk(arr, size){
    const out = [];
    for (let i=0; i<arr.length; i+=size) out.push(arr.slice(i, i+size));
    return out;
  }

  function findWraps(root){
    return (root || document).querySelectorAll('.p-wrap[data-pid]');
  }

  function insertDateIntoWrap(wrap, info){
    if (!wrap || wrap.dataset.mlitDateDone === '1') return;

    const title = wrap.querySelector('.entry-title') || wrap.querySelector('h3') || wrap.querySelector('h2');
    if (!title) return;

    // avoid duplicate if theme re-renders
    if (wrap.querySelector('.mlit-published-date--archive')) {
      wrap.dataset.mlitDateDone = '1';
      return;
    }

    const el = document.createElement('div');
    el.className = 'mlit-published-date mlit-published-date--archive';
    el.innerHTML = '<time datetime="'+ info.iso +'">'+ LABEL +': ' + info.human + '</time>';

    // "append to the grid item": place right after the title
    title.insertAdjacentElement('afterend', el);

    wrap.dataset.mlitDateDone = '1';
  }

  async function fetchDates(ids){
    // Only fetch those not in cache
    const missing = ids.filter(id => !cache.has(id));
    if (!missing.length) return;

    // WP REST per_page max is typically 100
    const groups = chunk(missing, 100);

    for (const g of groups){
      const url = new URL(REST);
      url.searchParams.set('include', g.join(','));
      url.searchParams.set('per_page', String(Math.min(100, g.length)));
      url.searchParams.set('_fields', 'id,date');

      let res;
      try{
        res = await fetch(url.toString(), { credentials: 'same-origin' });
      }catch(e){
        continue;
      }
      if (!res || !res.ok) continue;

      let data;
      try{
        data = await res.json();
      }catch(e){
        continue;
      }
      if (!Array.isArray(data)) continue;

      data.forEach(p => {
        const id = String(p.id);
        const iso = p.date || '';
        const human = formatHuman(iso);
        if (iso && human) cache.set(id, { iso, human });
      });
    }
  }

  async function decorate(root){
    const wraps = Array.from(findWraps(root));
    if (!wraps.length) return;

    const ids = Array.from(new Set(
      wraps.map(w => String(w.getAttribute('data-pid') || '')).filter(Boolean)
    ));

    await fetchDates(ids);

    wraps.forEach(w => {
      const id = String(w.getAttribute('data-pid') || '');
      const info = cache.get(id);
      if (info) insertDateIntoWrap(w, info);
    });
  }

  // Initial
  document.addEventListener('DOMContentLoaded', function(){
    decorate(document);
  });

  // Handle infinite scroll / ajax loaded grids
  const obs = new MutationObserver((mutations) => {
    for (const m of mutations){
      for (const node of m.addedNodes){
        if (node && node.querySelectorAll) {
          // if the node contains any p-wrap items, decorate that subtree
          if (node.matches && node.matches('.p-wrap[data-pid]')) {
            decorate(node.parentNode || document);
          } else if (node.querySelector && node.querySelector('.p-wrap[data-pid]')) {
            decorate(node);
          }
        }
      }
    }
  });
  obs.observe(document.documentElement, { childList:true, subtree:true });
})();
</script>
