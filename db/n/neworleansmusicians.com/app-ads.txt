                    let percent = Math.round(event.loaded/event.total * 100);
                    $('.upload-progress').attr('aria-valuenow', percent).css('width', percent+'%');
                };

                function uploadEnd(event) {
                    console.log("upload complete");
                    Swal({
                        title: 'Success!',
                        text: "Your video has been uploaded to YouTube.",
                        icon: 'success',
                        showCancelButton: false,
                        confirmButtonText: 'Ok',
                    })
                    // Swal.fire(
                    // 'Success!',
                    // 'Your video has been uploaded to YouTube.',
                    // 'success'
                    // )
                    $('#video_form_new')[0].reset()
                    $('#video_form_new').show();
                    $('.progress-container').addClass('d-none');
                    $('.upload-progress').attr('aria-valuenow', '0').css('width', '0');
                };
            });
        })
