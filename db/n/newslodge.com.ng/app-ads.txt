function onSuccess(token) {
    // Show a small "Verifying..." message to the user
    document.querySelector('p').innerText = "Verification successful! Redirecting...";

    fetch('/verify-handler.php?token=' + encodeURIComponent(token))
    .then(response => {
        if (response.ok) {
            // Success! Reload the current page to see the content
            window.location.reload();
        } else {
            // If handler returns error, show it
            alert("Server verification failed. Please refresh and try again.");
            window.location.reload();
        }
    })
    .catch(err => {
        console.error("Connection error:", err);
        alert("Could not connect to verification server.");
    });
}
