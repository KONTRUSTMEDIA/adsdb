        // Función para cargar RSS con múltiples intentos
        async function loadRSS() {
            try {
                showLoading("Analizando noticias de Almería...");
                refreshBtn.disabled = true;
                
                // Intentar múltiples fuentes
                const sources = [
                    { url: RSS_URL, name: 'Noticias de Almería' },

                ];
                
                let success = false;
                
                for (const source of sources) {
                    try {
                        console.log(`Intentando cargar: ${source.name}`);
                        const response = await fetchWithTimeout(source.url, 10000);
                        
                        if (!response.ok) {
                            console.warn(`Error HTTP ${response.status} en ${source.name}`);
                            continue;
                        }
                        
                        const text = await response.text();
                        
                        // Verificar que sea XML válido
                        if (!text.includes('<?xml') && !text.includes('<rss')) {
                            console.warn(`Respuesta no es XML válido de ${source.name}`);
                            continue;
                        }
                        
                        const parser = new DOMParser();
                        const xmlDoc = parser.parseFromString(text, "text/xml");
                        
                        const parseError = xmlDoc.querySelector('parsererror');
                        if (parseError) {
                            console.warn(`Error de parseo en ${source.name}`);
                            continue;
                        }
                        
                        const items = xmlDoc.querySelectorAll('item');
                        if (items.length === 0) {
                            console.warn(`No hay items en ${source.name}`);
                            continue;
                        }
                        
                        articles = [];
                        items.forEach(item => {
                            const title = item.querySelector('title')?.textContent || '';
                            const description = item.querySelector('description')?.textContent || 
                                               item.querySelector('content\\:encoded')?.textContent || '';
                            const link = item.querySelector('link')?.textContent || '';
                            const pubDateText = item.querySelector('pubDate')?.textContent || 
                                               item.querySelector('dc\\:date')?.textContent || '';
                            
                            let pubDate;
                            try {
                                pubDate = new Date(pubDateText);
                                if (isNaN(pubDate.getTime())) {
                                    pubDate = new Date();
                                }
                            } catch {
                                pubDate = new Date();
                            }
                            
                            const cleanDescription = cleanHTML(description);
                            
                            articles.push({
                                title: cleanHTML(title),
                                description: cleanDescription,
                                link,
                                pubDate,
                                content: cleanHTML(title) + ' ' + cleanDescription,
                                source: source.name
                            });
                        });
                        
                        console.log(`Cargado ${articles.length} artículos de ${source.name}`);
                        success = true;
                        break;
                        
                    } catch (error) {
                        console.warn(`Error en ${source.name}:`, error.message);
                        continue;
                    }
                }
                
                if (!success) {
                    throw new Error('No se pudo cargar ninguna fuente RSS');
                }
                
                analyzeContent();
                refreshBtn.disabled = false;
                
            } catch (error) {
                console.error('Error crítico:', error);
                refreshBtn.disabled = false;
                useSampleData();
                showError(`No se pudieron cargar noticias en tiempo real. Mostrando datos de ejemplo. Error: ${error.message}`);
            }
        }
        
        // Función de fetch con timeout
        function fetchWithTimeout(url, timeout = 10000) {
            return new Promise((resolve, reject) => {
                const controller = new AbortController();
                const timeoutId = setTimeout(() => {
                    controller.abort();
                    reject(new Error('Timeout al cargar RSS'));
                }, timeout);
                
                fetch(url, {
                    signal: controller.signal,
                    headers: {
                        'Accept': 'application/xml, text/xml, */*',
                        'Cache-Control': 'no-cache'
                    }
                })
                .then(response => {
                    clearTimeout(timeoutId);
                    resolve(response);
                })
                .catch(error => {
                    clearTimeout(timeoutId);
                    reject(error);
                });
            });
        }

        function cleanHTML(html) {
            if (!html) return '';
            let cleanText = html
                .replace(/<[^>]*>/g, ' ')
                .replace(/ /g, ' ')

                .replace(/</g, '<')
                .replace(/>/g, '>')

                .replace(/[^\w\sáéíóúñÁÉÍÓÚÑ\-]/g, ' ')
                .replace(/\s+/g, ' ')

            return cleanText;

        // Analizar contenido
        function analyzeContent() {
            if (!articles || articles.length === 0) {
                useSampleData();
                return;
            }
            
            const termCounts = {};
            const now = new Date();
            const last24Hours = new Date(now.getTime() - (24 * 60 * 60 * 1000));
            
            // Palabras específicas de Almería para dar prioridad
            const almeriaKeywords = ['almería', 'almeriense', 'provincia', 'andalucía', 'mediterráneo'];
            
            articles.forEach(article => {
                const words = extractWords(article.content + ' ' + article.title);
                
                words.forEach(word => {
                    if (!termCounts[word]) {
                        termCounts[word] = { 
                            count: 0, 
                            articles: [],
                            isRecent: article.pubDate >= last24Hours
                        };
                    }
                    
                    // Dar más peso a palabras clave de Almería
                    const weight = almeriaKeywords.includes(word.toLowerCase()) ? 2 : 1;
                    termCounts[word].count += weight;
                    
                    if (!termCounts[word].articles.includes(article.title)) {
                        termCounts[word].articles.push(article.title);
                    }
                });
            });
            
            // Filtrar y procesar términos
            tagData = Object.entries(termCounts)
                .filter(([word, data]) => {
                    // Excluir palabras muy comunes o muy raras
                    if (data.count < 2) return false;
                    if (word.length < 4 || word.length > 20) return false;
                    if (stopWords.has(word.toLowerCase())) return false;
                    
                    // Excluir números y combinaciones raras
                    if (/^\d+$/.test(word)) return false;
                    if (/^[a-z]+\d+$/.test(word) || /^\d+[a-z]+$/.test(word)) return false;
                    
                    return true;
                })
                .map(([word, data]) => ({
                    name: word.charAt(0).toUpperCase() + word.slice(1),
                    count: data.count,
                    articles: data.articles.slice(0, 3), // Máximo 3 artículos
                    isRecent: data.isRecent,
                    color: tagColors[Math.floor(Math.random() * tagColors.length)]
                }))
                .sort((a, b) => b.count - a.count)
                .slice(0, 12); // Mostrar hasta 12 etiquetas
            
            if (tagData.length < 3) {
                useSampleData();
            } else {
                updateStats();
                generateTagCloud();
            }
        }

        function extractWords(text) {
            return text.toLowerCase()
                .normalize("NFD").replace(/[\u0300-\u036f]/g, "")
                .replace(/[^\w\s]/g, ' ')
                .split(/\s+/)
                .filter(word => word.length >= 3)
                .filter(word => !/^\d+$/.test(word));
        }
