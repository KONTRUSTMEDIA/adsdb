                      else if (roller.rollDiv[id].timeId) roller.stopRolling(id);

                      var idx = roller.rollDiv[id].current;
                      document.getElementById("rollDiv" + id + idx).style.display = 'none';
                      if (idx + 1 > roller.rollDiv[id].total && roller.rollDiv[id].pa) {
                          var nidx = parseInt(id.substr(id.length - 1));

                          roller.rollDiv[id].term = 0;
                      }
                      idx = (idx + 1 > roller.rollDiv[id].total) ? 0 : idx + 1;
                      document.getElementById("rollDiv" + id + idx).style.display = 'block';
                      roller.rollDiv[id].current = idx;
                      if (roller.rollDiv[id].term > 0) roller.startRolling(id);
                  },

                  gotoDiv: function(id, idx) {
                      if (typeof(roller.rollDiv[id]) == 'undefined') roller.newRolling(id);

                      document.getElementById("rollDiv" + id + roller.rollDiv[id].current).style.display = 'none';

                  newRolling: function() {
                      var id = arguments[0];
                      var term = (arguments[1]) ? arguments[1] : 0;
                      var pa = (arguments[2]) ? arguments[2] : '';
                      for (var i = 0; document.getElementById("rollDiv" + id + i) != null; i++);
                      roller.rollDiv[id] = new Object();
                      roller.rollDiv[id].total = i - 1;

                      roller.rollDiv[id].timeId = 0;
                      roller.rollDiv[id].pa = pa;
                      if (term > 0) roller.startRolling(id);
                      //if(pa) roller.newRolling(pa);
                  },

                      clearTimeout(roller.rollDiv[id].timeId)

                      var imgs = document.getElementsByTagName("img");

// 슬라이크 전체 크기(width 구하기)
const slide3 = document.querySelector(".slide2");
let slideWidth3 = slide3.clientWidth;

// 버튼 엘리먼트 선택하기
const prevBtn3 = document.querySelector(".slide_prev_button2");
const nextBtn3 = document.querySelector(".slide_next_button2");

// 슬라이드 전체를 선택해 값을 변경해주기 위해 슬라이드 전체 선택하기
const slideItems3 = document.querySelectorAll(".slide_item2");
// 현재 슬라이드 위치가 슬라이드 개수를 넘기지 않게 하기 위한 변수
const maxSlide3 = slideItems3.length;

// 버튼 클릭할 때 마다 현재 슬라이드가 어디인지 알려주기 위한 변수
let currSlideddd = 1;

// 페이지네이션 생성
const pagination3 = document.querySelector(".slide_pagination2");

for (let i = 0; i < maxSlide3; i++) {
  if (i === 0) pagination3.innerHTML += `<li class="active">•</li>`;
  else pagination3.innerHTML += `<li>•</li>`;
}

const paginationItems3 = document.querySelectorAll(".slide_pagination2 > li");

function nextMove3() {
  currSlideddd++;
  // 마지막 슬라이드 이상으로 넘어가지 않게 하기 위해서
  if (currSlideddd <= maxSlide3) {
    // 슬라이드를 이동시키기 위한 offset 계산
    const offset3 = slideWidth3 * (currSlideddd - 1);
	
    // 각 슬라이드 아이템의 left에 offset3 적용
    slideItems3.forEach((i) => {
      i.setAttribute("style", `left: ${-offset3}px`);
    });
    // 슬라이드 이동 시 현재 활성화된 pagination 변경
    paginationItems3.forEach((i) => i.classList.remove("active"));
    paginationItems3[currSlideddd - 1].classList.add("active");
  } else {
    currSlideddd--;
  }
}



function prevMove3() {
  currSlideddd--;
  // 1번째 슬라이드 이하로 넘어가지 않게 하기 위해서
  if (currSlideddd > 0) {
    // 슬라이드를 이동시키기 위한 offset3 계산
    const offset3 = slideWidth * (currSlideddd - 1);
    // 각 슬라이드 아이템의 left에 offset3 적용
    slideItems3.forEach((i) => {
      i.setAttribute("style", `left: ${-offset3}px`);
    });
    // 슬라이드 이동 시 현재 활성화된 pagination 변경
    paginationItems3.forEach((i) => i.classList.remove("active"));
    paginationItems3[currSlideddd - 1].classList.add("active");
  } else {
    currSlideddd++;
  }
}

// 버튼 엘리먼트에 클릭 이벤트 추가하기
nextBtn3.addEventListener("click", () => {
  // 이후 버튼 누를 경우 현재 슬라이드를 변경
  nextMove3();
});
// 버튼 엘리먼트에 클릭 이벤트 추가하기
prevBtn3.addEventListener("click", () => {

  prevMove3();

// 각 페이지네이션 클릭 시 해당 슬라이드로 이동하기

  // 각 페이지네이션마다 클릭 이벤트 추가하기
  paginationItems3[i].addEventListener("click", () => {
    // 클릭한 페이지네이션에 따라 현재 슬라이드 변경해주기(currSlideddd는 시작 위치가 1이기 때문에 + 1)
    currSlideddd = i + 1;
    // 슬라이드를 이동시키기 위한 offset3 계산
    const offset3 = slideWidth * (currSlideddd - 1);
    // 각 슬라이드 아이템의 left에 offset3 적용
