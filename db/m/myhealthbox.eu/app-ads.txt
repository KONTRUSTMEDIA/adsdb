function runFilteredSearch() {

	if (document.getElementById("filter_language")) {
    	SearchLanguageCode  = document.getElementById("filter_language").value;
    }

	if (document.getElementById("filter_country")) {
    	SearchCountryCode  = document.getElementById("filter_country").value;
    }

    	search_string  = document.getElementById("search_bar").value;

	onSearchClick(false);

// setDefault = true means we set default values for language and country according to search settings
// setDefault = false means we do not set default values for language and country for example when we use search filters
function onSearchClick(setDefault) {

	if (typeof setDefault === 'undefined') { setDefault = true; }
	
	updateSearchVars(setDefault);
	
	var formData = getSearchArray();
	url = Object.keys(formData).map(function(k) {
    			return encodeURIComponent(k) + '=' + encodeURIComponent(formData[k])
				}).join('&')
	window.location.href = _UIBasePath+"search.php?"+url;	
}

// setDefault = true means we set default values for language and country according to search settings
// setDefault = false means we do not set default values for language for example when we use search filters
function updateSearchVars(setDefault) {

	if (SearchCountryCode > 0) 
		search_country = SearchCountryCode;
	else search_country = "";
	
	if (SearchLanguageCode > 0)	
		search_lang = SearchLanguageCode;
	else {
		if (setDefault) 
			search_lang = _UILanguageCode;
		else search_lang = "";
	}
}

function getSearchArray() {
	return { 
		"q":search_string, 
		"c": ""+search_country,
		"l": ""+search_lang
        };
}

function resetFilters() {

	if (document.getElementById("filter_language")) {
    	document.getElementById("filter_language").value = 0;
    }

	if (document.getElementById("filter_country")) {
    	document.getElementById("filter_country").value = 0;
    }
	
	onSearchClick(true);
}
