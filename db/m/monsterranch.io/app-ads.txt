<!-- ===== Minimal JS to open/close modal + tabs, plus btnRegister() helper ===== -->
<script>
(function(){
  const modal = document.getElementById('auth-modal');
  const openBtns = document.querySelectorAll('[data-open-auth]');
  const closeEls = modal.querySelectorAll('[data-close]');
  const tabLoginBtn = document.getElementById('tab-login-btn');
  const tabRegisterBtn = document.getElementById('tab-register-btn');
  const tabLogin = document.getElementById('tab-login');
  const tabRegister = document.getElementById('tab-register');
  
  function toggleRegisterFields(enable) {
	document.querySelectorAll('#tab-register input, #tab-register select')
	  .forEach(el => el.disabled = !enable);
  }

  function setTab(which){
	const loginActive = (which==='login');
	tabLoginBtn.classList.toggle('is-active', loginActive);
	tabRegisterBtn.classList.toggle('is-active', !loginActive);
	tabLogin.classList.toggle('is-active', loginActive);
	tabRegister.classList.toggle('is-active', !loginActive);
	tabLoginBtn.setAttribute('aria-selected', loginActive);
	tabRegisterBtn.setAttribute('aria-selected', !loginActive);
  
	// ✅ KEY LINE
	toggleRegisterFields(!loginActive);
  }
  function open(which){
	modal.setAttribute('aria-hidden','false');
	setTab(which==='register' ? 'register' : 'login');
	setTimeout(()=>modal.querySelector('input,button,select,textarea')?.focus(), 50);
	// Lock background scroll for nicer UX
	document.documentElement.style.overflow = 'hidden';
  }
  function close(){
	modal.setAttribute('aria-hidden','true');
	document.documentElement.style.overflow = '';
  }

  // Existing buttons
  openBtns.forEach(b=>b.addEventListener('click', e=>open(e.currentTarget.dataset.openAuth)));
  closeEls.forEach(el=>el.addEventListener('click', close));
  modal.addEventListener('click', e=>{ if(e.target.dataset.close!==undefined) close(); });
  document.addEventListener('keydown', e=>{ if(e.key==='Escape' && modal.getAttribute('aria-hidden')==='false') close(); });
  tabLoginBtn.addEventListener('click',()=>setTab('login'));
  tabRegisterBtn.addEventListener('click',()=>setTab('register'));
  

  // Provide your legacy helper so existing onclick="btnRegister()" still works
  window.btnRegister = function(){ open('register'); };
  // Ensure register fields are disabled on initial load
  toggleRegisterFields(false);
})();
</script>
