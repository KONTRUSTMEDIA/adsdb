    function getLocalStats() {
      try {
        return JSON.parse(localStorage.getItem('minesweeper_stats')) || {};
      } catch { return {}; }
    }

    function saveLocalStats(stats) {
      try {
        localStorage.setItem('minesweeper_stats', JSON.stringify(stats));
      } catch { }
    }

    function recordLocalGameComplete(difficulty, completionTime, won) {
      const stats = getLocalStats();
      if (!stats[difficulty]) {
        stats[difficulty] = { total: 0, wins: 0, best_time: null };
      }
      stats[difficulty].total++;
      if (won) {
        stats[difficulty].wins++;
        if (completionTime && (!stats[difficulty].best_time || completionTime < stats[difficulty].best_time)) {
          stats[difficulty].best_time = completionTime;
        }
      }
      
      if (won) {
        const today = new Date().toISOString().split('T')[0];
        if (!stats._win_dates) stats._win_dates = [];
        if (!stats._win_dates.includes(today)) {
          stats._win_dates.push(today);
          stats._win_dates.sort().reverse();
          
          stats._win_dates = stats._win_dates.slice(0, 365);
        }
      }
      saveLocalStats(stats);
      return stats;
    }

    function computeLocalStreak(stats) {
      const dates = stats._win_dates || [];
      if (dates.length === 0) return 0;
      const today = new Date().toISOString().split('T')[0];
      const yesterday = new Date(Date.now() - 86400000).toISOString().split('T')[0];
      if (dates[0] !== today && dates[0] !== yesterday) return 0;
      let streak = 0;
      let expected = new Date(dates[0]);
      for (const d of dates) {
        const expectedStr = expected.toISOString().split('T')[0];
        if (d === expectedStr) {
          streak++;
          expected = new Date(expected.getTime() - 86400000);
        } else {
          break;
        }
      }
      return streak;
    }

    function mergeStats(local, remote) {
      
      const merged = {};
      const diffs = new Set([...Object.keys(local), ...Object.keys(remote)]);
      for (const diff of diffs) {
        if (diff.startsWith('_')) continue;
        const l = local[diff] || { total: 0, wins: 0, best_time: null };
        const r = remote[diff] || { total: 0, wins: 0, best_time: null };
        
        const useLocal = l.total >= r.total;
        const chosen = useLocal ? { ...l } : { ...r };
        
        if (l.best_time && r.best_time) {
          chosen.best_time = Math.min(l.best_time, r.best_time);
        } else {
          chosen.best_time = l.best_time || r.best_time;
        }
        chosen.win_pct = chosen.total > 0 ? Math.round((chosen.wins / chosen.total) * 100) : 0;
        merged[diff] = chosen;
      }
      return merged;
    }

    function renderStats(stats, streak) {
      const map = { beginner: 'easy', intermediate: 'medium', expert: 'hard' };
      for (const [diff, label] of Object.entries(map)) {
        const s = stats[diff];
        document.getElementById(`stat-${label}-total`).textContent = s ? s.total : '0';
        document.getElementById(`stat-${label}-wins`).textContent = s ? s.wins : '0';
        document.getElementById(`stat-${label}-pct`).textContent = s ? `${s.win_pct || 0}%` : '0%';
        document.getElementById(`stat-${label}-best`).textContent = s ? formatStatsTime(s.best_time) : '-';
      }
      const streakEl = document.getElementById('stats-streak');
      if (streak > 0) {
        streakEl.innerHTML = `<span class="streak-fire">🔥</span> ${streak} day streak`;
        streakEl.style.display = '';
      } else {
        streakEl.style.display = 'none';
      }
    }

    function formatStatsTime(seconds) {
      if (!seconds) return '-';
      const m = Math.floor(seconds / 60);
      const s = Math.floor(seconds % 60);
      return m > 0 ? `${m}:${s.toString().padStart(2, '0')}` : `${s}s`;
    }

    async function fetchPlayerStats() {
      if (!myId) return;

      
      const localStats = getLocalStats();
      const localStreak = computeLocalStreak(localStats);
      renderStats(localStats, localStreak);

      
      try {
        const url = `${API_BASE}/api/player-stats?player_id=${encodeURIComponent(myId)}`;
        const res = await fetch(url);
        if (!res.ok) return;
        const data = await res.json();
        if (!data.success) return;
        const remoteStats = data.stats || {};
        const remoteStreak = data.streak || 0;

        
        const merged = mergeStats(localStats, remoteStats);
        const bestStreak = Math.max(localStreak, remoteStreak);
        renderStats(merged, bestStreak);
      } catch (e) {
        
      }
    }

      async startNewGame() {
        this.gameOver = false;
        this.gameWon = false;
        this.firstClick = true;
        this.timer = 0;
        this.flaggedCount = 0;
        this.bv = 0;
        this.stopTimer();
        this.startTime = null;

        
        if (isDaily && !IS_LOCAL_DEV) {
          try {
            const today = new Date().toISOString().split('T')[0];
            const response = await fetch(`/api/daily-puzzle?date=${today}&difficulty=${this.currentDifficulty}`);
            if (response.ok) {
              const data = await response.json();
              if (data.success) {
                this.dailyPuzzleConfig = data.board_config;
                this.rows = this.dailyPuzzleConfig.rows;
                this.cols = this.dailyPuzzleConfig.cols;
                this.mineCount = this.dailyPuzzleConfig.mines;
              } else {
                console.error('Failed to load daily puzzle:', data.error);
                
                this.setupNormalGame();
              }
            } else {
              console.error('Failed to fetch daily puzzle');
              this.setupNormalGame();
            }
          } catch (error) {
            console.error('Error loading daily puzzle:', error);
            this.setupNormalGame();
          }
        } else if (isDaily && IS_LOCAL_DEV) {
          
          console.log('Local dev mode: simulating daily puzzle');
          this.simulateLocalDailyPuzzle();
        } else {
          this.setupNormalGame();
        }

        this.updateDifficultyLabel();
        this.updateFlagCounter();
        this.updateTimer();
        this.updateGameStatus('');
        this.update3BVDisplay();
        this.createBoard();
        this.prePlaceMines();
        this.renderBoard();

        
        this.trackGameEvent('game_start');
      }

      
      simulateLocalDailyPuzzle() {
        
        const today = new Date();
        const dateString = today.toISOString().split('T')[0];
        const seed = this.hashCode(dateString);

        
        this.seededRandom = this.createSeededRandom(seed);

        
        this.dailyPuzzleConfig = {
          rows: this.rows,
          cols: this.cols,
          mines: this.mineCount,
          positions: this.generateDeterministicMinePositions()
        };

        console.log('Local daily puzzle config:', this.dailyPuzzleConfig);
      }

      
      generateDeterministicMinePositions() {
        const positions = [];
        const totalCells = this.rows * this.cols;
        const availablePositions = Array.from({ length: totalCells }, (_, i) => i);

        
        for (let i = availablePositions.length - 1; i > 0; i--) {
          const j = Math.floor(this.seededRandom() * (i + 1));
          [availablePositions[i], availablePositions[j]] = [availablePositions[j], availablePositions[i]];
        }

        
        for (let i = 0; i < this.mineCount; i++) {
          const pos = availablePositions[i];
          positions.push({
            row: Math.floor(pos / this.cols),
            col: pos % this.cols
          });
        }

        return positions;
      }

      
      createSeededRandom(seed) {
        return function () {
          seed = (seed * 9301 + 49297) % 233280;
          return seed / 233280;
        };
      }

      
      hashCode(str) {
        let hash = 0;
        for (let i = 0; i < str.length; i++) {
          const char = str.charCodeAt(i);
          hash = ((hash << 5) - hash) + char;
          hash = hash & hash; 
        }
        return Math.abs(hash);
      }

      setupNormalGame() {
        const config = this.difficulties[this.currentDifficulty];
        this.rows = config.rows;
        this.cols = config.cols;
        this.mineCount = config.mines;
        this.dailyPuzzleConfig = null;
      }

      createBoard() {
        this.board = [];
        for (let row = 0; row < this.rows; row++) {
          this.board[row] = [];
          for (let col = 0; col < this.cols; col++) {
            this.board[row][col] = {
              isMine: false,
              isRevealed: false,
              isFlagged: false,
              neighborMines: 0
            };
          }
        }
      }

      
      prePlaceMines() {
        if (isDaily && this.dailyPuzzleConfig) {
          
          this.dailyPuzzleConfig.minePositions.forEach(position => {
            const row = Math.floor(position / this.cols);
            const col = position % this.cols;
            this.board[row][col].isMine = true;
          });
        } else {
          
          let minesPlaced = 0;
          while (minesPlaced < this.mineCount) {
            const row = Math.floor(Math.random() * this.rows);
            const col = Math.floor(Math.random() * this.cols);
            if (!this.board[row][col].isMine) {
              this.board[row][col].isMine = true;
              minesPlaced++;
            }
          }
        }
        this.calculateNeighborMines();
      }

      
      placeMinesExcluding(excludeRow, excludeCol) {
        this.board.forEach(r => r.forEach(c => { c.isMine = false; c.neighborMines = 0; }));
        let minesPlaced = 0;
        while (minesPlaced < this.mineCount) {
          const row = Math.floor(Math.random() * this.rows);
          const col = Math.floor(Math.random() * this.cols);
          if (!this.board[row][col].isMine && !this.isAdjacentTo(row, col, excludeRow, excludeCol)) {
            this.board[row][col].isMine = true;
            minesPlaced++;
          }
        }
        this.calculateNeighborMines();
      }

      
      
      isBoardSolvable(startRow, startCol) {
        const rows = this.rows, cols = this.cols;
        
        const mine  = Array.from({ length: rows }, (_, r) =>
          Array.from({ length: cols }, (_, c) => this.board[r][c].isMine));
        const num   = Array.from({ length: rows }, (_, r) =>
          Array.from({ length: cols }, (_, c) => this.board[r][c].neighborMines));

        
        const state = Array.from({ length: rows }, () => new Uint8Array(cols));
        const totalSafe = rows * cols - this.mineCount;
        let revealedCount = 0;

        
        const neighbors = (r, c) => {
          const out = [];
          for (let dr = -1; dr <= 1; dr++)
            for (let dc = -1; dc <= 1; dc++) {
              if (dr === 0 && dc === 0) continue;
              const nr = r + dr, nc = c + dc;
              if (nr >= 0 && nr < rows && nc >= 0 && nc < cols) out.push([nr, nc]);
            }
          return out;
        };

        
        const reveal = (r, c) => {
          if (state[r][c] !== 0 || mine[r][c]) return;
          state[r][c] = 1;
          revealedCount++;
          if (num[r][c] === 0) {
            for (const [nr, nc] of neighbors(r, c)) reveal(nr, nc);
          }
        };

        
        reveal(startRow, startCol);

        
        let changed = true;
        while (changed) {
          changed = false;

          
          for (let r = 0; r < rows; r++) {
            for (let c = 0; c < cols; c++) {
              if (state[r][c] !== 1) continue; 
              const nbrs = neighbors(r, c);
              let unknowns = [], flagged = 0;
              for (const [nr, nc] of nbrs) {
                if (state[nr][nc] === 0) unknowns.push([nr, nc]);
                else if (state[nr][nc] === 2) flagged++;
              }
              if (unknowns.length === 0) continue;
              const remaining = num[r][c] - flagged;

              
              if (remaining === 0) {
                for (const [ur, uc] of unknowns) {
                  reveal(ur, uc);
                  changed = true;
                }
              }
              
              else if (remaining === unknowns.length) {
                for (const [ur, uc] of unknowns) {
                  state[ur][uc] = 2;
                  changed = true;
                }
              }
            }
          }

          
          
          
          
          
          for (let r1 = 0; r1 < rows && !changed; r1++) {
            for (let c1 = 0; c1 < cols && !changed; c1++) {
              if (state[r1][c1] !== 1) continue;
              const nbrs1 = neighbors(r1, c1);
              let unknowns1 = [], flagged1 = 0;
              for (const [nr, nc] of nbrs1) {
                if (state[nr][nc] === 0) unknowns1.push(nr * cols + nc);
                else if (state[nr][nc] === 2) flagged1++;
              }
              const rem1 = num[r1][c1] - flagged1;
              if (unknowns1.length === 0) continue;
              const set1 = new Set(unknowns1);

              
              for (let dr = -2; dr <= 2; dr++) {
                for (let dc = -2; dc <= 2; dc++) {
                  if (dr === 0 && dc === 0) continue;
                  const r2 = r1 + dr, c2 = c1 + dc;
                  if (r2 < 0 || r2 >= rows || c2 < 0 || c2 >= cols) continue;
                  if (state[r2][c2] !== 1) continue;
                  const nbrs2 = neighbors(r2, c2);
                  let unknowns2 = [], flagged2 = 0;
                  for (const [nr, nc] of nbrs2) {
                    if (state[nr][nc] === 0) unknowns2.push(nr * cols + nc);
                    else if (state[nr][nc] === 2) flagged2++;
                  }
                  const rem2 = num[r2][c2] - flagged2;
                  if (unknowns2.length === 0) continue;
                  const set2 = new Set(unknowns2);

                  
                  if (unknowns1.length <= unknowns2.length && unknowns1.every(k => set2.has(k))) {
                    
                    const diff = unknowns2.filter(k => !set1.has(k));
                    const diffMines = rem2 - rem1;
                    if (diffMines === 0) {
                      
                      for (const k of diff) { reveal(Math.floor(k / cols), k % cols); changed = true; }
                    } else if (diffMines === diff.length) {
                      
                      for (const k of diff) { state[Math.floor(k / cols)][k % cols] = 2; changed = true; }
                    }
                  }

                  
                  if (unknowns2.length <= unknowns1.length && unknowns2.every(k => set1.has(k))) {
                    const diff = unknowns1.filter(k => !set2.has(k));
                    const diffMines = rem1 - rem2;
                    if (diffMines === 0) {
                      for (const k of diff) { reveal(Math.floor(k / cols), k % cols); changed = true; }
                    } else if (diffMines === diff.length) {
                      for (const k of diff) { state[Math.floor(k / cols)][k % cols] = 2; changed = true; }
                    }
                  }
                }
              }
            }
          }
        }

        return revealedCount === totalSafe;
      }

      
      
      
      generateSolvableBoard(startRow, startCol) {
        const MAX_ATTEMPTS = 1000;
        for (let attempt = 0; attempt < MAX_ATTEMPTS; attempt++) {
          this.placeMinesExcluding(startRow, startCol);
          if (this.isBoardSolvable(startRow, startCol)) {
            return; 
          }
        }
        
        
        console.warn('Could not find a fully solvable board after', MAX_ATTEMPTS, 'attempts. Using best effort.');
      }

      
      ensureSafeFirstClick(row, col) {
        if (isDaily && this.dailyPuzzleConfig) {
          
          const minesToMove = [];
          for (let dr = -1; dr <= 1; dr++) {
            for (let dc = -1; dc <= 1; dc++) {
              const r = row + dr, c = col + dc;
              if (r >= 0 && r < this.rows && c >= 0 && c < this.cols && this.board[r][c].isMine) {
                minesToMove.push({ row: r, col: c });
                this.board[r][c].isMine = false;
              }
            }
          }
          for (const mine of minesToMove) {
            let placed = false;
            while (!placed) {
              const newRow = Math.floor(Math.random() * this.rows);
              const newCol = Math.floor(Math.random() * this.cols);
              if (!this.board[newRow][newCol].isMine && !this.isAdjacentTo(newRow, newCol, row, col)) {
                this.board[newRow][newCol].isMine = true;
                placed = true;
              }
            }
          }
          if (minesToMove.length > 0) this.calculateNeighborMines();
        } else {
          
          this.generateSolvableBoard(row, col);
        }
      }

      isAdjacentTo(row, col, targetRow, targetCol) {
        return Math.abs(row - targetRow) <= 1 && Math.abs(col - targetCol) <= 1;
      }

      calculateNeighborMines() {
        for (let row = 0; row < this.rows; row++) {
          for (let col = 0; col < this.cols; col++) {
            if (!this.board[row][col].isMine) {
              this.board[row][col].neighborMines = this.countNeighborMines(row, col);
            }
          }
        }
      }

      countNeighborMines(row, col) {
        let count = 0;
        for (let dr = -1; dr <= 1; dr++) {
          for (let dc = -1; dc <= 1; dc++) {
            const newRow = row + dr;
            const newCol = col + dc;
            if (newRow >= 0 && newRow < this.rows && newCol >= 0 && newCol < this.cols) {
              if (this.board[newRow][newCol].isMine) count++;
            }
          }
        }
        return count;
      }

      renderBoard() {
        const gameBoard = document.getElementById('game-board');
        gameBoard.innerHTML = '';
        const cellSize = getComputedStyle(document.documentElement).getPropertyValue('--cell-size').trim() || '24px';
        gameBoard.style.gridTemplateColumns = `repeat(${this.cols}, ${cellSize})`;

        for (let row = 0; row < this.rows; row++) {
          for (let col = 0; col < this.cols; col++) {
            const cell = document.createElement('div');
            cell.className = 'cell';
            cell.dataset.row = row;
            cell.dataset.col = col;

            cell.addEventListener('click', () => this.handleCellClick(row, col));
            cell.addEventListener('contextmenu', (e) => {
              e.preventDefault();
              this.handleRightClick(row, col);
            });

            gameBoard.appendChild(cell);
          }
        }
      }

      handleCellClick(row, col) {
        if (this.gameOver || this.gameWon) return;

        const cell = this.board[row][col];

        
        
        if (cell.isRevealed && !cell.isMine && cell.neighborMines > 0) {
          this.chord(row, col);
          return;
        }

        if (cell.isRevealed || cell.isFlagged) return;

        if (this.firstClick) {
          this.firstClick = false;
          this.ensureSafeFirstClick(row, col);
          this.bv = this.calculate3BV();
          this.update3BVDisplay();
          this.startTimer();
          this.startTime = Date.now(); 
        }

        if (cell.isMine) {
          this.endGame(false);
          return;
        }

        this.revealCell(row, col);
        this.checkWin();
      }

      chord(row, col) {
        const cell = this.board[row][col];
        
        let adjFlags = 0;
        for (let dr = -1; dr <= 1; dr++) {
          for (let dc = -1; dc <= 1; dc++) {
            if (dr === 0 && dc === 0) continue;
            const nr = row + dr, nc = col + dc;
            if (nr >= 0 && nr < this.rows && nc >= 0 && nc < this.cols) {
              if (this.board[nr][nc].isFlagged) adjFlags++;
            }
          }
        }
        
        if (adjFlags !== cell.neighborMines) return;
        
        for (let dr = -1; dr <= 1; dr++) {
          for (let dc = -1; dc <= 1; dc++) {
            if (dr === 0 && dc === 0) continue;
            const nr = row + dr, nc = col + dc;
            if (nr >= 0 && nr < this.rows && nc >= 0 && nc < this.cols) {
              const neighbor = this.board[nr][nc];
              if (!neighbor.isRevealed && !neighbor.isFlagged) {
                if (neighbor.isMine) {
                  
                  this.endGame(false);
                  return;
                }
                this.revealCell(nr, nc);
              }
            }
          }
        }
        this.checkWin();
      }

      handleRightClick(row, col) {
        if (this.gameOver || this.gameWon) return;

        const cell = this.board[row][col];
        if (cell.isRevealed) return;

        cell.isFlagged = !cell.isFlagged;
        this.flaggedCount += cell.isFlagged ? 1 : -1;
        this.updateFlagCounter();
        this.updateCellDisplay(row, col);
        this.checkWin();
      }

      revealCell(row, col) {
        const cell = this.board[row][col];
        if (cell.isRevealed || cell.isFlagged) return;

        cell.isRevealed = true;
        this.updateCellDisplay(row, col);

        
        if (cell.neighborMines === 0) {
          for (let dr = -1; dr <= 1; dr++) {
            for (let dc = -1; dc <= 1; dc++) {
              const newRow = row + dr;
              const newCol = col + dc;
              if (newRow >= 0 && newRow < this.rows && newCol >= 0 && newCol < this.cols) {
                this.revealCell(newRow, newCol);
              }
            }
          }
        }
      }

      updateCellDisplay(row, col) {
        const cell = this.board[row][col];
        const cellElement = document.querySelector(`[data-row="${row}"][data-col="${col}"]`);

        cellElement.className = 'cell';

        if (cell.isFlagged) {
          cellElement.classList.add('flagged');
        } else if (cell.isRevealed) {
          cellElement.classList.add('revealed');
          if (cell.isMine) {
            cellElement.classList.add('mine');
          } else if (cell.neighborMines > 0) {
            cellElement.dataset.count = cell.neighborMines;
          }
        }
      }

      checkWin() {
        let revealedCount = 0;
        for (let row = 0; row < this.rows; row++) {
          for (let col = 0; col < this.cols; col++) {
            if (this.board[row][col].isRevealed) revealedCount++;
          }
        }

        if (revealedCount === (this.rows * this.cols - this.mineCount)) {
          this.endGame(true);
        }
      }

      async endGame(won) {
        this.gameOver = !won;
        this.gameWon = won;
        this.stopTimer();

        
        let completionTime = null;
        if (this.startTime) {
          completionTime = (Date.now() - this.startTime) / 1000; 
        }

        
        await this.trackGameEvent('game_complete', completionTime, won);
        fetchPlayerStats();

        
        for (let row = 0; row < this.rows; row++) {
          for (let col = 0; col < this.cols; col++) {
            if (this.board[row][col].isMine) {
              this.board[row][col].isRevealed = true;
              this.updateCellDisplay(row, col);
            }
          }
        }

        if (won) {
          const bvs = (completionTime && this.bv) ? (this.bv / completionTime).toFixed(2) : null;
          this.updateGameStatus(`🎉 You Won!${bvs ? ` — ${this.bv} 3BV · ${bvs} 3BV/s` : ''}`, 'win');

          
          if (isDaily && myId) {
            await this.submitDailyResult(completionTime, false);
          }
        } else {
          this.updateGameStatus('💥 Game Over!', 'lose');

          
          if (isDaily && myId) {
            await this.submitDailyResult(null, true);
          }
        }

        
        window.dispatchEvent(new Event('minesweeperblast:game-complete'));
      }

        }

        try {

            }
          } else {

          }
        } catch (error) {

        }
      }

      async trackGameEvent(eventType, completionTime, won) {
        if (IS_LOCAL_DEV) return;

        
        if (eventType === 'game_complete') {
          recordLocalGameComplete(this.currentDifficulty, completionTime, won);

        try {
          await fetch('/api/save-game', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },

              event_type: eventType,
              player_id: myId || null,
              difficulty: this.currentDifficulty,

            }),
          });
        } catch (e) {
          
        }
      }

        this.timerInterval = setInterval(() => {
          this.timer++;
          this.updateTimer();
        }, 1000);
      }

        if (this.timerInterval) {
          clearInterval(this.timerInterval);
          this.timerInterval = null;
        }
      }

        const timerElement = document.getElementById('timer');
        const str = String(this.timer).padStart(3, '0');

        for (const digit of str) {
          const span = document.createElement('span');

        }
      }



      updateFlagCounter() {
        const flagsElement = document.getElementById('flags-count');
        
        const remaining = Math.max(0, this.mineCount - this.flaggedCount);
        const str = String(remaining).padStart(2, '0');
        flagsElement.innerHTML = '';
        for (const digit of str) {
          const span = document.createElement('span');
          span.className = `digit digit-${digit}`;
          flagsElement.appendChild(span);
        }
      }

      
      
      
      
      calculate3BV() {
        const R = this.rows, C = this.cols;
        const visited = Array.from({ length: R }, () => new Uint8Array(C));
        const openingMember = Array.from({ length: R }, () => new Uint8Array(C));
        let bv = 0;

        
        for (let r = 0; r < R; r++) {
          for (let c = 0; c < C; c++) {
            if (visited[r][c] || this.board[r][c].isMine || this.board[r][c].neighborMines !== 0) continue;
            
            bv++;
            const stack = [[r, c]];
            visited[r][c] = 1;
            while (stack.length) {
              const [cr, cc] = stack.pop();
              openingMember[cr][cc] = 1;
              for (let dr = -1; dr <= 1; dr++) {
                for (let dc = -1; dc <= 1; dc++) {
                  if (dr === 0 && dc === 0) continue;
                  const nr = cr + dr, nc = cc + dc;
                  if (nr < 0 || nr >= R || nc < 0 || nc >= C) continue;
                  if (visited[nr][nc] || this.board[nr][nc].isMine) continue;
                  visited[nr][nc] = 1;
                  
                  
                  
                  if (this.board[nr][nc].neighborMines === 0) {
                    stack.push([nr, nc]);
                  }
                  openingMember[nr][nc] = 1; 
                }
              }
            }
          }
        }

<h2 id="what-is-minesweeper">What Is Minesweeper?</h2>
<p>Minesweeper is a single-player logic puzzle video game. The player is presented with a rectangular grid of covered cells. Hidden beneath some of those cells are mines. The objective is to uncover every safe cell without detonating a mine, using numbered clues that indicate how many mines border each revealed cell.</p>
<p>Minesweeper belongs to the broader category of <strong>puzzle video games</strong> and is one of the most widely played computer games in history. It was bundled with Microsoft Windows from 1990 through Windows 7, making it instantly recognizable to billions of PC users worldwide. Today it remains popular as a free browser-based game, a mobile app, and a competitive speedrunning discipline.</p>
<p>The game tests <strong>logical deduction, pattern recognition, and risk assessment</strong>. At its core, every Minesweeper puzzle is a constraint-satisfaction problem — and in 2000, mathematician Richard Kaye proved that the general Minesweeper problem is <a href="https://web.mat.bham.ac.uk/R.W.Kaye/minesw/ordmsw.htm">NP-complete</a>, meaning it is at least as hard as any problem in the complexity class NP.</p>

<h2 id="minesweeper-strategy-from-beginner-to-expert">Minesweeper Strategy: From Beginner to Expert</h2>
<p>Mastering Minesweeper requires learning a progression of techniques — from basic number reading to advanced pattern shortcuts and probability estimation.</p>
<h3 id="beginner-strategies">Beginner Strategies</h3>
<ul>
<li><strong>Start in a corner or edge.</strong> Corners and edges have fewer neighbors, so they are more likely to produce an opening (a cascade of blank cells) that gives you a large area of information to work with.</li>
<li><strong>Read the numbers carefully.</strong> A &ldquo;1&rdquo; cell with only one unrevealed neighbor means that neighbor is definitely a mine. A &ldquo;1&rdquo; cell with all neighbors accounted for (either revealed or flagged) means all remaining covered neighbors are safe.</li>
<li><strong>Satisfy constraints one at a time.</strong> Look at each number and ask: &ldquo;Can I determine any adjacent cell with certainty?&rdquo; If yes, act. If not, move on to the next number.</li>
</ul>
<h3 id="intermediate-strategies">Intermediate Strategies</h3>
<ul>
<li><strong>Use subtraction logic.</strong> If a &ldquo;3&rdquo; cell has two flagged neighbors, exactly one of its remaining covered neighbors is a mine. This type of constraint propagation is the backbone of Minesweeper logic.</li>
<li><strong>Compare adjacent numbers.</strong> Two numbered cells that share overlapping neighbor sets allow you to narrow down mine positions. For example, a &ldquo;1&rdquo; next to a &ldquo;2&rdquo; along an edge — the difference in their counts tells you exactly where the extra mine must be.</li>
<li><strong>Work the boundary.</strong> Focus on the frontier between revealed and unrevealed cells. This is where information density is highest.</li>
</ul>
<h3 id="advanced-strategies--patterns">Advanced Strategies &amp; Patterns</h3>
<p>Experienced players recognize common cell configurations instantly. Memorizing these patterns dramatically increases speed and accuracy.</p>
<h4 id="the-1-1-pattern">The 1-1 Pattern</h4>
<p>Two adjacent &ldquo;1&rdquo; cells along a wall share only one unrevealed neighbor between them. That shared neighbor is safe because one mine must be elsewhere.</p>
<h4 id="the-1-2-1-pattern">The 1-2-1 Pattern</h4>
<p>A &ldquo;1-2-1&rdquo; configuration along an edge or wall forces mines into the cells adjacent to the two &ldquo;1&rdquo; values, leaving the cell adjacent to the &ldquo;2&rdquo; as safe. This is one of the most common and useful patterns.</p>
<h4 id="the-1-2-2-1-pattern">The 1-2-2-1 Pattern</h4>
<p>An extension of 1-2-1 — the mines must be placed at the ends, and the middle cells are safe.</p>
<p><strong>Want to master every pattern?</strong> See our <a href="/patterns/">complete visual pattern guide</a> with interactive step-by-step diagrams for all 12 essential Minesweeper patterns.</p>
<h4 id="chording-for-speed">Chording for Speed</h4>
<p><strong>Chording</strong> (also called &ldquo;double-clicking&rdquo;) is the most important technique for competitive play. Once you have correctly flagged the mines around a number, chord that number to instantly reveal all remaining neighbors. Skilled players chain chords across the board, clearing large sections in rapid succession.</p>
<h4 id="the-5050-guess">The 50/50 Guess</h4>
<p>Some board states are <strong>logically ambiguous</strong> — two cells are equally likely to contain a mine, and no amount of deduction can distinguish them. This is called a 50/50. Advanced players:</p>
<ul>
<li>Delay 50/50 guesses as long as possible. Solving other areas may resolve the ambiguity.</li>
<li>Choose the cell that, if safe, reveals the most new information.</li>
<li>Accept that some percentage of expert games are unwinnable without guessing.</li>
</ul>
<h4 id="probability-and-edge-cases">Probability and Edge Cases</h4>
<p>When forced to guess, estimate probabilities. A cell on the boundary with many numbered neighbors constraining it is generally safer than an isolated unrevealed cell in the interior. The <strong>effective probability</strong> of a cell being a mine depends on global mine count, not just local numbers.</p>

<h2 id="minesweeper-tips--tricks">Minesweeper Tips &amp; Tricks</h2>
<p>Quick-reference tips for players at every level:</p>
<ol>
<li><strong>Don&rsquo;t flag everything.</strong> Flagging takes time. If you can logically determine a cell is safe, just click it without flagging nearby mines first. Competitive players often use a &ldquo;no-flag&rdquo; (NF) style.</li>

<h2 id="the-history-of-minesweeper">The History of Minesweeper</h2>
<p>Minesweeper has a rich history stretching back over four decades. Understanding where the game came from helps appreciate why it remains one of the most enduring puzzle games ever created.</p>
<h3 id="origins-19831989">Origins (1983–1989)</h3>
<p>The earliest known mine-themed puzzle game is <strong>Mined-Out</strong>, created by Ian Andrew and released for the Sinclair ZX Spectrum in 1983. In Mined-Out, the player navigated a character across a field of hidden mines using directional clues. While the mechanics differ from modern Minesweeper, the conceptual DNA is clear.</p>
<p>Another precursor is <strong>Mines</strong>, a game written by Tom Anderson for SunOS (Unix) in 1987, later ported to X Window System in 1990. Mines used a grid-based reveal mechanic closer to what we recognize today.</p>
<h3 id="microsoft-minesweeper-1990present">Microsoft Minesweeper (1990–Present)</h3>
<p>The version of Minesweeper that conquered the world was created by <strong>Curt Johnson and Robert Donner</strong> at Microsoft. It first appeared in the <strong>Microsoft Entertainment Pack for Windows</strong> in 1990 and was bundled directly into <strong>Windows 3.1</strong> in 1992.</p>
<p>Microsoft included Minesweeper (along with Solitaire) for a practical reason: to teach users how to use a mouse. Minesweeper specifically trained <strong>left-click</strong> (reveal), <strong>right-click</strong> (flag), and precise cursor targeting — skills that were not intuitive to new PC users in the early 1990s.</p>
<p>Minesweeper remained a default Windows application through <strong>Windows 7</strong> (2009). Starting with <strong>Windows 8</strong> (2012), Microsoft replaced the classic version with <strong>Microsoft Minesweeper</strong>, a free-to-play app available through the Microsoft Store, featuring updated graphics, daily challenges, and adventure mode.</p>
<h3 id="the-humanitarian-controversy">The Humanitarian Controversy</h3>
<p>In 2001, the <strong>International Campaign to Ban Landmines (ICBL)</strong> asked Microsoft to remove Minesweeper from Windows, calling it &ldquo;an offence against the victims of mines.&rdquo; Microsoft responded in Windows Vista by offering an alternative &ldquo;Flowers&rdquo; theme that replaced mines with flower graphics.</p>
<h3 id="minesweeper-today">Minesweeper Today</h3>
<p>Minesweeper is now played primarily as a <strong>free online browser game</strong> or mobile app. Dozens of implementations exist, including ours at Minesweeper Blast. Google Search includes a built-in version accessible by searching &ldquo;Minesweeper.&rdquo; The game&rsquo;s competitive community is thriving, with international tournaments, official world rankings, and active speedrunning communities.</p>

<h2 id="the-mathematics-of-minesweeper">The Mathematics of Minesweeper</h2>
<p>Minesweeper is not just a game — it is a legitimate subject of mathematical and computer science research.</p>
<h3 id="np-completeness">NP-Completeness</h3>
<p>In 2000, <strong>Richard Kaye</strong> of the University of Birmingham published a landmark proof demonstrating that <strong>Minesweeper Consistency</strong> (determining whether a given partially-revealed board has a valid mine arrangement) is <strong>NP-complete</strong>. This means there is no known efficient algorithm to solve arbitrary Minesweeper puzzles, and finding one would resolve the famous P vs. NP problem.</p>
<h3 id="co-np-completeness">Co-NP-Completeness</h3>
<p>A 2011 paper by Allan Scott, Ulrike Stege, and Iris van Rooij showed that when a board is guaranteed to be consistent, the problem of solving it is <strong>co-NP-complete</strong> — a related but distinct complexity class.</p>
<h3 id="phase-transitions">Phase Transitions</h3>
<p>Research by Ross Dempsey and Charles Guinn (2020) found that Minesweeper exhibits a <strong>phase transition</strong> analogous to the satisfiability threshold in k-SAT problems: when mine density exceeds approximately 25%, the probability of needing to guess increases sharply.</p>
<h3 id="algorithmic-approaches">Algorithmic Approaches</h3>
<p>Minesweeper solvers typically use:</p>
<ul>
<li><strong>Constraint propagation</strong> — Applying the basic rules exhaustively.</li>
<li><strong>Backtracking search</strong> — Trying possible mine configurations when local rules are insufficient.</li>
<li><strong>Probabilistic estimation</strong> — Computing the probability of each unrevealed cell being a mine based on all known constraints.</li>
</ul>
<p>These techniques connect Minesweeper to core topics in artificial intelligence, combinatorial optimization, and computational complexity.</p>

<h2 id="frequently-asked-questions-about-minesweeper">Frequently Asked Questions About Minesweeper</h2>
<p><strong>Is Minesweeper a game of luck or skill?</strong>
Minesweeper is primarily a game of <strong>skill and logic</strong>. Most cells can be determined through deduction. However, some board configurations (especially at Expert difficulty) contain unavoidable 50/50 guesses, introducing a minor element of chance.</p>
<p><strong>What is the best first move in Minesweeper?</strong>
Click a <strong>corner cell</strong>. Corners have only three neighbors, making them the most likely to produce a zero (blank opening), which cascade-reveals a large safe area.</p>
<p><strong>What does &ldquo;chording&rdquo; mean in Minesweeper?</strong>
Chording is the technique of clicking both mouse buttons simultaneously on a numbered cell whose flagged neighbor count matches its number. This instantly reveals all unflagged neighbors. It is the #1 technique for reducing solve time.</p>
<p><strong>Can every Minesweeper game be won without guessing?</strong>
In traditional Minesweeper, no — some mine configurations create logically ambiguous board states where at least one guess is required. However, <strong>Minesweeper Blast uses no-guess board generation</strong>, meaning every puzzle on this site is verified to be solvable through pure logic before it is presented to you. If you are looking for no-guess Minesweeper, you are in the right place.</p>
<p><strong>What is NF (No Flag) Minesweeper?</strong>
NF is a competitive play style where the player never places flags. Instead, they rely entirely on left-clicks and chords triggered by satisfied number constraints from already-revealed cells. NF is faster for top players because it eliminates the time spent right-clicking.</p>
<p><strong>How is the mine counter calculated?</strong>
Mine counter = Total mines − Number of flags placed. If you place too many flags, the counter goes negative. The counter does not indicate whether your flags are correctly placed.</p>
<p><strong>Why was Minesweeper included in Windows?</strong>
Microsoft included Minesweeper in Windows 3.1 (1992) to teach new computer users how to <strong>left-click, right-click, and precisely target small on-screen elements</strong> with a mouse — fundamental skills for the graphical user interface era.</p>
