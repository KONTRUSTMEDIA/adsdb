                    <h2><i class="fas fa-newspaper"></i> ¿Qué es el Centro de Operaciones de Emergencia (COE)?</h2>
                    
                    <p>El Centro de Operaciones de Emergencia (COE) es una instancia estratégica del Instituto Nacional de Salud de Colombia que funciona como eje articulador de la respuesta ante situaciones de emergencia en salud pública. Este sistema integrado permite la coordinación efectiva entre diferentes entidades del sector salud, autoridades gubernamentales y organismos de socorro.</p>

                    <h3><i class="fas fa-exclamation-triangle"></i> Funciones Principales</h3>
                    
                    <div class="row">
                        <div class="col-md-6">
                            <ul>
                                <li><strong>Vigilancia epidemiológica:</strong> Monitoreo continuo de eventos de salud</li>
                                <li><strong>Alerta temprana:</strong> Detección precoz de riesgos</li>
                                <li><strong>Coordinación intersectorial:</strong> Trabajo con múltiples actores</li>
                            </ul>
                        </div>
                        <div class="col-md-6">
                            <ul>
                                <li><strong>Asistencia técnica:</strong> Apoyo a entidades territoriales</li>
                                <li><strong>Gestión de datos:</strong> Toma de decisiones basada en evidencia</li>
                                <li><strong>Capacitación:</strong> Fortalecimiento del talento humano</li>
                            </ul>
                        </div>
                    </div>

                    <h3><i class="fas fa-users"></i> Poblaciones Prioritarias</h3>

        // Cargar artículos recientes del blog
        function cargarArticulosRecientes() {
            const contenedor = document.getElementById('articulos-recientes');
            
            firebase.database().ref('BLOG/articulos')
                .orderByChild('fecha_creacion')
                .limitToLast(3)
                .once('value')
                .then((snapshot) => {
                    let articulos = [];
                    
                    snapshot.forEach((childSnapshot) => {
                        const articulo = childSnapshot.val();
                        if (articulo.publicado === true) {
                            articulos.push(articulo);
                        }
                    });
                    
                    // Ordenar por fecha (más recientes primero)
                    articulos.reverse();
                    
                    if (articulos.length === 0) {
                        contenedor.innerHTML = `
                            <div class="col-12 text-center py-5">
                                <i class="fas fa-newspaper" style="font-size: 3rem; color: #ccc;"></i>
                                <p class="mt-3 text-muted">No hay artículos publicados aún.</p>
                                <a href="blog/admin.html" class="btn btn-primary mt-3">
                                    <i class="fas fa-plus"></i> Crear Primer Artículo
                                </a>
                            </div>
                        `;
                        return;
                    }
                    
                    let html = '';
                    articulos.forEach(articulo => {
                        const fecha = new Date(articulo.fecha_creacion).toLocaleDateString('es-CO', {
                            year: 'numeric',
                            month: 'long',
                            day: 'numeric'
                        });
                        
                        const resumen = articulo.resumen || extraerResumen(articulo.contenido);
                        
                        html += `
                            <div class="col-md-4 mb-4">
                                <div class="card service-card h-100">
                                    <div class="card-body">
                                        ${articulo.categoria ? `<span class="category-badge mb-2">${articulo.categoria}</span>` : ''}
                                        <h5 class="article-title">${articulo.titulo}</h5>
                                        <p class="article-excerpt">${resumen}</p>
                                        <div class="article-meta mt-3">
                                            <small>
                                                <i class="fas fa-user"></i> ${articulo.autor}
                                            </small>
                                            <br>
                                            <small>
                                                <i class="fas fa-calendar"></i> ${fecha}
                                            </small>
                                        </div>
                                        <a href="blog/articulo.html?id=${articulo.id}" class="btn btn-primary btn-sm mt-3 w-100">
                                            Leer artículo <i class="fas fa-arrow-right"></i>
                                        </a>
                                    </div>
                                </div>
                            </div>
                        `;
                    });
                    
                    contenedor.innerHTML = html;
                })
                .catch((error) => {
                    console.error('Error al cargar artículos:', error);
                    contenedor.innerHTML = `
                        <div class="col-12 text-center py-5">
                            <p class="text-danger">Error al cargar los artículos.</p>
                        </div>
                    `;
                });
        }
        
        function extraerResumen(contenido, maxCaracteres = 120) {
            const textoPlano = contenido.replace(/<[^>]*>/g, '');
            if (textoPlano.length <= maxCaracteres) return textoPlano;
            return textoPlano.substring(0, maxCaracteres) + '...';
        }
        
        // Cargar artículos al cargar la página
        document.addEventListener('DOMContentLoaded', cargarArticulosRecientes);
