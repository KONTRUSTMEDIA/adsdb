document.addEventListener('DOMContentLoaded', function () {

  const config = {

    textColor: '#d42d7b',
    selectedCake: null,
    isDragging: false,
    fontStyle: "'Shivaji01', Arial, sans-serif",
  };

  // DOM Elements
  const elements = {
    cakeOptions: document.querySelectorAll('.cake-option'),
    customizationStep: document.querySelector('.customization-step'),
    previewSection: document.querySelector('.preview-section'),
    canvas: document.getElementById('cake-canvas'),
    ctx: document.getElementById('cake-canvas').getContext('2d'),
    nameInput: document.getElementById('recipient-name'),
    sizeSlider: document.getElementById('text-size'),
    sizeValue: document.getElementById('size-value'),
    textColorInput: document.getElementById('text-color'),
    fontStyleSelect: document.getElementById('font-style'),
    cakeForm: document.getElementById('cake-form'),
    downloadBtn: document.getElementById('download-cake'),
  };

  // Cake Images
  const cakeImages = {
    round: new Image(),
    heart: new Image(),
    square: new Image(),
  };

  // Load images
  cakeImages.round.src = 'https://marathipremi.in/wp-content/uploads/2025/04/Round-cake-base.png';
  cakeImages.heart.src = 'https://marathipremi.in/wp-content/uploads/2025/04/Heart-shaped-cake-base.png';

  // Initialize
  setupEventListeners();

  function setupEventListeners() {
    // Prevent Enter key for new lines
    elements.nameInput.addEventListener('keydown', function (e) {
      if (e.key === 'Enter') {
        e.preventDefault(); // Block newline
      }
    });

    // Cake selection
    elements.cakeOptions.forEach((option) => {
      option.addEventListener('click', function () {
        config.selectedCake = this.dataset.base;
        elements.cakeOptions.forEach((opt) => opt.classList.remove('active'));
        this.classList.add('active');
        elements.customizationStep.style.display = 'block';
      });
    });

    // Text size control
    elements.sizeSlider.addEventListener('input', function () {
      config.textSize = parseInt(this.value);
      elements.sizeValue.textContent = `${this.value}px`;
      drawCakePreview();
    });

    // Text color control
    elements.textColorInput.addEventListener('input', function () {
      config.textColor = this.value;
      drawCakePreview();
    });

    // Font style control
    elements.fontStyleSelect.addEventListener('change', function () {
      config.fontStyle = this.value;
      drawCakePreview();
    });

    // Position buttons
    document.querySelectorAll('.position-btn').forEach((btn) => {
      btn.addEventListener('click', function () {
        config.textPos.x += parseInt(this.dataset.dx || 0);
        config.textPos.y += parseInt(this.dataset.dy || 0);
        drawCakePreview();
      });
    });

    // Form submission
    elements.cakeForm.addEventListener('submit', function (e) {
      e.preventDefault();
      if (!config.selectedCake) {
        alert('कृपया प्रथम केकचा आकार निवडा');
        return;
      }

      let nameValue = elements.nameInput.value.trim().replace(/\n/g, ''); // Remove any newlines
      elements.nameInput.value = nameValue;

      if (!nameValue) {
        alert('कृपया प्रिय व्यक्तीचे नाव/संदेश टाका');
        return;
      }

      elements.previewSection.style.display = 'block';
      drawCakePreview();
    });

    // Canvas interactions
    elements.canvas.addEventListener('mousedown', startDrag);
    elements.canvas.addEventListener('mousemove', dragText);
    elements.canvas.addEventListener('mouseup', endDrag);
    elements.canvas.addEventListener('mouseleave', endDrag);

    // Download button
    elements.downloadBtn.addEventListener('click', downloadCake);
  }

  function drawCakePreview() {
    // Clear canvas
    elements.ctx.clearRect(0, 0, elements.canvas.width, elements.canvas.height);

    // Draw cake
    const cakeImg = cakeImages[config.selectedCake];
    if (cakeImg && cakeImg.complete && cakeImg.naturalWidth > 0) {
      const scale = Math.min(elements.canvas.width / cakeImg.width, elements.canvas.height / cakeImg.height);
      const width = cakeImg.width * scale;
      const height = cakeImg.height * scale;
      const x = (elements.canvas.width - width) / 2;
      const y = (elements.canvas.height - height) / 2;
      elements.ctx.drawImage(cakeImg, x, y, width, height);
    }

    // Draw name/message (single line only)
    let name = elements.nameInput.value.replace(/\n/g, '').trim();
    if (!name) return;

    elements.ctx.font = `${config.textSize}px ${config.fontStyle}`;
    elements.ctx.fillStyle = config.textColor;
    elements.ctx.textAlign = 'center';
    elements.ctx.fillText(name, config.textPos.x, config.textPos.y);

function populateDropdowns() {
    const daySelect = document.getElementById("day");
    const yearSelect = document.getElementById("year");

    for (let i = 1; i <= 31; i++) {
        const option = document.createElement("option");
        option.value = i;
        option.text = i;
        daySelect.appendChild(option);
    }

    const currentYear = new Date().getFullYear();
    for (let i = currentYear; i >= 1900; i--) {
        const option = document.createElement("option");
        option.value = i;
        option.text = i;
        yearSelect.appendChild(option);
    }
}

function calculateAge() {
    const day = parseInt(document.getElementById("day").value, 10);
    const month = parseInt(document.getElementById("month").value, 10);
    const year = parseInt(document.getElementById("year").value, 10);

    if (!day || !month || !year) {
        document.getElementById("result").innerText = "कृपया पूर्ण दिनांक निवडा.";
        return;
    }

    const birthdate = new Date(year, month, day);
    const today = new Date();
    let years = today.getFullYear() - birthdate.getFullYear();
    let months = today.getMonth() - birthdate.getMonth();
    let days = today.getDate() - birthdate.getDate();

    if (days < 0) {
        months--;
        days += new Date(today.getFullYear(), today.getMonth(), 0).getDate();
    }
    if (months < 0) {
        years--;
        months += 12;
    }

    document.getElementById("result").innerText =
        `तुमचे वय: ${years} वर्षे, ${months} महिने, ${days} दिवस`;
}

window.onload = populateDropdowns;
