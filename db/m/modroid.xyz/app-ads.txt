        function timeAgo(dateString) {
            if (!dateString) return '';
            const date = new Date(dateString);
            const now = new Date();
            const seconds = Math.round((now - date) / 1000);
            const minutes = Math.round(seconds / 60);
            const hours = Math.round(minutes / 60);
            const days = Math.round(hours / 24);

            if (days > 365) return `hace ${Math.floor(days / 365)} años`;
            if (days > 30) return `hace ${Math.floor(days / 30)} meses`;
            if (days > 7) return `hace ${Math.floor(days / 7)} semanas`;
            if (days > 1) return `hace ${days} días`;
            if (hours > 1) return `hace ${hours} horas`;
            if (minutes > 1) return `hace ${minutes} minutos`;
            return `hace un momento`;
        }

        async function renderRecommendedApps(category, currentSlug) {
            const recommendedContainer = document.getElementById('recommended-apps-container');
            if (!recommendedContainer) return;

            try {
                const q = query(collection(db, 'apps'), where("category", "==", category), limit(6));
                const querySnapshot = await getDocs(q);
                
                const recommendedApps = querySnapshot.docs
                    .map(doc => doc.data())
                    .filter(app => app.slug !== currentSlug);

                if (recommendedApps.length > 0) {
                    const appsHTML = recommendedApps.map(app => `
                        <a href="/${app.slug}" class="recommended-card flex items-center gap-4 -mx-2 px-2">
                            <img src="${app.imgSrc}" class="w-14 h-14 rounded-xl object-cover" alt="Icono de ${app.title}" onerror="this.onerror=null;this.src='https://placehold.co/64x64/1c1c1e/FFFFFF?text=Icon';">
                            <div>
                                <h3 class="font-semibold text-white">${app.title}</h3>
                                <p class="text-sm text-gray-400">${app.category}</p>
                            </div>
                        </a>
                    `).join('');
                    
                    recommendedContainer.innerHTML = `
                        <section class="space-y-4">
                            <h2 class="section-title">También podría gustarte</h2>
                            <div class="border-t border-[var(--divider-color)] pt-2">${appsHTML}</div>
                        </section>
                    `;
                }
            } catch (error) {
                console.error("Error loading recommended apps:", error);
            }
        }
        
        async function loadApp() {
            const slug = window.location.pathname.substring(1); 
            
            if (!slug || slug.endsWith('.html') || slug === 'admin') {
                loadingIndicator.style.display = 'none';
                if(window.location.pathname !== '/index.html') {
                    appContainer.innerHTML = '<p class="text-center text-red-500">URL inválida.</p>';
                }
                return;
            }

            try {
                const q = query(collection(db, 'apps'), where("slug", "==", slug), limit(1));
                const querySnapshot = await getDocs(q);

                if (!querySnapshot.empty) {
                    const appData = querySnapshot.docs[0].data();
                    document.title = `${appData.title ?? 'App'} | Modroid`;
                    document.querySelector('meta[name="description"]').setAttribute('content', (appData.description ?? '').substring(0, 160));
                    renderAppDetails(appData);
                    if (appData.category) {
                        renderRecommendedApps(appData.category, appData.slug);
                    }
                } else {
                    loadingIndicator.style.display = 'none';
                    appContainer.innerHTML = `<p class="text-center text-red-500">La aplicación "${slug}" no se ha encontrado. Verifica que la URL sea correcta.</p>`;
                }
            } catch (error) {
                console.error("Error al cargar la aplicación: ", error);
                loadingIndicator.style.display = 'none';
                appContainer.innerHTML = '<p class="text-center text-red-500">Ocurrió un error al cargar la aplicación. Revisa la consola para más detalles.</p>';
            }
        }
        
        const observer = new IntersectionObserver((entries) => {

                stickyInfo.classList.toggle('opacity-100', isSticky);
                stickyInfo.classList.toggle('pointer-events-auto', isSticky);
                stickyAction.classList.toggle('opacity-100', isSticky);
                stickyAction.classList.toggle('pointer-events-auto', isSticky);
                header.style.borderColor = isSticky ? 'var(--divider-color)' : 'transparent';
                
                backButton.classList.toggle('opacity-0', isSticky);
                backButton.classList.toggle('pointer-events-none', isSticky);
            });
        }, { rootMargin: "-56px 0px 0px 0px" }); // 56px es la altura del encabezado (h-14)

                    });

            }
