<script>
// 🔹 Variables globales
let artistasData = [];

// 🔹 Cargar JSON y llenar el filtro
async function cargarArtistas() {
  try {
    const response = await fetch('enlacesdescarga.json?v=' + Date.now());
    artistasData = await response.json();

    // Obtener nombres únicos ordenados A-Z
    const nombresOrdenados = [...new Set(artistasData.map(a => a.nombre))]
      .sort((a, b) => a.localeCompare(b, 'es', { sensitivity: 'base' }));

    const select = document.getElementById('filtroSelect');
    nombresOrdenados.forEach(nombre => {
      const option = document.createElement('option');
      option.value = nombre;
      option.textContent = nombre;
      select.appendChild(option);
    });
  } catch (error) {
    console.error("Error al cargar artistas:", error);
  }
}

// 🔹 Mostrar resultados al seleccionar un artista
function filtrarPorArtista() {
  const select = document.getElementById('filtroSelect');
  const artistaSeleccionado = select.value;

  if (!artistaSeleccionado) return; // Si no hay selección, no hace nada

  // Filtrar datos del artista seleccionado
  const resultados = artistasData.filter(item =>
    item.nombre.toLowerCase() === artistaSeleccionado.toLowerCase()
  );

  // Mostrar resultados
  mostrarResultados(resultados);
}

// 🔹 Función que muestra los resultados en pantalla (ajústala si tu HTML usa otro formato)
function mostrarResultados(lista) {
  const contenedor = document.getElementById('resultados');
  contenedor.innerHTML = "";

  if (lista.length === 0) {
    contenedor.innerHTML = "<p>No se encontraron resultados.</p>";
    return;
  }

  lista.forEach((item, index) => {
    const div = document.createElement('div');
    div.className = 'artista-card';
    div.innerHTML = `
      <div class="artista-item">
        <span class="numero">${index + 1}</span>
        <img src="${item.imagen}" alt="${item.nombre}">
        <div class="info">
          <p><strong>ID:</strong> ${item.id}</p>
          <p><strong>Nombre:</strong> ${item.nombre}</p>
        </div>
      </div>
    `;
    contenedor.appendChild(div);
  });
}

// 🔹 Ejecutar al cargar la página
document.addEventListener("DOMContentLoaded", cargarArtistas);
</script>
