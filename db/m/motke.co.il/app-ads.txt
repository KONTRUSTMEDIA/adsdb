/* pages_state */ 
/* PageState Factory */function WebsiteStates(cookieName){
    this.cookieName = cookieName;
    this.currentPage = '';
    this.currentCategoryName = '';

    /* The state object of all pages-categories */
    this.state = {
        pages: {}
    };

    /* Returns the state of all pages */
    this.getState = function() {
        return this.state;
    };
    
    /* Setting the current page for modify */
    this.setCurrentPage = function(page) {
        this.currentPage = page;
    };
    
    this.getCurrentPage = function() {
        return this.currentPage;
    };
    
    /* Setting the current category for modify */
    this.setCurrentCategoryName = function(category) {
        this.currentCategoryName = category;
    };

    this.getCurrentCategoryName = function() {
        return this.currentCategoryName;
    };

    /* Setting the state of the selected page-category */
    this.createCategoryState = function(catName, newStart, newLimit, newItem) {
        if(typeof this.state.pages[this.getCurrentPage()] != 'undefined'){
            var categoryState = new CategoryState(catName);
            categoryState.setStart(newStart);
            categoryState.setLimit(newLimit);
            categoryState.setItem(newItem);
            this.state.pages[this.getCurrentPage()]['categories'][catName] = categoryState.getState();
            return categoryState;
        } else {
            return {};
        }
    };

    this.getCurrentCategory = function() {
        if(this.isCurrentCategoryExist()){
            return this.state.pages[this.getCurrentPage()]['categories'][this.getCurrentCategoryName()];
        } else {
            return {};
        }
    };

    /* Return an object of current category */
    this.getCurrentCategoryObject = function(){
        if(this.isCurrentCategoryExist()){
            var category = new CategoryState(this.getCurrentCategoryName());
            var categoryState = this.state.pages[this.getCurrentPage()]['categories'][this.getCurrentCategoryName()];
            category.setStart(categoryState.start);
            category.setLimit(categoryState.limit);
            category.setItem(categoryState.item);
            return category;
        } else {
            return new CategoryState('');
        }
    };

    this.resetCurrentCategory = function(){
        if(this.isCurrentCategoryExist()){
            delete this.state.pages[this.getCurrentPage()]['categories'][this.getCurrentCategoryName()];
            this.toCookie();
        }
    };

    this.isCurrentCategoryExist = function(){
        if(typeof this.state.pages[this.getCurrentPage()] != 'undefined' && typeof this.state.pages[this.getCurrentPage()]['categories'][this.getCurrentCategoryName()] != 'undefined'){
            return true;
        } else {
            return false;
        }
    };

    this.createPageState = function(pageName) {
        return new PageState(pageName);
    };

    this.addPageState = function(pageState) {
        if (pageState instanceof PageState && typeof this.state.pages[pageState.getName()] === 'undefined') {
            this.state.pages[pageState.getName()] = pageState.getState();
            return true;
        } else {
            return false;
        }
    };

    this.stringifyState = function() {
        return JSON.stringify(this.state);
    };

    /* Getting the state object from cookie */
    this.fromCookie = function() {
        try {
            if(readCookie(this.cookieName) !== null){
                this.state = JSON.parse(readCookie(this.cookieName));
            } else {
                location.href = location.href.replace(location.hash, '').replace(location.search, '');
            }
        } catch (error) {}
        return this.state;
    };
    
    /* Setting the state object to cookie */
    this.toCookie = function() {
        createCookie(
            this.cookieName, 
            this.stringifyState(),
            365
        );
        return this.cookieName;
    };
}

function PageState(pageName) {
    this.name = pageName;
    this.state = {
      categories: {},
      source: /.*#back_pointer$/.test(location.href) ? "back" : "navigation"
    };
    /* Expects CategoryState instance */
    this.pushCategory = function(categoryState) {
        if(categoryState instanceof CategoryState){
            this.state.categories[categoryState.getName()] = categoryState.getState();
            return true;
        } else {
            return false;
        }
    };

    this.getName = function() {
        return this.name;
    };

    /* Returns Stage object */
    this.getState = function() {
        return this.state;
    };

    this.getString = function(){
        return JSON.stringify(this.getState());
    };
}

function CategoryState(categoryName) {
    this.name = categoryName;
    this.start = 0;
    this.limit = 0;
    this.item = 0;

    this.setStart = function(num) {
        this.start = parseInt(num ,10);
    };

    this.setLimit = function(num) {
        this.limit = parseInt(num ,10);
    };

    this.setItem = function(num) {
        this.item = parseInt(num ,10);
    };

    this.getName = function() {
        return this.name;
    };

    this.getState = function() {
        return {
            start: this.start,
            limit: this.limit,
            item: this.item
        };
    };

    this.getString = function(){
        return JSON.stringify(this.getState());
    };
}

function formatTime(time) {
    // Calculate total seconds
    const totalSeconds = Math.floor(time / 1000);

    // Calculate minutes and seconds
    const minutes = Math.floor(totalSeconds / 60);
    const seconds = totalSeconds % 60;

    // Format minutes and seconds as two digits
    const formattedMinutes = minutes.toString().padStart(2, '0');
    const formattedSeconds = seconds.toString().padStart(2, '0');

    // Return the formatted time
    return `${formattedMinutes}:${formattedSeconds}`;
}

function readCookie(name) {
    var nameEQ = name + "=";
    var ca = document.cookie.split(';');
    for(var i=0; i<ca.length; i++) {
        var c = ca[i];
        while (c.charAt(0)==' ') c = c.substring(1,c.length);
        if (c.indexOf(nameEQ) == 0) return c.substring(nameEQ.length,c.length);
    }
    return null;
}

function createCookie(name,value,days) {
    if (days) {
        var date = new Date();
        date.setTime(date.getTime()+(days*24*60*60*1000));
        var expires = "; expires="+date.toGMTString();
    }
    else var expires = "";
    document.cookie = name+"="+value+expires+"; path=/";
}

                    }
                }, 2000);
            }

            module_like.data.onList = true;

            switch (action) {
                case "hide":
                    next.hide();
                break;
                case "show": 
                    next.show();
                break;
                default:
                    next.toggle();
                break;
            }
        }
    },
    animend: function (container) {
        if ($(container).hasClass("appear")) {
            $(container).removeClass("appear");
        }
    }

function removeDuplicateItems(data){
    $('a[data-item-unique-added=false]').each(function(){
        /* Get current item id*/
        var currentUniqueId = $(this).attr('data-item-unique');

        /* Find id in array*/
        var isIdInArray = loadMoreItemsIds.find(function(id){
            return id == currentUniqueId;
        });

        /* Check if item already exist*/
        /* If exist, remove it, else, mark as added*/
        if(isIdInArray != undefined){
            $(this).remove();
        } else {
            loadMoreItemsIds.push(currentUniqueId);
            $(this).attr('data-item-unique-added', true);
        }
    });
}

        props.limit = aboveZero(props.container.attr("data-ajax-limit") || props.loadMoreBtnElem.attr("data-ajax-limit"));
        props.ajaxLimit = aboveZero(props.limit);
        props.params = props.container.attr("data-ajax-params") || props.loadMoreBtnElem.attr("data-ajax-params");
        props.start = 0;
        props.firstStart = 0;
        props.ajaxStart = 0;

        /* Load more direction */
        if(props.loadDirection === 'up'){
            /* UP */
            props.firstStart = aboveZero(props.container.attr("data-ajax-first-start"));
            if(props.firstStart < props.limit){
                props.ajaxLimit = props.firstStart;
            }
            props.firstStart = aboveZero(props.firstStart - props.limit);
            props.ajaxStart = props.firstStart;
        } else {
            /* DOWN */
            props.start = aboveZero(props.container.attr("data-ajax-start"));
            props.ajaxStart = props.start;
        }

        if(props.params){ /*equivelant to typeof params !== "undefined" && params!=""*/
            props.params = JSON.parse(props.params);
        }

        $.ajax({
            type: "POST",
            url: "https://www.motke.co.il/ajaxModule",
            data: {
                limit: props.ajaxLimit,
                start: props.ajaxStart,
                module: props.ajaxModule,
                func: props.ajaxFunc,
                params: props.params
            },
            success: function(data){
                var output = JSON.parse(data);
                handleMoreResults(output, props);
                var locationSearchPage = new RegExp('%D7%AA%D7%95%D7%A6%D7%90%D7%95%D7%AA_%D7%97%D7%99%D7%A4%D7%95%D7%A9');
                if (output.status === 'reload') location.reload();
                if (locationSearchPage.test(location.href))
                    $(function() {
                        $('.search-results-wrapper a[href]').attr('target', '_blank');
                    });
            }
        });
    });
    $(window).scroll(function() {
        /* Auto load more UP when scrolling */
        if(isScrolledIntoView('[data-tab-current=true]:not(.disable-auto-scroll) .moreBtn[data-load-direction=up]')){
            if($('.moreBtn[data-load-direction=up]').prop('disabled') === false){
                $('.moreBtn[data-load-direction=up]').click().prop('disabled', true);
            }
        }

        /* Auto load more DOWN when scrolling */
        if(isScrolledIntoView('[data-tab-current=true]:not(.disable-auto-scroll) .moreBtn[data-load-direction=down]')){
        if($('.moreBtn[data-load-direction=down]').prop('disabled') === false){
                $('.moreBtn[data-load-direction=down]').click().prop('disabled', true);
            }
        }
    });
});

function handleMoreResults(response, props) {
    var bannerWidth = 970;
    var bannerheight = 90;
    if (props.container.width() < 970) {
        bannerWidth = 728;
    }
    if (props.container.width() < 790) {
        bannerWidth = 320;
        bannerheight = 50;
    }
    var bannerWrapperSize = props.container.width() > 768 ? 'height: 124px' : 'height: 84px';
    if(response.status === 'ok'){
        var randomNum = Math.floor(Math.random() * (1000000 - 100) + 100);
        var banner = $("<div class='box box-bg-light w-100 justify-content-center d-flex align-items-center' style='"
            + bannerWrapperSize + "'>"
            +"<iframe src=" + 'https://www.motke.co.il/devices/desktop/template'
            +"/lib/banners/bannersIframe.php?width=" + bannerWidth + '&height=' + bannerheight + '&adUnit=' + UnitName + "&unique=" + randomNum + " width=\"" + (bannerWidth + 5)
            +'" height="' + bannerheight + '" frameborder="0"></iframe><\/div><\/div>');
        /* Append result items */
        if (props.ajaxFunc === "display_blog_post_items" && props.ajaxStart === 0) {
            banner = '';
        }
        if(props.container.children('.item').length > 0){
            if(props.loadDirection === 'up'){
                /* UP */
                props.container.children().first().after(response.html).fadeIn(200);
                props.container.children('.item').first().before(banner);
                /* Scroll to start items */
                scrollToElement(props.container.children('.item').first(), 200);
            } else {
                if(props.container.children('.item').last().length > 0){
                    /* DOWN */
                    if (response.total_items - props.start > props.ajaxLimit) {
                        props.container.children().last().before(response.html).before(banner).fadeIn(200);
                    } else {
                        props.container.children().last().before(response.html).fadeIn(200);
                        props.container.children().last().remove()
                    }
                } else {
                    /* Prepend */
                    props.container.prepend(banner).append(response.html).fadeIn(200);
                }
            }
        } else {
            if (props.ajaxFunc === "display_blog_post_items" && props.ajaxStart !== 0) {
                props.container.append(banner).append(response.html).fadeIn(200);
            } else {
                props.container.prepend(banner).prepend(response.html).fadeIn(200);
            }

        }

        /* Apply Appear Effect to new items */
        $('[data-start-group='+props.ajaxStart+']').addClass('appear');
        

        if(props.loadDirection === 'down'){
            if (props.ajaxFunc === "display_blog_post_items" && props.ajaxStart !== 0) {
                props.container.append(props.loadMoreBtnElem.parent());
            }
        }
        /* Update next start */
        if(props.loadDirection === 'up'){
            /* Update next up start */
            props.firstStart = aboveZero(props.firstStart) + aboveZero(props.limit) - aboveZero(response.results_count);
            props.container.attr("data-ajax-first-start", props.firstStart);
            $('[data-container-unique="'+props.containerSelector+'"]').attr("data-ajax-first-start", props.firstStart);
        } else {
            /* Update next down start */
            props.start = aboveZero(props.start) + aboveZero(response.results_count);
            props.container.attr("data-ajax-start", props.start);
            $('[data-container-unique="'+props.containerSelector+'"]').attr("data-ajax-start", props.start);
        }

        /* Hide load up btn if reached to start */
        if(props.loadDirection === 'up' && aboveZero(props.ajaxStart) == 0){
            props.loadMoreBtnElem.fadeOut(300);
        }

        /* Hide load down btn if reached the end */
        if(props.loadDirection === 'down' && aboveZero(props.start) >= response.total_items){
            props.loadMoreBtnElem.fadeOut(300);
        }

        removeDuplicateItems();
    } else {
        props.loadMoreBtnElem.fadeOut(300);
    }

    setTimeout(function() {
        disableDelayBtn(props.loadMoreBtnElem);
    }, 500);
}

/* ########################### */
/* ## generalMessage PLUGIN ## */
/* ########################### */
/*
###########################
## generalMessage Plugin ##
###########################

How to use?
    1. Create empty element with uniuqe id attr and place it where you want the message to be displayed.
    2. Call the plugin from your element selector. 
        example: $('#dating_optin_banner').generalMessage('#dating_join_message', 1)
        First param: your html template selector
        Second param: the message id

How to maintain?
    To add new template:
    1. In display.php duplicate the <template> tag.
    2. Change the id attribute to unique one.
    3. Create new instance and parse the unique id to the first parameter.
*/
(function ( $ ) {
 
    $.fn.generalMessage = function( templateId, messageId, btnCallback ) {
        this.templateId = templateId;
        this.templateContainer = $(this.templateId);
        this.messageId = messageId;
        this.btnCallback = btnCallback;
        this.init = function() {
            this.fetchData( this.messageId );
        };

        /* Clone template and append to selected element */
        this.clone = function() {
            $(this).html($(this.templateContainer).html());
        };

        /* Put content in cloned html */
        this.putContent = function(jsonContent ) {
            $(this).find('.message-img').attr( 'src', jsonContent.img );
            $(this).find('.message-title').html( jsonContent.title );
            $(this).find('.message-sub-title').html( jsonContent.sub_title );
            $(this).find('.message-content').html( jsonContent.content );
            $(this).find('.message-link').html( jsonContent.link_label );
            if(jsonContent.link.length > 0){
                $(this).find('.message-link').attr( 'href', jsonContent.link );
            } else if(typeof this.btnCallback === 'function'){
                $(this).find('.message-link').on('click', $.proxy(function(e){
                    e.preventDefault();
                    this.btnCallback()
                }, this));
            } else {
                $(this).find('.message-link').hide();
            }
        };

        /* Get content from DB */
        this.fetchData = function( messageId ){
            $.ajax({
                type: 'POST',
                url: 'https://www.motke.co.il/ajaxModule',
                data: {
                    func: 'get_message',
                    module: 'general_message',
                    messageId: messageId
                },
                success: $.proxy(function(data){
                    var output = JSON.parse(data);
                    if(output.status === 'success'){
                        this.clone();
                        this.putContent( output.data );
                    }
                }, this)
            });
        };

        this.init();
    };
 
}( jQuery ));

	########################

function DatingFilter(filterContainer, filterData){
	this.filterData = filterData;
	this.filterInputs = {
		'ranges': {
			'selector': ''
		}
	};

	/* Default limit amount. this will change on init() */
	this.limitAmount = 9;

	this.onload = function(){
		this.init();
	};

	this.defaultFilter = {
		age: {
			min: 50,
			max: 120
		},
		height: {
			min: 120,
			max: 210
		}
	};

	this.filterIsEmpty = function(filter) {
		return JSON.stringify(filter) === JSON.stringify(this.defaultFilter) || $.isEmptyObject(filter)
	};

	this.init = function(){
		this.limitAmount = parseInt($('#dating_items_container').attr('data-limit-amount') ,10);

		/* If url has filter data, fetch it */
		if($.isEmptyObject(this.filterData)){
			this.fetchFilterFromURI();
			if (this.filterIsEmpty(this.filterData)){ 
				this.fetchFilterFromCookie();
			}
		
			/* Add default values if not set */
			if(typeof this.filterData.age === 'undefined'){
				this.change('age', {
					'min': 50,
					'max': 120
				});
			}
			if(typeof this.filterData.height === 'undefined'){
				this.change('height', {
					'min': 120,
					'max': 210
				});
			}
		}

		/* When checkbox clicked, add choice to filterData object */
		$(filterContainer).on('click', '.dating-filter-checkbox', $.proxy(function(event){
			var elemClicked = event.target;
			var filterName = $(elemClicked).attr('name');
			var filterVal = $(elemClicked).attr('value');

			if($(elemClicked).is(':checked')){
				this.add(filterName, filterVal);
			} else {
				this.remove(filterName, filterVal);
			}

			/* Refresh feed with new filter settings */
			this.storeOnURI(this.filterData);
			this.storeOnCookie(this.filterData);
			
			this.refreshFeed();
		}, this));
		
		this.updateRange = function (name, currentMin, currentMax) {
			this.change(name, { min: currentMin, max: currentMax});
			this.storeOnURI(this.filterData);
			this.storeOnCookie(this.filterData);
			this.refreshFeed();
		};

		/* Range inputs */
		this.ageRangeMin = $('#dating-filter-range-min');
		this.ageRangeMax = $('#dating-filter-range-max');
		this.ageRangeMin.val(this.filterData.age.min);
		this.ageRangeMax.val(this.filterData.age.max);
		/** update on range change */
		this.ageSpecific = $.proxy(function (e) {
			var currentElem = e.target;
			var currentMin = parseInt(this.ageRangeMin.val() ,10);
			var currentMax = parseInt(this.ageRangeMax.val() ,10);
			if (currentElem.id === "dating-filter-range-min" && e.target.value <= 50) {
				currentMin = 50;
				e.target.value = 50;
			}
			else if (e.target.id === "dating-filter-range-min" && e.target.value >= 120) {
				currentMin = 119;
				e.target.value = 119;
			}
			else if (e.target.id === "dating-filter-range-max" && e.target.value  >= 120) {
				currentMax = 120;
				e.target.value = 120;
			}
			if ((currentMax - currentMin) <= 1) {
				currentMax = currentMin + 1;
				this.ageRangeMax.val(currentMax);
			}
			this.updateRange('age', currentMin, currentMax);
		}, this);

		/** update on range input */
		this.keyUpAge = $.proxy(function (e) {
			/** trigger only on number keys */
			if (e.keyCode != 8 && isNaN(e.key)) {
				return;
			}
			var currentMin = parseInt(this.ageRangeMin.val() ,10);
			var currentMax = parseInt(this.ageRangeMax.val() ,10);
			if (currentMin >= 50 && currentMin < 119 && currentMax < 120 && currentMax > currentMin) {
				this.updateRange('age', currentMin, currentMax);
				if ((currentMax - currentMin) <= 1) {
					currentMax = currentMin + 1;
					this.ageRangeMax.val(currentMax);
				}
			}
		}, this);

		this.ageRangeMin.change(this.ageSpecific);
		this.ageRangeMax.change(this.ageSpecific);
		this.ageRangeMin.on("keyup", this.keyUpAge);
		this.ageRangeMax.on("keyup", this.keyUpAge);

		if ($('#height-input-min').length > 0) {
			this.heightRangeMin = $('#height-input-min');
			this.heightRangeMax = $('#height-input-max');
			this.heightRangeMin.val(this.filterData.height.min === 0 ? 120 : this.filterData.height.min); /* probably dont need the condition here */
			this.heightRangeMax.val(this.filterData.height.max);
			
			/** update on range change */
			this.heightSpecific = $.proxy(function (e) {
				var currentMin = parseInt(this.heightRangeMin.val() ,10) === 120 ? 0 : parseInt(this.heightRangeMin.val() ,10);
				var currentMax = parseInt(this.heightRangeMax.val() ,10);
				if (e.target.id === "height-input-min" && e.target.value <= 120) {
					currentMin = 120;
					e.target.value = 120;
				}
				else if (e.target.id === "height-input-min" && e.target.value >= 210) {
					currentMin = 209;
					e.target.value = 209;
				}
				else if (e.target.id === "height-input-max" && e.target.value > 210) {
					currentMax = 210;
					e.target.value = 210;
				}
				if ((currentMax - currentMin) <= 1) {
					currentMax = currentMin + 1;
					this.heightRangeMax.val(currentMax);
				}
				this.updateRange('height', currentMin, currentMax);
			}, this);
			
			/** update on range input */
			this.keyUpHeight = $.proxy(function (e) {
				/** trigger only on number keys */
				if (e.which != 8 && isNaN(e.key)) {
					return;
				}
				var currentMin = parseInt(this.heightRangeMin.val() ,10);
				var currentMax = parseInt(this.heightRangeMax.val() ,10);
				if (currentMin >= 120 && currentMin < 210 && currentMax <= 210 && currentMax > currentMin) {
					if (e.target.id === "height-input-min" && parseInt(e.target.value ,10) === 120) {
						currentMin = 0;
					}
					if ((currentMax - currentMin) <= 1) {
						currentMax = currentMin + 1;
						this.heightRangeMax.val(currentMax);
					}
					this.updateRange('height', currentMin, currentMax);
					if (currentMin === 0) {e.target.value = 120};
				}
			}, this);
	
			this.heightRangeMin.change(this.heightSpecific);	
			this.heightRangeMax.change(this.heightSpecific);
			this.heightRangeMin.on("keyup", this.keyUpHeight);
			this.heightRangeMax.on("keyup", this.keyUpHeight);
		}
		this.updateInputs();
	};
	/* Adding new parameter to filter */
	this.add = function(filterName, filterVal){
		if( typeof this.filterData[filterName] === "undefined" ){
			this.filterData[filterName] = [];
		}

		if( this.filterData[filterName].indexOf(filterVal) === -1 ){
			this.filterData[filterName].push(filterVal);
		}
	};

	/* Removing new parameter from filter */
	this.remove = function(filterName, filterVal){
		if( typeof this.filterData[filterName] === "undefined" ){
			this.filterData[filterName] = [];
		}

		if(this.filterData[filterName].indexOf(filterVal) !== -1){
			var filterValIndex = this.filterData[filterName].indexOf(filterVal);
			this.filterData[filterName].splice(filterValIndex, 1);
		}
	};

	this.getFilter = function(){
		return this.filterData;
	};

	/* Change value */
	this.change = function(filterName, filterVal){
		this.filterData[filterName] = filterVal;
	};

	/* Tune filter inputs with current filter object parameters */
	this.updateInputs = function(){
		this.filterInputs.checkboxs = $(filterContainer).find('.dating-filter-checkbox');
		this.filterInputs.ranges.selector = $(filterContainer).find('.dating-filter-range-container');

		/* For each checkbox, check if exist in filterData object */
		$.each(this.filterInputs.checkboxs, $.proxy(function(checkboxIndex, checkboxId){
			var checkboxName = $(checkboxId).attr('name');
			var checkboxValue = $(checkboxId).attr('value');
			var isFilterDataExist = 0;

			/* Is data exist for this input */
			if(typeof this.filterData[checkboxName] === 'undefined'){
				return;
			}

			/* Is checkbox should be checked */
			if( this.filterData[checkboxName].indexOf(checkboxValue) !== -1 ){
				isFilterDataExist = 1;
			}

			/* Check or Uncheck */
			if(isFilterDataExist){
				$(checkboxId).prop('checked', true);
			} else {
				$(checkboxId).prop('checked', false);
			}
		}, this));


		
		/* For each range slider, check if exist in filterData object */
 
		$.each(this.filterInputs.ranges.selector, $.proxy(function(rangeSliderIndex, rangeSliderElem){
			if (rangeSliderElem.id.includes('min')) {
				rangeSliderElem.val(filterData.age.min);
			};
			if (rangeSliderElem.id.includes('max')) {
				rangeSliderElem.val(filterData.age.max);
			};
		}, this));

		this.refreshFeed();
	};

	this.encode = function(filter){
	    var filterString = JSON.stringify(filter);
	    return filterString;
	};

	this.decode = function(filterString){
	    var filter = JSON.parse(filterString);
	    return filter;
	};

	this.storeOnURI = function(filter){
	    var filterBase64 = this.encode(filter);
	    var newurl = urlParams.changeParam('dating_filter', filterBase64);
	    return;
	};

	this.fetchFilterFromURI = function(){
	    var filterString = getParameterByName('dating_filter');
	    if(filterString){
	        this.filterData = this.decode(filterString);
	    }
	    return this.filterData;
	};

	this.storeOnCookie = function(filter){
	    var filterBase64 = this.encode(filter);
	    var date = new Date();
	    createCookie('searchMeetingFilter', filterBase64,730);
	  
	};

	this.fetchFilterFromCookie = function(){
	    var filterString = readCookie('searchMeetingFilter');
	    if(filterString){
	        this.filterData = this.decode(filterString);
	    }
	    return this.filterData;
	};

	this.refreshFeed = function(){
		if(typeof getDatingMembers === 'function'){
            var limitStart = $(filterContainer).children('.item').length;

            /* If previous state exists */
            if(typeof websiteStates === 'object'){
                websiteStates.setCurrentCategoryName('cat_0');
                /* If category state exist */
                if(Object.keys(websiteStates.getCurrentCategory()).length){
                    this.limitAmount = websiteStates.getCurrentCategory().limit;
                    limitStart = websiteStates.getCurrentCategory().start;
                    itemNum = websiteStates.getCurrentCategory().item;
                    /* Update container start date */
                    $('#dating_items_container').attr('data-ajax-start', limitStart);
                    /* Update container first start */
                    $('#dating_items_container').attr('data-ajax-first-start', limitStart);
                }
            }

			getDatingMembers(this.filterData, limitStart, this.limitAmount, true);
			this.showHideCleanBtn();
		}
	};

	this.cleanAll = function(refresh){
		closeGroupItems('dating');
		this.filterInputs.checkboxs = $(filterContainer).find('.dating-filter-checkbox');
		this.filterInputs.ranges.selector = $(filterContainer).find('.dating-filter-range-container input');

		$.each(this.filterInputs.checkboxs, $.proxy(function(checkboxIndex, checkboxId){
			$(checkboxId).prop('checked', false);
		}, this));


 		$.each(this.filterInputs.ranges.selector, $.proxy(function(rangeSliderIndex, rangeSliderElem){
			if (rangeSliderElem.id.includes('min')) {
				this.filterInputs.ranges.selector[rangeSliderIndex].value = rangeSliderElem.min;
			};
			if (rangeSliderElem.id.includes('max')) {
				this.filterInputs.ranges.selector[rangeSliderIndex].value = rangeSliderElem.max;
			};
		}, this));

		this.clearFilter();
		this.storeOnCookie(this.filterData);
		this.storeOnURI(this.filterData);


		if(refresh){
			this.refreshFeed();
		}
	};

	this.clearFilter = function(defaultFilter){
		this.filterData =  Object.assign({}, this.defaultFilter);
		var newurl = urlParams.changeParam('dating_filter', JSON.stringify(this.filterData));
	};

	this.showHideCleanBtn = function(){

		}
	};

	this.disableFilter = function(){
		$('.dating-filter-control').addClass('disabled');
	};

	this.enableFilter = function(){

	};

	this.onload();
}

var auth_attempt = 0;
var auth_attempt_allowed = "99999";

function tryLoginAjax(){
    if(auth_attempt >= auth_attempt_allowed){
        callCustomAuthMessage();
        return;
    }
    $.ajax({
        type: "POST",
        url: "https://www.motke.co.il/ajaxModule",
        data: $('#loginAjaxForm').serialize() + "&module=login_form&func=try_login",
        success: function(data){
        	var output = JSON.parse(data);
            var tracker =  { send: function () { } };
            if ("ga" in window && ga.getAll) {/*in case GA is defined within tag manager - this is the way to use it!*/
                tracker = ga.getAll()[0];
            }
            if(output.status === 'ok'){
            	if(output.response === 'success'){
	                window.location = "https://www.motke.co.il/feed"
            	} else if(output.response === 'loggedIn') {
					clearValidationErrors();
					location.reload();
            	}
                auth_attempt = 0;
            } else {
                switch (output.status) {
                    case 'error|freeze':
                        invalidInput($("#loginEmail"), `החשבון שלך הוקפא. יש לפנות למערכת מוטק'ה לסיוע במייל info@motke.co.il`);
                        break;
                    default:
                        invalidInput($('#loginEmail'), "Password or email is incorrect. Please try another.");
                        auth_attempt++;
                        break;
                }
                if(auth_attempt >= auth_attempt_allowed){
                    callCustomAuthMessage();
                }
            }
        }
    });
    return false;
}

function callCustomAuthMessage(){
    $('#login_form').hide();
    $('#auth_error_msg').fadeIn();
}

$( function() {
    $.widget( "custom.combobox", {
      _create: function() {
        this.wrapper = $( "<span>" )
          .addClass( "custom-combobox" )
          .insertAfter( this.element );
        this.element.hide();
        this._createAutocomplete();
        this._createShowAllButton();
      },
 
      _createAutocomplete: function() {
        var selected = this.element.children( ":selected" ),
          value = selected.val() ? selected.text() : "";
 
        this.input = $( "<input>" )
          .appendTo( this.wrapper )
          .val( value )
          .attr( "title", "" )
          .addClass( "input-text custom-combobox-input ui-widget ui-widget-content ui-state-default ui-corner-left" )
          .autocomplete({
            delay: 0,
            minLength: 0,
            source: $.proxy( this, "_source" )
          })
          .tooltip({
            classes: {
              "ui-tooltip": "ui-state-highlight"
            }
          });
 
        this._on( this.input, {
          autocompleteselect: function( event, ui ) {
            ui.item.option.selected = true;
            this._trigger( "select", event, {
              item: ui.item.option
            });
          },
 
          autocompletechange: "_removeIfInvalid"
        });
      },
 
      _createShowAllButton: function() {
        var input = this.input,
          wasOpen = false;
 
        $( "<a>" )
          .attr( "tabIndex", -1 )
          .attr( "title", "Show All city" )
          .tooltip()
          .appendTo( this.wrapper )
          .removeClass( "ui-corner-all" )
          .addClass( "autocomplete-drop-btn icon-arrow-down" )
          .on( "mousedown", function() {
            wasOpen = input.autocomplete( "widget" ).is( ":visible" );
          })
          .on( "click", function() {
            input.trigger( "focus" );
 
            /* Close if already visible */
            if ( wasOpen ) {
              return;
            }
 
            /* Pass empty string as value to search for, displaying all results */
            input.autocomplete( "search", "" );
          });
      },
 
      _source: function( request, response ) {
        var matcher = new RegExp( $.ui.autocomplete.escapeRegex(request.term), "i" );
        response( this.element.children( "option" ).map(function() {
          var text = $( this ).text();
          if ( this.value && ( !request.term || matcher.test(text) ) )
            return {
              label: text,
              value: text,
              option: this
            };
        }) );
      },
 
      _removeIfInvalid: function( event, ui ) {
 
        /* Selected an item, nothing to do */
        if ( ui.item ) {
          return;
        }
 
        /* Search for a match (case-insensitive) */
        var value = this.input.val(),
          valueLowerCase = value.toLowerCase(),
          valid = false;
        this.element.children( "option" ).each(function() {
          if ( $( this ).text().toLowerCase() === valueLowerCase ) {
            this.selected = valid = true;
            return false;
          }
        });
 
        /* Found a match, nothing to do */
        if ( valid ) {
          return;
        }
 
        /* Remove invalid value */
        this.input
          .val( "" )
          .attr( "title", value + " didn't match any item" )
          .tooltip( "open" );
        this.element.val( "" );
        this._delay(function() {
          this.input.tooltip( "close" ).attr( "title", "" );
        }, 2500 );
        this.input.autocomplete( "instance" ).term = "";
      },
 
      _destroy: function() {
        this.wrapper.remove();
        this.element.show();
      }
    });
 
    $( "#combobox" ).combobox();
    $( "#toggle" ).on( "click", function() {
      $( "#combobox" ).toggle();
    });
  } );

var auth_attempt_allowed = "99999";

function trySubscribeAjax(event){
    event.stopPropagation();
    event.preventDefault();
    if(auth_attempt >= auth_attempt_allowed){
      callErrorAuthMessage();
        return;
    }

    /*client side validation*/
    var tracker =  { send: function () { } };
    if ("ga" in window && ga.getAll) {/*in case GA is defined within tag manager - this is the way to use it!*/
        tracker = ga.getAll()[0];
    }
    var validName = validateUsername($('#new_user_name'));
    if(!validName){
        if (typeof tracker != "undefined")
            tracker.send('event', 'signup', 'error', "Invalid-username ");
    }
    var validTermsCheck = validateCheckbox($('#termsCheckbox'), "חובה להסכים לתנאים.");
    if(!validTermsCheck){
        if (typeof tracker != "undefined")
             tracker.send( 'event', 'signup – step-3', 'error', "Not-Checked-Terms ");
    }
    var validPassword = validatePassword($('#signupPassword'));
    if(!validPassword){
        if (typeof tracker != "undefined")
             tracker.send( 'event', 'signup', 'error', "Invalid-password ");
    }
    var validRadio = validateRadio('genderRadios', "חובה לבחור זכר או נקבה.");
    if(!validRadio){
        if (typeof tracker != "undefined")
             tracker.send( 'event', 'signup', 'error', "Choose-Gender ");
    }
    var validBirthDay = validateSelect('#Birth_Date_dd');
    if(!validBirthDay){
        if (typeof tracker != "undefined")
             tracker.send( 'event', 'signup', 'error', "No-Birthday-day ");
    }
    var validBirthMonth = validateSelect('#Birth_Date_mm');
    if(!validBirthMonth){
        if (typeof tracker != "undefined")
             tracker.send( 'event', 'signup', 'error', "No-Birthday-month ");
    }
    var validBirthYear = validateSelect('#Birth_Date_yyyy');
    if(!validBirthYear){
        if (typeof tracker != "undefined")
             tracker.send( 'event', 'signup', 'error', "No-Birthday-year ");
    }
    var validCity = validateSelect('#wm_city');
    if(!validCity){
        if (typeof tracker != "undefined")
             tracker.send( 'event', 'signup', 'error', "No-City ");
    }
    var validEmail = validateEmail('#signupEmail');
    if(!validEmail){
        if (typeof tracker != "undefined")
             tracker.send( 'event', 'signup', 'error', "Invalid-Email ");
    }
    var validPassValidation = validateIdenticalInputs('#signupPassword', '#signupPasswordValidate');
    if(!validPassValidation){
        if (typeof tracker != "undefined")
             tracker.send( 'event', 'signup', 'error', "Invalid-Password-Validation ");
    }
    var validTerms = validateCheckbox('#termsCheckbox');
    if(!validTerms){
        if (typeof tracker != "undefined")
             tracker.send( 'event', 'signup', 'error', "Terms-Not-Argreed ");
    }

    var validCapcha = validateCapcha("#subscribeCaptcha", subscribeCaptcha);
    if(!validCapcha){
        if (typeof tracker != "undefined")
             tracker.send( 'event', 'signup', 'error', "Captcha-Error ");
    }

    if(
        validTermsCheck 
        && validName 
        && validRadio
        && validBirthDay 
        && validBirthMonth 
        && validBirthYear 
        && validCity
        && validEmail
        && validPassword 
        && validPassValidation
        && validCapcha
        && validTerms
    ){
        delayBtn($('#submitRegistrationForm'), 3600);

        var formData = new FormData();
        if(typeof signupFiles === "array"){
            $.each(signupFiles, function(key, value){
                formData.append(key, value);
            });
        }

        formData.append('module', 'signup_form');
        formData.append('func', 'try_signup');
        formData.append('g-recaptcha-response', grecaptcha.getResponse(subscribeCaptcha));
        formData.append('email', $('#registrationAjaxForm').find('[name=email]').val());
        formData.append('new_user_name', $('#registrationAjaxForm').find('[name=new_user_name]').val());
        formData.append('password', $('#signupPassword').val());
        formData.append('password_validation', $('#signupPasswordValidate').val());

        formData.append('Birth_Date_dd', $('#Birth_Date_dd').val());
        formData.append('Birth_Date_mm', $('#Birth_Date_mm').val());
        formData.append('Birth_Date_yyyy', $('#Birth_Date_yyyy').val());

        formData.append('gender', $('input[name=gender]:checked').val());
        formData.append('wm_city', $("#wm_city option:selected").val());

        formData.append('showMyAge', $('#showMyAge').is(':checked'));
        formData.append('showMyBitrthday', $('#showMyAge').is(':checked'));

        if($('#termsCheckbox').is(':checked')){
          formData.append('termsCheckbox', $('#termsCheckbox').is(':checked'));
        }
        if($('#offersCheckbox').is(':checked')){
          formData.append('offersCheckbox', $('#offersCheckbox').is(':checked'));
        }
        if($('#updatesCheckbox').is(':checked')){
          formData.append('updatesCheckbox', $('#offersCheckbox').is(':checked'));
        }

        formData.append('marital_status', $('input[name=marital_status]:checked').val());
        formData.append('mobile_phone_prefix', $('#mobile_phone_prefix').val());
        formData.append('mobile_phone', $('#mobile_phone').val());

        /*server side validation*/
        $.ajax({
          type: "POST",
          processData: false,
          contentType: false,
          cache: false,
          dataType: 'json',
          url: "https://www.motke.co.il/ajaxModule",
          data: formData,
          success: function(data){
            if(typeof data != "object"){
                var jsonOutput = JSON.parse(data);
            } else {
                var jsonOutput = data;
            }
            var tracker =  { send: function () { } };
            if ("ga" in window && ga.getAll) {/*in case GA is defined within tag manager - this is the way to use it!*/
                tracker = ga.getAll()[0];
            }
            if(Number(jsonOutput.success) > 0){
              if (typeof tracker != "undefined"){
                  tracker.send('event', 'signup', 'successful signup', "successful signup");
              }
              setTimeout(function(){ 
                  window.location = jsonOutput.after_signup_link;
              }, 500);
            } else {
              switch (jsonOutput.error) {/*custom GA event errors*/
                case 'aliasExists':
                    invalidInput($('#new_user_name'), "השם כבר קיים. נסה שם אחר, לדוגמה: "+jsonOutput.suggestion);
                    if (typeof tracker != "undefined")
                        tracker.send( 'event', 'signup – step 1', 'error', "השם כבר קיים. נסה שם אחר, לדוגמה: ");
                break;
                case 'emailExists':
                    invalidInput($('#signupEmail'), "כתובת המייל כבר קיימת במערכת");
                    if (typeof tracker != "undefined")
                        tracker.send( 'event', 'signup–step-1', 'error', "Email-is-already-exist. ");
                break;
                case 'emailInvalid':
                    invalidInput($('#signupEmail'), "יש להזין מייל תקין.");
                    if (typeof tracker != "undefined")
                        tracker.send( 'event', 'signup–step-1', 'error', "Email-is-invalid. ");
                break;
                case 'nicknameInvalid':
                    invalidInput($('#new_user_name'), "הכינוי אינו תקין. (רק אותיות ומספרים ומקסימום 20 תווים)");
                    if (typeof tracker != "undefined")
                        tracker.send( 'event', 'signup–step-1', 'error', "Nickname-is-invalid. ");
                break;
                case 'passInvalid':
                    invalidInput($('#signupPassword'), "יש להזין סיסמא בין 6 ל-30 תווים.");
                    if (typeof tracker != "undefined")
                        tracker.send( 'event', 'signup–step-1', 'error', "Password-is-invalid. ");
                break;
                case 'passNotMatch':
                    invalidInput($('#signupPasswordValidate'), "הסיסמא לא תואמת");
                    if (typeof tracker != "undefined")
                        tracker.send( 'event', 'signup–step-1', 'error', "Password-is-invalid. ");
                break;
                case 'birthdateInvalid':
                    invalidInput($('#Birth_Date_yyyy'), "יש לבחור תאריך לידה.");
                    if (typeof tracker != "undefined")
                        tracker.send( 'event', 'signup–step-1', 'error', "Birthdate-is-invalid. ");
                break;
                case 'emailConfirmationInvalid':
                  invalidInput($('#signupEmailConfirm'), "כתובת האימות אינה תואמת");
                  if (typeof tracker != "undefined")
                    tracker.send( 'event', 'signup–step-1', 'error', "email-not-identical ");
                break;
                case 'cityInvalid':
                  invalidInput($('#wm_city'), "בחר עיר");
                  if (typeof tracker != "undefined")
                    tracker.send( 'event', 'signup–step-1', 'error', "email-not-identical ");
                break;
                case 'phoneInvalid':
                  invalidInput($('#mobile_phone'), "נא להזין מספר טלפון תקין");
                  if (typeof tracker != "undefined")
                    tracker.send( 'event', 'signup–step-1', 'error', "נא להזין מספר טלפון תקין ");
                break;
                case 'phoneExists':
                  invalidInput($('#mobile_phone'), "מספר הטלפון כבר קיים במערכת");
                  if (typeof tracker != "undefined")
                    tracker.send( 'event', 'signup–step-1', 'error', "מספר הטלפון כבר קיים במערכת ");
                break;
                case 'captchaErr':
                  $('.captchaWrap > .alert').removeClass('hidden');
                  grecaptcha.reset(subscribeCaptcha);
                  if (typeof tracker != "undefined")
                    tracker.send( 'event', 'signup–step-2', 'error', "invalid-captcha ");
                break;
                default:
                  auth_attempt++;
                break;
              }
              if(auth_attempt >= auth_attempt_allowed){
                  callErrorAuthMessage();
                  return;
              }
            }
          }
        });
    } else {
      auth_attempt++;
    }
    setTimeout(function() {
      grecaptcha.reset(subscribeCaptcha);
    }, 100);
  return false;
}

function callErrorAuthMessage(){
  $('#registrationAjaxForm').fadeOut();
  $('#auth_error_msg').fadeIn();
  $("#wm_city").show();
  $(".popover").remove();
  $("#termsCheckbox, #signupPassword, #genderRadios, #Birth_Date_dd, #Birth_Date_mm, #Birth_Date_yyyy, #wm_city, #signupEmail, #signupPassword, #signupPasswordValidate, #termsCheckbox, #subscribeCaptcha, .custom-combobox").each(function(){
    if($(this).attr('id') != 'undefined'){
      clearValidationErrors($(this));
    }
  });
}

function processEmailShareForm_email_share(){
    var to = $('[name=to_email_share]').val();
    var from = $('[name=from_email_share]').val();
    var subject = $('[name=subject_email_share]').val();
    var content = $('[name=content_email_share]').val();
    var url = $('[name=url_email_share]').val();
    
    delayBtn($('.sendEmailShareDialogBtn'), 10000);

    $.ajax({
        type: 'POST',
        url: 'https://www.motke.co.il/ajaxModule',
        data: {
            module:'email_share',
            func:'send_email',
            to: to,
            from: from,
            subject: subject,
            content: content,
            url: url,
            recaptcha: grecaptcha.getResponse(shareRecaptcha)
        },
        success: function(data){
            var output = JSON.parse(data);
            if(output.success === 1){
                showAnswerText_email_share();
            }

            if(output.success === 0 && output.error === "recaptcha_error"){
                $('#recaptchaErrMsg').removeClass('hidden');
                disableDelayBtn($('.sendEmailShareDialogBtn'));
            }
        }
    });

    return false;
}
