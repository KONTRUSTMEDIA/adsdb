void addf(char(*name)[100], char(*author)[100], char(*publisher)[100],int book_count) {
	printf(&quot;책의 제목, 저자, 출판사 입력: &quot;);
	scanf(&quot;%s&quot;, name[book_count]);
	scanf(&quot;%s&quot;, author[book_count]);
	scanf(&quot;%s&quot;, publisher[book_count]);
}

int search(char(*name)[100], char(*author)[100], char(*publisher)[100], int book_count, int i) {
	int found = 0;
	switch (i) {
	case 1:
		char title_search[100];
		printf(&quot;제목 입력: &quot;);
		scanf(&quot;%s&quot;, title_search);

		for (int n = 0; n &lt; book_count; n++) {
			if (compare(name[n],title_search)) {
				printf(&quot;저자: %s, 출판사:%s\n&quot;, author[n], publisher[n]);
				found += 1;
			}
		}
		
		break;
	case 2:
		char author_search[100];
		printf(&quot;저자 입력: &quot;);
		scanf(&quot;%s&quot;, author_search);

		for (int n = 0; n &lt; book_count; n++) {
			if (compare(author[n], author_search)) {
				printf(&quot;제목: %s, 출판사:%s\n&quot;, name[n], publisher[n]);
				found += 1;
			}
		}
		break;
	case 3:
		char publisher_search[100];
		printf(&quot;출판사 입력: &quot;);
		scanf(&quot;%s&quot;, publisher_search);

		for (int n = 0; n &lt; book_count; n++) {
			if (compare(publisher[n], publisher_search)) {
				printf(&quot;제목: %s, 저자:%s\n&quot;, name[n], author[n]);
				found += 1;
			}
		}
	
		break;
	default:
		break;

	}
	if (!found) {
		printf(&quot;검색결과없음\n&quot;);
	}
	return 0;
}
int compare(char* a, char* b)
{
	int i = 0;

	while (a[i] != &#x27;\0&#x27; &amp;&amp; b[i] != &#x27;\0&#x27;) {
		if (a[i] != b[i])
			return 0;
		i++;
	}

	if (a[i] == &#x27;\0&#x27; &amp;&amp; b[i] == &#x27;\0&#x27;)
		return 1;

	return 0;
}
void borrow(char(*name)[100], int* borrowed, int book_count) {
	int found = 0;
	char borrow_search[100];
	printf(&quot;제목 입력: &quot;);

                      안녕하세요, 강의에서 사용한 코드 중 하나에 질문이 있습니다.

강의에서 지역변수의 reference를 리턴해서 문제가 생기는 경우인
int &amp;function() {
    int a = 2;
    return a;
}

int main() {
    int b = function();
    b = 3;

    return 0;
}
를 디스어셈블해보면, 다음과 같은 것을 확인했습니다. (-O0로 컴파일 했습니다)
function():
	; ...
	mov    DWORD PTR [rbp-0xc],0x2  
	mov    eax,0x0                  
        ; ...
	leave                           
	ret    
이것만 보면, function()이 애당초 a의 주소를 반환하는 대신 0을 반환하는 것으로 보이고, 실제로 main()에서 다음과 같이 0을 dereference하며
	mov    eax,DWORD PTR [rax]      
	mov    DWORD PTR [rbp-0x4],eax 
segmentation fault가 발생하는 것으로 보입니다. 혹시 컴파일러가 dangling reference를 반환하는 것을 감지해서 그냥 널을 리턴하는 것으로 코드를 변경한 것일까요?? g++로 컴퍼일한 결과인데 다른 컴파일러는 실제로 어떻게 동작하는지 궁금하네요. 만약 a의 주소를 리턴한다면 (물론 dangling reference기는 하나) 그 주소는 유효한 stack 주소일거고 segmentation fault가 날 이유가 없어 보여서요.
