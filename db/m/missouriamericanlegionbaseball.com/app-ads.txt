    // When the subnav exists, but the selected page has no children,
    // we try to display the siblings as subnav UNLESS, the parent node
    // is aso the root node. This means it woulld duplicate the main
    // nav, so we display nothing.
    var hasChildren = !!pageNav.children.length;
    var rootLevelPage = rootNav.parent.id === pageNav.parent.id
    // SUB NAV
    if (hasSubnav && (hasChildren || !rootLevelPage)) {
      NS.create('DropdownNav', {
        currentId: currentId,
        activeClass: 'has-sub-nav',
        minWidth: 1025,
        selector: '#topNav .theme-sub-nav',
        showTitle: true,
        showChildren: true,
        maxDepth: 0,
        data: pageNav,
        paywallImage: '/app_images/paywall-icon.svg?16579122511'
      });
    }
    else {
      $('#topNav').addClass('no-sub-nav');
    }
  })(jQuery);
  // legacy event... not sure it's needed
  $j(function(){
    $j('#topNav').trigger('topnav.loaded');
  });
