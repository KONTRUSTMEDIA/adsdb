  let YTReadyResolve;
  const YTReady = new Promise(resolve => { YTReadyResolve = resolve; });

  function onYouTubeIframeAPIReady() {
    YTReadyResolve(true);
  }
  window.onYouTubeIframeAPIReady = onYouTubeIframeAPIReady;

  (function loadYT(){
    if (window.YT && window.YT.Player) {
      YTReadyResolve(true);
      return;
    }
    const s = document.createElement('script');
    s.src = "https://www.youtube.com/iframe_api";
    document.head.appendChild(s);
  })();

  function fetchTitle(videoId, targetEl){
    fetch('https://noembed.com/embed?url=https://www.youtube.com/watch?v=' + videoId)
      .then(r => r.json())
      .then(d => { if (d && d.title) targetEl.textContent = d.title; else targetEl.textContent = 'Título no disponible'; })
      .catch(()=> targetEl.textContent = 'Título no disponible');
  }

  document.addEventListener('DOMContentLoaded', () => {
    const items = document.querySelectorAll('.video-item');
    const players = {};
    items.forEach((item, idx) => {
      const thumb = item.querySelector('.video-thumb');
      const titleBox = item.querySelector('.video-title');
      const videoId = thumb.dataset.id;
      const playlist = thumb.dataset.playlist || '';
      const img = document.createElement('img');
      img.src = 'https://img.youtube.com/vi/' + videoId + '/hqdefault.jpg';
      img.alt = 'Miniatura';
      img.loading = 'lazy';
      thumb.appendChild(img);
      fetchTitle(videoId, titleBox);
      thumb.addEventListener('click', async function handler(){
        thumb.classList.add('playing');
        if (players[idx]) {
          players[idx].playVideo();
          return;
        }
        const frameDiv = document.createElement('div');
        const playerDOMId = 'yt-player-' + idx + '-' + Math.random().toString(36).slice(2);
        frameDiv.id = playerDOMId;
        frameDiv.className = 'video-frame';
        thumb.innerHTML = '';
        thumb.appendChild(frameDiv);
        await YTReady;
        players[idx] = new YT.Player(playerDOMId, {
          videoId: videoId,
          playerVars: (playlist ? { autoplay: 1, listType: 'playlist', list: playlist, loop: 1 } : { autoplay: 1 }),
          events: {
            onReady: function(e){
              try { e.target.playVideo(); } catch(e){}
              const info = e.target.getVideoData();
              if (info && info.title) titleBox.textContent = info.title;
            },
            onStateChange: function(e){
              if (e.data === YT.PlayerState.PLAYING) {
                try {
                  const info = e.target.getVideoData();
                  if (info && info.title) titleBox.textContent = info.title;
                } catch(err){}
              }
            }
          }
        });
      });
    });
  });
