<script>
    // Cookie Consent Manager
    const cookieConsent = {
        // Cookie management
        setCookie: function(name, value, days) {
            const d = new Date();
            d.setTime(d.getTime() + (days * 24 * 60 * 60 * 1000));
            const expires = "expires=" + d.toUTCString();
            document.cookie = name + "=" + value + ";" + expires + ";path=/;SameSite=Lax";
        },

        getCookie: function(name) {
            const nameEQ = name + "=";
            const ca = document.cookie.split(';');
            for (let i = 0; i < ca.length; i++) {
                let c = ca[i];
                while (c.charAt(0) == ' ') c = c.substring(1, c.length);
                if (c.indexOf(nameEQ) == 0) return c.substring(nameEQ.length, c.length);
            }
            return null;
        },

        deleteCookie: function(name) {
            document.cookie = name + "=; expires=Thu, 01 Jan 1970 00:00:00 UTC; path=/;";
        },

        // Preferences management
        preferences: {
            necessary: true,
            analytics: false,
            targeting: false
        },

        loadPreferences: function() {
            const saved = this.getCookie('cookie_consent_prefs');
            if (saved) {
                try {
                    this.preferences = JSON.parse(saved);
                    return true;
                } catch (e) {
                    console.error('Error loading preferences:', e);
                }
            }
            return false;
        },

        savePreferences: function() {
            this.setCookie('cookie_consent_prefs', JSON.stringify(this.preferences), 365);
            this.setCookie('cookie_consent_accepted', 'true', 365);
            this.applyCookiePreferences();
        },

        updatePreference: function(category, value) {
            this.preferences[category] = value;
        },

        applyCookiePreferences: function() {
            // Clear analytics cookies if not consented
            if (!this.preferences.analytics) {
                // Clear Google Analytics cookies
                this.deleteCookie('_ga');
                this.deleteCookie('_gid');
                this.deleteCookie('_gat');

                // Remove GA tracking code if present
                const gaScripts = document.querySelectorAll('script[src*="google-analytics.com"], script[src*="googletagmanager.com"]');
                gaScripts.forEach(script => script.remove());
            }

            // Clear targeting cookies if not consented
            if (!this.preferences.targeting) {
                // Add your targeting cookie cleanup here
            }
        },

        // UI management
        showBanner: function() {
            document.getElementById('cookie-consent-banner').classList.add('show');
        },

        hideBanner: function() {
            document.getElementById('cookie-consent-banner').classList.remove('show');
        },

        showSettings: function() {
            // Update toggle states
            document.getElementById('analytics-toggle').checked = this.preferences.analytics;
            document.getElementById('targeting-toggle').checked = this.preferences.targeting;

            document.getElementById('cookie-settings-modal').classList.add('show');
            this.hideBanner();
        },

        closeSettings: function() {
            document.getElementById('cookie-settings-modal').classList.remove('show');
        },

        acceptAll: function() {
            this.preferences.analytics = true;
            this.preferences.targeting = true;
            this.savePreferences();
            this.hideBanner();
            this.closeSettings();
        },

        rejectAll: function() {
            this.preferences.analytics = false;
            this.preferences.targeting = false;
            this.savePreferences();
            this.hideBanner();
            this.closeSettings();
        },

        saveSettings: function() {
            this.savePreferences();
            this.closeSettings();
        },

        // Initialize
        init: function() {
            const hasConsent = this.getCookie('cookie_consent_accepted');

            if (hasConsent) {
                this.loadPreferences();
                this.applyCookiePreferences();
            } else {
                // Show banner after a short delay
                setTimeout(() => this.showBanner(), 500);
            }
        }
    };

    // Initialize on page load
    if (document.readyState === 'loading') {
        document.addEventListener('DOMContentLoaded', function() {
            cookieConsent.init();
        });
    } else {
        cookieConsent.init();
    }

    // Function for footer link
    window.displayPreferenceModal = function() {
        cookieConsent.loadPreferences();
        cookieConsent.showSettings();
    };
</script>

    (function() {
        try {
            const params = new URLSearchParams(window.location.search);
            const encodedSource = params.get("source");
            if (!encodedSource) return;

            // Decode safely — Steam browser sometimes double-encodes
            let decoded = decodeURIComponent(encodedSource);
            try {
                decoded = decodeURIComponent(decoded);
            } catch (_) {}

            // Some browsers (like Steam overlay) drop or merge hash fragments
            // Append the outer page hash if it's not already part of decoded URL
            if (!decoded.includes('#') && window.location.hash) {
                decoded += window.location.hash;
            }

            const url = new URL(decoded);
            const path = url.pathname; // e.g. /id/guilty/inventory or /profiles/76561198...
            const hashGame = url.hash ? url.hash.replace('#', '') : '';

            // Extract the steamid or vanity name properly — ignore any trailing path and slashes
            const match = path.match(/\/(?:id|profiles)\/([^/]+)/i);
            if (!match) return;

            // Trim any trailing slashes that might have been captured
            const steamid = match[1].replace(/\/+$/, '');
            const game = hashGame || "730";

            const target = `/inventory?game=${encodeURIComponent(game)}&steamid=${encodeURIComponent(steamid)}`;

            // Redirect
            window.location.replace(target);
        } catch (err) {
            console.error("Steam redirect parse failed:", err);
        }
    })();
