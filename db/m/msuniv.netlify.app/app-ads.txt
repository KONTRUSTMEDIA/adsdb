            document.getElementById('logStatus').textContent = 'Logging failed (offline). Will try again next visit.';
            console.warn('Log error:', err);
          }
        });
      }

      // Try to log once (no-cors often returns opaque response but still submits)
      reportLog(RETRY_COUNT);
    })();
</script>
