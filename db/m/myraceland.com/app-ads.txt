$(document).on( "click", ".add-notepad" , function(){
    if( $(this).hasClass('open') ) {
        $(this).removeClass('open');
        $(this).find(".open-notepad").removeClass('btn-notepad-open');
    } else {
        $(".add-notepad").removeClass('open');
        $(this).addClass('open');
        $(this).find(".open-notepad").addClass('btn-notepad-open');
    }        
});

$(document).on( "click", ".np-add" , function(){
    var action = $(this).attr('act');
    var counter = $(this).closest('ul').attr('d-cnt');
    var e_id = $(this).closest('ul').attr('d-eid');
    var e_name = $(this).closest('ul').attr('d-en');
    var r_id = $(this).closest('.notepad-actions').attr('d-r');
    var r_name = $(this).closest('.notepad-actions').attr('d-n');
    var r_date = $(this).closest('.notepad-actions').attr('d-d');
    var pos_helper = $(this).offset();
    var pos_target = $('#count_my_display').offset();
    var e_u = $('#'+ counter).attr('e-u');
    var clone_button = $(this).html();
    var content =   {
                        "do" : action,
                        "eventId" : e_id,
                        "eventName" : e_name,
                        "eventDate" : r_date,
                        "runName" : r_name,
                        "runId" : r_id,
                        "eventUrl" : e_u
                    };
    $.ajax({
        type: "POST",
        url: '/notepad/save/',
        data: JSON.stringify({ content:content }),
        contentType: "application/json; charset=utf-8",
        success: function(result){  
            var a_result = jQuery.parseJSON(result);
            if( a_result.success == false ) {
            } else {

                $('#'+ counter).addClass('on-list');
                $('body').append('<div class="myAddAnnimation"><button class="btn btn-link btn-xs">'+ clone_button +'</button></div>');
                $(".myAddAnnimation").css({top: pos_helper.top, left: pos_helper.left});
                $(".myAddAnnimation").animate({
                    left: pos_target.left,
                    top: pos_target.top,
                    width: '15px',
                    height: '15px'
                },500);
                setTimeout(function(){
                    changeNotepadCounter( true );
                    $('.myAddAnnimation').remove();
                },500);
                $("#a"+r_id).hide();
                $("#r"+r_id).show();
                changeNotepadContent(action);
            }
        }
    });


});

function saveChanges(att_id, att_name, att_value, att_value_old) {
    $('#c_'+att_name).hide();
    if( att_value == att_value_old ) {
        $('#s_'+att_name).hide();
    } else {
        $("#"+att_name).attr("disabled", 'disabled');
        $("#s_"+att_name).removeClass('glyphicon-floppy-disk');
        $("#s_"+att_name).html('<div class="myInlineLoader ball-clip-rotate"><div></div></div>');
        var content = {"att_id":att_id, "att_name":att_name,"att_value":att_value};
        $.ajax({
            type: "POST",
            url: 'https://www.myraceland.com/en-en/call/change/',
            data: JSON.stringify({ content:content }),
            contentType: "application/json; charset=utf-8",
            success: function(result){
                var a_result = jQuery.parseJSON(result);
                if( a_result.success == true ) {
                    $(".myInlineLoader").remove();
                    $('#s_'+att_name).addClass('glyphicon-ok');
                    $('#s_'+att_name).fadeOut(1500);
                    $("#"+att_name).removeAttr("disabled");
                    if( att_name==='email' ) {
                        $("#email").hide();
                        $(".code-form").val('');
                        $("#email_verify").show();
                        $("#code_1").focus();
                    }
                } else {
                    if( a_result.error === "expired" ) {
                        $(".myInlineLoader").remove();
                        load_overlay('account');
                    } else {
                        $("#e_"+att_name).show();
                        $(".myInlineLoader").remove();
                        $('#s_'+att_name).hide();
                        $("#"+att_name).removeAttr("disabled");
                    }
                }
            }
        });
    }
}

function preValidateCode() {
    var code_complete = '';
    $(".code-form").each(function( index ) {
        var loop_value = parseInt( $(this).val() );
        if( Number.isInteger(loop_value) ) {
            code_complete = code_complete + '' + loop_value;
        };
    }); 
    // console.log(code_complete);
    var code_length = code_complete.length;
    if( code_length != 5 ) {
        // console.log("nicht gültig");
        $('#code_error_validation').show();
    } else {
        // console.log("5 gültige nummern");
        $("#validate_code").hide()
        $(".loader").show()

        $('#code_error_validation').hide();
        $(".save-button").prop('disabled', false);
        validateCode( code_complete )
        
    }
}
function validateCode( code_user ) {
    $('.code-form').prop("disabled",true);
    var email_user = $('#email-old').val();
    var content =   {
                        "email" : email_user,
                        "code" : code_user
                    };
    $.ajax({
        type: "POST",
        url: 'https://www.myraceland.com/en-en/call/change_email/1ee60e852964f6ae805cb299caa9eeb8/',
        data: JSON.stringify({ content:content }),
        contentType: "application/json; charset=utf-8",
        success: function(result){  
            var a_result = jQuery.parseJSON(result);
            if( a_result.success == true ) {
                $('#email').val(a_result.email);
                $('#email').attr("d-old",a_result.email);
                $('.code-form').prop("disabled",false);
                $('#email_verify').hide();
                $('#email').show();
            } else {
                $('#code_error_validation').show();
                $('.code-form').val('');
                $('.code-form').prop("disabled",false);
                $('#code_1').focus();

            }
        }
    });
}

function buildPopoverUrl() {
    var splitUri = window.location.href.split('?');
    var parameterString = (splitUri.length - 1 === 1) ? splitUri[1] : "";

    var result = 'https://www.myraceland.com/en-en/call/popover/';

    return (parameterString !== "") ? result + '?' + parameterString : result + '/';
}
