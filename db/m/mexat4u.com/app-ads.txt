    <script>
        document.addEventListener("DOMContentLoaded", function () {
            let currentPage = 2;
            const maxPages = 6;
            let isLoading = false;

            function loadMoreArticles() {
                if (isLoading) return;
                isLoading = true;
                document.getElementById("loadingSpinner").style.display = "block";

                fetch(`/Home/LoadMoreArticles?page=${currentPage}`)
                    .then(response => response.json())
                    .then(data => {
                        if (data.length === 0) {
                            document.getElementById("loadMoreButton").style.display = "none";
                            return;
                        }

                        const newsContainer = document.getElementById("newsContainer");

                        data.forEach(news => {
                            const shortCategory = news.category?.length > 15 ? news.category.substring(0, 15) + "..." : news.category || "عام";
                            const formattedDate = new Date(news.postDate).toLocaleDateString('en-US', {
                                month: 'short',
                                day: '2-digit',
                                year: 'numeric'
                            });

                            let newsItem = `
                                <div class="col-lg-6 news-item">
                                    <div class="position-relative mb-3">
                                        <img src="/placeholder.gif" data-src="${news.imageUrl}" class="img-fluid w-100 lazyload" alt="${news.title}" title="${news.title}" style="object-fit: cover; height: 243px;" />
                                        <div class="bg-white border border-top-0 p-4">
                                            <div class="mb-2">
                                                <a class="badge badge-danger text-uppercase font-weight-semi-bold p-1 mr-2" href="/articles/category/${encodeURIComponent(news.categorySlug)}">${shortCategory}</a>
                                                <a class="text-body" href="#"><small>${formattedDate}</small></a>
                                            </div>
                                            <a class="h4 d-block mb-3 text-secondary text-uppercase font-weight-bold" style="font-size: 20px;" href="/articles/details/${encodeURIComponent(news.slug)}">${news.title}</a>
                                        </div>
                                    </div>
                                </div>
                            `;

                            newsContainer.insertAdjacentHTML("beforeend", newsItem);
                        });

                        setupLazyLoading();
                        currentPage++;
                        if (currentPage > maxPages) {
                            document.getElementById("loadMoreButton").style.display = "block";
                            window.removeEventListener("scroll", handleScroll);
                        }

                        document.getElementById("loadingSpinner").style.display = "none";
                        isLoading = false;
                    })
                    .catch(() => {
                        document.getElementById("loadingSpinner").style.display = "none";
                        isLoading = false;
                    });
            }

            function handleScroll() {
                if (window.innerHeight + window.scrollY >= document.body.offsetHeight - 100) {
                    loadMoreArticles();
                }
            }

            function setupLazyLoading() {
                const lazyImages = document.querySelectorAll("img.lazyload");

                const observer = new IntersectionObserver((entries, observer) => {
                    entries.forEach(entry => {
                        if (entry.isIntersecting) {
                            const img = entry.target;
                            img.src = img.getAttribute("data-src");
                            img.classList.remove("lazyload");
                            observer.unobserve(img);
                        }
                    });
                });

                lazyImages.forEach(img => observer.observe(img));
            }

            window.addEventListener("scroll", handleScroll);
            document.getElementById("loadMoreButton").addEventListener("click", loadMoreArticles);
        });
    </script>
