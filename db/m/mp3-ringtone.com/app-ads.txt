<script>
    let player = document.querySelector('#audio_player');
    $(document).on('click', '.audio_itemplay', function(e){
        e.preventDefault();
        let currentIcon = $(this).children('.material-icons').text();
        let icon_playing = 'pause';
        let icon_paused = 'play_arrow';
        let source = $(this).closest('.letaudioplay').attr('data-file').trim();
        let data_id = $(this).closest('.letaudioplay').attr('data-id').trim();
        data_id = parseInt(data_id);
        let player_idnow = $('#audio_player').attr('audio-id').trim();
        player_idnow = player_idnow == "" ? 0 : parseInt(player_idnow);

        let item_selector = '#audio_'+data_id;

        if(player_idnow != data_id){
            $('.audio_item_current_time').html('<!--removetiming-->');
            $('.playtrigericon').html(icon_paused);
            player.src = source;
            player.load();

            $('#audio_player').attr('audio-id', data_id);
            $('#audio_player').trigger('play');
            $(item_selector+' .playtrigericon').html(icon_playing);
        }
        else{
            if(player.paused){
                $(item_selector+' .playtrigericon').html(icon_playing);
                player.play();
            } else {
                $(item_selector+' .playtrigericon').html(icon_paused);
                player.pause();
            }
        }
    });

    function setSliderAndTimer() {
        //get minutes
        let minutes = Math.floor(player.currentTime / 60);
        if (minutes < 10) {
            minutes = "0" + minutes;
        }
        //get seconds
        let seconds = Math.floor(player.currentTime % 60);
        if (seconds < 10) {
            seconds = "0" + seconds;
        }
        let textContent = `${minutes}:${seconds}`;
        if (player.currentTime == player.duration) {
            player.pause();
        }
        let item_selector = '#audio_'+player.getAttribute('audio-id').trim();
        $(item_selector+' .audio_item_current_time').html(textContent);
    }

    player.addEventListener("timeupdate", setSliderAndTimer);
    player.onended = function(){
        let item_selector = '#audio_'+this.getAttribute('audio-id').trim();
        $(item_selector+' .audio_item_current_time').html('<!--hide-->');
        $(item_selector+' .playtrigericon').html('play_arrow');
    };
</script>
