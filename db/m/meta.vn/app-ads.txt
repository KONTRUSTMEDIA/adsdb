    window.MiniCartAutoSave = function ($section) {
        $('input,select,textarea', $section).on('change', function () {
            var data = window._cartInfo;
            if (typeof (data) == 'undefined' || data == null) {
                data = {};
                data.screenWidth = screen.width;
                data.screenHeight = screen.height;
            }
            $('input[data-link],select[data-link],textarea[data-link]').each(function () {
                var $inp = $(this);
                var linkId = $inp.data('link');
                if (linkId) {
                    if ($inp.attr('type') == 'checkbox') {
                        if ($inp.is(':checked'))
                            data[linkId] = '1';
                        else
                            data[linkId] = '0';
                    } else if ($inp.attr('type') == 'radio') {
                        if ($inp.is(':checked'))
                            data[linkId] = $inp.val();
                    }
                    else
                        data[linkId] = $inp.val();
                }
            });

            var products = [];
            var records = [];

            try {
                $('.product-viewed-item', $section).each(function () {
                    var pid = $(this).attr('data-pid');
                    var isKm = $(this).attr('data-record-id') == '0' ? '1' : '0';
                    var qty = $('.txtQty-cart', $(this)).val() || $('.txtQty-cart', $(this)).text();
                    products.push(pid + ':' + $.trim(qty) + ':' + isKm);
                    records.push($(this).attr('data-record-id') + ':' + qty)
                });
                data.products = products.join(',');
            } catch (e) { }

            try {
                $.post('/ajx/cart-spy.ashx', data, function (returnHTML) {
                    console.log(returnHTML);
                }).error(function () {
                    console.log(data);
                });
            } catch (e) {
                console.log(e);
            }
            window._cartInfo = data;

            //reload giỏ hàng

                    act: 'changecart',

        function showVisited() {

            if ($('.list-notications-box.cart .visited-list li').length > 0)
                return;

            try {
                var opt = {
                    container: ".list-notications-box.cart",
                    pageSize: 3,
                    pageIndex: 1,
                    fade: false,
                    deleteAlow: true,
                    pagger: function (rowsCount, pageSize, $box) {
                        $('.visited-more', $box).remove();
                        if (rowsCount > pageSize) {
                            $('.visited-inner', $box).append('<div class="product-viewed-more visited-more">'
                                + '<a href="/visited">Xem tất cả ' + rowsCount + ' sản phẩm</a></div>');
                        }
                    }
                };
                setTimeout(function () {
                    VisitedProducts().display(opt, "cart.ascx");
                    setTimeout(function () {
                        appendQuery();
                    }, 1000);
                }, 1000);
                countVisitedCart();
            } catch (e) { console.log("#head-cart-box " + e.message) }
        }

                    if (href != "#" && href.indexOf(".html") < 0)
                        $(this).attr('href', href + "?" + _moreQuery);
                });
            }
        }

        $(document).on('cart.firstload', function (event, params) {
            addClickActions();
            addCartEvents();
            showVisited();

<script>
    // Script để lấy thông tin time zone và thêm vào userTrackingData
    document.addEventListener('DOMContentLoaded', function () {
        window.userTrackingData = window.userTrackingData || {};
        const eventData = window.userTrackingData.eventData || {};
        const nav = {};

        // Thông tin mạng
        if (navigator.connection) {
            nav.connection = {
                connectionType: navigator.connection.effectiveType,
                bandwidth: navigator.connection.downlink,
            }
        }
        nav.language = navigator.language;
        nav.platform = navigator.platform;

        // Thông tin phần cứng
        nav.cpuCores = navigator.hardwareConcurrency;
        nav.deviceMemory = navigator.deviceMemory;

        eventData.navigator = nav;

        // Thông tin màn hình
        eventData.screen = {
            width: window.screen.width,
            height: window.screen.height,
            colorDepth: window.screen.colorDepth
        };

        // Thông tin thiết bị
        eventData.touchSupport = 'ontouchstart' in window;

        // Thông tin GPU
        try {
            var canvas = document.createElement('canvas');
            var gl = canvas.getContext('webgl') || canvas.getContext('experimental-webgl');
            if (gl) {
                var debugInfo = gl.getExtension('WEBGL_debug_renderer_info');
                if (debugInfo) {
                    eventData.gpuVendor = gl.getParameter(debugInfo.UNMASKED_VENDOR_WEBGL);
                    eventData.gpuRenderer = gl.getParameter(debugInfo.UNMASKED_RENDERER_WEBGL);
                }
            }
        } catch (e) {
            console.error('Unable to get GPU information', e);
        }

        window.userTrackingData.eventData = eventData;

        const extendData = {};

        extendData.userName = 'guest';

        window.userTrackingData.extendData = extendData;
    });

    (function () {
        const trackingDataUrl = 'https://metrics.meta.vn/collector/v1/track';
        SkyNetTrackingData(trackingDataUrl);
    })();
