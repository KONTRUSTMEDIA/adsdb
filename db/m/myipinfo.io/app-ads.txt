  <script>
    // Helper to load scripts with error handling
    const loadScript = (src, options = {}) => {
      return new Promise((resolve, reject) => {
        const script = document.createElement("script");
        script.src = src;
        if (options.async) script.async = true;
        if (options.defer) script.defer = true;
        if (options.id) script.id = options.id;
        if (options.attributes) {
          Object.entries(options.attributes).forEach(([key, value]) => {
            script.setAttribute(key, value);
          });
        }
        script.onload = () => resolve();
        script.onerror = () => {
          console.warn(`Failed to load script: ${src} (likely blocked by ad blocker)`);
          reject(new Error(`Script load failed: ${src}`));
        };
        document.body.appendChild(script);
      });
    };

    // Load Mapbox resources asynchronously after initial render
    setTimeout(() => {
      // Load Mapbox CSS (v3.10.0)
      const mapboxCss = document.createElement("link");
      mapboxCss.rel = "stylesheet";
      mapboxCss.href = "https://api.mapbox.com/mapbox-gl-js/v3.10.0/mapbox-gl.css";
      mapboxCss.media = "print";
      mapboxCss.onload = function() { this.media = "all"; this.onload = null; };
      document.head.appendChild(mapboxCss);

      // Load Mapbox JS (v3.10.0)
      loadScript("https://api.mapbox.com/mapbox-gl-js/v3.10.0/mapbox-gl.js", { defer: true })
        .catch(err => console.warn("Mapbox failed to load:", err.message));
