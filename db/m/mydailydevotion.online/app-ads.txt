    function isMobileOrFixedTypeOk(itiInstance) {
        if (!window.intlTelInputUtils || !itiInstance.getNumberType) return true;
        const t = itiInstance.getNumberType();
        return t === intlTelInputUtils.numberType.MOBILE ||
               t === intlTelInputUtils.numberType.FIXED_LINE ||
               t === intlTelInputUtils.numberType.FIXED_LINE_OR_MOBILE;
    }

    function validateUi(itiInstance, phoneEl) {
        phoneEl.setCustomValidity('');
        const validShape = itiInstance.isValidNumber();

        }

    }

    function allowDigits(phone)
    {
        phone.setAttribute('inputmode', 'numeric');
        phone.setAttribute('pattern', '\\d*');

        phone.addEventListener('beforeinput', (e) => {
            if (e.inputType && e.inputType.startsWith('delete')) return;
            if (typeof e.data === 'string' && /\D/.test(e.data)) {

            }
        });
    }

    function startCountdown(duration, redirectUrl) {
        var timer = duration, seconds;
        var countdownInterval = setInterval(function () {
            seconds = parseInt(timer, 10);
            $('#timer-seconds').text(seconds + '...');

            if (--timer < 0) {
                clearInterval(countdownInterval);
                window.location.href = redirectUrl;  // Redirect to the URL
            }
        }, 1000);
    }


    function sendClarityEvent(eventName, tags = {}) {
        if (typeof window.clarity !== 'function') {
            return;
        }
        Object.entries(tags).forEach(([key, val]) => {
            clarity('set', key, String(val));
        });
        clarity('event', eventName);
    }

    function sendClarityIdentity(identity) {
        if (typeof window.clarity !== 'function') {
            return;
        }
        if (identity) {
            clarity('identify', identity);
        }
    }

        document.querySelectorAll('.fiat-tags .tag').forEach(tag => {
            tag.addEventListener('click', function (e) {
                e.preventDefault();

                // Убираем active у всех
                document.querySelectorAll('.fiat-tags .tag').forEach(t => t.classList.remove('active'));
                // Добавляем active текущему
                this.classList.add('active');

                const fiat = this.getAttribute('data-fiat'); // aed | usd | sar

                // Обновляем каждую монету
                document.querySelectorAll('.list-item[data-coin]').forEach(item => {
                    const coin = item.getAttribute('data-coin');

                    // Берём данные для выбранной валюты, fallback на AED
                    const data = pricesData[fiat]?.[coin] || pricesData['aed'][coin];

                    if (!data) return;

                    // Основная цена
                    const mainPriceEl = item.querySelector('.item-price-main');
                    if (mainPriceEl) mainPriceEl.textContent = data.price;

                    // Вторичная цена (USD или AED)
                    const secondaryPriceEl = item.querySelector('.item-price-secondary');
                    if (secondaryPriceEl) secondaryPriceEl.textContent = data.secondary;

                    // Изменение в процентах
                    const changeEl = item.querySelector('.item-change > div');
                    if (changeEl) {
                        changeEl.textContent = data.change;

                        // Правильный класс: красный или зелёный
                        changeEl.className = 'text-bold ltr ' +
                            (data.change.includes('-') ? 'change-badge-danger' : 'change-badge-primary');
                    }
                });
            });
        });

        // По умолчанию активируем AED (если ещё не активирован)
        const defaultTag = document.querySelector('.fiat-tags .tag[data-fiat="aed"]');
        if (defaultTag && !defaultTag.classList.contains('active')) {
            defaultTag.click();
        }
    });
