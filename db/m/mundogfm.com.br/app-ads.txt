    async function getScheduleList(url) {
        const d = new Date();
        const response = await fetch(url)
        const data = await response.json();
        mountScheduleList(data[Object.keys(data)[d.getDay() - 1]], 'on')

    function getProgramasAtivos(schedule) {
        const now = new Date();
        const filteredProgramas = schedule.filter(programa => {
            const start = new Date(`${now.toDateString()} ${programa.inicio}`);
            const end = new Date(`${now.toDateString()} ${programa.fim}`);
            return start <= now && end >= now;
        });
        const futureProgramas = schedule.filter(programa => {
            const start = new Date(`${now.toDateString()} ${programa.inicio}`);
            const end = new Date(`${now.toDateString()} ${programa.fim}`);
            return start > now;
        }).slice(0, 2);
        return [...filteredProgramas, ...futureProgramas];
    } 

    function mountScheduleList(schedule, live) {
        let programas = getProgramasAtivos(schedule)
        const now = new Date();
        let horas = now.getHours().toString().padStart(2, '0');
        let minutos = now.getMinutes().toString().padStart(2, '0');
        let segundos = now.getSeconds().toString().padStart(2, '0');
        let horarioAtual = `${horas}:${minutos}:${segundos}`;
        const scheduleItems = programas.map(({ programa, descrição, inicio, fim, imagem, locutor }) => {
            const isPlaying = horarioAtual >= inicio && horarioAtual < fim;
            const isLive = live === 'on' && isPlaying;
            return `
      <div class="programa fade-in ${isLive ? 'liveNow' : ''}">
        <figure>
          <img loading="lazy" src="${imagem}" alt="${programa}" />
        </figure>

        </button>
        <p>${descrição}</p>
      </div>
    `;

        const initializePoll339 = async () => {
            const pollForm = document.querySelector(`[data-poll-id="339"] .mw-survey-form`);
            const pollButton = document.querySelector(`[data-poll-id="339"] .mw-survey-btn`);

            if (!pollForm || !pollButton) {

            const showPollResults = async (pollItemId) => {
                // console.log("chamou aq")
                try {
                    const response = await fetch(`/enquete/poll-results?pollId=339`);
                    const html = await response.text();

                    // console.log(html)

                    pollForm.outerHTML = html;

                    const pollItem = document.querySelector(`[data-poll-id="339"] [data-poll-item-id="${pollItemId}"]`);
                    if (pollItem) {
                        pollItem.classList.add('votedItem')
                    }

                    pollButton.onclick = null;
                    pollButton.style.visibility = 'hidden';
                } catch (error) {
                    console.error(error);
                }
            };

            const registerPollVote = async (formData) => {
                try {
                    const response = await fetch('/enquete/register-poll-vote', {
                        method: 'POST',
                        body: formData,
                    });

                    const { errors } = await response.json();
                    return !errors;
                } catch (error) {
                    console.error(error);
                    return false;
                }
            };

            for (const cookie of document.cookie.split('; ')) {
                const [name, value] = cookie.split('=');
                if (name === `pool-vote-339` && value) {
                    showPollResults(value);
                }
            }

            pollButton.onclick = async (event) => {
                event.preventDefault();

                const formData = new FormData(pollForm);
                const pollItemId = formData.get('pollItemId');

                if (!pollItemId) {
                    return;
                }

                const isVoteRegistered = await registerPollVote(formData);

                showPollResults(pollItemId);
                if (isVoteRegistered) {
                    const expiration = new Date();
                    expiration.setTime(expiration.getTime() + 10 * 365 * 24 * 60 * 60 * 1000);

                    document.cookie = `pool-vote-339=${pollItemId}; expires=${expiration.toUTCString()}; path=/`;
                    showPollResults(pollItemId);
                }
            };

        if (typeof window.initializePoll339 !== 'function') {
