    <script>
        document.addEventListener('DOMContentLoaded', function () {
            const menuLinks = document.querySelectorAll('.main-navigation ul li a');
            const activeMenuKey = 'tosinoActiveMenuHref'; // localStorage에 저장할 고유한 키

            function setActiveMenu(activeHref) {
                let menuActivated = false;
                menuLinks.forEach(function(link) {
                    link.classList.remove('active-menu'); // 일단 모든 메뉴 비활성화
                    if (link.getAttribute('href') === activeHref) {
                        link.classList.add('active-menu');
                        menuActivated = true;
                    }
                });
                return menuActivated;
            }

            // 1. 페이지 로드 시 localStorage에서 마지막 활성 메뉴 Href 가져오기
            let lastActiveHref = localStorage.getItem(activeMenuKey);

            // 2. 가져온 Href로 메뉴 활성화 시도
            let activated = false;
            if (lastActiveHref) {
                activated = setActiveMenu(lastActiveHref);
            }

            // 3. localStorage에 정보가 없거나, 해당 Href의 메뉴가 현재 페이지에 없다면
            //    첫 번째 메뉴를 활성화하고, 그 정보를 localStorage에 저장
            if (!activated && menuLinks.length > 0) {
                const firstMenuHref = menuLinks[0].getAttribute('href');
                setActiveMenu(firstMenuHref);
                localStorage.setItem(activeMenuKey, firstMenuHref); // 첫 번째 메뉴를 기본 활성으로 저장
            }

            // 4. 각 메뉴 링크에 클릭 이벤트 리스너 추가
            menuLinks.forEach(function(link) {
                link.addEventListener('click', function() {
                    // 클릭된 링크의 href를 localStorage에 저장
                    const clickedHref = this.getAttribute('href');
                    localStorage.setItem(activeMenuKey, clickedHref);

                    // 즉시 시각적 업데이트 (실제 페이지 이동 전에 반영)
                    setActiveMenu(clickedHref);

                    // 링크의 기본 동작(페이지 이동)은 막지 않음
                });
            });
        });
    </script>
