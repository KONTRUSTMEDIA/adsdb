(function () {
  let hasTriggered = false;

  function getScrollPercent() {
    const scrollTop = window.scrollY || document.documentElement.scrollTop;
    const docHeight = document.documentElement.scrollHeight - window.innerHeight;
    return (scrollTop / docHeight) * 100;
  }

  function shouldShowPopup() {
    const lastShown = localStorage.getItem("popupLastShown");
    if (!lastShown) return true;
    const elapsed = (Date.now() - parseInt(lastShown, 10)) / (1000 * 60 * 60 * 24);
    return elapsed >= 1;
  }

  function showPopup() {
    const trigger = document.querySelector('[fs-modal-element="open-1"]');
    if (trigger) {
      console.log("🎯 Trigger found — opening modal");
      trigger.click();
    } else {
      console.warn("❌ Modal trigger not found — retrying...");
      setTimeout(showPopup, 1000);
    }
  }

  function trackModalClose() {
    document.addEventListener("click", (e) => {
      if (e.target.closest('[fs-modal-element="close"]')) {
        localStorage.setItem("popupLastShown", Date.now().toString());
        console.log("❌ Modal closed by user — timestamp saved");
      }
    });
  }
  
  function handleTimeTrigger() {
  if (hasTriggered) return;

  setTimeout(() => {
    if (hasTriggered) return;
    hasTriggered = true;

    if (shouldShowPopup()) {
      showPopup();
    }
  }, 5000); // 5 seconds
}

  function handlePopupLogic() {
    handleTimeTrigger();
    trackModalClose();
  }

  document.addEventListener("DOMContentLoaded", handlePopupLogic);
})();
