$(function(){
	$btn_scroll_top = $('.btn_back-top');

		btn_scroll_h_pos = 170;
	}

	var moveHandler = function(){
			var scroll_top_pos = 80;
			if($(window).scrollTop() > scroll_top_pos){
			if(!scroll){
				$btn_scroll_top.queue([]);
				$btn_scroll_top.stop();
				$btn_scroll_top.css('opacity','0.9');
				$btn_scroll_top.show();

			}
		}else{
			$btn_scroll_top.queue([]);
			$btn_scroll_top.stop();
			$btn_scroll_top.hide();
		}
	}

	$(window).on('scroll',moveHandler);
	$(window).on('resize',moveHandler);


	moveHandler();

})
