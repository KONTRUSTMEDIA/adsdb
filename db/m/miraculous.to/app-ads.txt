<script>
(function(){
  const MODAL_ID = "promo-modal";
  const COOKIE_NAME = "MTOPROMO3";
  const SHOW_AFTER_MS = 800;
  const SUPPRESS_DAYS = 1;

  const modal = document.getElementById(MODAL_ID);
  if(!modal) return;

  const sheet = modal.querySelector(".mto-promo__sheet");
  const closeEls = modal.querySelectorAll("[data-close]");
  const link = modal.querySelector(".mto-promo__link");

  let lastFocused = null;

  function setCookie(name, value, days){
    const d = new Date();
    d.setTime(d.getTime() + (days*24*60*60*1000));
    document.cookie = `${name}=${encodeURIComponent(value)}; expires=${d.toUTCString()}; path=/; domain=miraculous.to; secure; samesite=Strict`;
  }
  function getCookie(name){
    const m = document.cookie.match(new RegExp('(^| )'+name+'=([^;]+)'));
    return m ? decodeURIComponent(m[2]) : '';
  }
  function lockScroll(yes){
    document.documentElement.style.overflow = yes ? "hidden" : "";
    document.body.style.overflow = yes ? "hidden" : "";
  }

  function trapFocus(e){
    if(e.key !== "Tab") return;
    const focusables = sheet.querySelectorAll('a, button, [tabindex]:not([tabindex="-1"])');
    if(!focusables.length) return;
    const first = focusables[0], last = focusables[focusables.length-1];
    if(e.shiftKey && document.activeElement === first){
      e.preventDefault(); last.focus();
    }else if(!e.shiftKey && document.activeElement === last){
      e.preventDefault(); first.focus();
    }
  }

  function openModal(){
    if(modal.hasAttribute("hidden")){
      modal.removeAttribute("hidden");
      modal.classList.add("mto-open");
      lastFocused = document.activeElement;
      lockScroll(true);
      (link || sheet).focus?.();
      document.addEventListener("keydown", onKey);
      sheet.addEventListener("keydown", trapFocus);
    }
  }
  function closeModal(){
    modal.classList.remove("mto-open");
    lockScroll(false);
    modal.setAttribute("hidden","true");
    document.removeEventListener("keydown", onKey);
    sheet.removeEventListener("keydown", trapFocus);
    if(lastFocused && lastFocused.focus) lastFocused.focus();
    setCookie(COOKIE_NAME, "1", SUPPRESS_DAYS);
  }
  function onKey(e){
    if(e.key === "Escape" || e.key === "Esc" || e.key === "Backspace"){
      e.preventDefault(); closeModal();
    }
  }

  closeEls.forEach(el => el.addEventListener("click", closeModal));
  modal.addEventListener("click", function(e){
    if(e.target && e.target.dataset && e.target.dataset.close) closeModal();
  });

  window.MTOpromo = { open: openModal, close: closeModal };

  if(!getCookie(COOKIE_NAME)){
    window.addEventListener("load", () => setTimeout(openModal, SHOW_AFTER_MS));
  }
})();
</script>

  function toggleElement(element){
    var el = document.getElementById(element);
    if (!el) return;
    el.style.display = (el.style.display === 'none') ? '' : 'none';
  }

  var userTimezone = Intl.DateTimeFormat().resolvedOptions().timeZone;
  var userLang = navigator.language || navigator.userLanguage;

  function updateCountdown(episodeId, isoUtcString, titleElement) {
    var episodeDate = new Date(isoUtcString);
    if (isNaN(episodeDate)) {
      var cd = document.getElementById('countdown-' + episodeId);
      if (cd) cd.textContent = '—';
      return;
    }

    var now = new Date();
    var diff = episodeDate.getTime() - now.getTime();

    if (diff <= 0) {
      document.getElementById('countdown-' + episodeId).innerHTML =
        '<span style="color: green; font-weight: bold;">RELEASED</span>';
      if (titleElement) {
        var originalTitle = titleElement.getAttribute('data-original-title') || titleElement.textContent;
        titleElement.innerHTML = originalTitle + ' <span style="color:green; font-weight:bold;">RELEASED</span>';
      }
    } else {
      var days = Math.floor(diff / 86400000);
      var hours = Math.floor((diff % 86400000) / 3600000);
      var minutes = Math.floor((diff % 3600000) / 60000);

      if (titleElement) {
        var originalTitle = titleElement.getAttribute('data-original-title') || titleElement.textContent;
        if (days < 7) {
          titleElement.innerHTML = originalTitle + ' <span style="color:red; font-weight:bold;">UPCOMING</span>';
        } else {
          titleElement.innerHTML = originalTitle;
        }
      }

      var countdownText = '<span style="color: red; font-weight: bold;">'
        + days + ' Day(s), '
        + hours + ' Hour(s), '
        + minutes + ' Minute(s)</span>';

      document.getElementById('countdown-' + episodeId).innerHTML = countdownText;
    }

    var options = {
	  timeZone: userTimezone,
	  year: 'numeric',
	  month: 'long',
	  day: 'numeric',
	  hour: 'numeric',
	  minute: 'numeric',
	  hour12: false,
	  timeZoneName: 'short'
};
    var formattedDate = new Intl.DateTimeFormat(userLang, options).format(episodeDate);
    document.getElementById('episode-date-' + episodeId).textContent = formattedDate;
  }

  function startCountdowns() {
    var episodes = document.querySelectorAll('[data-episode-date]');
    episodes.forEach(function(element) {
      var episodeId = element.getAttribute('data-episode-id');
      var isoUtc = element.getAttribute('data-episode-date');
      var titleElement = document.getElementById('title-' + episodeId);
      updateCountdown(episodeId, isoUtc, titleElement);
    });
  }

  document.addEventListener('DOMContentLoaded', function() {
    startCountdowns();
    setInterval(startCountdowns, 1000);
  });
