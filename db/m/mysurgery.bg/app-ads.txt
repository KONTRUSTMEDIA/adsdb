<script>
document.addEventListener("DOMContentLoaded", function () {
  const voiceSelect = document.getElementById("voiceSelect");
  const readBtn = document.getElementById("readArticleBtn");
  const stopBtn = document.getElementById("stopReadBtn");
  const indicator = document.getElementById("readingIndicator");

  let voices = [];
  let isReading = false;
  let stopReading = false;

  function populateVoices() {
    voices = speechSynthesis.getVoices();
    voiceSelect.innerHTML = '';

    voices.forEach((voice, i) => {
      if (voice.lang.startsWith("bg")) {
        const option = document.createElement("option");
        option.value = i;
        option.textContent = `${voice.name} (${voice.lang})`;
        voiceSelect.appendChild(option);
      }
    });

    if (voiceSelect.options.length === 0) {
      const option = document.createElement("option");
      option.textContent = "❌ Няма налични български гласове";
      option.disabled = true;
      voiceSelect.appendChild(option);
    }
  }

  speechSynthesis.onvoiceschanged = populateVoices;
  populateVoices();

  function speakSentences(text, voice) {
    const sentences = text.match(/[^.!?]+[.!?]+/g) || [text];
    let index = 0;
    stopReading = false;
    indicator.style.display = "block";

    function speakNext() {
      if (stopReading || index >= sentences.length) {
        isReading = false;
        indicator.style.display = "none";
        return;
      }

      const utterance = new SpeechSynthesisUtterance(sentences[index].trim());
      utterance.voice = voice;
      utterance.lang = voice?.lang || 'bg-BG';
      utterance.rate = 1;

      utterance.onend = () => {
        index++;
        speakNext();
      };

      utterance.onerror = () => {
        stopReading = true;
        indicator.style.display = "none";
      };

      speechSynthesis.speak(utterance);
    }

    isReading = true;
    speakNext();
  }

  readBtn.addEventListener("click", () => {
    if (isReading) return;

    const article = document.querySelector('.article-content');
    if (!article || voiceSelect.selectedIndex < 0) return;

    const text = article.innerText || article.textContent;
    const selectedVoice = voices[voiceSelect.value];

    speechSynthesis.cancel();
    speakSentences(text, selectedVoice);
  });

  stopBtn.addEventListener("click", () => {
    stopReading = true;
    speechSynthesis.cancel();
    indicator.style.display = "none";
  });
});
</script>
