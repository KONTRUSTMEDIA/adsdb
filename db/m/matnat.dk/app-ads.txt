        document.addEventListener("DOMContentLoaded", () => {
        const toTopBtn = document.querySelector("#to-top-btn");
        let isVisible = false;

        // Klik for at scrolle til top
        toTopBtn.addEventListener("click", () => {
            window.scrollTo({ top: 0, behavior: "smooth" });
        });

        // Scroll observer
        document.addEventListener("scroll", () => {
            if (window.scrollY > 200 && !isVisible) {

                toTopBtn.classList.add("opacity-100", "pointer-events-auto", "scale-100");
                isVisible = true;
            } else if (window.scrollY <= 200 && isVisible) {
                toTopBtn.classList.add("opacity-0", "pointer-events-none", "scale-75");
                toTopBtn.classList.remove("opacity-100", "pointer-events-auto", "scale-100");
                isVisible = false;
            }
        });
    });

</script>
