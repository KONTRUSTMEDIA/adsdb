        $(document).ready(function(){
            $('.accordion-button').click(function(){
                var contentId = $(this).data('content-id');
                var $content = $('#' + contentId);
                var $button = $(this);
                
                // 1. Cierra todos los contenidos que no sean el actual
                $('.accordion-content').each(function() {
                    // Obtiene el botón asociado al contenido
                    var otherButton = $(this).parent().find('.accordion-button');
                    
                    if (this.id !== contentId) {
                        // Cierra el contenido
                        $(this).css('max-height', 0);
                        // Remueve la clase 'active' del botón
                        otherButton.removeClass('active');
                    }
                });

                // 2. Despliega o cierra el contenido actual
                if (!$button.hasClass('active')) {
                    // Si el botón no está activo, lo abrimos:
                    var innerHeight = $content.find('.accordion-content-inner').prop('scrollHeight');
                    
                    // Abre el contenido, asegurando un espacio extra para el padding/margen
                    $content.css('max-height', innerHeight + 40 + 'px'); 
                    $button.addClass('active');
                } else {
                    // Si el botón ya está activo, lo cerramos:
                    $content.css('max-height', 0);
                    $button.removeClass('active');
                }
            });
        });
    </script>
