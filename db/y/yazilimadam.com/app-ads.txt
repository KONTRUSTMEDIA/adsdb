<script>
    (function () {
        const openBtnDesktop = document.getElementById('openQuoteBtn');
        const openBtnMobile  = document.getElementById('openQuoteMobileBtn');
        const closeBtn       = document.getElementById('closeQuoteBtn');
        const overlay        = document.getElementById('quoteOverlay');
        const modal          = document.getElementById('quoteModal');
        const form           = document.getElementById('quoteForm');
        const msgEl          = document.getElementById('quoteMessage');

        function openModal() {
            if (!modal) return;
            modal.classList.add('active');
            document.body.classList.add('quote-modal-open');

            msgEl.textContent = '';
            msgEl.className = 'quote-message';
            form.reset();

            document.querySelectorAll('.quote-option-btn').forEach(function (btn) {
                btn.classList.remove('active');
            });
        }

        function closeModal() {
            if (!modal) return;
            modal.classList.remove('active');
            document.body.classList.remove('quote-modal-open');
        }

        if (openBtnDesktop) {
            openBtnDesktop.addEventListener('click', function (e) {
                e.preventDefault();
                openModal();
            });
        }

        if (openBtnMobile) {
            openBtnMobile.addEventListener('click', function (e) {
                e.preventDefault();
                openModal();
            });
        }

        if (closeBtn) {
            closeBtn.addEventListener('click', closeModal);
        }
        if (overlay) {
            overlay.addEventListener('click', closeModal);
        }
        document.addEventListener('keydown', function (e) {
            if (e.key === 'Escape') {
                closeModal();
            }
        });

        const optionButtons = document.querySelectorAll('.quote-option-btn');
        optionButtons.forEach(function (btn) {
            btn.addEventListener('click', function () {
                const group = this.dataset.group;
                const value = this.dataset.value;

                optionButtons.forEach(function (b) {
                    if (b.dataset.group === group) {
                        b.classList.remove('active');
                    }
                });
                this.classList.add('active');

                if (group === 'ecommerce') {
                    document.getElementById('q_ecommerce').value = value;
                }
                if (group === 'heard') {
                    document.getElementById('q_heard_from').value = value;
                }
            });
        });

        if (form) {
            form.addEventListener('submit', function (e) {
                e.preventDefault();

                const name    = document.getElementById('q_name').value.trim();
                const phone   = document.getElementById('q_phone').value.trim();
                const email   = document.getElementById('q_email').value.trim();
                const message = document.getElementById('q_project').value.trim();

                if (!name || !phone || !message) {
                    msgEl.textContent = 'Lütfen Ad Soyad, Telefon ve Proje Detaylarınız alanlarını doldurunuz.';
                    msgEl.className = 'quote-message error';
                    return;
                }

                if (email && !/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email)) {
                    msgEl.textContent = 'Lütfen geçerli bir e-posta adresi giriniz veya bu alanı boş bırakınız.';
                    msgEl.className = 'quote-message error';
                    return;
                }

                msgEl.textContent = 'Gönderiliyor...';
                msgEl.className = 'quote-message';

                const formData = new FormData(form);

                fetch('send-quote.php', {
                    method: 'POST',
                    body: formData
                })
                    .then(function (response) { return response.json(); })
                    .then(function (data) {
                        if (data.success) {
                            msgEl.textContent = data.message || 'Formunuz başarıyla gönderildi.';
                            msgEl.className = 'quote-message success';
                            form.reset();
                            document.querySelectorAll('.quote-option-btn').forEach(function (btn) {
                                btn.classList.remove('active');
                            });
                        } else {
                            msgEl.textContent = data.message || 'Bir hata oluştu. Lütfen tekrar deneyin.';
                            msgEl.className = 'quote-message error';
                        }
                    })
                    .catch(function () {
                        msgEl.textContent = 'Sunucuya ulaşılamadı. Lütfen daha sonra tekrar deneyin.';
                        msgEl.className = 'quote-message error';
                    });
            });
        }
    })();
</script>
