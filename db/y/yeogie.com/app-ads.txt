            if (e.keyCode == 38) {

                focus = -1;

                chkKeyin = setTimeout(function() {
                    $("#KeywordAutoComplete").html('');

    var globalCursor = -1;
    var globalRuntime = 300;
    var globalHeight = 0;
    var globalAct = false;
    var globalObj = null;
    var globalTimer = 5000;

    var globalList = new Array();

    var GlobalPrev = function() {
        if (globalAct == false) {
            clearTimeout(globalObj);
            globalAct = true;
            globalCursor--;
            globalCursor = globalCursor < 0 ? (globalList.length - 1) : globalCursor;

            $('#global div.list li').animate({
                'top': globalHeight + 'px'
            }, globalRuntime, function() {
                $(this).remove();
            });
            var newList = globalList[globalCursor].clone();
            $('#global div.list ul').append(newList);
            newList.css({
                'top': (globalHeight * -1) + 'px'
            }).animate({
                'top': '0px'
            }, globalRuntime, function() {
                $('#global span.now').text(globalCursor + 1);
                globalAct = false;
                globalObj = setTimeout(GlobalNext, globalTimer);
            });
        };
    };

    var GlobalNext = function() {
        if (globalAct == false) {
            clearTimeout(globalObj);
            globalAct = true;
            globalCursor++;
            globalCursor = globalCursor >= globalList.length ? 0 : globalCursor;

            $('#global div.list li').animate({
                'top': (globalHeight * -1) + 'px'
            }, globalRuntime, function() {
                $(this).remove();
            });
            var newList = globalList[globalCursor].clone();
            $('#global div.list ul').append(newList);
            newList.css({
                'top': globalHeight + 'px'
            }).animate({
                'top': '0px'
            }, globalRuntime, function() {
                globalAct = false;
                globalObj = setTimeout(GlobalNext, globalTimer);
            });
        };
    };

    var GlobalInit = function() {
        $('#global div.list li').each(function() {
            globalList[globalList.length] = $(this).clone();
        });
        globalHeight = $('#global div.list').height();
        $('#global div.list li').remove();

        GlobalNext();
    };

    $(function() {
        GlobalInit();
        $('#global button.prev').click(GlobalPrev);
        $('#global button.next').click(GlobalNext);
    });

<script>
    var marketList = [];
    var marketCursor = 0;
    var marketObj = null;
    var marketRuntime = 300;
    var marketDelay = 4000;
    var MarketInit = function() {
        $.ajax({
            'type': 'post',
            // 'url':'/common/datas/market.list.html',
            'url': '/ygMarket/datas/product/recents.html',
            'dataType': 'json',
            'success': function(req) {
                if (req.ERROR) {
                    console.log(req.ERROR);
                } else {
                    marketList = req.LIST;
                    $(marketList.reverse()).each(function(key) {
                        if (key >= 3) return false;
                        marketCursor = marketList.length - key;

                        var newItem = MarketItem(this);
                        newItem.css({
                            'opacity': '0'
                        });
                        $('#marketList').prepend(newItem);
                        newItem.animate({
                            'opacity': '1'
                        }, marketRuntime);
                    });

                    marketObj = setTimeout(MarketItemAdd, marketDelay);
                };
            },
            'error': function() {
                UnLoading();
                console.log('제품정보를 불러오던 중 오류가 발생했습니다');
            }
        });
    };

    var MarketItem = function(data) {
        var newItem = $('<div>').addClass('item');
        var newAnchor = $('<a>').attr({
            'href': 'https://market.yeogie.com/article/' + data.CODE,
            'title': data.NAME,
            'target': '_blank'
        }).appendTo(newItem);

        $('<span>').addClass('thumb').css({
            'background-size': 'cover',
            'background-image': 'url(\'' + data.THUMB + '\')'
        }).appendTo(newAnchor);

        var newInfo = $('<span>').addClass('info').appendTo(newAnchor);
        $('<span>').addClass('divide').addClass(data.DIVIDE).text(data.DNAME).appendTo(newInfo);
        $('<span>').addClass('status').addClass(data.STATUS).text(data.SNAME).appendTo(newInfo);
        $('<span>').addClass('title').text(data.NAME).appendTo(newInfo);
        if (data.PRICE) $('<span>').addClass('price').appendTo(newInfo).html('<span class="number">' + AddCommas(data.PRICE) + '</span> <span class="unit">원</span>');
        else $('<span>').addClass('price').appendTo(newInfo).text('가격협의');

        return newItem;
    };

    var MarketItemAdd = function() {
        clearTimeout(marketObj);
        marketCursor--;
        if (marketCursor < 0) marketCursor = marketList.length - 1;

        var newItem = MarketItem(marketList[marketCursor]);
        newItem.css({
            'margin-top': '-107px'
        });
        $('#marketList').prepend(newItem);
        newItem.animate({
            'margin-top': '0'
        }, marketRuntime, function() {
            $('#marketList div.item:last-child').remove();
            marketObj = setTimeout(MarketItemAdd, marketDelay);
        });
    };

    $(function() {
        MarketInit();
    });
</script>
