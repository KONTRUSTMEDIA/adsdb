function render(data) {

    const el = document.getElementById('songs-list'); 

    let html = '';

    

    if(!data.length){ el.innerHTML='<p style="text-align:center; color:#666; margin-top:20px;">Tapılmadı...</p>'; return; }

    currentPlaylist = data;

    data.forEach((s, i) => {

        // Reklam (Hər 7 mahnıdan bir)

        if(i > 0 && i % 7 === 0) {

            html += `<div class="item ad-container" style="display:block; min-height:80px; padding:0; background:#fff;">${typeof adCodeHTML !== 'undefined' ? adCodeHTML : ''}</div>`;

        }

        

        const cover = s.cover_image && s.cover_image.length > 20 ? s.cover_image : defaultLogo;

        const flag = s.flag ? `<span class="flag-icon">${s.flag}</span>` : '';

        

        // --- BADGE MƏNTİQİ ---

        let badges = '';

        const now = new Date();

        const created = new Date(s.created_at);

        const diffHours = (now - created) / 36e5; // Saat fərqi

        

        // 1. YENİ (24 saat)

        if (diffHours < 24) badges += `<div class="badge-tag b-new">🔥 YENİ</div>`;

        

        // 2. TREND (1000+ Play)

        if (parseInt(s.plays) > 1000) badges += `<div class="badge-tag b-trend">⚡ TREND</div>`;

        

        // 3. SEVİMLİ (50+ Like - Rəqəmi dəyişə bilərsən)

        if (parseInt(s.likes) > 50) badges += `<div class="badge-tag b-love">❤️ SEVİMLİ</div>`;

        

        const badgesHTML = badges ? `<div class="badges-con">${badges}</div>` : '';

        

        html += `

        <div class="item" onclick="playFromIndex(${i})">

            ${badgesHTML}

            <img src="${cover}" loading="lazy" decoding="async" width="60" height="60">

            <div class="info">

                <h3>${s.title}</h3>

                <p>${flag}${s.artist}</p>

            </div>

            <div style="font-size:10px; color:#666; margin-right:5px; text-align:right;">

                <div style="margin-top:2px;"><i class="fas fa-heart"></i> ${s.likes}</div>

        </div>`;

    });

    

    el.innerHTML = html;

    // Mini player açılan kimi ("active" class-ı gələn kimi), bizim düyməni də açır.

    document.addEventListener("DOMContentLoaded", function() {

        const targetNode = document.getElementById('miniPlayer');

        const hintNode = document.getElementById('mini-dl-hint');

        if (targetNode && hintNode) {

            const config = { attributes: true, attributeFilter: ['class'] };

            const callback = function(mutationsList, observer) {

                for(const mutation of mutationsList) {

                    if (mutation.type === 'attributes' && mutation.attributeName === 'class') {

                        if (targetNode.classList.contains('active')) {

                            hintNode.classList.add('active');

                        } else {

                            hintNode.classList.remove('active');

                        }

                    }

                }

            };

            const observer = new MutationObserver(callback);

            observer.observe(targetNode, config);

        }

    });

function prepareUpload(files) { 

    const c = document.getElementById('upload-previews'); c.innerHTML=''; 

    

    Array.from(files).forEach(f => { 

        const d = document.createElement('div'); 

        d.style.padding='8px'; d.style.borderBottom='1px solid #333'; d.style.fontSize='13px';

        d.innerHTML = `<i class="fas fa-spinner fa-spin"></i> ${f.name} <span style="color:#888;">(Analiz edilir...)</span>`;

        c.appendChild(d);

        

        // JSMediaTags ilə MP3 daxilini oxu

        if(typeof jsmediatags !== 'undefined') {

            jsmediatags.read(f, {

                onSuccess: (tag) => {

                    const t = tag.tags;

                    let base64 = "";

                    if (t.picture) {

                        let data = t.picture.data;

                        let format = t.picture.format;

                        let str = "";

                        for (let i = 0; i < data.length; i++) str += String.fromCharCode(data[i]);

                        base64 = `data:${format};base64,${window.btoa(str)}`;

                    }

                    metaData[f.name] = { title: t.title||f.name.replace(/\.[^/.]+$/, ""), artist: t.artist||"Naməlum", coverData: base64 };

                    d.innerHTML = `<i class="fas fa-music" style="color:#F72585"></i> <b>${t.artist||'Naməlum'}</b> - ${t.title||f.name} <i class="fas fa-check" style="color:green; float:right;"></i>`;

                },

                onError: (e) => {

                    metaData[f.name] = { title: f.name, artist: "Naməlum", coverData: "" };

                    d.innerHTML = `<i class="fas fa-music"></i> ${f.name}`;

                }

            });

        } else {

             metaData[f.name] = { title: f.name, artist: "Naməlum", coverData: "" };

             d.innerHTML = `<i class="fas fa-music"></i> ${f.name}`;

        }

    });

}

function startUpload() { 

    const bar = document.getElementById('prog-bar'); 

    const cont = document.getElementById('prog-cont'); 

    const fileInput = document.getElementById('u-file');

    if(fileInput.files.length === 0) { alert("Mahnı seçilməyib!"); return; }

    cont.style.display='block'; 

    bar.style.width='0%'; bar.innerText='0%';

    

    const fd = new FormData(); 

    fd.append('action','upload'); 

    for(let f of fileInput.files) fd.append('musicFiles[]',f); 

    fd.append('metaData', JSON.stringify(metaData)); 

    fd.append('genre', document.getElementById('u-genre').value); 

    

    const xhr = new XMLHttpRequest(); 

    xhr.open('POST','',true); 

    

    // Yükləmə Faizi Hesablanması

    xhr.upload.onprogress = e => { 

        if (e.lengthComputable) {

            const percent = Math.round((e.loaded / e.total) * 100);

            bar.style.width = percent + '%';

            bar.innerText = percent + '%';

        }

    }; 

    

    xhr.onload = () => { 

        try {

            const r = JSON.parse(xhr.responseText);

            if(r.success) { alert("Uğurla Yükləndi!"); location.reload(); } 

            else { alert('Xəta: ' + r.error); }

        } catch(e) { alert("Server Xətası!"); }

    }; 

    xhr.send(fd); 

}

    let userLang = (navigator.language || navigator.userLanguage).split('-')[0];

    if (!langDB[userLang]) userLang = 'en';

    

    // Əgər Azərbaycan dilidirsə (az), heçnə dəyişmə (Default)

    if (userLang === 'az') return;

function updateTextByContent(oldText, newText) {

    // Bütün elementləri gəzirik

    const walker = document.createTreeWalker(document.body, NodeFilter.SHOW_TEXT, null, false);

    let node;

    while(node = walker.nextNode()) {

        if(node.nodeValue.trim() === oldText) {

            node.nodeValue = newText;

        }

    }

}

// --- AUDIO FIX ENGINE (High Quality + Booster) ---

let audioCtx, analyser, source, canvas, ctx;

let audioInitialized = false;

function initAudioVisualizer() {

    const audioEl = document.getElementById('audio');

    // 1. Audio Context yaradiriq (Əgər yoxdursa)

    if (!audioCtx) {

        // YÜKSƏK KEYFİYYƏT REJİMİ (LATENCY HINT)

        const AudioContext = window.AudioContext || window.webkitAudioContext;

        audioCtx = new AudioContext({

            latencyHint: 'playback', // Əsas düzəliş: Səsi hamarlayır və keyfiyyəti artırır

            sampleRate: 44100 // Standart CD keyfiyyəti

        });

        

        // 2. Vizualizator üçün qoşulma

        // CrossOrigin vacibdir, amma bəzi serverlərdə xəta verir. 

        audioEl.crossOrigin = "anonymous";

        source = audioCtx.createMediaElementSource(audioEl);

        analyser = audioCtx.createAnalyser();

        

        // 3. SƏS GÜCLƏNDİRİCİ (GAIN NODE) - Batıq səslər üçün

        const gainNode = audioCtx.createGain();

        gainNode.gain.value = 1.0; // Səs səviyyəsi (lazım olsa 1.2 edib artıra bilərsən)

        

        // 4. ZƏNCİRİ QURURUQ: Source -> Gain -> Analyser -> Destination

        source.connect(gainNode);

        gainNode.connect(analyser);

        analyser.connect(audioCtx.destination); 

        

        analyser.fftSize = 256;

        canvas = document.getElementById('rhythm-canvas');

        if(canvas) {

            ctx = canvas.getContext('2d');

            window.addEventListener('resize', resizeCanvas);

            resizeCanvas();

            animateVisualizer();

        }

    }

    // 4. Brauzer bloku qoyubsa, onu qırırıq (RESUME)

    if (audioCtx.state === 'suspended') {

        audioCtx.resume();

    }

    

    audioInitialized = true;

}

function resizeCanvas() {

    if(canvas) {

        canvas.width = window.innerWidth;

        canvas.height = window.innerHeight;

    }

}

function animateVisualizer() {

    if (!audioCtx || !analyser) return;

    requestAnimationFrame(animateVisualizer);

    

    const bufferLength = analyser.frequencyBinCount;

    const dataArray = new Uint8Array(bufferLength);

    analyser.getByteFrequencyData(dataArray);

    

    // Canvas təmizlə

    ctx.fillStyle = 'rgba(15, 17, 26, 0.2)';

    ctx.fillRect(0, 0, canvas.width, canvas.height);

    

    // Bass gücünü hesabla

    let bass = 0;

    for (let i = 0; i < 10; i++) { bass += dataArray[i]; }

    bass = bass / 10;

    

    const centerX = canvas.width / 2;

    const centerY = canvas.height / 2;

    const radius = 100 + (bass * 1.5); // Ritmə görə böyümə

    

    // Dairəni çək

    const gradient = ctx.createRadialGradient(centerX, centerY, radius * 0.1, centerX, centerY, radius);

    gradient.addColorStop(0, `rgba(247, 37, 133, ${bass/255})`);

    gradient.addColorStop(1, 'rgba(0,0,0,0)');

    

    ctx.beginPath();

    ctx.arc(centerX, centerY, radius, 0, 2 * Math.PI);

    ctx.fillStyle = gradient;

    ctx.fill();

}

        audioCtx.resume();

    }

});

function renderProfileList(songsToRender) {

    const list = document.getElementById('pp-list');

    list.innerHTML = '';

    if(songsToRender.length === 0) {

        list.innerHTML = '<p style="text-align:center; color:#666; margin-top:30px;">Bu kateqoriyada mahnı yoxdur.</p>';

        return;

    }

    songsToRender.forEach((s, index) => {

        // "YENİ" Məntiqi (24 Saat)

        const createdDate = new Date(s.created_at);

        const now = new Date();

        const diffHours = (now - createdDate) / (1000 * 60 * 60);

        

        let badgeHTML = '';

        if(diffHours < 24) {

            badgeHTML = '<div class="pp-badge-new">🔥 YENİ</div>';

function sortProfileSongs(type, tabElement) {

    // Tab dizaynını dəyiş

    document.querySelectorAll('.p-tab').forEach(t => t.classList.remove('active'));

    tabElement.classList.add('active');

    

    let sortedSongs = [...currentUserSongs]; // Kopya alırıq (orijinalı pozmamaq üçün)

    if (type === 'plays') {

        // Çoxdan aza dinləmə

        sortedSongs.sort((a, b) => b.plays - a.plays);

    } else if (type === 'likes') {

        // Çoxdan aza bəyənmə

        sortedSongs.sort((a, b) => b.likes - a.likes);

    } else if (type === 'trend') {

        // Trend: Həm yeni (son 7 gün), həm də çox dinlənən (>100)

        // Və ya sadəcə ən populyar ilk 10

        sortedSongs.sort((a, b) => b.plays - a.plays);

        sortedSongs = sortedSongs.slice(0, 10); // Top 10

    } else {

        // Hamısı (Default ID sırası)

        // Heç nə etmirik, olduğu kimi qalır

    }

    renderProfileList(sortedSongs);

}

function submitAuth(type) {

    const u = document.getElementById(type === 'login' ? 'l-user' : 'r-user').value;

    const p = document.getElementById(type === 'login' ? 'l-pass' : 'r-pass').value;

    

    if(!u || !p) { alert("Xanaları doldurun!"); return; }

    

    // Düyməni deaktiv et

    const btn = event.target;

    const oldText = btn.innerText;

    btn.innerText = "Gözləyin...";

    btn.disabled = true;

    const fd = new FormData();

    fd.append('action', type);

    fd.append('username', u);

    fd.append('password', p);

    fetch('', {method:'POST', body:fd})

    .then(r => r.text()) // Əvvəl text kimi al

    .then(text => {

        try {

            const d = JSON.parse(text);

            if(d.success) {

                location.reload(); // Uğurludursa səhifəni yenilə

            } else {

                alert(d.error || "Xəta baş verdi.");

                btn.innerText = oldText;

                btn.disabled = false;

            }

        } catch(e) {

            console.error(text);

            alert("Sistem xətası! Konsola baxın.");

            btn.innerText = oldText;

            btn.disabled = false;

        }

    });

}

    if(el) el.innerText = parseInt(el.innerText||0) + 1;

    // Paylaşım

    const text = currentSong.artist + " - " + currentSong.title;

    const url = window.location.origin + "?song=" + currentSong.id;

    

    if(navigator.share) {

        navigator.share({ title: text, text: text, url: url }).catch(e=>{});

    } else {

        prompt("Linki kopyala:", url); // Prompt modalı z-index sayəsində üstdə olacaq

    }

}

let lastCheckId = 0;

// Siyahıdakı ən son ID-ni tapan köməkçi funksiya
function updateLastId(list) {
    if(list && list.length > 0) {
        // Mövcud siyahıdan ən böyük ID-ni tapırıq
        const max = Math.max(...list.map(s => parseInt(s.id)));
        if(max > lastCheckId) lastCheckId = max;
    }
}

    if(typeof songs !== 'undefined') {
        updateLastId(songs);
    }
});
