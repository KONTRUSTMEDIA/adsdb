        // Mobile Menu Toggle
        const mobileMenuBtn = document.getElementById('mobileMenuBtn');
        const mobileMenuDrawer = document.getElementById('mobileMenuDrawer');
        const mobileMenuOverlay = document.getElementById('mobileMenuOverlay');
        const mobileMenuClose = document.getElementById('mobileMenuClose');

        function openMobileMenu() {
            mobileMenuDrawer.classList.add('active');
            mobileMenuOverlay.classList.add('active');
            document.body.style.overflow = 'hidden';
        }

        function closeMobileMenu() {
            mobileMenuDrawer.classList.remove('active');
            mobileMenuOverlay.classList.remove('active');
            document.body.style.overflow = '';
        }

        if (mobileMenuBtn) {
            mobileMenuBtn.addEventListener('click', openMobileMenu);
        }
        if (mobileMenuClose) {
            mobileMenuClose.addEventListener('click', closeMobileMenu);
        }
        if (mobileMenuOverlay) {
            mobileMenuOverlay.addEventListener('click', closeMobileMenu);
        }

        // Mobile Notification Toggle
        const mobileNotificationBtn = document.getElementById('mobileNotificationBtn');
        const mobileNotificationOverlay = document.getElementById('mobileNotificationOverlay');
        const mobileNotificationClose = document.getElementById('mobileNotificationClose');
        const welcomeNotification = document.getElementById('welcomeNotification');
        const notificationBadge = document.querySelector('.mobile-notification-badge');

        function openMobileNotification() {
            mobileNotificationOverlay.classList.add('active');
            document.body.style.overflow = 'hidden';

            // Bildirimi okundu olarak işaretle
            if (welcomeNotification) {
                setTimeout(() => {
                    welcomeNotification.classList.remove('unread');
                }, 300);
            }

            // Badge'i gizle
            if (notificationBadge) {
                setTimeout(() => {
                    notificationBadge.style.display = 'none';
                }, 300);
            }
        }

        function closeMobileNotification() {
            mobileNotificationOverlay.classList.remove('active');
            document.body.style.overflow = '';
        }

        if (mobileNotificationBtn) {
            mobileNotificationBtn.addEventListener('click', openMobileNotification);
        }
        if (mobileNotificationClose) {
            mobileNotificationClose.addEventListener('click', closeMobileNotification);
        }

        // ESC key to close
        document.addEventListener('keydown', function(e) {
            if (e.key === 'Escape') {
                closeMobileMenu();
                closeMobileNotification();
            }
        });

        // Mobile App Banner
        const mobileAppBanner = document.getElementById('mobileAppBanner');
        const mobileAppBannerClose = document.getElementById('mobileAppBannerClose');

        // Check if banner was dismissed
        const bannerDismissed = localStorage.getItem('mobileAppBannerDismissed');
        const isMobile = window.innerWidth < 768;

        if (!bannerDismissed && isMobile && mobileAppBanner) {
            // Show banner after 2 seconds
            setTimeout(() => {
                mobileAppBanner.classList.add('show');
            }, 2000);
        }

        // Close banner
        if (mobileAppBannerClose) {
            mobileAppBannerClose.addEventListener('click', function() {
                mobileAppBanner.classList.remove('show');
                localStorage.setItem('mobileAppBannerDismissed', 'true');
            });
        }

        // Also dismiss on app button click
        const appButtons = document.querySelectorAll('.mobile-app-btn');
        appButtons.forEach(btn => {
            btn.addEventListener('click', function() {
                mobileAppBanner.classList.remove('show');
                localStorage.setItem('mobileAppBannerDismissed', 'true');
            });
        });

        if(typeof Swiper==='undefined'){console.warn('Swiper not loaded');}else{
        const heroSlider = new Swiper('.hero-slider', {
            loop: true,
            autoplay: {
                delay: 5000,
                disableOnInteraction: false,
            },
            effect: 'fade',
            fadeEffect: {
                crossFade: true
            },
            navigation: {
                nextEl: '.swiper-button-next',
                prevEl: '.swiper-button-prev',
            },
            on: {
                init: function() {
                    updateSliderNumbers(this);
                },
                slideChange: function() {
                    updateSliderNumbers(this);
                }
            }
        });

        // Number item click
        document.querySelectorAll('.slider-number-item').forEach((item, index) => {
            item.addEventListener('click', () => {
                heroSlider.slideToLoop(index);
            });
        });

        // Update number indicators
        function updateSliderNumbers(swiper) {
            const currentIndex = swiper.realIndex;

            document.querySelectorAll('.slider-number-item').forEach((item, index) => {
                item.classList.remove('active', 'passed');
                if (index < currentIndex) {
                    item.classList.add('passed');
                } else if (index === currentIndex) {
                    item.classList.add('active');
                }
            });
        }

        } // end Swiper check

        // Mobile Navigation Toggle
        const navToggle = document.getElementById('navToggle');
        const navList = document.getElementById('navList');

        if (navToggle && navList) {
            navToggle.addEventListener('click', function() {
                navList.classList.toggle('active');
                const icon = this.querySelector('i');
                icon.classList.toggle('fa-bars');
                icon.classList.toggle('fa-times');
            });
        }
    </script>

</body>
</html>
