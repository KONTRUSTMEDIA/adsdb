          isOpen = !isOpen;
          
          if(isOpen) {
              // Open menu
              collapseMenu.classList.add('show');
              collapseMenu.classList.add("fff");
              menuBtn.classList.add('active');
              nav.classList.add('active-radius');

          } else {
              // Close menu
              collapseMenu.classList.remove("fff");
              collapseMenu.classList.remove('show');
              menuBtn.classList.remove('active');
              nav.classList.remove('active-radius');
          }
      });

      // Close menu when clicking outside
      document.addEventListener('click', function(event) {
          const isClickInsideMenu = collapseMenu.contains(event.target);
          const isClickOnButton = menuBtn.contains(event.target);
          
          if (!isClickInsideMenu && !isClickOnButton && isOpen) {
              collapseMenu.classList.remove('show');
              menuBtn.classList.remove('active');
              nav.classList.remove('active-radius');
              isOpen = false;
          }
      });
  </script>
