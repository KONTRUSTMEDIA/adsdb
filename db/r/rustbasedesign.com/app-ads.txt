<script>
moment.locale("en");
function flask_moment_render(elem) {{
    const timestamp = moment(elem.dataset.timestamp);
    const func = elem.dataset.function;
    const format = elem.dataset.format;
    const timestamp2 = elem.dataset.timestamp2;
    const no_suffix = elem.dataset.nosuffix;
    const units = elem.dataset.units;
    let args = [];
    if (format)
        args.push(format);
    if (timestamp2)
        args.push(moment(timestamp2));
    if (no_suffix)
        args.push(no_suffix);
    if (units)
        args.push(units);
    elem.textContent = timestamp[func].apply(timestamp, args);
    elem.classList.remove('flask-moment');
    elem.style.display = "";
}}
function flask_moment_render_all() {{
    const moments = document.querySelectorAll('.flask-moment');
    moments.forEach(function(moment) {{
        flask_moment_render(moment);
        const refresh = moment.dataset.refresh;
        if (refresh && refresh > 0) {{
            (function(elem, interval) {{
                setInterval(function() {{
                    flask_moment_render(elem);
                }}, interval);
            }})(moment, refresh);
        }}
    }})
}}
document.addEventListener("DOMContentLoaded", flask_moment_render_all);
</script>

<script>
document.addEventListener("DOMContentLoaded", () => {
    const toggler = document.getElementById("hamburger");
    const navbar = document.getElementById("navbarSupportedContent");

    if (!navbar || !toggler) {
        console.error("Navbar or Toggler element not found!");
        return;
    }

    toggler.addEventListener("click", () => {
        const expanded = toggler.getAttribute("aria-expanded") === "true";

        if (expanded) {
            // Smooth collapse
            navbar.style.height = `${navbar.scrollHeight}px`; // Set height for animation
            setTimeout(() => {
                navbar.style.height = "0"; // Collapse to 0 height
                navbar.classList.remove("show");
                toggler.setAttribute("aria-expanded", "false");
            }, 10); // Small delay to ensure transition starts
        } else {
            // Smooth expand
            navbar.style.height = "0"; // Start from collapsed height
            navbar.classList.add("show");
            toggler.setAttribute("aria-expanded", "true");
            setTimeout(() => {
                navbar.style.height = `${navbar.scrollHeight}px`; // Expand to full height
            }, 10); // Small delay to ensure transition starts
        }

        navbar.addEventListener(
            "transitionend",
            () => {
                if (!navbar.classList.contains("show")) {
                    navbar.style.height = ""; // Reset height after collapse
                }
            },
            { once: true } // Only trigger once per transition
        );
    });
});


</script>
