In addition to the readability text tools above, we have a separate readability calculator called, the <a class="regular-links-inherit" href="https://readabilityformulas.com/calculator-fry-graph-readability-formula.php">Fry Graph</a> and <a class="regular-links-inherit" href="https://readabilityformulas.com/calculator-raygor-graph-readability-formula.php">Raygor Graph Text Tool</a>, which will score a sample of your text and visually plot the grade level onto each graph.
<br><br> 
The Fry readability formula measures English texts to provide you with a reading/grade level based on two criteria: 1) the average number of sentences, and 2) syllables per hundred words. We plot these averages onto The Fry Graph to visually show you the grade level of your readers.
<br><br> 
Similar to The Fry Graph, <b>The Raygor readability formula</b> also measures text based on the average number of sentences and letters per hundred words. We plot these two averages on the Raygor Graph to visually show you the reading/grade level of your readers. The Raygor Graph is largely used in secondary education to assess teaching materials and books, whereas <b>The Fry Graph</b> is commonly used on a broader range of texts.

<script>
let slideIndex = 0;
const slides = document.querySelectorAll(".slide");

const slideTrack = document.querySelector(".slide-track");
const totalSlides = slides.length; // This will automatically count the number of slides
const leftArrow = document.querySelector(".left-arrow");
const rightArrow = document.querySelector(".right-arrow");


function restartAnimation(element) {
    element.style.animation = 'none'; // Remove the animation
    element.offsetHeight; // Force a reflow
    element.style.animation = ''; // Re-add the animation by removing the inline style
}

const pagination = document.querySelector(".pagination");

// Function to update the active pagination dot
function updatePagination() {
    const activeDot = document.querySelector(".pagination-dot.active");
    if (activeDot) {
        activeDot.classList.remove("active");
    }
    pagination.children[slideIndex].classList.add("active");
}

// Add pagination dots dynamically based on the number of slides
for (let i = 0; i < totalSlides; i++) {
    const dot = document.createElement("div");
    dot.classList.add("pagination-dot");
    dot.addEventListener('click', () => {
        slideIndex = i;
        moveSlide();
        updatePagination();
    });
    pagination.appendChild(dot);
}

// Initially, mark the first dot as active
updatePagination();



function moveSlide() {
    slideTrack.style.transform = 'translateX(' + (-slideIndex * 100) + '%)';

    // Restart the animation for the text container of the current slide
    const currentTextContainer = slides[slideIndex].querySelector('.text-container');
    restartAnimation(currentTextContainer);

    updatePagination();
}

leftArrow.addEventListener('click', () => {
    slideIndex = (slideIndex <= 0) ? totalSlides - 1 : slideIndex - 1;
    moveSlide();
});

rightArrow.addEventListener('click', () => {
    slideIndex = (slideIndex >= totalSlides - 1) ? 0 : slideIndex + 1;
    moveSlide();
});

// Autoplay slider
setInterval(() => {
    slideIndex = (slideIndex >= totalSlides - 1) ? 0 : slideIndex + 1;
    moveSlide();
    updatePagination();
}, 10000);


</script>
