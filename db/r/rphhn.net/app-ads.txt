  function pickQueryFields(url) {
    const targetKeys = ['num_words', 'sort', 'state'];
    const params = new URL(url || window.location.href).searchParams;
    const result = {};

    targetKeys.forEach(key => {
      if (params.has(key)) {
        result[key] = params.get(key); // 只取第一个值（与 params.has 保持一致）
      }
    });

    return result;
  }
