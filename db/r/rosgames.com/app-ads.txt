const isApp = window.navigator.standalone || window.matchMedia('(display-mode: standalone)').matches;

// 3. 逻辑控制函数
function closePWABanner() {
    const banner = document.getElementById('pwa-banner');
    if (banner) {
        banner.classList.remove('is-mobile-active');
        sessionStorage.setItem('pwa_banner_closed', 'true');
    }
}

function installPWA() {
    if (!deferredPWAEvent) return;
    deferredPWAEvent.prompt();
    deferredPWAEvent.userChoice.then((choice) => {
        if (choice.outcome === 'accepted') closePWABanner();
        deferredPWAEvent = null;
    });
}

// 4. 事件监听核心
window.addEventListener('load', () => {
    const banner = document.getElementById('pwa-banner');
    if (!banner || !isMobile() || isApp) return; // 基础拦截：无元素、非移动端、已安装
    
    // 检查本次会话是否关闭过
    if (sessionStorage.getItem('pwa_banner_closed') === 'true') return;

    if (isIos) {
        // iOS 逻辑：直接显示说明文案
        banner.classList.add('is-mobile-active');
        const iosTxt = document.getElementById('pwa-ios-txt');
        if (iosTxt) iosTxt.classList.remove('d-none');
    } else {
        // Android/Chrome 逻辑：等待系统触发安装提示
        window.addEventListener('beforeinstallprompt', (e) => {
            e.preventDefault();
            deferredPWAEvent = e;
            banner.classList.add('is-mobile-active');
            
            const andTxt = document.getElementById('pwa-android-txt');
            const btn = document.getElementById('pwa-install-btn');
            if (andTxt) andTxt.classList.remove('d-none');
            if (btn) btn.classList.remove('d-none');
        });
    }
});
