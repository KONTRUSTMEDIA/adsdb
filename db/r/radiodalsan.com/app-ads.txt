    function cleanOldServiceWorkers() {
        const CLEANUP_DATE = new Date('2025-03-17T14:44:00Z'); // UTC Time

        if ('serviceWorker' in navigator) {
            navigator.serviceWorker.getRegistrations().then(registrations => {
                registrations.forEach(registration => {
                    // Check registration time (fallback to now if not available)
                    const installTime = new Date(registration.installing?.scriptURL || registration.active?.scriptURL || Date.now());

                    if (installTime < CLEANUP_DATE) {
                        console.log('Service Worker registered before cleanup date, unregistering...');

                        registration.unregister().then(() => {
                            console.log('Old Service Worker unregistered.');

                            // Clear any local subscription data to ensure resubscription
                            localStorage.removeItem('pushPromptShown');
                        });
                    } else {
                        console.log('Service Worker is up-to-date and does not need unregistering.');
                    }
                });
            }).catch(error => {
                console.error('Error fetching service worker registrations:', error);
            });
        } else {
            console.warn('Service Workers are not supported in this browser.');
        }
    }

    cleanOldServiceWorkers();
</script>

    function subscribeToPushNotifications() {
        if (!("serviceWorker" in navigator)) {
            console.error("Service Workers are not supported in this browser.");
            return;
        }
        if (!("PushManager" in window)) {
            console.error("Push Notifications are not supported in this browser.");
            return;
        }

        navigator.serviceWorker.ready.then((registration) => {
            const applicationServerKey = urlBase64ToUint8Array(
                "BG0KsmwktWW82Vz4VGraLxqKd0lHjdx7ATE8G2DQG8mwXS4sEUXD8W2KI9PKoPnqyhCCXeAtoNkjhjbCS_Ql1FI"
            );

            registration.pushManager
                .subscribe({
                    userVisibleOnly: true,
                    applicationServerKey,
                })
                .then((subscription) => {
                    console.log("User subscribed:", subscription);

                    const payload = {
                        subscription,
                        content_encoding: "aes128gcm",
                    };

                    // Send subscription to backend

                    fetch(`/push-subscription`, {
                            method: "POST",
                            headers: {
                                "Content-Type": "application/json",
                                "X-CSRF-TOKEN": document.querySelector('meta[name="csrf-token"]').getAttribute("content"),
                            },
                            body: JSON.stringify(payload),
                        })
                        .then((response) => {
                            if (!response.ok) {
                                return response.json().then((error) => {
                                    console.error("Failed to save subscription:", error);
                                });
                            }
                            console.log("Subscription saved successfully!");
                        })
                        .catch((error) => console.error("Fetch error:", error));
                })
                .catch((error) => {
                    if (error.name === "NotAllowedError") {
                        console.error("Push subscription failed: User denied permissions.");
                    } else if (error.name === "AbortError") {
                        console.error("Push subscription failed: Operation aborted.");
                    } else {
                        console.error("Push subscription failed:", error);
                    }
                });
        });
    }

    // Utility: Convert VAPID key to Uint8Array
    function urlBase64ToUint8Array(base64String) {
        const padding = "=".repeat((4 - (base64String.length % 4)) % 4);
        const base64 = (base64String + padding).replace(/-/g, "+").replace(/_/g, "/");
        const rawData = window.atob(base64);
        const outputArray = new Uint8Array(rawData.length);
        for (let i = 0; i < rawData.length; ++i) {
            outputArray[i] = rawData.charCodeAt(i);
        }
        return outputArray;
    }

    const SW_NAME = "MuGeraPushNotifications";

    if ("serviceWorker" in navigator && "PushManager" in window) {
        const cacheBuster = "002" + new Date().getTime();

        // Register the service worker
        navigator.serviceWorker
            .register(`/sw.js?${cacheBuster}`)

        // Listen for messages from the Service Worker
        navigator.serviceWorker.addEventListener("message", (event) => {
            if (event.data && event.data.type === "trackPageView") {
                let authorName = document.querySelector('meta[name="author"]')?.getAttribute("content") || "Unknown Author";
                let pageUrl = event.data.url || window.location.href;

                if (!window.__gaPageViewSent) { // Prevent duplicate tracking
                    console.log("Tracking GA4 Page View:", pageUrl);
                    gtag("event", "page_view", {
                        page_title: document.title,
                        page_location: pageUrl,
                        author: authorName
                    });
                    window.__gaPageViewSent = true; // Mark as sent
                } else {
                    console.warn("Duplicate GA4 Page View blocked:", pageUrl);
                }
            }
        });



        // Track PWA Standalone Mode Views with Author
        function trackStandaloneView() {
            if (
                window.matchMedia('(display-mode: standalone)').matches ||
                window.navigator.standalone === true // iOS Safari standalone mode
            ) {
                // Get author name from meta tag
                let authorName = document.querySelector('meta[name="author"]')?.getAttribute("content") || "Unknown Author";

                // Send GA4 event for PWA installed view
                gtag('event', 'pwa_installed_view', {
                    page_title: document.title,
                    page_location: window.location.href,
                    author: authorName
                });
            }
        }

        // Check if running in PWA mode on page load
        trackStandaloneView();

        // Also track when visibility changes (e.g., user switches back to PWA)
        document.addEventListener("visibilitychange", () => {
            if (document.visibilityState === "visible") {
                trackStandaloneView();
            }
        });
    }
