                    if( a_result.error === "expired" ) {
                        $(".myLoader").remove();
                        load_overlay('account');
                    } else {
                        $("#e_"+att_name).show();
                        $(".loader_"+att_name).remove(); 
                    }
                    // Daten konnten nicht gespeichert werden.
                }
            }
        });        
    }
        
});

function saveChanges(att_id, att_name, att_value, att_value_old) {
    $('#c_'+att_name).hide();
    if( att_value == att_value_old ) {
        $('#s_'+att_name).hide();
    } else {
        $("#"+att_name).attr("disabled", 'disabled');
        $("#s_"+att_name).removeClass('glyphicon-floppy-disk');
        $("#s_"+att_name).html('<div class="myInlineLoader ball-clip-rotate"><div></div></div>');
        var content = {"att_id":att_id, "att_name":att_name,"att_value":att_value};
        $.ajax({
            type: "POST",
            url: 'https://www.runme.de/call/change/',
            data: JSON.stringify({ content:content }),
            contentType: "application/json; charset=utf-8",
            success: function(result){
                var a_result = jQuery.parseJSON(result);
                if( a_result.success == true ) {
                    $(".myInlineLoader").remove();
                    $('#s_'+att_name).addClass('glyphicon-ok');
                    $('#s_'+att_name).fadeOut(1500);
                    $("#"+att_name).removeAttr("disabled");
                    if( att_name==='email' ) {
                        $("#email").hide();
                        $(".code-form").val('');
                        $("#email_verify").show();
                        $("#code_1").focus();
                    }
                } else {
                    if( a_result.error === "expired" ) {
                        $(".myInlineLoader").remove();
                        load_overlay('account');
                    } else {
                        $("#e_"+att_name).show();
                        $(".myInlineLoader").remove();
                        $('#s_'+att_name).hide();
                        $("#"+att_name).removeAttr("disabled");
                    }
                }
            }
        });
    }
}



function changeNotepadCounter( add ) {
    var count_old = parseInt($('#count_my').val());
    if( add ) {
        var count_new = count_old + 1;
    } else {
        var count_new = count_old - 1;
    }
    $('#count_my').val(count_new);
    $('#count_my_display').html(count_new);
}

function preValidateCode() {
    var code_complete = '';
    $(".code-form").each(function( index ) {
        var loop_value = parseInt( $(this).val() );
        if( Number.isInteger(loop_value) ) {
            code_complete = code_complete + '' + loop_value;
        };
    }); 
    // console.log(code_complete);
    var code_length = code_complete.length;
    if( code_length != 5 ) {
        // console.log("nicht gültig");
        $('#code_error_validation').show();
    } else {
        // console.log("5 gültige nummern");
        $("#validate_code").hide()
        $(".loader").show()

        $('#code_error_validation').hide();
        $(".save-button").prop('disabled', false);
        validateCode( code_complete )
        
    }
}
function validateCode( code_user ) {
    $('.code-form').prop("disabled",true);
    var email_user = $('#email-old').val();
    var content =   {
                        "email" : email_user,
                        "code" : code_user
                    };
    $.ajax({
        type: "POST",
        url: 'https://www.runme.de/call/change_email/d18354cca8f13a1c3cde5089441f9e17/',
        data: JSON.stringify({ content:content }),
        contentType: "application/json; charset=utf-8",
        success: function(result){  
            var a_result = jQuery.parseJSON(result);
            if( a_result.success == true ) {
                $('#email').val(a_result.email);
                $('#email').attr("d-old",a_result.email);
                $('.code-form').prop("disabled",false);
                $('#email_verify').hide();
                $('#email').show();
            } else {
                $('#code_error_validation').show();
                $('.code-form').val('');
                $('.code-form').prop("disabled",false);
                $('#code_1').focus();

            }
        }
    });
}

function buildPopoverUrl() {
    var splitUri = window.location.href.split('?');
    var parameterString = (splitUri.length - 1 === 1) ? splitUri[1] : "";

    var result = 'https://www.runme.de/call/popover/';

    return (parameterString !== "") ? result + '?' + parameterString : result + '/';
}
