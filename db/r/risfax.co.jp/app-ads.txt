			console.log('Slick function restored');
		}

		console.log('$.fn.slick:', typeof $.fn.slick);

		if ($('.bnr-slider').length > 0 && typeof $.fn.slick === 'function') {
			// スライド数を確認
			var slideCount = $('.bnr-slider .bnr-card').length;
			console.log('Current slide count:', slideCount);

			// スライドが0の場合は矢印を非表示
			if (slideCount === 0) {
				$('.bnr-slider-arrow').hide();
				console.log('No slides, hiding arrows');
				return;
			}

			try {
				$('.bnr-slider').slick({
					arrows: true,
					appendArrows: $('.bnr-slider-arrow'),
					prevArrow: '<div class="btn-arrow prev-arrow"></div>',
					nextArrow: '<div class="btn-arrow next-arrow"></div>',
					dots: false,
					autoplay: true,
					autoplaySpeed: 4000,
					infinite: true,
					speed: 1000,
					pauseOnFocus: false,
					pauseOnHover: false,
					pauseOnVisibility: false,
					swipe: true,
					swipeToSlide: true,
					centerMode: true,
					slidesToShow: 1,
					slidesToScroll: 1,
					variableWidth: true,
				});

				$('.bnr-slider').on('afterChange', function(event, slick, currentSlide) {
					if (slick.options.autoplay && !slick.autoPlayTimer) {
						console.log('Banner slider: Resuming autoplay after interaction');
						slick.slickPlay();
					}
				});

				$('.bnr-slider').on('touchend swipe', function() {
					var slickInstance = $(this).slick('getSlick');
					if (slickInstance.options.autoplay) {
						console.log('Banner slider: Resuming autoplay after touch/swipe');
						setTimeout(function() {
							slickInstance.slickPlay();
						}, 100);
					}
				});

				console.log('Slick initialized successfully');
			} catch(e) {
				console.error('Error initializing slick:', e);
			}
		} else {
			console.error('Slider not found or slick not available');
		}
	});
