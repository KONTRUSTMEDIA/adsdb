(function() {
    const CONFIG = {
        title: "Support by Shopping",
        subtitle: "Verified Amazon Merchandise",
        itemsPerBox: 2,
        selector: ".manga-ad-spot"
    };

    const path = window.location.pathname;
    const match = path.match(/\/chapter\/(.*?)-chapter-/);
    const slug = match ? match[1] : 'default';

    fetch(`/fetch-ads.php?series=${slug}`)
        .then(res => res.json())
        .then(data => {
            if (!data || data.length === 0) return;

            // This finds every div you placed in the body
            const targetSlots = document.querySelectorAll(CONFIG.selector);
            let pointer = 0;

            targetSlots.forEach(slot => {
                const slice = data.slice(pointer, pointer + CONFIG.itemsPerBox);
                if (slice.length < CONFIG.itemsPerBox) return;

                let html = `
                    <div class="manga-merch-box">
                        <div class="manga-merch-header">
                            <b>${CONFIG.title}</b>
                            <span>${CONFIG.subtitle}</span>
                        </div>`;

                slice.forEach(item => {
                    html += `
                        <div class="manga-merch-item">
                            <a href="${item.link}" target="_blank" rel="nofollow">
                                <img src="${item.image}" alt="${item.title}" loading="lazy">
                            </a>
                        </div>`;
                });

                html += `</div>`;
                slot.innerHTML = html;
                pointer += CONFIG.itemsPerBox;
            });
        })
        .catch(() => {});
})();

<script>
(function() {
    const STORAGE_KEY = 'manga_history_v1';
    const path = window.location.pathname;

     function sanitize(str) {
        const temp = document.createElement('div');
        temp.textContent = str;
        return temp.innerHTML;
    }

     const match = path.match(/\/chapter\/(.*?)-chapter-/);
    if (match) {
        const slug = match[1];
        const cleanName = slug.split('-').map(word => word.charAt(0).toUpperCase() + word.slice(1)).join(' ');
        const chapterTitle = document.title.split('|')[0].trim();

        const currentManga = {
            name: cleanName,
            chapter: chapterTitle,
            url: window.location.href
        };

        let history = [];
        try {
            const data = localStorage.getItem(STORAGE_KEY);
            history = data ? JSON.parse(data) : [];
            if (!Array.isArray(history)) history = []; // Poison Protection
        } catch (e) { history = []; }

        history = history.filter(item => item.name !== currentManga.name);
        history.unshift(currentManga);
        localStorage.setItem(STORAGE_KEY, JSON.stringify(history.slice(0, 10)));
    }

     const sidebar = document.getElementById('manga-history-sidebar');
    if (sidebar) {
        let historyData = [];
        try {
            historyData = JSON.parse(localStorage.getItem(STORAGE_KEY) || '[]');
        } catch (e) { historyData = []; }

        const currentPath = window.location.href;
        const displayList = historyData.filter(item => item.url !== currentPath).slice(0, 5);

        if (displayList.length > 0) {
             let html = '<div class="mb-3 rounded bg-bg-secondary related">' +
                       '<div class="p-3 border-b border-border-primary flex justify-between items-center">' +
                       '<div class="text-lg font-black flex items-center gap-2">' +
                       '<svg width="18" height="18" fill="none" stroke="currentColor" stroke-width="3" viewBox="0 0 24 24"><path d="M12 8v4l3 3m6-3a9 9 0 11-18 0 9 9 0 0118 0z" stroke-linecap="round" stroke-linejoin="round"/></svg>' +
                       'Last Read</div>' +
                       '<button onclick="clearMangaHistory()" class="text-[10px] uppercase font-bold text-red-500 hover:text-red-400">Clear</button>' +
                       '</div>' +
                       '<div class="grid grid-cols gap-3 p-3">';

            displayList.forEach(function(item) {
                 const safeName = sanitize(item.name);
                const safeChapter = sanitize(item.chapter);

                html += '<div class="pb-2 border-b border-border-primary last:border-0 last:pb-0">' +
                        '<a href="' + item.url + '" class="group">' +
                        '<div class="font-bold text-sm group-hover:text-primary transition-colors">' + safeName + '</div>' +
                        '<div class="text-xs text-text-muted mt-1">' + safeChapter + '</div>' +
                        '</a></div>';
            });

            html += '</div></div>';
            sidebar.innerHTML = html;
        }
    }
})();

 function clearMangaHistory() {
    localStorage.removeItem('manga_history_v1');
    const sidebar = document.getElementById('manga-history-sidebar');
    if (sidebar) sidebar.style.display = 'none';
}
