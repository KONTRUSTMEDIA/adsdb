	$m("#modal_trigger").leanModal({top : 200, overlay : 0.6, closeButton: ".modal_close" });

	$m(function(){
		// Calling Login Form
		$m("#login_form").click(function(){
			$m(".social_login").hide();
			$m(".user_login").show();
			return false;
		});

		// Calling Register Form
		$m("#register_form").click(function(){
			$m(".social_login").hide();
			$m(".user_register").show();
			$m(".header_title").text('Register');
			return false;
		});

		// Going back to Social Forms
		$m(".back_btn").click(function(){
			$m(".user_login").hide();
			$m(".user_register").hide();
			$m(".social_login").show();
			$m(".header_title").text('Get my Printable Resume');
			return false;
		});

	})


$j = jQuery.noConflict();
function periodical() {
	$j('.icon1').effect('shake', { times:3 }, 200);
}
$j(document).ready(function() {
	$j('.icon1').hide().css('display','').fadeIn(600);
	var shake = setInterval(periodical, 5000);
	
	/* updated: click clear */
	$j('.icon1').click(function() {
		clearInterval(shake);
	})	
});
