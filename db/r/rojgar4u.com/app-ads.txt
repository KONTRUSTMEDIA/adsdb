<script>
// === STARTUP SOUND ===
// Autoplay policy ke karan ho sakta hai ki sound hamesha play na ho
window.addEventListener('load', () => {
    const sound = document.getElementById('startSound');
    if (sound) {
        sound.volume = 0.4; 
        sound.play().catch(() => {
            console.log('Autoplay blocked');
        }); 
    }
});

// === REDIRECT CONTROL (FIXED) ===
// Tagline ka animation 3.5s par shuru hota hai aur 0.8s lagta hai. 
// Yani poora animation 4.3s par khatam hota hai.
const ANIMATION_END_TIME = 4300; 

// ** FIX 2: Text ko 3 seconds tak dikhane ke liye delay badhaya **
const ADDITIONAL_VISIBLE_TIME = 1000; // 3 seconds extra
const FADE_OUT_DURATION = 600;

// Total delay before redirect = Animation End + Extra Visible Time + Fade Out Time
const TOTAL_REDIRECT_DELAY = ANIMATION_END_TIME + ADDITIONAL_VISIBLE_TIME + FADE_OUT_DURATION; // Lagbhag 7.9 seconds

// 1. Fade out the preloader
setTimeout(() => {
    const container = document.getElementById('preloader-container');
    if (container) container.style.opacity = 0;
}, ANIMATION_END_TIME + ADDITIONAL_VISIBLE_TIME); // Extra time ke baad fade out shuru karein

// 2. मेन वेबसाइट पर रीडायरेक्ट करें
setTimeout(() => { 
    const targetUrl = '/';
    
    // ** FIX 3: Redirection ke liye ek aur majboot tarika **
    if (targetUrl && targetUrl !== 'undefined') {
         window.location.replace(targetUrl); 
    } else {
        // Fallback to home page if target URL is somehow missing
        window.location.replace('/');
    }
    
}, TOTAL_REDIRECT_DELAY); 
</script>
