document.addEventListener("DOMContentLoaded", function () {
  if (window.location.pathname !== "/" && window.location.pathname !== "/home") {
    return; 
  }
  const overlay = document.getElementById("customModalOverlay");
  const closeButtons = [
    document.getElementById("customCloseModal"),
    document.getElementById("customCloseModalFooter")
  ];

  const today = new Date().toDateString(); 
  const maxShowsPerDay = 3; 

  let modalData = localStorage.getItem("modalData");
  modalData = modalData ? JSON.parse(modalData) : { date: today, count: 0 };

  if (modalData.date !== today) {
    modalData = { date: today, count: 0 };
  }

  if (modalData.count < maxShowsPerDay) {
    overlay.style.display = "flex";
    modalData.count += 1;
    localStorage.setItem("modalData", JSON.stringify(modalData));
  }

  closeButtons.forEach(btn => {
    btn.addEventListener("click", () => {
      overlay.style.display = "none";
    });
  });

  overlay.addEventListener("click", e => {
    if (e.target === overlay) overlay.style.display = "none";
  });
});
</script>

<script>
document.addEventListener('DOMContentLoaded', function() {
    const track = document.getElementById('eventsCarousel');
    const prevBtn = document.getElementById('prevBtn');
    const nextBtn = document.getElementById('nextBtn');
    
    if (!track || !prevBtn || !nextBtn) return;
    
    const items = track.querySelectorAll('.event-carousel-item');
    const totalItems = items.length;
    
    let currentIndex = 0;
    let itemsPerView = 3;
    
    // Calcular itens por visualização baseado no tamanho da tela
    function updateItemsPerView() {
        const width = window.innerWidth;
        if (width < 768) {
            itemsPerView = 1;
        } else if (width < 992) {
            itemsPerView = 2;
        } else {
            itemsPerView = 3;
        }
    }
    
    function updateCarousel() {
        const itemWidth = items[0].offsetWidth;
        const gap = 20;
        const offset = -(currentIndex * (itemWidth + gap));
        track.style.transform = `translateX(${offset}px)`;
    }
    
    function moveNext() {
        currentIndex++;
        if (currentIndex > totalItems - itemsPerView) {
            currentIndex = 0;
        }
        updateCarousel();
    }
    
    function movePrev() {
        currentIndex--;
        if (currentIndex < 0) {
            currentIndex = Math.max(0, totalItems - itemsPerView);
        }
        updateCarousel();
    }
    
    // Event listeners
    nextBtn.addEventListener('click', moveNext);
    prevBtn.addEventListener('click', movePrev);
    
    // Atualizar em resize
    let resizeTimer;
    window.addEventListener('resize', function() {
        clearTimeout(resizeTimer);
        resizeTimer = setTimeout(function() {
            updateItemsPerView();
            if (currentIndex > totalItems - itemsPerView) {
                currentIndex = Math.max(0, totalItems - itemsPerView);
            }
            updateCarousel();
        }, 250);
    });
    
    // Suporte para teclado (acessibilidade)
    document.addEventListener('keydown', function(e) {
        if (e.key === 'ArrowLeft') movePrev();
        if (e.key === 'ArrowRight') moveNext();
    });
    
    // Inicializar
    updateItemsPerView();
    updateCarousel();
});
</script>

        function calculateSlides() {
            if (larguraTela <= 425) {
                totalSlides = totalItems; // 1 item por slide no mobile
            } else {
                totalSlides = Math.ceil(totalItems / 5); // 5 itens por slide no desktop
            }

        calculateSlides();
        
        window.addEventListener('resize', () => {
            larguraTela = window.innerWidth;
            calculateSlides();
            
            // Recalcular o slide atual baseado no novo modo
            const items = document.querySelectorAll('.slider-item.agreement');
            const itemsPerSlide = larguraTela <= 425 ? 1 : 5;
            const maxSlide = Math.ceil(items.length / itemsPerSlide) - 1;
            
            // Ajustar currentSlide para o novo modo se necessário
            if (currentSlide > maxSlide) {
                currentSlide = 0;
            }
            
            // Forçar recalculo da posição
            setTimeout(() => {
                updateMargin();
            }, 100);
        });

        function goPrev() {
            const items = document.querySelectorAll('.slider-item.agreement');
            const itemsPerSlide = larguraTela <= 425 ? 1 : 5;
            const maxSlide = Math.ceil(items.length / itemsPerSlide) - 1;
            
            currentSlide--;
            if (currentSlide < 0) {
                currentSlide = maxSlide;
            }
            updateMargin();
        }

        function goNext() {
            const items = document.querySelectorAll('.slider-item.agreement');
            const itemsPerSlide = larguraTela <= 425 ? 1 : 5;
            const maxSlide = Math.ceil(items.length / itemsPerSlide) - 1;
            
            currentSlide++;
            if (currentSlide > maxSlide) {
                currentSlide = 0;
            }
            updateMargin();
        }

        function updateMargin() {
            const slider = document.querySelector('.slider-width');
            const items = document.querySelectorAll('.slider-item.agreement');
            
            if (items.length === 0 || !slider) return;
            
            // Atualizar largura da tela atual
            larguraTela = window.innerWidth;
            
            let itemsPerSlide, slideWidth;
            
            if (larguraTela <= 425) {
                itemsPerSlide = 1;
                slideWidth = items[0].offsetWidth + 20; // largura do item + margin
            } else {
                itemsPerSlide = 5;
                slideWidth = (items[0].offsetWidth + 20) * itemsPerSlide; // largura de 5 itens
            }
            
            const maxSlide = Math.ceil(items.length / itemsPerSlide) - 1;
            
            // Garantir que currentSlide não ultrapasse os limites
            if (currentSlide > maxSlide) {
                currentSlide = 0;
            }
            if (currentSlide < 0) {
                currentSlide = maxSlide;
            }
            
            const newMargin = currentSlide * slideWidth;
            slider.style.marginLeft = `-${newMargin}px`;
            
            // Debug para ajudar a identificar problemas
            console.log(`Largura: ${larguraTela}, Items por slide: ${itemsPerSlide}, Slide atual: ${currentSlide}, Max slide: ${maxSlide}, Margin: ${newMargin}px`);
        }

        function readMore(element) {
            const button = element;
            const sliderItem = button.closest('.slider-item');
            sliderItem.classList.toggle('active');

            if (sliderItem.classList.contains('active')) {
                return element.textContent = 'Voltar';
            } else {
                return element.textContent = 'Veja mais';
            }
        }

        const slider = document.querySelector('.slider-width');

        });

                goNext();
            }, 4000);
