const scrollable = document.getElementById('scrollable');

if (scrollable) { // Check if the element exists
  let isDragging = false; // Flag für Dragging
  let startX; // Startposition der Maus
  let scrollLeft; // Anfangs-Scrollposition
  let dragTimeout; // Timeout für das Halten der Maus
  let isClickPrevented = false; // Verhindert Klicks während Dragging

  // Hilfsfunktion: Alle Links wiederherstellen
  function resetButtonStyles() {
    const links = scrollable.querySelectorAll('a');
    links.forEach((link) => {
      link.style.pointerEvents = '';
      link.style.userSelect = '';
    });
  }

  // Mousedown: Potenzielles Starten von Dragging
  scrollable.addEventListener('pointerdown', (e) => {
    startX = e.pageX - scrollable.offsetLeft;
    scrollLeft = scrollable.scrollLeft;

    // Timeout setzen, um Dragging nur bei längerem Halten zu starten
    dragTimeout = setTimeout(() => {
      isDragging = true; // Dragging aktivieren
      isClickPrevented = true; // Klick auf Link verhindern
      const target = e.target.closest('a');
      if (target) {
        // Ändere die Stile, um Interaktionen zu blockieren
        target.style.pointerEvents = 'none';
        target.style.userSelect = 'none';
      }
    }, 100); // Zeit, die für Dragging erforderlich ist
  });

  // Mouseup: Dragging stoppen oder Klick erlauben
  document.addEventListener('pointerup', (e) => {
    clearTimeout(dragTimeout); // Timeout abbrechen
    if (isDragging) {
      isDragging = false; // Dragging beenden
      resetButtonStyles(); // Alle Links zurücksetzen
      e.preventDefault(); // Klick während Dragging verhindern
      e.stopImmediatePropagation(); // Weitere Events stoppen
    }
    isClickPrevented = false; // Normalen Klick wieder zulassen
  });

  // Mousemove: Scrollen bei Dragging
  document.addEventListener('pointermove', (e) => {
    if (!isDragging) return; // Nur scrollen, wenn Dragging aktiv ist
    e.preventDefault();
    const x = e.pageX - scrollable.offsetLeft;
    const walk = (x - startX) * 1.5; // Geschwindigkeit des Scrollens
    scrollable.scrollLeft = scrollLeft - walk;
  });

  // Zusätzliche Sicherheit: Klicks blockieren, wenn Dragging aktiv war
  scrollable.querySelectorAll('a').forEach((link) => {
    // Drag & Drop blockieren
    link.addEventListener('dragstart', (e) => {
      e.preventDefault(); // Dragging von Links verhindern
    });

    // Klickblockierung prüfen
    link.addEventListener('click', (e) => {
      if (isClickPrevented) {
        e.preventDefault(); // Klick verhindern, wenn Dragging aktiv war
        isClickPrevented = false; // Blockierung zurücksetzen
      }
    });
  });
} else {
  console.warn("No scrollable element found on the page.");
}

// Moderne async Version für bessere Performance
async function initConsentManager() {
    // Verhindere mehrfache Initialisierung
    if (window.cmpInitialized) return;
    window.cmpInitialized = true;

    // Einmal-Guard gegen mehrfaches Feuern
    let proceeded = false;
    const proceedOnce = () => {
        if (proceeded) return false;
        proceeded = true;
        return true;
    };

    // Hilfs-Timeout, damit Ads nie das Iframe blockieren
    const settleWithTimeout = (promises, ms) => {
        let timer;
        const timeout = new Promise((resolve) => {
            timer = setTimeout(() => resolve('timeout'), ms);
        });
        return Promise.race([
            Promise.allSettled(promises).then((r) => {
                clearTimeout(timer);
                return r;
            }),
            timeout
        ]);
    };

    const loadAdsParallel = async () => {
        try {
            await settleWithTimeout([
                loadGoogleAds(),
                loadAzerionAds(), 
            ], 100); // 3s Failsafe – anpassen bei Bedarf
        } catch (err) {
            console.error('Ad loading error:', err);
        } finally {
            try { showIframe(); } catch(e) { console.error('showIframe error:', e); }
        }
    };
    
    // Check welche CMP-Version verfügbar ist
    const hasCMPv1 = typeof __cmp !== 'undefined';
    const hasTCFv2 = typeof __tcfapi !== 'undefined';
    
    if (hasTCFv2) {
        // Bevorzuge TCF v2 (neuer Standard)
        __tcfapi('addEventListener', 2, (tcData, success) => {
            if (!success || !tcData) return;

            // Fallback: Wenn GDPR NICHT gilt (z.B. USA), alles genau 1x laden
            if (tcData.gdprApplies === false) {
                if (!proceedOnce()) return;
                console.log('GDPR does not apply (TCF v2) – proceeding immediately');
                loadAdsParallel();
                return;
            }

            // EU-Flow
            if (tcData.gdprApplies) {
                if (tcData.eventStatus === 'tcloaded' || tcData.eventStatus === 'useractioncomplete') {
                    if (!proceedOnce()) return;
                    console.log('TCF v2 consent received');
                    loadAdsParallel();
                }
            }
        });
    } else if (hasCMPv1) {
        // Fallback auf CMP v1
        __cmp('addEventListener', ['consent', () => {
            const status = __cmp('consentStatus');
            if (status?.consentExists) {
                if (!proceedOnce()) return;
                console.log('CMP v1 consent received');
                loadAdsParallel();
            }
        }, false], null);
    } else {
        // Kein CMP gefunden - lade nach Delay, aber nur 1x
        console.log('No CMP found - using timeout fallback');
        setTimeout(() => {
            if (!proceedOnce()) return;
            loadAdsParallel();
        }, 2000);
    }
}

// Start bei DOMContentLoaded für optimales Timing
if (document.readyState === 'loading') {
    document.addEventListener('DOMContentLoaded', initConsentManager);
} else {
    initConsentManager();
}
