  <script>
    const $ = (id) => document.getElementById(id);
    const logEl = $("log");

    function log(msg){
      const ts = new Date().toLocaleTimeString();
      logEl.innerHTML += `[${ts}] ${escapeHtml(msg)}<br>`;
      logEl.scrollTop = logEl.scrollHeight;
    }
    function escapeHtml(s){
      return String(s)
        .replaceAll("&","&amp;")
        .replaceAll("<","&lt;")
        .replaceAll(">","&gt;");
    }

    function setStatus(ok, text){
      $("statusText").textContent = text;
      const dot = $("statusDot");
      dot.classList.remove("ok","bad");
      dot.classList.add(ok ? "ok" : "bad");
    }

    function ping(){
      log("UI ping ✅");
    }

    async function copyInfo(){
      const info =
`Host: ${location.host}
Path: ${location.pathname}
Time: ${new Date().toString()}
UA: ${navigator.userAgent}`;
      try{
        await navigator.clipboard.writeText(info);
        log("Copied to clipboard ✅");
      }catch(e){
        log("Clipboard blocked (not HTTPS / permissions).");
      }
    }

    function clearLog(){ logEl.innerHTML = ""; log("Log cleared."); }

    let contrast = false;
    function toggleTheme(){
      contrast = !contrast;
      document.documentElement.style.setProperty("--card", contrast ? "rgba(0,0,0,.35)" : "rgba(255,255,255,.08)");
      document.documentElement.style.setProperty("--stroke", contrast ? "rgba(255,255,255,.22)" : "rgba(255,255,255,.14)");
      log(`Contrast: ${contrast ? "ON" : "OFF"}`);
    }

    function init(){
      $("host").textContent = location.host || "local-file";
      $("path").textContent = location.pathname || "/";
      $("time").textContent = new Date().toLocaleString();
      $("ua").textContent = navigator.userAgent;
      $("vp").textContent = `${window.innerWidth}×${window.innerHeight}`;
      $("lang").textContent = navigator.language || "—";
      $("online").textContent = navigator.onLine ? "yes" : "no";
      $("build").textContent = "dev-" + Math.random().toString(16).slice(2,8);

      setStatus(true, "ready");
      log("Page loaded: OK");
    }

    window.addEventListener("resize", () => $("vp").textContent = `${window.innerWidth}×${window.innerHeight}`);
    window.addEventListener("online",  () => $("online").textContent = "yes");
    window.addEventListener("offline", () => $("online").textContent = "no");
    init();
  </script>
