        function verifyTurnstile() {
        let turnstile_verified = localStorage.getItem("turnstile_verified");
        function renderTurnstile(turnstileModal) {
            turnstile.ready(function () {
                let widgetId = turnstile.render("#cf-turnstile", {
                    sitekey: "0x4AAAAAAA1mpoKsbQ6UbuNP",
                    retry: "never",
                    "refresh-expired": "never",
                    "refresh-timeout": "never",
                    "error-callback": turnstileError,
                    callback: function(token) {
                        $.ajax({
                            type: "POST",
                            data: {
                                "cf-turnstile-response": token
                            },
                            url: "/api/verification",
                            complete: function (xhr) {
                                let res = xhr.responseJSON;
                                if (res !== undefined && res !== "" && res.result !== undefined && res.result !== "") {
                                    if (xhr.status === 200) {
                                        localStorage.setItem("turnstile_verified", res.result)
                                        turnstileModal.hide()
                                        $("#turnstileModal").remove()
                                    } else {
                                        addAlert("danger", "Verification failed. Try again")
                                        displayAlerts()
                                    }
                                } else {
                                    addAlert("danger", "Unexpected error")
                                    displayAlerts()
                                }
                            }
                        });
                    }
                });
                $(document).on("click", "#refresh-turnstile", function() {
                    turnstile.remove(widgetId)
                    $("#cf-turnstile").empty()
                    renderTurnstile(turnstileModal)
                });
                function turnstileError(error) {
                    console.log(error)
                    return true
                }
            });
        }
        if (turnstile_verified === null || Math.floor($.now() / 1000) - parseInt(turnstile_verified) >= 300 - 10) {
            if ($("#turnstileModal").length === 0) {
                $("body").append(`<div class="modal fade user-select-none" data-bs-backdrop="static" data-bs-keyboard="false" id="turnstileModal"><div class="modal-dialog"><div class="modal-content"><div class="modal-header"><h1 class="modal-title fs-5">Checking your browser before accessing</h1></div><div class="modal-body"><p>This process is automatic. Your browser will return to your requested content shortly.</p><div id="cf-turnstile" data-theme="dark""></div></div><div class="modal-footer"><button id="refresh-turnstile" class="btn btn-primary">Refresh</button></div></div></div></div>`)
                let turnstileModal = new bootstrap.Modal("#turnstileModal")
                turnstileModal.show()
                renderTurnstile(turnstileModal)
            }
        }
        let refresh = window.setInterval(function () {
            clearInterval(refresh);
            verifyTurnstile();
        }, 5000);
    }
</script>

<script>
    displayLoading("[data-search=results]")
    let searchTitles = []
    $("[data-search=input]").on("click", function () {
        if (searchTitles.length === 0) {
            $.getJSON("https://infinityscans.org/api/search", function (data) {
                searchTitles = data.result
            })
        }
    })
    let typingTimeout = null
    $("[data-search=input]").on("input", function () {
        if ($("[data-search=results]").find("[data-loading-placeholder]").length === 0) {
            $("[data-search=results]").append(`<div data-loading-placeholder></div>`)
            displayLoading("[data-search=results]")
        }
        $("[data-search=results-wrapper]").removeClass("d-none")
        let search = $(this).val().toLowerCase()
        clearTimeout(typingTimeout);
        typingTimeout = setTimeout(function() {
            let filteredSearchTitles = searchTitles.filter(title => {
                let alt_titles = title.alt_titles === null ? [] : JSON.parse(title.alt_titles)
                return title.title.toLowerCase().includes(search) || alt_titles.some(altTitle => altTitle.toLowerCase().includes(search));
            })
            if (filteredSearchTitles.length > 0) {
                $("[data-search=results]").html(filteredSearchTitles.map((item) => {
                    let type = item.type === 1 ? "comic" : "novel"
                    let link = `https://infinityscans.org/${type}/${item.id}/${item.slug}/d19af2e11df1c199849c9f4370d8e5f5`
                    let votes = `${item.rating_count} votes`
                    if (item.rating_count === 1) {
                        votes = "1 vote"
                    }
                    return `<div class="bg-body-secondary d-flex align-items-center overflow-hidden">
                                        <div class="card flex-grow-1">
                                            <div class="row g-0">
                                                <div class="col-auto">
                                                    <a class="position-relative d-flex align-items-center rounded-start overflow-hidden" style="width: 80px; aspect-ratio: 1/1.25" href="${link}">
                                                        <img src="https://cdn.infinityscans.org/${item.id}/${item.cover}?_=${item.updated}" class="img-fluid rounded-start" alt="${item.title} Cover" loading="lazy" />
                                                        <span class="badge text-bg-primary position-absolute top-0 m-1 rounded-pill text-uppercase small">${type}</span>
                                                    </a>
                                                </div>
                                                <div class="col px-2 py-1 overflow-hidden">
                                                    <a class="d-block fw-bold text-decoration-none text-truncate border-bottom" href="${link}" title="${item.title}">${item.title}</a>
                                                    <ul class="list-group list-group-flush">
                                                        <li class="list-group-item bg-body p-0 small d-flex flex-nowrap align-items-center border-0"><span class="d-stars" style="--rating: ${item.rating}" title="${item.rating} stars out of 5"></span><span class="d-inline-block text-truncate ps-1" title="${votes}">${votes}</span></li>
                                                        <li class="list-group-item bg-body p-0 small border-0">Views: ${item.views}</li>
                                                        <li class="list-group-item bg-body p-0 small border-0">Bookmarks: ${item.bookmarks}</li>
                                                    </ul>
                                                </div>
                                            </div>
                                        </div>
                                    </div>`
                }).join(""))
            } else {
                $("[data-search=results]").html(`<div style="height: 75px" class="d-flex align-items-center justify-content-center">No results found. Maybe try a different spelling?</div>`)
            }
            removeLoading("[data-search=results]")
        }, 500);
    })
    function searchDisplayNone() {
        $("[data-search=results-wrapper]").addClass("d-none")
        $("[data-search=results]").html(`<div style="height: 75px" data-loading-placeholder></div>`)
        $("[data-search=input]").val("").blur()
        displayLoading("[data-search=results]")
    }
    $(document).on("click", function (e) {
        if ($(e.target).parents("[data-search=results-wrapper]").length === 0 && !$(e.target).is("[data-search=input]")) {
            searchDisplayNone()
        }
    })
    let windowWidth = window.innerWidth;
    $(window).resize(function() {
        if (window.innerWidth != windowWidth) {
            windowWidth = window.innerWidth;
            searchDisplayNone();
            $(".dropdown-toggle").removeClass("show");
            $(".dropdown-menu").removeClass("show");
        }
    });
</script>
