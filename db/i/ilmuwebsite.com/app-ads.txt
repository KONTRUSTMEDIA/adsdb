    (function() {
        const slides = Array.from(document.querySelectorAll('.slide'));
        if (!slides.length) return;

        slides.forEach((s, i) => {
            s.classList.toggle('hidden', i !== 0);
            s.classList.toggle('visible', i === 0);
        });

        function next() {
            slides[idx].classList.add('hidden');
            slides[idx].classList.remove('visible');
            idx = (idx + 1) % slides.length;
            slides[idx].classList.remove('hidden');
            slides[idx].classList.add('visible');
        }

        if (slides.length > 1) {
            setInterval(next, 4500);

    function formatTanggalIndonesiaLengkap(datetime) {
        const hari = ['Minggu', 'Senin', 'Selasa', 'Rabu', 'Kamis', 'Jumat', 'Sabtu'];
        const bulan = {
            1: 'Januari',
            2: 'Februari',
            3: 'Maret',
            4: 'April',
            5: 'Mei',
            6: 'Juni',
            7: 'Juli',
            8: 'Agustus',
            9: 'September',
            10: 'Oktober',
            11: 'November',
            12: 'Desember'
        };

        const date = new Date(datetime);
        const namaHari = hari[date.getDay()];
        const tgl = date.getDate();
        const namaBulan = bulan[date.getMonth() + 1];
        const tahun = date.getFullYear();
        const jam = ('0' + date.getHours()).slice(-2);
        const menit = ('0' + date.getMinutes()).slice(-2);

        return `${namaHari}, ${tgl} ${namaBulan} ${tahun} ${jam}.${menit} WIB`;
