                    webhookUrl: 'https://n8n.alexanderoviedofadul.dev/webhook/429c7c2c-c27c-45bc-ae4d-b2e74c8f7362/chat',

                    chatSessionKey: 'sessionId',
                    defaultLanguage: 'es',

                            footer: 'Isabella EA - Soluciones tecnológicas a medida',
                            getStarted: 'Nueva Conversación',
                            inputPlaceholder: 'Escribe tu pregunta...',
                            errorMessage: 'Lo sentimos, el servicio de chat no está disponible en este momento. Por favor, contáctanos por WhatsApp.'
                        }
                    },
                    webhookConfig: {
                        method: 'POST',
                        headers: {
                            'Content-Type': 'application/json',
                            'Accept': 'application/json'
                        },
                        timeout: 30000, // 30 segundos de timeout
                        mode: 'cors', // Especificar el modo de la solicitud
                        credentials: 'omit', // No enviar cookies para evitar problemas de CORS
                        cache: 'no-cache' // No usar caché para evitar problemas con respuestas antiguas
                    },
                    // Manejar errores de conexión
                    onError: (error) => {
                        console.error('Error en el chat:', error);

                        // Determinar el tipo de error
                        let errorMessage = 'Lo sentimos, nuestro servicio de chat está experimentando problemas técnicos.';
                        let errorTitle = 'Servicio de chat no disponible';

                        // Si es un error 500, mostrar un mensaje específico
                        if (error && (
                            (error.message && error.message.includes('500')) ||
                            (error.status === 500) ||
                            (error.statusCode === 500) ||
                            (error.stack && error.stack.includes('500'))
                        )) {
                            errorMessage = 'El servidor de chat ha encontrado un error interno (Error 500). Esto indica que el servidor N8N está funcionando pero hay un problema con la configuración del webhook o con el flujo de trabajo.';
                            errorTitle = 'Error interno en el servidor de chat';
                            console.log('Error 500 detectado en el servidor N8N: https://n8n.alexanderoviedofadul.dev/webhook/429c7c2c-c27c-45bc-ae4d-b2e74c8f7362/chat');

                            // Mostrar un mensaje adicional con instrucciones para el usuario
                            const chatElement = document.getElementById('isabella-chat');
                            if (chatElement) {
                                chatElement.innerHTML = `
                                    <div style="padding: 20px; background-color: #f8f9fa; border-radius: 10px; text-align: center; margin-top: 20px;">
                                        <h4>${errorTitle}</h4>
                                        <p>${errorMessage}</p>
                                        <p>Posibles causas:</p>
                                        <ol style="text-align: left; display: inline-block;">
                                            <li>El flujo de trabajo en N8N puede estar mal configurado</li>
                                            <li>El webhook puede no tener los permisos necesarios</li>
                                            <li>Puede haber un error en el código del flujo de trabajo</li>
                                            <li>El servidor N8N puede estar sobrecargado</li>
                                        </ol>
                                        <p style="margin-top: 15px;">Por favor, contacta al administrador del sistema para revisar la configuración del servidor N8N.</p>
                                    </div>
                                `;
                                // Evitar que se siga ejecutando el resto del código
                                return;
                            }
                        }

                        // Si es un error de CORS, mostrar un mensaje específico
                        if (error && error.message && error.message.includes('CORS')) {
                            errorMessage = 'Hay un problema de configuración en el servidor. Estamos trabajando para solucionarlo.';
                            errorTitle = 'Error de configuración';
                            console.log('Error CORS detectado');
                        }

                        // Si es un error de conexión (Failed to fetch, Network Error, etc.)
                        if (error && error.message && (
                            error.message.includes('Failed to fetch') ||
                            error.message.includes('Network Error') ||
                            error.message.includes('Mixed Content') ||
                            error.message.includes('blocked:mixed') ||
                            error.message.includes('ERR_CONNECTION_TIMED_OUT')
                        )) {
                            // Verificar si es un error de timeout específicamente
                            if (error.message.includes('ERR_CONNECTION_TIMED_OUT') ||
                                (error.stack && error.stack.includes('ERR_CONNECTION_TIMED_OUT'))) {
                                errorMessage = 'La conexión con el servidor de chat ha excedido el tiempo de espera. El servidor n8n.alexanderoviedofadul.dev no está respondiendo.';
                                errorTitle = 'Error de tiempo de espera en la conexión';
                                console.log('Error de timeout detectado: El servidor N8N en https://n8n.alexanderoviedofadul.dev no responde');
                            } else {
                                errorMessage = 'No se puede conectar con el servidor de chat en este momento. Estamos intentando conectar con https://n8n.alexanderoviedofadul.dev pero parece que hay un problema de conexión.';
                                errorTitle = 'Error de conexión con el servidor';
                                console.log('Error de conexión detectado: No se puede conectar con el servidor N8N en https://n8n.alexanderoviedofadul.dev');
                            }

                            // Mostrar un mensaje adicional con instrucciones para el usuario
                            const chatElement = document.getElementById('isabella-chat');
                            if (chatElement) {
                                chatElement.innerHTML = `
                                    <div style="padding: 20px; background-color: #f8f9fa; border-radius: 10px; text-align: center; margin-top: 20px;">
                                        <h4>${errorTitle}</h4>
                                        <p>${errorMessage}</p>

                                        </ol>
                                        <p style="margin-top: 15px;">Por favor, contáctanos por WhatsApp para asistencia o inténtalo más tarde</p>
                                    </div>
                                `;

                                return;
                            }

                try {

                    console.log('Verificando estado del servidor N8N...');

                    // Crear una función para asegurarse de que los datos tengan la estructura correcta
                    const transformRequest = function(data) {
                        // Asegurarse de que siempre haya un sessionId
                        if (!data.sessionId) {
                            data.sessionId = "session-" + Date.now();
                        }
                        // Asegurarse de que siempre haya un chatInput
                        if (!data.chatInput && !data.message) {
                            data.chatInput = "Hola";
                        } else if (data.message && !data.chatInput) {
                            data.chatInput = data.message;
                        }
                        return data;
                    };

                    // Agregar la función de transformación al objeto chatConfig
                    chatConfig.transformRequest = transformRequest;

                    // Preparar los datos para la solicitud de prueba
                    const testData = transformRequest({
                        chatInput: "Hola",
                        sessionId: "test-session-" + Date.now()
                    });

                    const response = await fetch(chatConfig.webhookUrl, {
                        method: 'POST', // Usar POST en lugar de HEAD para obtener más información
                        headers: {
                            'Content-Type': 'application/json',
                            'Accept': 'application/json'
                        },
                        // Enviar un payload con la estructura correcta que espera el flujo de trabajo
                        body: JSON.stringify(testData),
                        mode: 'cors',
                        credentials: 'omit'
                    });

                    console.log('Respuesta del servidor N8N:', response.status, response.statusText);

                    if (response.status === 500) {
                        // Intentar obtener más información sobre el error 500
                        let errorDetails = '';
                        try {
                            const errorBody = await response.text();
                            console.log('Detalles del error 500:', errorBody);
                            errorDetails = errorBody;

                            // Si el error es "Error in workflow", podría ser porque el flujo de trabajo espera campos específicos
                            if (errorBody.includes('Error in workflow')) {
                                console.log('El error parece estar relacionado con el formato de datos que espera el flujo de trabajo.');
                            }
                        } catch (e) {
                            console.error('No se pudo obtener el cuerpo de la respuesta del error:', e);
                        }

                        throw new Error(`El servidor respondió con un error 500 (Error interno del servidor). Detalles: ${errorDetails}`);
                    } else if (response.status !== 200 && response.status !== 204) {
                        // Intentar obtener más información sobre el error
                        let errorDetails = '';
                        try {
                            const errorBody = await response.text();
                            console.log(`Detalles del error ${response.status}:`, errorBody);
                            errorDetails = errorBody;
                        } catch (e) {
                            console.error('No se pudo obtener el cuerpo de la respuesta del error:', e);
                        }

                        throw new Error(`El servidor respondió con un código de estado ${response.status} (${response.statusText}). Detalles: ${errorDetails}`);
                    }

                    // Si llegamos aquí, el servidor está respondiendo correctamente
                    console.log('Servidor N8N respondiendo correctamente. Inicializando chat...');

                    // Crear una versión personalizada de createChat que aplique la transformación
                    const originalCreateChat = createChat;
                    const customCreateChat = (config) => {
                        // Guardar la configuración original
                        const originalConfig = {...config};

                        // Modificar la configuración para interceptar las solicitudes
                        const originalFetch = window.fetch;
                        window.fetch = function(url, options) {
                            // Solo interceptar solicitudes al webhook
                            if (url === config.webhookUrl && options && options.body) {
                                try {
                                    // Parsear el cuerpo de la solicitud
                                    const body = JSON.parse(options.body);

                                    // Aplicar la transformación
                                    const transformedBody = config.transformRequest ? config.transformRequest(body) : body;
                                    console.log('Interceptando solicitud al webhook. Datos originales:', body);
                                    console.log('Datos transformados:', transformedBody);

                                    // Reemplazar el cuerpo de la solicitud
                                    options.body = JSON.stringify(transformedBody);
                                } catch (e) {
                                    console.error('Error al transformar la solicitud:', e);
                                }
                            }

                            // Llamar a la función original de fetch
                            return originalFetch.apply(this, arguments);
                        };

                        // Llamar a la función original de createChat con la configuración modificada
                        const chatInstance = originalCreateChat(originalConfig);

                        // Restaurar la función original de fetch después de un breve retraso
                        // para asegurar que el chat se haya inicializado completamente
                        setTimeout(() => {
                            window.fetch = originalFetch;
                            console.log('Restaurada la función original de fetch');
                        }, 5000);

                        return chatInstance;
                    };

                    // Usar la versión personalizada de createChat
                    customCreateChat(chatConfig);
                } catch (error) {
                    console.error('Error al verificar el estado del servidor N8N:', error);
                    // Intentar inicializar el chat de todos modos, el manejador de errores se encargará de mostrar el mensaje adecuado
                    // Usar la versión personalizada de createChat si está definida, o la original si no
                    if (typeof customCreateChat === 'function') {
                        customCreateChat(chatConfig);
                    } else {
                        // Crear una versión personalizada de createChat que aplique la transformación
                        const originalCreateChat = createChat;
                        const customCreateChat = (config) => {
                            // Guardar la configuración original
                            const originalConfig = {...config};

                            // Modificar la configuración para interceptar las solicitudes
                            const originalFetch = window.fetch;
                            window.fetch = function(url, options) {
                                // Solo interceptar solicitudes al webhook
                                if (url === config.webhookUrl && options && options.body) {
                                    try {
                                        // Parsear el cuerpo de la solicitud
                                        const body = JSON.parse(options.body);

                                        // Aplicar la transformación
                                        const transformedBody = config.transformRequest ? config.transformRequest(body) : body;
                                        console.log('Interceptando solicitud al webhook. Datos originales:', body);
                                        console.log('Datos transformados:', transformedBody);

                                        // Reemplazar el cuerpo de la solicitud
                                        options.body = JSON.stringify(transformedBody);
                                    } catch (e) {
                                        console.error('Error al transformar la solicitud:', e);
                                    }
                                }

                                // Llamar a la función original de fetch
                                return originalFetch.apply(this, arguments);
                            };

                            // Llamar a la función original de createChat con la configuración modificada
                            const chatInstance = originalCreateChat(originalConfig);

                            // Restaurar la función original de fetch después de un breve retraso
                            // para asegurar que el chat se haya inicializado completamente
                            setTimeout(() => {
                                window.fetch = originalFetch;
                                console.log('Restaurada la función original de fetch');
                            }, 5000);

                            return chatInstance;
                        };

                        // Usar la versión personalizada de createChat
                        customCreateChat(chatConfig);
                    }
                }

            } catch (error) {
                console.error('Error al inicializar el chat:', error);
                // Mostrar un mensaje genérico de error
                const chatElement = document.getElementById('isabella-chat');
                if (chatElement) {
                    chatElement.innerHTML = `
                        <div style="padding: 20px; background-color: #f8f9fa; border-radius: 10px; text-align: center; margin-top: 20px;">
                            <h4>Error en el servicio de chat</h4>
                            <p>Ha ocurrido un error inesperado al cargar el chat.</p>
                            <p>Por favor, contáctanos por WhatsApp o envíanos un correo a <a href="mailto:info@isabellaea.com">info@isabellaea.com</a></p>
                        </div>
                    `;
                }
            }
        }
