var lang = "hu";

function toggleLeiras(id) {
    var shortDiv = document.getElementById('short-' + id);
    var fullDiv  = document.getElementById('full-' + id);
    var btn      = document.getElementById('btn-' + id);

    if (!shortDiv || !fullDiv || !btn) return;

    if (fullDiv.style.display === 'none' || fullDiv.style.display === '') {
        fullDiv.style.display = 'block';
        shortDiv.style.display = 'none';
        btn.textContent = (lang === 'hu') ? 'Kevesebb...' : 'Less...';
    } else {
        fullDiv.style.display = 'none';
        shortDiv.style.display = 'block';
        btn.textContent = (lang === 'hu') ? 'Bővebben...' : 'More...';
    }
}

document.addEventListener('DOMContentLoaded', function() {
    document.querySelectorAll('.leiras-full').forEach(function(div){
        div.style.display='none';
    });
});
