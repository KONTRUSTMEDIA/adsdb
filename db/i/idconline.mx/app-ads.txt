                            text: "Por favor, completa el CAPTCHA.",
                            style: {
                                background: 'none'
                            },
                            duration: 3000,
                            position: 'center',
                            className: "bg-danger",
                        }).showToast();
                        submitButton.disabled = false;
                        return;
                    }

                    const formData = new FormData(formContact);
                    formData.append('acc', 'saveContactUs');

                    try {
                        const request = await fetch('./?a=home.main', {
                            method: 'POST',
                            body: formData
                        });

                        const response = await request.json();

                        const bgColor = response.error ? "danger" : "success";
                        Toastify({
                            text: response.msg,
                            style: {
                                background: 'none'
                            },
                            duration: 3000,
                            position: 'center',
                            className: "bg-" + bgColor,
                        }).showToast();

                        if(!response.error){
                            formContact.reset();
                            grecaptcha.reset();
                            modalContact.modal('hide');
                        }

                    } catch (e) {
                        console.error(e);
                        Toastify({
                            text: "Ha ocurrido un error en el servidor.",
                            style: {
                                background: 'none'
                            },
                            duration: 3000,
                            position: 'center',
                            className: "bg-danger",
                        }).showToast();
                    } finally {
                        submitButton.disabled = false;
                    }
                });
            });
        }
        return {
            init: function () {
                handleForm();
            }
