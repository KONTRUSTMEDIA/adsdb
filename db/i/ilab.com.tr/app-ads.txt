    function showTeaser() {
      const teaserThumbnail = document.getElementById('teaser-thumbnail');
      const videoWrapper = document.getElementById('teaser-video');
      const videoEl = videoWrapper.querySelector('video');
      const teaserButton = document.getElementById('teaser-button');
      teaserThumbnail.classList.add('d-none');
      videoWrapper.classList.remove('d-none');
      teaserButton.classList.add('d-none');
      videoEl.play();
      videoWrapper.addEventListener('click', () => {
        if (videoEl.paused) {
          videoEl.play();
        } else {
          videoEl.pause();
        }
        teaserButton.classList.toggle('d-none');
      });
      videoEl.addEventListener('ended', () => {
        teaserButton.classList.remove('d-none');
      });
    }
