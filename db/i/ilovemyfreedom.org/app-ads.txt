  function calculateProgress(currentVal, targetVal) {
    var progressBar = document.querySelectorAll('.cart-shippingThreshold__progress');
    var progressNum = document.querySelectorAll('.cart-shipping__num');
    var progressOuter = document.querySelectorAll('.cart-shipping__numOuter');
    var successMsg = document.querySelectorAll('.cart-shipping__success');
    var result = Math.round((100 * currentVal) / targetVal).toFixed(2);

    progressBar.forEach(function (el) {
      el.style.width = result + '%';
    });

    var newProgressNum = (targetVal - currentVal) / 100;
    if (newProgressNum <= 0) {
      successMsg.forEach(function (el) {
        el.style.display = 'block';
      });
      progressOuter.forEach(function (el) {
        el.style.display = 'none';
      });
    } else {
      successMsg.forEach(function (el) {
        el.style.display = 'none';
      });
      progressOuter.forEach(function (el) {
        el.style.display = 'block';
      });
    }

    progressNum.forEach(function (el) {
      el.textContent = newProgressNum;
    });
  }

  function fetchCartDataAndCalculateProgress() {
    fetch('/cart.js', {
        method: 'GET',
        headers: {
            'Content-Type': 'application/json',
            'X-Requested-With': 'XMLHttpRequest',
        },
    })
    .then((response) => response.json())
    .then((cart) => {
        var productIdsToCheck = [7761620304041];
        var productIdsInCart = productIdsToCheck.some(function (productId) {
            return cart.items.some(function (item) {
                return item.product_id === productId;
            });
        });

        var customerTags = cart.customer ? cart.customer.tags.split(', ') : [];
        var targetValue;

        if (productIdsInCart) {
            targetValue = 2900;
        } else {
            if (customerTags.includes('ALLIANCE-BRONZE-TRIAL') || customerTags.includes('ALLIANCE-BRONZE-MEMBER')) {
                targetValue = 3900;

                targetValue = 0;
            } else {
                targetValue = 4900;
            }
        }

        calculateProgress(cart.total_price, targetValue);
    })
    .catch((error) => {
    });
}

<script>
let previousCartState = null;

function fetchCart() {
  return fetch('/cart.js')
    .then(response => response.json())
    .then(cart => {
      return cart;
    });
}

function fetchProductTags(handle) {
  return fetch(`/products/${handle}.js`)
    .then(response => response.json())
    .then(product => {
      return product.tags || [];
    })
    .catch(error => {
      return [];
    });
}

function checkMembershipProductRemoval(currentCart) {
  const membershipProductData = JSON.parse(sessionStorage.getItem('membershipProductData'));

  if (!membershipProductData) {
    return;
  }

  if (previousCartState) {
      const membershipProductWasInCart = previousCartState.items.some(item => {
      const isMatchingProduct = item.variant_id == membershipProductData.id;
      const isMatchingSellingPlan = item.selling_plan_allocation && item.selling_plan_allocation.selling_plan.id == membershipProductData.sellingPlan;
      
      return isMatchingProduct && isMatchingSellingPlan;
    });

      const membershipProductIsStillInCart = currentCart.items.some(item => {
      const isMatchingProduct = item.variant_id == membershipProductData.id;
      const isMatchingSellingPlan = item.selling_plan_allocation && item.selling_plan_allocation.selling_plan.id == membershipProductData.sellingPlan;
      
      return isMatchingProduct && isMatchingSellingPlan;
    });

    if (membershipProductWasInCart && !membershipProductIsStillInCart) {
      sessionStorage.removeItem('membershipProductData');
      removeLightningDealProducts(currentCart);
    }
  }

  previousCartState = currentCart;
}

function removeLightningDealProducts(cart) {
  const lightningDealLines = [];

  // Log full cart to see available keys
  console.log('Full cart items:', cart.items);

  // Gather all line numbers for products with the 'LightningDeal' tag
  const promises = cart.items.map((item, index) => {
    const productHandle = item.handle;

    return fetchProductTags(productHandle).then(productTags => {
      console.log('Product tags for', productHandle, productTags);

      if (productTags.includes('LightningDeal')) {
        lightningDealLines.push(index + 1); // Add line number to the list
        console.log('LightningDeal product found:', productHandle, 'at line:', index + 1);
      }
    }).catch(error => console.error('Error fetching product tags:', error));
  });

  // Once all tag-fetching promises are resolved
  Promise.all(promises).then(() => {
    if (lightningDealLines.length === 0) {
      console.log('No LightningDeal products to remove.');
    } else if (lightningDealLines.length === 1) {
      // If only one LightningDeal product, remove by line number
      console.log('Removing single LightningDeal product at line:', lightningDealLines[0]);
      removeProductByLine(lightningDealLines[0]);
    } else {
      // If multiple LightningDeal products, clear the entire cart
      console.log('Multiple LightningDeal products found. Clearing the cart.');
      clearCart();
    }
  }).catch(error => console.error('Error processing promises:', error));
}

function removeProductByLine(line) {
  fetch('/cart/change.js', {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
    },
    body: JSON.stringify({
      line: line,
      quantity: 0,
    }),
  })
  .then(response => response.json())
  .then(cart => {
    document.dispatchEvent(new CustomEvent('cart:build'));
    fetchCartTotal();
    fetchCartDataAndCalculateProgress();
    return cart;
  })
  .catch(error => console.error('Error removing product:', error));
}

function clearCart() {
  fetch('/cart/clear.js', {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
    },
  })
  .then(response => response.json())
  .then(cart => {
    console.log('Cart cleared successfully:', cart);
    document.dispatchEvent(new CustomEvent('cart:build'));
    fetchCartTotal();
    fetchCartDataAndCalculateProgress();
  })
  .catch(error => console.error('Error clearing cart:', error));
}

function initializeCartState() {
  fetchCart().then(currentCart => {
    previousCartState = currentCart;
  });
}

function trackCartUpdates() {
  fetchCart().then(currentCart => {
    checkMembershipProductRemoval(currentCart); 
  });
}

document.addEventListener('DOMContentLoaded', function() {
  initializeCartState();

  document.addEventListener('cart:updated', function() {
    trackCartUpdates();
  });

  document.body.addEventListener('click', function(event) {
