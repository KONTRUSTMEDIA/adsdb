<script>
(function () {

    function normaliser(str) {
        return (str || "")
            .toLowerCase()
            .normalize("NFD")                 // sépare les accents
            .replace(/[\u0300-\u036f]/g, "")  // supprime les accents
            .replace(/\u00a0/g, " ")          // remplace les espaces insécables
            .replace(/\s+/g, " ")             // compresse les espaces
            .trim();
    }

    function trouverConteneurBloc(el) {
        var blocs = ["P", "DIV", "BLOCKQUOTE", "SECTION", "ARTICLE", "LI"];
        var courant = el;
        while (courant && courant !== document.body) {
            if (blocs.indexOf(courant.tagName) !== -1) {
                return courant;
            }
            courant = courant.parentElement;
        }
        return null;
    }

    function appliquerEncadres() {
        // mots déclencheurs au début du texte (version normalisée, sans accents)
      var regexp = /^(eclairage|note|repere|focus|reaction)\b/;

        // on scanne tout le contenu hors script/style
        var elements = document.querySelectorAll("body *:not(script):not(style)");

        elements.forEach(function (el) {
            var txt = el.innerText || el.textContent || "";
            var txtNorm = normaliser(txt);

            if (!regexp.test(txtNorm)) {
                return;
            }

            var conteneur = trouverConteneurBloc(el) || el;

            // mot-clé normalisé = premier mot (focus/eclairage/note/repere)
            var motCle = txtNorm.split(" ")[0];
            conteneur.setAttribute("data-keyword", motCle);

            conteneur.classList.add("encadre-gris-auto");
        });
    }

    if (document.readyState === "complete" || document.readyState === "interactive") {
        setTimeout(appliquerEncadres, 0);
    } else {
        document.addEventListener("DOMContentLoaded", appliquerEncadres);
    }

})();
</script>

    selectors.forEach(sel => {
      document.querySelectorAll(sel).forEach(el => {
        const t = (el.innerText || el.textContent || "").trim();
        if (t.length > bestLen) { best = el; bestLen = t.length; }

  async function getLinks(maxItems) {
    let res;
    try { res = await fetch(RSS_URL, { cache: "no-store" }); } catch (e) { return null; }
    if (!res.ok) return null;

    const xml = new DOMParser().parseFromString(await res.text(), "text/xml");
    const items = Array.from(xml.querySelectorAll("item"));
    if (!items.length) return null;

    const currentUrl = (location.href || "").split("#")[0];
    const links = [];

    for (const it of items) {
      const title = it.querySelector("title")?.textContent.trim();
      const link  = it.querySelector("link")?.textContent.trim();
      if (!title || !link) continue;
      if (link === currentUrl) continue;
      links.push({ title, link });
      if (links.length >= maxItems) break;
    }
    return links.length ? links : null;
  }

  function buildBox(links) {
    ensureBoxCSS();
    const box = document.createElement("section");

    if (alreadyInjected()) return true;

      if (old) old.remove();
      boot();
    }
  });
})();
