<script>
(function(){
  "use strict";

  // Racine probable du contenu d'article
  function findRoot(){
    return document.querySelector("article")
        || document.querySelector("#article,.article,.entry-content,.post-content,#content,.content,main")
        || document.body;
  }

  // Mots-clés reconnus en début de bloc
  var KEY_MAP = [
    { key: "focus",     regex: /^(FOCUS)\s*[:\-–]?\s*/i },
    { key: "eclairage", regex: /^(ECLAIRAGE|ÉCLAIRAGE)\s*[:\-–]?\s*/i },
    { key: "note",      regex: /^(NOTE)\s*[:\-–]?\s*/i },
    { key: "repere",    regex: /^(REPERE|REPÈRE)\s*[:\-–]?\s*/i },
    { key: "reaction",  regex: /^(REACTION|RÉACTION)\s*[:\-–]?\s*/i },
    { key: "asavoir",   regex: /^(A\s+SAVOIR|À\s+SAVOIR)\s*[:\-–]?\s*/i },
    { key: "precision", regex: /^(PRECISION|PRÉCISION)\s*[:\-–]?\s*/i },

    { key: "brief",      regex: /^(BRIEF)\s*[:\-–]?\s*/i },
    { key: "infoplus",   regex: /^(INFO\+|INFO PLUS)\s*[:\-–]?\s*/i },
    { key: "analyse",    regex: /^(ANALYSE)\s*[:\-–]?\s*/i },
    { key: "contexte",   regex: /^(CONTEXTE)\s*[:\-–]?\s*/i },
    { key: "memoire",    regex: /^(POUR\s+MEMOIRE|POUR\s+MÉMOIRE)\s*[:\-–]?\s*/i },
    { key: "rappel",     regex: /^(RAPPEL)\s*[:\-–]?\s*/i },
    { key: "decryptage", regex: /^(DECRYPTAGE|DÉCRYPTAGE)\s*[:\-–]?\s*/i },
    { key: "enchiffres", regex: /^(EN\s+CHIFFRES)\s*[:\-–]?\s*/i },
    { key: "aretenir",   regex: /^(A\s+RETENIR|À\s+RETENIR)\s*[:\-–]?\s*/i }
  ];

  function normalizeStart(str){
    return (str || "").replace(/^\s+/, "");
  }

  // Essaie de trouver un mot-clé en début de texte
  function detectKey(txt){
    var start = normalizeStart(txt || "");
    for (var i = 0; i < KEY_MAP.length; i++){
      if (KEY_MAP[i].regex.test(start)){
        return KEY_MAP[i].key;
      }
    }
    return null;
  }

  // Traite un bloc (p, div, blockquote, etc.)
  function processBlock(el){
    if (!el) return;

    var txt = el.textContent || "";
    var key = detectKey(txt);
    if (!key) return;

    // Ajoute la classe d'encadré si absente
    if (!el.classList.contains("encadre-gris-auto")){
      el.classList.add("encadre-gris-auto");
    }

    // Ajoute le data-keyword si absent
    if (!el.hasAttribute("data-keyword")){
      el.setAttribute("data-keyword", key);
    }
  }

  function init(){
    var root = findRoot();
    if (!root) return;

    // On cible uniquement les blocs de texte "simples"
    var blocks = root.querySelectorAll("p, div, blockquote");
    for (var i = 0; i < blocks.length; i++){
      processBlock(blocks[i]);
    }
  }

  if (document.readyState === "loading"){
    document.addEventListener("DOMContentLoaded", init);
  } else {
    init();
  }
})();
</script>

<script>
(function () {
  const RSS_URL = "https://www.infonormandie.com/xml/syndication.rss";
  const BLOCK_ID = "infonormandie-autres-infos";
  const STYLE_ID = "in-autres-infos-css";

  function alreadyInjected() { return !!document.getElementById(BLOCK_ID); }
  function isMobile() { return window.matchMedia && window.matchMedia("(max-width: 768px)").matches; }
  function getMaxItems() { return isMobile() ? 3 : 4; }

  function findBestContainer() {
    const selectors = ["article","#article",".article",".entry-content",".post-content","#content",".content","main",".main"];
    let best = null, bestLen = 0;
    selectors.forEach(sel => {
      document.querySelectorAll(sel).forEach(el => {
        const t = (el.innerText || el.textContent || "").trim();
        if (t.length > bestLen) { best = el; bestLen = t.length; }
      });
    });
    return { el: best, len: bestLen };
  }

  function ensureBoxCSS() {
    if (document.getElementById(STYLE_ID)) return;
    const size = isMobile() ? "15px" : "15.5px";
    const st = document.createElement("style");
    st.id = STYLE_ID;
    st.textContent =
      "#" + BLOCK_ID + " ul{font-size:" + size + " !important; line-height:1.5 !important;}" +
      "#" + BLOCK_ID + " li{font-size:inherit !important; line-height:inherit !important;}" +
      "#" + BLOCK_ID + " a{font-size:inherit !important; line-height:inherit !important;}";
    document.head.appendChild(st);
  }

  async function getLinks(maxItems) {
    let res;
    try { res = await fetch(RSS_URL, { cache: "no-store" }); } catch (e) { return null; }
    if (!res.ok) return null;

    const xml = new DOMParser().parseFromString(await res.text(), "text/xml");
    const items = Array.from(xml.querySelectorAll("item"));
    if (!items.length) return null;

    const currentUrl = (location.href || "").split("#")[0];
    const links = [];

    for (const it of items) {
      const title = it.querySelector("title")?.textContent.trim();
      const link  = it.querySelector("link")?.textContent.trim();
      if (!title || !link) continue;
      if (link === currentUrl) continue;
      links.push({ title, link });
      if (links.length >= maxItems) break;
    }
    return links.length ? links : null;
  }

  function buildBox(links) {
    ensureBoxCSS();
    const box = document.createElement("section");
    box.id = BLOCK_ID;
    box.style.cssText = "margin-top:20px;padding:16px;border:1px solid #e6e6e6;border-radius:12px;background:#fafafa;";
    const header = document.createElement("div");
    header.style.cssText = "display:flex;align-items:center;gap:8px;margin:0 0 12px 0;";

    const icon = document.createElement("span");
    icon.setAttribute("aria-hidden", "true");
    icon.style.cssText = "display:inline-flex;opacity:.75;";

      '<path d="M10 2H6a2 2 0 0 0-2 2v16a2 2 0 0 0 2 2h12a2 2 0 0 0 2-2v-8"/>' +

      '</svg>';

    header.appendChild(icon); header.appendChild(h);
    box.appendChild(header);

    const ul = document.createElement("ul");
    ul.style.cssText = "margin:0;padding-left:18px;display:flex;flex-direction:column;gap:10px;";
    links.forEach(l => {
      const li = document.createElement("li");
      const a = document.createElement("a");
      a.href = l.link;
      a.textContent = l.title;
      a.style.cssText = "font-weight:400;color:inherit;text-decoration:none;border-bottom:1px solid rgba(0,0,0,.2);";
      li.appendChild(a);
      ul.appendChild(li);
    });
    box.appendChild(ul);
    return box;
  }

  async function tryInjectOnce() {
    if (alreadyInjected()) return true;
    const c = findBestContainer();
    if (!c.el || c.len < 200) return false;
    const links = await getLinks(getMaxItems());
    if (!links) return false;
    const old = document.getElementById(BLOCK_ID);
    if (old) old.remove();
    c.el.appendChild(buildBox(links));
    return true;
  }

  async function boot() {
    if (await tryInjectOnce()) return;
    let tries = 0;
    const timer = setInterval(async () => {
      tries++;
      const ok = await tryInjectOnce();
      if (ok || tries >= 20) clearInterval(timer);
    }, 500);
  }

  if (document.readyState === "loading") document.addEventListener("DOMContentLoaded", boot);
  else boot();

  let lastMobile = isMobile();
  window.addEventListener("resize", () => {
    const nowMobile = isMobile();
    if (nowMobile !== lastMobile) {
      lastMobile = nowMobile;
      const old = document.getElementById(BLOCK_ID);
      if (old) old.remove();
      boot();
    }
  });
})();
</script>

(function(){
  "use strict";

  var CAP_CLASS = "infono-cap2";

    return document.querySelector("article")
      || document.querySelector("#article,.article,.entry-content,.post-content,#content,.content,main");
  }

    document.querySelectorAll("."+CAP_CLASS).forEach(function(x){
      if(x.tagName === "DIV") {

      }

    var root = findRoot();
    if(!root) return;

    var photos = Array.from(root.querySelectorAll("div.photo"));
    if(photos.length < 2) return;

    var nativeInside = photo2.querySelector("figcaption,.legende,.légende,.caption,.credit,.legend,[class*='legende'],[class*='caption'],[class*='credit']");
    if(nativeInside){
      nativeInside.style.display = "block";

      nativeInside.style.opacity = "1";

    }

    // 2) Sinon, on regarde juste après le bloc .photo (cas anciens)

      next.style.display = "block";
      next.style.visibility = "visible";
      next.style.opacity = "1";

           || Array.from(photo2.querySelectorAll("img")).find(function(i){

    cap.textContent = txt;

      clearTimeout(t);
      t=setTimeout(run, 250);
    }).observe(document.documentElement, {childList:true, subtree:true});
  }
})();
