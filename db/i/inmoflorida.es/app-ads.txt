      // Endpoint Plausible self-hosted
      var plausibleAPI = "https://1727851714667.hostinghouse.it/api/event";

      // Dominio principale (dal data-domain dello script HTML)
      var primaryDomain = currentScript.getAttribute("data-domain");

      // Dominio "aggregato" fisso
      var aggregatorDomain = "totale";

      // Referrer forzato per il progetto "totale"

      /**
       * Funzione generica per inviare l'evento a Plausible
       */
      function sendPlausibleEvent(eventName, domain, ref, options) {
        // Ignora contesti di test o localhost
        if (/^localhost$|^127(\.[0-9]+){0,2}\.[0-9]+$|^\[::1?\]$/.test(loc.hostname) || loc.protocol === "file:") {
          if (options && options.callback) options.callback();
          return;

          if (options && options.callback) options.callback();
          return;

        try {
          if (window.localStorage.plausible_ignore === "true") {
            if (options && options.callback) options.callback();
            return;
          }
        } catch(e) {}

        var payload = {
          n: eventName,        // "pageview" o altro
          u: loc.href,         // URL reale della pagina
          d: domain,           // Dominio di destinazione (primary o aggregator)
          r: ref || null       // Referrer (vero o forzato)
        };
        if (options && options.meta)  payload.m = JSON.stringify(options.meta);
        if (options && options.props) payload.p = options.props;

        var xhr = new XMLHttpRequest();
        xhr.open("POST", plausibleAPI, true);
        xhr.setRequestHeader("Content-Type", "text/plain");
        xhr.send(JSON.stringify(payload));
      }

      /**
       * Invia 2 eventi "pageview":
       * 1) sul dominio principale, con referrer reale
       * 2) sul dominio "totale", con referrer forzato (solo se presente)
       */
      function trackPageview() {
        // Evento per "dominio reale" (con referrer vero)
        sendPlausibleEvent("pageview", primaryDomain, doc.referrer);

        // Se l'utente ha inserito un referrer fittizio (forcedReferrerForAggregator non vuoto)
        if (forcedReferrerForAggregator) {
          sendPlausibleEvent("pageview", aggregatorDomain, forcedReferrerForAggregator);
        }

       * Rileva se cambia il pathname e invia "pageview" (utile per Single Page Application)

      function onPathChange() {
        if (lastPath !== loc.pathname) {
          lastPath = loc.pathname;
          trackPageview();
        }
      }

      var originalPushState = window.history.pushState;
      if (originalPushState) {

        onPathChange();
