    var h = Math.floor(seconds / 3600);
    var m = Math.floor((seconds % 3600) / 60);
    var s = Math.floor(seconds % 60);
    return (h < 10 ? "0" + h : h) + ":" +
           (m < 10 ? "0" + m : m) + ":" +
           (s < 10 ? "0" + s : s);
  }

  function pushToDataLayer(status, percent, video) {
    const includePercent = (status === '⏳ progress');
      
    var data = {
      event: 'video_watch_event',
      videoStatus: status,
      ...(includePercent && percent !== null && percent !== undefined ? { videoPercent: percent } : {}),
      videoCurrentTime: formatTime(video.currentTime),
      videoDuration: video.duration ? formatTime(video.duration) : null,
      videoTitle: getPageVideoTitle(),
      videoUrl: window.location.href,
      videoId: getVideoID(),
      activeUserId: getActiveUserId()
    };
    console.log('[Video Event]', data); // Debug log
    window.dataLayer = window.dataLayer || [];
    window.dataLayer.push(data);
  }

  function setupVideoTracking(video) {
    if (video.__trackingSet) return;
    video.__trackingSet = true;

    var milestones = [25, 50, 75, 100];
    var reported = {};

    video.addEventListener('play', function () {
      pushToDataLayer('▶️ play', null, video);
    });

    video.addEventListener('pause', function () {
      // Don't report pause if video is at or near end
      if (video.duration && video.currentTime >= video.duration - 0.5) {
        return;
      }
      pushToDataLayer('⏸️ paused', null, video);
    });

    video.addEventListener('ended', function () {
      if (!reported[100]) {
        reported[100] = true;
        console.log('✅ ended event fired');
        pushToDataLayer('✅ completed', 100, video);
      }
    });

    video.addEventListener('timeupdate', function () {
      if (!video.duration) return;
      var percent = Math.floor((video.currentTime / video.duration) * 100);

      for (var i = 0; i < milestones.length; i++) {
        var m = milestones[i];
        if (percent >= m && !reported[m]) {
          reported[m] = true;
          const status = m === 100 ? '✅ completed' : '⏳ progress';
          pushToDataLayer(status, m, video);
        }
      }

      // Fallback in case 'ended' event doesn't fire
      if (!reported[100] && video.duration - video.currentTime < 1) {
        reported[100] = true;
        console.log('✅ fallback: near end detected');
        pushToDataLayer('✅ completed', 100, video);
      }
    });
  }

  // Watch for dynamically added videos
  var observer = new MutationObserver(function () {
    var videos = document.querySelectorAll('video');
    for (var i = 0; i < videos.length; i++) {
      setupVideoTracking(videos[i]);
    }
  });

  observer.observe(document.body, { childList: true, subtree: true });

  // Setup existing videos on load
  var initialVideos = document.querySelectorAll('video');
  for (var i = 0; i < initialVideos.length; i++) {
    setupVideoTracking(initialVideos[i]);
  }
})();

function PT_LikeComments(self,type,comment_id){
  if (!self || (type != 'up' && type != 'down') || !comment_id){
    return false;
  }

  else if (!$('#main-container').attr('data-logged') && $('#main-url').val()) {
    window.location.href = PT_Ajax_Requests_File() + 'login?to=' + $('#main-url').val();
    return false;
  }

  var self = $(self);
  $.ajax({
    url: 'https://wesharez.com/aj/comment-like-system/' + type,
    type: 'POST',
    dataType: 'json',
    data: {id:comment_id}
  })
  .done(function(data) {
    if (data.status == 200 && type == 'up' && data.type == 'added_like'){
      self.find('span').text((data.up));
      $('[data-comment-dislikes="'+comment_id+'"]').find('span').text((data.down));
      self.find('svg.feather').addClass('active');
      self.siblings().find('svg.feather').removeClass('active');
      
    }

    else if (data.status == 200 && type == 'up' && data.type == 'deleted_like'){
      self.find('span').text((data.up));
      $('[data-comment-dislikes="'+comment_id+'"]').find('span').text((data.down));
      self.find('svg.feather').removeClass('active');
      self.siblings().find('svg.feather').removeClass('active');
    }

    else if (data.status == 200 && type == 'down' && data.type == 'added_dislike'){
      self.find('span').text((data.down));
