<script>
document.addEventListener('DOMContentLoaded', function() {
    const items = document.querySelectorAll('.slider-item');
    const dots = document.querySelectorAll('.dot');
    let current = 0;

    if (items.length <= 1) return;

    // Ustawienie pierwszej kropki jako aktywnej
    if(dots.length > 0) dots[0].style.background = '#fff';

    function changeSlide(next) {
        // Ukryj obecny
        items[current].style.opacity = '0';
        items[current].style.zIndex = '1';
        if(dots.length > 0) dots[current].style.background = 'rgba(255,255,255,0.4)';

        current = next;

        // Pokaż nowy
        items[current].style.opacity = '1';
        items[current].style.zIndex = '5';
        if(dots.length > 0) dots[current].style.background = '#fff';
    }

    function autoPlay() {
        let next = (current + 1) % items.length;
        changeSlide(next);
    }

    let timer = setInterval(autoPlay, 10000);

    // Klikanie w kropki
    dots.forEach((dot, index) => {
        dot.addEventListener('click', () => {
            clearInterval(timer);
            changeSlide(index);
            timer = setInterval(autoPlay, 10000);
        });
    });
});
