// ===== 유틸 =====
const ensureHttp = (u) => {
  if (!u) return "";
  return /^https?:\/\//i.test(u) ? u : `https://${u}`;
};
const toHostname = (u) => {
  try { return new URL(ensureHttp(u)).host.replace(/^www\./, ''); } catch { return ''; }
};
const hexToRgba = (hex, a=1) => {
  if (!hex) return `rgba(138,180,255,${a})`;
  const h = hex.replace('#','');
  const bigint = parseInt(h.length===3 ? h.split('').map(c=>c+c).join('') : h, 16);
  const r = (bigint >> 16) & 255, g = (bigint >> 8) & 255, b = bigint & 255;
  return `rgba(${r},${g},${b},${a})`;
};
const getUrls = (t) => {
  const list = Array.isArray(t.url) ? t.url : [t.url];
  return list.filter(Boolean).map(ensureHttp);
};

function tileEl(t) {
  const a = document.createElement('a');
  a.className = 'tile';

  const urls = getUrls(t);
  a.href = urls[0] || '#';
  a.target = '_blank';
  a.rel = 'noopener noreferrer';
  a.style.setProperty('--tile-accent', hexToRgba(t.color || '#8ab4ff', .55));
  a.dataset.id = t.id;

  const hosts = urls.map(toHostname).filter(Boolean);
  const metaText = hosts.length <= 2 ? hosts.join(' • ') : `${hosts[0]} 외 ${hosts.length - 1}개`;
  a.setAttribute('aria-label', `${t.title} 열기 (${metaText})`);

  // === 핵심: 배열의 모든 URL 새 탭으로 열기 ===
  a.addEventListener('click', (e) => {
    if (urls.length > 1) {
      e.preventDefault();
      urls.forEach(u => {
        const win = window.open(u, '_blank', 'noopener');
        if (win) { try { win.opener = null; } catch(e){} }
      });
    }
  });

  let iconEl;
  if (t.img) {
    const bg = document.createElement('div');
    bg.className = 'bg';
    bg.style.backgroundImage = `url(${t.img})`;
    a.prepend(bg);
    iconEl = document.createElement('div');
    iconEl.className = 'emoji';
    iconEl.textContent = t.emoji || '🔗';
    iconEl.style.position = 'relative';
  } else {
    iconEl = document.createElement('div');
    iconEl.className = 'emoji';
    iconEl.textContent = t.emoji || '🔗';
  }

  const title = document.createElement('h3');
  title.className = 'title';
  title.textContent = t.title;

  const meta = document.createElement('div');
  meta.className = 'meta';
  meta.textContent = metaText || toHostname(urls[0] || '');

  if (Array.isArray(t.url) && t.url.length > 1) {
    const badge = document.createElement('div');
    badge.className = 'badge';
    badge.textContent = `${t.url.length}개 링크`;
    a.append(badge);
  }

  a.append(iconEl, title, meta);
  return a;
}
