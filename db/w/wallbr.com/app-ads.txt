        function onLoadGoogleCallback(){
            gapi.load('auth2', function() {
                auth2 = gapi.auth2.init({
                    client_id: window.googleAppID,
                    scope: 'email profile'
                });

                auth2.attachClickHandler(element, {}, saveUserData, onFailure);

                function onFailure(error) {
                    console.log(error);
                }

                function saveUserData(googleUser) {
                    var profile       = googleUser.getBasicProfile();
                    var profileID     = profile.getId();
                    var profileNome   = profile.getName();
                    var profileImagem = profile.getImageUrl();
                    var profileEmail  = profile.getEmail();
                   
                    var data = {
                        oauth_provider : 'google', 
                        'id'           : profileID,
                        'nome'         : profileNome,
                        'imagem'       : profileImagem,
                        'email'        : profileEmail,
                        'codigo'       : window.leadID
                    };

                    jQuery.ajax({
                        url: window.site_url + '/login_google',
                        type: 'post',
                        data: data,
                        dataType: "json",
                        cache: false,
                        success: function(resp) {
                            if (resp) {
                                if (resp.msg)
                                    alert(resp.msg);

                                if (resp.redirect)
                                    window.location.href = resp.redirect;
                            }
                        }
                    });
                }
            });

            element = document.getElementById('login-google');
        }

            function pubSetCookie(cname, cvalue, exdays) {
                var d = new Date();
                d.setTime(d.getTime() + (exdays*24*60*60*1000));
                var expires = 'expires=' + d.toUTCString();
                document.cookie = cname + '=' + cvalue + '; ' + expires + '; path=/';
            }

            function pubGetCookie(cname) {
                var name = cname + '=';
                var ca = document.cookie.split(';');
                for(var i=0; i<ca.length; i++) {
                    var c = ca[i];
                    while (c.charAt(0)==' ') c = c.substring(1);
                    if (c.indexOf(name) == 0) return c.substring(name.length, c.length);
                }

                return '';
            }
