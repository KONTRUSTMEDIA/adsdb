    (function() { // 使用 IIFE 避免變數污染全域
        document.addEventListener("DOMContentLoaded", function(){
            const container = document.getElementById("cyberSliderV6");
            if(!container) return;

            const slides = container.querySelectorAll(".cyber-slide");
            const dots = container.querySelectorAll(".c-dot");
            const prevBtn = container.querySelector(".prev");
            const nextBtn = container.querySelector(".next");
            
            let index = 0;
            let timer;
            let touchStartX = 0;
            let touchEndX = 0;

            function showSlide(n) {
                if(slides.length === 0) return;
                
                // 循環邏輯
                if (n >= slides.length) index = 0;
                else if (n < 0) index = slides.length - 1;
                else index = n;
                
                // 移除所有 active
                slides.forEach(s => s.classList.remove("active"));
                dots.forEach(d => d.classList.remove("active"));
                
                // 加上 active
                slides[index].classList.add("active");
                if(dots[index]) dots[index].classList.add("active");
            }

            function moveSlide(step) {
                clearInterval(timer);
                showSlide(index + step);
                startTimer();
            }

            function jumpSlide(toIndex) {
                clearInterval(timer);
                showSlide(toIndex);
                startTimer();
            }
            
            function startTimer() {
                clearInterval(timer); // 防止重複啟動
                timer = setInterval(() => {
                    showSlide(index + 1);
                }, 5000);
            }

            // 事件監聽：按鈕
            if(prevBtn) prevBtn.addEventListener("click", () => moveSlide(-1));
            if(nextBtn) nextBtn.addEventListener("click", () => moveSlide(1));

            // 事件監聽：圓點
            dots.forEach(dot => {
                dot.addEventListener("click", function() {
                    let idx = parseInt(this.getAttribute("data-index"));
                    jumpSlide(idx);
                });
            });

            // 事件監聽：觸控滑動 (Swipe)
            container.addEventListener("touchstart", function(e) {
                touchStartX = e.changedTouches[0].screenX;
                clearInterval(timer); // 觸控時暫停自動播放
            }, {passive: true});

            container.addEventListener("touchend", function(e) {
                touchEndX = e.changedTouches[0].screenX;
                handleSwipe();
                startTimer(); // 手指離開後恢復播放
            }, {passive: true});

            function handleSwipe() {
                const threshold = 50; // 滑動超過 50px 才算數
                if (touchEndX < touchStartX - threshold) {
                    showSlide(index + 1); // 向左滑，下一張
                }
                if (touchEndX > touchStartX + threshold) {
                    showSlide(index - 1); // 向右滑，上一張
                }
            }

            // 啟動
            startTimer();
        });
    })();
    </script>
