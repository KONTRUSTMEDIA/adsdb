          AD_WIDTH: 960,
          AD_HEIGHT: 540,
          AD_DISPLAY: 'default', //default, fullscreen, center, fill
          LOADING_TEXT: 'loading advertisement',
          PREROLL_ELEM: function () { return document.getElementById('preroll') },
          AIP_COMPLETE: function (evt) {
            console.log(evt);
            /*******************
             ***** WARNING *****
            *******************
            Please do not remove the PREROLL_ELEM
            from the page, it will be hidden automaticly.
            If you do want to remove it use the AIP_REMOVE callback.

            window.dispatchEvent(event);
          },
          AIP_REMOVE: function () {
            // Here it's safe to remove the PREROLL_ELEM from the page if you want. But it's not recommend.
          }
        });

      const REQUEST_CONFIG = {
        method: 'HEAD',
        mode: 'no-cors'
      };
      let noDollars = false;
      fetch(`https://ads.google.com?=${new Date().getTime()}`, REQUEST_CONFIG)
        .then((response) => {
          noDollars = response.redirected;
        })
        .catch((e) => {
          noDollars = true;
        });

      window.isMonetizedBlocked = () => {
        if (!noDollars) {
          const event = new Event("aip-load-succeeded");
          window.dispatchEvent(event);
          return false;
        } else {
          //Adlib didnt load this could be due to an adblocker, timeout etc.

          window.dispatchEvent(event);
          return true;
        }
      }

      window.show_preroll = () => {
        //check if the adslib is loaded correctly or blocked by adblockers etc.
        if (!noDollars) {
          aiptag.cmd.player.push(function () { aiptag.adplayer.startPreRoll(); });
