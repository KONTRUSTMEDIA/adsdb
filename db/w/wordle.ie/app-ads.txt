          const prevRow = Math.floor((index - 1) / 5);
          const currentRow = Math.floor(index / 5);
          if (prevRow === currentRow) {
            inputs[index - 1].focus();
          }
        }
        // Handle Enter to analyze
        if (e.key === 'Enter') {
          analyzeSolver();
        }
      });
    });

    // Button handlers
    document.getElementById('analyze-btn').addEventListener('click', analyzeSolver);
    document.getElementById('reset-btn').addEventListener('click', resetSolver);
    document.getElementById('quick-hint-btn').addEventListener('click', showQuickHint);
  }

  function cycleTileState(tile) {
    const states = ['empty', 'absent', 'present', 'correct'];
    const currentState = tile.dataset.state;
    const currentIndex = states.indexOf(currentState);
    const nextIndex = (currentIndex + 1) % states.length;
    tile.dataset.state = states[nextIndex];

    // Auto-fill with sample letter for demonstration if empty
    const input = tile.querySelector('.solver-input');
    if (states[nextIndex] !== 'empty' && !input.value) {
      const sampleLetters = ['S', 'T', 'A', 'R', 'E'];
      const position = parseInt(tile.dataset.position);
      input.value = sampleLetters[position] || 'A';
    } else if (states[nextIndex] === 'empty') {
      input.value = '';
    }
  }

  function resetSolver() {
    const tiles = document.querySelectorAll('.solver-tile');
    const inputs = document.querySelectorAll('.solver-input');
    
    tiles.forEach(tile => {
      tile.dataset.state = 'empty';
    });
    
    inputs.forEach(input => {
      input.value = '';
    });

    currentGuesses = [];
    showEmptyResults();
    hideAnalytics();
  }

  function collectGuessData() {
    const rows = document.querySelectorAll('.solver-row');
    const guesses = [];

    rows.forEach((row, rowIndex) => {
      const tiles = row.querySelectorAll('.solver-tile');
      const inputs = row.querySelectorAll('.solver-input');
      let hasContent = false;
      let guess = {
        word: '',
        feedback: []
      };

      tiles.forEach((tile, tileIndex) => {
        const input = inputs[tileIndex];
        const letter = input.value.toUpperCase();
        const state = tile.dataset.state;

        if (letter && state !== 'empty') {
          hasContent = true;
          guess.word += letter;
          guess.feedback.push({
            letter: letter,
            position: tileIndex,
            state: state
          });
        } else if (hasContent) {
          guess.word += ' ';
          guess.feedback.push({
            letter: ' ',
            position: tileIndex,
            state: 'empty'
          });
        }
      });

      if (hasContent && guess.word.length === 5) {
        guesses.push(guess);
      }
    });

    return guesses;
  }

  function analyzeSolver() {
    if (isAnalyzing) return;
    
    const guessData = collectGuessData();
    if (guessData.length === 0) {
      showError('Please enter at least one guess with tile colors!');
      return;
    }

    isAnalyzing = true;

    // Generate suggestions immediately - no fake delay needed
    const suggestions = generateSuggestions(guessData);
    showSuggestions(suggestions);
    showAnalytics();
    isAnalyzing = false;

    // Track usage for analytics
    if (typeof gtag !== 'undefined') {
      gtag('event', 'wordle_solver_used', {
        'guesses_entered': guessData.length,
        'page_location': window.location.href
      });
    }
  }

  function generateSuggestions(guesses) {
    const suggestions = [];
    const usedLetters = new Set();
    const correctLetters = new Map(); // position -> letter
    const presentLetters = new Set();
    const absentLetters = new Set();

    // Analyze constraints from guesses
    guesses.forEach(guess => {
      guess.feedback.forEach(fb => {
        if (fb.letter === ' ') return;
        
        usedLetters.add(fb.letter);
        
        switch (fb.state) {
          case 'correct':
            correctLetters.set(fb.position, fb.letter);
            break;
          case 'present':
            presentLetters.add(fb.letter);
            break;
          case 'absent':
            absentLetters.add(fb.letter);
            break;
        }
      });
    });

    // Filter words based on constraints
    const validWords = COMMON_WORDS.filter(word => {
      // Check if word contains absent letters
      for (let letter of absentLetters) {
        if (word.includes(letter) && !presentLetters.has(letter) && !Array.from(correctLetters.values()).includes(letter)) {
          return false;
        }
      }

      // Check if word has correct letters in correct positions
      for (let [position, letter] of correctLetters) {
        if (word[position] !== letter) {
          return false;
        }
      }

      // Check if word contains all present letters but not in guessed positions
      for (let letter of presentLetters) {
        if (!word.includes(letter)) {
          return false;
        }
      }

      // Check that present letters are not in positions where they were marked yellow
      for (let guess of guesses) {
        for (let fb of guess.feedback) {
          if (fb.state === 'present' && word[fb.position] === fb.letter) {
            return false;
          }
        }
      }

      return true;
    });

    // Score words based on letter frequency and remaining possibilities
    const scoredWords = validWords.map(word => {
      let score = 0;
      const letterCounts = {};
      
      // Calculate frequency score
      for (let letter of word) {
        score += LETTER_FREQUENCY[letter] || 0.001;
        letterCounts[letter] = (letterCounts[letter] || 0) + 1;
      }

      // Penalty for repeated letters (unless we know there are repeats)
      for (let count of Object.values(letterCounts)) {
        if (count > 1) {
          score *= 0.8;
        }
      }

      // Bonus for not using already-guessed letters (exploration)
      let newLetters = 0;
      for (let letter of word) {
        if (!usedLetters.has(letter)) {
          newLetters++;
        }
      }
      score += newLetters * 0.1;

      return {
        word: word,
        score: score,
        newLetters: newLetters,
        probability: Math.min(95, Math.round(score * 100))
      };
    });

    // Sort by score and return top suggestions
    scoredWords.sort((a, b) => b.score - a.score);
    return scoredWords.slice(0, 8);
  }

  function animateCounter(element, target, suffix) {
    let current = 0;
    const increment = target / 50;
    const timer = setInterval(() => {
      current += increment;
      if (current >= target) {
        current = target;
        clearInterval(timer);
      }
      
      if (suffix === '%') {
        element.textContent = Math.floor(current) + suffix;
      } else if (target % 1 !== 0) {
        element.textContent = current.toFixed(1) + suffix;
      } else {
        element.textContent = Math.floor(current).toLocaleString() + suffix;
      }
    }, 20);
  }

        if (a !== answer) {
          a.classList.remove('active');

      // Toggle current FAQ
      if (isExpanded) {
        this.setAttribute('aria-expanded', 'false');

          let current = 0;
          const increment = target / 100;
          const timer = setInterval(() => {
            current += increment;
            if (current >= target) {
              current = target;
              clearInterval(timer);
            }
            if (typeof target === 'string' && target.includes('%')) {
              element.textContent = Math.floor(current) + '%';
            } else if (typeof target === 'string' && target.includes('.')) {
              element.textContent = current.toFixed(1);
            } else {
              element.textContent = Math.floor(current).toLocaleString();
            }
          }, 20);

                number.classList.add('animated');

                let targetValue;

                  targetValue = parseInt(originalText.replace('%', ''));
                } else if (originalText.includes('.')) {
                  targetValue = parseFloat(originalText);
                } else {
                  targetValue = parseInt(originalText.replace(/,/g, ''));
