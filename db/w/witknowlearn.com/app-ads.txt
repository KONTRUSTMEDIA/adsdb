        function r_download(id, link) {
            // Disable both the top and bottom download links
            document.getElementById('top-download').style.pointerEvents = 'none';
            document.getElementById('bottom-download').style.pointerEvents = 'none';

            // Perform the fetch request
            fetch('/resource/download?id=' + id)

                .then(fileUrls => {
                    if (fileUrls.is_show_popup) {
                        let image = $("#img_card").attr('src');
                        $("#popup_image").attr('src', image);
                        $('#afterDownloadModal').modal('show');
                        // return;
                    }
                    if (fileUrls.is_show_popup2) {
                        // alert('sf');
                        // location.reload();
                        // return;
                        download_content(fileUrls.price, fileUrls.board_name, fileUrls.grade_name, fileUrls
                            .subject_name)
                        return false;
                    }
                    fileUrls.fileUrls.forEach((url, index) => {
                        const downloadLink = document.createElement('a');
                        downloadLink.href = url;
                        downloadLink.download = url.split('/').pop();
                        document.body.appendChild(downloadLink);
                        downloadLink.click();
                        document.body.removeChild(downloadLink);
                    });
                })
                .finally(() => {
                    setTimeout(() => {
                        document.getElementById('top-download').style.pointerEvents = 'auto';
                        document.getElementById('bottom-download').style.pointerEvents = 'auto';
                    }, 5000); // 5000 milliseconds = 5 seconds
                });
        }

        function send_otp() {
            var email = document.getElementById("email_fp").value;
            var emailError = document.getElementById("emailError");

            // Clear previous error message
            emailError.classList.add("d-none");

            // Validate email
            if (!email) {
                emailError.textContent = "Please enter your email address.";
                emailError.classList.remove("d-none");
                return;
            }

            // AJAX call to send OTP
            fetch('/send/otp', {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json',
                        'X-CSRF-TOKEN': "Q1teAhOA8alne3P3aHpLCt8K0HGc2llg84M8epNw", // Include CSRF token for security
                    },
                    body: JSON.stringify({
                        email: email,
                    })
                })
                .then(response => response.json())
                .then(data => {
                    if (data.status === "success") {
                        // Redirect to the password reset page
                        $('#forgotPasswordModal').modal('hide');
                        $('#passwordResetModal').modal('show');
                    } else {
                        emailError.textContent = data.message || 'An error occurred';
                        emailError.classList.remove("d-none");
                    }
                })
                .catch(error => {
                    console.error('Error:', error);
                    emailError.textContent = "Something went wrong, please try again.";
                    emailError.classList.remove("d-none");
                });
        }

    <script>
        function reset_password() {
            // Clear previous errors
            document.getElementById("otp_error").innerText = "";
            document.getElementById("new_password_error").innerText = "";
            document.getElementById("confirm_password_error").innerText = "";
            document.getElementById("form_error").innerText = "";

            var email = document.getElementById("email_fp").value;
            var otp = document.getElementById("otp").value;
            var newPassword = document.getElementById("new_password").value;
            var confirmPassword = document.getElementById("confirm_password").value;

            let hasError = false;

            // Client-side validation
            if (!otp || otp.length !== 6) {
                document.getElementById("otp_error").innerText = "OTP must be 6 digits.";
                hasError = true;
            }

            if (!newPassword || newPassword.length < 8) {
                document.getElementById("new_password_error").innerText = "Password must be at least 8 characters.";
                hasError = true;
            }

            if (newPassword !== confirmPassword) {
                document.getElementById("confirm_password_error").innerText = "Passwords do not match.";
                hasError = true;
            }

            if (hasError) return;

            // Send AJAX request
            $.ajax({
                url: '/reset/password',
                type: 'POST',
                dataType: 'json',
                headers: {
                    'X-CSRF-TOKEN': "Q1teAhOA8alne3P3aHpLCt8K0HGc2llg84M8epNw" // Include CSRF token for security
                },
                data: {
                    email: email,
                    otp: otp,
                    new_password: newPassword
                },
                success: function(data) {
                    if (data.status === "success") {
                        document.getElementById("form_error").classList.remove("text-danger");
                        document.getElementById("form_error").classList.add("text-success");
                        document.getElementById("form_error").innerText = "Password reset successfully!";
                        $('#passwordResetModal').modal('hide');
                        $('#loginModal').modal('show');
                    } else {
                        document.getElementById("form_error").innerText = data.message ||
                            "Error resetting password.";
                    }
                },
                error: function(xhr) {
                    if (xhr.responseJSON && xhr.responseJSON.errors) {
                        let errors = xhr.responseJSON.errors;
                        if (errors.otp) document.getElementById("otp_error").innerText = errors.otp[0];
                        if (errors.new_password) document.getElementById("new_password_error").innerText =
                            errors.new_password[0];
                        if (errors.email) document.getElementById("form_error").innerText = errors.email[0];
                    } else {
                        console.log(xhr.responseJSON)
                        console.log(xhr.responseJSON.status)
                        if (xhr.responseJSON.status == 'error') {
                            document.getElementById("form_error").innerText = xhr.responseJSON.message;
                        } else {
                            document.getElementById("form_error").innerText =
                                "Something went wrong. Please try again.";
                        }
                    }
                }
            });
        }
    </script>
