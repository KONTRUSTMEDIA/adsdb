    $(function(){
        var admin = $("#admin-view");
        var pub = $("#public-view");
        if (typeof(CKEDITOR) != "undefined"){
            var instances = [];
            for (i in CKEDITOR.instances){
                var inst = CKEDITOR.instances[i].name;
                instances.push(inst);
            }
        }

        admin.on("click", function(){
            if (!$(this).hasClass("active")){
                $(this).addClass("btn-primary");
                pub.removeClass("btn-primary");
            }
            $(".admin-ctrl").show();
            $("#navlist > ul > li > a > strike").parent("a").show();
            $("a.expired").show();
            $("strike.expired-article").parents("div[class*='article']").show();
            $("strike.expired-headline").parents("table.headline-table").show();
            $("strike.expired-article-newspage").parents("div.glasstop").show();
            $("strike.expired-headline-newspage").parents("div[class*='art']").show();
            $("td.news_headline_text_cell > a > strike").parents("div.listitem").show();
            if (typeof(CKEDITOR) != "undefined"){
                for (i=0; i<instances.length; i++){
                    var inst = instances[i];
                    div = $("#"+inst);
                    div.attr("contenteditable", true);
                    CKEDITOR.inline(inst);
                }
            }
        });

        pub.on("click", function(){
            if (!$(this).hasClass("active")){
                $(this).addClass("btn-primary");
                admin.removeClass("btn-primary");
            }
            $(".admin-ctrl").hide();
            $("#navlist > ul > li > a > strike").parent("a").hide();
            $("a.expired").hide();
            $("strike.expired-article").parents("div[class*='article']").hide();
            $("strike.expired-headline").parents("table.headline-table").hide();
            $("strike.expired-article-newspage").parents("div.glasstop").hide();
            $("strike.expired-headline-newspage").parents("div[class*='art']").hide();
            $("td.news_headline_text_cell > a > strike").parents("div.listitem").hide();
            if (typeof(CKEDITOR) != "undefined"){
                for (i=0; i<instances.length; i++){
                    var inst = instances[i];
                    div = $("#"+inst);
                    CKEDITOR.instances[inst].destroy();
                    div.attr("contenteditable", false);
                }
            }
        });
    });

// Redirects to a secure LA address version of the current page.
function gotoSecureURL(sessionID) {

    // Get their currrent URL
    var loc = location.href;
    
    //If they login from the Password page, take them to the homepage.
    if (loc.toLowerCase().indexOf("/cpanel/password.asp") != -1){
    	loc = loc.toLowerCase().replace("/cpanel/password.asp", "/default.asp");
    }

    //If they login from SecretWord.asp, take them to the selected page
    if (loc.toLowerCase().indexOf("/secretword.asp?file=teamdocs.asp") != -1){
        loc = loc.toLowerCase().replace("/secretword.asp?file=teamdocs.asp&", "/teamdocs.asp?");
    }else if (loc.toLowerCase().indexOf("/secretword.asp?file=schedule.asp") != -1){
        loc = loc.toLowerCase().replace("/secretword.asp?file=schedule.asp&", "/schedule.asp?");
    }else if (loc.toLowerCase().indexOf("/secretword.asp?file=pictures.asp") != -1){
        loc = loc.toLowerCase().replace("/secretword.asp?file=pictures.asp", "/pictures.asp");
    }else if (loc.toLowerCase().indexOf("/secretword.asp") != -1){
        loc = loc.toLowerCase().replace("/secretword.asp", "/roster.asp");
    }
    // Replace the host name portion with LeagueAthletics.com
    var host = location.hostname.toLowerCase();
    if (host.indexOf("dev.leagueathletics.com") == -1)
        loc = loc.replace(host, "LeagueAthletics.com");

    // Change protocal to https or add it if not present.
    loc = loc.replace("http://", "https://");                
    if (loc.indexOf("https://") == -1 )
        loc = "https://" + loc;

    // Strip off any existing snid= values.
    var iSnid = loc.indexOf("&snid=");
    if (iSnid >= 0) {
        var iNd = loc.indexOf("&", iSnid + 1)
        if (iNd == -1) iNd = loc.length - 1;
        loc = loc.substr(0, iSnid) + loc.substr(iNd);
    }
                
    // Add org= parameter if not already there.
    if (loc.indexOf("org=") == -1) {
        if (loc.indexOf("?") == -1) {
            loc += "?";
        } else {
            loc += "&";
        }               

        loc += "org=westwoodlittleleague.com";
    }

    // Redirect to secure version of the page.
    var aLoc = loc.split("#");
    var hashLocation = aLoc[1] || '';
    if (hashLocation) hashLocation = '#' + hashLocation;
    var tempLoc = aLoc[0] + "&snid=" + sessionID + hashLocation;
    location.href = tempLoc;
    //location.href = loc + "&snid=" + sessionID;
}

<p><a href="http://positivecoach.org/">Positive Coaching Alliance</a> is a national non-profit developing &ldquo;Better Athletes, Better People&rdquo; by working to provide all youth and high school athletes a positive, character-building youth sports experience.<br />
<br />
Since its 1998 launch at Stanford University by Founder and CEO Jim Thompson, PCA has impacted more than 5 million youth athletes. PCA reaches youth and high school sport leaders, coaches, athletes, parents, and officials through partnerships with schools and youth sports organizations &nbsp;nationwide that comprise live workshops, online courses , books , and e-communications, providing free Tips and Tools.<br />
<br />
At the center of &nbsp;PCA&rsquo;s &nbsp;work are three models:<br />
&nbsp;</p>

<ul>
	<li>The <strong>Double-Goal Coach</strong>&reg;, who strives to win while also pursuing the more-important goal of life lessons through sports;</li>
	<li>The <strong>Second-Goal Parent</strong>&reg;, who concentrates on life lessons, while letting coaches and athletes focus on competing; and</li>
	<li>The <strong>Triple-Impact Competitor</strong>&reg;, who strives to impact sport on three levels by improving oneself, teammates and the game as a whole.</li>
</ul>

<p>PCA&rsquo;s &nbsp;allies include a National Advisory Board &nbsp;of top athletes, coaches, academicians and business leaders; prominent national youth sports organizations; and corporations and foundations &nbsp;that support its&nbsp; work.&nbsp;<br />
<br />
Westwood Little League provided two PCA led opportunities for parents and coaches in 2013.&nbsp;&nbsp; In 2014, PCA provided WLL&rsquo;s board of directors a Leadership Seminar.&nbsp;&nbsp; All parents and coaches are encouraged to become familiar with PCA principles -- check out the <a href="http://positivecoach.org/">PCA website</a>!</p>
