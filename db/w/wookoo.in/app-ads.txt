	var dataAvailable = false;
	var currentPageNumber = 1;
	$(document).ready(function(){
		$.ajax({
			type: 'GET',
			dataType:'json',
			url: 'data.json',		
		}).done(function(response){
			gameData = response;
			setGameData(currentPageNumber);
		});
		$(window).scroll(async function () {
			if (dataAvailable == true) {
				if ($(window).scrollTop() + 105 >= $(document).height() - $(window).height() - $(".about-div")[0].offsetHeight) {
					currentPageNumber++;
					dataAvailable = false;
					setGameData(currentPageNumber);
				}
			}
		});
	});	

	function setGameData(page){
		game_list = "";
		if(gameData){
			var singlePageData = 30;
			var start = (page - 1) * singlePageData;
			var end = start + singlePageData;
			if(start == 0) { start = -1; }
			var filter_gameData = gameData.filter((el,i)=> i > start && i <= end);
			filter_gameData.forEach(games => {
				game_list += `<div class="col span_1_of_6">
									<div class="box-gallery">
										<div class="box-content-gallery">
											<a href="${games.url}">
											<div class="row-fluid">
												<div class="col span_4_of_4">								
													<img src="${games.img}" class="box-icon-gallery">
												</div>	
												<div class="box-short-description-gallery">${games.name.replace(/-/g,' ')}</div>
											</div>	
											</a>
										</div>
									</div>
								</div>`;
			});
			$("#game_list").append(game_list);
			if(filter_gameData && filter_gameData.length > (singlePageData - 1)){
				dataAvailable = true;
			}
		}
	}
</script>
