		function rearrangeArticles(container) {
			const articles = container.querySelectorAll('.article');
			const leftColumn = document.createElement('div');
			const rightColumn = document.createElement('div');
			leftColumn.classList.add('index-left');
			rightColumn.classList.add('index-right');
			
			articles.forEach((article, index) => {
				if (index < 1) {
					leftColumn.appendChild(article.cloneNode(true));
				} else {
					rightColumn.appendChild(article.cloneNode(true));
				}
			});
			
			container.innerHTML = '';
			container.appendChild(leftColumn);
			container.appendChild(rightColumn);
		}
		
		function rearrange(container) {
			const articles = container.querySelectorAll('.article');
			const aboveColumn = document.createElement('div');
			const underColumn = document.createElement('div');
			aboveColumn.classList.add('index-above');
			underColumn.classList.add('index-under');
			
			articles.forEach((article, index) => {
				if (index < 2) {
					aboveColumn.appendChild(article.cloneNode(true));
				} else {
					underColumn.appendChild(article.cloneNode(true));
				}
			});
			
			container.innerHTML = '';
			container.appendChild(aboveColumn);
			container.appendChild(underColumn);
		}
		
		function processContainers(containers, rearrangeFunction) {
			containers.forEach(container => {
				if (container) {
					rearrangeFunction(container);
				}
			});
		}
		
		processContainers(containersReport, rearrangeArticles);
		processContainers(containersEvent, rearrangeArticles);
		processContainers(containersCompany, rearrange);
