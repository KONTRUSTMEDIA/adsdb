const timetable = {"کراچی":{"seher":"05:43","iftar":"18:31"},"لاہور":{"seher":"05:15","iftar":"17:57"},"اسلام آباد":{"seher":"05:20","iftar":"18:00"},"پشاور":{"seher":"05:26","iftar":"18:05"},"کوئٹہ":{"seher":"05:44","iftar":"18:27"}};
const displayCities = ["کراچی","لاہور","اسلام آباد","پشاور","کوئٹہ"];

const currentCityName = document.getElementById('currentCityName');
const currentCityTime = document.getElementById('currentCityTime');
const currentCityCard = document.getElementById('mobileCurrentCity');

let currentIndex = 0;
const totalCities = displayCities.length;
let isFlashing = false;
let flashTimeout = null;
let flashingCityIndex = null; // which city is currently flashing

function formatAMPM(hourStr, minStr){
    let hours = parseInt(hourStr,10);
    let minutes = parseInt(minStr,10);
    let ampm = hours >= 12 ? 'PM' : 'AM';
    let displayHour = hours % 12; 
    if(displayHour === 0) displayHour = 12;
    return `<span style="direction:ltr; unicode-bidi:bidi-override">(${displayHour.toString().padStart(2,'0')}:${minutes.toString().padStart(2,'0')} ${ampm})</span>`;
}

// Render mobile card content
function renderMobileCity(index){
    const city = displayCities[index];
    if(!timetable[city]) return;

    const seherParts = timetable[city].seher.split(':');
    const iftarParts = timetable[city].iftar.split(':');
    const now = new Date();

    const seher = new Date(); seher.setHours(parseInt(seherParts[0]), parseInt(seherParts[1]), 0, 0);
    const iftar = new Date(); iftar.setHours(parseInt(iftarParts[0]), parseInt(iftarParts[1]), 0, 0);

    currentCityName.textContent = city;

    if(now < seher){
        showCountdown(currentCityTime, seher - now, "انتہاۓسحر", seherParts);
    } else if(now >= seher && now < iftar){
        showCountdown(currentCityTime, iftar - now, "وقتِ افطار", iftarParts);
    } else {
        let tomorrowSeher = new Date(seher.getTime() + 24*60*60*1000);
        showCountdown(currentCityTime, tomorrowSeher - now, "انتہاۓسحر", seherParts);
    }
}

// Trigger a flash for 30 seconds
function triggerFlash(cityElement, message, flashCityIndex, callback){
    if(!cityElement) return;

    flashingCityIndex = flashCityIndex;
    currentIndex = flashCityIndex;

    // Force render the flashing city
    const city = displayCities[currentIndex];
    currentCityName.textContent = city;
    cityElement.innerHTML = `
        <div class="city-name">${city}</div>
        <div class="time-display">${message}</div>
    `;
    cityElement.classList.add("flash");

    isFlashing = true;

    if(flashTimeout) clearTimeout(flashTimeout);
    flashTimeout = setTimeout(()=>{
        cityElement.classList.remove("flash");
        isFlashing = false;
        flashingCityIndex = null;

        // After flash, restore normal timer
        renderMobileCity(currentIndex);
        if(callback) callback();
    }, 30000);
}

function showCountdown(element, diff, label, timeParts){
    let h = Math.floor(diff / 3600000),
        m = Math.floor((diff % 3600000) / 60000),
        s = Math.floor((diff % 60000) / 1000);
    h = h.toString().padStart(2,'0'); 
    m = m.toString().padStart(2,'0'); 
    s = s.toString().padStart(2,'0');
    element.innerHTML = `${formatAMPM(timeParts[0],timeParts[1])} ${label}<br>${h}:${m}:${s}`;
}

/* Mobile Timer */
function updateCurrentCityTimer(){
    const now = new Date();
    const city = displayCities[currentIndex];
    if(!timetable[city]) return;

    const seherParts = timetable[city].seher.split(':');
    const iftarParts = timetable[city].iftar.split(':');

    const seher = new Date(); seher.setHours(parseInt(seherParts[0]), parseInt(seherParts[1]),0,0);
    const iftar = new Date(); iftar.setHours(parseInt(iftarParts[0]), parseInt(iftarParts[1]),0,0);

    // If already flashing, pause auto-update
    if(isFlashing) {
        if(flashingCityIndex !== null && currentIndex !== flashingCityIndex){
            currentIndex = flashingCityIndex;
            renderMobileCity(currentIndex);
        }
        return;
    }

    // Flash for Seher end
    if(now >= seher && now < seher.getTime() + 30000 && !isFlashing){
        triggerFlash(currentCityCard, "سحری کا وقت ختم ہو گیا ہے", currentIndex);
        return;
    }

    // Flash for Iftar end
    if(now >= iftar && now < iftar.getTime() + 30000 && !isFlashing){
        triggerFlash(currentCityCard, "روزہ افطار ہو گیا ہے", currentIndex);
        return;
    }

    // Normal countdown
    renderMobileCity(currentIndex);
}

/* Desktop Timer */
function updateDesktopTimers(){
    const desktopCities = document.querySelectorAll('.desktop-city');
    const now = new Date();

    desktopCities.forEach(cityCard=>{
        const city = cityCard.dataset.city;
        const display = cityCard.querySelector('.time-display');
        if(!timetable[city] || !display) return;

        const seherParts = timetable[city].seher.split(':');
        const iftarParts = timetable[city].iftar.split(':');

        const seher = new Date(); seher.setHours(parseInt(seherParts[0]), parseInt(seherParts[1]),0,0);
        const iftar = new Date(); iftar.setHours(parseInt(iftarParts[0]), parseInt(iftarParts[1]),0,0);

        // Seher flash
        if(now >= seher && now < seher.getTime() + 30000){
            cityCard.classList.add("flash");
            display.innerHTML = "سحری کا وقت ختم ہو گیا ہے";
            return;
        }

        // Iftar flash
        if(now >= iftar && now < iftar.getTime() + 30000){
            cityCard.classList.add("flash");
            display.innerHTML = "روزہ افطار ہو گیا ہے";
            return;
        }

        // Normal countdown
        cityCard.classList.remove("flash");
        if(now < seher){
            showCountdown(display, seher - now, "انتہاۓسحر", seherParts);
        } else if(now >= seher && now < iftar){
            showCountdown(display, iftar - now, "وقتِ افطار", iftarParts);
        } else {
            let tomorrowSeher = new Date(seher.getTime() + 24*60*60*1000);
            showCountdown(display, tomorrowSeher - now, "انتہاۓسحر", seherParts);
        }
    });
}

/* Auto slide mobile cities every 5 sec */
let autoSlideInterval = setInterval(()=>{
    if(!isFlashing){
        currentIndex = (currentIndex + 1) % totalCities;
        renderMobileCity(currentIndex);
    }
}, 5000);

/* Swipe support - disabled during flash */
let touchStartX = 0;
currentCityCard.addEventListener('touchstart', function(e){ 
    touchStartX = e.changedTouches[0].screenX; 
});
currentCityCard.addEventListener('touchend', function(e){
    let touchEndX = e.changedTouches[0].screenX;
    let delta = touchEndX - touchStartX;
    if(Math.abs(delta) > 30 && !isFlashing){
        if(delta < 0){
            currentIndex = (currentIndex + 1) % totalCities;
        } else {
            currentIndex = (currentIndex - 1 + totalCities) % totalCities;
        }
        renderMobileCity(currentIndex);
    }
});

/* Update every second */
setInterval(()=>{
    updateCurrentCityTimer();
    updateDesktopTimers();
}, 1000);

/* Initial call */
renderMobileCity(currentIndex);
updateCurrentCityTimer();
updateDesktopTimers();

});
</script>
