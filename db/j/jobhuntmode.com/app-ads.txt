  <script>
    ( function ()
    {
      const alertEl = document.querySelector( '.popFloatStay' );
      if ( !alertEl ) return;

      let shown = false;
      let ticking = false;

      // Show when the user scrolls a bit, not immediately.
      const threshold = Math.max( 75, window.innerHeight * 0.05 )

      function maybeShow ()
      {
        if ( shown ) return;
        if ( window.scrollY > threshold )
        {
          alertEl.classList.add( 'is-visible' );
          shown = true;
          window.removeEventListener( 'scroll', onScroll, { passive: true } );
        }
      }

      function onScroll ()
      {
        if ( ticking ) return;
        ticking = true;
        requestAnimationFrame( function ()
        {
          maybeShow();
          ticking = false;
        } );
      }

      // In case the page loads scrolled (e.g., back/forward cache)
      maybeShow();

      window.addEventListener( 'scroll', onScroll, { passive: true } );
    } )();
  </script>
