// Function to format currency
function formatCurrencyValue(amount) {
    return '₹' + new Intl.NumberFormat('en-IN').format(Math.round(amount));
}

// Function to calculate EMI and update results
function calculateEmiOnChange() {
    // Get current values
    const loanAmount = parseFloat(document.getElementById('loan-amount-input').value);
    const interestRate = parseFloat(document.getElementById('interest-rate-input').value);
    const loanTenure = parseFloat(document.getElementById('loan-tenure-input').value);
    
    // Calculate EMI
    const monthlyRate = interestRate / 12 / 100;
    const tenureMonths = loanTenure * 12;
    const emi = loanAmount * monthlyRate * Math.pow(1 + monthlyRate, tenureMonths) / (Math.pow(1 + monthlyRate, tenureMonths) - 1);
    const totalAmount = emi * tenureMonths;
    const totalInterest = totalAmount - loanAmount;
    
    // Calculate percentages for chart
    const principalPercentage = Math.round((loanAmount / totalAmount) * 100);
    const interestPercentage = 100 - principalPercentage;
    
    // Update display values
    document.getElementById('monthly-emi').textContent = formatCurrencyValue(emi);
    document.getElementById('principal-amount').textContent = formatCurrencyValue(loanAmount);
    document.getElementById('total-interest').textContent = formatCurrencyValue(totalInterest);
    document.getElementById('total-amount').textContent = formatCurrencyValue(totalAmount);
    
    // Update schedule values
    document.getElementById('schedule-loan-amount').textContent = formatCurrencyValue(loanAmount);
    document.getElementById('schedule-interest-rate').textContent = interestRate + '%';
    document.getElementById('schedule-tenure').textContent = loanTenure + ' Years';
    document.getElementById('schedule-emi').textContent = formatCurrencyValue(emi);
    
    // Update chart
    if (emiChart) {
        emiChart.data.datasets[0].data = [principalPercentage, interestPercentage];
        emiChart.update();
    }
}

// Function to update amortization schedule table
function updateAmortizationSchedule() {
    const loanAmount = parseFloat(document.getElementById('loan-amount-input').value);
    const interestRate = parseFloat(document.getElementById('interest-rate-input').value);
    const loanTenure = parseFloat(document.getElementById('loan-tenure-input').value);
    
    const monthlyRate = interestRate / 12 / 100;
    const tenureMonths = loanTenure * 12;
    const emi = loanAmount * monthlyRate * Math.pow(1 + monthlyRate, tenureMonths) / (Math.pow(1 + monthlyRate, tenureMonths) - 1);
    
    let tableHtml = '';
    let remainingPrincipal = loanAmount;
    
    // Generate rows for the first 12 months or full term if shorter
    for (let month = 1; month <= Math.min(12, tenureMonths); month++) {
        const interestForMonth = remainingPrincipal * monthlyRate;
        const principalForMonth = emi - interestForMonth;
        remainingPrincipal -= principalForMonth;
        
        tableHtml += `<tr>
            <td>${month}</td>
            <td>${formatCurrencyValue(emi)}</td>
            <td>${formatCurrencyValue(principalForMonth)}</td>
            <td>${formatCurrencyValue(interestForMonth)}</td>
            <td>${formatCurrencyValue(Math.max(0, remainingPrincipal))}</td>
        </tr>`;
    }
    
    document.getElementById('amortization-table-body').innerHTML = tableHtml;
}

// Function to format currency
function formatInvestmentCurrencyValue(amount) {
    return '₹' + new Intl.NumberFormat('en-IN').format(Math.round(amount));
}

// Function to calculate future value
function calculateInvestmentFutureValue(principal, annualAddition, rateOfReturn, years) {
    const monthlyRate = rateOfReturn / 12 / 100;
    let futureValue = principal * Math.pow(1 + monthlyRate, years * 12);
    
    if (annualAddition > 0) {
        // Calculate with annual additions (simplified for annual additions)
        for (let i = 1; i <= years; i++) {
            futureValue += annualAddition * Math.pow(1 + monthlyRate, (years - i) * 12);
        }
    }
    
    return futureValue;
}

// Function to calculate investment returns and update results
function calculateReturnsOnChange() {
    // Get current values
    const initialInvestment = parseFloat(document.getElementById('initial-investment-input').value);
    const investmentPeriod = parseFloat(document.getElementById('investment-period-returns-input').value);
    const annualAddition = parseFloat(document.getElementById('annual-addition-input').value);

    const debtReturn = parseFloat(document.getElementById('debt-return-input').value);
    const goldReturn = parseFloat(document.getElementById('gold-return-input').value);
    const realEstateReturn = parseFloat(document.getElementById('real-estate-return-input').value);
    
    // Calculate future values
    const equityFutureValue = calculateInvestmentFutureValue(initialInvestment, annualAddition, equityReturn, investmentPeriod);
    const debtFutureValue = calculateInvestmentFutureValue(initialInvestment, annualAddition, debtReturn, investmentPeriod);
    const goldFutureValue = calculateInvestmentFutureValue(initialInvestment, annualAddition, goldReturn, investmentPeriod);
    const realEstateFutureValue = calculateInvestmentFutureValue(initialInvestment, annualAddition, realEstateReturn, investmentPeriod);
    
    // Calculate total investment
    const totalInvestment = initialInvestment + (annualAddition * investmentPeriod);
    
    // Update display values
    document.getElementById('display-initial-investment').textContent = formatInvestmentCurrencyValue(initialInvestment);

    document.getElementById('real-estate-value').textContent = formatInvestmentCurrencyValue(realEstateFutureValue);
    
    // Calculate and update return percentages

            <td>${formatInvestmentCurrencyValue(debtFutureValue - totalInvestment)}</td>
            <td>${debtReturnPercent.toFixed(1)}%</td>

            <td>${formatInvestmentCurrencyValue(goldFutureValue)}</td>

// Function to calculate FD and update results
function calculateFdOnChange() {
    // Get current values
    const principalAmount = parseFloat(document.getElementById('principal-amount-input').value);
    const interestRate = parseFloat(document.getElementById('fd-interest-rate-input').value);
    const tenureYears = parseFloat(document.getElementById('tenure-years-input').value);
    const compoundFrequency = document.getElementById('compound-frequency').value;
    
    // Calculate compounding frequency
    let compoundingPeriodsPerYear = 1; // default yearly
    switch (compoundFrequency) {
        case 'monthly':
            compoundingPeriodsPerYear = 12;
            break;
        case 'quarterly':
            compoundingPeriodsPerYear = 4;
            break;
        case 'half-yearly':
            compoundingPeriodsPerYear = 2;
            break;
        default:
            compoundingPeriodsPerYear = 1; // yearly
    }
    
    // Calculate maturity amount
    const totalCompoundingPeriods = tenureYears * compoundingPeriodsPerYear;
    const ratePerPeriod = interestRate / (compoundingPeriodsPerYear * 100);
    const maturityAmount = principalAmount * Math.pow((1 + ratePerPeriod), totalCompoundingPeriods);
    const interestEarned = maturityAmount - principalAmount;
    
    // Calculate percentages for chart
    const principalPercentage = Math.round((principalAmount / maturityAmount) * 100);
    const interestPercentage = 100 - principalPercentage;
    
    // Update display values
    document.getElementById('maturity-amount').textContent = formatFdCurrencyValue(maturityAmount);
    document.getElementById('fd-principal-amount').textContent = formatFdCurrencyValue(principalAmount);
    document.getElementById('interest-earned').textContent = formatFdCurrencyValue(interestEarned);
    document.getElementById('compounding-frequency').textContent = compoundFrequency.charAt(0).toUpperCase() + compoundFrequency.slice(1);
    
    // Update modal details
    document.getElementById('growth-principal-amount').textContent = formatFdCurrencyValue(principalAmount);
    document.getElementById('growth-interest-rate').textContent = interestRate + '%';
    document.getElementById('growth-tenure').textContent = tenureYears + ' Years';
    document.getElementById('growth-maturity-amount').textContent = formatFdCurrencyValue(maturityAmount);
    
    // Update insights
    const returnPercentage = (interestEarned / principalAmount) * 100;
    document.getElementById('fd-growth-insights').innerHTML = `
        <li>Your investment of ${formatFdCurrencyValue(principalAmount)} will grow to ${formatFdCurrencyValue(maturityAmount)} in ${tenureYears} years</li>
        <li>You will earn ${formatFdCurrencyValue(interestEarned)} as interest</li>
        <li>That's a ${returnPercentage.toFixed(2)}% return on your principal</li>
    `;
    
    // Update chart
    if (fdChart) {
        fdChart.data.datasets[0].data = [principalPercentage, interestPercentage];
        fdChart.update();
    }
}

      const price = parseFloat(stock.ltp || 0).toFixed(2);
      const percent = parseFloat(stock.pChange || 0).toFixed(2);
      const sign = parseFloat(stock.pChange) >= 0 ? "+" : "";
      const arrow = parseFloat(stock.pChange) >= 0 ? "▲" : "▼";

      return `
        <div class="stockCard ${colorClass}">
          <div class="symbol">${stock.symbol}</div>
          <div class="price">₹${price}</div>
          <div class="change">${arrow} ${sign}${percent}%</div>
        </div>
      `;
    }).join("");

    // duplicate content for seamless loop
    ticker.innerHTML = html + html;

  } catch (err) {
    console.error("Error loading stocks:", err);
    ticker.innerHTML = "⚠️ Unable to load live data.";

<script>
async function loadIndexChart(symbol, canvasId, valueId, colorTop, colorBottom) {
  try {
    const response = await fetch(`get_index_data.php?symbol=${encodeURIComponent(symbol)}`);
    const json = await response.json();

    if (!json.chart?.result?.[0]) {
      document.getElementById(valueId).innerHTML = "⚠️ Data not available";
      return;
    }

    const result = json.chart.result[0];
    const timestamps = result.timestamp || [];
    const prices = result.indicators?.quote?.[0]?.close || [];

    if (timestamps.length === 0 || prices.length === 0) return;

    // Format timestamps → readable times
    const labels = timestamps.map(t => {
      const d = new Date(t * 1000);
      return d.getHours() + ":" + String(d.getMinutes()).padStart(2, '0');
    });

    const lastPrice = prices[prices.length - 1];
    const firstPrice = prices[0];
    const change = lastPrice - firstPrice;
    const pChange = ((change / firstPrice) * 100).toFixed(2);
    const arrow = change >= 0 ? "▲" : "▼";
    const color = change >= 0 ? "#00ff99" : "#ff4d4d";

    // Update text under chart
    document.getElementById(valueId).innerHTML = `
      <b style="font-size:20px;">₹${lastPrice.toFixed(2)}</b>
      <span style="color:${color};font-weight:600;margin-left:8px;">
        ${arrow} ${pChange}%
      </span>
    `;

    // Chart setup
    const ctx = document.getElementById(canvasId).getContext('2d');
    const gradient = ctx.createLinearGradient(0, 0, 0, 200);
    gradient.addColorStop(0, colorTop);
    gradient.addColorStop(1, colorBottom);

    new Chart(ctx, {
      type: 'line',
      data: {
        labels: labels,
        datasets: [{
          data: prices,
          borderColor: '#fff',
          backgroundColor: gradient,
          fill: true,
          tension: 0.35,
          borderWidth: 2,
          pointRadius: 0
        }]
      },
      options: {
        responsive: true,
        plugins: { legend: { display: false } },
        scales: {
          x: { display: false },
          y: {
            ticks: { color: '#fff', font: { size: 12 } },
            grid: { color: 'rgba(255,255,255,0.1)' }
          }
        }
      }
    });

  } catch (err) {
    console.error("Error loading chart:", symbol, err);
    document.getElementById(valueId).innerHTML = "⚠️ Unable to load data";
  }
}

(function() {
    'use strict';
    window.addEventListener('load', function() {
        var forms = document.getElementsByClassName('needs-validation');
        var validation = Array.prototype.filter.call(forms, function(form) {
            form.addEventListener('submit', function(event) {
                if (form.checkValidity() === false) {
                    event.preventDefault();
                    event.stopPropagation();
                }
                form.classList.add('was-validated');
            }, false);
        });
    }, false);
})();
</script>
