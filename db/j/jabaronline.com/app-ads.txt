                const now = new Date();

                let target = null, name = "";

                for(let k in m){
                    const [h, mi] = m[k].split(':'); 
                    const pt = new Date(); pt.setHours(h, mi, 0); 
                    if(pt > now){ target = pt; name = k; break; }
                }
                
                if(!target){
                    const [h, mi] = m["Imsak"].split(':');
                    target = new Date(); target.setDate(target.getDate()+1); target.setHours(h, mi, 0);
                    name = "Imsak";
                }

                const diff = target - now;
                const hh = String(Math.floor(diff/3600000)).padStart(2, '0');
                const mm = String(Math.floor((diff%3600000)/60000)).padStart(2, '0');
                const ss = String(Math.floor((diff%60000)/1000)).padStart(2, '0');

                document.getElementById('f-label').innerText = name;

                document.querySelectorAll('.sh-item').forEach(e => e.classList.remove('active'));

            }, 1000);
        } catch(e){}
    }

<script>
document.addEventListener('DOMContentLoaded', function() {
    const container = document.getElementById('sliderScrollContainer');
    const prevBtn = document.getElementById('sliderPrevBtn');
    const nextBtn = document.getElementById('sliderNextBtn');
    
    if (!container) return;

    let autoScrollInterval;
    const scrollSpeed = 0.5; // Kecepatan pelan
    const scrollDelay = 40;
    let isAutoScrolling = true;

    // Fungsi Auto Scroll
    function startAutoScroll() {
        if (!isAutoScrolling) return;
        
        stopAutoScroll();
        autoScrollInterval = setInterval(() => {
            // Cek apakah sudah mentok kanan
            if (container.scrollLeft + container.clientWidth >= container.scrollWidth - 1) {
                // BERHENTI saat mencapai akhir
                stopAutoScroll();
                isAutoScrolling = false;
            } else {
                container.scrollLeft += scrollSpeed;
            }
        }, scrollDelay);
    }

    function stopAutoScroll() {
        clearInterval(autoScrollInterval);
    }

    // Jalankan auto scroll saat load (delay 1.5 detik)
    setTimeout(startAutoScroll, 1500);

    // Hentikan saat user hover/touch
    container.addEventListener('mouseenter', stopAutoScroll);
    container.addEventListener('touchstart', stopAutoScroll, {passive: true});

    // Manual Navigation
    if (prevBtn) {
        prevBtn.addEventListener('click', () => {
            stopAutoScroll();
            container.scrollBy({ left: -180, behavior: 'smooth' });
        });
    }

    if (nextBtn) {
        nextBtn.addEventListener('click', () => {
            stopAutoScroll();
            container.scrollBy({ left: 180, behavior: 'smooth' });
        });
    }
});
</script>

<script>
document.addEventListener('DOMContentLoaded', function() {
    const container = document.getElementById('sliderScrollContainer');
    const prevBtn = document.getElementById('sliderPrevBtn');
    const nextBtn = document.getElementById('sliderNextBtn');
    
    if (!container) return;

    let autoScrollInterval;
    const scrollSpeed = 0.5; // Kecepatan pelan
    const scrollDelay = 40;
    let isAutoScrolling = true;

    // Fungsi Auto Scroll
    function startAutoScroll() {
        if (!isAutoScrolling) return;
        
        stopAutoScroll();
        autoScrollInterval = setInterval(() => {
            // Cek apakah sudah mentok kanan
            if (container.scrollLeft + container.clientWidth >= container.scrollWidth - 1) {
                // BERHENTI saat mencapai akhir
                stopAutoScroll();
                isAutoScrolling = false;
            } else {
                container.scrollLeft += scrollSpeed;
            }
        }, scrollDelay);
    }

    function stopAutoScroll() {
        clearInterval(autoScrollInterval);
    }

    // Jalankan auto scroll saat load (delay 1.5 detik)
    setTimeout(startAutoScroll, 1500);

    // Hentikan saat user hover/touch
    container.addEventListener('mouseenter', stopAutoScroll);
    container.addEventListener('touchstart', stopAutoScroll, {passive: true});

    // Manual Navigation
    if (prevBtn) {
        prevBtn.addEventListener('click', () => {
            stopAutoScroll();
            container.scrollBy({ left: -180, behavior: 'smooth' });
        });
    }

    if (nextBtn) {
        nextBtn.addEventListener('click', () => {
            stopAutoScroll();
            container.scrollBy({ left: 180, behavior: 'smooth' });
        });
    }
});
</script>

<script>
// Auto-scroll for premium cards on mobile
document.addEventListener('DOMContentLoaded', function() {
    const container = document.querySelector('.premium-cards-container');
    
    if (!container) return;
    
    // Only enable auto-scroll on mobile
    function isMobile() {
        return window.innerWidth <= 768;
    }
    
    let scrollInterval;
    let isUserScrolling = false;
    let scrollTimeout;
    
    function startAutoScroll() {
        if (!isMobile()) return;
        
        scrollInterval = setInterval(() => {
            if (isUserScrolling) return;
            
            const cardWidth = container.querySelector('.premium-card-item')?.offsetWidth || 0;
            const gap = 12; // 0.75rem gap
            const scrollAmount = cardWidth + gap;
            
            // Check if we're at the end
            if (container.scrollLeft + container.clientWidth >= container.scrollWidth - 10) {
                // Reset to start
                container.scrollTo({
                    left: 0,
                    behavior: 'smooth'
                });
            } else {
                // Scroll to next card
                container.scrollBy({
                    left: scrollAmount,
                    behavior: 'smooth'
                });
            }
        }, 3000); // Auto-scroll every 3 seconds
    }
    
    function stopAutoScroll() {
        if (scrollInterval) {
            clearInterval(scrollInterval);
            scrollInterval = null;
        }
    }
    
    // Detect user scrolling
    container.addEventListener('scroll', () => {
        isUserScrolling = true;
        
        // Clear existing timeout
        if (scrollTimeout) {
            clearTimeout(scrollTimeout);
        }
        
        // Resume auto-scroll after 5 seconds of no user interaction
        scrollTimeout = setTimeout(() => {
            isUserScrolling = false;
        }, 5000);
    });
    
    // Start auto-scroll on mobile
    if (isMobile()) {
        startAutoScroll();
    }
    
    // Handle window resize
    window.addEventListener('resize', () => {
        stopAutoScroll();
        if (isMobile()) {
            startAutoScroll();
        }
    });
    
    // Pause on hover (desktop) or touch (mobile)
    container.addEventListener('mouseenter', () => {
        if (!isMobile()) return;
        isUserScrolling = true;
    });
    
    container.addEventListener('mouseleave', () => {
        if (!isMobile()) return;
        setTimeout(() => {
            isUserScrolling = false;
        }, 1000);
    });
});
</script>

<script>
document.addEventListener('DOMContentLoaded', function() {
    const container = document.getElementById('mnwScrollContainer');
    const prevBtn = document.getElementById('mnwPrevBtn');
    const nextBtn = document.getElementById('mnwNextBtn');
    
    if (!container) return;

    let autoScrollInterval;
    const scrollSpeed = 0.5; // KECEPATAN DIKURANGI (sebelumnya 1)
    const scrollDelay = 40; // Delay diperlambat juga
    let isAutoScrolling = true;

    // Fungsi Auto Scroll
    function startAutoScroll() {
        if (!isAutoScrolling) return;
        
        stopAutoScroll(); // Clear existing interval
        autoScrollInterval = setInterval(() => {
            // Cek apakah sudah mentok kanan (dengan toleransi 1px)
            if (container.scrollLeft + container.clientWidth >= container.scrollWidth - 1) {
                // BERHENTI saat mencapai akhir
                stopAutoScroll();
                isAutoScrolling = false; // Flag finished
            } else {
                container.scrollLeft += scrollSpeed;
            }
        }, scrollDelay);
    }

    function stopAutoScroll() {
        clearInterval(autoScrollInterval);
    }

    // Jalankan auto scroll saat load
    // Beri sedikit delay awal agar user sadar ada gerakan
    setTimeout(startAutoScroll, 1500);

    // Hentikan saat user menyentuh/hover container
    container.addEventListener('mouseenter', stopAutoScroll);
    container.addEventListener('touchstart', stopAutoScroll, {passive: true});

    // Manual Navigation
    if (prevBtn) {
        prevBtn.addEventListener('click', () => {
            stopAutoScroll();
            container.scrollBy({ left: -220, behavior: 'smooth' }); // Scroll width card desktop
        });
    }

    if (nextBtn) {
        nextBtn.addEventListener('click', () => {
            stopAutoScroll();
            container.scrollBy({ left: 220, behavior: 'smooth' });
        });
    }
});
</script>

                document.addEventListener('DOMContentLoaded', function() {

                    const body = document.body;
                    const html = document.documentElement; // Get HTML element

                    // Function to set the theme
                    function setTheme(theme) {
                        if (theme === 'dark') {
                            body.classList.add('dark-mode');
                            html.classList.add('dark-mode'); // Add to HTML tag
                            if (darkModeToggleDesktop) darkModeToggleDesktop.innerHTML = '<i class="bi bi-sun-fill"></i>';
                            if (darkModeToggleMobile) darkModeToggleMobile.innerHTML = '<i class="bi bi-sun-fill"></i>';
                            if (darkModeToggleMobile) darkModeToggleMobile.innerHTML = '<i class="bi bi-sun-fill"></i>';
                        } else {
                            body.classList.remove('dark-mode');
                            html.classList.remove('dark-mode'); // Remove from HTML tag
                            if (darkModeToggleDesktop) darkModeToggleDesktop.innerHTML = '<i class="bi bi-moon-fill"></i>';
                            if (darkModeToggleMobile) darkModeToggleMobile.innerHTML = '<i class="bi bi-moon-fill"></i>';
                        }
                        localStorage.setItem('theme', theme);
                    }

                    // Check for saved theme preference
                    const savedTheme = localStorage.getItem('theme');
                    if (savedTheme) {
                        setTheme(savedTheme);
                    } else if (window.matchMedia && window.matchMedia('(prefers-color-scheme: dark)').matches) {

            <!-- Global Image Error Handler - Replace broken images with default -->
            <script>
            (function() {
                const defaultImage = 'https://jabaronline.com/assets/img/modib-img-default.jpg';
                
                // Handle all img tags that fail to load
                document.addEventListener('DOMContentLoaded', function() {
                    // Add error handler to all existing images
                    document.querySelectorAll('img').forEach(function(img) {
                        img.addEventListener('error', function() {
                            // Prevent infinite loop if default image also fails
                            if (this.src !== defaultImage) {
                                console.log('Image failed to load, using default:', this.src);
                                this.src = defaultImage;
                            }
                        });
                    });
                    
                    // Handle background images (for premium widget and others)
                    document.querySelectorAll('[style*="background-image"]').forEach(function(el) {
                        const bgStyle = el.style.backgroundImage;
                        const urlMatch = bgStyle.match(/url\(['"]?([^'"]+)['"]?\)/);
                        
                        if (urlMatch && urlMatch[1]) {
                            const bgUrl = urlMatch[1];
                            // Don't check if already using default image
                            if (bgUrl !== defaultImage) {
                                const testImg = new Image();
                                testImg.onerror = function() {
                                    console.log('Background image failed to load, using default:', bgUrl);
                                    el.style.backgroundImage = "url('" + defaultImage + "')";
                                };
                                testImg.src = bgUrl;
                            }
                        }
                    });
                });
                
                // Also handle dynamically added images
                if (window.MutationObserver) {
                    const observer = new MutationObserver(function(mutations) {
                        mutations.forEach(function(mutation) {
                            mutation.addedNodes.forEach(function(node) {
                                if (node.nodeType === 1) { // Element node
                                    if (node.tagName === 'IMG') {
                                        node.addEventListener('error', function() {
                                            if (this.src !== defaultImage) {
                                                this.src = defaultImage;
                                            }
                                        });
                                    }
                                    // Check for images within added nodes
                                    node.querySelectorAll && node.querySelectorAll('img').forEach(function(img) {
                                        img.addEventListener('error', function() {
                                            if (this.src !== defaultImage) {
                                                this.src = defaultImage;
                                            }
                                        });
                                    });
                                }
                            });
                        });
                    });
                    
                    observer.observe(document.body, {
                        childList: true,
                        subtree: true
                    });
                }
            })();
            </script>
