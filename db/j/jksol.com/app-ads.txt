    function signup_newsl() {

        var email = $("#nemail").val();
        var patt = /^\w+([\.-]?\w+)*@\w+([\.-]?\w+)*(\.\w{2,4})+$/;

        var mr = email.match(patt);

        if (mr == null) {
            return false;
        }

        $.ajax({
            url: "https://jksol.com/site/newsletter",
            type: 'POST',
            data: {
                email: email
            },
            success: function(data, textStatus, xhr) {
                if (data == 1) {
                    $('#succ_news').html("Your have successfully signup for newsletter");
                } else {
                    $('#succ_news').html("Your have already signup for newsletter");
                }
            }
        });

        var timer = setInterval(function() {
            var h = $('#succ_news').html();
            if (h != "") {
                $('#succ_news').html("");
                clearInterval(timer);
            }
        }, 2000);
    }
