if (window.location.search && window.location.search.indexOf("?aspxerrorpath=") > -1) {
	var path = window.location.search.substring("?aspxerrorpath=".length);
	if (path.indexOf("/tc.jsp") > -1) {
		window.location.href="https://www.jobmarket.com.hk/Home/TermsofUse/";
	} else if (path.indexOf("/thestandardjobs") > -1) {
		window.location.href="http://www.thestandard.com.hk/Career.php";
	} else if (path.indexOf("/privacy_statement.jsp") > -1) {
		window.location.href="https://www.jobmarket.com.hk/Home/PrivacyStatement/";
	} else if (path.indexOf("/instantpost") > -1) {
		window.location.href="https://www.jobmarket.com.hk/Employer/";
	} else if (path.indexOf("/company/") > -1) {
		window.location.href="https://www.jobmarket.com.hk/Job/?CompanyID=" + path.match(/\/company\/[^\/]+\/(\d+)/)[1];
	} else if (path.indexOf("/job/") > -1) {
		window.location.href="https://www.jobmarket.com.hk/Job/JobDetail?jobid=" + path.match(/\/job\/[^\/]+\/(\d+)/)[1];
	} else {
		window.location.href="https://www.jobmarket.com.hk";
	}
} else {
	window.location.href="https://www.jobmarket.com.hk";
}
