        $(document).ready(function() {
            $('#btnSubscribe').on('click', function(){
                var email = $('#InputNewsletter').val();

                $('#btnSubscribe').addClass('hidden');
                $('.loading-subscribe').removeClass('hidden');

                var formdata = new FormData();
                formdata.append("email", email);

                var myHeaders = {
                    "X-CSRF-TOKEN": $('meta[name="csrf-token"]').attr('content')
                }

                var requestOptions = {
                    method: 'POST',
                    headers: myHeaders,
                    body: formdata
                };

                fetch("https://japanesestation.com/subscribe", requestOptions)
                    .then(response => response.text())
                    .then(result => {
                        var data = JSON.parse(result);
                        console.log(data);
                        $('#btnSubscribe').removeClass('hidden');
                        $('.loading-subscribe').addClass('hidden');

                        if(data.response.status == 'subscribed') {
                            $('#SubscribeModalSuccess').modal('show');

                            $('#InputNewsletter').val("");
                        } else {
                            if(data.response.detail != null) {
                                $(".message-error").empty();
                                $(".message-error").append(data.response.detail);
                                $('.title-error').empty();
                                $('.title-error').append(data.response.title);
                                $('#SubscribeModalError').modal('show');
                            } else {
                                $('#SubscribeModalError').modal('show');
                            }
                        }
                    })
                    .catch(error => {
                        console.log('error', error);
                        $('#btnSubscribe').removeClass('hidden');
                        $('.loading-subscribe').addClass('hidden');

                        $('#SubscribeModalError').modal('show');
                    });

            });

        });
    </script>
