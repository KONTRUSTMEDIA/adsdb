                function hledat_action() {
                    $(vyber_check).prop('checked', false);
                    $(vyber_backdrop + ', ' + hledat_cancel_btn + ', ' + hledat_send_btn).show();
                    $(separator).hide();
                    $(separator_wrap).addClass('w-100');
                }

                    $(hledat_backdrop + ', ' + vyber_cancel_btn).hide();
                    $(separator).show();
                    $(separator_wrap).removeClass('w-100');
                    $(vyber_block + ' label.list-group-item + .collapse').addClass('show');
                    $(vyber_check).prop('checked', false);

                function vyber_action() {
                    if ($(vyber_check + ':checked').length > 0) {
                        $(hledat_backdrop + ', ' + vyber_cancel_btn).show();
                        $(separator).hide();
                    } else {
                        $(hledat_backdrop + ', ' + vyber_cancel_btn).hide();
                        $(separator).show();
                    }
                }

                function vyber_cancel() {
                    $(vyber_backdrop + ', ' + hledat_cancel_btn + ', ' + hledat_send_btn).hide();
                    $(separator).show();
                    $(separator_wrap).removeClass('w-100');
                    //$(hledat_input).val('');

                $(vyber_block + ' label.list-group-item .form-check-input').each(function() {
                    if ($(this).is(':checked')) {
                        var targetDiv = $($(this).closest('.list-group-item').data('bs-target'));
                        targetDiv.removeClass('show');
                    }
                });

                $(vyber_block + ' label.list-group-item .form-check-input').change(function () {
                    if ($(this).is(':checked')) {
                        var targetDiv = $($(this).closest('.list-group-item').data('bs-target'));
                        targetDiv.find('.form-check-input').prop('checked', false);
                    }
                });

                $(hledat_backdrop + ', ' + hledat_cancel_btn + ', ' + hledat_send_btn).hide();

                // Jen pro testování

                    hledat_action();
                    $(hledat_input).val('742 41 Ostrava');
                });

                $(hledat_input).on('input',function(e){
                    if($(this).val()) {
                        hledat_action();
                    } else {
                        $(vyber_backdrop + ', ' + hledat_cancel_btn + ', ' + hledat_send_btn).hide();
                        $(separator_wrap).removeClass('w-100');
                        $(separator).show();
                    }

                });

                if($(hledat_input).val() && $(hledat_input).val().trim() !== '') {
                    hledat_action();

                $(hledat_cancel_btn).on('click', function () {
                    vyber_cancel();
                    $('#lokalita_id').val('');

                $(hledat_send_btn).on('click', function () {
                    $("#sendForm").click();

                vyber_action();

                    vyber_action();
