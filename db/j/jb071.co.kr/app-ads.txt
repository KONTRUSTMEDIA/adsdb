    const loadMoreButton = document.querySelector('.ultp-loadmore-action');

    if (!loadMoreButton) {

        return;
    }

    // 쓰로틀링 함수 정의 (이전과 동일)
    function throttle(func, limit) {
        let inThrottle;
        return function() {
            const args = arguments;
            const context = this;
            if (!inThrottle) {
                func.apply(context, args);
                inThrottle = true;
                setTimeout(() => inThrottle = false, limit);
            }
        }
    }

    // 스크롤 위치 감지 및 버튼 클릭 함수
    function checkScrollPosition() { // 함수 이름도 내용에 맞게 살짝 바꿔봤어!
        const documentHeight = document.documentElement.scrollHeight; // 문서 전체 높이
        const windowHeight = window.innerHeight; // 현재 화면 높이
        const scrollPosition = window.scrollY; // 현재 스크롤된 위치

        // 문서 높이의 90% 지점을 계산합니다.
        const triggerHeight = documentHeight * 0.8; // 0.7는 90%를 의미해요.

        // 현재 스크롤 위치(화면 상단 기준)가 문서 높이의 90% 지점을 넘어섰는지 확인
        // 또는 현재 스크롤된 영역(화면 하단 기준)이 문서 높이의 90% 지점을 넘어섰는지 확인
        // 여기서는 화면 하단이 90% 지점을 넘어설 때를 기준으로 할게요!
        // (현재 스크롤 위치 + 현재 화면 높이)가 문서 전체 높이의 90% 보다 크거나 같을 때!
        const isBeyond90Percent = (scrollPosition + windowHeight >= triggerHeight);


        // 스크롤 위치가 90% 지점 또는 그 이상에 도달했고, 버튼이 보인다면 클릭합니다.
        if (isBeyond90Percent) {
            loadMoreButton.click();
            console.log('스크롤 90% 지점 도달! 더 보기 버튼을 클릭했습니다.');

            // 참고: 이 부분도 이전처럼 '더 보기' 버튼이 한 번만 눌리고 더 이상 필요 없다면
            // 아래 주석을 풀어서 스크롤 이벤트 리스너를 제거할 수 있습니다.
            // window.removeEventListener('scroll', throttledCheck);
        }
    }

    // 스크롤 이벤트에 쓰로틀링 적용 (200ms 간격)
    const throttledCheck = throttle(checkScrollPosition, 200);

    // 스크롤 이벤트 리스너 등록 (쓰로틀링 적용된 함수 사용)
    window.addEventListener('scroll', throttledCheck);

    // 페이지 로드 시 초기 상태 체크 (쓰로틀링 없이 바로 실행)
    // 혹시 페이지 길이가 짧아서 로드되자마자 90% 지점에 도달해 있다면 바로 로드되게!
    checkScrollPosition();
});
