        // Load data
        let appData = [];

        // Fetch and load data
        async function loadData() {
            try {
                // 首页使用精简版数据文件，减小首屏传输体积
                const response = await fetch('results_index.json');
                const rawData = await response.json();

                // Normalize results_index.json to internal appData schema
                appData = rawData.map((item, index) => {
                    // 使用 results_index.json 中记录的原始 id，与 detail/<id>.html 对应
                    const originalId = item.iso_id || item.id || (index + 1);
                    const screenshots = Array.isArray(item.preview_img) ? item.preview_img : [];

                    // Build title_image full path based on preview_img[0] prefix: "data/<id>/"
                    const firstPreview = screenshots[0] || '';
                    const basePrefix = firstPreview
                        ? `${firstPreview.split('/').slice(0, 2).join('/')}/`
                        : '';

                    const titleImageFull = item.title_image
                        ? (String(item.title_image).includes('/') ? String(item.title_image) : `${basePrefix || 'data/'}${item.title_image}`)
                        : '';

                    // Prefer title_image as icon (matches data model), fallback to first preview image
                    const iconPath = titleImageFull || firstPreview || '';

                    // Rating & installs
                    const score = item.rating ? parseFloat(item.rating) : null;
                    const installs = item.usage_count || '';

                    // Updated date (normalize to timestamp for filters/sorting)
                    const updatedTs = item.updated ? Date.parse(item.updated) : NaN;

                    // Text fields
                    const descriptionHTML = item.overview || '';
                    const about_app = item.about || '';
                    const editors_review = item.worth || item.seo_description || '';
                    const seo_description = item.seo_description || '';

                    // Pros & Cons
                    const pros = item.pros_cons && Array.isArray(item.pros_cons.pros) ? item.pros_cons.pros : [];
                    const cons = item.pros_cons && Array.isArray(item.pros_cons.cons) ? item.pros_cons.cons : [];

                    // FAQs
                    const faqs = Array.isArray(item.faqs) ? item.faqs : [];

                    // Comments
                    const comments = Array.isArray(item.comments) ? item.comments : [];

                    // Category & type (lable is now a single string in results.json)
                    let labels = [];
                    if (Array.isArray(item.lable)) {
                        labels = item.lable;
                    } else if (typeof item.lable === 'string' && item.lable.trim()) {
                        labels = [item.lable.trim()];
                    }
                    const category = labels[0] || 'Extensions';

                    return {
                        // IDs（与 detail/<id>.html 一一对应）
                        iso_id: originalId,
                        appId: item.expansion_url || item.web_url || String(index + 1),
                        slug: item.slug || '',
                        // Basic info
                        title: item.title_name || item.seo_title || 'Unknown Extension',
                        icon: iconPath,
                        title_image: titleImageFull,
                        screenshots,
                        url: item.expansion_url || item.web_url || '',
                        expansion_url: item.expansion_url || '',
                        website: item.website || '',
                        developer: item.developer || item.web_text || '',
                        category,
                        type: 'app',
                        // Stats
                        score,
                        installs,
                        version: item.version || '',
                        updated: Number.isFinite(updatedTs) ? updatedTs : null,
                        // Text content
                        descriptionHTML,
                        about_app,
                        editors_review,
                        seo_description,
                        what_is_it: descriptionHTML,
                        how_to_use: '',
                        benefits: item.features || [],
                        features: item.features || [],
                        quick_verdict: item.worth || '',
                        // Structured fields
                        pros,
                        cons,
                        faqs,
                        comments
                    };
                });

                // Remove duplicates by title_name (keep the first occurrence)
                const seenTitles = new Set();
                appData = appData.filter((item) => {
                    const key = (item.title || '').trim().toLowerCase();
                    if (!key) return true;
                    if (seenTitles.has(key)) return false;
                    seenTitles.add(key);
                    return true;
                });

                initHomePage();
            } catch (error) {
                console.error('Error loading data:', error);
                const body = document.body;
                if (body) {
                    body.textContent = '';
                    const msg = document.createElement('div');
                    msg.className = 'loading';
                    msg.textContent = 'Error loading data. Please check if results.json exists.';
                    body.appendChild(msg);
                }
            }
        }

            // Featured Collections - Now rendered as static HTML

        // Parse installs string (e.g., "100,000,000+" -> 100000000)
        function parseInstalls(installs) {
            if (!installs) return 0;
            const num = installs.replace(/,/g, '').replace(/\+/g, '');
            return parseInt(num) || 0;
        }

        // Display Rising Stars
        function displayRisingStars(period = 'week') {
            const container = document.getElementById(period === 'month' ? 'risingStarsMonthGrid' : 'risingStarsWeekGrid');
            if (!container) return;

            // Helper: deduplicate by appId/title
            const dedupeById = (items) => {
                const seen = new Set();
                const result = [];
                for (const item of items) {
                    const key = item.appId || item.iso_id || item.title;
                    if (!key || seen.has(key)) continue;
                    seen.add(key);
                    result.push(item);
                }
                return result;
            };

            // Sort all apps by score and installs
            const allApps = dedupeById([...appData])
                .filter(item => item.score) // Only include apps with scores
                .sort((a, b) => {
                    const scoreA = a.score || 0;
                    const scoreB = b.score || 0;
                    const installsA = parseInstalls(a.installs);
                    const installsB = parseInstalls(b.installs);
                    // Sort by score (60%) and installs (40%)
                    const rankA = scoreA * 0.6 + Math.log10(installsA + 1) * 0.4;
                    const rankB = scoreB * 0.6 + Math.log10(installsB + 1) * 0.4;
                    return rankB - rankA;
                });

            // Last Week: first 6 apps
            // Last Month: next 6 apps (to avoid duplicates)
            const startIndex = period === 'week' ? 0 : 6;
            const risingStars = allApps.slice(startIndex, startIndex + 6);

            renderAppCards(container, risingStars);
        }

        function displayEditorsChoice() {
            const carousel = document.getElementById('editorsChoiceCarousel');
            const dotsContainer = document.getElementById('editorsChoiceDots');
            if (!carousel || !dotsContainer) return;

            // Find apps with editors_review or top rated (2-3 items)
            const candidates = [...appData]
                .filter(item => (item.editors_review || item.seo_description) && item.score > 4.0)
                .sort((a, b) => b.score - a.score)
                .slice(0, 3);

            if (candidates.length === 0) {
                // Fallback to top rated apps
                editorsChoiceItems = [...appData]
                    .sort((a, b) => b.score - a.score)
                    .slice(0, 3);
            } else {
                editorsChoiceItems = candidates;
            }

            if (editorsChoiceItems.length === 0) {
                carousel.parentElement.parentElement.style.display = 'none';
                return;
            }

            // Clear carousel
            carousel.innerHTML = '';
            dotsContainer.innerHTML = '';

            // Create cards
            const tpl = document.getElementById('tplEditorsChoiceCard');
            if (!tpl) return;

            editorsChoiceItems.forEach((item, index) => {
                const card = createEditorsChoiceCard(tpl, item, index);
                if (index === 0) {
                    card.classList.add('active');
                }
                carousel.appendChild(card);
                
                // Create dot
                const dot = document.createElement('button');
                dot.className = 'editors-choice-dot';
                if (index === 0) dot.classList.add('active');
                dot.setAttribute('aria-label', `Go to slide ${index + 1}`);
                dot.addEventListener('click', () => goToEditorsChoiceSlide(index));
                dotsContainer.appendChild(dot);
            });

            // Initialize first slide
            editorsChoiceCurrentIndex = 0;
            
            // Setup navigation
            setupEditorsChoiceNavigation();
        }

        function createEditorsChoiceCard(tpl, item, index) {
            const node = tpl.content.firstElementChild.cloneNode(true);

            // Set background color based on index
            const bgClasses = ['editors-choice-bg-1', 'editors-choice-bg-2', 'editors-choice-bg-3'];
            node.classList.add(bgClasses[index % 3]);

            // Set link target for whole card
            const cardUrl = item.url || '';
            if (cardUrl) {
                node.href = cardUrl;
                node.target = '_blank';
                node.rel = 'noopener noreferrer';
                node.title = item.title ? `Open ${item.title} in store` : 'Open store page';
            } else {
                node.href = '#';
                node.removeAttribute('target');
                node.setAttribute('aria-disabled', 'true');
            }

            // Set icon
            const iconEl = node.querySelector('.editors-choice-card-icon');
            if (iconEl) {
                iconEl.src = item.icon || 'data/logo.png';
                iconEl.alt = item.title || '';
                iconEl.onerror = () => {
                    iconEl.src = 'data/logo.png';
                };
            }

            // Set title
            const titleEl = node.querySelector('.editors-choice-card-title');
            if (titleEl) {
                titleEl.textContent = item.title || '';
            }

            // Set reason (editors_review or seo_description)
            const reasonEl = node.querySelector('.editors-choice-card-reason');
            if (reasonEl) {
                const reason = item.editors_review || item.seo_description || '';
                reasonEl.textContent = reason.length > 150 ? `${reason.substring(0, 150)}...` : reason;
            }

            // Set button text visibility; card already links out
            const btn = node.querySelector('.editors-choice-card-btn');
            if (btn) {
                if (item.url) {
                    btn.title = item.title ? `Open ${item.title} in store` : 'Open store page';
                } else {
                    btn.style.display = 'none';
                }
            }

            return node;
        }

        function setupEditorsChoiceNavigation() {
            const prevBtn = document.getElementById('editorsChoicePrev');
            const nextBtn = document.getElementById('editorsChoiceNext');
            const carousel = document.getElementById('editorsChoiceCarousel');

            if (!carousel && !prevBtn && !nextBtn) return;

            const goNext = () => {
                if (!editorsChoiceItems.length) return;
                editorsChoiceCurrentIndex = (editorsChoiceCurrentIndex + 1) % editorsChoiceItems.length;
                goToEditorsChoiceSlide(editorsChoiceCurrentIndex);
            };

            const goPrev = () => {
                if (!editorsChoiceItems.length) return;
                editorsChoiceCurrentIndex = (editorsChoiceCurrentIndex - 1 + editorsChoiceItems.length) % editorsChoiceItems.length;
                goToEditorsChoiceSlide(editorsChoiceCurrentIndex);
            };

            // Buttons
            if (prevBtn) {
                prevBtn.addEventListener('click', () => {
                    goPrev();
                    resetAutoPlay();
                });
            }

            if (nextBtn) {
                nextBtn.addEventListener('click', () => {
                    goNext();
                    resetAutoPlay();
                });
            }

            // Auto-play
            let autoPlayInterval = null;
            const startAutoPlay = () => {
                if (autoPlayInterval) return;
                autoPlayInterval = setInterval(goNext, 5000);
            };
            const stopAutoPlay = () => {
                if (autoPlayInterval) {
                    clearInterval(autoPlayInterval);
                    autoPlayInterval = null;
                }
            };
            const resetAutoPlay = () => {
                stopAutoPlay();
                startAutoPlay();
            };

            startAutoPlay();

            // Pause on hover
            if (carousel && carousel.parentElement) {
                carousel.parentElement.addEventListener('mouseenter', stopAutoPlay);
                carousel.parentElement.addEventListener('mouseleave', startAutoPlay);
            }

            // Swipe / drag navigation (mouse & touch via pointer events)
            if (carousel) {
                let pointerDown = false;
                let startX = 0;
                const SWIPE_THRESHOLD = 40; // px
                let suppressClick = false;

                const onPointerDown = (e) => {
                    pointerDown = true;
                    startX = e.clientX || (e.touches && e.touches[0]?.clientX) || 0;
                    suppressClick = false;
                };

                const onPointerUp = (e) => {
                    if (!pointerDown) return;
                    pointerDown = false;
                    const endX = e.clientX || (e.changedTouches && e.changedTouches[0]?.clientX) || 0;
                    const deltaX = endX - startX;
                    if (Math.abs(deltaX) < SWIPE_THRESHOLD) return;

                    if (deltaX < 0) {
                        goNext();
                    } else {
                        goPrev();
                    }
                    resetAutoPlay();
                    suppressClick = true;
                };

                const onPointerLeave = () => {
                    pointerDown = false;
                };

                // Pointer events (covers mouse + touch on modern browsers)
                carousel.addEventListener('pointerdown', onPointerDown);
                carousel.addEventListener('pointerup', onPointerUp);
                carousel.addEventListener('pointerleave', onPointerLeave);

                // Fallback for older touch-only environments
                carousel.addEventListener('touchstart', onPointerDown, { passive: true });
                carousel.addEventListener('touchend', onPointerUp);
                carousel.addEventListener('touchcancel', onPointerLeave);

                // Prevent navigation click when swipe detected
                carousel.addEventListener('click', (e) => {
                    if (suppressClick) {
                        e.preventDefault();
                        e.stopPropagation();
                        suppressClick = false;
                    }
                }, true);
            }
        }

                });
            }

            dots.forEach((dot, i) => {

            });
        }

        function setupTrendingTabs() {
            const tabs = document.querySelectorAll('.trending-tab-btn');
            const loadMoreBtn = document.getElementById('trendingLoadMoreBtn');
            const contents = {
                canonical: document.getElementById('trendingCanonical'),
                favicon: document.getElementById('trendingFavicon'),
                opencard: document.getElementById('trendingOpencard')
            };

            let currentTab = 'canonical';
            const DEFAULT_SHOWN = 6;
            const shownCountByTab = {
                canonical: DEFAULT_SHOWN,
                favicon: DEFAULT_SHOWN,
                opencard: DEFAULT_SHOWN
            };

            // Cache results.json for Load More
            let trendingResultsCache = null;
            let trendingResultsPromise = null;

            // Prefer per-tab cursors so each tab loads different items first
            // (0-based indexes into results.json array)
            const cursorByTab = {
                canonical: 4,  // after ids 1-4
                favicon: 10,   // after ids 7-10
                opencard: 16   // after ids 13-16
            };

            const usedIdsGlobal = new Set();
            const usedIdsByTab = {
                canonical: new Set(),
                favicon: new Set(),
                opencard: new Set()
            };

            const parseIdFromHref = (href) => {
                try {
                    const u = new URL(href, window.location.href);
                    const id = Number(u.searchParams.get('id'));
                    return Number.isFinite(id) ? id : null;
                } catch {
                    return null;
                }
            };

            const initUsedIdsFromDOM = () => {
                Object.entries(contents).forEach(([tabName, content]) => {
                    if (!content) return;
                    const cards = Array.from(content.querySelectorAll('.app-card'));
                    cards.forEach((card) => {
                        const id = parseIdFromHref(card.getAttribute('href') || '');
                        if (!id) return;
                        usedIdsGlobal.add(id);
                        usedIdsByTab[tabName]?.add(id);
                    });
                });
            };

            const loadTrendingResults = async () => {
                if (trendingResultsCache) return trendingResultsCache;
                if (!trendingResultsPromise) {
                    trendingResultsPromise = fetch('results.json', { cache: 'no-cache' })
                        .then((r) => {
                            if (!r.ok) throw new Error(`Failed to load results.json: ${r.status}`);
                            return r.json();
                        })
                        .then((data) => {
                            trendingResultsCache = Array.isArray(data) ? data : [];
                            return trendingResultsCache;
                        })
                        .catch((err) => {
                            console.error(err);
                            trendingResultsCache = [];
                            return trendingResultsCache;
                        });
                }
                return trendingResultsPromise;
            };

            const mapResultToCardItem = (id, raw) => {
                const title = raw?.title_name || raw?.title || '';
                const seo_description = raw?.seo_description || raw?.description || raw?.overview || '';
                // results.json uses `rating` (string) for stars; fall back to other possible fields
                const scoreRaw =
                    raw?.score ??
                    raw?.rating ??
                    raw?.avg_rating ??
                    raw?.rating_value ??
                    raw?.stars ??
                    raw?.star ??
                    raw?.rate ??
                    null;
                const score = scoreRaw != null && scoreRaw !== '' ? Number(scoreRaw) : null;

                const previews = Array.isArray(raw?.preview_img) ? raw.preview_img : [];
                const firstPreview = previews[0] || '';
                const base = firstPreview ? String(firstPreview).split('/').slice(0, 2).join('/') + '/' : 'data/';
                const icon = raw?.title_image
                    ? (String(raw.title_image).includes('/') ? String(raw.title_image) : base + String(raw.title_image))
                    : (firstPreview || '');

                const installs =
                    raw?.usage_count ||
                    raw?.installs ||
                    raw?.users ||
                    raw?.user_count ||
                    raw?.installs_text ||
                    '';

                const url =
                    raw?.url ||
                    raw?.expansion_url ||
                    raw?.web_url ||
                    raw?.store_url ||
                    raw?.store_link ||
                    raw?.link ||
                    '';

                // Extract slug from raw data (same as static cards use)
                const slug = raw?.slug || '';

                return {
                    iso_id: id,
                    title,
                    icon,
                    seo_description,
                    score: Number.isFinite(score) ? score : null,
                    installs: installs ? String(installs) : '',
                    url: url ? String(url) : '',
                    slug: slug
                };
            };

            const appendMoreCardsFromResults = async (tabName, needCount) => {
                const content = contents[tabName];
                if (!content) return 0;
                const grid = content.querySelector('.apps-grid');
                const tpl = document.getElementById('tplIndexAppCard');
                if (!grid || !tpl) return 0;

                const results = await loadTrendingResults();
                if (!Array.isArray(results) || results.length === 0) return 0;

                let appended = 0;
                let cursor = cursorByTab[tabName] || 0;

                // Try to find "needCount" unused items starting from cursor.
                while (appended < needCount && cursor < results.length) {
                    const id = cursor + 1; // results.json is 0-based array, ids are 1-based
                    const raw = results[cursor];
                    cursor += 1;

                    if (usedIdsGlobal.has(id)) continue;
                    // Avoid missing title (skip empty)
                    const title = raw?.title_name || raw?.title || '';
                    if (!title) continue;

                    const item = mapResultToCardItem(id, raw);
                    const node = createAppCardNode(tpl, item);
                    node.dataset.trendingDynamic = '1';
                    grid.appendChild(node);

                    usedIdsGlobal.add(id);
                    usedIdsByTab[tabName]?.add(id);
                    appended += 1;
                }

                cursorByTab[tabName] = cursor;
                return appended;
            };

            const hasMoreToLoad = async (tabName) => {
                const results = await loadTrendingResults();
                if (!Array.isArray(results) || results.length === 0) return false;
                // Quick scan forward from cursor for any unused item with a title
                let cursor = cursorByTab[tabName] || 0;
                while (cursor < results.length) {
                    const id = cursor + 1;
                    const raw = results[cursor];
                    if (!usedIdsGlobal.has(id) && (raw?.title_name || raw?.title)) return true;
                    cursor += 1;
                }
                return false;
            };

            const applyTabVisibility = (tabName) => {
                const content = contents[tabName];
                if (!content) return;
                const cards = Array.from(content.querySelectorAll('.app-card'));
                const shown = shownCountByTab[tabName] || DEFAULT_SHOWN;
                cards.forEach((card, idx) => {
                    card.style.display = idx < shown ? '' : 'none';
                });
                if (loadMoreBtn) loadMoreBtn.style.display = 'inline-block';
            };
            
            const resetTab = (tabName) => {
                shownCountByTab[tabName] = DEFAULT_SHOWN;
                const content = contents[tabName];
                if (!content) return;
                const grid = content.querySelector('.apps-grid');
                if (!grid) {
                    applyTabVisibility(tabName);
                    return;
                }
                const currentCount = grid.querySelectorAll('.app-card').length;
                const need = DEFAULT_SHOWN - currentCount;
                if (need <= 0) {
                    applyTabVisibility(tabName);
                } else {
                    // 动态补齐到 DEFAULT_SHOWN 条，再应用可见性
                    appendMoreCardsFromResults(tabName, need).then(() => {
                        applyTabVisibility(tabName);
                    });
                }
            };
            
            tabs.forEach(tab => {
                tab.addEventListener('click', () => {
                    // Remove active class from all tabs
                    tabs.forEach(t => t.classList.remove('active'));
                    tab.classList.add('active');
                    
                    // Hide all content
                    Object.values(contents).forEach(content => {
                        if (content) content.style.display = 'none';
                    });
                    
                    // Show selected content
                    const tabName = tab.dataset.tab;
                    if (contents[tabName]) {
                        contents[tabName].style.display = 'block';
                        currentTab = tabName;
                        resetTab(tabName);
                    }
                });
            });

            if (loadMoreBtn) {
                loadMoreBtn.addEventListener('click', async () => {
                    const content = contents[currentTab];
                    if (!content) return;

                    const shownNow = shownCountByTab[currentTab] || DEFAULT_SHOWN;
                    const targetShown = shownNow + 4;
                    const cardsNow = Array.from(content.querySelectorAll('.app-card'));

                    // If we already have enough cards in DOM, just reveal next 4.
                    if (cardsNow.length >= targetShown) {
                        shownCountByTab[currentTab] = targetShown;
                        applyTabVisibility(currentTab);
                        // Hide button if no more can be loaded and all visible
                        if (loadMoreBtn) {
                            const more = await hasMoreToLoad(currentTab);
                            loadMoreBtn.style.display = more ? 'inline-block' : 'none';
                        }
                        return;
                    }

                    // Otherwise, fetch results.json and append missing cards
                    const need = targetShown - cardsNow.length;
                    await appendMoreCardsFromResults(currentTab, need);

                    // Reveal up to target
                    shownCountByTab[currentTab] = targetShown;
                    applyTabVisibility(currentTab);

                    if (loadMoreBtn) {
                        const more = await hasMoreToLoad(currentTab);
                        // If still can't load more and we've revealed everything we have, hide
                        loadMoreBtn.style.display = more ? 'inline-block' : 'none';
                    }
                });
            }

            // Initialize default tab visibility (show 4)
            initUsedIdsFromDOM();
            resetTab(currentTab);

            // Initial button visibility
            if (loadMoreBtn) {
                loadMoreBtn.style.display = 'inline-block';
            }
        }

        // Render app cards using template in index.html
        function renderAppCards(container, items) {
            if (!container) return;
            const tpl = document.getElementById('tplIndexAppCard');
            if (!tpl) {
                container.textContent = '';
                items.forEach(i => {
                    const div = document.createElement('div');
                    div.className = 'app-card';
                    div.textContent = i.title || '';
                    container.appendChild(div);
                });
                return;
            }
            // 使用 requestAnimationFrame 优化渲染，减少布局偏移
            requestAnimationFrame(() => {
                container.innerHTML = '';
                const frag = document.createDocumentFragment();
                items.forEach(item => {
                    const node = createAppCardNode(tpl, item);
                    frag.appendChild(node);
                });
                container.appendChild(frag);
            });
        }

        // 背景色类计数器，用于循环分配不同的背景色
        let bgClassCounter = 0;

        function createAppCardNode(tpl, item) {
            const node = tpl.content.firstElementChild.cloneNode(true);
            const slugForLink = item.slug || (item.iso_id ? String(item.iso_id) : '');
            const href = slugForLink ? `detail/${slugForLink}.html` : 'detail.html';
            node.setAttribute('href', href);
            node.setAttribute('title', item.title ? `View details for ${item.title}` : 'View app details');
            
            // 为每个卡片分配不同的淡渐变背景色
            const bgClassIndex = (bgClassCounter % 8) + 1;
            node.classList.add(`app-card-bg-${bgClassIndex}`);
            bgClassCounter++;

            const iconEl = node.querySelector('.app-card-icon');
            if (iconEl) {
                iconEl.width = 64;
                iconEl.height = 64;
                iconEl.loading = 'lazy';
                iconEl.decoding = 'async';
                iconEl.style.width = '64px';
                iconEl.style.height = '64px';
                iconEl.style.minWidth = '64px';
                iconEl.style.minHeight = '64px';
                iconEl.style.maxWidth = '64px';
                iconEl.style.maxHeight = '64px';
                iconEl.src = item.icon || 'data/logo.png';
                iconEl.alt = item.title || '';
                iconEl.onerror = () => {
                    iconEl.src = 'data/logo.png';
                    iconEl.style.width = '64px';
                    iconEl.style.height = '64px';
                };
            }

            const titleEl = node.querySelector('.app-card-title');
            if (titleEl) titleEl.textContent = item.title || '';

            const badgeEl = node.querySelector('.app-card-rating-badge');
            if (badgeEl) {
                const rating = Number.isFinite(item.score) ? Number(item.score).toFixed(1) : '—';
                badgeEl.textContent = `★ ${rating}`;
            }

            const descEl = node.querySelector('.app-card-description');
            if (descEl) {
                // Use seo_description field directly
                let descText = item.seo_description || '';
                // Clean up text (CSS will handle line clamping and ellipsis)
                descText = descText.trim();
                descEl.textContent = descText || '';
                descEl.style.display = descText ? 'block' : 'none';
            }

            const aboutEl = node.querySelector('.app-card-about');
            if (aboutEl) {
                let aboutText = '';
                // Check about_app field (mapped from item.about in loadData)
                const aboutRaw = item.about_app || item.about || '';
                if (aboutRaw) {
                    // Extract text from HTML if needed
                    if (typeof aboutRaw === 'string' && aboutRaw.includes('<')) {
                        const tempDiv = document.createElement('div');
                        tempDiv.innerHTML = aboutRaw;
                        aboutText = tempDiv.textContent || tempDiv.innerText || '';
                    } else {
                        aboutText = String(aboutRaw);
                    }
                    // Remove the Jsplugin editor note section if present
                    aboutText = aboutText.split('---')[0].trim();
                }
                aboutEl.textContent = aboutText || '';
                aboutEl.style.display = aboutText ? 'block' : 'none';
            }

            const installsText = item.installs || '';
            const installsWrap = node.querySelector('.app-card-meta-item');
            const installsEl = node.querySelector('.app-card-meta-installs');
            if (installsEl) installsEl.textContent = installsText;
            if (installsWrap) installsWrap.style.display = installsText ? 'inline-flex' : 'none';

            const btn = node.querySelector('.app-card-download-btn');
            if (btn) {
                // Always show button to keep layout consistent with static cards.
                // If we have a store URL, clicking the button goes to store; otherwise it behaves like the card link.
                if (item.url) btn.dataset.url = item.url;
                else btn.removeAttribute('data-url');
                    btn.style.display = '';
            }

            return node;
        }

        // Setup carousel controls
        let currentSlide = 0;
        let slideInterval;

        function setupCarouselControls() {
            const prevBtn = document.querySelector('.carousel-btn-prev');
            const nextBtn = document.querySelector('.carousel-btn-next');
            const dots = document.querySelectorAll('.carousel-dot');
            const slides = document.querySelectorAll('.carousel-slide');

            const showSlide = (index) => {
                slides.forEach((slide, i) => {
                    slide.classList.toggle('active', i === index);
                });
                dots.forEach((dot, i) => {
                    dot.classList.toggle('active', i === index);
                });
                currentSlide = index;
            };

            const nextSlide = () => {
                const next = (currentSlide + 1) % slides.length;
                showSlide(next);
            };

            const prevSlide = () => {
                const prev = (currentSlide - 1 + slides.length) % slides.length;
                showSlide(prev);
            };

            if (nextBtn) {
                nextBtn.addEventListener('click', () => {
                    nextSlide();
                    resetInterval();
                });
            }

            if (prevBtn) {
                prevBtn.addEventListener('click', () => {
                    prevSlide();
                    resetInterval();
                });
            }

            dots.forEach((dot, index) => {
                dot.addEventListener('click', () => {
                    showSlide(index);
                    resetInterval();
                });
            });

            const resetInterval = () => {
                clearInterval(slideInterval);
                slideInterval = setInterval(nextSlide, 5000);
            };

            // Auto-play carousel
            if (slides.length > 1) {
                resetInterval();
            }
        }

        // Initialize Statistics
        function initStatistics() {
            // Calculate total downloads
            const totalDownloads = appData.reduce((sum, item) => {
                return sum + parseInstalls(item.installs);
            }, 0);

            // Format number with commas
            const formatNumber = (num) => {
                return num.toLocaleString('en-US');
            };

            // Update download count
            const downloadEl = document.getElementById('totalDownloads');
            if (downloadEl) {
                // Animate number
                let current = 0;
                const target = totalDownloads;
                const duration = 2000;
                const increment = target / (duration / 16);

                const updateCount = () => {
                    current += increment;
                    if (current < target) {
                        downloadEl.textContent = formatNumber(Math.floor(current));
                        requestAnimationFrame(updateCount);
                    } else {
                        downloadEl.textContent = formatNumber(target);
                    }
                };
                updateCount();
            }

            // Update total programs
            const programsEl = document.getElementById('totalPrograms');
            if (programsEl) {
                programsEl.textContent = formatNumber(appData.length);
            }

            // Update last update time
            const updateEl = document.getElementById('lastUpdate');
            if (updateEl) {
                const now = new Date();
                updateEl.textContent = 'Just now';

                // Update every minute
                setInterval(() => {
                    const minutesAgo = Math.floor((new Date() - now) / 60000);
                    if (minutesAgo === 0) {
                        updateEl.textContent = 'Just now';
                    } else if (minutesAgo === 1) {
                        updateEl.textContent = '1 minute ago';
                    } else {
                        updateEl.textContent = `${minutesAgo} minutes ago`;
                    }
                }, 60000);
            }
        }
