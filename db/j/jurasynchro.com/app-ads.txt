(function(){
  const KEY = 'js_consent';
  const MAX_AGE_DAYS = 183; // ~6 months

  function setCookie(name, value, days){
	const d = new Date();
	d.setTime(d.getTime() + days*24*60*60*1000);
	document.cookie = name + '=' + encodeURIComponent(value) + '; expires=' + d.toUTCString() + '; path=/; SameSite=Lax';
  }
  function getCookie(name){
	const m = document.cookie.match(new RegExp('(?:^|; )' + name.replace(/[$()*+.?[\\\]^{|}]/g,'\\$&') + '=([^;]*)'));
	return m ? decodeURIComponent(m[1]) : '';
  }

  function applyConsent(value){
	// value: "all" or "none"
	setCookie(KEY, value, MAX_AGE_DAYS);
	window.dispatchEvent(new CustomEvent('js:consent', { detail: { value } }));
  }

  function showBanner(){
	const el = document.getElementById('jsCookieBanner');
	if (el) el.style.display = 'block';
  }
  function hideBanner(){
	const el = document.getElementById('jsCookieBanner');
	if (el) el.style.display = 'none';
  }

  // Public API so /cookies page can open it if needed
  window.JSConsent = {
	open: showBanner,
	accept: function(){ applyConsent('all'); hideBanner(); },
	reject: function(){ applyConsent('none'); hideBanner(); },
	value: function(){ return getCookie(KEY) || ''; }
  };

  // init
  const existing = getCookie(KEY);
  if (!existing) showBanner();

  const btnA = document.getElementById('jsCookieAccept');
  const btnR = document.getElementById('jsCookieReject');
  if (btnA) btnA.addEventListener('click', window.JSConsent.accept);
  if (btnR) btnR.addEventListener('click', window.JSConsent.reject);

})();
</script>
