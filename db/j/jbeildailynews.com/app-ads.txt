<script language="javascript" type="text/javascript">

var mainSliderWidth = 0;
var mainSliderHeight = 0;
var mainSliderCurrent = 1;
var mainSliderTotal;
var mainSliderPosition = 0;
var mainSliderTimeInterval = 5;
var mainSliderTimeout;

var longestNewsBlock;

var startCount = 0;
var maxArticlesPerLoad = 5;
var isLoadingMore = false;

$(document).ready(function(){
	mainSliderTotal = $('#slidesContainer a').size();
});

$(window).load(function(){
	handleResize();
	handleResize_custom();
	automateMainSlider();
});

$(window).resize(function(){
    handleResize();
	handleResize_custom();
});

function handleResize_custom() {
	mainSliderWidth = $('#mainSlider').width();
	mainSliderHeight = Math.round(mainSliderWidth * 0.66);
	mainSliderPosition = -(mainSliderWidth * (mainSliderCurrent-1));
	$('#mainSlider').css({'height':mainSliderHeight+'px'});
	$('#slidesContainer').css({'right':mainSliderPosition+'px'});
	$('#mainSlider #slidesContainer a').css({'width':mainSliderWidth+'px'});
	
	longestNewsBlock = 0;
	$('#newsBlocks ul li').css({'height':'auto'});
	$('#newsBlocks ul li').each(function(index, element) {
        if ($(this).height() > longestNewsBlock) {
			longestNewsBlock = $(this).height();
		}
    });
	if (windowWidth > 1100) {
		$('#newsBlocks ul li').css({'height':longestNewsBlock+'px'});
	} else {
		$('#newsBlocks ul li').css({'height':'auto'});
	}
	$('#mainSliderContainer #newsBlocks ul li').css({'height':'auto'});
}

function navMainSlider(dir) {
	clearTimeout(mainSliderTimeout);
	if (dir === 'next') {
		if (mainSliderCurrent == mainSliderTotal) {
			mainSliderCurrent = 1;	
		} else {
			mainSliderCurrent++;	
		}
	} else if (dir === 'prev') {
		if (mainSliderCurrent == 1) {
			mainSliderCurrent = mainSliderTotal;	
		} else {
			mainSliderCurrent--;	
		}
	}
	mainSliderPosition = -(mainSliderWidth * (mainSliderCurrent-1));
	$('#slidesContainer').animate({right:mainSliderPosition+'px'},500,'easeInOutCubic');
	automateMainSlider();
}

function automateMainSlider() {
	mainSliderTimeout = setTimeout(function(){
		navMainSlider('next');
	}, mainSliderTimeInterval*1000);	
}

function loadMoreJdArticles(rootPath) {
	if (startCount >= (maxArticlesPerLoad * 2)) {
		document.location.href = rootPath+'جبيل-دايلي-نيوز';
	} else {
		isLoadingMore = true;
		doLoadMoreJdArticles(rootPath);
	}
}

function doLoadMoreJdArticles(rootPath) {
	startCount += maxArticlesPerLoad;
	$('#mainSliderContainer #newsBlocks ul').append('<div class="loadingIcon"></div>');
	$('#mainSliderContainer #newsBlocks .loadMoreContainer').css({'opacity':0});
	$.ajax({
		url: rootPath+'includes/load_jd_articles.php?start_count='+startCount+'&max_articles_per_load='+maxArticlesPerLoad+'&loadmore=1',
		type: 'GET',
		success: function(data) {
			setTimeout(function(){
				$('#mainSliderContainer #newsBlocks ul .loadingIcon').remove();
				$('#mainSliderContainer #newsBlocks ul li .loadMoreJd').append(data);
				$('#mainSliderContainer #newsBlocks .loadMoreContainer').removeAttr('style');
				isLoadingMore = false;
			}, 1000);
		}
	});
}
