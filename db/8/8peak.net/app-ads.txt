document.getElementById('companyBriefForm').addEventListener('submit', function(e) {
    e.preventDefault();
    const email = document.getElementById('briefEmail').value;
    const messageDiv = document.getElementById('downloadMessage');
    
    // Send email to server
    fetch('/company-brief.php', {
        method: 'POST',
        headers: {
            'Content-Type': 'application/json',
        },
        body: JSON.stringify({ email: email })
    }).then(response => {
        if (response.ok) {
            // Download the PDF
            window.location.href = '/presentation/8peaks-presentation.pdf';
            messageDiv.style.display = 'block';
            messageDiv.style.color = '#10b981';
            messageDiv.textContent = '✓ Thank you! Check your email and the download will begin shortly.';
            document.getElementById('briefEmail').value = '';
        } else {
            throw new Error('Failed to process request');
        }
    }).catch(error => {
        // Even if API fails, allow download
        window.location.href = '/presentation/8peaks-presentation.pdf';
        messageDiv.style.display = 'block';
        messageDiv.style.color = '#10b981';
        messageDiv.textContent = '✓ Thank you! Your download will begin shortly.';
        document.getElementById('briefEmail').value = '';
        
        // Log email for manual follow-up
        console.log('Company brief requested by:', email);
    });
});
</script>
