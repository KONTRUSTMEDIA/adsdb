<script>
(function() {
  const COMPONENT_ID = 'aahrudwvldkhuavdwcaigenblockbedf5f8mkquhp';
  
  class VideoCarousel extends HTMLElement {
    constructor() {
      super();
      this.track = this.querySelector(`.video-carousel__track-${COMPONENT_ID}`);
      this.cards = [...this.querySelectorAll(`.video-carousel__card-${COMPONENT_ID}`)];
      this.videos = [...this.querySelectorAll(`.video-carousel__video-${COMPONENT_ID}`)];
      this.dotsContainer = this.querySelector(`.video-carousel__dots-${COMPONENT_ID}`);
      this.prevBtn = this.querySelector(`.video-carousel__arrow--prev-${COMPONENT_ID}`);
      this.nextBtn = this.querySelector(`.video-carousel__arrow--next-${COMPONENT_ID}`);
      
      this.currentVideo = null;
      this.currentIndex = 0;
      this.desktopColumns = parseInt(this.dataset.desktopColumns) || 3;
      this.isMobile = window.innerWidth < 750;
      this.scrollTimeout = null;
    }

    connectedCallback() {
      if (this.cards.length === 0) return;
      
      this.generatePosters();
      this.initVideoControls();
      this.initNavigation();
      this.updateState();
      
      // Debounced scroll handler
      this.track.addEventListener('scroll', () => {
        clearTimeout(this.scrollTimeout);
        this.scrollTimeout = setTimeout(() => this.updateState(), 50);
      }, { passive: true });
      
      // Resize handler
      let resizeTimeout;
      window.addEventListener('resize', () => {
        clearTimeout(resizeTimeout);
        resizeTimeout = setTimeout(() => {
          const wasMobile = this.isMobile;
          this.isMobile = window.innerWidth < 750;
          if (wasMobile !== this.isMobile) {
            this.initNavigation();
            this.updateState();
          }
        }, 150);
      }, { passive: true });
    }

    generatePosters() {
      // For videos without custom posters, capture first frame
      this.cards.forEach((card) => {
        const video = card.querySelector(`.video-carousel__video-${COMPONENT_ID}`);
        const poster = card.querySelector(`.video-carousel__poster-${COMPONENT_ID}`);
        
        if (!video || !poster) return;
        if (poster.dataset.hasPoster === 'true') return; // Skip if custom poster exists
        
        const canvas = poster;
        const ctx = canvas.getContext('2d');
        
        // Wait for video metadata to load
        const captureFrame = () => {
          // Set canvas size to match video
          canvas.width = video.videoWidth || 720;
          canvas.height = video.videoHeight || 1280;
          
          // Seek to 1 second (or 0 if video is shorter)
          video.currentTime = Math.min(1, video.duration || 1);
        };
        
        const drawFrame = () => {
          try {
            ctx.drawImage(video, 0, 0, canvas.width, canvas.height);
            // Reset video to start
            video.currentTime = 0;
          } catch (e) {
            // CORS or other error - leave canvas blank
            console.warn('Could not generate video poster:', e);
          }
        };
        
        if (video.readyState >= 1) {
          captureFrame();
        } else {
          video.addEventListener('loadedmetadata', captureFrame, { once: true });
        }
        
        video.addEventListener('seeked', drawFrame, { once: true });
      });
    }

    initVideoControls() {
      this.cards.forEach((card, index) => {
        const video = card.querySelector(`.video-carousel__video-${COMPONENT_ID}`);
        if (!video) return;

        // Click on card to play/pause
        card.addEventListener('click', (e) => {
          // Don't interfere with default behavior for links, etc.
          if (e.target.closest('a, button')) return;
          
          if (video.paused) {
            this.playVideo(video, card);
          } else {
            video.pause();
          }
        });

        // Video ended - reset
        video.addEventListener('ended', () => {
          video.currentTime = 0;
          card.classList.remove('is-playing');
          if (this.currentVideo === video) {
            this.currentVideo = null;
          }
        });

        // Sync playing state
        video.addEventListener('play', () => card.classList.add('is-playing'));
        video.addEventListener('pause', () => {
          if (video.currentTime === 0 || video.ended) {
            card.classList.remove('is-playing');
          }
        });
      });
    }

    playVideo(video, card) {
      // Pause any currently playing video
      if (this.currentVideo && this.currentVideo !== video) {
        this.currentVideo.pause();
        this.currentVideo.currentTime = 0;
        const prevCard = this.currentVideo.closest(`.video-carousel__card-${COMPONENT_ID}`);
        if (prevCard) prevCard.classList.remove('is-playing');
      }

      // Unmute and play
      video.muted = false;
      video.play().catch(() => {
        // Autoplay policy fallback - play muted
        video.muted = true;
        video.play().catch(() => {});
      });
      
      this.currentVideo = video;
    }

    initNavigation() {
      this.dotsContainer.innerHTML = '';
      
      const totalItems = this.cards.length;
      const itemsPerPage = this.isMobile ? 1 : this.desktopColumns;
      const totalPages = Math.ceil(totalItems / itemsPerPage);
      
      // Only show dots if there's more than one page
      if (totalPages > 1) {
        const dotCount = Math.min(totalPages, 4);
        
        for (let i = 0; i < dotCount; i++) {
          const dot = document.createElement('button');
          dot.className = `video-carousel__dot-${COMPONENT_ID}`;
          dot.setAttribute('role', 'tab');
          dot.setAttribute('aria-label', `Go to ${this.isMobile ? 'video' : 'page'} ${i + 1}`);
          dot.addEventListener('click', () => this.goToPage(i));
          this.dotsContainer.appendChild(dot);
        }
      }

      // Arrow handlers
      if (this.prevBtn) {
        this.prevBtn.onclick = () => this.navigate(-1);
      }
      if (this.nextBtn) {
        this.nextBtn.onclick = () => this.navigate(1);
      }
    }

    navigate(direction) {
      const itemsPerPage = this.isMobile ? 1 : this.desktopColumns;
      const totalPages = Math.ceil(this.cards.length / itemsPerPage);
      const newPage = Math.max(0, Math.min(totalPages - 1, this.currentIndex + direction));
      this.goToPage(newPage);
    }

    goToPage(pageIndex) {
      const itemsPerPage = this.isMobile ? 1 : this.desktopColumns;
      const targetCardIndex = pageIndex * itemsPerPage;
      
      if (this.cards[targetCardIndex]) {
        const card = this.cards[targetCardIndex];
        const trackRect = this.track.getBoundingClientRect();
        const cardRect = card.getBoundingClientRect();
        
        // Calculate scroll position
        let scrollPos;
        if (this.isMobile) {
          // Center the card on mobile
          scrollPos = card.offsetLeft - (trackRect.width - cardRect.width) / 2;
        } else {
          // Align to start on desktop
          scrollPos = card.offsetLeft;
        }
        
        this.track.scrollTo({
          left: scrollPos,
          behavior: 'smooth'
        });
      }
    }

    updateState() {
      const trackRect = this.track.getBoundingClientRect();
      const scrollLeft = this.track.scrollLeft;
      const maxScroll = this.track.scrollWidth - this.track.clientWidth;
      
      // Find current visible card/page
      const itemsPerPage = this.isMobile ? 1 : this.desktopColumns;
      
      if (this.isMobile) {
        // Find the card closest to center
        const trackCenter = trackRect.left + trackRect.width / 2;
        let closestIndex = 0;
        let closestDist = Infinity;
        
        this.cards.forEach((card, i) => {
          const cardRect = card.getBoundingClientRect();
          const cardCenter = cardRect.left + cardRect.width / 2;
          const dist = Math.abs(cardCenter - trackCenter);
          if (dist < closestDist) {
            closestDist = dist;
            closestIndex = i;
          }
        });
        
        this.currentIndex = closestIndex;
      } else {
        // Desktop: calculate page based on scroll position
        if (this.cards[0]) {
          const cardWidth = this.cards[0].offsetWidth;
          const gap = parseInt(getComputedStyle(this.track).gap) || 0;
          const pageWidth = itemsPerPage * cardWidth + (itemsPerPage - 1) * gap;
          this.currentIndex = Math.round(scrollLeft / pageWidth);
        }
      }

      // Update dots
      const dots = [...this.dotsContainer.querySelectorAll(`.video-carousel__dot-${COMPONENT_ID}`)];
      const totalPages = Math.ceil(this.cards.length / itemsPerPage);
      const activeDotIndex = Math.min(this.currentIndex, dots.length - 1);
      
      dots.forEach((dot, i) => {
        dot.classList.toggle('is-active', i === activeDotIndex);
        dot.setAttribute('aria-selected', i === activeDotIndex);
      });

      // Update arrows (desktop)
      if (this.prevBtn) {
        this.prevBtn.classList.toggle('is-visible', scrollLeft > 10);
      }
      if (this.nextBtn) {
        this.nextBtn.classList.toggle('is-visible', scrollLeft < maxScroll - 10);
      }
    }
  }

  // Register custom element
  if (!customElements.get(`video-carousel-${COMPONENT_ID}`)) {
    customElements.define(`video-carousel-${COMPONENT_ID}`, VideoCarousel);
  }
})();
