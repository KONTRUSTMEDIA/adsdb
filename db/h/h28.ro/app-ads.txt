<script>
(function() {
    var tiers = [
        { name: 'Basic', min: 1, max: 50, pricePer: 10 },
        { name: 'Standard', min: 51, max: 150, pricePer: 15 },
        { name: 'Premium', min: 151, max: 9999, pricePer: 25 }
    ];
    var input = document.getElementById('pricing-nr-angajati');
    var tierEl = document.getElementById('pricing-tier-name');
    var totalEl = document.getElementById('pricing-total');
    var detailEl = document.getElementById('pricing-detail');
    var originalLine = document.getElementById('pricing-original-line');
    var originalEl = document.getElementById('pricing-original');
    var discountLabelEl = document.getElementById('pricing-discount-label');
    var codInput = document.getElementById('pricing-cod-reducere');
    var aplicaBtn = document.getElementById('pricing-aplica-cod');
    var eliminaBtn = document.getElementById('pricing-elimina-cod');
    var codBadge = document.getElementById('pricing-cod-badge');
    var codMsg = document.getElementById('pricing-cod-msg');

    var applied = null;

    function getTotal() {
        var n = parseInt(input.value, 10) || 0;
        if (n < 1) n = 1;
        if (n > 999) n = 999;
        var tier = tiers.find(function(t) { return n >= t.min && n <= t.max; }) || tiers[0];
        return { n: n, tier: tier, total: n * tier.pricePer };
    }

    function update() {
        var data = getTotal();
        var n = data.n, tier = data.tier, total = data.total;
        if (n < 1) n = 1;
        if (n > 999) n = 999;
        input.value = n;

        var displayTotal = total;
        if (applied) {
            var discount = applied.type === 'percent' ? total * applied.value / 100 : Math.min(total, applied.value);
            displayTotal = Math.max(0, Math.round(total - discount));
            originalLine.classList.remove('hidden');
            originalEl.textContent = total.toLocaleString('ro-RO');
            totalEl.textContent = displayTotal.toLocaleString('ro-RO');
            discountLabelEl.textContent = ' ' + applied.label;
            discountLabelEl.classList.remove('hidden');
        } else {
            originalLine.classList.add('hidden');
            totalEl.textContent = total.toLocaleString('ro-RO');
            discountLabelEl.classList.add('hidden');
        }

        tierEl.textContent = tier.name;
        detailEl.textContent = n + ' angajați × ' + tier.pricePer + ' RON/angajat';
    }

    function showMsg(text, isError) {
        codMsg.textContent = text || '';
        codMsg.classList.toggle('error', !!isError);
        codMsg.classList.toggle('hidden', !text);
    }

    if (input) { input.addEventListener('input', update); input.addEventListener('change', update); }

    if (aplicaBtn && codInput) {
        aplicaBtn.addEventListener('click', function() {
            var code = (codInput.value || '').trim();
            if (!code) { showMsg('Introdu codul de reducere.', true); return; }
            var total = getTotal().total;
            var url = '/api/check-discount.php?code=' + encodeURIComponent(code) + '&amount=' + total;
            aplicaBtn.disabled = true;
            fetch(url).then(function(r) { return r.json(); }).then(function(res) {
                aplicaBtn.disabled = false;
                if (res.valid) {
                    applied = { code: code, type: res.type, value: res.value, label: res.label };
                    codBadge.textContent = 'Cod: ' + code;
                    codBadge.classList.remove('hidden');
                    eliminaBtn.classList.remove('hidden');
                    codInput.value = '';
                    showMsg('Cod aplicat: ' + res.label);
                    update();
                } else { showMsg(res.error || 'Cod invalid.', true); }
            }).catch(function() { aplicaBtn.disabled = false; showMsg('Eroare la verificare cod.', true); });
        });
    }

    if (eliminaBtn) {
        eliminaBtn.addEventListener('click', function() {
            applied = null;
            codBadge.classList.add('hidden');
            eliminaBtn.classList.add('hidden');
            showMsg('');
            update();
        });
    }

    update();
})();
</script>
