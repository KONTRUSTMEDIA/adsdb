let currentSliderIndex = 0;
const totalSliderItems = 16;
let sliderInterval;

function showSliderItem(index) {
    if (!totalSliderItems) return;
    for (let i=0; i<totalSliderItems; i++){
        const slide=document.querySelector(`.slider-item[data-index="${i}"]`);
        const pager=document.querySelector(`.pager-link[data-slide-index="${i}"]`);
        if(slide) { slide.style.opacity = i===index? '1':'0'; slide.style.zIndex = i===index? '50':'0'; }
        if(pager) { pager.style.color = i===index?'#fff':'#7c7c7c'; pager.style.backgroundPosition = i===index?'0 -59px':'0 7px'; }
    }
    currentSliderIndex=index;
}

function sliderNext(){currentSliderIndex=(currentSliderIndex+1)%totalSliderItems; showSliderItem(currentSliderIndex);}
function sliderPrev(){currentSliderIndex=(currentSliderIndex-1+totalSliderItems)%totalSliderItems; showSliderItem(currentSliderIndex);}
function sliderGoTo(index,redirect=false){showSliderItem(index); if(redirect){const link=document.querySelector(`.pager-link[data-slide-index="${index}"]`); if(link&&link.dataset.href) window.location.href=link.dataset.href;}}

let touchStartX=0;
const slider=document.querySelector('.slider-wrapper');
if(slider){slider.addEventListener('touchstart', e=>touchStartX=e.touches[0].clientX);
    slider.addEventListener('touchend', e=>{const diff=touchStartX-e.changedTouches[0].clientX; if(Math.abs(diff)>50) diff>0?sliderNext():sliderPrev();});}

sliderInterval=setInterval(sliderNext,5000);
document.querySelectorAll('.pager-link').forEach(link=>{
    link.addEventListener('mouseenter',()=>clearInterval(sliderInterval));
    link.addEventListener('mouseleave',()=>sliderInterval=setInterval(sliderNext,5000));
});

document.addEventListener('keydown', e=>{if(e.key==='ArrowLeft') sliderPrev(); if(e.key==='ArrowRight') sliderNext();});
</script>
