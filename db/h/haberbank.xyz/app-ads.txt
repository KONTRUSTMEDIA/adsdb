        if (typeof window.clarity !== 'function') {
            return;
        }
        Object.entries(tags).forEach(([key, val]) => {
            clarity('set', key, String(val));
        });
        clarity('event', eventName);
    }

    function sendClarityIdentity(identity) {
        if (typeof window.clarity !== 'function') {
            return;
        }

            clarity('identify', identity);
        }
