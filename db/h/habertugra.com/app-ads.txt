<script>
(function() {
    var imsakTimestamp = 1771989300 * 1000;
    var iftarTimestamp = 1771948560 * 1000;

    function updateTimers() {
        var now = new Date().getTime();
        updateSingleTimer(imsakTimestamp, 'imsak-timer', now);
        updateSingleTimer(iftarTimestamp, 'iftar-timer', now);
    }

    function updateSingleTimer(target, elemId, now) {
        var distance = target - now;
        if(distance < 0) distance = (target + 86400000) - now;

        var hours = Math.floor((distance % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));
        var minutes = Math.floor((distance % (1000 * 60 * 60)) / (1000 * 60));
        var seconds = Math.floor((distance % (1000 * 60)) / 1000);

        var el = document.getElementById(elemId);
        if(el) {
            el.innerText = 
                (hours < 10 ? '0' + hours : hours) + ' : ' +
                (minutes < 10 ? '0' + minutes : minutes) + ' : ' +
                (seconds < 10 ? '0' + seconds : seconds);
        }
    }
    
    setInterval(updateTimers, 1000);
    updateTimers();

    // City Change -> Save Name to Cookie
    document.getElementById('ramazanCitySelect').addEventListener('change', function() {
        var cityName = this.value;
        if(cityName) {
            var d = new Date();
            d.setTime(d.getTime() + (30*24*60*60*1000));
            // Store City NAME in cookie
            document.cookie = "ramazan_city_name=" + encodeURIComponent(cityName) + ";expires=" + d.toUTCString() + ";path=/";
            window.location.reload();
        }
    });
})();

function openWidgetModal(title, type, encodedContent) {
    try {
        var myModal = new bootstrap.Modal(document.getElementById('widgetWidgetModal'));
        document.getElementById('widgetWidgetLabel').innerText = title;
        var body = document.getElementById('widgetWidgetBody');
        
        // Decode base64 with UTF-8 support
        var content = decodeURIComponent(escape(window.atob(encodedContent)));
        
        if(type === 'api') {
            body.innerHTML = '<iframe src="'+content+'" style="width:100%; height:100%; border:none;"></iframe>';
        } else if(type === 'html') {
            // Clear body
            body.innerHTML = '';
            
            // Create iframe
            var iframe = document.createElement('iframe');
            iframe.style.width = '100%';
            iframe.style.height = '100%';
            iframe.style.border = 'none';
            body.appendChild(iframe);
            
            // Write content to iframe document
            var doc = iframe.contentWindow.document;
            doc.open();
            doc.write('<!DOCTYPE html><html><head><meta charset="UTF-8"><style>body{margin:0;padding:10px;font-family:sans-serif;}</style></head><body>');
            doc.write(content);
            doc.write('</body></html>');
            doc.close();
        }
        
        myModal.show();
    } catch(e) {
        console.error("Widget Error:", e);
        alert("Widget yüklenirken bir hata oluştu.");
    }
}

<script>
document.querySelectorAll('.follow-btn').forEach(btn => {
    btn.addEventListener('click', function(e) {
        e.preventDefault();
        e.stopPropagation();
        const userId = this.getAttribute('data-user-id');
        const self = this;
        if (self.disabled) return;
        self.disabled = true;

        const formData = new FormData();
        formData.append('user_id', userId);

        fetch('https://habertugra.com/Follow/toggle', {
            method: 'POST',
            body: formData
        })
        .then(response => response.json())
        .then(data => {
            if (data.status === 'success') {
                document.querySelectorAll(`.follow-btn[data-user-id="${userId}"]`).forEach(b => {
                    const icon = b.querySelector('i');
                    if (data.follow_status === 'followed') {
                        b.classList.remove('btn-danger', 'btn-primary');
                        b.classList.add('btn-success');
                        if(icon) { icon.classList.remove('fa-plus'); icon.classList.add('fa-check'); }
                    } else {
                        b.classList.remove('btn-success');
                        b.classList.add('btn-danger');
                        if(icon) { icon.classList.remove('fa-check'); icon.classList.add('fa-plus'); }
                    }
                });
            } else {
                alert(data.message);
            }
        })
        .catch(error => console.error('Error:', error))
        .finally(() => self.disabled = false);
    });
});
</script>

<script>
    let scale = 1;
    let isDragging = false;
    let startX, startY;
    let translateX = 0, translateY = 0;
    
    function openEgazeteModal() {
        var myModal = new bootstrap.Modal(document.getElementById('egazeteModal'));
        myModal.show();
    }

    function zoomIn() {
        scale += 0.2;
        updateTransform();
    }

    function zoomOut() {
        if(scale > 0.4) scale -= 0.2;
        updateTransform();
    }

    function updateTransform() {
        const img = document.getElementById('egazeteImage');
        if(img) {
            img.style.transform = `translate(${translateX}px, ${translateY}px) scale(${scale})`;
        }
    }

    // Simple Drag Logic for Image
    const container = document.getElementById('egazeteContainer');
    const img = document.getElementById('egazeteImage');
    
    if(img && container) {
        img.addEventListener('mousedown', (e) => {
            if(scale <= 1) return; // Only drag if zoomed
            isDragging = true;
            startX = e.clientX - translateX;
            startY = e.clientY - translateY;
            img.style.cursor = 'grabbing';
            e.preventDefault();
        });

        window.addEventListener('mouseup', () => {
            isDragging = false;
            if(img) img.style.cursor = 'grab';
        });

        window.addEventListener('mousemove', (e) => {
            if (!isDragging) return;
            translateX = e.clientX - startX;
            translateY = e.clientY - startY;
            updateTransform();
        });
    }
</script>
