    <h2 class="section-title">عن هوامير البورصة</h2>
    <p class="about-text">
  منذ تأسيسها عام <strong>2005</strong>، رسخت <strong>منتديات هوامير البورصة السعودية</strong> مكانتها كأكبر منصة عربية متخصصة في <strong>أسواق المال والاستثمار</strong>،

</p>

<p class="about-text">
  وخلال مسيرتها الممتدة لأكثر من عقدين، تحولت <strong>هوامير البورصة</strong> إلى منبر اقتصادي مؤثر يعكس <strong>نبض الأسواق</strong> ويواكب أبرز التطورات المالية المحلية والعالمية،
  عبر ملايين المشاركات والتحليلات والنقاشات اليومية التي يثريها الأعضاء والخبراء.
</p>

<p class="about-text">
  اليوم، تُعد المنصة وجهة رئيسية <strong>للشركات المالية والمصارف وشركات التقنية والاتصالات</strong> الراغبة في الوصول إلى جمهور نوعي يتمتع بالوعي الاستثماري والخبرة السوقية العالية.
</p>

function animateCountUp(elementId, finalValue, duration = 2000) {
  var element = typeof elementId === 'string' ? document.getElementById(elementId) : elementId;
  if (!element) return;

  finalValue = Number(finalValue) || 0;

  var startTime = null;
  var range = finalValue;

  function step(timestamp) {
    if (!startTime) startTime = timestamp;
    var progress = Math.min((timestamp - startTime) / duration, 1);
    var current = Math.floor(progress * range);

    element.textContent = current.toLocaleString('en-US');

    if (progress < 1) {
      requestAnimationFrame(step);
    } else {
      element.textContent = finalValue.toLocaleString('en-US');
    }
  }
  requestAnimationFrame(step);
}

    function formatTime(dateStr) {
        const d = new Date(dateStr.replace(" ", "T").replace(/-/g, "/"));
        const diff = (Date.now() - d.getTime()) / 1000;
        if (isNaN(diff) || diff < 0) return dateStr;
        if (diff < 60) return "الآن";
        if (diff < 3600) return `قبل ${Math.floor(diff/60)} دقيقة`;
        if (diff < 86400) return `قبل ${Math.floor(diff/3600)} ساعة`;
        const days = Math.floor(diff/86400);
        if (days === 1) return "قبل يوم";
        if (days === 2) return "قبل يومين";
        if (days <= 10) return `قبل ${days} أيام`;
        return `قبل ${days} يوم`;
    }

let statsAnimated = false;

function startStatsCounter() {
  // استدعاء بيانات JSON (نفس الملف الذي تستخدمه)
  fetch('cachehome/home_blocks_cache.json')
    .then(res => res.json())
    .then(data => {
      if (!data || !data.stats) return;

      // تشغيل العد فقط مرة واحدة
      if (!statsAnimated) {
        statsAnimated = true;
        animateCountUp('members-count', data.stats.members, 2000);
        animateCountUp('threads-count', data.stats.threads, 2000);
        animateCountUp('posts-count', data.stats.posts, 2000);
      }
    })
    .catch(console.error);
}

document.addEventListener('DOMContentLoaded', () => {
  const statsSection = document.querySelector('.about-hawamer');
  if (!statsSection) return;

  const observer = new IntersectionObserver((entries, observer) => {
    entries.forEach(entry => {
      if (entry.isIntersecting) {
        startStatsCounter();
        observer.unobserve(statsSection); // نوقف المراقبة بعد التشغيل
      }
    });
  }, { threshold: 0.4 }); // يبدأ العد عندما يظهر 40% من البلوك

  observer.observe(statsSection);
});
