            runSpeechRecog = () => {
            var SpeechRecognition = window.SpeechRecognition || window.webkitSpeechRecognition;
            if (!SpeechRecognition) {
                alert("Trình duyệt của bạn không hỗ trợ tìm kiếm bằng giọng nói.");
                return;
            }
            
            var search = document.getElementById('search');
            document.getElementById("search").placeholder = "...";
            
            // Check for iOS non-Safari browsers (Chrome, Firefox, Zalo, Facebook)
            var isIOS = /iPad|iPhone|iPod/.test(navigator.userAgent) && !window.MSStream;
            var isSafari = /Safari/.test(navigator.userAgent) && !/Chrome/.test(navigator.userAgent) && !/CriOS/.test(navigator.userAgent) && !/FxiOS/.test(navigator.userAgent);
            
            if (isIOS && !isSafari) {
                alert("Rất tiếc, Apple chỉ cho phép sử dụng Tìm kiếm Giọng nói trên trình duyệt Safari gốc của máy.\n\nVui lòng mở trang web này bằng Safari để tiếp tục nhé!");
                document.getElementById("search").placeholder = "Nhập tên bài hát, tên ca sĩ / @tên nhạc sĩ / lời";
                return;
            }
            
            try {
                let recognization = new SpeechRecognition();
                recognization.lang = "vi-VN";
                
                recognization.onstart = () => {
                   search.value = "Đang lắng nghe ...";
                }
                
                recognization.onresult = (e) => {
                   var transcript = e.results[0][0].transcript;
                   search.value = transcript;
                   search.classList.remove("hide");
                   document.getElementById("searchform").submit();
                }
                
                recognization.onerror = (e) => {
                   console.error("Speech recognition error: ", e.error);
                   search.value = "";
                   document.getElementById("search").placeholder = "Nhập tên bài hát, tên ca sĩ / @tên nhạc sĩ / lời";
                   
                   if (e.error === 'not-allowed' || e.error === 'audio-capture') {
                       alert("Không thể truy cập Micro (" + e.error + ")!\n\nCách khắc phục:\n1. Bật 'Siri & Đọc chính tả' trong Cài đặt chung (nếu dùng iPhone).\n2. Cho phép trình duyệt truy cập Micro.\n3. Nếu đang dùng Chrome/Zalo trên iPhone, hãy mở web bằng Safari.");
                   } else {
                       alert("Lỗi nhận dạng giọng nói (" + e.error + "). Vui lòng thử lại.");
                   }
                }
                
                recognization.start();
            } catch (err) {
                alert("Lỗi khởi tạo giọng nói: " + err.message);
            }
         }

        </script>
