    <script>
        (() => {
            const section = document.querySelector('.developers-code');
            if (!section) return;

            const bgLayers = section.querySelectorAll('.bg-layer');
            const content = section.querySelector('.content');

            let hasAnimated = false;
            let isSnapped = false;

            function handleScroll() {
                const rect = section.getBoundingClientRect();
                const vh = window.innerHeight;

                /* -------------------------
                Auto-snap when entering section
                -------------------------- */
                if (!isSnapped && rect.top < vh * 0.5 && rect.top > -100) {
                    isSnapped = true;
                    section.scrollIntoView({ behavior: 'smooth', block: 'start' });
                }

                // Reset snap when fully scrolled past
                if (rect.bottom < 0) {
                    isSnapped = false;
                }

                /* -------------------------
                Background transitions
                -------------------------- */
                const totalScroll = rect.height - vh;
                if (totalScroll > 0) {
                    const scrolled = Math.min(
                        totalScroll,
                        Math.max(0, vh - rect.top)
                    );
                    const progress = scrolled / totalScroll;

                    bgLayers.forEach((layer, index) => {
                        const step = 1 / bgLayers.length;
                        const start = index * step;
                        const end = start + step;

                        let opacity = 0;
                        if (progress >= end) opacity = 1;
                        else if (progress > start) {
                            opacity = (progress - start) / step;
                        }

                        layer.style.opacity = opacity;
                    });
                }

                /* -------------------------
                Content fade-in when it enters viewport
                -------------------------- */
                if (!hasAnimated) {
                    const contentRect = content.getBoundingClientRect();
                    if (contentRect.top < vh * 0.8) {
                        content.classList.add('animate-in', 'dark-bg');
                        hasAnimated = true;
                    }
                }
            }

            let ticking = false;
            function onScroll() {
                if (!ticking) {
                    requestAnimationFrame(() => {
                        handleScroll();
                        ticking = false;
                    });
                    ticking = true;
                }
            }

            window.addEventListener('scroll', onScroll, { passive: true });
            window.addEventListener('resize', onScroll);
            handleScroll();
        })();
    </script>
