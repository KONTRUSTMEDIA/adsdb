    window.BackgroundHelper = function () {
        var slides = [];
        var direction = "next";
        var motion = "horizontal";
        var width = 0;
        var height = 0;
        var multiplier = 1;
        var originalWidth = 0;
        var originalHeight = 0;
        var transitionDuration = "";

        this.init = function (motionType, dir, duration) {
            direction = dir;
            motion = motionType;
            slides = [];
            width = 0;
            height = 0;
            multiplier = 1;
            originalWidth = 0;
            originalHeight = 0;
            transitionDuration = duration;
        };

        this.processSlide = function (element, modify) {
            this.updateSize(element, null);
            var pos = [];

            var bgPosition = element.css("background-position");
            var positions = bgPosition.split(",");
            $.each(positions, function (i) {
                var position = $.trim(this);
                var point = position.split(" ");
                var zeroValue = browser.ie && browser.version >= 10 ? 0.1 : 0;
                if (point.length > 1) {
                    var x = point[0].indexOf('%') === -1 ? parseFloat(point[0], 10) : zeroValue;
                    var y = point[1].indexOf('%') === -1 ? parseFloat(point[1], 10) : zeroValue;
                    pos.push({ x: x, y: y });
                } else {
                    pos.push({ x: zeroValue, y: zeroValue });
                }
            });

            slides.push({
                "images": element.css("background-image"),
                "sizes": element.css("background-size"),
                "positions": pos
            });

            if (modify)
                element.css("background-image", "none");
        };

        this.updateSize = function (element, initialSize) {
            width = element.outerWidth(false);
            height = element.outerHeight();
            if (initialSize && parseInt(initialSize.width, 10) !== 0) {
                originalWidth = parseInt(initialSize.width, 10);
                originalHeight = parseInt(initialSize.height, 10);
                if (motion === "fade") {
                    $.each(element.children(), function (i) {
                        $(this).css("background-position", getCssPositions(slides[i].positions, { x: 0, y: 0 }));
                    });
                }
            }
        };

        this.setBackground = function (element, items) {
            var bg = [];
            var sizes = [];
            $.each(items, function (i, o) {
                bg.push(o.images);
                sizes.push(o.sizes);
            });
            element.css({
                "background-image": bg.join(", "),
                //"background-size": sizes.join(", "),
                "background-repeat": "no-repeat"
            });
        };

        this.setPosition = function (element, items) {
            var pos = [];
            $.each(items, function (i, o) {
                pos.push(o.positions);
            });
            element.css({
                "background-position": pos.join(", ")
            });
        };

        this.current = function (index) {
            return slides[index] || null;
        };

        this.next = function (index) {
            var next;
            if (direction === "next") {
                next = (index + 1) % slides.length;
            } else {
                next = index - 1;
                if (next < 0) {
                    next = slides.length - 1;
                }
            }
            return slides[next];
        };

        this.items = function (prev, next, move) {
            var prevItem = { x: 0, y: 0 };
            var nextItem = { x: 0, y: 0 };
            var isDirectionNext = direction === "next";
            var verticalOffset = -(originalHeight - height) / 2;
            var horizontalOffset = -(originalWidth - width) / 2;
            if (motion === "horizontal") {
                prevItem.y = nextItem.y = -(originalHeight - height) / 2;
                prevItem.x = horizontalOffset;
                nextItem.x = (isDirectionNext ? originalWidth : -originalWidth) + horizontalOffset;
                if (move) {
                    prevItem.x += isDirectionNext ? -originalWidth : originalWidth;
                    nextItem.x += isDirectionNext ? -originalWidth : originalWidth;
                }
            } else if (motion === "vertical") {
                prevItem.x = nextItem.x = horizontalOffset;
                prevItem.y = verticalOffset;
                nextItem.y = (isDirectionNext ? originalHeight : -originalHeight) + verticalOffset;
                if (move) {
                    prevItem.y += isDirectionNext ? -originalHeight : originalHeight;
                    nextItem.y += isDirectionNext ? -originalHeight : originalHeight;
                }
            }
            var result = [];
            if (!!prev) {
                result.push({ images: prev.images, positions: getCssPositions(prev.positions, prevItem), sizes: prev.sizes });
            }
            if (!!next) {
                result.push({ images: next.images, positions: getCssPositions(next.positions, nextItem), sizes: next.sizes });
            }

            if (direction === "next") {
                result.reverse();
            }

            return result;
        };

        this.transition = function (container, on) {
            container.css($.support.themeTransition.prefix + "transition", on ? "background-position " + transitionDuration + " ease-in-out" : "");
        };

        function getCssPositions(positions, offset) {
            var result = [];
            if (positions === undefined) {
                return "";
            }
            offset.x = offset.x || 0;
            offset.y = offset.y || 0;
            for (var i = 0; i < positions.length; i++) {
                result.push((positions[i].x * 1 + offset.x) + "px " + (positions[i].y * 1 + offset.y) + "px");
            }
            return result.join(", ");
        }
    };


    var ThemeSlider = function (element, settings) {

        var interval = null;
        var active = false;
        var children = element.find(".active").parent().children();
        var last = false;
        var running = false;

        this.settings = $.extend({}, {
            "animation": "horizontal",
            "direction": "next",
            "speed": 600,
            "pause": 2500,
            "auto": true,
            "repeat": true,
            "navigator": null,
            "clickevents": true,
            "hover": true,
            "helper": null
        }, settings);

        this.move = function (direction, next) {
            var activeItem = element.find(".active"),
                nextItem = next || activeItem[direction](),
                innerDirection = this.settings.direction === "next" ? "forward" : "back",
                reset = direction === "next" ? "first" : "last",
                moving = interval,
                slider = this, tmp;

            active = true;

            if (moving) { this.stop(true); }

            if (!nextItem.length) {
                nextItem = element.find(".art-slide-item")[reset]();
                if (!this.settings.repeat) { last = true; active = false; return; }
            }

            if ($.support.themeTransition) {
                nextItem.addClass(this.settings.direction);
                tmp = nextItem.get(0).offsetHeight;

                activeItem.addClass(innerDirection);
                nextItem.addClass(innerDirection);

                element.trigger("beforeSlide", children.length);

                element.one($.support.themeTransition.event, function () {
                    nextItem.removeClass(slider.settings.direction)
                        .removeClass(innerDirection)
                        .addClass("active");
                    activeItem.removeClass("active")
                        .removeClass(innerDirection);
                    active = false;
                    setTimeout(function () {
                        element.trigger("afterSlide", children.length);
                    }, 0);
                });
            } else {
                element.trigger("beforeSlide", children.length);

                activeItem.removeClass("active");
                nextItem.addClass("active");
                active = false;

                element.trigger("afterSlide", children.length);
            }

            this.navigate(nextItem);

            if (moving) { this.start(); }
        };

        this.navigate = function (position) {
            var index = children.index(position);
            $(this.settings.navigator).children().removeClass("active").eq(index).addClass("active");
        };

        this.to = function (index) {
            var activeItem = element.find(".active"),
                children = activeItem.parent().children(),
                activeIndex = children.index(activeItem),
                slider = this;

            if (index > (children.length - 1) || index < 0) {
                return;
            }

            if (active) {
                return element.one("afterSlide", function () {
                    slider.to(index);
                });
            }

            if (activeIndex === index) {
                return;
            }

            this.move(index > activeIndex ? "next" : "prev", $(children[index]));
        };

        this.next = function () {
            if (!active) {
                if (last) { this.stop(); return; }
                this.move("next");
            }
        };

        this.prev = function () {
            if (!active) {
                if (last) { this.stop(); return; }
                this.move("prev");
            }
        };

        this.start = function (force) {
            if (!!force) {
                setTimeout($.proxy(this.next, this), 10);
            }
            interval = setInterval($.proxy(this.next, this), this.settings.pause);
            running = true;
        };

        this.stop = function (pause) {
            clearInterval(interval);
            interval = null;
            running = !!pause;
            active = false;
        };

        this.active = function () {
            return running;
        };

        this.moving = function () {
            return active;
        };

        this.navigate(children.filter(".active"));

        if (this.settings.clickevents) {
            $(this.settings.navigator).on("click", "a", { slider: this }, function (event) {
                var activeIndex = children.index(children.filter(".active"));
                var index = $(this).parent().children().index($(this));
                if (activeIndex !== index) {
                    event.data.slider.to(index);
                }
                event.preventDefault();
            });
        }

        if (this.settings.hover) {
            var slider = this;
            element.add(this.settings.navigator)
                   .add(element.siblings(".art-shapes")).hover(function () {
                       if (element.is(":visible") && !last) { slider.stop(true); }
                   }, function () {
                       if (element.is(":visible") && !last) { slider.start(); }
                   });
        }
    };

                data.start();
            }
        });
    };

})(jQuery);

// used to apply compicated values in style like '!important!
function applyCss(object, param, value) {
    var rg = new RegExp(param + '\s*:\s*[^;]+;', "i");
    var style = object.attr('style');
    var str = param + ': ' + value + ';';
    if (rg.test(style)) {
        style = style.replace(rg, str);
    }
    else {
        style += '; ' + str;
    }

    object.attr('style', style);
}

// convert universal coord to pixels
function uniToPx(uni, size, parentSize) {
    uni = parseFloat(uni || '0');
    if (uni < 0) {
        uni = uni * size;
    } else if (uni >= 1) {
        uni = parentSize - (2 - uni) * size;
    } else {
        uni = uni * (parentSize - size);
    }

    return uni;
}

        }
    }

    return 0;
}

            var pageSliderQuery = '.art-pageslider .art-slide-item>*';
            if (headerObjectResizer.isPreview) {
                headerQuery = 'header .art-slider';
                pageSliderQuery = '.art-pageslider .art-slider, .art-pageslider .art-textblock';
            }
            $(headerQuery + ', ' + pageSliderQuery).each(function () {
                var object = $(this);
                height = object.parent().height();

                var off = sheetLeftFunc(sheet, object);

                var cls = object.attr('class').split(' ');
                $.each(cls, function (key, val) {
                    val = $.trim(val);
                    if (val.length === 0) return;
                    if (val.indexOf(cssPrefix) !== 0) return;

                    val = val.substring(cssPrefix.length);
                    var data = resizeData[val];
                    if (typeof data === 'undefined') return;

                    if (cleanUpStyles) {
                        object.css('display', '');
                        object.css('left', '');
                        object.css('margin-left', '');
                    }

                    var respData = data.responsive[responsiveType];
                    if (respData.visible) {
                        object.css('display', '');
                    } else {
                        applyCss(object, 'display', 'none !important');
                    }

                    if (cleanUpStyles || !respData.visible) return false;

                    var x = uniToPx(respData.left, data.autoWidth ? object.width() : data.width, sheetWidth);
                    x += off;

                    var y = uniToPx(respData.top, data.height, height);

                    object.css('left', x + 'px');
                    object.css('top', y + 'px');
                    applyCss(object, 'margin-left', '0px !important');
                    return false;
                });
            });

            // move images in slide's background-images
            var slides = $('.art-slide-item').add(header);
            if (browser.ie && browser.version <= 8) {
                slides = slides.add('.art-slide-item .ie8fix');
            }
            $.each(slides, function (slideIdx, slide) {
                slide = $(slide);

                if (slide.closest('.art-collage').length > 0 || cleanUpStyles) {
                    slide.css('background-position', '');
                    return;
                }

                var slideVisible = slide.is(':visible');
                if (!slideVisible && browser.ie) {
                    slide.css('display', 'block');
                }

                var off = sheetLeftFunc(sheet, slide);

                if (browser.ie && browser.version <= 8) {
                    var s = slide.attr('style');
                    if (s) {
                        s = s.replace(/background\-position[^;]+/, '');
                        slide.attr('style', s);
                    }
                } else {
                    slide.css('background-position', '');
                }
                slide.css('background-size', '');

                var bgImage = slide.css('background-image') ? slide.css('background-image').split(',') : [];
                var bgPosition = slide.css('background-position') && (slide.css('background-position').replace(/[0][^\d]+/gi, '')).length > 0 ?
                    slide.css('background-position').split(',') :
                    [];
                if (bgImage.length !== bgPosition.length) {
                    slide.css('display', '');
                    return;
                }

                height = slide.height();
                if (height === 0) height = slide.parent().height();

                $.each(bgImage, function (idx, val) {
                    var findImageIdx = val.lastIndexOf('images/');
                    var findDotIdx = val.lastIndexOf('.');
                    if (findImageIdx === -1 || findDotIdx === -1) return;

                    var name = val.substring(findImageIdx + 7, findDotIdx);

                    var data = resizeData[name];
                    if (typeof data === 'undefined') return;

                    var respData = data.responsive[responsiveType];
                    // big default coordinates for hiding
                    var x = 9999, y = 9999;
                    if (respData.visible) {
                        x = uniToPx(respData.left, data.width, sheetWidth);
                        x += off + data.area.x;

                        y = uniToPx(respData.top, data.height, height);
                        y += data.area.y;
                    }

                    bgPosition[idx] = x + 'px ' + y + 'px';
                });

                slide.css('background-position', bgPosition.join(','));

                if (!slideVisible && browser.ie) {
                    slide.css('display', '');
                }
            });

        };

<script>
(function(){
  // Verificar se está na página inicial
  var path = window.location.pathname;
  var isHome = path === "/" || path === "/index.html" || path === "" || 
               window.location.search.includes("m=1");
  
  if(!isHome) {
    document.getElementById("destaques-wrapper").style.display = "none";
    return;
  }

  // Elementos DOM
  var wrapper = document.getElementById("destaques-wrapper");
  var track = document.getElementById("destaquesTrack");
  var dotsContainer = document.getElementById("destaquesDots");
  var prevBtn = wrapper.querySelector(".destaques-nav.prev");
  var nextBtn = wrapper.querySelector(".destaques-nav.next");
  var slider = wrapper.querySelector(".destaques-slider");
  
  var index = 0;
  var isAnimating = false;
  var autoSlideInterval;
  var userInteracted = false;
  var interactionTimeout;
  
  // Configuração
  var CACHE_KEY = "destaques_cache_final_v4";
  var CACHE_TTL = 2 * 60 * 60 * 1000; // 2 horas
  
  // Função para mostrar que está carregando
  function showLoading() {
    wrapper.classList.remove('loaded');
  }
  
  // Função para esconder loading e mostrar conteúdo
  function showContent() {
    wrapper.classList.add('loaded');
  }
  
  // Iniciar mostrando loading
  showLoading();
  
  // Funções do slider
  function updateSlider() {
    if(isAnimating) return;
    isAnimating = true;
    
    track.style.transform = "translateX(" + (-index * 100) + "%)";
    
    // Atualizar dots
    var dots = dotsContainer.querySelectorAll('.destaques-dot:not(.placeholder-dot)');
    dots.forEach((dot, i) => {
      dot.classList.toggle('active', i === index);
    });
    
    setTimeout(() => { isAnimating = false; }, 500);
  }
  
  function handleUserInteraction() {
    // Marcar que usuário interagiu
    userInteracted = true;
    
    // Parar auto-slide temporariamente
    stopAutoSlide();
    
    // Limpar timeout anterior
    clearTimeout(interactionTimeout);
    
    // Reativar auto-slide após 10 segundos de inatividade
    interactionTimeout = setTimeout(function() {
      userInteracted = false;
      startAutoSlide();
    }, 10000);
  }
  
  function initControls() {
    prevBtn.addEventListener("click", function(e){
      e.preventDefault();
      if(isAnimating) return;
      
      // Registrar interação do usuário
      handleUserInteraction();
      
      var items = track.querySelectorAll('.destaques-item:not(.placeholder)').length;
      if(items === 0) return;
      index = (index - 1 + items) % items;
      updateSlider();
    });
    
    nextBtn.addEventListener("click", function(e){
      e.preventDefault();
      if(isAnimating) return;
      
      // Registrar interação do usuário
      handleUserInteraction();
      
      var items = track.querySelectorAll('.destaques-item:not(.placeholder)').length;
      if(items === 0) return;
      index = (index + 1) % items;
      updateSlider();
    });
    
    dotsContainer.addEventListener("click", function(e){
      if(e.target.classList.contains('destaques-dot') && 
         !e.target.classList.contains('placeholder-dot') && 
         !isAnimating) {
        
        // Registrar interação do usuário
        handleUserInteraction();
        
        var dots = dotsContainer.querySelectorAll('.destaques-dot:not(.placeholder-dot)');
        index = Array.from(dots).indexOf(e.target);
        updateSlider();
      }
    });
    
    // Pausar auto-slide quando mouse estiver sobre o slider
    slider.addEventListener('mouseenter', function() {
      if(autoSlideInterval) {
        stopAutoSlide();
      }
    });
    
    // Retomar auto-slide quando mouse sair do slider (se usuário não interagiu)
    slider.addEventListener('mouseleave', function() {
      if(!userInteracted) {
        startAutoSlide();
      }
    });
    
    // Iniciar auto-slide
    startAutoSlide();
  }
  
  function startAutoSlide() {
    // Só iniciar se usuário não interagiu recentemente
    if(userInteracted) return;
    
    stopAutoSlide();
    
    autoSlideInterval = setInterval(function(){
      var items = track.querySelectorAll('.destaques-item:not(.placeholder)').length;
      if(items > 1) {
        index = (index + 1) % items;
        updateSlider();
      }
    }, 5000);
  }
  
  function stopAutoSlide() {
    clearInterval(autoSlideInterval);
    autoSlideInterval = null;
  }
  
  // Cache
  function loadCache() {
    try {
      var data = localStorage.getItem(CACHE_KEY);
      if(!data) return null;
      
      var cached = JSON.parse(data);
      if(!cached || !cached.html || !cached.dots || 
         Date.now() - cached.time > CACHE_TTL) {
        return null;
      }
      
      return cached;
    } catch(e) {
      return null;
    }
  }
  
  function saveCache(html, dots) {
    try {
      localStorage.setItem(CACHE_KEY, JSON.stringify({
        time: Date.now(),
        html: html,
        dots: dots
      }));
    } catch(e) {}
  }
  
  // Iniciar carregamento
  initDestaques();
  
  async function initDestaques() {
    try {
      // 1. Tentar carregar do cache
      var cached = loadCache();
      
      // 2. Buscar novos dados (mesmo se tiver cache)
      var newData = await loadNewContent();
      
      // 3. Se não conseguiu novos dados mas tem cache, usar cache
      if(!newData && cached) {
        renderSlider(cached);
        showContent();
        return;
      }
      
      // 4. Se não tem cache nem novos dados, manter placeholder
      if(!newData) {
        // Tentar novamente após 5 segundos
        setTimeout(initDestaques, 5000);
      }
      
    } catch(e) {
      console.error('Erro ao inicializar destaques:', e);
      // Tentar novamente após 10 segundos
      setTimeout(initDestaques, 10000);
    }
  }
  
  function renderSlider(cached) {
    if(!cached) return;
    
    track.innerHTML = cached.html;
    dotsContainer.innerHTML = cached.dots;
    
    // Mostrar navegação
    prevBtn.classList.add('visible');
    nextBtn.classList.add('visible');
    
    // Ativar primeiro item
    var items = track.querySelectorAll('.destaques-item');
    if(items.length > 0) {
      updateSlider();
      initControls();
    }
    
    showContent();
  }
  
  // Carregar novo conteúdo
  async function loadNewContent() {
    try {
      var feed = await fetchFeed();
      if(!feed || feed.length === 0) return null;
      
      var posts = await processPosts(feed);
      if(posts.length === 0) return null;
      
      // Gerar HTML
      var html = posts.map(post => `
        <div class="destaques-item">
          <img src="${post.image}" alt="${post.title}" loading="lazy">
          <a class="destaques-overlay" href="${post.link}">
            <div class="destaques-post-title">${post.title}</div>
          </a>
        </div>
      `).join('');
      
      var dots = Array(posts.length).fill('<div class="destaques-dot"></div>').join('');
      
      // Atualizar slider
      track.innerHTML = html;
      dotsContainer.innerHTML = dots;
      
      // Inicializar controles
      index = 0;
      updateSlider();
      initControls();
      prevBtn.classList.add('visible');
      nextBtn.classList.add('visible');
      
      // Salvar cache
      saveCache(html, dots);
      
      showContent();
      
      return posts;
      
    } catch(e) {
      console.error('Erro ao carregar conteúdo:', e);
      return null;
    }
  }
  
  async function fetchFeed() {
    try {
      var response = await fetch("/feeds/posts/default?alt=json&max-results=30&orderby=published");
      if(!response.ok) throw new Error('Erro no feed');
      
      var data = await response.json();
      return data.feed.entry || [];
    } catch(e) {
      return [];
    }
  }
  
  async function processPosts(entries) {
    var validPosts = [];
    var processedIds = new Set();
    
    for(var i = 0; i < entries.length && validPosts.length < 5; i++) {
      var entry = entries[i];
      
      // Verificar se é post válido
      var title = entry.title?.$t || "";
      if(title.includes("(Fotos)") || title.includes("(Preview)")) continue;
      
      // Obter link
      var linkObj = entry.link?.find(l => l.rel === "alternate");
      if(!linkObj) continue;
      var link = linkObj.href;
      
      // Evitar duplicados
      if(processedIds.has(link)) continue;
      processedIds.add(link);
      
      // Buscar imagem
      var image = await getPostImage(link);
      if(!image) continue;
      
      validPosts.push({
        title: title.replace(/"/g, '&quot;'),
        link: link,
        image: image
      });
    }
    
    return validPosts;
  }
  
async function getPostImage(url) {
  try {
    var response = await fetch(url);
    var text = await response.text();

    var tempDiv = document.createElement('div');
    tempDiv.innerHTML = text;

    function normalizeBloggerUrl(u) {
      if (!u) return u;
      if (u.startsWith('//')) u = 'https:' + u;

      // Blogspot / Blogger (path-based)
      if (u.includes('blogspot.') || u.includes('blogger.com')) {
        // Força um tamanho grande SEM &#8220;cortar&#8221; (-c) e SEM reduzir.
        // Exemplos: /s72-c/, /s320/, /w300-h300/, /w1200-h630-p-k-no-nu/
        u = u.replace(/\/s\d+(-c)?\//, '/s1600/');
        u = u.replace(/\/w\d+-h\d+[^/]*\//, '/s1600/');
        return u;
      }

      // Googleusercontent (param-based)
      if (u.includes('googleusercontent.com')) {
        // Remove parâmetros de tamanho comuns e força s1600
        u = u.replace(/=w\d+-h\d+[^&]*/g, '');
        u = u.replace(/=s\d+[^&]*/g, '');
        // Se já existir "=", substitui tudo após o "=" por s1600 (mais confiável)
        if (u.includes('=')) {
          u = u.replace(/=.*/, '=s1600');
        } else {
          u += '=s1600';
        }
        return u;
      }

      return u;
    }

    function firstMetaImage() {
      // 1) og:image
      var og = tempDiv.querySelector('meta[property="og:image"], meta[property="og:image:url"]');
      if (og && og.content) return og.content;

      // 2) twitter:image
      var tw = tempDiv.querySelector('meta[name="twitter:image"], meta[name="twitter:image:src"]');
      if (tw && tw.content) return tw.content;

      return null;
    }

    function jsonLdImage() {
      var scripts = Array.from(tempDiv.querySelectorAll('script[type="application/ld+json"]'));
      for (var i = 0; i < scripts.length; i++) {
        try {
          var raw = scripts[i].textContent || scripts[i].innerText || '';
          if (!raw.trim()) continue;

          var data = JSON.parse(raw);

          // Pode vir como array
          var items = Array.isArray(data) ? data : [data];

          for (var j = 0; j < items.length; j++) {
            var it = items[j];
            if (!it) continue;

            var img = it.image;

            // image pode ser string, array, ou objeto
            if (typeof img === 'string') return img;
            if (Array.isArray(img) && img.length) {
              if (typeof img[0] === 'string') return img[0];
              if (img[0] && typeof img[0].url === 'string') return img[0].url;
            }
            if (img && typeof img.url === 'string') return img.url;
          }
        } catch(e) {}
      }
      return null;
    }

    function parseSrcset(srcset) {
      if (!srcset) return [];
      return srcset.split(',').map(s => s.trim()).map(part => {
        var pieces = part.split(/\s+/);
        var u = pieces[0] || '';
        var w = 0;
        if (pieces[1] && pieces[1].endsWith('w')) {
          var n = parseInt(pieces[1].replace('w',''), 10);
          if (!isNaN(n)) w = n;
        }
        return { url: u, w: w };
      }).filter(x => x.url);
    }

    function pickBestFromImg(imgEl) {
      var candidates = [];

      var dataOriginal = imgEl.getAttribute('data-original') || imgEl.getAttribute('data-orig-src');
      var dataSrc = imgEl.getAttribute('data-src') || imgEl.getAttribute('data-lazy-src') || imgEl.getAttribute('data-lazyload');
      var srcAttr = imgEl.getAttribute('src');

      if (dataOriginal) candidates.push({ url: dataOriginal, w: 99999 });
      if (dataSrc) candidates.push({ url: dataSrc, w: 5000 });
      if (srcAttr) candidates.push({ url: srcAttr, w: 1000 });

      var srcset = imgEl.getAttribute('srcset');
      parseSrcset(srcset).forEach(c => candidates.push(c));

      if (imgEl.src) candidates.push({ url: imgEl.src, w: 900 });

      // dedupe
      var seen = new Set();
      candidates = candidates.filter(c => c.url && !seen.has(c.url) && (seen.add(c.url), true));

      if (!candidates.length) return null;
      candidates.sort((a,b) => (b.w||0) - (a.w||0));
      return candidates[0].url;
    }

    // 0) Tenta pegar a imagem &#8220;principal&#8221; do post (normalmente é a correta e grande)
    var metaImg = firstMetaImage();
    if (metaImg) return normalizeBloggerUrl(metaImg);

    var ldImg = jsonLdImage();
    if (ldImg) return normalizeBloggerUrl(ldImg);

    // 1) Fallback: buscar no conteúdo do post
    var content = tempDiv.querySelector('.post-body, .post-content, .entry-content, article');
    if (!content) return null;

    var allImgs = Array.from(content.querySelectorAll('img'));

    var imgs = allImgs.filter(img => {
      var raw = (img.getAttribute('data-original') || img.getAttribute('data-src') || img.getAttribute('src') || img.src || '').toLowerCase();
      return raw &&
        !raw.includes('imgur.com') &&
        !raw.includes('addtoany.com') &&
        !raw.includes('icon') &&
        !raw.includes('avatar') &&
        !raw.includes('emoji') &&
        !raw.includes('logo') &&
        !raw.includes('share') &&
        !raw.includes('widget');
    });

    if (!imgs.length) return null;

    // Pega a &#8220;maior&#8221; entre as imagens do body
    var bestUrl = null;
    var bestScore = -1;

    imgs.forEach(img => {
      var chosen = pickBestFromImg(img);
      if (!chosen) return;

      var score = 0;

      // srcset max width
      var parsed = parseSrcset(img.getAttribute('srcset'));
      if (parsed.length) {
        var maxW = parsed.reduce((m, c) => Math.max(m, c.w || 0), 0);
        score += maxW;
      }

      // /s###/
      var m1 = chosen.match(/\/s(\d+)(-c)?\//);
      if (m1 && m1[1]) score += parseInt(m1[1], 10);

      // /w###-h###/
      var m2 = chosen.match(/\/w(\d+)-h(\d+)/);
      if (m2 && m2[1]) score += parseInt(m2[1], 10);

      // bônus anti-thumbnail
      if (!chosen.includes('s72') && !chosen.includes('s100') && !chosen.includes('s160')) score += 300;

      if (score > bestScore) {
        bestScore = score;
        bestUrl = chosen;
      }
    });

    if (!bestUrl) return null;

    return normalizeBloggerUrl(bestUrl);
  } catch(e) {
    return null;
  }
}
})();
</script>
