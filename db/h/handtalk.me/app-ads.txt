              const alturaOriginal = lottieInstance.animationData.h;

              lottie_player.style.setProperty('width', novaLargura + 'px', 'important');

        document.addEventListener('DOMContentLoaded', function() {
          // Controle do menu mobile
          const mobileToggle = document.getElementById('custom-mobile-toggle');
          const mobileClose = document.getElementById('custom-mobile-close');
          const mobileContainer = document.getElementById('custom-mobile-container');
          const toggleIcon = document.getElementById('mobile-toggle-icon');

          mobileToggle.addEventListener('click', function() {
            if (mobileContainer.style.display === 'block') {
              mobileContainer.style.display = 'none';
              document.body.style.overflow = 'auto';
              toggleIcon.classList.remove('fa-times');
              toggleIcon.classList.add('fa-bars');
            } else {
              mobileContainer.style.display = 'block';
              document.body.style.overflow = 'hidden';
              toggleIcon.classList.remove('fa-bars');
              toggleIcon.classList.add('fa-times');
            }
          });

          // Fechar menu mobile
          mobileClose.addEventListener('click', function() {
            mobileContainer.style.display = 'none';
            document.body.style.overflow = 'auto';
            toggleIcon.classList.remove('fa-times');
            toggleIcon.classList.add('fa-bars');
          });

          document.querySelectorAll('.custom-mobile-nav a').forEach(link => {
            link.addEventListener('click', function() {
              mobileContainer.style.display = 'none';
              document.body.style.overflow = 'auto';
              toggleIcon.classList.remove('fa-times');
              toggleIcon.classList.add('fa-bars');
              // Fecha todos os submenus
              document.querySelectorAll('.custom-nav-submenu').forEach(submenu => {
                submenu.style.maxHeight = '0';
                submenu.style.padding = '0';
                submenu.style.boxShadow = "none";
              });
            });
          });

          // Seleciona todos os botões do menu
          const navButtons = document.querySelectorAll('.custom-nav-button');

          // Adiciona evento de clique a cada botão
          navButtons.forEach(button => {
            button.addEventListener('click', function() {
              // Remove a classe 'active' de todos os botões
              navButtons.forEach(btn => btn.classList.remove('active'));

              // Adiciona a classe 'active' apenas ao botão clicado
              this.classList.add('active');
            });
          });

          // Fecha o menu mobile quando um link é clicado (mantendo o destaque)
          document.querySelectorAll('.custom-mobile-nav a').forEach(link => {
            link.addEventListener('click', function() {
              const mobileContainer = document.getElementById('custom-mobile-container');
              const toggleIcon = document.getElementById('mobile-toggle-icon');

              mobileContainer.style.display = 'none';
              document.body.style.overflow = 'auto';
              if (toggleIcon) {
                toggleIcon.classList.remove('fa-times');
                toggleIcon.classList.add('fa-bars');
              }
            });
          });

          // Função para controlar submenus
          function setupSubmenu(button) {
            button.addEventListener('click', function(e) {
              e.stopPropagation();
              const expanded = this.getAttribute('aria-expanded') === 'true';
              const submenu = document.getElementById(this.getAttribute('aria-controls'));
              const isMobile = button.closest('.custom-mobile-nav'); // Verifica se é mobile

              // Fecha outros submenus abertos no mesmo nível
              const parentMenu = this.closest('ul');
              if (parentMenu) {
                parentMenu.querySelectorAll('.custom-nav-button[aria-expanded="true"]').forEach(openButton => {
                  if (openButton !== this) {
                    openButton.setAttribute('aria-expanded', 'false');
                    const openSubmenu = document.getElementById(openButton.getAttribute('aria-controls'));
                    if (openSubmenu) {
                      openSubmenu.style.maxHeight = '0';
                      if (!openButton.closest('.custom-mobile-nav')) { // Só aplica padding no desktop
                        openSubmenu.style.padding = '0';
                      }
                      openSubmenu.style.boxShadow = 'none';
                    }
                  }
                });
              }

              // Alterna o submenu atual
              this.setAttribute('aria-expanded', !expanded);
              if (!expanded) {
                // Animação de ABERTURA
                submenu.style.maxHeight = submenu.scrollHeight + 150 + 'px';
                if (!isMobile) { // Aplica padding apenas no desktop
                  submenu.style.padding = '24px 40px';
                  submenu.style.boxShadow = '0px 2px 3px rgba(0, 0, 0, 0.15), 0px 6px 10px 4px rgba(0, 0, 0, 0.1)';
                }
              } else {
                // Animação de FECHAMENTO
                submenu.style.maxHeight = '0';
                if (!isMobile) { // Remove padding apenas no desktop
                  submenu.style.padding = '0';
                  submenu.style.boxShadow = 'none';
                }
              }
            });
          }

          // Aplicar a todos os botões de submenu (desktop e mobile)
          document.querySelectorAll('.custom-desktop-nav .custom-nav-button, .custom-mobile-nav .custom-nav-button').forEach(setupSubmenu);

          // Fechar submenus ao clicar fora (desktop)
          document.addEventListener('click', function(e) {
            if (!e.target.closest('.custom-nav-item')) {
              document.querySelectorAll('.custom-nav-button[aria-expanded="true"]').forEach(button => {
                button.setAttribute('aria-expanded', 'false');
                const submenu = document.getElementById(button.getAttribute('aria-controls'));
                if (submenu) {
                  submenu.style.maxHeight = '0';
                  submenu.style.padding = '0'; // Reseta o padding
                  submenu.style.boxShadow = "none";
                }
              });
            }
          });

          // Fechar submenus ao pressionar Esc
          document.addEventListener('keydown', function(e) {
            if (e.key === 'Escape') {
              document.querySelectorAll('.custom-nav-button[aria-expanded="true"]').forEach(button => {
                button.setAttribute('aria-expanded', 'false');
                const submenu = document.getElementById(button.getAttribute('aria-controls'));
                if (submenu) {
                  submenu.style.maxHeight = '0';
                  submenu.style.padding = '0';
                  submenu.style.boxShadow = "none"
                }
              });
            }
          });
        });
