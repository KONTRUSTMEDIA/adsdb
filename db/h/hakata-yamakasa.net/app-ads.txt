<!-- 年のみ<span class="yearCountOnly"></span>    -->
<script>
  const startYear = 1241;
  const baseMonth = 6; // 7月
  const baseDay = 15;

  const now = new Date();
  const currentYear = now.getFullYear();
  const currentMonth = now.getMonth() + 1;
  const currentDay = now.getDate();

  // 今年の基準日
  const thisYearBase = new Date(currentYear, baseMonth, baseDay);

  // 何年目か計算
  let nthYear = currentYear - startYear + 1;
  if (now < thisYearBase) {
      nthYear -= 1;
  }

  // 注釈付き表示
  const displayText = `${nthYear}年（※${currentYear}年${currentMonth}月${currentDay}日現在）`;

  // class="yearCount" に反映
  document.querySelectorAll('.yearCount').forEach(el => {
    el.textContent = displayText;
  });

  // class="yearCountOnly" に反映
  document.querySelectorAll('.yearCountOnly').forEach(el => {
    el.textContent = `${nthYear}年`;
  });
</script>
