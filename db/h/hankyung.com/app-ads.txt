                $(document).ready(function () {
                  // 탭 리스트 추출 ['tab1', 'tab2', 'tab3']
                  const tabList = $(".topten-module .topten-menu li")
                    .map(function () {
                      return $(this).data("menu");
                    })
                    .get();

                  let isTabClick = false; // 탭을 클릭해서 슬라이드 이동하는지 체크
                  let isLocked = false; // 버튼 연타 방지를 위한 잠금 플래그
                  const lockDuration = 500; // 잠금 지속 시간

                  // TOPTEN slick 슬라이드 초기화
                  $(".toptenSlide").slick({
                    slidesToShow: 1,
                    slidesToScroll: 1,
                    infinite: true,
                    arrows: true,
                  });

                  // 슬라이드 변경 완료 후 탭 활성화
                  $(".toptenSlide").on(
                    "afterChange",
                    function (event, slick, currentSlide) {
                      if (!isTabClick) {
                        const newTab = tabList[currentSlide % tabList.length];
                        activateTab(newTab);
                      }
                      isTabClick = false;
                      unlock();
                    }
                  );

                  // 탭메뉴 클릭 시, 활성화 및 해당 슬라이드로 이동
                  $(".topten-module .topten-menu li").click(function () {
                    const targetTab = $(this).data("menu");
                    const targetIndex = tabList.indexOf(targetTab);

                    isTabClick = true;
                    activateTab(targetTab);
                    $(".toptenSlide").slick("slickGoTo", targetIndex, true);
                  });

                  // 슬라이드 이전, 다음 버튼 이동
                  $(".toptenSlide .slick-prev").click(function () {
                    if (isLocked) return;
                    lock();
                    $(".toptenSlide").slick("slickPrev");
                  });
                  $(".toptenSlide .slick-next").click(function () {
                    if (isLocked) return;
                    lock();
                    $(".toptenSlide").slick("slickNext");
                  });

                  // 탭 활성화
                  function activateTab(tabName) {
                    $(".topten-module .topten-menu li").removeClass("on");
                    $(
                      '.topten-module .topten-menu li[data-menu="' +
                        tabName +
                        '"]'
                    ).addClass("on");

                    $(".topten-tab-cont").removeClass("current");
                    $("." + tabName).addClass("current");
                  }

                  // 버튼 연타 방지
                  function lock() {
                    isLocked = true;
                    setTimeout(unlock, lockDuration);
                  }
                  function unlock() {
                    isLocked = false;
                  }
                });
              </script>
