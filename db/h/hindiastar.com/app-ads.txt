<!-- JavaScript for Pagination -->
<script>
document.addEventListener('DOMContentLoaded', function() {
    
    // Function to show the first page of each list
    function initializePagination() {
        document.querySelectorAll('.questions-box').forEach(box => {
            const firstPage = box.querySelector('.question-page[data-page-num="1"]');
            if (firstPage) {
                firstPage.style.display = 'block';
            }
        });
    }

    // Function to handle page changes
    function changePage(boxId, direction) {
        const box = document.getElementById(boxId);
        if (!box) return;

        const paginationFooter = box.querySelector('.pagination-footer');
        const list = box.querySelector('.questions-list');
        
        let currentPage = parseInt(paginationFooter.dataset.currentPage);
        const totalPages = parseInt(paginationFooter.dataset.totalPages);
        
        // Calculate new page number
        let newPage = (direction === 'next') ? currentPage + 1 : currentPage - 1;
        if (newPage < 1) newPage = 1;
        if (newPage > totalPages) newPage = totalPages;
        if (newPage === currentPage) return; // No change

        // Find page elements
        const currentPageEl = list.querySelector('.question-page[data-page-num="' + currentPage + '"]');
        const newPageEl = list.querySelector('.question-page[data-page-num="' + newPage + '"]');

        // Swap visibility
        if (currentPageEl) currentPageEl.style.display = 'none';
        if (newPageEl) newPageEl.style.display = 'block';

        // Update footer data
        paginationFooter.dataset.currentPage = newPage;
        
        // Update text
        paginationFooter.querySelector('.current-page-num').textContent = newPage;

        // Update button states
        const prevButton = paginationFooter.querySelector('.prev');
        const nextButton = paginationFooter.querySelector('.next');
        
        prevButton.disabled = (newPage === 1);
        nextButton.disabled = (newPage === totalPages);
    }

    // Add click event listeners to all pagination buttons
    document.querySelectorAll('.pagination-button').forEach(button => {
        button.addEventListener('click', function() {
            const boxId = this.closest('.pagination-footer').dataset.paginationControls;
            
            if (this.classList.contains('next')) {
                changePage(boxId, 'next');
            } else if (this.classList.contains('prev')) {
                changePage(boxId, 'prev');
            }
        });
    });

    // Show Page 1 on load
    initializePagination();
});
</script>
