<script>
function searchBus() {
    // Get input values
    const fromCity = document.getElementById('fromCity').value.trim();
    const toCity = document.getElementById('toCity').value.trim();
    const button = document.querySelector('.search-box button');
    const errorContainer = document.querySelector('.error-message');
    
    // Remove previous error messages
    if (errorContainer) {
        errorContainer.remove();
    }
    
    // Remove error styling
    document.querySelectorAll('.search-box input').forEach(input => {
        input.classList.remove('error');
    });
    
    // Validation
    if (!fromCity && !toCity) {
        showError('Please enter both departure and destination cities');
        document.getElementById('fromCity').classList.add('error');
        document.getElementById('toCity').classList.add('error');
        return;
    }
    
    if (!fromCity) {
        showError('Please enter departure city');
        document.getElementById('fromCity').classList.add('error');
        document.getElementById('fromCity').focus();
        return;
    }
    
    if (!toCity) {
        showError('Please enter destination city');
        document.getElementById('toCity').classList.add('error');
        document.getElementById('toCity').focus();
        return;
    }
    
    // Additional validation for minimum length
    if (fromCity.length < 2) {
        showError('From city must be at least 2 characters');
        document.getElementById('fromCity').classList.add('error');
        document.getElementById('fromCity').focus();
        return;
    }
    
    if (toCity.length < 2) {
        showError('To city must be at least 2 characters');
        document.getElementById('toCity').classList.add('error');
        document.getElementById('toCity').focus();
        return;
    }
    
    // Check if cities are the same
    if (fromCity.toLowerCase() === toCity.toLowerCase()) {
        showError('Departure and destination cities cannot be the same');
        document.getElementById('fromCity').classList.add('error');
        document.getElementById('toCity').classList.add('error');
        return;
    }
    
    // Show loading state
    button.textContent = 'Searching';
    button.classList.add('loading');
    button.disabled = true;
    
    // Create search URL with proper encoding
    const searchUrl = `https://haryanabusinfo.in/timetable/?from_city=${encodeURIComponent(fromCity)}&to_city=${encodeURIComponent(toCity)}`;
    
    // Add small delay for better UX
    setTimeout(() => {
        // Track search event (if you have analytics)
        if (typeof gtag !== 'undefined') {
            gtag('event', 'search', {
                'search_term': `${fromCity} to ${toCity}`,
                'custom_map': {'custom_parameter_1': 'bus_search'}
            });
        }
        
        // Redirect to search results
        window.location.href = searchUrl;
    }, 500);
}

function showError(message) {
    const searchBox = document.querySelector('.search-box');
    const errorDiv = document.createElement('div');
    errorDiv.className = 'error-message';
    errorDiv.textContent = message;
    searchBox.insertAdjacentElement('afterend', errorDiv);
    
    // Auto-remove error after 5 seconds
    setTimeout(() => {
        if (errorDiv.parentNode) {
            errorDiv.remove();
        }
    }, 5000);
}
