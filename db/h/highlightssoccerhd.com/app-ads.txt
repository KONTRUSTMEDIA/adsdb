            function slugify(text) {
                return text.toString().toLowerCase().trim().replace(/\s+/g, '-').replace(/[^\w\-]/g, '');
            }

            document.getElementById('load-more-btn').addEventListener('click', function() {
                const btn = this;
                const currentPageInput = document.getElementById('current-page');
                const totalPagesInput = document.getElementById('total-pages');
                const countrySlugInput = document.getElementById('country-slug');
                
                const currentPage = parseInt(currentPageInput.value);
                const totalPages = parseInt(totalPagesInput.value);
                const countrySlug = countrySlugInput.value;
               
                
                if (currentPage >= totalPages) {
                    btn.disabled = true;
                    btn.textContent = 'No more matches';
                    return;
                }
                
                const nextPage = currentPage + 1;
                const date = new Date().toISOString().split('T')[0]; // Format: YYYY-MM-DD
                
                // Call ajax.php with parameters
                const ajaxUrl = 'https://highlightssoccerhd.com/ajax.php?obj=get-load-more&tbl=country&p=' + nextPage + '&slug=' + encodeURIComponent(countrySlug) + '&date=' + date;
                console.log('Loading matches from:', ajaxUrl);
                
                btn.textContent = 'Loading...';
                btn.disabled = true;
                
                // Function to fetch with timeout and retry
                const fetchWithTimeout = (url, options = {}, timeout = 15000, retries = 3) => {
                    return new Promise((resolve, reject) => {
                        const controller = new AbortController();
                        const { signal } = controller;
                        
                        const timeoutId = setTimeout(() => {
                            controller.abort();
                            reject(new Error('Request timeout'));
                        }, timeout);
                        
                        fetch(url, { ...options, signal })
                            .then(response => {
                                clearTimeout(timeoutId);
                                resolve(response);
                            })
                            .catch(error => {
                                clearTimeout(timeoutId);
                                reject(error);
                            });
                    });
                };
                
                const loadMatches = (retryCount = 0) => {
                    fetchWithTimeout(ajaxUrl, {}, 15000) // 15 second timeout
                        .then(response => {
                            if (!response.ok) {
                                throw new Error(`HTTP ${response.status}: ${response.statusText}`);
                            }
                            return response.json();
                        })
                        .then(data => {
                            console.log('Response from ajax.php:', data);
                            
                            if (data && data.status === 1 && data.data && Array.isArray(data.data)) {
                                const matchList = document.getElementById('match-list');
                                const fragment = document.createDocumentFragment();
                                
                                data.data.forEach(match => {
                                    const tempDiv = document.createElement('div');
                                    tempDiv.innerHTML = createMatchCard(match);
                                    fragment.appendChild(tempDiv.firstElementChild);
                                });
                                
                                matchList.appendChild(fragment);
                                currentPageInput.value = nextPage;
                                
                                if (nextPage >= totalPages) {
                                    btn.textContent = 'No more matches';
                                } else {
                                    btn.textContent = 'Load More Matchs';
                                    btn.disabled = false;
                                }
                            } else {
                                throw new Error(data.message || 'Invalid response format');
                            }
                        })
                        .catch(error => {
                            console.error('Error loading matches:', error);
                            
                            // Retry logic
                            if (retryCount < 2) { // Retry up to 2 times
                                console.log(`Retrying... Attempt ${retryCount + 1}/3`);
                                btn.textContent = `Retrying... (${retryCount + 1}/3)`;
                                setTimeout(() => {
                                    loadMatches(retryCount + 1);
                                }, 2000); // Wait 2 seconds before retry
                            } else {
                                // Final failure
                                btn.textContent = 'Load More Matchs';
                                btn.disabled = false;
                                
                                let errorMessage = 'Error loading matches: ';
                                if (error.name === 'AbortError') {
                                    errorMessage += 'Request timed out. Please check your internet connection.';
                                } else if (error.message.includes('HTTP')) {
                                    errorMessage += error.message;
                                } else {
                                    errorMessage += error.message || 'Unknown error occurred';
                                }
                                
                                alert(errorMessage);
                            }
                        });
                };
                
                // Start loading
                loadMatches();
            });

            function createMatchCard(match) {
                const separator = ' vs ';
                const homeLogo = match.home_team_logo || 'https://placehold.co/80x80/cccccc/000000?text=?';
                const awayLogo = match.away_team_logo || 'https://placehold.co/80x80/cccccc/000000?text=?';
                
                const matchSlug = match.slug ? match.slug + '-' + match.event_date : slugify(match.event_home_team + ' vs ' + match.event_away_team + ' ' + match.event_date);
                const linkSlug = 'https://highlightssoccerhd.com/match/' + matchSlug;
                
                const leagueName = match.league_name ? match.league_name.split(' - ')[0] : 'Unknown League';
                const linkLeague = 'https://highlightssoccerhd.com/' + slugify(match.country_name) + '/' + slugify(leagueName);
                
                const matchDate = new Date(match.event_date).toLocaleDateString('en-US', { month: 'short', day: '2-digit' });
                
                const bgColors = ['#e74c3c', '#f1c40f', '#2980b9', '#e67e22', '#16a085', '#2c3e50'];
                const bgColor = bgColors[Math.floor(Math.random() * bgColors.length)];
                
                return `
                    <article class="post-card">
                        <a href="${linkSlug}">
                            <div class="post-thumb" style="background-color: ${bgColor}; background: url('template/img/bg-fixture.jpg') no-repeat center center / cover; position: relative;">
                                <div style="display:flex; align-items:center; justify-content:center; gap:12px; padding:10px;">
                                    <img src="${homeLogo}" alt="${escapeHtml(match.event_home_team)}" style="width:64px; height:64px; object-fit:contain; border-radius:6px;">
                                    <img src="${awayLogo}" alt="${escapeHtml(match.event_away_team)}" style="width:64px; height:64px; object-fit:contain; border-radius:6px; margin-left: 77%;">
                                </div>
                                <div class="play-icon">▶</div>
                            </div>
                        </a>
                        <div class="post-content">
                            <div class="post-meta">
                                <span class="category-tag"><a href="${linkLeague}" title="${escapeHtml(leagueName)}">${escapeHtml(leagueName)}</a></span>
                                <span class="date">${matchDate}</span>
                            </div>
                            <h3 class="post-title">
                                <a href="${linkSlug}">${escapeHtml(match.event_home_team)}${separator}${escapeHtml(match.event_away_team)} Highlights and Goals</a>
                            </h3>
                        </div>
                    </article>
                `;
            }

            function escapeHtml(text) {
                const div = document.createElement('div');
                div.textContent = text;
                return div.innerHTML;
            }

        const mobileToggle = document.getElementById('mobile-toggle');

        function toggleMenu() {

            menuOverlay.classList.toggle('active');
            // Khóa cuộn trang
            document.body.style.overflow = navBar.classList.contains('active') ? 'hidden' : '';
        }

        mobileToggle.addEventListener('click', (e) => { e.stopPropagation(); toggleMenu(); });

        // Accordion cho Mobile Menu
        document.querySelectorAll('.has-child > a, .has-sub > a').forEach(item => {
            item.addEventListener('click', function(e) {
                if (window.innerWidth < 992) {
                    e.preventDefault(); 
                    const parentLi = this.parentElement;
                    const subMenu = this.nextElementSibling; 
                    parentLi.classList.toggle('open');
                    if (subMenu) subMenu.classList.toggle('show');
                }
            });
        });
