                        function timestamp() {

                                    //$('#clock-wrapper').html(data);

                                        now.setSeconds(now.getSeconds() + 1);
                                        var h = now.getHours();
                                        var m = now.getMinutes();
                                        var s = now.getSeconds();
                                        m = checkTime(m);
                                        s = checkTime(s);
                                        document.getElementById('clock-wrapper').innerHTML = h + ":" + m + ":" + s;
                                        var t = setTimeout(startTime, 1000);
                                    }
                                    function checkTime(i) {
                                        if (i < 10) {

                                        }

                                    }
