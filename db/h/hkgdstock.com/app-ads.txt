        function zeroPadding(value, len) {
            if (value.length < len) {
                //return '0'.repeat(len-value.length) + value;
                return Array(len-value.length+1).join('0') + value;
            }
        }
        
        function validate(type, value) {
            switch(type) {
                case "stocks":
                    switch (true) {
                        case (value.match(/^[0-9]{1,4}$/) !== null):
                            return true;
                        break;
                        case (value.match(/,/) !== null && value.match(/(?:\n|\r\n)/) !== null):
                            return false;
                        break;
                        case (value.match(/,/) !== null):
                            var stock_nums = value.split(",");
                            if (stock_nums.length > 200) {
                                return false;
                            }
                            for (var i=0; i<stock_nums.length; i++) {
                                var stock_num = stock_nums[i].trim();
                                if (!validate('number-ALZ', stock_num)) {
                                    return false;
                                }
                            }
                        break;
                        case (value.match(/(?:\n|\r\n)/) !== null):
                            var stock_nums = value.split(/(?:\n|\r\n)/);
                            if (stock_nums.length > 200) {
                                return false;
                            }
                            for (var i=0; i<stock_nums.length; i++) {
                                var stock_num = stock_nums[i].trim();
                                if (!validate('number-ALZ', stock_num)) {
                                    return false;
                                }
                            }
                        break;
                        default:
                            return false;
                    }
                break;
                case "number-ALZ": // accept leading zero
                   if (value.match(/^[0-9]{1,4}$/) === null) {
                       return false;
                   }
                break;
                case "number-NLZ": // no leading zero
                   if (value.match(/^[1-9][0-9]{0,3}$/) === null) {
                       return false;
                   }
                break;
                case "price":
                    if (value.match(/^[1-9][0-9]{0,}$/) === null) {
                        if (value.match(/^0[0-9]+\./) !== null) {
                            return false;
                        }
                        if (value.match(/^[0-9]+\.[0-9]+$/) === null) {
                            return false;
                        }
                    }
                break;
            }
            return true;
        }
