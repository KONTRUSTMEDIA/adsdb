//滚动事件
function oscroll(wrapper_id,scroll_id,scroll_span_id,type){
	var oConter = document.getElementById(wrapper_id);
	if(wrapper_id == 'coming_soon_wrapper'){
		var oUl = oConter.getElementsByTagName('ul')[1];
		var visible_width = oUl.offsetWidth;
	}else{
		var oUl = oConter.getElementsByTagName('ul')[0];
		var visible_width = oConter.offsetWidth;
	}
	var oLi = oConter.getElementsByTagName('li');
	var oScroll = document.getElementById(scroll_id);
	var oSpan	= document.getElementById(scroll_span_id);
	var oLiWidth = $("."+wrapper_id+" > ul > li").width();
	var oLiNum   = Math.ceil($("."+wrapper_id).attr('total'));
	var oUlWidth = oLiWidth * oLiNum;
	var offWidthMax = (oScroll.offsetWidth - oSpan.offsetWidth)/(oUlWidth / visible_width - 1);

	if(type == 1){ //加
		var l = oSpan.offsetLeft + offWidthMax;
	}else{ //减
		var l = oSpan.offsetLeft - offWidthMax;
	}
	if(l < 0){
		l = 0;
	}else if(l > oScroll.offsetWidth - oSpan.offsetWidth){
		l = oScroll.offsetWidth - oSpan.offsetWidth;
	}
	var scrol = l / (oScroll.offsetWidth - oSpan.offsetWidth);
	sMove(oSpan, 'left', Math.ceil(l));
	var new_left = Math.ceil((oUlWidth - visible_width) * scrol);
		var error_value = (Math.floor(new_left/visible_width) * visible_width) - new_left;
		new_left += error_value;
	sMove(oUl, 'left', '-'+new_left);
	
}
//拖拽滚动条
function initialize_scroll_drag(wrapper_id,scroll_id,scroll_span_id){
	var oConter = document.getElementById(wrapper_id);
	if(wrapper_id == 'coming_soon_wrapper'){
		var oUl = oConter.getElementsByTagName('ul')[1];
		var visible_width = oUl.offsetWidth;
	}else{
		var oUl = oConter.getElementsByTagName('ul')[0];
		var visible_width = oConter.offsetWidth;
	}
	var oLi = oConter.getElementsByTagName('li')[0];
	var oScroll = document.getElementById(scroll_id);
	var oSpan	= document.getElementById(scroll_span_id);
	var iX 		= 0;
	var oUlWidth = 0;
	var oLiNum  = 0;
	
	oSpan.onmousedown = function(e){
		var oEvent = e || event;
		iX = oEvent.clientX - oSpan.offsetLeft;
		//oUlWidth = oLi.offsetWidth;
		var oLiWidth = $("."+wrapper_id+" > ul > li").width();
		var oLiNum   = Math.ceil($("."+wrapper_id).attr('total'));
		//console.log(oLiWidth);
		document.onmousemove = function(e){
			var oUlWidth = oLiWidth * oLiNum;
			var oEvent = e || event;
			var l = oEvent.clientX - iX;
			if(l < 0){
				l = 0;
			}else if(l > oScroll.offsetWidth - oSpan.offsetWidth){
				l = oScroll.offsetWidth - oSpan.offsetWidth;
			}
			var scrol = l / (oScroll.offsetWidth - oSpan.offsetWidth);
			oSpan.style.left = l+'px';
			oUl.style.left = '-'+(oUlWidth - visible_width) * scrol +'px';
			return false;
		};
		document.onmouseup = function(){
			echo.render();
			document.onmousemove = null;
			document.onmouseup = null;
		};
		return false;
	};
}

//运动框架
function getStyle(obj, attr){
	return obj.currentStyle ? obj.currentStyle[attr] : getComputedStyle(obj, false)[attr];
}
function sMove(obj, attr, iT, onEnd){
	clearInterval(obj.timer);
	obj.timer = setInterval(function(){
		dMove(obj, attr, iT, onEnd);
	},30);
}
function dMove(obj, attr, iT, onEnd){
	var iCur = 0;
	attr == 'opacity' ? iCur = parseInt(parseFloat(getStyle(obj, attr)*100)) : iCur = parseInt(getStyle(obj, attr));
	var iS = (iT - iCur) / 5;
	iS = iS > 0 ? Math.ceil(iS) : Math.floor(iS);
	if(iCur == iT){
		clearInterval(obj.timer);
		if(onEnd){
			onEnd();
		}
	}else{
		if(attr == 'opacity'){
			obj.style.ficter = 'alpha(opacity:'+(iCur + iS)+')';
			obj.style.opacity = (iCur + iS) / 100;
		}else{
			obj.style[attr] = iCur + iS +'px';
		}
	}
	echo.render();
}

function setCookie(name,value,time)
{
	var exp = new Date(); 
	exp.setTime(exp.getTime() + time);
	document.cookie = name + "="+ escape (value) + ";path=/;expires=" + exp.toGMTString();
}
function getCookie(name)
{
	var arr,reg=new RegExp("(^| )"+name+"=([^;]*)(;|$)");
	if(arr=document.cookie.match(reg)) return unescape(arr[2]);
	else return null;
}
function delCookie(name)
{
	var exp = new Date();
	exp.setTime(exp.getTime() - 15000);
	var cval=getCookie(name);
	if(cval!=null) document.cookie= name + "="+cval+";path=/;expires="+exp.toGMTString();
}
