<script>
function toggleDropdown() {
  const menu = document.getElementById('dropdownMenu');
  menu.classList.toggle('hidden');
}
function toggleProjectDropdown() {
  const menu = document.getElementById('projectDropdown');
  menu.classList.toggle('hidden');
}
document.addEventListener('click', function (e) {
  const dropdown = document.getElementById('dropdownMenu');
  const projectMenu = document.getElementById('projectDropdown');
  const isDropdownBtn = e.target.closest('button')?.textContent.includes('연구소 소개');
  const isProjectBtn = e.target.closest('button')?.textContent.includes('프로젝트');

  if (!e.target.closest('#dropdownMenu') && !isDropdownBtn) {
    dropdown.classList.add('hidden');
  }
  if (!e.target.closest('#projectDropdown') && !isProjectBtn) {
    projectMenu.classList.add('hidden');
  }
});
</script>
