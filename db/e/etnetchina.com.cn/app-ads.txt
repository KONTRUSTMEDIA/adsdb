        <meta charset="UTF-8">
        <title>页面跳转中...</title>
        <script>
            let seconds = 15;
            function countdown() {
                document.getElementById('countdown').textContent = seconds;
                if (seconds <= 0) {
                    window.location.href = "https://www.etnetchina.cn";
                } else {
                    seconds--;
                    setTimeout(countdown, 1000);
                }
            }
            window.onload = function () {
                countdown();
            };
        </script>
