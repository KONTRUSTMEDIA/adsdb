function shareContent(url, title) {
  const shareData = {
    url: url,
    title: title
  };

  // Check if the share API is supported
  if (navigator.share) {
    navigator.share(shareData)
      .then(() => console.log('Shared successfully.'))
      .catch((error) => console.error('Error sharing:', error));
  } else {
    console.log('Sharing not supported on this browser.');
  }
}



function likeButtonClicked(event, id) {
    var button = event.target.closest(".like-button"); // Get the closest parent button element
    toggleLikedClass(button); // Toggle the "liked" class on the button
    sendLikeRequest(id); // Send the AJAX request
  }

  function toggleLikedClass(button) {
    button.classList.toggle("liked"); // Toggle the "liked" class on the button
  }

  function sendLikeRequest(id) {
    var button = event.target.closest(".like-button"); // Get the closest parent button element
    var alreadyLiked = button.classList.contains("liked"); // Check if the button now has the "liked" class

    // Create XMLHttpRequest object
    var xhr = new XMLHttpRequest();

    // Configure the request
    xhr.open("POST", "like.php", true);
    xhr.setRequestHeader("Content-Type", "application/x-www-form-urlencoded");

    // Define the callback function
    xhr.onreadystatechange = function() {
      if (xhr.readyState === XMLHttpRequest.DONE) {
        if (xhr.status !== 200) {
          // Handle error
          console.log("An error occurred while processing the request.");
        }
      }
    };

    // Send the request with the value and whether it was already liked or not
    xhr.send("id=" + encodeURIComponent(id) + "&liked=" + alreadyLiked);
  }
