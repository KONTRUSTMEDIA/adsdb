<script>
  // Function untuk menampilkan atau menyembunyikan jendela chat
  function toggleChat() {
    const chatWindow = document.getElementById("chatWindow");
    chatWindow.style.display = chatWindow.style.display === "none" || chatWindow.style.display === "" ? "block" : "none";
  }

  // Function untuk mengirim pesan ke WhatsApp
  function sendMessage() {
    const messageInput = document.getElementById("userMessage");
    const messageText = messageInput.value;

    if (messageText.trim() !== "") {
      // Arahkan ke WhatsApp dengan isi pesan
      const phoneNumber = "6282123460040"; // Nomor WhatsApp tujuan
      const waURL = `https://wa.me/${phoneNumber}?text=${encodeURIComponent(messageText)}`;
      window.open(waURL, "_blank");
      messageInput.value = ""; // Kosongkan input setelah terkirim
    }
  }
</script>

<script>
document.addEventListener('DOMContentLoaded', function () {
    const popups = Array.from(document.querySelectorAll('.popup-ad'));
    let currentIndex = 0;

    // Update countdown untuk semua popup
    popups.forEach(popup => {
        const id = popup.dataset.id;
        const countdownEl = document.getElementById(`countdown-${id}`);
        function updateCountdown() {
            const now = new Date();
            const midnight = new Date();
            midnight.setHours(24, 0, 0, 0);
            const diff = midnight - now;

            if (!countdownEl) return;

            if (diff <= 0) {
                countdownEl.textContent = "Waktu habis!";
                return;
            }

            const h = Math.floor(diff / (1000 * 60 * 60));
            const m = Math.floor((diff % (1000 * 60 * 60)) / (1000 * 60));
            const s = Math.floor((diff % (1000 * 60)) / 1000);

            countdownEl.textContent =
                `Sisa waktu: ${String(h).padStart(2, '0')}:${String(m).padStart(2, '0')}:${String(s).padStart(2, '0')}`;
        }
        updateCountdown();
        setInterval(updateCountdown, 1000);
    });

    // Menampilkan popup satu per satu
    function showNextPopup() {
        if (currentIndex >= popups.length) return;

        const popup = popups[currentIndex];
        const id = popup.dataset.id;
        const delay = parseInt(popup.dataset.delay) || 0;

        const today = new Date().toISOString().split('T')[0];
        const lastShownDate = localStorage.getItem('popupLastShownDate-' + id);
        let shownCount = parseInt(localStorage.getItem('popupShownCount-' + id)) || 0;

        if (lastShownDate !== today) {
            shownCount = 0;
            localStorage.setItem('popupShownCount-' + id, 0);
            localStorage.setItem('popupLastShownDate-' + id, today);
        }

        if (shownCount < 2) {
            setTimeout(() => {
                popup.style.display = 'flex';
                setTimeout(() => {
                    popup.style.opacity = '1'; // Fade in 3s
                }, 50);

                localStorage.setItem('popupShownCount-' + id, shownCount + 1);
            }, delay * 1000);
        } else {
            currentIndex++;
            showNextPopup();
        }
    }

    // Tutup popup dan tampilkan popup berikutnya
    window.closePopup = function (id) {
        const popup = document.getElementById('popup-ad-' + id);
        if (!popup) return;

        popup.style.opacity = '0'; // Fade out
        setTimeout(() => {
            popup.style.display = 'none';
            currentIndex++;
            showNextPopup();
        }, 500); // Waktu fade-out
    };

    // Mulai tampilkan
    showNextPopup();
});
</script>
