    function removeHighlights() {
      document.querySelectorAll(".highlight").forEach((el) => {
        const parent = el.parentNode;
        parent.replaceChild(document.createTextNode(el.textContent), el);
      });
    }

    function highlightMatches(query) {
      const walker = document.createTreeWalker(document.body, NodeFilter.SHOW_TEXT, {
        acceptNode: (node) =>
          node.nodeValue.toLowerCase().includes(query) ? NodeFilter.FILTER_ACCEPT : NodeFilter.FILTER_REJECT,
      });

      const highlights = [];
      while (walker.nextNode()) {
        const node = walker.currentNode;
        const parent = node.parentNode;

        const regex = new RegExp(`(${query})`, "gi");
        const fragment = document.createDocumentFragment();

        // Split the text node and wrap matches in a span
        node.nodeValue.split(regex).forEach((part) => {
          if (regex.test(part)) {
            const span = document.createElement("span");
            span.className = "highlight";
            span.textContent = part;
            fragment.appendChild(span);
            highlights.push(span);
          } else {
            fragment.appendChild(document.createTextNode(part));
          }
        });

        parent.replaceChild(fragment, node);
      }

      return highlights;
    }

    searchButton.addEventListener("click", () => {
      const query = searchInput.value.trim().toLowerCase();
      if (!query) {
        alert("Please enter a search term.");
        return;
      }

      // Remove previous highlights
      removeHighlights();

      // Highlight new matches
      const highlights = highlightMatches(query);

      if (highlights.length > 0) {
        searchResults.textContent = `Found ${highlights.length} matches for "${query}"!`;

        // Scroll to the first match and add hover effect
        const firstMatch = highlights[0];
        firstMatch.classList.add("hover-highlight");

          firstMatch.style.backgroundColor = "orange";
        });

        firstMatch.addEventListener("mouseout", () => {

        });
      } else {
        searchResults.textContent = `No matches found for "${query}".`;
      }
