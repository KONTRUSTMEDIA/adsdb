    <section id="features" class="features-area pb-70">
        <div class="container">
            <!-- Based on Pandapower -->
            <div class="analysis-framework-section wow animate__animated animate__fadeInUp" data-wow-delay=".2s">
                <div class="analysis-framework-header">
                    <h3>Based on <a style="color:#2f5bea" href="http://www.pandapower.org/">Pandapower</a></h3>
                    <div class="bar"></div>
                </div>
                <div class="row analysis-grid">
                    <div class="col-lg-4 col-md-6">
                        <div class="analysis-card">
                            <h2>Load Flow</h2>
                            <p>Power flow solver based on the Newton-Raphson method. The implementation was originally based on PYPOWER, but has been improved with respect to robustness, runtime and usability. Backward/forward sweep, Iwamoto, fast decoupled and Gauss-Seidel methods are also available.</p>
                        </div>
                    </div>
                    <div class="col-lg-4 col-md-6">
                        <div class="analysis-card">
                            <h2>Short Circuit</h2>
                            <p>Short-circuit calculation with two engine options: <strong>Pandapower</strong> (IEC 60909) for three-phase, two-phase and single-phase fault currents with power converter modeling (PV, wind); and <strong>OpenDSS</strong> fault study for conventional fault analysis on all buses.</p>
                        </div>
                    </div>
                    <div class="col-lg-4 col-md-6">
                        <div class="analysis-card">
                            <h2>Optimal Power Flow</h2>
                            <p>Optimal power flow constrained by AC or DC loadflow equations. Branch constraints represent maximum apparent power loading of transformers and line current loadings. Bus constraints cover voltage magnitude and angle limits.</p>
                        </div>
                    </div>
                    <div class="col-lg-4 col-md-6">
                        <div class="analysis-card">
                            <h2>Contingency Analysis</h2>
                            <p>Define N-1 cases as contingencies. Lines and transformers can be switched off one at a time. Power flow calculations determine min/max values among all N-1 cases for relevant variables.</p>
                        </div>
                    </div>
                    <div class="col-lg-4 col-md-6">
                        <div class="analysis-card">
                            <h2>Controller Simulation</h2>
                            <p>The control module simulates elements controlled by power flows. Examples: tap changer controllers adapting transformer tap based on bus voltage, or droop controllers in PV plants adapting reactive power based on bus voltage.</p>
                        </div>
                    </div>
                    <div class="col-lg-4 col-md-6">
                        <div class="analysis-card">
                            <h2>Time Series Simulation</h2>
                            <p>Time-based simulation linked to the control module. Controllers update element values in each time step in a loop, ideal for daily or seasonal analysis.</p>
                        </div>
                    </div>
                </div>
            </div>

            <!-- Based on OpenDSS -->
            <div class="analysis-framework-section wow animate__animated animate__fadeInUp" data-wow-delay=".3s">
                <div class="analysis-framework-header">
                    <h3>Based on <a style="color:#2f5bea" href="https://opendss.epri.com/">OpenDSS</a></h3>
                    <div class="bar"></div>
                </div>
                <div class="row analysis-grid">
                    <div class="col-lg-4 col-md-6">
                        <div class="analysis-card">
                            <h2>Load Flow</h2>
                            <p>Comprehensive electrical system simulation for utility distribution systems. Supports RMS steady-state analyses including snapshot, daily, yearly power flow, harmonics, dynamics, fault study, and Monte Carlo analysis.</p>
                        </div>
                    </div>
                    <div class="col-lg-4 col-md-6">
                        <div class="analysis-card">
                            <h2>Short Circuit</h2>
                            <p>Fault study computes fault currents and short-circuit impedances at every bus. Use <em>Solve Mode=FaultStudy</em> for initial symmetrical short-circuit current (I″<sub>k</sub>), peak current (i<sub>p</sub>), thermal current (I<sub>th</sub>), and R<sub>k</sub>, X<sub>k</sub>. Select the OpenDSS tab in the Short Circuit dialog.</p>
                        </div>
                    </div>
                    <div class="col-lg-4 col-md-6">
                        <div class="analysis-card">
                            <h2>Harmonic Analysis</h2>
                            <p>Computes voltage and current harmonic distortion. Use <em>Solve Mode=Harmonic</em> in the Power Flow dialog. Specify harmonic orders (e.g. 3, 5, 7, 11, 13) and view magnitude spectra. Supports harmonic loads (TCR, HVDC spectra), single-tuned filters, and THD results.</p>
                        </div>
                    </div>
                </div>
            </div>

        // Currency Conversion - Stripe-like approach
        (function() {
            // Prevent double execution
            if (window._currencyConversionDone) return;
            window._currencyConversionDone = true;
            
            const USD_PRICE = 5;
            let userCurrency = 'USD';
            let exchangeRate = 1;
            
            // Detect user's currency based on locale
            function detectCurrency() {
                try {
                    const locale = navigator.language || navigator.userLanguage;
                    let region;
                    
                    // Try Intl.Locale (newer browsers)
                    try {
                        const localeObj = new Intl.Locale(locale);
                        region = localeObj.region;
                        // If region is undefined, use language code as fallback
                        if (!region) {
                            region = localeObj.language.toUpperCase();
                        }
                    } catch (e) {
                        // Fallback: extract region from locale string (e.g., "en-US" -> "US", "pl" -> "PL")
                        const parts = locale.split('-');
                        region = parts.length > 1 ? parts[1].toUpperCase() : parts[0].toUpperCase();
                    }
                    
                    // Map language codes to regions (for language-only locales like "pl", "cs", etc.)
                    const langToRegion = {
                        'PL': 'PL', 'CS': 'CZ', 'CZ': 'CZ', 'HU': 'HU', 'RO': 'RO', 'BG': 'BG',
                        'HR': 'HR', 'SV': 'SE', 'SE': 'SE', 'NB': 'NO', 'NO': 'NO', 'DA': 'DK', 'DK': 'DK',
                        'SK': 'SK', 'SL': 'SI', 'SI': 'SI', 'ET': 'EE', 'EE': 'EE', 
                        'LV': 'LV', 'LT': 'LT', 'FI': 'FI', 'UK': 'UA', 'UA': 'UA',
                        'RU': 'RU', 'TR': 'TR', 'JA': 'JP', 'JP': 'JP', 'ZH': 'CN', 
                        'KO': 'KR', 'KR': 'KR', 'VI': 'VN', 'TH': 'TH', 'ID': 'ID',
                        'PT': 'BR', 'BR': 'BR'
                    };
                    if (langToRegion[region]) {
                        region = langToRegion[region];
                    }
                    
                    // Map common regions to currencies (similar to Stripe)
                    const regionToCurrency = {
                        'US': 'USD', 'CA': 'CAD', 'GB': 'GBP', 'AU': 'AUD', 'NZ': 'NZD',
                        'JP': 'JPY', 'CN': 'CNY', 'IN': 'INR', 'KR': 'KRW', 'SG': 'SGD',
                        'HK': 'HKD', 'TW': 'TWD', 'MY': 'MYR', 'TH': 'THB', 'ID': 'IDR',
                        'PH': 'PHP', 'VN': 'VND', 'BR': 'BRL', 'MX': 'MXN', 'AR': 'ARS',
                        'CL': 'CLP', 'CO': 'COP', 'PE': 'PEN', 'ZA': 'ZAR', 'EG': 'EGP',
                        'AE': 'AED', 'SA': 'SAR', 'IL': 'ILS', 'TR': 'TRY', 'RU': 'RUB',
                        'PL': 'PLN', 'CZ': 'CZK', 'HU': 'HUF', 'RO': 'RON', 'BG': 'BGN',
                        'HR': 'HRK', 'SE': 'SEK', 'NO': 'NOK', 'DK': 'DKK', 'CH': 'CHF',
                        'UA': 'UAH',
                        'AT': 'EUR', 'BE': 'EUR', 'DE': 'EUR', 'ES': 'EUR', 'FR': 'EUR',
                        'IE': 'EUR', 'IT': 'EUR', 'NL': 'EUR', 'PT': 'EUR', 'FI': 'EUR',
                        'GR': 'EUR', 'LU': 'EUR', 'MT': 'EUR', 'CY': 'EUR', 'SK': 'EUR',
                        'SI': 'EUR', 'EE': 'EUR', 'LV': 'EUR', 'LT': 'EUR'
                    };
                    
                    userCurrency = regionToCurrency[region] || 'USD';
                    
                    // Debug logging (can be removed in production)
                    console.log('Currency detection:', { locale, region, userCurrency });
                    
                    // Try to get currency from Intl API
                    if (!regionToCurrency[region]) {
                        const formatter = new Intl.NumberFormat(locale, {
                            style: 'currency',
                            currency: 'USD'
                        });
                        const parts = formatter.formatToParts(1);
                        const currencyCode = parts.find(part => part.type === 'currency');
                        if (currencyCode && currencyCode.value !== 'USD') {
                            // Try to detect from currency symbol
                            const currencyMap = {
                                '€': 'EUR', '£': 'GBP', '¥': 'JPY', '₹': 'INR',
                                'R$': 'BRL', '₽': 'RUB', '₩': 'KRW', '₪': 'ILS',
                                '₦': 'NGN', '₨': 'PKR', '₫': 'VND', '₱': 'PHP'
                            };
                            // This is a fallback - we'll use region detection primarily
                        }
                    }
                } catch (e) {
                    console.log('Currency detection error:', e);
                    userCurrency = 'USD';
                }
            }
            
            // Fetch exchange rate
            async function fetchExchangeRate() {
                if (userCurrency === 'USD') {
                    exchangeRate = 1;
                    return;
                }
                
                try {
                    // Using a free public API (no key required)
                    const response = await fetch(`https://api.exchangerate-api.com/v4/latest/USD`);
                    const data = await response.json();
                    exchangeRate = data.rates[userCurrency] || 1;
                } catch (e) {
                    console.log('Exchange rate fetch error:', e);
                    // Fallback: use approximate rates for major currencies
                    const fallbackRates = {
                        'EUR': 0.92, 'GBP': 0.79, 'JPY': 150, 'CAD': 1.35,
                        'AUD': 1.52, 'CHF': 0.88, 'CNY': 7.2, 'INR': 83,

                        'SEK': 10.5, 'NOK': 10.8, 'DKK': 6.9, 'TRY': 32,
                        'RUB': 90, 'UAH': 41, 'BGN': 1.8, 'HRK': 7.0,
                        'KRW': 1350, 'TWD': 32, 'THB': 36, 'IDR': 16000,
                        'PHP': 56, 'VND': 25000, 'MYR': 4.7, 'ZAR': 18
                    };
                    exchangeRate = fallbackRates[userCurrency] || 1;
                }
            }
            
            // Format price in local currency
            function formatPrice(amount, currency) {
                try {
                    const formatter = new Intl.NumberFormat(navigator.language || 'en-US', {
                        style: 'currency',
                        currency: currency,
                        minimumFractionDigits: currency === 'JPY' ? 0 : 2,
                        maximumFractionDigits: currency === 'JPY' ? 0 : 2
                    });
                    return formatter.format(amount);
                } catch (e) {
                    // Fallback formatting
                    const symbols = {
                        'USD': '$', 'EUR': '€', 'GBP': '£', 'JPY': '¥',
                        'CNY': '¥', 'INR': '₹', 'KRW': '₩', 'BRL': 'R$',
                        'MXN': '$', 'CAD': '$', 'AUD': '$', 'SGD': 'S$',
                        'PLN': 'zł', 'CZK': 'Kč', 'HUF': 'Ft', 'RON': 'lei',
                        'SEK': 'kr', 'NOK': 'kr', 'DKK': 'kr', 'CHF': 'CHF',
                        'TRY': '₺', 'RUB': '₽', 'UAH': '₴', 'THB': '฿',
                        'VND': '₫', 'PHP': '₱', 'IDR': 'Rp', 'MYR': 'RM',
                        'ZAR': 'R', 'TWD': 'NT$', 'HKD': 'HK$'
                    };
                    const symbol = symbols[currency] || currency + ' ';
                    return symbol + amount.toFixed(currency === 'JPY' ? 0 : 2);
                }
            }
            
            // Update all price displays
            function updatePrices() {
                // Skip if user is in US (no conversion needed)
                if (userCurrency === 'USD') return;
                
                const convertedPrice = USD_PRICE * exchangeRate;
                const formattedPrice = formatPrice(convertedPrice, userCurrency);
                const priceText = formattedPrice + '/month';
                const priceTextShort = formattedPrice.replace(/\.\d{2}/, '') + '/mo';
                const priceNum = convertedPrice.toFixed(userCurrency === 'JPY' ? 0 : 2);
                
                // Helper to safely replace text in HTML
                function replaceInHTML(html, search, replace) {
                    return html.replace(new RegExp(search.replace(/[.*+?^${}()|[\]\\]/g, '\\$&'), 'gi'), replace);
                }
                
                // Update hero section - Cloud Edition heading
                const heroHeading = document.querySelector('.pricing-highlight h4');
                if (heroHeading && heroHeading.innerHTML.includes('$5/month')) {
                    heroHeading.innerHTML = replaceInHTML(heroHeading.innerHTML, '$5/month', priceText);
                }
                
                // Update pricing section - main price display
                const priceSpan = document.querySelector('.price span[style*="56px"], .price span[style*="font-size: 56px"]');
                if (priceSpan) {
                    priceSpan.textContent = priceNum;
                }
                
                // Update all price superscripts (currency symbol) - update ALL pricing cards
                const priceSups = document.querySelectorAll('.price sup');
                if (priceSups.length > 0) {
                    // Get currency symbol - use symbol map for reliability
                    const currencySymbols = {
                        'USD': '$', 'EUR': '€', 'GBP': '£', 'JPY': '¥',
                        'CNY': '¥', 'INR': '₹', 'KRW': '₩', 'BRL': 'R$',
                        'MXN': '$', 'CAD': '$', 'AUD': '$', 'SGD': 'S$',
                        'PLN': 'zł', 'CZK': 'Kč', 'HUF': 'Ft', 'RON': 'lei',
                        'SEK': 'kr', 'NOK': 'kr', 'DKK': 'kr', 'CHF': 'CHF',
                        'TRY': '₺', 'RUB': '₽', 'UAH': '₴', 'THB': '฿',
                        'VND': '₫', 'PHP': '₱', 'IDR': 'Rp', 'MYR': 'RM',
                        'ZAR': 'R', 'TWD': 'NT$', 'HKD': 'HK$'
                    };
                    const currencySymbol = currencySymbols[userCurrency] || '$';
                    priceSups.forEach(function(sup) {
                        sup.textContent = currencySymbol;
                    });
                }
                
                // Update all text nodes containing $5 (but not already converted)
                const walker = document.createTreeWalker(
                    document.body,
                    NodeFilter.SHOW_TEXT,
                    null,
                    false
                );
                
                const textNodes = [];
                let node;
                while (node = walker.nextNode()) {
                    // Only collect nodes with unconverted prices ($5 not followed by .)
                    if (/\$5(?!\.)/.test(node.textContent)) {
                        textNodes.push(node);
                    }
                }
                
                textNodes.forEach(function(textNode) {
                    let text = textNode.textContent;
                    // Replace in specific order to avoid double replacement
                    text = text.replace(/\$5\/month/g, priceText);
                    text = text.replace(/\$5\/mo\b/g, priceTextShort);
                    // Only match $5 when NOT followed by a digit or decimal
                    text = text.replace(/\$5(?![0-9.])/g, formattedPrice);
                    textNode.textContent = text;
                });
                
                // Update floating CTA button
                const floatingCta = document.getElementById('floatingCta');
                if (floatingCta) {
                    floatingCta.innerHTML = replaceInHTML(floatingCta.innerHTML, '$5/mo', priceTextShort);
                }
                
                // Update Cloud Service description
                const cloudCards = document.querySelectorAll('.opensource-card');
                cloudCards.forEach(function(card) {
                    const cloudIcon = card.querySelector('.fa-cloud');
                    if (cloudIcon) {
                        const p = card.querySelector('p');
                        if (p && p.textContent.includes('$5/month')) {
                            p.innerHTML = replaceInHTML(p.innerHTML, '$5/month', priceText);
                        }
                    }
                });
            }
            
            // Initialize
            async function init() {
                detectCurrency();
                await fetchExchangeRate();
                updatePrices();
            }
            
            // Run on page load
            if (document.readyState === 'loading') {
                document.addEventListener('DOMContentLoaded', init);
            } else {
                init();
            }
        })();
