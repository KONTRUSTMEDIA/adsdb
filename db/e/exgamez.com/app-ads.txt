(function () {
  // Cursor DOM'u oluştur
  const cursor = document.createElement('div');
  cursor.id = 'exg-cursor';
  cursor.innerHTML = `
    <img class="exg-cursor-inner"
         src="https://exgamez.com/wp-content/uploads/2026/01/cursor2.png"
         alt="">
  `;
  document.body.appendChild(cursor);

  let shown = false;

  // Uç nokta (hotspot)
  const hotspot = { x: 6, y: 6 };

  function setPos(x, y) {
    cursor.style.transform = `translate3d(${x - hotspot.x}px, ${y - hotspot.y}px, 0)`;
  }

  function isTextTarget(el) {
    return el && el.closest('input, textarea, select, [contenteditable="true"]');
  }

  function isInteractive(el) {
    return el && el.closest('a, button, .game-item, .play-button, [role="button"]');
  }

  // Hareket: text-mode değilken takip et
  function onMove(e) {
    if (document.documentElement.classList.contains('exg-text-mode')) return;

    setPos(e.clientX, e.clientY);

    if (!shown) {
      cursor.style.opacity = '1';
      shown = true;
    }
  }

  window.addEventListener('pointermove', onMove, { passive: true });

  // Text-mode aç/kapat + hover büyüme
  document.addEventListener('pointerover', (e) => {
    // input/textarea üstünde: text-mode ON
    if (isTextTarget(e.target)) {
      document.documentElement.classList.add('exg-text-mode');
      cursor.classList.remove('is-hover', 'is-down');
      return;
    }

    // text-mode OFF (diğer alanlar)
    document.documentElement.classList.remove('exg-text-mode');

    // interaktif öğe üstünde büyüme
    if (isInteractive(e.target)) cursor.classList.add('is-hover');
  }, { passive: true });

  document.addEventListener('pointerout', (e) => {
    if (isInteractive(e.target)) cursor.classList.remove('is-hover');
  }, { passive: true });

  // Click efekti (text-mode değilken)
  window.addEventListener('pointerdown', () => {
    if (document.documentElement.classList.contains('exg-text-mode')) return;
    cursor.classList.add('is-down');
  }, { passive: true });

  window.addEventListener('pointerup', () => {
    cursor.classList.remove('is-down');
  }, { passive: true });

})();
