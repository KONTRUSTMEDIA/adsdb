            // load more

            let lastPostID = -1

            var lastScrollTop = 0;
            var lastTriggerPosition = 0;
            var loading = false;
            var threshold = 3500; // distância do final da página para acionar

            $(window).on('scroll', function() {

                if( !$('.post_list_container').length ) return;

                var scrollTop = $(window).scrollTop();
                var windowHeight = $(window).height();
                var documentHeight = $(document).height();

                var scrollDirection = scrollTop > lastScrollTop ? 'down' : 'up';
                lastScrollTop = scrollTop;

                // Só continua se estiver rolando pra baixo
                if (scrollDirection !== 'down') return;

                // console.log( $('.post_list_container').outerHeight() )

                // console.log('ScrollTop:', scrollTop)

                // altura da div
                let divTotal = $('.post_list_container').outerHeight()

                // se a altura da div - threshold < que o scrollTop, já carrega
                if ( divTotal - threshold < scrollTop ) {

                    if (loading) return

                    if ( scrollTop > lastTriggerPosition ) {

                        loading = true;
                        lastTriggerPosition = scrollTop;

                        // console.log('Carrega +')

                        carregarMaisConteudo().then(function() { loading = false; });

                    }

                }

            });
