        const words = [

        ];

        const element = document.getElementById("typewriter");
        let wordIndex = 0, charIndex = 0;

        function type() {
            if (charIndex < words[wordIndex].length) {
                element.textContent += words[wordIndex].charAt(charIndex);
                charIndex++;
                setTimeout(type, 100);
            } else {
                setTimeout(() => {
                    element.textContent = '';
                    charIndex = 0;
                    wordIndex = (wordIndex + 1) % words.length;
                    type();
                }, 1000);
            }
        }
