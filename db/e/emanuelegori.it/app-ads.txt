            function wrapNode(node, url){
                if (!url) return;
                if (!node || !node.parentNode) return;

                // Évite double-wrap
                if (node.closest && node.closest('span.open-img')) {
                    // Met quand même à jour le handler du wrapper existant
                    var wrapper = node.closest('span.open-img');
                    wrapper.onclick = function(e){
                        e.preventDefault();
                        window.open(url, '_blank');
                    };
                    wrapper.style.cursor = 'pointer';
                    return;

                var wrapper = makeWrapper(url);
                node.parentNode.insertBefore(wrapper, node);
                wrapper.appendChild(node);

                var fromImg = mediaEl && mediaEl.getAttribute && mediaEl.getAttribute('data-open-url');
                var fromContainer = container && container.getAttribute && container.getAttribute('data-open-url');

                pictures.forEach(function(pic){
                    var url = getUrlFor(container, pic);
                    if (url) wrapNode(pic, url);

                // Pour les <img> qui ne sont pas déjà dans un <picture> ou déjà wrap
                imgs.forEach(function(img){
                    if (img.closest('picture')) return; // déjà géré via <picture>

                    if (url) wrapNode(img, url);
