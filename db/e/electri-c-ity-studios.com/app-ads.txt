<script type="bb706e8cd301a84a48166331-text/javascript">
(() => {
  const root = document.querySelector('[data-ecity-now][data-station]');
  if (!root || root.__ecityNowInit) return;
  root.__ecityNowInit = true;

  const station = root.getAttribute('data-station');
  const textEl  = root.querySelector('[data-ecity-now-text]');
  const statusEl= root.querySelector('[data-ecity-now-status]');
  const updEl   = root.querySelector('[data-ecity-now-updated]');

  if (!station || !textEl || !statusEl || !updEl) return;

  const urlNow = `https://api.laut.fm/station/${encodeURIComponent(station)}/current_song`;

  const fmt = (s) => {
    const artist = s?.artist?.name ?? s?.artist ?? 'Unknown Artist';
    const title  = s?.title ?? 'Unknown Title';
    return `${artist} – ${title}`;
  };

  const humanTime = () =>
    new Date().toLocaleString(undefined, { hour: "2-digit", minute: "2-digit", second: "2-digit" });

  let hasTrack = false;
  let inFlight = false;
  let lastId = null;
  let timer = null;

  function clearTimer() {
    if (timer) { clearTimeout(timer); timer = null; }
  }

  // “Benutzerfreundlich”: smart refresh -> kurz nach Song-Ende, sonst moderater Fallback.
  function scheduleNext(nowObj) {
    clearTimer();

    // Default: moderat (lebt, aber spamt nicht)
    let nextMs = 35000;

    // Wenn ends_at verfügbar: refresh kurz nach Ende
    if (nowObj?.ends_at) {
      const ends = Date.parse(nowObj.ends_at);
      if (!Number.isNaN(ends)) {
        const diff = ends - Date.now();
        // nicht zu aggressiv: min 6s, max 60s
        nextMs = Math.min(60000, Math.max(6000, diff + 1500));
      }
    }

    timer = setTimeout(update, nextMs);
  }

  async function update() {
    if (inFlight) return;
    inFlight = true;

    const ctrl = new AbortController();
    const hardTimeout = setTimeout(() => ctrl.abort(), 8000);

    try {
      statusEl.textContent = hasTrack ? "Live" : "Loading…";

      // Cache-Busting, damit kein Proxy/Browser dir alte Daten serviert
      const url = `${urlNow}?t=${Date.now()}`;

      const res = await fetch(url, {
        cache: "no-store",
        headers: { "Accept": "application/json" },
        signal: ctrl.signal
      });

      if (!res.ok) throw new Error(`HTTP ${res.status}`);

      const now = await res.json();

      // Falls API mal “leer” liefert
      if (!now || (!now.title && !now.artist && !now?.artist?.name)) {
        if (!hasTrack) {
          statusEl.textContent = "Idle";
          textEl.textContent = "No track info";
          updEl.textContent = "—";
        } else {
          statusEl.textContent = "Live";
          updEl.textContent = `No update (${humanTime()})`;
        }
        scheduleNext(now);
        return;
      }

      // Nur bei Änderung neu setzen (ruhiger / weniger Layout-Jitter)
      const id = now.id ?? `${now?.artist?.name ?? now?.artist ?? ""}|${now?.title ?? ""}`;
      if (id !== lastId) {
        textEl.textContent = fmt(now);
        lastId = id;
      }

      hasTrack = true;
      statusEl.textContent = "Live";
      updEl.textContent = `Updated ${humanTime()}`;

      scheduleNext(now);
    } catch (e) {
      console.warn("[ecity-now] fetch failed:", e);

      if (!hasTrack) {
        statusEl.textContent = "API error";
        textEl.textContent = "Could not load current track";
        updEl.textContent = "—";
      } else {
        statusEl.textContent = "Live";
        updEl.textContent = `Reconnecting… (${humanTime()})`;
      }

      // Bei Fehler: etwas schneller retry, aber nicht spammen
      clearTimer();
      timer = setTimeout(update, 15000);
    } finally {
      clearTimeout(hardTimeout);
      inFlight = false;
    }
  }

  // Initial load
  update();

  // Wenn Tab wieder aktiv wird: sofort einmal refreshen (fühlt sich “live” an)
  document.addEventListener("visibilitychange", () => {
    if (document.visibilityState === "visible") update();
  }, { passive: true });

  // Optional: Fokus-Refresh (Desktop)
  window.addEventListener("focus", () => update(), { passive: true });

})();
</script>

<script type="bb706e8cd301a84a48166331-text/javascript">
(() => {
  // Safe boot (in case WP moves scripts)
  const boot = () => {
    const root = document.querySelector('[data-ecity-player]');
    if (!root || root.__ecityInit) return;
    root.__ecityInit = true;

    // ✅ Stable direct stream host (your working URL base)
    const STREAMS = [
      "https://electri-c-ity-studios.stream.laut.fm/electri-c-ity-studios",
      "https://electri-c-ity-studios.stream.laut.fm/electri-c-ity-studios.mp3",
      "https://stream.laut.fm/electri-c-ity-studios" // last fallback (may be non-audio for some setups)
    ];

    const audio  = root.querySelector('[data-ecity-audio]');
    const play   = root.querySelector('[data-ecity-play]');
    const pause  = root.querySelector('[data-ecity-pause]');
    const vol    = root.querySelector('[data-ecity-vol]');
    const mute   = root.querySelector('[data-ecity-mute]');
    const status = root.querySelector('[data-ecity-status]');
    const timeEl = root.querySelector('[data-ecity-time]');
    const badge  = root.querySelector('[data-ecity-badge]');
    const hintEl = root.querySelector('[data-ecity-hint]');
    const diagEl = root.querySelector('[data-ecity-diag]');

    if (!audio || !play || !pause || !vol || !mute || !status || !timeEl || !badge) return;

    const setDiag = (t) => { if (diagEl) diagEl.textContent = t || ""; };

    let started = false;
    let startAt = 0;
    let total = 0;
    let lastPlayAttemptAt = 0;

    audio.preload = "none";
    audio.crossOrigin = "anonymous";
    audio.volume = parseFloat(vol.value || "0.85");

    const fmt = (sec) => {
      const mm = String((sec/60)|0).padStart(2,"0");
      const ss = String(sec%60).padStart(2,"0");
      return mm + ":" + ss;
    };

    setInterval(() => {
      const ms = total + (started ? (Date.now() - startAt) : 0);
      timeEl.textContent = fmt((ms/1000)|0);
    }, 500);

    const bookTimeIfRunning = () => {
      if (started) {
        total += (Date.now() - startAt);
        started = false;
      }
    };

    const markPlayingUI = () => {
      root.classList.add("is-playing");
      badge.textContent = "LIVE";
      status.textContent = "Live • läuft";
      if (hintEl) hintEl.textContent = "Stream läuft.";
    };

    const markStoppedUI = (txt) => {
      root.classList.remove("is-playing");
      badge.textContent = "OFF";
      status.textContent = txt;
    };

    const setSrc = (url) => {
      // cache-buster forces reconnect and avoids stale connections
      const u = url + (url.includes("?") ? "&" : "?") + "t=" + Date.now();
      audio.src = u;
      try { audio.load(); } catch(_) {}
    };

    const explainError = (e) => {
      const name = e?.name || "Error";
      const msg  = e?.message || String(e);
      return `${name}: ${msg}`;
    };

    async function tryPlay(){
      for (const url of STREAMS){
        try{
          setDiag("Try: " + url);
          markStoppedUI("Verbinde…");
          setSrc(url);

          // small delay for network start
          await new Promise(r => setTimeout(r, 200));

          await audio.play();

          if (!started) { started = true; startAt = Date.now(); }
          markPlayingUI();
          setDiag("OK: " + url);
          return true;
        }catch(e){
          setDiag(`Failed: ${url} | ${explainError(e)}`);
          // try next candidate
        }
      }
      markStoppedUI("Stream startet nicht");
      if (hintEl) hintEl.textContent = "Start fehlgeschlagen. Bitte erneut tippen.";
      return false;
    }

    play.addEventListener("click", async () => {
      const now = Date.now();
      if (now - lastPlayAttemptAt < 700) return;
      lastPlayAttemptAt = now;

      await tryPlay();
    });

    pause.addEventListener("click", () => {
      try { audio.pause(); } catch(_) {}
      bookTimeIfRunning();
      markStoppedUI("Pausiert");
      setDiag("Paused.");
      if (hintEl) hintEl.textContent = "Pausiert. Tippe Play zum Fortsetzen.";
    });

    vol.addEventListener("input", () => {
      audio.volume = parseFloat(vol.value);
    });

    mute.addEventListener("click", () => {
      audio.muted = !audio.muted;
      mute.textContent = audio.muted ? "🔊" : "🔇";
    });

    audio.addEventListener("playing", () => {
      if (!started) { started = true; startAt = Date.now(); }
      markPlayingUI();
    });

    audio.addEventListener("waiting", () => setDiag("Buffering…"));
    audio.addEventListener("stalled", () => setDiag("Stalled…"));

    audio.addEventListener("pause", () => {
      if (started && audio.paused) {
        bookTimeIfRunning();
        markStoppedUI("Unterbrochen");
        setDiag("Interrupted pause.");
      }
    });

    audio.addEventListener("error", () => {
      bookTimeIfRunning();
      const err = audio.error;
      const code = err ? err.code : 0; // 1 abort 2 network 3 decode 4 not supported
      markStoppedUI("Stream-Fehler");
      setDiag("Audio error code=" + code + " | src=" + (audio.currentSrc || audio.src || ""));
      if (hintEl) hintEl.textContent = "Stream-Fehler. Bitte erneut auf Play tippen.";
    });

    audio.addEventListener("ended", () => {
      bookTimeIfRunning();
      markStoppedUI("Beendet");
      setDiag("Ended.");
      if (hintEl) hintEl.textContent = "Stream beendet/unterbrochen. Bitte erneut auf Play tippen.";
    });

    // Impression
    setDiag("Ready. Click Play.");
  };

  if (document.readyState === "loading") {
    document.addEventListener("DOMContentLoaded", boot, { once: true });
  } else {
    boot();
  }
})();
</script>

(() => {
  const root = document.querySelector('.ecity-drop[data-station]');
  if (!root) return;

  const station = root.getAttribute('data-station');
  const bar = root.querySelector('.ecity-drop-bar');
  const panel = root.querySelector('#ecity-drop-panel');
  const nowEl = root.querySelector('#ecity-drop-now');
  const listEl = root.querySelector('#ecity-drop-list');
  const statusEl = root.querySelector('#ecity-drop-status');
  const updatedEl = root.querySelector('#ecity-drop-updated');

  const urlNow  = `https://api.laut.fm/station/${station}/current_song`;
  const urlLast = `https://api.laut.fm/station/${station}/last_songs`;

  const fmt = (s) =>
    `${s?.artist?.name ?? s?.artist ?? 'Unknown Artist'} – ${s?.title ?? 'Unknown Title'}`;

  async function update(){
    try{
      const [now, last] = await Promise.all([
        fetch(urlNow).then(r => r.json()),
        fetch(urlLast).then(r => r.json())
      ]);

      nowEl.textContent = fmt(now);

      listEl.innerHTML = '';
      last.slice(0, 7).forEach(s => {
        const li = document.createElement('li');
        li.className = 'ecity-drop-item';
        li.textContent = fmt(s);
        listEl.appendChild(li);
      });

      statusEl.textContent = 'Live';
      updatedEl.textContent = 'Updated';
    }catch{
      statusEl.textContent = 'API error';
    }
  }

  bar.addEventListener('click', () => {
    panel.hidden = !panel.hidden;
  });

  update();
  setInterval(update, 25000);
})();
