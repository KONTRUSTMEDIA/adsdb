    function testimonialSlider(totalTestimonials, visibleCount) {
        return {
            currentSlide: 0,
            totalTestimonials: totalTestimonials,
            visibleCount: visibleCount,
            windowWidth: window.innerWidth,
            get isMobile() {
                return this.windowWidth < 768;
            },
            get itemsPerSlide() {
                return this.isMobile ? 1 : this.visibleCount;
            },
            get totalPages() {
                return Math.ceil(this.totalTestimonials / this.itemsPerSlide);
            },
            get slideWidth() {
                return this.isMobile ? 100 : (100 / this.visibleCount);
            },
            get canGoNext() {
                return true; // Always true since we loop
            },
            nextSlide() {
                const maxSlide = this.isMobile ? this.totalTestimonials - 1 : this.totalPages - 1;
                if (this.currentSlide >= maxSlide) {
                    this.currentSlide = 0; // Loop back to start
                } else {
                    this.currentSlide++;
                }
            },
            previousSlide() {
                const maxSlide = this.isMobile ? this.totalTestimonials - 1 : this.totalPages - 1;
                if (this.currentSlide <= 0) {
                    this.currentSlide = maxSlide; // Loop to end
                } else {
                    this.currentSlide--;
                }
            },
            goToPage(page) {
                this.currentSlide = page;
            },
            init() {
                // Reset slide position on resize and update windowWidth to trigger reactivity
                window.addEventListener('resize', () => {
                    this.windowWidth = window.innerWidth;
                    this.currentSlide = 0;
                });
            }
        }
    }
