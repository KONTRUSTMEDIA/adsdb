	function submitForm() {
		var data = $("#logInForm").serialize();
		$.ajax({
			type: 'POST',
			url: 'https://www.eduparc.in/ajax-files/OrgLogin.php',

	function ValidateEmail(Email) {
		var expr = /^([\w-\.]+)@((\[[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}\.)|(([\w-]+\.)+))([a-zA-Z]{2,4}|[0-9]{1,3})(\]?)$/;
		return expr.test(Email);
	};


	$('#mc-embedded-subscribe').click(function() {

		var hiddenData = $('#hiddenData').val();

		var Email = $('#mce-EMAIL').val();
		var captchaval = $('#captcha_code').val();

		ValidateEmail(Email);

		if (Email == "") {
			$('#mce-EMAIL').css('border-color', 'red');
			$('#mce-EMAIL').html('Please enter email.');
			flag = false;
		} else if (captchaval == "") {
			$('#captcha_code').css('border-color', 'red');
			$('.requiredCaptcha').html('Please Fill The Required Field.');
			flag = false;
		} else {
			flag = true;
		}

		//organize the data properly
		var form_data =
			'hiddenData=' + hiddenData +
			'&Email=' + Email +
			'&captchaval=' + captchaval;

		if (flag) {
			$.ajax({
				url: "https://www.eduparc.in/ajax-files/newssettler.php",
				type: "POST",
				data: form_data,

				success: function(html) {
					if (html == 1) {
						$('.responceMessage').html('<div class="successMsg" style="color:#4CAF50;background:#4caf5030;border-radius:2px;padding:10px 10px;">Thanks! Your inquiry has been submitted!</div>').fadeIn('slow');
						$('.required').fadeOut();
						setTimeout(function() {
							window.location = window.location
						}, 2000);
					}
					if (html == 0) {
						$('.responceMessage').html('<div class="errorMsg" style="color:#F44336;background:#f4433626;border-radius:2px;padding:10px 10px;">Sorry, unexpected error. Please try again !</div>').fadeIn('slow');
						$('.required').fadeOut();
					}
					if (html == 2) {
						$('.responceMessage').html('<div class="errorMsg" style="color:#F44336;background:#f4433626;border-radius:2px;padding:10px 10px;">Sorry, You have inserted the wrong phrase in the captcha box !</div>').fadeIn('slow');
						$('.required').fadeOut();
					}
				}
			});
			return false;
		}
	});
