    <script type="bd79a78cb88104d35ae290b6-text/javascript">
        const canvas = document.getElementById('snowCanvas');
        const ctx = canvas.getContext('2d');
        const snowToggle = document.getElementById('snowToggle');
        
        let width, height;
        let particles = [];
        let animationId; // Para controlar o loop
        let isSnowing = true; // Controla se a neve CONTINUA sendo gerada

        const particleCount = 100;

        function resize() {
            width = canvas.width = window.innerWidth;
            height = canvas.height = window.innerHeight;
        }
        
        function initParticles() {
            particles = [];
            for (let i = 0; i < particleCount; i++) {
                particles.push({ 
                    x: Math.random() * width, 
                    y: Math.random() * height, 
                    r: Math.random() * 2 + 0.5, 
                    s: Math.random() * 0.5 + 0.2 // Velocidade
                });
            }
        }

        function draw() {
            ctx.clearRect(0, 0, width, height);
            ctx.fillStyle = "rgba(255, 255, 255, 0.6)";
            ctx.beginPath();
            
            let activeParticles = 0; // Contador de partículas visíveis

            for (let i = 0; i < particleCount; i++) {
                const p = particles[i];
                
                // Só desenha se estiver dentro ou perto da tela
                if (p.y <= height + 10) {
                    ctx.moveTo(p.x, p.y);
                    ctx.arc(p.x, p.y, p.r, 0, Math.PI * 2, true);
                    activeParticles++;
                }
            }
            ctx.fill();
            update();
            
            // LÓGICA DE PARADA NATURAL:
            // Continua animando se o toggle estiver ligado (isSnowing)
            // OU se ainda existirem flocos caindo (activeParticles > 0)
            if (isSnowing || activeParticles > 0) {
                animationId = requestAnimationFrame(draw);
            } else {
                // Para totalmente quando tudo sumir
                cancelAnimationFrame(animationId);
                animationId = null; 
            }
        }

        function update() {
            for (let i = 0; i < particleCount; i++) {
                const p = particles[i];
                p.y += p.s; 
                
                // LÓGICA DE REGENERAÇÃO:
                // Se saiu da tela E o switch está ligado, volta pro topo.
                // Se o switch está desligado, ele ignora e deixa o floco cair pra sempre.
                if (p.y > height && isSnowing) {
                    particles[i] = { 
                        x: Math.random() * width, 
                        y: -5, 
                        r: p.r, 
                        s: p.s 
                    };
                }
            }
        }

        function toggleSnow() {
            if (snowToggle.checked) {
                // LIGAR
                isSnowing = true;
                canvas.classList.remove('snow-off'); // Fade-in do fundo escuro
                
                // Se a animação estava parada, reinicia
                if (!animationId) {
                    initParticles(); // Reinicia posições (opcional, ou deixe cair do céu)
                    draw();
                }
            } else {
                // DESLIGAR
                isSnowing = false;
                canvas.classList.add('snow-off'); // Fade-out do fundo para transparente
                // NÃO paramos a animação aqui. Deixamos o draw() terminar naturalmente.
            }
        }

        // Inicialização
        window.addEventListener('resize', () => {
            resize();
        });
        
        snowToggle.addEventListener('change', toggleSnow);

        resize();
        initParticles();
        
        // Estado inicial
        if (snowToggle.checked) {
            draw();
        } else {
            canvas.classList.add('snow-off');
        }
    </script>
