function getCookie(name) {
    let cookieArr = document.cookie.split(";");
    for (let i = 0; i < cookieArr.length; i++) {
        let cookiePair = cookieArr[i].split("=");
        if (name === cookiePair[0].trim()) {
            return decodeURIComponent(cookiePair[1]);
        }
    }
    return null;
}

function setCookie(name, value, days) {
    let expires = "";
    if (days) {
        const date = new Date();
        date.setTime(date.getTime() + (days * 24 * 60 * 60 * 1000));
        expires = "; expires=" + date.toUTCString();
    }
    document.cookie = name + "=" + value + expires + "; path=/";
}

function getCookie(name) {
    const nameEQ = name + "=";
    const ca = document.cookie.split(';');
    for (let i = 0; i < ca.length; i++) {
        let c = ca[i];
        while (c.charAt(0) === ' ') c = c.substring(1);
        if (c.indexOf(nameEQ) === 0) return c.substring(nameEQ.length);
    }
    return null;
}

<script>
document.addEventListener('DOMContentLoaded', function () {

    const params = new URLSearchParams(window.location.search);

    const lat  = params.get('lat');
    const long = params.get('long');
    const grad = params.get('grad');

    let sessionUser = {};

    // Load existing sessionUser if it exists
    const existing = sessionStorage.getItem('sessionUser');
    if (existing) {
        sessionUser = JSON.parse(existing);
    }

    // 1️⃣ Check lat & long
    if (lat && long) {
        sessionUser.lat = lat;
        sessionUser.long = long;
    }

    // 2️⃣ If grad exists, save as city
    if (grad) {
        sessionUser.city = grad;
    }

    // 3️⃣ Determine location
    sessionUser.location = !!(sessionUser.lat && sessionUser.long);

    // 4️⃣ Detect platform (mobile or desktop)
    const ua = navigator.userAgent || navigator.vendor || window.opera;

    if (/Mobi|Android|iPhone|iPad|iPod|BlackBerry|IEMobile|Opera Mini/i.test(ua)) {
        sessionUser.platform = 'mobile';
    } else {
        sessionUser.platform = 'desktop';
    }

    // 5️⃣ Detect OS
    if (/android/i.test(ua)) {
        sessionUser.platform_os = 'Android';
    } else if (/iPad|iPhone|iPod/.test(ua) && !window.MSStream) {
        sessionUser.platform_os = 'iOS';
    } else if (/Win/.test(ua)) {
        sessionUser.platform_os = 'Windows';
    } else if (/Mac/.test(ua)) {
        sessionUser.platform_os = 'Mac';
    } else if (/Linux/.test(ua)) {
        sessionUser.platform_os = 'Linux';
    } else {
        sessionUser.platform_os = 'Unknown';
    }

    // 6️⃣ Detect if user is using your app
    if (ua.includes('EvontiPhoneAplikacija')) {
        sessionUser.platform_app = 'iPhone App';
    } else if (ua.includes('EvontAndroidApp')) {
        sessionUser.platform_app = 'Android App';
    } else {
        sessionUser.platform_app = 'PWA App';
    }

    // Save sessionUser
    sessionStorage.setItem('sessionUser', JSON.stringify(sessionUser));

    // Optional: clean URL

document.addEventListener("DOMContentLoaded", function() {

    const cookieBanner = document.getElementById("cookie-banner");
    const pushBanner = document.getElementById("push-banner");
    const isDiscover = window.location.pathname.includes('/discover/');

    // --- Check if browser supports push notifications ---
    const pushSupported = ('Notification' in window) && ('PushManager' in window);
    if (!pushSupported) {
        console.log("🚫 Push notifications not supported in this browser. Hiding push banner.");
        pushBanner.style.display = "none";
        return; // stop further push banner logic
    }

    // --- Read userSettings safely ---
    let userSettings = {};
    try {
        const stored = localStorage.getItem('userSettings');
        userSettings = stored ? JSON.parse(stored) : {};
    } catch(e) {
        userSettings = {};
        console.warn('Error parsing userSettings', e);
    }

    // --- Show cookie banner if not accepted ---
    if (!userSettings.cookie) {
        cookieBanner.style.display = "flex";
    }

    // --- Function to show push banner with 1.5s delay ---
    function showPushBanner() {
        if (isDiscover && userSettings.pushNotifications === undefined) {
            setTimeout(() => {
                pushBanner.style.display = "flex";
            }, 1500);
        }
    }

    // --- Cookie "U redu" click ---
    document.getElementById("accept-cookie").addEventListener("click", function() {
        try {
            const stored = localStorage.getItem('userSettings');
            const currentSettings = stored ? JSON.parse(stored) : {};

            const updatedSettings = {
                ...currentSettings,
                cookie: true
            };
            localStorage.setItem('userSettings', JSON.stringify(updatedSettings));
            userSettings = updatedSettings;

            // Hide cookie banner
            cookieBanner.style.animation = "slideDownCookie 0.5s";
            setTimeout(() => cookieBanner.style.display = "none", 300);

            console.log("🍪 Cookie accepted.");

            // Show push banner after 1.5s (only if supported)
            showPushBanner();

        } catch(e) {
            console.error('Error updating userSettings', e);
        }
    });

    // --- Push "Accept" click ---
    document.getElementById("accept-push").addEventListener("click", async function() {
        try {
            const stored = localStorage.getItem('userSettings');
            const currentSettings = stored ? JSON.parse(stored) : {};

            const updatedSettings = {
                ...currentSettings,
                pushNotifications: true,
                onesignal_prompted: true
            };
            localStorage.setItem('userSettings', JSON.stringify(updatedSettings));

            // Hide push banner
            pushBanner.style.animation = "slideDownCookie 0.5s";
            setTimeout(() => pushBanner.style.display = "none", 500);

            // Trigger OneSignal native prompt immediately
            OneSignalDeferred.push(async function(OneSignal) {
                try {
                    await OneSignal.Notifications.requestPermission();
                    console.log("🔔 OneSignal permission requested immediately");
                } catch(e) {
                    console.warn("Push permission request failed:", e);
                }
            });

        } catch(e) {
            console.error('Error updating userSettings', e);
        }
    });

    // --- Push "Not Now" click ---
    document.getElementById("not-now-push").addEventListener("click", function() {
        try {
            const stored = localStorage.getItem('userSettings');
            const currentSettings = stored ? JSON.parse(stored) : {};

            const updatedSettings = {
                ...currentSettings,
                pushNotifications: false,
                onesignal_prompted: true
            };
            localStorage.setItem('userSettings', JSON.stringify(updatedSettings));

            // Hide push banner
            pushBanner.style.animation = "slideDownCookie 0.5s";
            setTimeout(() => pushBanner.style.display = "none", 500);

            console.log("Push banner dismissed.");
        } catch(e) {
            console.error('Error updating userSettings', e);
        }
    });

    // --- Show push banner on page load if cookies accepted but push not decided yet ---
    if (pushSupported && userSettings.cookie === true && userSettings.pushNotifications === undefined) {
        showPushBanner();
    }

});
