<script>
	function handle_mega_dropdowns(id, prefetch) {
		if (prefetch === undefined) {
			prefetch = false;
		}

		if (id != 'destinations') {
			$("#destinations_submenu").hide();
		}
		if (id != 'offers') {
			$("#offers_submenu").hide();
		}
		if (id != 'proposals') {
			$("#proposals_submenu").hide();
		}
		if (id != 'accommodation') {
			$("#accommodation_submenu").hide();
		}
		if (id != 'food') {
			$("#food_submenu").hide();
		}
		if (id != 'booking') {
			$("#booking_submenu").hide();
		}

		// MYTP.templates.main['menuIsFetched'+id] set to true in ajaxProcess GET_MENU
		if (MYTP.templates.main['menuIsFetched' + id] === undefined) {
			// console.log('Need to fetch menu ' + id);

			if (!prefetch) {
				$('#' + id + '_submenu').append('Loading...');
			}

			
			// console.log('Generic style menus');
			MYTP.ajaxCalls(
				MYTP.ajax.config.url,
				// params: menuKindId, tpId
				{
					'toDos': [{
						'what': 'GET_MENU',
						'params': [id, prefetch, 21497]
					}]
				}
			);
		} else {
			// No need to fetch
		}

		if (!prefetch) {
			MYTP.templates.main.toggle_visibility(id + "_submenu");
		}

		/*
			$("#line3").mouseleave(function(){
			    $("#"+id+"_submenu").hide();
			});
		  */
	}
</script>
