function reklamphp()
{
	var c = phpAds_findObj('kayanreklam');

	if (!c)
		return false;
	
	if (c.style)
		c = c.style;
	
	if (window.innerHeight)
		ih = window.innerHeight;
	else
		ih = document.body.clientHeight;
	
	if (window.innerWidth)
		iw = window.innerWidth;
	else
		iw = document.body.clientWidth;
	
	
	if (document.all) { 
		
		c.pixelLeft = iw + document.body.scrollLeft - 130;
		c.pixelTop = -5 + document.body.scrollTop;
	
	} else {
		c.left = iw + window.pageXOffset - 130 - 16;
		c.top = 0 + window.pageYOffset;
	
	}

	c.visibility = phpAds_adlayers_visible_e6c8d575;
}


function phpAds_simplepop_e6c8d575(what)
{
	var c = phpAds_findObj('kayanreklam');

	if (!c)
		return false;

	if (c.style)
		c = c.style;

	switch(what)
	{
		case 'close':
			phpAds_adlayers_visible_e6c8d575 = 'hidden';

			reklamphp();
			phpAds_adlayers_timerid_e6c8d575 = window.setInterval('reklamphp()', 10);


			break;
	}
}
