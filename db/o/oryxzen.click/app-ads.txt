    // 动态加载脚本函数
    function loadScript(src) {
      return new Promise((resolve, reject) => {
        const script = document.createElement('script');
        script.src = src;
        script.onload = resolve;
        script.onerror = reject;
        document.head.appendChild(script);
      });
    }

    window.addEventListener('DOMContentLoaded', () => {
      setTimeout(() => {
        Promise.all([
          loadScript('./assets/js/vue/vue3.global.prod.js'),
          loadScript('./assets/js/game/indexGameData.js'),
          // loadScript('./assets/js/tailwind.js'),
        ]).then(() => {
          // 确保initApp函数已定义再调用
          if (typeof initApp === 'function') {
            initApp();
          } else {
            console.error('initApp function not defined');
          }
        }).catch(error => {
          console.error('Failed to load required scripts:', error);
        });
      }, 100);
    });

    const lazyLoader = {
      observer: null,
      images: [],
      sections: [],
      rafId: null,

      init() {
        if ('IntersectionObserver' in window) {
          this.observer = new IntersectionObserver(this.handleIntersect.bind(this), {
            root: null,
            rootMargin: '00px',
            threshold: 0.01
          });
        }
      },

      handleIntersect(entries) {
        entries.forEach(entry => {
          if (entry.isIntersecting) {
            const target = entry.target;

            if (target.classList.contains('lazy-image')) {
              this.loadImage(target);
            }
            else if (target.dataset.lazySection) {
              this.loadSection(target);
            }

            if (target.classList.contains('lazy-image')) {
              this.observer.unobserve(target);
            }
          }
        });
      },

      loadImage(img) {
        const src = img.getAttribute('data-src');
        if (src) {
          requestAnimationFrame(() => {
            const newImg = new Image();
            newImg.onload = () => {
              img.setAttribute('src', src);
              img.classList.add('loaded', 'fade-in');
              const indicator = img.nextElementSibling;
              if (indicator && indicator.classList.contains('lazy-loading-indicator')) {
                indicator.style.display = 'none';
              }
              if (img.dataset.gameId) {
                window.lazyLoadCallbacks?.forEach(callback => callback(img.dataset.gameId));
              }
            };
            newImg.src = src;
          });
        }
      },

      loadSection(section) {
        requestAnimationFrame(() => {
          section.classList.add('loaded');
          const images = section.querySelectorAll('.lazy-image');
          images.forEach(img => this.observer.observe(img));
        });
      },

      observeImages(images) {
        if (this.observer) {
          images.forEach(img => this.observer.observe(img));
        } else {
          images.forEach(img => this.loadImage(img));
        }
      },

      observeSections(sections) {
        if (this.observer) {
          sections.forEach(section => this.observer.observe(section));
        } else {
          sections.forEach(section => this.loadSection(section));
        }
      },

      throttleScroll(callback) {
        if (!this.rafId) {
          this.rafId = requestAnimationFrame(() => {
            callback();
            this.rafId = null;
          });
        }
      }
    };

    lazyLoader.init();

    function initApp() {
      const { createApp, ref, onMounted, onUnmounted, nextTick, computed } = Vue;

      createApp({
        setup() {
          const list = ref([]);
          const allGames = ref([]);

          const currentSlide = ref(0);
          const track = ref(null);
          const videos = ref([]);
          const indicators = ref([]);
          const videoItems = ref([]);
          
          // 搜索侧边栏状态
          const searchSidebarOpen = ref(false);
          // 搜索相关变量
          const searchQuery = ref('');
          const searchResults = ref([]);
          const showSearchResults = ref(false);
          
          // 打开搜索侧边栏
          const openSearchSidebar = () => {
            searchSidebarOpen.value = true;
            document.body.style.overflow = 'hidden';
            // 打开搜索侧边栏时聚焦到搜索框
            nextTick(() => {
              const searchInput = document.querySelector('.search-header input');
              if (searchInput) {
                searchInput.focus();
              }
            });
          };
          
          // 关闭搜索侧边栏
          const closeSearchSidebar = () => {
            searchSidebarOpen.value = false;
            document.body.style.overflow = '';
            // 关闭时清空搜索
            searchQuery.value = '';
            searchResults.value = [];
            showSearchResults.value = false;
          };
          
          // 执行搜索
          const performSearch = () => {
            if (!searchQuery.value.trim()) {
              searchResults.value = [];
              showSearchResults.value = false;
              return;
            }
            
            const query = searchQuery.value.toLowerCase().trim();
            // 在游戏数据中搜索标题、描述或标签
            searchResults.value = allGames.value.filter(game => {
              const title = (game.short_title || '').toLowerCase();
              const description = (game.description || '').toLowerCase();
              const tags = (game.tags || '').toLowerCase();
              const category = (game.category || '').toLowerCase();
              
              return title.includes(query) || 
                     description.includes(query) || 
                     tags.includes(query) || 
                     category.includes(query);
            });
            
            showSearchResults.value = true;
          };
          
          // 处理搜索输入变化
          const handleSearchInput = () => {
            performSearch();
          };
          
          // 处理搜索按钮点击
          const handleSearchClick = () => {
            performSearch();
          };
          
          // 处理回车键搜索
          const handleKeyPress = (event) => {
            if (event.key === 'Enter') {
              performSearch();
            }
          };


          window.lazyLoadCallbacks = [];

          let shuffledGames = null;

          const getData = () => {
            // 只在首次或数据为空时进行随机排序
            if (!shuffledGames || shuffledGames.length === 0) {
              // 使用更高效的Fisher-Yates洗牌算法
              shuffledGames = [...filter_data];
              for (let i = shuffledGames.length - 1; i > 0; i--) {
                const j = Math.floor(Math.random() * (i + 1));
                [shuffledGames[i], shuffledGames[j]] = [shuffledGames[j], shuffledGames[i]];
              }
            }

            // 初始化数据时添加加载状态标记
            allGames.value = shuffledGames.map(game => ({
              ...game,
              loaded: false
            }));
            list.value = [...allGames.value];
          };

          const addImageLoadCallback = (callback) => {
            window.addEventListener('lazyImageLoaded', (e) => {
              if (e.detail && e.detail.gameId) {
                callback(e.detail.gameId);
              }
            });
          };

          const handleImageLoaded = (gameId) => {
            // 使用requestAnimationFrame延迟DOM更新
            requestAnimationFrame(() => {
              const gameIndex = allGames.value.findIndex(g => g.id === gameId);
              if (gameIndex !== -1) {
                allGames.value[gameIndex].loaded = true;
              }
            });
          };

          const newGames = computed(() => {
            return [...allGames.value].sort(() => Math.random() - 0.5).slice(0, 16);
          });

          const popularGames = computed(() => {
            const games = [...allGames.value].sort(() => Math.random() - 0.5).slice(0, 16);
            return games.map(game => ({
              ...game,
              playCount: (Math.floor(Math.random() * 20) + 1) + 'k'
            })).sort((a, b) => {
              const numA = parseInt(a.playCount);
              const numB = parseInt(b.playCount);
              return numB - numA;
            });
          });

          const puzzlesGames = computed(() => {
            return [...allGames.value].filter(game => game.category === 'Puzzles').slice(0, 16);
          });

          const girlsGames = computed(() => {
            return [...allGames.value].filter(game => game.category === 'Girls').slice(0, 16);
          });

          const arcadeGames = computed(() => {
            return [...allGames.value].filter(game => game.category === 'Arcade').slice(0, 16);
          });
          const hypercasualGames = computed(() => {
            return [...allGames.value].filter(game => game.category === 'Hypercasual').slice(0, 16);
          });

          const categories = computed(() => {
            // 用于跟踪已处理过的分类
            const seenCategories = new Set();
            // 返回每个分类的第一个游戏对象
            return allGames.value.filter(game => {
              // 如果这个分类还没处理过
              if (!seenCategories.has(game.category)) {
                // 将该分类添加到已处理集合
                seenCategories.add(game.category);
                // 返回这个游戏对象
                return true;
              }
              return false;
            });
          });
          const getTagsList = (tags) => {
            if (!tags) return [];
            if (Array.isArray(tags)) return tags;
            if (typeof tags === 'string') return tags.split(/[,，]/).map(tag => tag.trim()).filter(tag => tag);
            return [];
          };

          const navigateToDetail = (gameId, shortTitle = '') => {
            const encodedTitle = encodeURIComponent(shortTitle);
            window.location.href = `./detail.html?id=${gameId}&title=${encodedTitle}`;
          };


          // 固定的推荐游戏列表，不再随机变化
          const recommendedGames = computed(() => {
            if (!allGames.value || allGames.value.length === 0) return [];
            // 直接返回前6个游戏，不再随机排序
            return allGames.value.slice(0, 6);
          });

          // 移动端菜单开关状态
          const menuOpen = ref(false);

          // 轮播图方法 - Vue实现
          const startVideos = () => {
            videos.value.forEach((video, index) => {
              // 设置为静音以提高自动播放成功率
              video.muted = true;
              video.autoplay = true;
              video.loop = true;

              // 尝试播放视频
              const playPromise = video.play();

              if (playPromise !== undefined) {
                playPromise.then(() => {
                }).catch(e => {
                  document.addEventListener('click', function onFirstClick() {
                    video.play().catch(err => {
                    });
                    document.removeEventListener('click', onFirstClick);
                  });
                });
              }
            });
          };

          const getItemWidth = () => {
            if (window.innerWidth <= 768) {
              return 280; // 移动端固定宽度
            }
            if (videoItems.value.length > 0) {
              return videoItems.value[0].offsetWidth + 6; // 包含左右margin
            }
            return 300; // 默认值
          };

          const updateCarousel = (index) => {
            if (!track.value || videoItems.value.length === 0 || indicators.value.length === 0) return;

            // 确保索引在有效范围内
            const validIndex = Math.max(0, Math.min(index, indicators.value.length - 1));
            currentSlide.value = validIndex;

            // 计算滚动位置
            const itemWidth = getItemWidth();
            const containerWidth = track.value.parentElement.offsetWidth;
            const maxScrollableIndex = Math.max(0, videoItems.value.length - Math.floor(containerWidth / itemWidth));
            const targetIndex = Math.min(validIndex, maxScrollableIndex);
            const scrollPosition = targetIndex * itemWidth;

            // 使用scrollTo方法实现平滑滚动
            track.value.scrollTo({
              left: scrollPosition,
              behavior: 'smooth'
            });

            // 更新指示器状态 - 确保只有一个激活
            updateIndicatorStates(validIndex);
          };

          // 统一的指示器状态更新函数
          const updateIndicatorStates = (activeIndex) => {
            if (indicators.value.length === 0) return;

            // 先重置所有指示器
            indicators.value.forEach(indicator => {
              indicator.classList.remove('active', 'bg-white');
              indicator.classList.add('bg-white/50');
            });

            // 只激活指定的指示器
            if (activeIndex >= 0 && activeIndex < indicators.value.length) {
              indicators.value[activeIndex].classList.add('active', 'bg-white');
              indicators.value[activeIndex].classList.remove('bg-white/50');
            }
          };

          const handleIndicatorClick = (index) => {
            if (!isNaN(index)) {
              updateCarousel(index);
            }
          };

          // 处理拖动时更新指示器
          const handleScroll = () => {
            if (!track.value || videoItems.value.length === 0 || indicators.value.length === 0) return;

            const itemWidth = getItemWidth();
            const scrollPosition = track.value.scrollLeft;

            // 计算当前显示的主要幻灯片索引
            const currentIndex = Math.round(scrollPosition / itemWidth);

            // 确保索引在有效范围内
            const validIndex = Math.max(0, Math.min(currentIndex, indicators.value.length - 1));

            // 更新指示器状态，但不触发滚动（避免循环更新）
            if (validIndex !== currentSlide.value) {
              currentSlide.value = validIndex;
              // 使用统一的指示器更新函数
              updateIndicatorStates(validIndex);
            }
          };

          const initCarousel = () => {
            // 获取DOM元素
            track.value = document.getElementById('videoCarouselTrack');
            videos.value = Array.from(document.querySelectorAll('.video-item video'));
            indicators.value = Array.from(document.querySelectorAll('#carouselIndicators div'));
            videoItems.value = Array.from(document.querySelectorAll('.video-item'));

            // 启动视频播放
            startVideos();

            // 初始化轮播状态
            updateCarousel(0);

            // 为指示器添加点击事件
            indicators.value.forEach((indicator, index) => {
              indicator.addEventListener('click', () => handleIndicatorClick(index));
            });

            // 添加滚动事件监听，用于拖动时更新指示器
            if (track.value) {
              track.value.addEventListener('scroll', handleScroll);
            }

            // 响应式处理
            window.addEventListener('resize', () => updateCarousel(currentSlide.value));
          };

          // ESC键事件处理函数
          const handleEscKey = (event) => {
            if (event.key === 'Escape' && searchSidebarOpen.value) {
              closeSearchSidebar();
            }
          };
          
          onMounted(() => {
            getData();
            // 添加ESC键事件监听器
            document.addEventListener('keydown', handleEscKey);
            
            nextTick(() => {
              // 注册图片加载回调
              addImageLoadCallback(handleImageLoaded);

              requestAnimationFrame(() => {
                const lazyImages = document.querySelectorAll('.lazy-image');
                lazyLoader.observeImages(lazyImages);

                const gameSections = [
                  document.getElementById('funny-games'),
                  document.getElementById('amazing-games'),
                  document.getElementById('online-games'),
                  document.getElementById('recommended')
                ];

                gameSections.forEach(section => {
                  if (section) {
                    section.dataset.lazySection = 'true';
                    lazyLoader.observeSections([section]);
                  }
                });

                // 初始化Vue轮播图
                if (document.getElementById('videoCarouselTrack')) {
                  initCarousel();
                }
              });
            });
          });

          onUnmounted(() => {
            clearInterval(autoPlayTimer);
            // 移除ESC键事件监听器
            document.removeEventListener('keydown', handleEscKey);
          });

          return {
            list,
            getTagsList,
            currentSlide,
            handleIndicatorClick,
            newGames,
            puzzlesGames,
            popularGames,
            girlsGames,
            arcadeGames,
            hypercasualGames,
            categories,
            recommendedGames,
            menuOpen,
            addImageLoadCallback,
            handleImageLoaded,
            // 搜索侧边栏相关
            searchSidebarOpen,
            openSearchSidebar,
            closeSearchSidebar,
            // 搜索功能相关
            searchQuery,
            searchResults,
            showSearchResults,
            handleSearchInput,
            handleSearchClick,
            handleKeyPress
          };
        },
      }).mount("#app");
    }
