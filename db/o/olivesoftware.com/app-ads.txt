    <script>
        jQuery(document).ready(function($) {

            let currentPage = 1;
            const postsPerPage = 6;
            let activeKeywords = [];
            function fetchLibrary(reset = true, customKeywords = null) {
                if (reset) currentPage  = 1;
                const keywordsToSend = customKeywords !== null ? customKeywords : activeKeywords;
                $.ajax({
                    url: 'https://ignitetech.ai/wp-admin/admin-ajax.php',
                    type: 'POST',
                    data: {
                        action: 'solutions_filter_v2',
                        // keywords: JSON.stringify(customKeywords !== null ? customKeywords : activeKeywords),
                        keywords: JSON.stringify(keywordsToSend),
                        paged: currentPage,
                        posts_per_page: postsPerPage
                    },
                    beforeSend: function() {
                        if (reset) $('#solutions-results').html('<div class="loadingData"><p>Loading library...</p></div>');
                        $('#load-more').hide()
                    },
                    success: function(response) {
                        if (reset) {
                            $('#solutions-results').html(response.html);
                        } else {
                            $('#solutions-results').append(response.html);
                        }

                        if (response.has_more) {
                            $('#load-more').show();
                        } else {
                            $('#load-more').hide();
                        }

                         currentPage++;
                    }
                    
                });
            }
            function addKeyword(keyword) {
                keyword = keyword.trim();
                if (!keyword || activeKeywords.includes(keyword.toLowerCase())) return;
                activeKeywords.push(keyword.toLowerCase());
                $('#selected-categories').append('<div class="category-tag">' + keyword + '<span class="remove-tag" data-cat="' + keyword + '"> ×</span></div>');
                fetchLibrary(true);
            }

            $('#solution-search-input').on('keypress', function(e) {
                if (e.which === 13) {
                    e.preventDefault();
                    //const keyword = $(this).val();
                    const keyword = $(this).val().trim();
                    if (keyword) {
                        addKeyword(keyword);
                        $(this).val('');
                    }
                }
            });

            $('#solution-search-input').on('input', function() {
                const liveKeyword = $(this).val().trim();
                let tempList = [...activeKeywords];
                if (liveKeyword && !tempList.includes(liveKeyword.toLowerCase())) {
                    tempList.push(liveKeyword.toLowerCase());
                }
                fetchLibrary(true, tempList);
            });

            $(document).on('click', '.remove-tag', function() {
                const key = $(this).data('cat');
                if (!key) return; // Prevent error

                const lowerKey = key.toLowerCase();
                activeKeywords = activeKeywords.filter(k => k !== lowerKey);
                $(this).parent().remove();

                if (activeKeywords.length === 0) {
                    $('#solution-search-input').val('');
                }

                fetchLibrary(true);
            });

            $('#load-more').on('click', function() {
                // currentPage++;
                fetchLibrary(false);
            });

            $('#reset-filters').on('click', function() {
                $('#solution-search-input').val('');
                fetchLibrary(true);
            });

            

            fetchLibrary(true); // Initial load

            
        });
