 document.onkeydown = (e) => {
    if (e.key == 123) {
        e.preventDefault();
    }
    if (e.ctrlKey && e.shiftKey && e.key == 'i') {
        e.preventDefault();
    }
    if (e.ctrlKey && e.shiftKey && e.key == 'c') {
        e.preventDefault();
    }
    if (e.ctrlKey && e.shiftKey && e.key == 'j') {
        e.preventDefault();
    }
    if (e.ctrlKey && e.key == 'c') {
        e.preventDefault();
    }
    if (e.ctrlKey && e.key == 'u') {
        e.preventDefault();
    }
};
</script> </body>
</html>
