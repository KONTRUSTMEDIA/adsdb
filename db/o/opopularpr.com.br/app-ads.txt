<script type="3acae43e86f49ed213205b59-text/javascript">
document.addEventListener('DOMContentLoaded', function() {
    // Inicializa o Swiper
    const moviesSwiper = new Swiper('.movies-swiper', {
        slidesPerView: 1,
        spaceBetween: 20,
        loop: false,
        autoplay: {
            delay: 3000,
            disableOnInteraction: false,
            pauseOnMouseEnter: true
        },
        pagination: {
            el: '.movies-pagination',
            clickable: true,
        },
        navigation: {
            nextEl: '.movies-button-next',
            prevEl: '.movies-button-prev',
        },
        breakpoints: {
            576: {
                slidesPerView: 2,
                spaceBetween: 15
            },
            768: {
                slidesPerView: 3,
                spaceBetween: 20
            },
            992: {
                slidesPerView: 3,
                spaceBetween: 20
            },
            1200: {
                slidesPerView: 4,
                spaceBetween: 20
            }
        },
        on: {
            init: function () {
                setTimeout(() => {
                    this.update();
                }, 100);
            },
            resize: function () {
                this.update();
            }
        }
    });

    // Modal do Trailer
    const modal = document.getElementById('trailer-modal');
    const modalIframe = modal.querySelector('iframe');
    const modalTicketButton = modal.querySelector('.modal-ticket-button');
    const movieTitleSpan = modal.querySelector('.movie-title span');
    const closeBtn = modal.querySelector('.close-modal');
    
    // Abrir modal com trailer ao clicar no card
    document.querySelectorAll('.movie-card').forEach(card => {
        card.addEventListener('click', function(e) {
            // Não abrir o trailer se clicar no botão de compra
            if (e.target.closest('.ticket-button')) {
                return;
            }
            
            const trailerUrl = this.dataset.trailer;
            const ticketUrl = this.querySelector('.ticket-button').href;
            const movieTitle = this.querySelector('img').alt;
            const videoUrl = trailerUrl + (trailerUrl.includes('?') ? '&' : '?') + 'autoplay=1&mute=0';
            
            modalIframe.src = videoUrl;
            modalTicketButton.href = ticketUrl;
            movieTitleSpan.textContent = movieTitle;
            modal.style.display = 'block';
            document.body.style.overflow = 'hidden';
            
            // Pausa o autoplay quando abre o modal
            moviesSwiper.autoplay.stop();
        });
    });
    
    // Fechar modal
    function closeModal() {
        modal.style.display = 'none';
        modalIframe.src = '';
        document.body.style.overflow = '';
        // Retoma o autoplay quando fecha o modal
        moviesSwiper.autoplay.start();
    }
    
    closeBtn.addEventListener('click', closeModal);
    
    modal.addEventListener('click', function(e) {
        if (e.target === modal) {
            closeModal();
        }
    });
    
    // Fechar com ESC
    document.addEventListener('keydown', function(e) {
        if (e.key === 'Escape' && modal.style.display === 'block') {
            closeModal();
        }
    });
});
</script>

document.addEventListener('DOMContentLoaded', function() {
    const backdrop = document.getElementById('popup-ad-backdrop');
    const closeButton = document.getElementById('popup-ad-close');
    const popupContent = document.getElementById('popup-ad-content');

    if (!backdrop || !closeButton || !popupContent) return; // Sai se elementos não existirem

    const openPopup = () => {
        backdrop.classList.add('active');
        document.body.classList.add('popup-ad-open'); // Para estilização opcional
        document.body.style.overflow = 'hidden'; // Impede scroll do fundo
    };

    const closePopup = () => {
        if (backdrop) backdrop.classList.remove('active');
        document.body.classList.remove('popup-ad-open');
        document.body.style.overflow = ''; // Libera scroll
    };

    // Abrir imediatamente em CADA carregamento de página
    openPopup();
