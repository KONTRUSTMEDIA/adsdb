    function setData(data) {
        if (data && data.length) {
            // Remapping object
            data = data.map((d) => {
                if (!d) return d
                const today = new Date(d.date)
                const todayDate = today.getDate().toString().padStart(2, '0');
                const todayMonth = (today.getMonth() + 1).toString().padStart(2, '0');
                const todayYear = today.getFullYear();
                const todayDay = days[today.getDay()];

                d.oriDate = d.date
                d.date = `${todayDate}-${todayMonth}-${todayYear} (${todayDay})`;
                d.draw_no = d.draw_lap ? `${d.draw_no}/${d.draw_lap}` : d.draw_no;

                Object.keys(d).forEach((key) => {
                    if (key.includes('jackpot_prize') && d[key]) {
                        let currency = "RM"
                        if (d.category_id === 17) currency = "USD" // Added mylotto
                        d[key] = `${currency} ${numberWithCommas(d[key])}`
                    }
                })

                if (d.numbers) {
                    const object = d.numbers.reduce((n, item) => ({
                        ...n,
                        [`${item.type}-${item.position}`]: item.number
                    }), {})

                    return {
                        ...d,
                        ...object
                    }
                }

                // Added mylotto
                if (d.category_id === 17 && !d.six_number)
                    d.six_number = "      ";

                if (d.six_number) {
                    const sixNo = d.six_number.split("") || []
                    const mylotto6d = sixNo.reduce((n, no, index) => ({
                        ...n,
                        [`d6_first_number_${index+1}`]: no
                    }), {})

                    return {
                        ...d,
                        ...mylotto6d
                    }
                }

                return d
            })

            // Updating data
            data.forEach((d) => {
                if (!d) return // check if data is returned

                Object.keys(d).forEach((key) => {
                    let table = document.querySelector(`#table-${d.category_id}-${d.oriDate}`);
                    // only those main page on left sidebar
                    if (!table) table = document.querySelector(`.table-${d.category_id}`);
                    if (table) {
                        if (key === "jackpot_number" && d.category_id === 17) {
                            const divs = table.querySelectorAll('.number');
                            const div = [...divs].find(e => e.innerText === d[key])
                            if (div) div.classList.add("jackpot_indicator");
                        } else {
                            // querySelectorAll because of "Magnum 4D jackpot gold"
                            const elements = table.querySelectorAll(`[data-id=${key}]`)
                            if (elements.length) {
                                elements.forEach((element) => {
                                    if (key === 'zodiac') element.src = `${templateUrl}/assets${d[key]}`;
                                    else element.innerHTML = d[key] || "&nbsp;";
                                })
                            }
                        }
                    }

                })
            })
        }
    }

    async function fetchResult() {
        try {
            showModal()
            let host = window.location.host
            host = host.includes('localhost') ? '4dresultco.com' : host;
            const url = `https://api.${host}/api/result?date=${new Date().getTime()}`
            const response = await fetch(url)
            const data = await response.json()
            setData(data)
        } catch (err) {
            console.log("Error happened live result --> ", err)
        } finally {
            setTimeout(() => hideModal(), 500)
        }
    }
