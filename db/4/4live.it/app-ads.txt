	function decodeHTMLEntities(text) {
		const entities = {
			'&#8217;': "'",
			'&#8220;': '"',
			'&#8221;': '"',
			'&#8211;': '–',
			'&#8212;': '—',
			'&#8230;': '…',
			'&amp;': '&',

			'&gt;': '>',
			'&quot;': '"',
			'&apos;': "'",
			'&#39;': "'",
			'&#34;': '"'
		}

		return text.replace(/&#?\w+;/g, function (match) {
			return entities[match] || match
		})
	}
