<script type="f887b9f339f09517d4da0ce0-text/javascript">
    const data = [
      "Bangalore",
      "Hyderabad",
      "Mumbai",
      "Delhi",
      "Chennai",
      "Trivandrum",
      "Kochi",
      "Kozhikode",
      "Dubai",
      "Abu Dhabi",
      "Pune",
      "Mangalore",
    ];

    $(document).ready(function () {
      const $searchInput = $("#searchInput");
      const $dropdownMenu = $("#dropdownMenu");
      const $noResultsButton = $("#noResultsButton");

      // Populate dropdown dynamically
      function populateDropdown(items) {
        $dropdownMenu.empty(); // Clear existing items
        if (items.length > 0) {
          items.forEach((item) => {
            $dropdownMenu.append(
              `<li><button class="dropdown-item">${item}</button></li>`
            );
          });
          $noResultsButton.hide(); // Hide "No Results" button
          $dropdownMenu.show(); // Show the dropdown
        } else {
          $dropdownMenu.hide(); // Hide dropdown if no items
          $noResultsButton.show(); // Show "No Results" button
        }
      }

      // Filter dropdown items based on input
      $searchInput.on("input", function () {
        const query = $(this).val().toLowerCase();
        const filteredData = data.filter((item) =>
          item.toLowerCase().includes(query)
        );
        populateDropdown(filteredData);
      });

      // Handle item selection
      $dropdownMenu.on("click", ".dropdown-item", function () {
        const selectedItem = $(this).text();
        $searchInput.val(selectedItem); // Set input value
        $dropdownMenu.hide(); // Hide dropdown
        $noResultsButton.hide(); // Hide "No Results" button
      });
    });
  </script>
