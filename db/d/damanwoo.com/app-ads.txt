    const initTheme = () => {
        const theme = (() => {
            if (
                typeof localStorage !== "undefined" &&
                localStorage.getItem("theme")
            ) {
                return localStorage.getItem("theme");
            }
            if (window.matchMedia("(prefers-color-scheme: dark)").matches) {
                return "dark";
            }
            return "light";
        })();

        if (theme === "dark") {
            document.documentElement.classList.add("dark");
        } else {
            document.documentElement.classList.remove("dark");
        }

        // Re-bind toggle button every time page loads/swaps
        const themeToggle = document.getElementById("themeToggle");
        if (themeToggle && !themeToggle.hasAttribute("data-theme-bound")) {
            themeToggle.setAttribute("data-theme-bound", "true");
            themeToggle.addEventListener("click", () => {
                document.documentElement.classList.toggle("dark");
                const isDark =
                    document.documentElement.classList.contains("dark");
                localStorage.setItem("theme", isDark ? "dark" : "light");
            });
        }
    };

    // Run on initial load
    initTheme();

    // Run after every View Transition swap
    document.addEventListener("astro:after-swap", initTheme);
    // Explicitly for astro:page-load if script is cached
    document.addEventListener("astro:page-load", initTheme);
