    function debounce(fn, wait, immediate) {
        var timeout;
        return function () {
            var _this = this;
            var args = arguments;
            if (timeout) clearTimeout(timeout);
            if (immediate) {
                var callNow = !timeout;
                timeout = setTimeout(function () {
                    timeout = null;
                }, wait);
                if (callNow) {
                    fn.apply(_this, args);
                }
            } else {
                timeout = setTimeout(function () {
                    fn.apply(_this, args);
                }, wait);
            }
        };
    }

    function initCalculatorArry() {
        var sku ="530300";
        var qty =$("#inputQuantity").val();
        var countryId = $("#countryN .country_tit").attr("data-countryid");
        var countrycode = $("#countryN .country_tit").attr("data-countrycode");
        var countryName = $("#countryN .country_tit").attr("data-countryname");
        var url = calculatorURL + "?sku=" + sku + "&qty=" + qty + "&countryId=" + countryId + "&countryCode=" + countrycode;
        calculatorData =calculatorData ||{};
        if(calculatorData.calculatorArry  && calculatorData.Qty== qty && calculatorData.CountryId == countryId){
            setCalculatorHtml(calculatorData.calculatorArry);
        }else{
            calculatorData.Qty= qty;
            calculatorData.CountryId= countryId;

            $.ajax({
                url: url,
                type: "GET",
                dataType: "jsonp",
                beforeSend: function () {
                    $("#loadingShipping").addClass("loading");
                },
                success: function (data) {
                    if (data.isForbidSale) {
                        $("#loadingShipping").html("<b class='free_icon'> Cannot ship </b> to <span id='countryTit' class='country_tit' rel='nofollow'>" + countryName + "</span>");
                        if ($("#calculatorShip").find("dd").length > 0) {
                            $("#calculatorShip").find("dd").remove();
                        }
                    } else {
                        $("#loadingShipping").html("<b class='free_icon'></b> to <span id='countryTit' class='country_tit' rel='nofollow'>United States via UPS  Expedited</span><span class='shipping_box'> <b class='busdays'></b></span>");
                        if (data) {
                            calculatorData.calculatorArry = data.calculatorArry;
                            setCalculatorHtml(calculatorData.calculatorArry);

                            //ReSet Ship From and Options
                            var isHasShip = (calculatorData.calculatorArry && calculatorData.calculatorArry.length > 0) ? true : false;
                            setShipFromHtml(data.shipFromWareHouses, data.optionsProducts, sku, data.saleState, data.isOutOfStock,
                                data.isShowPresale, data.isShip, data.collectionNumber, isHasShip);

                        } else {
                            setCalculatorHtml(null);
                        }

                    }



                },
                complete: function () {
                    changeShipping();
                    $("#loadingShipping").removeClass("loading");
                },
                error: function (data) {
                    console.info("error: " + data.responseText);
                }
            });
        }
    };

    function getBlogPageName(url) {
        try {
            const u = new URL(url);
            const path = u.pathname.replace(/^\/|\/$/g, '');
            const parts = path.split('/');
            //判断首页
            if (parts.length === 0 || (parts.length === 1 && parts[0] === 'geekgiz')) {
                return '导购主页';
            }
            for (let i = parts.length - 1; i >= 0; i--) {
                if (parts[i]) {
                    return parts[i];
                }
            }
            return '导购主页';
        } catch (e) {
            return '导购主页';
        }
    }
    //是否博客首页
    function isBlogHome(url) {
        try {
            const u = new URL(url);
            const path = u.pathname.replace(/\/+$/, '');
            if (path === '/geekgiz' || path === '/geekgiz/page') {
                return true;
            }
            //匹配page带分页的 例如page/2等
            const match = path.match(/^\/geekgiz\/page\/\d+$/);
            if (match) {
                return true;
            }

            return false;
        } catch {
            return false;
        }
    }

    function decodeHtmlEntities(str, objText) {
        const textarea = document.createElement('textarea');
        textarea.innerHTML = str;
        $(objText).text(textarea.value);
    }
    function translationsText(reviewId, languageTarget, objText, revewTranslate) {
            $.ajax({
                url: "/ProductReview/Translate?reviewId="+ reviewId +"&languageCode=" + languageTarget,
                type: "get",
                beforeSend: function () {
                    $(revewTranslate).text("Translating...").addClass("active").show();
                    $(objText).parents(".revew_translate_content").hide();
                },
                success: function (data) {
                    if (data.Success && data.Data) {
                        var translateData = JSON.parse(data.Data);
                        decodeHtmlEntities(translateData.data.translations[0].translatedText, objText);
                        $(objText).parents(".revew_translate_content").show();
                        $(revewTranslate).text("Translate").removeClass("active").hide();
                    } else {
                        $(revewTranslate).text("Translate").removeClass("active");
                        $.curContent.gkAlert("Sorry, Operation failed. Please try again after refresh.", false, "340px", "40px");
                    }
                }
            });

        }

    function translateLang(event,obj){
        event.stopPropagation();
