// Pricing Tab Switcher
function showPricingTab(tab, btn) {
    document.getElementById('pricing-tab-orgs').style.display = 'none';
    document.getElementById('pricing-tab-schools').style.display = 'none';
    document.getElementById('pricing-tab-coaches').style.display = 'none';
    document.getElementById('pricing-tab-' + tab).style.display = 'block';
    document.querySelectorAll('#pricingTabs button').forEach(function(b) {
        b.classList.remove('btn-primary', 'active');
        b.classList.add('btn-outline-primary');
    });
    btn.classList.remove('btn-outline-primary');
    btn.classList.add('btn-primary', 'active');

// Organization Search
document.addEventListener('DOMContentLoaded', function() {
    const searchInput = document.getElementById('orgSearchInput');
    const resultsDiv = document.getElementById('orgSearchResults');
    let searchTimeout;

    searchInput.addEventListener('input', function() {
        clearTimeout(searchTimeout);
        const query = this.value.trim();

        if (query.length < 2) {
            resultsDiv.innerHTML = '';
            return;
        }

        searchTimeout = setTimeout(() => {
            fetch(`api/searchOrgs.api.php?q=${encodeURIComponent(query)}`)
                .then(response => response.json())
                .then(data => {
                    if (data.success && data.orgs.length > 0) {
                        resultsDiv.innerHTML = '<div class="list-group">' +
                            data.orgs.map(org =>
                                `<a href="${org.publicslug}" class="list-group-item list-group-item-action">
                                    <div class="d-flex w-100 justify-content-between align-items-center">
                                        <h6 class="mb-0">${org.orgname}</h6>
                                        <i class="bi bi-arrow-right"></i>
                                    </div>
                                </a>`
                            ).join('') +
                            '</div>';
                    } else {
                        resultsDiv.innerHTML = '<div class="alert alert-info mb-0"><i class="bi bi-info-circle me-2"></i>No organizations found. Try a different search term.</div>';
                    }
                })
                .catch(error => {
                    resultsDiv.innerHTML = '<div class="alert alert-danger mb-0"><i class="bi bi-exclamation-triangle me-2"></i>Error searching. Please try again.</div>';
                });
        }, 300);
    });

    // Clear search when modal is closed
    document.getElementById('orgFinderModal').addEventListener('hidden.bs.modal', function() {
        searchInput.value = '';
        resultsDiv.innerHTML = '';
    });
});

        <script>
// KoachSynk Brand Colors
const KS_COLORS = {
    primary: '#0a6aa6',    // Blue - normal confirmations
    danger: '#8A0514',     // Red - destructive actions
    secondary: '#6c757d',  // Gray - cancel buttons
    success: '#198754',    // Green - success toasts
    warning: '#FCC201'     // Yellow - warning icon
};

/**
 * Show a confirmation modal with blue confirm button (for normal actions)
 * @param {Object} options - Modal options
 * @param {string} options.title - Modal title
 * @param {string} options.text - Modal message (plain text)
 * @param {string} options.html - Modal message (HTML, use instead of text)
 * @param {string} options.confirmText - Confirm button text (default: 'Confirm')
 * @param {string} options.cancelText - Cancel button text (default: 'Cancel')
 * @param {string} options.icon - Icon type: 'warning', 'question', 'info' (default: 'warning')
 * @returns {Promise<boolean>} - True if confirmed, false if cancelled
 */
async function ksConfirm(options = {}) {
    const result = await Swal.fire({
        title: options.title || 'Confirm',
        text: options.text || undefined,
        html: options.html || undefined,
        icon: options.icon || 'warning',
        showCancelButton: true,
        confirmButtonColor: KS_COLORS.primary,
        cancelButtonColor: KS_COLORS.secondary,
        confirmButtonText: options.confirmText || 'Confirm',
        cancelButtonText: options.cancelText || 'Cancel',
        reverseButtons: false
    });
    return result.isConfirmed;
}

/**
 * Show a confirmation modal with red confirm button (for destructive actions)
 * @param {Object} options - Same options as ksConfirm
 * @returns {Promise<boolean>} - True if confirmed, false if cancelled
 */
async function ksConfirmDanger(options = {}) {
    const result = await Swal.fire({
        title: options.title || 'Confirm',
        text: options.text || undefined,
        html: options.html || undefined,
        icon: options.icon || 'warning',
        showCancelButton: true,
        confirmButtonColor: KS_COLORS.danger,
        cancelButtonColor: KS_COLORS.secondary,
        confirmButtonText: options.confirmText || 'Confirm',
        cancelButtonText: options.cancelText || 'Cancel',
        reverseButtons: false
    });
    return result.isConfirmed;
}

/**
 * Show a toast notification (auto-dismisses)
 * @param {string} message - The message to display
 * @param {string} type - Toast type: 'success', 'error', 'warning', 'info' (default: 'success')
 * @param {number} duration - Duration in milliseconds (default: 3000)
 */
function ksToast(message, type = 'success', duration = 3000) {
    Swal.fire({
        toast: true,
        position: 'top-end',
        icon: type,
        title: message,
        showConfirmButton: false,
        timer: duration,
        timerProgressBar: true
    });
}

/**
 * Show an alert modal (informational, single OK button)
 * @param {Object} options - Modal options
 * @param {string} options.title - Modal title
 * @param {string} options.text - Modal message
 * @param {string} options.icon - Icon type: 'success', 'error', 'warning', 'info'
 */
async function ksAlert(options = {}) {
    await Swal.fire({
        title: options.title || 'Notice',
        text: options.text || undefined,
        html: options.html || undefined,
        icon: options.icon || 'info',
        confirmButtonColor: KS_COLORS.primary,
        confirmButtonText: options.confirmText || 'OK'
    });
}

        <script>
        document.addEventListener('DOMContentLoaded', function() {
            // Check for pending email change requests that need attention
            const notificationBadge = document.getElementById('email-change-notification-badge');
            const dropdownBadge = document.getElementById('email-change-dropdown-badge');

            if (notificationBadge && dropdownBadge) {
                fetch('https://koachsynk.com/api/getPendingEmailChanges.api.php')
                    .then(res => res.json())
                    .then(data => {
                        if (data.success && data.actionCounts) {
                            const needsApproval = data.actionCounts.needsApproval || 0;
                            const needsVerification = data.actionCounts.needsVerification || 0;
                            const totalPending = needsApproval + needsVerification;

                            if (needsApproval > 0) {
                                // Show notification badge for approvals needed
                                notificationBadge.textContent = needsApproval;
                                notificationBadge.style.display = 'inline-block';
                            }

                            if (totalPending > 0) {
                                // Show dropdown badge for any pending action
                                dropdownBadge.textContent = totalPending;
                                dropdownBadge.style.display = 'inline-block';
                            }
                        }
                    })
                    .catch(err => {
                        console.error('Error checking email change notifications:', err);
                    });
            }
        });
        </script>
