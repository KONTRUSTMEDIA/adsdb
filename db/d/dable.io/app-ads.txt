    <script>
    function readProp(msg, prop) {
      return msg.replace(
        new RegExp("^.*" + prop + "=([^\&]+).*$"),
        '$1'
      );
    }
    simplePostMessage.receiveMessage( function (e) {
      if (e.data && typeof e.data == "string") {
        var scrollHeight = document.body.scrollHeight;
        var received_data = decodeURIComponent(e.data);
        var fid = readProp(received_data, 'id');
        if (!fid) return;

        // adjust height message
        if (received_data.indexOf('if_height=') > -1) {
          h = Number(readProp(received_data, 'if_height'));
          if (!isNaN(h) && h > 0) {
            document.getElementById(fid)
                    .getElementsByTagName("iframe")[0].height = h;
          }
          sendHeight(scrollHeight + h);
          sendFormHeight(h);
        } else if (received_data.indexOf('force_redraw=1') > -1) {
          // 카카오톡 인앱 브라우저 이슈
          // https://trello.com/c/6WN5p4Po
          // 간헐적으로 actionform frame의 Paint가 동작하지 않는다..
          // DOM 영역에는 분명히 잡히나 브라우저뷰에 Paint되지 않아 비어있는 화면으로 나옴.
          // Hack 사용: 강제 Redraw event
          setTimeout( function() {
            redrawForm();
          }, 0);
        }
      }
    } );

    function sendHeight(height) {
      $(function() {
        // iframe으로 들어간 경우 높이 조정을 위한 postMessage
        if (top !== window) {
          try {
            window.parent.postMessage(JSON.stringify({
              type: 'syncHeight',
              contentHeight: height || document.body.scrollHeight
            }), '*');
          } catch(e) {
            // ignore errors
          }
        }
      } );
    }

    function sendFormHeight(formHeight) {
      $(function() {
        $('[id^=_dable_actionform]').each( function() {
          var formElemOffset = $(this).offset();
          var formOffset = (formElemOffset && formElemOffset.top) || 0;
          // iframe으로 들어간 경우 높이 조정을 위한 postMessage
          if (top !== window) {
            try {
              window.parent.postMessage(JSON.stringify({
                type: 'formHeight',
                formHeight: formHeight,
                formOffset: formOffset
              }), '*');
            } catch(e) { /* ignore errors */ }
          }
        });
      } );
    }

    function getFormHeight() {
      var formElm = document.getElementsByTagName("iframe")[0];
      if (formElm) {
        formElm.contentWindow.postMessage(JSON.stringify({
          type: 'getFormHeight'
        }), '*');
      }
    }

    function redrawForm() {
      var formElms = document.getElementsByTagName("iframe");
      if (formElms && formElms.length > 0) {
        for (var i = 0; i < formElms.length; i++) {
          if (formElms[i])
            $(formElms[i]).force_redraw();
        }
      }
    }

    function init() {
      // floatingbar on
      if ( ACTION_FORM.enabled && ACTION_FORM.is_floating_bar_on ) {
        window.parent.postMessage(JSON.stringify({
          type: 'enableFloating'
        }), '*');
      }
    }

    $(window).load( function() {
      getFormHeight();
      sendHeight();
      init();
    } );
