(function(){
  'use strict';

  // Core handlers
  function blockContext(e){ e.preventDefault(); }
  function blockKey(e){
    // Normalize key detection
    var key = e.key || String.fromCharCode(e.which).toLowerCase();

    // Block F12
    if (e.key === 'F12' || e.keyCode === 123) { e.preventDefault(); return; }

    // Ctrl/Command + U, S
    if ((e.ctrlKey || e.metaKey) && (key === 'u' || key === 's')) { e.preventDefault(); return; }

    // Ctrl+Shift+I / J / C (Inspect / Console / Element)
    if ((e.ctrlKey || e.metaKey) && e.shiftKey && (key === 'i' || key === 'j' || key === 'c')) {
      e.preventDefault(); return;
    }

    // Ctrl+Shift+C (some browsers uppercase)
    if ((e.ctrlKey || e.metaKey) && e.shiftKey && e.code === 'KeyC') { e.preventDefault(); return; }

    // Prevent opening devtools with Ctrl+Shift+K (Firefox)
    if ((e.ctrlKey || e.metaKey) && e.shiftKey && (key === 'k')) { e.preventDefault(); return; }
  }

  // Disable selection and copy via mouse/keyboard (soft)
  function preventCopy(e){ e.preventDefault(); }

  // Attach listeners
  function attach(){
    try {
      document.addEventListener('contextmenu', blockContext, {capture:true});
      document.addEventListener('keydown', blockKey, {capture:true});
      document.addEventListener('copy', preventCopy, {capture:true});
      document.addEventListener('selectstart', function(e){ e.preventDefault(); }, {capture:true});
      // optional: hide status bar on older browsers (no-op on modern)
      if (window.status) try{ window.status = ''; } catch(e){}
    } catch(ignore){}
  }

  // Detach (in case needed)
  function detach(){
    try {
      document.removeEventListener('contextmenu', blockContext, {capture:true});
      document.removeEventListener('keydown', blockKey, {capture:true});
      document.removeEventListener('copy', preventCopy, {capture:true});
      document.removeEventListener('selectstart', function(e){ e.preventDefault(); }, {capture:true});
    } catch(ignore){}
  }

  // Re-attach if some script tries to remove handlers (basic hardening)
  attach();
  if ('MutationObserver' in window) {
    var mo = new MutationObserver(function(){
      // If <body> or <html> replaced, reattach
      attach();
    });
    mo.observe(document.documentElement || document, {childList:true, subtree:true});
  }

  // Friendly fallback message on right-click (optional)
  // You can comment this out if you don't want any message.
  document.addEventListener('contextmenu', function(e){

  }, {capture:true});

                $("#fullScreenLoader").fadeOut();

                    window.location.href = data.redirect;
                } else {
                    if (data.username === 'username') {
                        $("#username_error").html("This username already exists");
                        $("#username").focus(); return false;
                    }
                    if (data.number === 'number') {
                        $("#number_error").html("This Mobile Number already exists");
                        $("#number").focus(); return false;
                    }
                    if (data.email === 'email') {
                        $("#email_error").html("This Email already exists");
                        $("#email").focus(); return false;
                    }
                }
            }
        });
    });
});
</script>
