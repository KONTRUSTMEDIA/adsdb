  <script>
    document.addEventListener("DOMContentLoaded", function() {

      const mainEl = document.querySelector(".product-details-active");
      const navEl = document.querySelector(".product-details-nav");
      if (!mainEl || !navEl) return;

      // Çift init olmasın
      if (mainEl.swiper || navEl.swiper) return;

      const slideCount = mainEl.querySelectorAll(".swiper-slide").length;

      // Thumbs + autoplay ile loop, az slide'da kaydırma hatası yapar
      const useLoop = slideCount >= 6; // güvenli eşik
      const useAutoplay = true;

      const thumbs = new Swiper(navEl, {
        spaceBetween: 10,
        slidesPerView: 4,
        watchSlidesProgress: true,
        slideToClickedSlide: true,
        freeMode: true,
        watchSlidesVisibility: true,
        breakpoints: {
          0: {
            slidesPerView: 3
          },
          576: {
            slidesPerView: 4
          }
        }
      });

      const main = new Swiper(mainEl, {
        slidesPerView: 1,
        spaceBetween: 10,
        centeredSlides: false,
        watchOverflow: true,

        loop: useLoop,
        loopAdditionalSlides: useLoop ? Math.max(6, slideCount) : 0,

        autoplay: useAutoplay ? {
          delay: 2000,
          disableOnInteraction: false,
          pauseOnMouseEnter: true
        } : false,

        thumbs: {
          swiper: thumbs
        },

        observer: true,
        observeParents: true,
      });

      // Autoplay + thumbs + loop durumunda senkronu sağlamlaştır
      main.on("slideChangeTransitionEnd", () => {
        if (thumbs && thumbs.update) thumbs.update();
      });

      window.addEventListener("load", () => {
        main.update();
        thumbs.update();
        if (!useLoop) main.slideTo(0, 0);
      });
    });
  </script>
