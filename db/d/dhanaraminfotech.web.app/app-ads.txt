        // Update Statistics
        async function updateStatistics() {
            try {
                // Get job applications
                const jobSnapshot = await database.ref('jobApplications').once('value');
                const jobs = jobSnapshot.val() || {};
                const totalApplications = Object.keys(jobs).length;
                
                // Get contact messages
                const contactSnapshot = await database.ref('contactMessages').once('value');
                const contacts = contactSnapshot.val() || {};
                const totalMessages = Object.keys(contacts).length;
                
                // Calculate most popular position
                const positionCount = {};
                Object.values(jobs).forEach(app => {
                    positionCount[app.position] = (positionCount[app.position] || 0) + 1;
                });
                
                let popularPosition = 'None';
                let maxCount = 0;
                Object.entries(positionCount).forEach(([position, count]) => {
                    if (count > maxCount) {
                        maxCount = count;
                        popularPosition = position;
                    }
                });
                
                // Update UI
                document.getElementById('totalApplications').textContent = totalApplications;
                document.getElementById('totalMessages').textContent = totalMessages;
                document.getElementById('popularPosition').textContent = popularPosition;
                
            } catch (error) {
                console.error('Error updating statistics:', error);
            }
        }
