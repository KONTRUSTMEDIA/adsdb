  <!--
    Standalone fallback detector (no dependencies).
    Runs BEFORE React mounts. If React's own detection takes over, this hides itself.
    Catches: extension-based blockers (DOM), DNS-level blockers (network), Brave Shields.

  <script type="2bb8326d7d72962ce1145dfb-text/javascript">
    (function () {
      // ── Early API gate ─────────────────────────────────────────
      // Block ALL API requests immediately. The gate is released once
      // detection resolves clean, or settings say detection is disabled.
      // This prevents data from loading behind the overlay.
      // BUT: if user already acknowledged (soft-block skip), let them through.
      var _ack = false;
      try { _ack = localStorage.getItem('_pref_density') === '1' } catch (e) { }

      // Check if user is premium (active or expired) from JWT in cookie
      var _isPremium = false, _premiumExpired = false, _isStaff = false;
      var _staffRoles = ['admin', 'super-admin', 'moderator', 'manga', 'team', 'team_manager', 'comment', 'user_manager', 'editor'];
      try {
        var m = document.cookie.match(/(?:^|;\s*)auth-token=([^;]*)/);
        if (m) {
          var parts = m[1].split('.');
          if (parts.length === 3) {
            var p = JSON.parse(atob(parts[1].replace(/-/g, '+').replace(/_/g, '/')));
            var u = p.user || p;
            // Check permissions for staff roles
            var perms = u.permissions || p.permissions || [];
            if (Array.isArray(perms)) {
              for (var i = 0; i < perms.length; i++) {
                if (_staffRoles.indexOf(perms[i]) !== -1) { _isStaff = true; break; }
              }
            }
            if (u.is_premium) {
              var exp = u.membership_expires_at;
              if (exp && new Date(exp).getTime() >= Date.now()) {
                _isPremium = true;      // active premium
              } else if (exp) {
                _premiumExpired = true;  // was premium, now expired
              }
            }
          }
        }
      } catch (e) { }

      // Gate starts OFF — only turned ON when detection actually finds a blocker.
      // This avoids blocking API requests before we even know if detection is enabled.
      window.__adblock_blocked = false;

      // Skip if React overlay already took over
      if (window.__adblock_react_active) return;

      // Users who are premium, staff, or already acknowledged skip detection entirely
      var _skipDetection = _ack || _isPremium || _premiumExpired || _isStaff;

      var TIMEOUT = 6000, SETTLE = 400;
      var score = 0, isDNS = false, isBrave = false, hasExt = false;
      // Defaults — will be overridden by API settings if available
      var THRESHOLD = 35, DETECTION_ENABLED = true, DETECTION_MODE = 'soft-block';

      // --- Fetch settings from ads_settings API ---
      function fetchSettings() {
        return new Promise(function (resolve) {
          // Try API first, then fallback to local ads-config.json
          var urls = ['/api/ads_settings', '/ads-config.json'];
          var tried = 0;
          function tryNext() {
            if (tried >= urls.length) { resolve(); return }
            var url = urls[tried++];
            if (!window.fetch) { resolve(); return }
            fetch(url, { cache: 'no-store' }).then(function (r) {
              if (!r.ok) throw new Error('HTTP ' + r.status);
              return r.json();
            }).then(function (data) {
              // Unwrap common server wrappers
              var d = data && data.data ? data.data : data && data.settings ? data.settings : data;
              if (!d) { resolve(); return }
              var ab = d.adblockDetection || d.adblock_detection || {};
              if (typeof ab.enabled === 'boolean') DETECTION_ENABLED = ab.enabled;
              if (typeof ab.confidenceThreshold === 'number') THRESHOLD = ab.confidenceThreshold;
              if (typeof ab.confidence_threshold === 'number') THRESHOLD = ab.confidence_threshold;
              if (typeof ab.mode === 'string') DETECTION_MODE = ab.mode;
              resolve();
            }).catch(function () { tryNext() });
          }
          tryNext();
          // Don't wait forever for settings
          setTimeout(function () { resolve() }, 3000);
        });
      }

      // --- Brave detection ---
      function checkBrave(cb) {
        try {
          if (navigator.brave && typeof navigator.brave.isBrave === 'function') {
            navigator.brave.isBrave().then(function (r) { cb(!!r) }).catch(function () { cb(false) });
            return;
          }
          if (navigator.userAgentData && navigator.userAgentData.brands) {
            for (var i = 0; i < navigator.userAgentData.brands.length; i++) {
              if (navigator.userAgentData.brands[i].brand.toLowerCase().indexOf('brave') !== -1) { cb(true); return }
            }
          }
          if ('brave' in navigator) { cb(true); return }
        } catch (e) { }
        cb(false);
      }

      // --- Bait check ---
      function checkBaits() {
        var ids = ['ad-bait-primary', 'ad-bait-secondary', 'ad-bait-tertiary', 'ad-bait-banner', 'ad-bait-sidebar'];
        var hidden = 0;
        for (var i = 0; i < ids.length; i++) {
          var el = document.getElementById(ids[i]);
          if (!el) { hidden++; continue }
          var s = window.getComputedStyle(el);
          if (s.display === 'none' || s.visibility === 'hidden' || el.offsetHeight === 0 && el.offsetParent === null) hidden++;
        }
        return hidden >= 2 ? Math.min(hidden * 18, 100) : hidden === 1 ? 15 : 0;
      }

      // --- Network probe (catches DNS-level + extension blocking) ---
      function networkProbe(url) {
        return new Promise(function (resolve) {
          var done = false;
          // Method 1: fetch
          if (window.fetch) {
            fetch(url, { method: 'HEAD', mode: 'no-cors', cache: 'no-store' }).then(function () {
              if (!done) { done = true; resolve(0) }
            }).catch(function () {
              if (!done) { done = true; resolve(60) }
            });
          }
          // Method 2: image pixel (fallback + parallel)
          var img = new Image();
          img.onload = function () { if (!done) { done = true; resolve(0) } };
          img.onerror = function () { if (!done) { done = true; resolve(55) } };
          img.src = url + '?_=' + Date.now();
          // Timeout
          setTimeout(function () { if (!done) { done = true; resolve(30) } }, TIMEOUT);
        });
      }

      // --- Multi-URL probe (hits multiple ad domains for reliability) ---
      function multiNetworkProbe() {
        var urls = [
          'https://pagead2.googlesyndication.com/pagead/js/adsbygoogle.js',
          'https://www.googleadservices.com/pagead/conversion.js',
          'https://static.ads-twitter.com/uwt.js'
        ];
        return Promise.all(urls.map(networkProbe)).then(function (scores) {
          // If 2+ domains blocked, high confidence it's DNS-level blocking
          var blocked = scores.filter(function (s) { return s > 0 });
          if (blocked.length >= 2) { isDNS = true; return Math.max.apply(null, scores) }
          return blocked.length === 1 ? Math.floor(scores.reduce(function (a, b) { return Math.max(a, b) }, 0) * 0.6) : 0;
        });
      }

      // --- Dynamic bait ---
      function dynamicBait() {
        return new Promise(function (resolve) {
          var el = document.createElement('div');
          el.innerHTML = '&nbsp;';
          el.className = 'adsbox ad-unit adsbygoogle banner_ad';
          el.style.cssText = 'position:absolute;top:-10000px;left:-10000px;width:1px;height:1px;z-index:-1;pointer-events:none;';
          document.body.appendChild(el);
          setTimeout(function () {
            var gone = !document.body.contains(el);
            var s = window.getComputedStyle(el);
            var hid = s.display === 'none' || s.visibility === 'hidden' || el.offsetHeight === 0;
            el.remove();
            resolve(gone ? 55 : hid ? 45 : 0);
          }, SETTLE);
        });
      }

      // --- Show overlay ---
      function showOverlay() {
        if (window.__adblock_react_active) return; // React took over
        var ov = document.getElementById('_fb_overlay');
        if (!ov) return;
        ov.classList.add('active');
        document.documentElement.style.overflow = 'hidden';
        document.body.style.overflow = 'hidden';

        // Block all API requests globally (same flag the React API layer checks)
        window.__adblock_blocked = true;

        // Show ALL relevant instruction panels and type labels
        var extI = document.getElementById('_fb_instruct_ext');
        var dnsI = document.getElementById('_fb_instruct_dns');
        var braveI = document.getElementById('_fb_instruct_brave');
        var typeBrave = document.getElementById('_fb_type_brave');
        var typeDNS = document.getElementById('_fb_type_dns');
        var typeExt = document.getElementById('_fb_type_ext');
        var typeGeneric = document.getElementById('_fb_type_generic');
        var multiNotice = document.getElementById('_fb_multi_notice');

        // Reset all
        if (extI) extI.style.display = 'none';
        if (dnsI) dnsI.style.display = 'none';
        if (braveI) braveI.style.display = 'none';

        var reasonCount = 0;

        if (isBrave) {
          if (braveI) braveI.style.display = 'block';
          if (typeBrave) typeBrave.style.display = 'block';
          reasonCount++;
        }
        if (isDNS) {
          if (dnsI) dnsI.style.display = 'block';
          if (typeDNS) typeDNS.style.display = 'block';
          reasonCount++;
        }
        if (hasExt) {
          if (extI) extI.style.display = 'block';
          if (typeExt) typeExt.style.display = 'block';
          reasonCount++;
        }
        // If nothing specific detected, show generic
        if (reasonCount === 0) {
          if (extI) extI.style.display = 'block';
          if (typeGeneric) typeGeneric.style.display = 'block';
        }
        // Show multi-notice if 2+ reasons
        if (reasonCount >= 2 && multiNotice) {
          multiNotice.style.display = 'block';
        }

        // Continue button — hidden in hard-block mode
        var btn = document.getElementById('_fb_continue');
        var cdEl = document.getElementById('_fb_countdown');
        if (DETECTION_MODE === 'hard-block') {
          // Hard-block: no continue button at all
          if (btn) btn.style.display = 'none';
        } else {
          // Soft-block / warn: show countdown then enable
          var cd = 7;
          var iv = setInterval(function () {
            cd--;
            if (cdEl) cdEl.textContent = '(' + cd + ')';
            if (cd <= 0) {
              clearInterval(iv);
              if (cdEl) cdEl.style.display = 'none';
              if (btn) { btn.disabled = false; btn.classList.remove('disabled') }
            }
          }, 1000);
          if (btn) {
            btn.onclick = function () {
              if (this.disabled) return;
              // Lift the API gate
              window.__adblock_blocked = false;
              // Store acknowledgment
              try { localStorage.setItem('_pref_density', '1') } catch (e) { }
              // Reload so the UI fetches fresh data
              location.reload();
            };
          }
        }
      }

      // --- Main ---
      function run() {
        if (window.__adblock_react_active) return;
        // Premium, staff, or acknowledged users skip detection entirely
        if (_skipDetection) return;
        // Fetch settings first, then run detection
        fetchSettings().then(function () {
          if (!DETECTION_ENABLED) { return; }
          if (window.__adblock_react_active) return; // React manages gate
          // In warn mode, don't block API — users see warning but can browse
          if (DETECTION_MODE === 'warn') window.__adblock_blocked = false;

          checkBrave(function (brave) {
            isBrave = brave;
            var baitScore = checkBaits();
            score += baitScore;

            Promise.all([multiNetworkProbe(), dynamicBait()]).then(function (results) {
              if (window.__adblock_react_active) return;
              var netScore = results[0], dynScore = results[1];
              score = Math.max(score, baitScore);

              // DNS: network probes fired
              if (netScore >= 50) isDNS = true;
              // Extension: bait or dynamic bait was hidden/removed
              if (baitScore > 0 || dynScore > 0) hasExt = true;

              // Combine: take the max of network and DOM signals, weight them
              var finalScore = Math.min(
                Math.round(Math.max(baitScore, dynScore) * 0.5 + netScore * 0.5),
                100
              );
              // If any single vector is very high, use it directly
              finalScore = Math.max(finalScore, baitScore, netScore, dynScore);
              finalScore = Math.min(finalScore, 100);

              if (finalScore >= THRESHOLD) {
                // Check if user already acknowledged (soft-block skip)
                if (_ack) {
                  window.__adblock_blocked = false;
                  return;
                }
                showOverlay();
              } else {
                // No blocker detected — release the API gate
                window.__adblock_blocked = false;
              }
            }).catch(function () { window.__adblock_blocked = false; });
          });
        });
      }

      // Wait for DOM then run
      if (document.readyState === 'loading') {
        document.addEventListener('DOMContentLoaded', function () { setTimeout(run, 500) });
      } else {
        setTimeout(run, 500);
