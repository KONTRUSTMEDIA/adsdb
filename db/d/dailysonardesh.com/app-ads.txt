<script>
document.addEventListener('DOMContentLoaded',()=>{
  const off    = document.getElementById('uttOffcanvas');
  const back   = document.getElementById('uttBackdrop');
  const search = document.getElementById('uttSearchOverlay');
  const input  = document.getElementById('uttSearchInput');

  const openMenuBtn  = document.getElementById('uttOpenMenu');
  const closeMenuBtn = document.getElementById('uttCloseMenu');

  function lock(){ document.body.style.overflow='hidden'; }
  function unlock(){ document.body.style.overflow=''; }

  function closeMenu(){
    off.classList.remove('active');
    back.classList.remove('active');
    unlock();
  }

  function openMenu(){
    off.classList.add('active');
    back.classList.add('active');
    lock();
  }

  function openSearch(){
    search.classList.add('active');
    lock();
    setTimeout(()=>{ if(input) input.focus(); }, 50);
  }

  function closeSearch(){
    search.classList.remove('active');
    unlock();
  }

  // ✅ offcanvas only from left hamburger
  openMenuBtn && (openMenuBtn.onclick = openMenu);
  closeMenuBtn && (closeMenuBtn.onclick = closeMenu);
  back && (back.onclick = closeMenu);

  // search open/close
  const openSearchBtn  = document.getElementById('uttOpenSearch');
  const closeSearchBtn = document.getElementById('uttCloseSearch');
  openSearchBtn && (openSearchBtn.onclick = openSearch);
  closeSearchBtn && (closeSearchBtn.onclick = closeSearch);

  // Esc key
  document.addEventListener('keydown', (e)=>{
    if(e.key === 'Escape'){
      if(search.classList.contains('active')) closeSearch();
      if(off.classList.contains('active')) closeMenu();
    }
  });

  // submenu toggle (child থাকলে টগল, না থাকলে ক্লিকেই যাবে)
  off.querySelectorAll('.utt-off-item>a').forEach(a=>{
    a.onclick = (e)=>{
      const sub = a.parentElement.querySelector('.utt-off-sub');
      if(sub){
        e.preventDefault();
        a.parentElement.classList.toggle('open');
      }else{
        closeMenu();
      }
    };
  });
});
</script>
