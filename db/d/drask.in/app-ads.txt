    function Login() {
        $('#add').html('Please wait..');
        var Name = $('#kullaniciadi').val();
        var Password = $('#Passwords').val();


        $.ajax({
            url: '/JSLogin',
            type: 'GET',
            data: {
                Name: Name,
                Password: Password
            },
            success: function (data) {
                if (data.s == "Ok") {
                    $('#add').html('Login Successful, You Are Redirected, Please Wait ...');
                    setTimeout(login, 1);
                }
                if (data.s == "NameError") {
                    $('#add').html('Your username is incorrect...');
                }
                if (data.s == "PasswordError") {
                    $('#add').html('Your Password is Incorrect...');
                }
                if (data.s == "Error") {
                    $('#add').html('A Problem Occurred, Please Try Again Later...');
                }
                if (data.s == "KayitYok") {
                    $('#add').html('Username or Password Incorrect...');
                }
            }
        });
    }
    </script>

                }
                if (data.s == "error") {
                    $('#addMailEkle').html('A Problem Occurred, Please Try Again Later...');
                }
            }
        });
    }
