<script>

    $("#codecyan-form").on('submit', function(e) {
        e.preventDefault();
        
      var itemUrl = $("#codecyan-download-link").val();
      var gToken = $("#codecyan-token-hidden").val();
        $.ajax({
            url: 'https://getpaidstock.com/api.php?url=' + encodeURIComponent(itemUrl) + '&token=' + gToken,
            type: 'GET',
            dataType: 'JSON',
            success: function(data) {
                if (data.success) {
                    $("#codecyan-error").removeClass("d-block").addClass("d-none"); //hide any error alerts
                    $("#codecyan-success").html('<a href="' + data.result + '"><button type="button" class="secondary-button" aria-hidden="true">Download Generated Photo</button></a>');
                    $("#codecyan-success").removeClass("d-none").addClass("d-block");
                } else {
                  $("#codecyan-success").removeClass("d-block").addClass("d-none"); //hide any error alerts
                    $("#codecyan-error").text(data.result);
                    $("#codecyan-error").removeClass("d-none").addClass("d-block");
                }
            },
            error: function(error) {
              console.log(error);
                $("#codecyan-error").text("Oops! There is an issue at our end. Please try again.");
                $("#codecyan-error").removeClass("d-none").addClass("d-block");
            }
        });
    });

    $(document).ready(function () {
      $(document).ajaxStart(function () {
          $("#codecyan-download-btn-default").hide();
          $("#codecyan-download-btn-processing").show();
        }).ajaxStop(function () {
          $("#codecyan-download-btn-processing").hide();
          $("#codecyan-download-btn-default").show();
          });
    });
