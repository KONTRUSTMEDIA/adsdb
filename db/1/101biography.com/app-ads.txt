    /* FETCH GENERIC */
    async function fetchFromCMS(ep, elId, fn, empId){
      const el=document.getElementById(elId); if(!el)return;
      try{
        const d=await fetchJson(`/cms/api/${ep}`);
        if(d.ok && d.items && d.items.length>0){ 
          if(ep.includes('gallery')) GALLERY_ITEMS=d.items; 
          el.innerHTML=fn(d.items); 
        } else { document.getElementById(empId)?.classList.remove('hidden'); el.innerHTML=''; }
      }catch{ el.innerHTML='<div class="text-red-500 text-xs text-center col-span-full">Gagal memuat data.</div>'; }
    }

    /* COUNTDOWN */
    function startCountdown(d){
      const t=new Date(d).getTime();
      if(countdownTimer)clearInterval(countdownTimer);
      countdownTimer=setInterval(()=>{
        const now=new Date().getTime(), dist=t-now;
        if(dist<0){ document.getElementById("cdDays").innerText="0"; return;}
        document.getElementById("cdDays").innerText=Math.floor(dist/(864e5));
        document.getElementById("cdHours").innerText=Math.floor((dist%864e5)/(36e5)).toString().padStart(2,'0');
        document.getElementById("cdMins").innerText=Math.floor((dist%36e5)/(6e4)).toString().padStart(2,'0');
        document.getElementById("cdSecs").innerText=Math.floor((dist%6e4)/1e3).toString().padStart(2,'0');
      },1000);
    }
