    jQuery(document).ready(function() {
        const apiUrl = '/wp-json/mediaset-app-api/v1/programmaOnair';

        function decodeHtmlEntities(text) {
            const textarea = document.createElement('textarea');
            textarea.innerHTML = text;
            return textarea.value;
        }
        fetch(apiUrl)
            .then(response => {
                if (!response.ok) {
                    throw new Error(`Network response was not ok: ${response.statusText}`);
                }
                return response.json();
            })
            .then(data => {
                console.log(data);
                if (data && data.onair && data.onair.length > 0) {
                    const onAirProgram = data.onair[0];
                    const titleFM = decodeHtmlEntities(onAirProgram.titoloFM);
                    const imageUrl = onAirProgram.immagineXL;
                    const titleLinkMobile = document.querySelectorAll("#onair-title-link-mobile");
                    if (titleLinkMobile) {
                        titleLinkMobile.forEach(title => {
                            title.textContent = titleFM;
                        });
                    }
                    const titleLinkDesktop = document.querySelectorAll('#onair-title-link-desktop');
                    if (titleLinkDesktop) {
                        titleLinkDesktop.forEach(title => {
                            title.textContent = titleFM;
                        });
                    }
                    const onairImage = document.querySelectorAll("[id='onair-image']");
                    if (onairImage) {
                        onairImage.forEach(title => {
                            title.src = imageUrl;
                            title.alt = titleFM;
                        });
                    }
                } else {}
            })
            .catch(error => {
                const titleLinkMobile = document.querySelectorAll("#onair-title-link-mobile");
                const titleLinkDesktop = document.querySelectorAll('#onair-title-link-desktop');
                const defaultText = 'On Air';

                if (titleLinkMobile) {
                    titleLinkMobile.forEach(title => {
                        title.textContent = defaultText;

                }
                if (titleLinkDesktop) {
                    titleLinkDesktop.forEach(title => {
                        title.textContent = defaultText;
                    });
                }

                const onairImage = document.querySelectorAll("#onair-image");
                if (onairImage) {

                        title.style.display = 'none';

            });
    });

function handleUserLoginSuccessful(payload) {
    user_logged = true;

  rtiCastingPlugin.join = async function (url) {

    }

    // Listener una tantum

      () => rtiCastingPlugin.redirectWithToken(url),

    );

    try {

      if (isUserLogged) {
        rtiCastingPlugin.log("debug", "logged");
        await rtiCastingPlugin.redirectWithToken(url);

        rtiCastingPlugin.log("debug", "userNotLogged");

      }
    } catch (e) {
      rtiCastingPlugin.log("error", "login check failed", e);
      handlerClickShowLogin();
    }
  };

  // Eseguo davvero il join
  await rtiCastingPlugin.join(url);
}
