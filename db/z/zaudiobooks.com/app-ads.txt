<script>
let excludeIds = []; 
let loadingRandom = false;

$('#btn-load-more-random').on('click', function () {
    if (loadingRandom) return;

    loadingRandom = true;
    const $btn = $(this);
    const originalText = $btn.text();
    $btn.text('Loading......');

    $.getJSON(
        'https://naudios.com/API/see_more_random.php',
        {
            key: 'NAUDIOS_KEY_2024',
            lastID: excludeIds.slice(-100).join(',') 
        },
        function (res) {
            if (res.success && res.data.length > 0) {
                res.data.forEach(function (post) {
                    excludeIds.push(post.id);
                    $('#post-list').append(`
                        <a href="/watch/${post.id}" class="post-item">
                            <div class="post-thumb">
                                <img src="${BASE_URL}img/thumb/${post.thumb}" 
                                     alt="${post.title}" 
                                     class="detail-thumb">
                            </div>
                            <div class="flex-grow-1">
                                <div class="post-title">${post.title}</div>
                            </div>
                        </a>
                    `);
                });

                if (excludeIds.length > 100) {
                    excludeIds = excludeIds.slice(-100);
                }

                $btn.text(originalText);
            } else {
                $btn.text('No more random items').prop('disabled', true);
            }

            loadingRandom = false;
        }
    ).fail(function() {
        $btn.text('Error, try again');
        loadingRandom = false;
    });
});
</script>
