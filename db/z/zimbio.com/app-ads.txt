<script>
document.addEventListener('DOMContentLoaded', function () {

    // ==== Tabs ====
    document.querySelectorAll('.tabs-wrapper').forEach(function (wrapper) {
        var tabs = wrapper.querySelectorAll('.tabs-nav li');
        var contents = wrapper.querySelectorAll('.tab-content');

        tabs.forEach(function (tab) {
            tab.addEventListener('click', function () {
                tabs.forEach(function (t) { t.classList.remove('active'); });
                contents.forEach(function (c) { c.classList.remove('active'); });

                this.classList.add('active');
                var targetId = this.getAttribute('data-tab');
                var target = wrapper.querySelector('#' + targetId);
                if (target) {
                    target.classList.add('active');
                }
            });
        });
    });

    // ==== Mobile hamburger menu ====
    var toggleBtn = document.querySelector('.hamburger');
    var menu = document.querySelector('.header-menu');
    if (toggleBtn && menu) {
        toggleBtn.addEventListener('click', function () {
            menu.classList.toggle('active');
        });
    }

    // ==== Mobile submenu tap behavior ====
    var breakpoint = 768;
    if (window.innerWidth <= breakpoint) {
        var menuItems = document.querySelectorAll('.nav-menu li.menu-item-has-children > a');
        if (menuItems.length) {
            menuItems.forEach(function (link) {
                var tappedOnce = false;

                link.addEventListener('click', function (e) {
                    var parentLi = this.parentElement;

                    if (!tappedOnce) {
                        e.preventDefault();
                        tappedOnce = true;

                        menuItems.forEach(function (otherLink) {
                            if (otherLink !== link) {
                                otherLink.parentElement.classList.remove('submenu-open');
                                otherLink.dataset.tapped = 'false';
                            }
                        });

                        parentLi.classList.add('submenu-open');

                        setTimeout(function () {
                            tappedOnce = false;
                        }, 1500);
                    }
                });
            });
        }
    }
});
</script>
