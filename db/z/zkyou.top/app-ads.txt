    <script>
      new Vue({
        el: "#app",
        components: { "header-component": HeaderComponent, "menu-component": MenuComponent, "footer-component": FooterComponent },
        data: { 
          games: [], 
          displayedGames: [], // 当前显示的游戏
          gamesPerPage: 12, // 每页显示的游戏数量
          currentPage: 1, // 当前页码
          showSearch: false, 
          searchValue: "", 
          selectedCategories: [],
          showMenu: false,
          showCategoryMenu: false, //
          selectedCategory: null //  
        },
        computed: {
          // 随机游戏列表（内部先打乱一次）
          randomizedGames() {
            return [...this.games].sort(() => 0.5 - Math.random());
          },
          categories() {
                    // 获取所有不重复的游戏分类
                    return [...new Set(this.games.map(game => game.category))].sort();
                },
          // 精选游戏（随机取6个）
          randomizedGames() {
            return [...this.games].sort(() => 0.5 - Math.random());
          },
          featuredGames() {
            return this.randomizedGames.slice(0, 3);
          },
          hotGames() {
            return this.randomizedGames.slice(6, 14);
          },
          // 是否还有更多游戏可以加载
          hasMoreGames() {
            return this.displayedGames.length < this.games.length;
          }
        },
        mounted() {
          window.addEventListener("scroll", this.handleScroll);
          this.getData();
        },
        beforeDestroy() {
          window.removeEventListener("scroll", this.handleScroll);
        },
        methods: {
          toggleSearch() {
            if (this.searchValue) {
              // 搜索功能
              const filteredGames = this.games.filter(game => 
                game.title.toLowerCase().includes(this.searchValue.toLowerCase()) ||
                game.category.toLowerCase().includes(this.searchValue.toLowerCase()) ||
                game.tags.toLowerCase().includes(this.searchValue.toLowerCase())
              );
              // 跳转到搜索结果页面
              window.location.href = `./allGames.html?search=${encodeURIComponent(this.searchValue)}`;
            } else {
              this.showSearch = !this.showSearch;
            }
          },
          performSearch() {
            if (this.searchValue.trim()) {
              // 跳转到搜索结果页面
              window.location.href = `./allGames.html?search=${encodeURIComponent(this.searchValue)}`;
            }
          },
          // 按分类筛选 - 跳转到所有游戏页面并传递分类参数
          filterByCategory(category) {
            window.location.href = `./allGames.html?category=${encodeURIComponent(category)}`;
          },
          handleScroll() {
            const backToTopButton = document.getElementById("back-to-top");
            if (document.body.scrollTop > 20 || document.documentElement.scrollTop > 20) {
              backToTopButton.style.display = "flex";
            } else {
              backToTopButton.style.display = "none";
            }
          },
          scrollToTop() {
            window.scrollTo({ top: 0, behavior: "smooth" });
          },
          toggleCategoryMenu() {
              this.showCategoryMenu = !this.showCategoryMenu;
          },
          selectCategory(category) {
              this.filterByCategory(category);
              this.toggleCategoryMenu(); // 选择分类后关闭菜单
          },
          getData() {
            this.games = games;
            // 初始化显示的游戏
            this.loadMoreGames();
          },
          playGame(game) {
            // 打开游戏URL
            //window.open(game.url, '_blank');
            window.location.href = "./detail.html?id="+game.id;
          },
          toHome() {
            window.scrollTo({ top: 0, behavior: 'smooth' });
          },
          toogleMenu() {
            this.showMenu = !this.showMenu;
          },
          // 加载更多游戏
          loadMoreGames() {
            const startIndex = this.displayedGames.length;
            const endIndex = startIndex + this.gamesPerPage;
            const newGames = this.games.slice(startIndex, endIndex);
            
            this.displayedGames = [...this.displayedGames, ...newGames];
            this.currentPage++;
          },
          // 跳转到所有游戏页面（原toAllList功能）
          toAllGames() {
            window.location.href = "./allGames.html";
          }
        },
      });
