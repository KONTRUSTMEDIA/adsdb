<script>
  (function() {
    var cards = document.querySelectorAll('.testimonial-card');
    var buttons = document.querySelectorAll('.testimonials-nav button');
    var currentIndex = 0;
    buttons.forEach(function(btn) {
      btn.addEventListener('click', function() {
        var newIndex = parseInt(btn.dataset.index);
        if (newIndex === currentIndex) return;
        cards.forEach(function(card, i) {
          card.classList.remove('active', 'exiting');
          if (i === newIndex) card.classList.add('active');
          else if (i === currentIndex) card.classList.add('exiting');
        });
        buttons.forEach(function(b, i) { b.classList.toggle('active', i === newIndex); });
        currentIndex = newIndex;
      });
    });
  })();
</script>

    const expanded = this.getAttribute('aria-expanded') === 'true';
    this.setAttribute('aria-expanded', !expanded);
    this.closest('.nav').querySelector('.nav-links').classList.toggle('nav-open');
  });
  // Close menu on link click (mobile)
  document.querySelectorAll('.nav-links a').forEach(function(link) {
    link.addEventListener('click', function() {
      const toggle = document.querySelector('.nav-toggle');
      if (toggle && toggle.getAttribute('aria-expanded') === 'true') {
        toggle.setAttribute('aria-expanded', 'false');
        toggle.closest('.nav').querySelector('.nav-links').classList.remove('nav-open');
      }
    });
  });
  // Scroll-triggered fade-in animations
  (function() {
    var targets = document.querySelectorAll('.service-card, .pricing-card, .guarantee-card, .team-member, .faq-item, .process-step, .contact-card, .blog-card, .book-card, .essentials-tier, .royalty-option');
    if (!targets.length || !('IntersectionObserver' in window)) return;
    targets.forEach(function(el) { el.classList.add('fade-target'); });
    var observer = new IntersectionObserver(function(entries) {
      entries.forEach(function(entry) {
        if (entry.isIntersecting) {
          entry.target.classList.add('fade-in');
          observer.unobserve(entry.target);
        }
      });
    }, { threshold: 0.1, rootMargin: '0px 0px -40px 0px' });
    targets.forEach(function(el) { observer.observe(el); });
  })();
  </script>
