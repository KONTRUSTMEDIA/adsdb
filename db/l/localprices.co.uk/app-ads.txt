                } catch (error) {
                    logError('Error fetching businesses:', error);
                }
            };
        }

        // Function to dynamically load Google Maps script
        async function loadGoogleMapsScript() {
            try {
                const response = await fetch('/api/get-maps-script');
                const data = await response.json();
                if (data.scriptUrl) {
                    const script = document.createElement('script');
                    script.src = data.scriptUrl;
                    script.async = true;
                    script.defer = true;
                    script.onload = initMap;
                    document.head.appendChild(script);
                } else {
                    console.error('No script URL returned from API');
                    document.getElementById('mapSection').style.display = 'none';
                }
            } catch (error) {
                console.error('Error fetching Google Maps script URL:', error);
                document.getElementById('mapSection').style.display = 'none';
            }
        }

        // Ensure original functions work
        window.addEventListener('load', function () {
            loadGoogleMapsScript();
        });
    </script>
