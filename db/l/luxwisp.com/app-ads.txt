document.addEventListener('DOMContentLoaded', function() {

    // Get Elements
    var container = document.getElementById('bpc-app'); // Get container for checks
    var petTypeSelect = document.getElementById('bpc-pet-type');
    var dogSizeArea = document.getElementById('bpc-dog-size-area');

    var petAgeInput = document.getElementById('bpc-pet-age');

    var errorDiv = document.getElementById('bpc-error');

    var summaryInfoSpan = document.getElementById('bpc-summary-info');
    var humanAgeSpan = document.getElementById('bpc-human-age');
    var funFactSpan = document.getElementById('bpc-fun-fact');

    // Check essential elements exist
    if (!container || !petTypeSelect || !dogSizeArea || !dogSizeSelect || !petAgeInput || !calcButton || !resetButton || !errorDiv || !resultArea || !summaryInfoSpan || !humanAgeSpan || !funFactSpan) {
        console.log("BPC Error: HTML structure missing/broken.");
        if(document.getElementById('bpc-error')) { // Try to display error
             document.getElementById('bpc-error').innerHTML = 'Calculator Error: Init failed.';
             document.getElementById('bpc-error').style.display = 'block'; // Make sure error div is visible
        }
        return;
    }

    var animationFrameId = null; // For count-up animation

    // --- Helper Functions ---
    function applyErrorStyles(errors) { // Pass error messages to check which field failed
        // Clear previous
        if(petTypeSelect) petTypeSelect.classList.remove('bpc-input-error');
        if(petAgeInput) petAgeInput.classList.remove('bpc-input-error');
        if(dogSizeSelect) dogSizeSelect.classList.remove('bpc-input-error');

        // Apply based on error text
        if (errors.indexOf('species') > -1 && petTypeSelect) petTypeSelect.classList.add('bpc-input-error');
        if (errors.indexOf('age') > -1 && petAgeInput) petAgeInput.classList.add('bpc-input-error');
        if (errors.indexOf('size') > -1 && dogSizeSelect) dogSizeSelect.classList.add('bpc-input-error');
    }

    function clearErrorStyles() {
        if(petTypeSelect) petTypeSelect.classList.remove('bpc-input-error');
        if(petAgeInput) petAgeInput.classList.remove('bpc-input-error');
        if(dogSizeSelect) dogSizeSelect.classList.remove('bpc-input-error');
    }

    function animateCountUp(element, endValue, duration) {
        if (animationFrameId) { cancelAnimationFrame(animationFrameId); }
        if (!element) { return; }
        var startValue = parseFloat(element.innerHTML) || 0; var startTime = null;
        function step(timestamp) {
            if (!startTime) startTime = timestamp; var elapsed = timestamp - startTime;
            var progress = Math.min(elapsed / duration, 1);
            var currentValue = progress * (endValue - startValue) + startValue;
            var isWhole = endValue % 1 === 0;
            element.innerHTML = isWhole ? Math.round(currentValue) : currentValue.toFixed(1);
            if (progress < 1) { animationFrameId = requestAnimationFrame(step); }
            else { element.innerHTML = isWhole ? endValue : endValue.toFixed(1); animationFrameId = null; }
        }
        animationFrameId = requestAnimationFrame(step);
    }

    // --- Event Handlers ---
    petTypeSelect.onchange = function() {
        var isDog = petTypeSelect.value === 'dog';
        dogSizeArea.style.display = isDog ? 'block' : 'none';
        if (!isDog) { dogSizeSelect.value = ''; if(dogSizeSelect) dogSizeSelect.classList.remove('bpc-input-error'); }
         if (errorDiv.innerHTML !== '') { errorDiv.innerHTML = ''; clearErrorStyles(); } // Clear errors and styles on type change
    };

    };

    calcButton.onclick = function() {
        var petType = petTypeSelect.value; var ageString = petAgeInput.value; var petAge = parseFloat(ageString); var dogSize = ''; var humanAge = 0; var isValid = true; var errors = [];

        errorDiv.innerHTML = ''; errorDiv.style.display = 'none'; // Hide error div initially
        resultArea.classList.remove('visible'); // Hide results before calc
        if (animationFrameId) { cancelAnimationFrame(animationFrameId); animationFrameId = null; humanAgeSpan.innerHTML = '0'; }

        // --- Validation ---
        if (petType === '') { isValid = false; errors.push('Please select a pet species.'); }
        if (ageString === '' || isNaN(petAge) || petAge < 0 || petAge > 30) { isValid = false; errors.push('Please enter a valid age (0-30).'); }
        if (petType === 'dog') { dogSize = dogSizeSelect.value; if (dogSize === '') { isValid = false; errors.push('Please select the dog\'s size.'); } }

        if (!isValid) {
            errorDiv.innerHTML = errors.join('<br>');
            errorDiv.style.display = 'block'; // Show error div
            applyErrorStyles(errorDiv.innerHTML); // Apply styles based on errors found
            return;
        } else {
             clearErrorStyles(); // Clear styles if valid
        }

        // --- Calculation Logic (Unchanged - Safe Syntax) ---
        var firstYearHumanEq = { cat: 15, dog: 15, rabbit: 21, guinea_pig: 14, hamster: 30 };
        if (petAge < 1) { humanAge = petAge * (firstYearHumanEq[petType] || 15); }
        else { switch (petType) { /* Calculation cases as before */
                 case 'cat': if (petAge === 1) { humanAge = 15; } else if (petAge <= 2) { humanAge = 15 + (petAge - 1) * 9; } else { humanAge = 24 + (petAge - 2) * 4; } break;
                 case 'dog': if (petAge === 1) { humanAge = 15; } else if (petAge <= 2) { humanAge = 15 + (petAge - 1) * 9; } else { var factor = 5; if (dogSize === 'small') { factor = 4; } else if (dogSize === 'large') { factor = 6; } humanAge = 24 + (petAge - 2) * factor; } break;
                 case 'rabbit': if (petAge === 1) { humanAge = 21; } else { humanAge = 21 + (petAge - 1) * 6; } break;
                 case 'guinea_pig': if (petAge === 1) { humanAge = 14; } else { humanAge = 14 + (petAge - 1) * 8; } break;
                 case 'hamster': humanAge = petAge * 30; break; default: humanAge = petAge * 7; }
        } humanAge = Math.round(Math.max(0, humanAge) * 10) / 10;

        // --- Get Fun Fact (Unchanged - Safe Syntax) ---
        var stage = ""; var emj = { cat: '🐈', dog: '🐕', rabbit: '🐇', guinea_pig: '🐹', hamster: '🐹' }[petType] || '🐾'; var young = petAge < 1;
        if (petType === 'hamster') { if (humanAge > 60) { stage = 'a very wise old ' + emj + '!'; } else if (humanAge > 30) { stage = 'a mature adult ' + emj + '.'; } }
        if (stage === "") { if(young) { stage = 'a tiny baby ' + emj + '.'; } else if(humanAge <= 5) { stage = 'a playful toddler ' + emj + '.'; } else if(humanAge <= 12) { stage = 'a curious kid ' + emj + '.'; } else if(humanAge <= 19) { stage = 'a sassy teenager ' + emj + '.'; } else if(humanAge <= 35) { stage = 'a young adult ' + emj + '.'; } else if(humanAge <= 55) { stage = 'a mature adult ' + emj + '.'; } else if(humanAge <= 70) { stage = 'a wise senior ' + emj + '.'; } else { stage = 'a venerable elder ' + emj + '!'; } }
        var funFact = stage;

        // --- Display Results with Animation Trigger ---
        var petTypeName = petType.charAt(0).toUpperCase() + petType.slice(1).replace('_', ' ');
        var summaryText = petTypeName + ' (' + petAge + ' yrs'; if (petType === 'dog') { summaryText += ', ' + dogSize + ' size'; } summaryText += ')';
        summaryInfoSpan.innerHTML = summaryText; funFactSpan.innerHTML = funFact;

        // Use class to trigger CSS animations for results container/boxes

             resultArea.classList.add('visible');

        // Start count-up animation for the number

    if (petTypeSelect.value === 'dog') { dogSizeArea.style.display = 'block'; } else { dogSizeArea.style.display = 'none'; }
     resultArea.classList.remove('visible'); // Ensure results start hidden
