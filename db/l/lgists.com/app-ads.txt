    document.addEventListener('DOMContentLoaded', () => {
        const consentBanner = document.getElementById('consent-banner');
        const consentModal = document.getElementById('consent-modal');
        const consentOverlay = document.getElementById('consent-overlay');
        const acceptAllButton = document.getElementById('accept-all');
        const rejectAllButton = document.getElementById('reject-all');
        const customizeButton = document.getElementById('customize');
        const cancelCustomizeButton = document.getElementById('cancel-customize');
        const consentForm = document.getElementById('consent-form');
        const analyticsCookiesCheckbox = document.getElementById('analytics-cookies');
    
        // Check if consent has already been given
        const consent = localStorage.getItem('cookie-consent');
        if (!consent) {
            consentBanner.classList.add('show');
        } else {
            const preferences = JSON.parse(consent);
            updateGTMConsent(preferences.analytics);
        }
    
        // Update Google Tag Manager consent settings
        function updateGTMConsent(analyticsConsent) {
            window.dataLayer = window.dataLayer || [];
            window.dataLayer.push({
                'event': 'consent_update',
                'analytics_storage': analyticsConsent ? 'granted' : 'denied'
            });
        }
    
        function showModal() {
            consentModal.classList.add('show');
            consentOverlay.classList.add('show');
            consentModal.setAttribute('aria-hidden', 'false');
        }
    
        function hideModal() {
            consentModal.classList.remove('show');
            consentOverlay.classList.remove('show');
            consentModal.setAttribute('aria-hidden', 'true');
            consentBanner.classList.add('show');
        }
    
        function hideBanner() {
            consentBanner.classList.remove('show');
        }
    
        acceptAllButton.addEventListener('click', () => {
            const preferences = { analytics: true };
            localStorage.setItem('cookie-consent', JSON.stringify(preferences));
            updateGTMConsent(true);
            hideBanner();
        });
    
        rejectAllButton.addEventListener('click', () => {
            const preferences = { analytics: false };
            localStorage.setItem('cookie-consent', JSON.stringify(preferences));
            updateGTMConsent(false);
            hideBanner();
        });
    
        customizeButton.addEventListener('click', () => {
            const consent = localStorage.getItem('cookie-consent');
            if (consent) {
                const preferences = JSON.parse(consent);
                analyticsCookiesCheckbox.checked = preferences.analytics;
            }
            consentBanner.classList.remove('show');
            showModal();
        });
    
        cancelCustomizeButton.addEventListener('click', hideModal);
    
        consentForm.addEventListener('submit', (e) => {
            e.preventDefault();
            const preferences = {
                analytics: analyticsCookiesCheckbox.checked
            };
            localStorage.setItem('cookie-consent', JSON.stringify(preferences));
            updateGTMConsent(preferences.analytics);
            hideModal();
            hideBanner();
        });
    
        consentOverlay.addEventListener('click', hideModal);
    
        document.addEventListener('keydown', (e) => {
            if (e.key === 'Escape' && consentModal.classList.contains('show')) {
                hideModal();
            }
        });
    });
    </script>







<script>
document.addEventListener('DOMContentLoaded', () => {
    // Newsletter Form Logic
    console.log('Newsletter script loaded');
    const form = document.querySelector('.newsletter-form');
    // if (!form) {
    //     console.error('Newsletter form not found');
    //     showToast('Form not found. Please check the page.', 'error');
    //     return;
    // }

    console.log('Form HTML:', form.outerHTML);
    console.log('All form inputs:', Array.from(form.querySelectorAll('input')).map(input => ({
        name: input.name,
        value: input.value
    })));

    const emailInput = form.querySelector('input[name="subscriber_email"]');
    const actionInput = form.querySelector('input[name="form_action"]');
    const button = form.querySelector('.newsletter-button');
    const csrfInput = form.querySelector('input[name="csrfmiddlewaretoken"]');
    const newsletterUrl = '/newsletter/subscribe/';

    if (!emailInput || !actionInput || !button || !csrfInput) {
        console.error('Form elements missing:', {
            emailInput: !!emailInput,
            actionInput: !!actionInput,
            button: !!button,
            csrfInput: !!csrfInput
        });
        showToast('Form setup error. Please check the page.', 'error');
        return;
    }

    const emailInputs = document.querySelectorAll('input[name="subscriber_email"]');
    if (emailInputs.length > 1) {
        console.error('Multiple subscriber_email inputs found:', emailInputs.length);
        showToast('Multiple email fields detected. Please fix the form.', 'error');
    }

    emailInput.addEventListener('change', () => {
        console.log('Email input value:', emailInput.value);
    });

    async function checkSubscriptionStatus(email) {
        if (!email || !email.includes('@') || !email.includes('.')) {
            button.textContent = 'Subscribe';
            actionInput.value = 'subscribe';
            return;
        }
        try {
            const body = `subscriber_email=${encodeURIComponent(email)}&form_action=check&csrfmiddlewaretoken=${csrfInput.value}`;
            console.log('Check status body:', body);
            const response = await fetch(newsletterUrl, {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/x-www-form-urlencoded',
                    'X-CSRFToken': csrfInput.value,
                    'X-Requested-With': 'XMLHttpRequest',
                },
                body: body
            });
            const data = await response.json();
            console.log('Check status response:', data);
            button.textContent = data.is_subscribed ? 'Unsubscribe' : 'Subscribe';
            actionInput.value = data.is_subscribed ? 'unsubscribe' : 'subscribe';
        } catch (error) {
            console.error('Error checking subscription:', error);
            showToast('Error checking subscription.', 'error');
        }
    }

    emailInput.addEventListener('input', () => {
        const email = emailInput.value.trim();
        console.log('Email input:', email);
        checkSubscriptionStatus(email);
    });

    form.addEventListener('submit', async (e) => {
        e.preventDefault();
        button.disabled = true;
        button.classList.add('processing');
        const email = emailInput.value.trim();
        console.log('Form submitted with email:', email, 'action:', actionInput.value);

        console.log('Form state before submission:', {
            subscriber_email: emailInput.value,
            form_action: actionInput.value,
            csrfmiddlewaretoken: csrfInput.value
        });

        const formData = new FormData();
        formData.append('subscriber_email', email);
        formData.append('form_action', actionInput.value);
        formData.append('csrfmiddlewaretoken', csrfInput.value);
        console.log('FormData:', Object.fromEntries(formData));

        try {
            const response = await fetch(newsletterUrl, {
                method: 'POST',
                body: formData,
                headers: {
                    'X-CSRFToken': csrfInput.value,
                    'X-Requested-With': 'XMLHttpRequest',
                },
            });

            const data = await response.json();
            console.log('Submission response:', data);
            showToast(data.message, data.status);

            if (data.status === 'success') {
                button.textContent = data.action === 'subscribe' ? 'Unsubscribe' : 'Subscribe';
                actionInput.value = data.action === 'subscribe' ? 'unsubscribe' : 'subscribe';
                if (data.action === 'subscribe') {
                    form.reset();
                }
            }
        } catch (error) {
            console.error('Submission error:', error);
            showToast('An error occurred. Please try again.', 'error');
        } finally {
            button.disabled = false;
            button.classList.remove('processing');
        }
    });

    function showToast(message, status) {
        console.log('Showing toast:', message, status);
        const existingToast = document.querySelector('#notification');
        if (existingToast) existingToast.remove();

        const toast = document.createElement('div');
        toast.id = 'notification';
        toast.className = `toast ${status}`;
        toast.textContent = message;
        document.body.appendChild(toast);

        setTimeout(() => {
            toast.classList.add('show');
        }, 100);

        setTimeout(() => {
            toast.classList.remove('show');
            setTimeout(() => toast.remove(), 300);
        }, 5000);
    }

    // Mobile Menu Logic
    const hamburger = document.querySelector('.hamburger');
    const navMenu = document.querySelector('.nav-menu');
    const mobileOverlay = document.querySelector('.mobile-overlay');
    const mobileCloseBtn = document.querySelector('.mobile-close-btn');
    const body = document.body;

    function toggleMobileMenu() {
        const isActive = navMenu.classList.contains('active');
        if (!isActive) {
            navMenu.classList.add('active');
            hamburger.classList.add('active');
            mobileOverlay.classList.add('active');
            mobileCloseBtn.classList.add('active');
            body.style.overflow = 'hidden';
        } else {
            navMenu.classList.remove('active');
            hamburger.classList.remove('active');
            mobileOverlay.classList.remove('active');
            mobileCloseBtn.classList.remove('active');
            body.style.overflow = '';
        }
    }

    if (hamburger && navMenu && mobileOverlay && mobileCloseBtn) {
        hamburger.addEventListener('click', toggleMobileMenu);
        mobileCloseBtn.addEventListener('click', toggleMobileMenu);
        mobileOverlay.addEventListener('click', toggleMobileMenu);

        navMenu.querySelectorAll('a').forEach(link => {
            link.addEventListener('click', () => {
                navMenu.classList.remove('active');
                hamburger.classList.remove('active');
                mobileOverlay.classList.remove('active');
                mobileCloseBtn.classList.remove('active');
                body.style.overflow = '';
            });
        });
    }

    window.addEventListener('resize', () => {
        if (window.innerWidth > 768 && navMenu) {
            navMenu.classList.remove('active');
            hamburger.classList.remove('active');
            mobileOverlay.classList.remove('active');
            mobileCloseBtn.classList.remove('active');
            body.style.overflow = '';
        }
    });

    // Smooth Scroll for Anchor Links
    document.querySelectorAll('a[href^="#"]').forEach(anchor => {
        anchor.addEventListener('click', function (e) {
            if (navMenu && !navMenu.classList.contains('active')) {
                e.preventDefault();
                const target = document.querySelector(this.getAttribute('href'));
                if (target) {
                    target.scrollIntoView({
                        behavior: 'smooth',
                        block: 'start'
                    });
                }
            }
        });
    });

    // Lazy Loading Images
    if ('IntersectionObserver' in window) {
        const imageObserver = new IntersectionObserver((entries, observer) => {
            entries.forEach(entry => {
                if (entry.isIntersecting) {
                    const img = entry.target;
                    img.src = img.dataset.src;
                    img.classList.remove('lazy');
                    imageObserver.unobserve(img);
                }
            });
        });

        document.querySelectorAll('img.lazy').forEach(img => {
            imageObserver.observe(img);
        });
    }
});
</script>
