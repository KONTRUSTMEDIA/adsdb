  <script>
  (function(){
    var MIN_W = 1024;
    var ITEM_SEL = '.posts__item';
    // Selettori “intelligenti” per titolo e descrizione (adattabili al tema)
    var TITLE_SEL = '.posts__title, .entry-title, h2, h3';
    var DESC_SEL  = '.posts__excerpt, .entry-summary, .excerpt, .post__excerpt, .card-excerpt';

    function $all(s, c){ return (c||document).querySelectorAll(s); }
    function $(s, c){ return (c||document).querySelector(s); }
    function debounce(fn, d){ var t; return function(){ clearTimeout(t); t = setTimeout(fn, d); }; }

    // Crea (una volta) un wrapper che contiene titolo + descrizione
    function ensureCenterBlock(item){
      if (item.querySelector('.posts__centerblock')) return item.querySelector('.posts__centerblock');

      var title = item.querySelector(TITLE_SEL);
      var desc  = item.querySelector(DESC_SEL);

      if (!title && !desc) return null;

      // wrapper "neutro"
      var wrap = document.createElement('div');
      wrap.className = 'posts__centerblock';

      // Inseriamo wrap nel punto dove si trova il titolo (o prima del desc)
      var anchor = title || desc;
      if (!anchor || !anchor.parentNode) return null;

      anchor.parentNode.insertBefore(wrap, anchor);
      if (title) wrap.appendChild(title);
      if (desc)  wrap.appendChild(desc);

      return wrap;
    }

    // Equalizza PER RIGA (stesso offsetTop) per evitare righe vuote
    function equalizePerRow(scope){
      var items = $all(ITEM_SEL, scope);
      if (!items.length) return;

      // reset
      items.forEach(function(el){
        el.style.height = 'auto';
        el.style.minHeight = '0';
        el.classList.remove('is-short-text'); // pulizia
      });

      if (window.innerWidth < MIN_W) return;

      // Raggruppa per parent
      var byParent = new Map();
      items.forEach(function(el){
        var p = el.parentElement || document.body;
        if (!byParent.has(p)) byParent.set(p, []);
        byParent.get(p).push(el);
      });

      byParent.forEach(function(group){
        // Raggruppa per offsetTop = riga visiva
        var rows = new Map();
        group.forEach(function(el){
          var top = el.offsetTop;
          if (!rows.has(top)) rows.set(top, []);
          rows.get(top).push(el);
        });

        // Per ogni riga: imposta la stessa altezza e centra l'item col testo minore
        rows.forEach(function(rowItems){
          // 1) altezza uguale per riga
          var maxH = 0;
          rowItems.forEach(function(el){
            el.style.height = 'auto';
            var h = el.getBoundingClientRect().height;
            if (h > maxH) maxH = h;
          });
          var hpx = Math.ceil(maxH) + 'px';
          rowItems.forEach(function(el){ el.style.height = hpx; });

          // 2) trova l'item col minor "blocco testo" (titolo+descrizione) e centrarlo
          var minTextH = Infinity, candidates = [];
          rowItems.forEach(function(el){
            var wrap = ensureCenterBlock(el);
            // Se il tema non usa questi selettori, wrap può essere nullo → saltiamo
            if (!wrap) return;
            // reset auto prima di misurare
            wrap.style.marginTop = '';
            wrap.style.marginBottom = '';

            var h = wrap.getBoundingClientRect().height;
            if (h < minTextH){
              minTextH = h;
              candidates = [el];
            } else if (h === minTextH){
              candidates.push(el); // in caso di pari merito
            }
          });

          // rimuovi la classe da tutti, poi aggiungi a chi ha meno testo
          rowItems.forEach(function(el){ el.classList.remove('is-short-text'); });
          candidates.forEach(function(el){ el.classList.add('is-short-text'); });
        });
      });
    }

    function rerun(){
      equalizePerRow(document.body);
    }

    function init(){
      // primo run
      rerun();

      // immagini (anche lazy)

      // MutationObserver: se entrano contenuti (anche se NON ads in .posts__item ora esclusi) ricalcola
      if ('MutationObserver' in window) {

        setTimeout(rerun, 2000);
      });
