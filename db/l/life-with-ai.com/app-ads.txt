        function loadInputsFromUrl() {
            const urlParams = new URLSearchParams(window.location.search);

            $('input, select, textarea').each(function () {
                const inputId = $(this).attr('id');
                const urlParamValue = urlParams.get(inputId);

                if (urlParamValue !== null) {
                    if ($(this).is(':checkbox')) {
                        // Set checkbox state
                        const isChecked = urlParamValue === 'true';
                        $(this).prop('checked', isChecked).trigger('change');
                    } else if ($(this).is(':radio')) {
                        // Set the radio button state
                        if (urlParamValue === 'on' && $(this).val() === 'on') {
                            $(this).prop('checked', true).trigger('change');
                        }
                    } else {
                        // Set the value for other input types
                        $(this).val(urlParamValue).trigger('change');
                    }
                }
            });
        }

        function saveInputsToUrl(goToNewUrl = true) {
            const urlParams = new URLSearchParams();
            urlParams.set("lan", "en")

            $('input, select, textarea').each(function () {
                const inputId = $(this).attr('id');

                if ($(this).is(':checkbox')) {
                    // Save checkbox state (true or false)
                    const isChecked = $(this).is(':checked');
                    urlParams.set(inputId, isChecked);
                } else if ($(this).is(':radio')) {
                    // Save radio button state
                    if ($(this).is(':checked')) {
                        urlParams.set(inputId, $(this).val());
                    }
                } else {
                    // Save the value for other input types
                    const inputValue = $(this).val();
                    urlParams.set(inputId, inputValue);
                }
            });

            const newUrl = `${window.location.pathname}?${urlParams.toString()}`;
            if (goToNewUrl)
                history.pushState({}, '', newUrl);
            else
                return newUrl
        }
