    document.addEventListener('DOMContentLoaded', function() {
        const faqItems = document.querySelectorAll('.faq-question');
        
        faqItems.forEach(question => {
            question.addEventListener('click', () => {
                const parentItem = question.parentElement;
                const answer = parentItem.querySelector('.faq-answer');
                const plusSign = question.querySelector('span');
                
                // Check if this item is already active
                const isActive = parentItem.classList.contains('active');
                
                // Close all items
                document.querySelectorAll('[style*="background: rgba(255, 255, 255, 0.1)"]').forEach(item => {
                    item.classList.remove('active');
                    item.querySelector('.faq-answer').style.maxHeight = '0';
                    item.querySelector('.faq-answer').style.padding = '0 35px';
                    item.querySelector('.faq-question span').textContent = '+';
                    item.querySelector('.faq-question span').style.transform = 'rotate(0deg)';
                    item.querySelector('.faq-question span').style.background = 'rgba(255, 255, 255, 0.2)';
                    item.querySelector('.faq-question span').style.color = '#2c8fb3';
                    item.querySelector('.faq-question span').style.border = '2px solid #87ceeb';
                });
                
                // If this item wasn't active, open it
                if (!isActive) {
                    parentItem.classList.add('active');
                    answer.style.maxHeight = answer.scrollHeight + 'px';
                    answer.style.padding = '30px 35px';
                    plusSign.textContent = '−';
                    plusSign.style.transform = 'rotate(180deg)';
                    plusSign.style.background = 'linear-gradient(135deg, #87ceeb, #2c8fb3)';
                    plusSign.style.color = '#fff';
                    plusSign.style.border = '2px solid transparent';
                }
            });
        });
        
        // Add hover effects
        document.querySelectorAll('[style*="background: rgba(255, 255, 255, 0.1)"]').forEach(item => {
            item.addEventListener('mouseenter', function() {
                this.style.transform = 'translateX(10px)';
                this.style.boxShadow = '0 12px 35px rgba(135, 206, 235, 0.3)';
                this.style.border = '1px solid rgba(135, 206, 235, 0.8)';
            });
            
            item.addEventListener('mouseleave', function() {
                if (!this.classList.contains('active')) {
                    this.style.transform = 'translateX(0)';
                    this.style.boxShadow = '0 5px 20px rgba(135, 206, 235, 0.15)';
                    this.style.border = '1px solid rgba(173, 216, 230, 0.5)';
                }
            });
        });
        
        // Add hover effect to questions
        document.querySelectorAll('.faq-question').forEach(question => {
            question.addEventListener('mouseenter', function() {
                if (!this.parentElement.classList.contains('active')) {
                    this.style.color = '#1e3a5f';
                }
            });
            
            question.addEventListener('mouseleave', function() {
                if (!this.parentElement.classList.contains('active')) {
                    this.style.color = '#070707';
                }
            });
        });
    });
