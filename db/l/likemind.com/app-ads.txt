$(document).ready(function() {

  let splitType = new SplitType(".animate_text", {
    types: "words, chars",
    tagName: "span"
  });

  $(".animate_text_wrap").each(function (index) {
    const wrapper = this;
    const heading = $(this).find(".animate_text");

    // Add a flag to prevent double execution
    if ($(this).data('animated')) return;
    $(this).data('animated', true);

    // Set text to invisible by default
    gsap.set(heading.find(".char"), { y: "1em", opacity: 0 });

    // Create timeline that animates in once
    let tl = gsap.timeline({ 
      paused: true,
      repeat: 0 // Explicitly set to not repeat
    });

    tl.to(heading.find(".char"), {
      y: 0, 
      opacity: 1, 
      stagger: { amount: 0.3 },
      ease: "power2.out"
    });

    // Create ScrollTrigger
    ScrollTrigger.create({
      trigger: wrapper,
      start: "top 80%",
      once: true, // This should prevent multiple triggers
      onEnter: () => {
        tl.restart(); // Use restart instead of play to ensure it starts from beginning
      }
    });
  });
});
