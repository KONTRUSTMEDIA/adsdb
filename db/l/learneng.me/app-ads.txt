async function findTeacherRecord(user) {
  const uid = user.uid;
  const email = (user.email || "").toLowerCase().trim();

  // 1) users (uid)
  const usersDoc = await db.collection("users").doc(uid).get();
  if (usersDoc.exists) return { source: "users", doc: usersDoc, data: usersDoc.data() };

  // 2) appTeachers (uid)
  const appDoc = await db.collection("appTeachers").doc(uid).get();
  if (appDoc.exists) return { source: "appTeachers", doc: appDoc, data: appDoc.data() };

  // 3) donorTeachers (uid) (rare but supported)
  const donorUidDoc = await db.collection("donorTeachers").doc(uid).get();
  if (donorUidDoc.exists) return { source: "donorTeachers", doc: donorUidDoc, data: donorUidDoc.data() };

  // 4) donorTeachers (email legacy) ✅ your random-id donor docs
  if (email) {
    const q = await db.collection("donorTeachers").where("email", "==", email).limit(1).get();
    if (!q.empty) return { source: "donorTeachers", doc: q.docs[0], data: q.docs[0].data() };
  }

  // 5) appTeachers (email fallback - optional safety)
  if (email) {
    const q2 = await db.collection("appTeachers").where("email", "==", email).limit(1).get();
    if (!q2.empty) return { source: "appTeachers", doc: q2.docs[0], data: q2.docs[0].data() };
  }

  return null;
}

function computeRoleFromRecord(found) {
  // defaults
  let roleLabel = "Free Teacher";
  let roleColor = "#999";
  let userRole = "free";
  let adFree = false;

  if (!found) return { roleLabel, roleColor, userRole, adFree };

  // ✅ PLUS: users paid
  if (found.source === "users" && found.data?.paid === true) {
    roleLabel = "Plus Teacher 🚀";
    roleColor = "#667eea";
    userRole = "plus";
    adFree = true;
    return { roleLabel, roleColor, userRole, adFree };
  }

  // ✅ PRO/APPS: appTeachers paid => plus yetkisi (senin mantık)
  if (found.source === "appTeachers" && found.data?.paid === true) {
    roleLabel = "PRO Member 🚀";
    roleColor = "#667eea";
    userRole = "plus";
    adFree = true;
    return { roleLabel, roleColor, userRole, adFree };
  }

  // ✅ DONOR: only if paid true (admin onaylı donor)
  // (Eğer paid:false iken "donor pending" istiyorsan söyle, 1 satırla değiştiririm.)
  if (found.source === "donorTeachers" && found.data?.paid === true) {
    roleLabel = "Donor Teacher 🎁";
    roleColor = "#009688";
    userRole = "donor";
    adFree = true;
    return { roleLabel, roleColor, userRole, adFree };
  }

  return { roleLabel, roleColor, userRole, adFree };
}

async function refreshStudentProfile() {
  if (!AUTH_USER) return;

  // Loading placeholders
  const setTxt = (id, v) => { const el = document.getElementById(id); if (el) el.textContent = v; };
  setTxt("stuClassName", "Loading...");
  setTxt("stuTeacherLine", "Loading...");

  try {
    // 1) Student doc
    const sDoc = await db.collection("students").doc(AUTH_USER.uid).get();
    if (!sDoc.exists) {
      setTxt("stuFullName", "Student");
      setTxt("stuStars", "0");
      setTxt("stuNo", "—");
      setTxt("stuClassCode", "—");
      setTxt("stuClassName", "No class info found");
      setTxt("stuTeacherLine", "");
      return;
    }

    const s = sDoc.data() || {};
    const fullName = `${(s.name || "Student").trim()} ${(s.surname || "").trim()}`.trim();

    // Stars safe parse
    let stars = 0;
    if (s.stars !== undefined && s.stars !== null) {
      const n = parseInt(s.stars, 10);
      stars = isNaN(n) ? 0 : n;
    }

    setTxt("stuFullName", fullName || "Student");
    setTxt("stuStars", String(stars));
    setTxt("stuNo", s.stuNo || "—");
    setTxt("stuClassCode", s.classCode || "—");

    // 2) Resolve class (best: classId; fallback: classCode)
    let classData = null;
    let classId = s.classId || null;

    if (classId) {
      const cDoc = await db.collection("classes").doc(classId).get();
      if (cDoc.exists) classData = { id: cDoc.id, ...cDoc.data() };
    }

    if (!classData && s.classCode) {
      const q = await db.collection("classes").where("classCode", "==", s.classCode).limit(1).get();
      if (!q.empty) {
        classData = { id: q.docs[0].id, ...q.docs[0].data() };
        classId = q.docs[0].id;
      }
    }

    // Update class lines
    if (classData) {
      setTxt("stuClassName", classData.name || "Unnamed Class");
      const tId = classData.teacherId || s.teacherId || "";
      setTxt("stuTeacherLine", tId ? `Teacher: ${maskUid(tId)}` : "");
    } else {
      setTxt("stuClassName", "Class not found");

async function openUserProfile() {
  if (!AUTH_USER) return;

  document.getElementById('userProfileModal').style.display = 'flex';
  document.getElementById('profileEmail').innerText = AUTH_USER.email || "-";
  document.getElementById('profileUid').innerText = AUTH_USER.uid;

  try {
    let totalClasses = 0;
    let totalStudents = 0;

    // ✅ Plan label (PRO yok)
    const role = localStorage.getItem("userRole") || "free";
    const adFree = localStorage.getItem("adFree") === "true";

    let planName = "Free Teacher";
    if (role === "plus") planName = "LearnEng Plus Member 💎";
    else if (role === "donor") planName = "Donor Teacher 🎁";
    else planName = adFree ? "Member (Ad-free)" : "Free Teacher";

    // ✅ Member since (users doc varsa al, yoksa auth creation time)
    let memberSince = AUTH_USER.metadata?.creationTime ? new Date(AUTH_USER.metadata.creationTime) : new Date();
    const usersDoc = await db.collection("users").doc(AUTH_USER.uid).get();

if (usersDoc.exists) {
  const d = usersDoc.data();
  if (d.createdAt?.toDate) memberSince = d.createdAt.toDate();
}

// ✅ Admin button visibility
const adminBtn = document.getElementById("adminPanelBtn");
if (adminBtn) {
  const isAdmin = usersDoc.exists && usersDoc.data()?.admin === true;
  adminBtn.style.display = isAdmin ? "flex" : "none";
}


    // ✅ Expiry info
    let expiryText = "";
    if (role === "plus") expiryText = "Unlimited access";
    else if (role === "donor") expiryText = adFree ? "Ad-free active" : "Pending / Not active";
    else expiryText = "";

    // ✅ Classes: teacherId == uid
    const classSnap = await db.collection("classes")
      .where("teacherId", "==", AUTH_USER.uid)
      .get();

    totalClasses = classSnap.size;

    // ✅ Collect classCodes (skip empty)
    const classCodes = classSnap.docs
      .map(d => (d.data().classCode || "").trim())
      .filter(Boolean);

    // ✅ Students: teacherId == uid AND classCode IN [...]
    if (classCodes.length > 0) {
      const chunks = chunkArray(classCodes, 10); // safe for Firestore 'in'
      let count = 0;

      for (const part of chunks) {
        const stuSnap = await db.collection("students")
          .where("teacherId", "==", AUTH_USER.uid)
          .where("classCode", "in", part)
          .get();
        count += stuSnap.size;
      }

      totalStudents = count;
    } else {
      // teacher has no classes => show 0
      totalStudents = 0;
    }

    // ✅ UI Update
    document.getElementById('profilePlanBadge').innerText = planName;
    document.getElementById('profileExpiryInfo').innerText = expiryText;
    document.getElementById('statClasses').innerText = totalClasses;
    document.getElementById('statStudents').innerText = totalStudents;

    const days = Math.ceil(Math.abs(new Date() - memberSince) / (1000 * 60 * 60 * 24));
    document.getElementById('statDays').innerText = days;

  } catch (e) {
    console.error("Profile load error", e);
  }
}

// ✅ helper
function chunkArray(arr, size) {
  const out = [];
  for (let i = 0; i < arr.length; i += size) out.push(arr.slice(i, i + size));
  return out;
}

      <div style="grid-column:1/-1; padding:40px; text-align:center; color:#078084;">
        ⏳ Loading...
      </div>
    </div>
  `;

  // ✅ ACCESS GATE (senin kuralın)
  const access = resolveClassesAccessFast();
  const grid = document.getElementById("classesGrid");
  const newBtn = document.getElementById("newClassBtn");

  // 1) Student ise hiç gösterme
  if (access.mode === "blocked") {
    if (grid) grid.innerHTML = `
      <div style="grid-column:1/-1; padding:30px; text-align:center; background:#fff; border-radius:16px; border:1px solid #eee;">
        <div style="font-size:46px;">🔒</div>
        <h3 style="margin:10px 0 0; color:#334155;">Students cannot access this section</h3>
        <p style="margin:6px 0 0; color:#64748b;">Please use the student panel.</p>
      </div>
    `;
    if (newBtn) newBtn.style.display = "none";
    return;
  }

  // 2) Demo ise demo göster
  if (access.mode === "demo") {
    if (grid) grid.innerHTML = renderDemoClassesCards();
    if (newBtn) newBtn.style.display = "none";
    return;
  }

  // 3) Real ise gerçek sınıfları yükle (sadece users paid:true + appTeachers paid:true)
  const user = firebase.auth().currentUser;
  if (!user) {
    // teorik olarak buraya düşmez ama güvenlik için
    if (grid) grid.innerHTML = renderDemoClassesCards();
    if (newBtn) newBtn.style.display = "none";
    return;
  }

  try {
    const classesSnap = await db.collection("classes")
      .where("teacherId", "==", user.uid)
      .orderBy("name")
      .get();

    grid.innerHTML = "";

    if (classesSnap.empty) {
      grid.innerHTML = `
        <div style="grid-column: 1/-1; text-align: center; padding: 60px; background: #fff; border-radius: 16px; border: 2px dashed #e0e0e0;">
          <div style="font-size: 50px; margin-bottom: 15px;">📭</div>
          <h3 style="color: #555; margin:0;">No classes yet</h3>
          <p style="color:#888; margin-bottom:20px;">Create your first class to get started.</p>
          <button onclick="openCreateClassModal()" class="donor-action-btn" style="width:auto;">Create Class</button>
        </div>
      `;
      return;
    }

    const cardsHtml = await Promise.all(classesSnap.docs.map(async (doc) => {
      const cls = doc.data();
      const classId = doc.id;

      const safeName = cls.name ? cls.name.replace(/'/g, "\\'") : "Class";
      const safeCode = (cls.classCode || "").trim();

      let studentSnap = null;
      if (safeCode) {
        studentSnap = await db.collection("students")
          .where("teacherId", "==", user.uid)
          .where("classCode", "==", safeCode)
          .get();
      }

      let classStudents = [];
      if (studentSnap) {
        studentSnap.forEach((sDoc) => {
          const s = sDoc.data();
          let starCount = parseInt(s.stars ?? 0, 10);
          if (isNaN(starCount)) starCount = 0;
          const firstName = ((s.name || "Student").trim().split(" ")[0]) || "Student";
          classStudents.push({ firstName, stars: starCount });
        });
      }

      classStudents.sort((a,b) => b.stars - a.stars);
      const top3 = classStudents.slice(0,3);
      const remainingCount = Math.max(0, classStudents.length - 3);

      let listHtml = "";
      if (top3.length > 0) {
        listHtml = `<div class="cc-top-list">`;
        top3.forEach((s, idx) => {
          let icon = idx===0 ? "🥇" : idx===1 ? "🥈" : "🥉";
          listHtml += `
            <div class="cc-student-row">
              <span style="color:#444; font-weight:500;">${icon} ${s.firstName}</span>
              <span style="font-weight:700; color:#f59e0b; font-size:0.9rem;">⭐ ${s.stars}</span>
            </div>
          `;
        });
        listHtml += `</div>`;
        if (remainingCount > 0) {
          listHtml += `<div style="font-size:0.8rem; color:#94a3b8; margin-top:-5px; text-align:right;">+ ${remainingCount} more students</div>`;
        }
      } else {
        listHtml = `<div style="text-align:center; padding:20px 0; color:#cbd5e1; font-style:italic;">No students visible</div>`;
      }

      return `
        <div class="class-card-modern">
          <div class="cc-body">
            <div class="cc-header-row">
              <span class="cc-emoji">🏫</span>
              <div class="cc-title" title="${cls.name || "Class"}">${cls.name || "Class"}</div>

              <button class="cc-link-btn" title="Create student join link"
                onclick="openJoinLinkModal('${safeName}', '${safeCode}')">🔗</button>
            </div>

            <div class="cc-code-row">
              <span class="cc-code-badge" title="Class Code">${safeCode || "-"}</span>
            </div>

            ${listHtml}
          </div>

          <div class="cc-footer">
            <button class="cc-view-btn" onclick="openClassDetails('${classId}', '${safeName}', '${safeCode}')">View Whole Class</button>
            <button class="cc-settings-btn" onclick="openClassActions('${classId}', '${safeName}', '${safeCode}')">⚙️Class Actions</button>
          </div>
        </div>
      `;
    }));

    grid.innerHTML = cardsHtml.join("");

  } catch (err) {
    console.error("Classes Load Error:", err);
    if (grid) grid.innerHTML = `
      <div style="grid-column:1/-1; padding:20px; color:red; text-align:center;">
        ⚠️ Error loading classes.<br><small>${err.message}</small>
      </div>`;
  }
}

async function fetchPrintables() {
  folderTree.printables.folders = {};
  folderTree.printables.items = [];
  ROOT_FOLDERS.forEach(({ path, label }) => {
    folderTree.printables.folders[label] = {
      _storagePath: path,
      _loaded: false,
      folders: {},
      items: []
    };
  });
}

function insertToFolderTree(tree, pathArray, item) {
  let node = tree;
  pathArray.forEach(part => {
    if (!node.folders) node.folders = {};
    if (!node.folders[part]) node.folders[part] = { folders: {}, items: [] };
    node = node.folders[part];
  });
  if (!node.items) node.items = [];
  node.items.push(item);
}

async function fetchStorageContents(node) {
  try {
    const res = await firebase.storage().ref(node._storagePath).listAll();

    // --- KLASÖRLER (Aynı kalıyor) ---
    res.prefixes.forEach(f => {
      if (!node.folders[f.name]) {
        node.folders[f.name] = {
          _storagePath: f.fullPath,
          _loaded: false,
          folders: {},
          items: []
        };
      }
    });

    // --- DOSYALAR (KRİTİK GÜNCELLEME BURADA) ---
    
    // 1. Storage'dan gelen dosyaları hazırla
    const storageFiles = res.items.map(itemRef => ({
      title: itemRef.name,
      url: null, // Tıklandığında URL alınacak
      storagePath: itemRef.fullPath,
      isBuilder: false // Bunlar normal dosya
    }));

    // 2. Klasörde ZATEN VAR OLAN Builder içeriklerini (isBuilder: true) sakla!
    // Eğer bunu yapmazsak, klasör yüklenince Builder oyunları silinir.
    const existingBuilderItems = (node.items || []).filter(item => item.isBuilder);

    // 3. İki listeyi BİRLEŞTİRip klasöre koy
    node.items = [...existingBuilderItems, ...storageFiles];

    node._loaded = true;

  } catch (e) {
    console.error("fetchStorageContents error:", e);
  }
}

async function isVerifiedTeacher(user) {
  if (!user) return false;

  // ⛔ class-code öğrenci (local access)
 if (localStorage.getItem("localAccessRole") === "student") return false;

  // ⛔ students collection = student
  try {
    const s = await db.collection("students").doc(user.uid).get();
    if (s.exists) return false;
  } catch (e) {
    // sessiz geç
  }

  const uid = user.uid;
  const email = (user.email || "").toLowerCase().trim();

  // 1) users
  try {
    const u = await db.collection("users").doc(uid).get();
    if (u.exists) return true;
  } catch (e) {}

  // 2) appTeachers
  try {
    const a = await db.collection("appTeachers").doc(uid).get();
    if (a.exists) return true;
  } catch (e) {}

  // 3) donorTeachers (uid)
  try {
    const d1 = await db.collection("donorTeachers").doc(uid).get();
    if (d1.exists) return true;
  } catch (e) {}

  // 4) donorTeachers (email fallback legacy)
  if (email) {
    try {
      const q = await db.collection("donorTeachers").where("email", "==", email).limit(1).get();
      if (!q.empty) return true;
    } catch (e) {}
  }

  // (optional) appTeachers email fallback – bazı legacy durumlar için
  if (email) {
    try {
      const q2 = await db.collection("appTeachers").where("email", "==", email).limit(1).get();
      if (!q2.empty) return true;
    } catch (e) {}
  }

  return false;
}

// 3. Update Star Count (Real-time & Optimistic)
async function updateStar(studentId, change) {
    const label = document.getElementById(`star-val-${studentId}`);
    if(!label) return;

    let current = parseInt(label.innerText);
    let newVal = current + change;
    
    if(newVal < 0) return; // Prevent negative stars

    // 1. Optimistic Update (Change UI immediately for speed)
    label.innerText = newVal;

    try {
        // 2. Update Database
        await db.collection("students").doc(studentId).update({
            stars: firebase.firestore.FieldValue.increment(change)
        });
    } catch (err) {
        console.error("Star update failed", err);
        // Revert UI if failed
        label.innerText = current; 
        alert("Connection error: Could not update star.");
    }
}

// 3. E-Posta ve Şifre Oluşturucu (Türkçe Karakter Temizleme)
async function generateCredentialFor(sid, name, no) {
    if(!confirm(`Create/Reset login for ${name}?`)) return;

    // Türkçe karakter haritası
    const trMap = {
        'ç': 'c', 'Ç': 'c', 'ğ': 'g', 'Ğ': 'g', 'ı': 'i', 'I': 'i', 'İ': 'i', 'ö': 'o', 'Ö': 'o', 'ş': 's', 'Ş': 's', 'ü': 'u', 'Ü': 'u', ' ': ''
    };

    // İsmi temizle (Küçük harf, Türkçe karakter yok, boşluk yok)
    let cleanName = name.toLowerCase().split(' ')[0]; // Sadece ilk ad
    cleanName = cleanName.replace(/[çğışüöÇĞİŞÜÖ ]/g, function(s) { return trMap[s] || s; });

// 4. Toplu Oluşturma (GÜVENLİ VERSİYON)
async function generateAllCredentials() {
    const list = document.getElementById("studentManagerList");
    
    // 1. Önce sınıfın güncel listesini çekelim
    const snapshot = await db.collection("students")
                             .where("classId", "==", currentManageClassId)
                             .get();

    if (snapshot.empty) return alert("No students in this class.");

    // 2. Hesabı OLMAYANLARI tespit et (Filtreleme)
    const studentsNeedAccount = [];
    
    snapshot.forEach(doc => {
        const s = doc.data();
        // E-postası VEYA şifresi olmayanları listeye al
        if (!s.email || !s.generatedPassword) {
            studentsNeedAccount.push({ id: doc.id, ...s });
        }
    });

    // 3. Eğer herkesin hesabı varsa işlemi durdur
    if (studentsNeedAccount.length === 0) {
        return alert("✅ All students in this class already have login credentials. No action needed.");
    }

    // 4. Kullanıcıdan Onay Al (Sayı belirterek)
    const confirmMsg = `Found ${studentsNeedAccount.length} students without login info.\n\nGenerate credentials for these ${studentsNeedAccount.length} students?\n(Students who already have accounts will NOT be affected.)`;
    
    if (!confirm(confirmMsg)) return;

    // 5. İşlemi Başlat
    // Kullanıcıya işlemin sürdüğünü gösterelim
    const oldListHTML = list.innerHTML;
    list.innerHTML = `<div style="text-align:center; padding:40px;">
                        <h3>⚡ Generating...</h3>
                        <p>Creating accounts for ${studentsNeedAccount.length} students.</p>
                        <p>Please wait, do not close this window.</p>
                      </div>`;

    let successCount = 0;

    // Tek tek hepsine istek at (Promise.all kullanmıyoruz, sunucuyu yormamak için sırayla yapıyoruz)
    for (const s of studentsNeedAccount) {
        try {
            console.log(`Generating for: ${s.name}...`);
            
            // Cloud Function'a göndermek için gerekli verileri hazırla
            // (saveNewStudentSimple içindeki mantığın aynısını burada kullanıyoruz)
            const trMap = {'ç':'c','Ç':'c','ğ':'g','Ğ':'g','ı':'i','I':'i','İ':'i','ö':'o','Ö':'o','ş':'s','Ş':'s','ü':'u','Ü':'u',' ':'','-':''};
            
            const cleanName = s.name.toLowerCase().replace(/[çğışüöÇĞİŞÜÖ \-]/g, char => trMap[char]||char);
            const cleanSurname = (s.surname || "").toLowerCase().replace(/[çğışüöÇĞİŞÜÖ \-]/g, char => trMap[char]||char);
            
            // Eğer soyadı yoksa "student" kullan
            const finalSurname = cleanSurname || "student";
            
            const email = `${cleanName}${s.stuNo}@learneng.me`;
            const password = `${finalSurname}${s.stuNo}`;
            const user = firebase.auth().currentUser;

            // Cloud Function Çağrısı
            const resp = await fetch("https://europe-west1-learneng-plus.cloudfunctions.net/registerStudent", {
                method: "POST",
                headers: { "Content-Type": "application/json" },
                body: JSON.stringify({ 
                    name: s.name, 
                    surname: s.surname, 
                    stuNo: s.stuNo, 
                    email, 
                    password, 
                    classId: currentManageClassId, 
                    teacherId: user.uid, 
                    classCode: currentManageClassCode 
                })
            });

            if (result.error) throw new Error(result.error);

            // Hata olsa bile döngü devam etsin, diğerlerini yakmasın
        }
    }

    alert(`✅ Process Completed!\nSuccessfully generated credentials for ${successCount} students.`);

// 3. Metni Satır Satır Analiz Et (E-OKUL ALGORİTMASI)
function parseOCRToList(text) {
    parsedStudents = [];
    const lines = text.split('\n');

    lines.forEach(line => {
    let cleanLine = line.replace(/[^a-zA-Z0-9çğıöşüÇĞİÖŞÜ\s]/g, "").trim();
    if (cleanLine.length < 3) return;

    let parts = cleanLine.split(/\s+/);
    if (parts.length < 2) return; // En az No ve Ad olmalı

    // Başlangıç analiz
    const isFirstNum = /^\d+$/.test(parts[0]);
    const isSecondNum = /^\d+$/.test(parts[1]);

    let studentNo = "";
    let nameRaw = "";
    let surnameRaw = "";
    let genderRaw = "";

    // Cinsiyet Bulma (Sondan bak)
    const lastPart = parts[parts.length - 1].toUpperCase();
    let hasGender = false;
    
    if (['K', 'KIZ', 'G'].includes(lastPart)) { genderRaw = "girl"; hasGender = true; }
    else if (['E', 'ERKEK', 'B'].includes(lastPart)) { genderRaw = "boy"; hasGender = true; }

    const endIndex = hasGender ? parts.length - 2 : parts.length - 1;

    // SENARYO A: E-Okul (İki sayı ile başlar: 1 209 ...)
    if (isFirstNum && isSecondNum) {
        studentNo = parts[1]; // İkincisi okul nosudur
        surnameRaw = parts[endIndex];
        nameRaw = parts.slice(2, endIndex).join(" ");
    }
    // SENARYO B: Manuel Giriş (Tek sayı ile başlar: 209 ALİ ...)
    else if (isFirstNum && !isSecondNum) {
        studentNo = parts[0]; // İlk sayı okul nosudur
        surnameRaw = parts[endIndex];
        // Ad, 1. indexten (numaradan sonra) soyada kadar olan kısım
        nameRaw = parts.slice(1, endIndex).join(" ");
    }
    else {
        return; // Sayı ile başlamayan satırları atla (Başlık vs olabilir)
    }

    // Listeye Ekle
    if (studentNo && nameRaw && surnameRaw) {
        parsedStudents.push({
            no: studentNo,
            name: titleCase(nameRaw),
            surname: titleCase(surnameRaw),
            gender: genderRaw,
            valid: true
        });
    }
});

    renderOCRTable();
}

// 4. Tabloyu Çiz (Cinsiyet Kolonu Eklendi)
function renderOCRTable() {
    document.getElementById("ocrLoading").style.display = "none";
    document.getElementById("ocrResultArea").style.display = "block";
    const tbody = document.getElementById("ocrTableBody");
    tbody.innerHTML = "";

    if (parsedStudents.length === 0) {
        tbody.innerHTML = "<tr><td colspan='6' style='text-align:center; padding:20px; color:red;'>Hiç öğrenci bulunamadı.<br>Listenin net ve 'Sıra No - No - Ad - Soyad' formatında olduğundan emin olun.</td></tr>";
        document.getElementById("saveOCRBtn").disabled = true;
        return;
    }

    parsedStudents.forEach((stu, index) => {
        // Çakışma Kontrolü
        let isDuplicate = existingStudentNumbers.includes(stu.no);
        let rowClass = isDuplicate ? "row-error" : "";
        let statusHtml = isDuplicate ? "<span class='error-msg'>⚠️ Kayıtlı</span>" : "<span style='color:green;'>Uygun</span>";
        let checkDisabled = isDuplicate ? "disabled" : "checked"; // Çakışanları seçili getirme
        if(isDuplicate) checkDisabled = ""; // Checkbox'ı boş bırak (kullanıcı isterse seçemesin veya zor olsun)

        // Cinsiyet Gösterimi
        let genderIcon = "";
        if(stu.gender === 'girl') genderIcon = "👧 K";
        else if(stu.gender === 'boy') genderIcon = "👦 E";

        const tr = document.createElement("tr");
        tr.className = rowClass;
        tr.innerHTML = `
            <td><input type="checkbox" class="ocr-checkbox" data-index="${index}" ${!isDuplicate ? 'checked' : ''} ${isDuplicate ? 'disabled' : ''}></td>
            <td><input type="text" value="${stu.no}" onchange="updateParsedStudent(${index}, 'no', this.value)"></td>
            <td><input type="text" value="${stu.name}" onchange="updateParsedStudent(${index}, 'name', this.value)"></td>
            <td><input type="text" value="${stu.surname}" onchange="updateParsedStudent(${index}, 'surname', this.value)"></td>
            <td><input type="text" value="${stu.gender === 'girl' ? 'K' : (stu.gender === 'boy' ? 'E' : '')}" placeholder="K/E" onchange="updateParsedGender(${index}, this.value)" style="width:40px; text-align:center;"></td>
            <td>${statusHtml}</td>
        `;
        tbody.appendChild(tr);
    });

    document.getElementById("saveOCRBtn").disabled = false;
}

// Kullanıcı cinsiyeti elle düzeltirse
function updateParsedGender(index, val) {
    val = val.toUpperCase();
    if (val === 'K' || val === 'KIZ') parsedStudents[index].gender = 'girl';
    else if (val === 'E' || val === 'ERKEK') parsedStudents[index].gender = 'boy';
    else parsedStudents[index].gender = '';
}

// 6. Seçilenleri Kaydet (Cinsiyet Dahil)
async function saveOCRResults() {
    const checkboxes = document.querySelectorAll('.ocr-checkbox:checked');
    if (checkboxes.length === 0) return alert("Lütfen kaydedilecek öğrencileri seçin.");

    if (!currentManageClassId) return alert("Sınıf bilgisi eksik. Sayfayı yenileyip tekrar deneyin.");

    const btn = document.getElementById("saveOCRBtn");
    const originalText = btn.innerText;
    btn.innerText = `İşleniyor...`;
    btn.disabled = true;

    const user = firebase.auth().currentUser;
    let addedCount = 0;
    let skippedCount = 0;
    
    // Türkçe karakter haritası (Sadece harfler)
    const trMap = {'ç':'c','Ç':'c','ğ':'g','Ğ':'g','ı':'i','I':'i','İ':'i','ö':'o','Ö':'o','ş':'s','Ş':'s','ü':'u','Ü':'u'};

    try {
        let safeClassCode = "";
        const classDoc = await db.collection("classes").doc(currentManageClassId).get();
        if (classDoc.exists) safeClassCode = classDoc.data().classCode;

        for (const cb of checkboxes) {
            const index = cb.getAttribute('data-index');
            const s = parsedStudents[index];

            // --- ADIM 1: İSİM TEMİZLİĞİ (Boşlukları Yok Et) ---
            
            // 1. Önce küçük harfe çevir
            let rawName = s.name.toLowerCase();
            let rawSurname = (s.surname || "student").toLowerCase();

            // 2. Türkçe karakterleri İngilizceye çevir
            rawName = rawName.replace(/[çğışüöÇĞİŞÜÖ]/g, char => trMap[char] || char);
            rawSurname = rawSurname.replace(/[çğışüöÇĞİŞÜÖ]/g, char => trMap[char] || char);

            // 3. KRİTİK ADIM: Tüm boşlukları, tireleri ve nokta işaretlerini SİL (Regex: \s = boşluk)
            // "alperen yuşa" -> "alperenyusa" olur.
            const cleanName = rawName.replace(/[\s\-\.]/g, ''); 
            const cleanSurname = rawSurname.replace(/[\s\-\.]/g, '');

            // E-mail ve Şifre (Artık içinde boşluk olması imkansız)
            const email = `${cleanName}${s.no}@learneng.me`;
            
            // Şifre: Ad + No (Çift isimlerde ad uzun olduğu için güvenli)
            let password = `${cleanName}${s.no}`; 
            
            // Ekstra Güvenlik: Şifre yine de 6 karakterden kısaysa uzat
            if (password.length < 6) password += "123";

            try {
                // KONTROL: Zaten var mı?
                const existingSnap = await db.collection("students").where("email", "==", email).limit(1).get();

                if (!existingSnap.empty) {
                    console.log(`⚠️ Atlandı (Zaten kayıtlı): ${s.name}`);
                    skippedCount++;
                    continue; 
                }

                // OLUŞTUR
                const resp = await fetch("https://europe-west1-learneng-plus.cloudfunctions.net/registerStudent", {
                    method: "POST",
                    headers: { "Content-Type": "application/json" },
                    body: JSON.stringify({ 
                        name: s.name,     // Ekranda görünen isim (Boşluklu kalabilir, sorun yok)
                        surname: s.surname, 
                        stuNo: s.no, 
                        email: email,     // Temizlenmiş email (Boşluksuz)
                        password: password, 
                        classId: currentManageClassId, 
                        teacherId: user.uid, 
                        classCode: safeClassCode,
                        gender: s.gender || "" 
                    })
                });

                if (!resp.ok) throw new Error(`Server Error: ${resp.status}`);

                // Firestore Güncelleme
                db.collection("students").where("email", "==", email).get().then(snap => {
                    if(!snap.empty) {
                        snap.docs[0].ref.update({ 
                            generatedPassword: password,
                            gender: s.gender || ""
                        });
                    }
                });

                addedCount++;
                console.log(`✅ Eklendi: ${s.name}`);

            } catch (err) {
                console.error(`Hata (${s.name}):`, err);
            }
        }

        let msg = `✅ İşlem Tamamlandı!`;
        if (addedCount > 0) msg += `\n• ${addedCount} yeni öğrenci eklendi.`;
        if (skippedCount > 0) msg += `\n• ${skippedCount} öğrenci zaten kayıtlı olduğu için atlandı.`;
        
        alert(msg);
        closeOCRModal();
        
        if (typeof loadManagerStudentList === "function") {
            loadManagerStudentList(currentManageClassId);
        }

    } catch (globalErr) {
        alert("Genel Hata: " + globalErr.message);

function titleCase(str) {
    if (!str) return "";
    return str.toLocaleLowerCase('tr-TR').split(' ').map(word => 
        word.charAt(0).toLocaleUpperCase('tr-TR') + word.slice(1)
    ).join(' ');
}

async function studentCreateAccountFromJoin() {
  const btn = document.getElementById("joinCreateBtn");
  const name = (document.getElementById("joinName")?.value || "").trim();
  const surname = (document.getElementById("joinSurname")?.value || "").trim();
  const stuNo = (document.getElementById("joinStuNo")?.value || "").trim();
  const email = (document.getElementById("joinEmail")?.value || "").trim().toLowerCase();
  const pass = (document.getElementById("joinPass")?.value || "").trim();

  if (!JOIN_CLASS_CODE) return alert("Sınıf kodu yok. Öğretmen linkini yeniden aç.");
  if (!name || !surname || !stuNo || !email || !pass) return alert("Lütfen tüm alanları doldur.");

  if (btn) { btn.disabled = true; btn.textContent = "Kaydediliyor..."; }

  try {
    // 1) classCode -> classId & teacherId
    const classQ = await db.collection("classes").where("classCode", "==", JOIN_CLASS_CODE).limit(1).get();
    if (classQ.empty) {
      throw new Error("invalid-class-code");
    }
    const classDoc = classQ.docs[0];
    const classId = classDoc.id;
    const classData = classDoc.data();
    const teacherId = classData.teacherId || null;

    // 2) Create Auth user
    const cred = await auth.createUserWithEmailAndPassword(email, pass);
    const uid = cred.user.uid;

    // 3) Create students/{uid}
    await db.collection("students").doc(uid).set({
      name,
      surname,
      stuNo, // ✅ okul numarası alanın zaten bu
      email,
      classCode: JOIN_CLASS_CODE,
      classId,
      teacherId,
      avatarKey: "learnEngPenguin",
      stars: 0,
      createdAt: firebase.firestore.FieldValue.serverTimestamp()
    }, { merge: true });

    alert("✅ Kayıt tamam! Artık e-posta ve şifrenle giriş yapabilirsin.");
    closeStudentJoinModal();
    // Login zaten createUser sonrası otomatik oluyor
    closeAuthModal();

  } catch (e) {
    console.error(e);

    if (e.code === "auth/email-already-in-use") {
      alert("Bu e-posta zaten kayıtlı. Öğrenci giriş ekranından giriş yapmayı dene.");
    } else if (String(e.message || "").includes("invalid-class-code")) {
      alert("Geçersiz sınıf kodu. Öğretmeninden doğru linki iste.");
    } else {
      alert("⚠️ Kayıt başarısız. Lütfen tekrar dene.");
    }
  } finally {
    if (btn) { btn.disabled = false; btn.textContent = "✅ Kaydol"; }
  }
}

async function findTeacherRecord(user) {
  const uid = user.uid;
  const email = (user.email || "").toLowerCase().trim();

  const usersDoc = await db.collection("users").doc(uid).get();
  if (usersDoc.exists) return { source: "users", doc: usersDoc, data: usersDoc.data() };

  const appDoc = await db.collection("appTeachers").doc(uid).get();
  if (appDoc.exists) return { source: "appTeachers", doc: appDoc, data: appDoc.data() };

  const donorUidDoc = await db.collection("donorTeachers").doc(uid).get();
  if (donorUidDoc.exists) return { source: "donorTeachers", doc: donorUidDoc, data: donorUidDoc.data() };

  if (email) {
    const q = await db.collection("donorTeachers").where("email", "==", email).limit(1).get();
    if (!q.empty) return { source: "donorTeachers", doc: q.docs[0], data: q.docs[0].data() };
  }

  if (email) {
    const q2 = await db.collection("appTeachers").where("email", "==", email).limit(1).get();
    if (!q2.empty) return { source: "appTeachers", doc: q2.docs[0], data: q2.docs[0].data() };
  }

  return null;
}

function computeRoleFromTeacher(found) {
  let roleLabel = "Free Teacher";
  let roleColor = "#999";
  let userRole = "free";
  let adFree = false;

  if (!found) return { roleLabel, roleColor, userRole, adFree };

  if (found.source === "users" && found.data?.paid === true) {
    roleLabel = "Plus Teacher 🚀";
    roleColor = "#667eea";
    userRole = "plus";
    adFree = true;
    return { roleLabel, roleColor, userRole, adFree };
  }

  if (found.source === "appTeachers" && found.data?.paid === true) {
    roleLabel = "PRO Member 🚀";   // sadece label; rol yine plus
    roleColor = "#667eea";
    userRole = "plus";
    adFree = true;
    return { roleLabel, roleColor, userRole, adFree };
  }

  if (found.source === "donorTeachers" && found.data?.paid === true) {
    roleLabel = "Donor Teacher 🎁";
    roleColor = "#009688";
    userRole = "donor";
    adFree = true;
    return { roleLabel, roleColor, userRole, adFree };
  }

  return { roleLabel, roleColor, userRole, adFree };
}

async function getStudentsOfClass(classId, teacherId){
  const cSnap = await db.collection("classes").doc(classId).get();
  const classCode = (cSnap.exists ? (cSnap.data().classCode || "").trim() : "");

  let q = db.collection("students").where("teacherId","==", teacherId);
  if (classCode) q = q.where("classCode","==", classCode);
  else q = q.where("classId","==", classId);

  const snap = await q.get();
  return snap.docs.map(d=>{
    const s = d.data() || {};
    return { id:d.id, name:`${s.name||""} ${s.surname||""}`.trim() || "Student" };
  });
}

async function getAssignmentCountsForClass(assignmentId, classId, teacherId){
  const students = await getStudentsOfClass(classId, teacherId);

  let submittedCount = 0;
  try{
    const subsSnap = await db.collection("assignmentSubmissions")
      .where("teacherId","==", teacherId)
      .where("classId","==", classId)
      .where("assignmentId","==", assignmentId)
      .get();

    subsSnap.forEach(doc=>{
      const s = doc.data() || {};
      if (s.status === "submitted") submittedCount++;
    });
  }catch(e){
    console.warn("getAssignmentCountsForClass:", e);
  }

  const totalStudents = students.length;
  const pendingCount = Math.max(0, totalStudents - submittedCount);
  return { totalStudents, submittedCount, pendingCount };
}

function guessGradeFromText(text){
  const t = String(text||"").toLowerCase();
  const m = t.match(/(?:^|\D)(5|6|7|8)(?:\D|$)/);
  if (m) return parseInt(m[1],10);
  if (t.includes("5th")) return 5;
  if (t.includes("6th")) return 6;
  if (t.includes("7th")) return 7;
  if (t.includes("8th")) return 8;
  return null;
}

async function openSendAssignmentModal(payload){
  // ✅ only plus
  if (localStorage.getItem("userRole") !== "plus") return;

  const user = firebase.auth().currentUser;
  if (!user) { 
    alert("Please login first."); 
    if (typeof openAuthModal === "function") try { openAuthModal(); } catch(_) {}
    return;
  }

  window.__sendAssign.payload = payload || {};

  const label = document.getElementById("samTargetLabel");
  const wrap  = document.getElementById("samClassesWrap");
  const hint  = document.getElementById("samClassHint");
  const dueInput = document.getElementById("samDueDate");

  // target label
  if (label) {
    const kind = (payload?.targetKind === "folder") ? "Folder" : "File";
    const ttl = payload?.title || "Material";
    label.innerHTML = `Target: <b>${kind}</b> — <span style="font-weight:900; color:#0f172a;">${escapeHtml(ttl)}</span>`;
  }

  // due default +7
  if (dueInput) {
    const d = new Date();
    d.setDate(d.getDate() + 7);
    dueInput.value = d.toISOString().slice(0,10);
  }

  // classes
  const classes = await loadMyClassesOnce();
  if (wrap) wrap.innerHTML = "";

  const bc = (payload?.breadcrumb || []).join(" / ");
  const guessedGrade = guessGradeFromText(bc) || guessGradeFromText(payload?.title || "");

  const selectedIds = new Set();
  if (guessedGrade != null) {
    classes.forEach(c => {
      const nm = (c.name || "");
      if (guessGradeFromText(nm) === guessedGrade) selectedIds.add(c.id);
    });
  }

  classes.forEach(c => {
    const row = document.createElement("label");
    row.style.cssText = "display:flex; gap:10px; align-items:center; padding:10px; border:1px solid #e2e8f0; border-radius:12px; background:#fff; cursor:pointer;";
    row.innerHTML = `
      <input type="checkbox" class="samClassChk" value="${c.id}" ${selectedIds.has(c.id) ? "checked":""} style="transform:scale(1.15);">
      <div style="flex:1;">
        <div style="font-weight:900; color:#0f172a;">${escapeHtml(c.name || "Class")}</div>
        <div style="color:#94a3b8; font-size:.85rem;">${escapeHtml((c.classCode||"").trim() || "-")}</div>
      </div>
    `;
    wrap.appendChild(row);
  });

  if (hint) {
    hint.textContent = guessedGrade
      ? `Default selected: Grade ${guessedGrade} classes (based on current folder).`
      : `Tip: Navigate from a grade folder (e.g., 8th Grade) to auto-select matching classes.`;
  }

  const m = document.getElementById("sendAssignmentModal");
  if (m) m.style.display = "flex";
}

async function sendDigitalAssignment(){
  const btn = document.getElementById("samSendBtn");
  if (btn) { btn.disabled = true; btn.textContent = "Sending..."; }

  try {
    if (localStorage.getItem("userRole") !== "plus") {
      alert("Plus membership required.");
      return;
    }

    const user = firebase.auth().currentUser;
    if (!user) { alert("Please login first."); return; }

    const payload = window.__sendAssign.payload || {};
    const dueStr = (document.getElementById("samDueDate")?.value || "").trim();
    if (!dueStr) { alert("Please select a deadline."); return; }

    const classIds = Array.from(document.querySelectorAll(".samClassChk:checked")).map(x => x.value);
    if (!classIds.length) { alert("Please select at least one class."); return; }

    // Due: end of day
    const dueDate = new Date(dueStr + "T23:59:59");
    const dueAt = firebase.firestore.Timestamp.fromDate(dueDate);

    // ✅ targetUrl yoksa ama storagePath (targetId) varsa şimdi üret (FILE için)
    let targetUrl = payload.targetUrl || "";
    if (!targetUrl && payload.targetId && payload.targetKind === "file") {
      try {
        targetUrl = await firebase.storage().ref(payload.targetId).getDownloadURL();
      } catch (e) {
        console.warn("getDownloadURL failed, saving storagePath only:", e);
      }
    }

    const doc = {
      type: "digital",
      teacherId: user.uid,

      title: payload.title || "Digital Assignment",
      targetKind: payload.targetKind || "file",

      // file için: url + storagePath, folder için: storagePath (folder path)
      targetUrl: targetUrl || null,
      targetStoragePath: payload.targetId || null,   // ✅ always keep this
      breadcrumb: payload.breadcrumb || [],

      classIds,
      dueAt,

      baseStars: 1,
      bonusStarsIfScore: 1,

      createdAt: firebase.firestore.FieldValue.serverTimestamp()
    };

    await db.collection("assignments").add(doc);

    alert("✅ Assignment sent!");
    closeSendAssignmentModal();

  } catch (e) {
    console.error(e);
    alert("⚠️ Failed: " + (e.message || e));
  } finally {
    if (btn) { btn.disabled = false; btn.textContent = "✅ Send"; }
  }
}

  return String(s||"").replace(/[&<>"']/g, m => ({
    "&":"&amp;","<":"&lt;",">":"&gt;",'"':"&quot;","'":"&#39;"
  }[m]));
}

  const t2 = document.getElementById("tabPending");

function asmApplyFilter(){
  const q = (document.getElementById("asmSearch")?.value || "").toLowerCase().trim();
  const listId = (__asm.tab==="submitted") ? "asmSubmitted" : "asmPending";
  const el = document.getElementById(listId);
  if (!el) return;

  const rows = (__asm.tab==="submitted") ? __asm.submittedRows : __asm.pendingRows;
  if (!q){
    el.innerHTML = rows.join("") || `<div style="color:#94a3b8;">No items.</div>`;
    return;
  }
  const filtered = rows.filter(html => html.toLowerCase().includes(q));
  el.innerHTML = filtered.join("") || `<div style="color:#94a3b8;">No match.</div>`;
}

function asmRender(mode){
  const target = document.getElementById("asmList");
  if (!target) return;

  let rows = [];
  if (mode === "all") rows = __asm.submittedRows.concat(__asm.pendingRows);
  if (mode === "submitted") rows = __asm.submittedRows;
  if (mode === "pending") rows = __asm.pendingRows;

  target.innerHTML = rows.join("") || `<div style="color:#94a3b8;">No items.</div>`;
}

function seenKey(teacherId, classId, assignmentId){
  return `seen_${teacherId}_${classId}_${assignmentId}`;
}
function getSeenMs(teacherId, classId, assignmentId){
  const v = localStorage.getItem(seenKey(teacherId, classId, assignmentId));
  const n = v ? parseInt(v,10) : 0;
  return isNaN(n) ? 0 : n;
}
function setSeenNow(teacherId, classId, assignmentId){
  localStorage.setItem(seenKey(teacherId, classId, assignmentId), String(Date.now()));
}

// cache in memory (per teacher session)
window.__asgStatsCache = {
  teacherId: null,
  byKey: new Map(),       // key = `${assignmentId}__${classId}` => stats
  byClass: new Map(),     // classId => array of stats
  loadedAt: 0
};

// loads all stats for teacher ONCE (fast)
async function loadAllAssignmentStatsForTeacher(){
  const user = firebase.auth().currentUser;
  if (!user) return window.__asgStatsCache;

  // cache valid for 20 seconds (avoid spam)
  const now = Date.now();
  if (window.__asgStatsCache.teacherId === user.uid && (now - window.__asgStatsCache.loadedAt) < 20000) {
    return window.__asgStatsCache;
  }

  const snap = await db.collection("assignmentClassStats")
    .where("teacherId","==", user.uid)
    .get();

  const byKey = new Map();
  const byClass = new Map();

  snap.forEach(doc=>{
    const s = doc.data() || {};
    const assignmentId = s.assignmentId;
    const classId = s.classId;
    if (!assignmentId || !classId) return;
    const k = `${assignmentId}__${classId}`;
    byKey.set(k, s);

    if (!byClass.has(classId)) byClass.set(classId, []);
    byClass.get(classId).push(s);
  });

  window.__asgStatsCache = {
    teacherId: user.uid,
    byKey,
    byClass,
    loadedAt: now
  };

  return window.__asgStatsCache;
}

function isUnseenForStat(teacherId, stat){
  const last = stat.lastSubmittedAt?.toDate ? stat.lastSubmittedAt.toDate() : null;
  if (!last) return false;
  const seenMs = getSeenMs(teacherId, stat.classId, stat.assignmentId);
  return last.getTime() > seenMs;
}

async function paintSidebarAssignmentsBadge(){
  const user = firebase.auth().currentUser;
  if (!user) return;

  const cache = await loadAllAssignmentStatsForTeacher();
  let hasUnseen = false;

  cache.byKey.forEach(stat=>{
    if (isUnseenForStat(user.uid, stat)) hasUnseen = true;
  });

  const badge = document.getElementById("asgNotifBadge");
  if (badge) badge.style.display = hasUnseen ? "inline-flex" : "none";
}

// For class card: show if any assignment in that class is unseen
function classHasUnseen(teacherId, classId){
  const arr = window.__asgStatsCache.byClass.get(classId) || [];
  return arr.some(stat => isUnseenForStat(teacherId, stat));
}

// For assignment card: unseen flag + submittedCount (fast)
function getStat(assignmentId, classId){
  const k = `${assignmentId}__${classId}`;
  return window.__asgStatsCache.byKey.get(k) || null;
}

async function openEditAssignmentModal(assignmentId){
  const user = firebase.auth().currentUser;
  if (!user) return alert("Please login again.");

  const m = document.getElementById("editAssignmentModal");
  if (!m) return alert("editAssignmentModal missing in HTML.");

  const meta = document.getElementById("eamMeta");
  const titleEl = document.getElementById("eamTitle");
  const dueEl = document.getElementById("eamDue");

  try{
    const doc = await db.collection("assignments").doc(assignmentId).get();
    if (!doc.exists) return alert("Assignment not found.");

    const a = doc.data() || {};
    if (a.teacherId && a.teacherId !== user.uid){
      return alert("You can only edit your own assignments.");
    }

    window.__editAssign.id = assignmentId;

    if (meta) meta.textContent = `ID: ${assignmentId}`;
    if (titleEl) titleEl.value = a.title || "";

    // dueAt -> date input (YYYY-MM-DD)
    const due = a.dueAt?.toDate ? a.dueAt.toDate() : (a.dueAt ? new Date(a.dueAt) : null);
    if (dueEl){
      dueEl.value = due ? new Date(due.getTime() - (due.getTimezoneOffset()*60000)).toISOString().slice(0,10) : "";
    }

    m.style.display = "flex";
  }catch(e){
    console.error(e);
    alert("Could not open edit modal. Check Console.");
  }
}

async function saveEditedAssignment(){
  const user = firebase.auth().currentUser;
  if (!user) return alert("Please login again.");

  const id = window.__editAssign.id;
  if (!id) return;

  const btn = document.getElementById("eamSaveBtn");
  const titleEl = document.getElementById("eamTitle");
  const dueEl = document.getElementById("eamDue");

  const newTitle = (titleEl?.value || "").trim();
  const dueStr = (dueEl?.value || "").trim();

  if (!newTitle) return alert("Please enter a title.");
  if (!dueStr) return alert("Please select a deadline.");

  // Save as end of day
  const dueDate = new Date(dueStr + "T23:59:59");
  const dueAt = firebase.firestore.Timestamp.fromDate(dueDate);

  try{
    if (btn){ btn.disabled = true; btn.textContent = "Saving..."; }

    // security: ensure owner
    const ref = db.collection("assignments").doc(id);
    await db.runTransaction(async (tx)=>{
      const snap = await tx.get(ref);
      if (!snap.exists) throw new Error("Assignment not found.");
      const a = snap.data() || {};
      if (a.teacherId && a.teacherId !== user.uid) throw new Error("Not allowed.");

      tx.update(ref, {
        title: newTitle,
        dueAt,
        updatedAt: firebase.firestore.FieldValue.serverTimestamp()
      });
    });

    closeEditAssignmentModal();

    // ✅ Refresh teacher UI
    // If you're currently on class assignments page, reload it:
    if (typeof window.__lastOpenClassAssignments === "object" && window.__lastOpenClassAssignments?.classId){
      const x = window.__lastOpenClassAssignments;
      await openClassAssignments(x.classId, x.className, x.classCode);
    } else {
      // fallback: repaint cards
      await openAssignmentsModule();
    }

  }catch(e){
    console.error(e);
    alert("Save failed: " + (e.message || e));
  }finally{
    if (btn){ btn.disabled = false; btn.textContent = "✅ Save Changes"; }
  }
}

async function confirmDeleteAssignment(assignmentId, title){
  const user = firebase.auth().currentUser;
  if (!user) return alert("Please login again.");

  let safeTitle = title || "Assignment";
  try{
    if (!title){
      const snap = await db.collection("assignments").doc(assignmentId).get();
      safeTitle = snap.exists ? (snap.data()?.title || "Assignment") : "Assignment";
    }
  }catch(e){ /* ignore */ }

  const ok = confirm(`Delete this assignment?\n\n${safeTitle}\n\nThis will remove it from students' dashboards.`);
  if (!ok) return;

  await deleteAssignment(assignmentId);
}

async function deleteAssignment(assignmentId){
  const user = firebase.auth().currentUser;
  if (!user) return alert("Please login again.");

  try{
    const ref = db.collection("assignments").doc(assignmentId);

    // ensure owner
    const snap = await ref.get();
    if (!snap.exists) return alert("Assignment not found.");
    const a = snap.data() || {};
    if (a.teacherId && a.teacherId !== user.uid) return alert("Not allowed.");

    await ref.delete();

    // ✅ Refresh current page
    if (typeof window.__lastOpenClassAssignments === "object" && window.__lastOpenClassAssignments?.classId){
      const x = window.__lastOpenClassAssignments;
      await openClassAssignments(x.classId, x.className, x.classCode);
    } else {
      await openAssignmentsModule();
    }

    // ✅ Update sidebar bell
    if (typeof paintSidebarAssignmentsBadge === "function"){
      await paintSidebarAssignmentsBadge();
    }

    alert("✅ Assignment deleted.");
  }catch(e){
    console.error(e);
    alert("Delete failed: " + (e.message || e));
  }
}

async function proceedToDownload() {
    const btn = document.getElementById("proceedDownloadBtn");
    const originalText = btn.innerHTML;
    btn.innerHTML = "Açılıyor... ⏳";
    btn.disabled = true;

    const data = window.__pendingDownload;
    if (!data) return closePrintableDownloadModal();

    try {
        if (data.url) {
            window.open(data.url, "_blank");
        } else if (data.storagePath) {
            const url = await firebase.storage().ref(data.storagePath).getDownloadURL();
            window.open(url, "_blank");
            // Cache the url for next time (optional but good for performance)
            data.url = url; 
        } else {
            alert("⚠️ Dosya bağlantısı bulunamadı.");
        }
    } catch (e) {
        console.error("Dosya açma hatası:", e);
        alert("⚠️ Dosya açılırken bir hata oluştu.");
    } finally {
        btn.innerHTML = originalText;
        btn.disabled = false;
        closePrintableDownloadModal();
        if(data.restoreFunc) data.restoreFunc();
    }
}

let adSenseInjected = false;

function shouldShowAds() {
  // 1. Öğretmen Plus/Pro (reklamsız) mi?
  const adFree = localStorage.getItem("adFree") === "true";
  // 2. E-posta ile giriş yapmış kayıtlı öğrenci mi?
  const isStudent = localStorage.getItem("userRole") === "student";

  const isLocalStudent = localStorage.getItem("localAccessRole") === "student";

  if (adFree || isStudent || isLocalStudent) {
    return false;
  }
  
  // Hiçbiri değilse (Ziyaretçi veya Ücretsiz Öğretmen) REKLAM GÖSTER
  return true;
}

function checkAndLoadAds() {
  // Eğer daha önce yüklendiyse (sayfa içinde tekrar çağrılırsa) durdur
  if (adSenseInjected) return;

  if (shouldShowAds()) {
    console.log("Reklamlar aktif ediliyor...");
    const script = document.createElement("script");
    script.async = true;
    script.src = "https://pagead2.googlesyndication.com/pagead/js/adsbygoogle.js?client=ca-pub-3790447039448781";
    script.crossOrigin = "anonymous";
    document.head.appendChild(script);
    adSenseInjected = true;
  } else {
    console.log("Reklamsız deneyim aktif. AdSense yüklenmedi.");
  }
}

// Promo Modal Reklamı (Manuel çağrılan özel alan)
function pushAdsenseSafely(boxId) {
  const box = document.getElementById(boxId);
  if (!box) return;

  if (box.dataset.loaded === "true") return;

  box.style.display = "block";

  const ins = box.querySelector("ins.adsbygoogle");
  const width = ins ? ins.offsetWidth : box.offsetWidth;

  if (width <= 0) {
    setTimeout(() => pushAdsenseSafely(boxId), 250);
    return;
  }

  try {
    (window.adsbygoogle = window.adsbygoogle || []).push({});
    box.dataset.loaded = "true";
  } catch (e) {
    console.warn("AdSense push warning:", e);
  }
}

function hideAdsense(boxId) {
  const box = document.getElementById(boxId);
  if (box) box.style.display = "none";
}
