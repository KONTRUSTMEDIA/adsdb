  <!-- Auto-recovery: stale cache detection + chunk load failures -->
  <script>
    (function() {
      function clearCachesAndReload() {
        if ('caches' in window) {
          caches.keys().then(function(names) {
            return Promise.all(names.map(function(n) { return caches.delete(n); }));
          }).then(function() { window.location.reload(true); });
        } else {
          window.location.reload(true);
        }
      }

      window.addEventListener('error', function(e) {
        // Only auto-retry once per session to prevent infinite loops
        if (sessionStorage.getItem('_jsRetried')) return;

        // Detect "Unexpected token '<'" — JS file served as HTML
        if (e.message && e.message.indexOf("Unexpected token '<'") !== -1) {
          sessionStorage.setItem('_jsRetried', '1');
          clearCachesAndReload();
          return;
        }

        // Detect deferred chunk load failures (stale cache after deploy)
        // These show up as "Loading chunk X failed" or "Failed to fetch" on .part.js files
        var msg = (e.message || '').toLowerCase();
        var src = (e.filename || '').toLowerCase();
        if (
          (msg.indexOf('loading chunk') !== -1) ||
          (msg.indexOf('loading module') !== -1) ||
          (src.indexOf('.part.js') !== -1 && (msg.indexOf('failed') !== -1 || msg.indexOf('error') !== -1)) ||
          (msg.indexOf('failed to fetch dynamically imported module') !== -1)
        ) {
          sessionStorage.setItem('_jsRetried', '1');
          clearCachesAndReload();
        }
      });
    })();
  </script>

    window.onerror = function(message, source, lineno, colno, error) {
      // Ignore errors from third-party ad/tracking scripts
      if (source && (
        source.includes('adthrive') ||
        source.includes('ads.adthrive.com') ||
        source.includes('googletagmanager') ||
        source.includes('google-analytics')
      )) {
        console.warn('Suppressed third-party error:', message);
        return true; // Prevent the error from propagating
      }
      // Ignore common third-party variable errors (Raptive ads)
      if (message && message.includes('adthrive')) {
        console.warn('Suppressed ad script error:', message);
        return true;
      }

      // Log app errors to GA4 (skip third-party)
      try {
        if (typeof gtag === 'function') {
          var desc = String(message || 'unknown').substring(0, 150);
          gtag('event', 'exception', {
            description: desc,
            fatal: true,
            error_context: 'window.onerror',
            source: source ? source.substring(0, 100) : 'unknown',
            line: lineno || 0,
          });
        }
      } catch(_) {}

      return false; // Let other errors propagate normally
    };

    // Catch unhandled Promise rejections
    window.onunhandledrejection = function(event) {
      try {
        if (typeof gtag === 'function') {
          var reason = event.reason ? String(event.reason).substring(0, 150) : 'unknown';
          gtag('event', 'exception', {
            description: reason,
            fatal: false,
            error_context: 'unhandled_rejection',
          });
        }
      } catch(_) {}
    };

    // Memory monitoring (Chrome/Edge only — covers ~70% of users)
    // Expose for Dart JS interop
    window.getMemoryInfo = function() {
      if (performance && performance.memory) {
        return Math.round(performance.memory.usedJSHeapSize / 1048576 * 10) / 10;
      }
      return null;
    };

    // Periodic memory check — log to GA4 if approaching limits
    (function() {
      var _memThresholdsFired = {};
      setInterval(function() {
        var mb = window.getMemoryInfo();
        if (mb === null) return;
        var thresholds = [100, 150, 200];
        for (var i = 0; i < thresholds.length; i++) {
          var t = thresholds[i];
          if (mb >= t && !_memThresholdsFired[t]) {
            _memThresholdsFired[t] = true;
            try {
              if (typeof gtag === 'function') {
                gtag('event', 'memory_warning', {
                  threshold_mb: t,
                  actual_mb: mb,
                  context: 'periodic_check',
                  route: location.pathname || 'unknown',
                });
              }
            } catch(_) {}
          }
        }
      }, 30000); // Check every 30 seconds
    })();

    // Smart horizontal scroll handling - prevent browser navigation only when actively scrolling
    (function() {
      let scrollableElement = null;
      let startX = 0;
      let startScrollLeft = 0;

      document.addEventListener('pointerdown', function(e) {
        // Find if we clicked on a horizontally scrollable element
        let element = e.target;
        while (element && element !== document.body) {
          const hasHorizontalScroll = element.scrollWidth > element.clientWidth;
          if (hasHorizontalScroll) {
            startX = e.clientX;
            startScrollLeft = element.scrollLeft;
            scrollableElement = element;
            break;
          }
          element = element.parentElement;
        }
      }, { passive: true });

      document.addEventListener('pointermove', function(e) {
        if (!scrollableElement) return;

        const deltaX = e.clientX - startX;
        const scrollLeft = scrollableElement.scrollLeft;
        const maxScrollLeft = scrollableElement.scrollWidth - scrollableElement.clientWidth;

        // Determine if we can scroll in the desired direction
        const movingRight = deltaX > 0; // Pointer moving right = scroll content left
        const movingLeft = deltaX < 0;  // Pointer moving left = scroll content right

        const canScrollLeft = scrollLeft > 0; // Can scroll content left (we're not at leftmost)
        const canScrollRight = scrollLeft < maxScrollLeft; // Can scroll content right (we're not at rightmost)

        // Only consume the event if we can scroll in the desired direction
        if (movingRight && canScrollLeft) {
          // Moving pointer right and we can scroll content left - handle it, block navigation
          e.stopPropagation();
        } else if (movingLeft && canScrollRight) {
          // Moving pointer left and we can scroll content right - handle it, block navigation
          e.stopPropagation();
        }
        // Otherwise, let the browser handle navigation (we're at an edge or can't scroll)
      }, { passive: true });

      document.addEventListener('pointerup', function() {
        scrollableElement = null;
      }, { passive: true });

      document.addEventListener('pointercancel', function() {
        scrollableElement = null;
      }, { passive: true });

      // Touch events for mobile
      document.addEventListener('touchstart', function(e) {
        let element = e.target;
        while (element && element !== document.body) {
          const hasHorizontalScroll = element.scrollWidth > element.clientWidth;
          if (hasHorizontalScroll) {
            startX = e.touches[0].clientX;
            startScrollLeft = element.scrollLeft;
            scrollableElement = element;
            break;
          }
          element = element.parentElement;
        }
      }, { passive: true });

      document.addEventListener('touchmove', function(e) {
        if (!scrollableElement) return;

        const currentX = e.touches[0].clientX;
        const deltaX = currentX - startX;
        const scrollLeft = scrollableElement.scrollLeft;
        const maxScrollLeft = scrollableElement.scrollWidth - scrollableElement.clientWidth;

        const movingRight = deltaX > 0;
        const movingLeft = deltaX < 0;

        const canScrollLeft = scrollLeft > 0;
        const canScrollRight = scrollLeft < maxScrollLeft;

        // Only prevent default if we're actively scrolling within bounds
        if ((movingRight && canScrollLeft) || (movingLeft && canScrollRight)) {
          // Don't prevent default - let Flutter handle it, but stop propagation
          e.stopPropagation();
        }
      }, { passive: true });

      document.addEventListener('touchend', function() {
        scrollableElement = null;
      }, { passive: true });
    })();
