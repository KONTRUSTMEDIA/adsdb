        // Toggle View Menu
        function toggleViewMenu(e) {
            if (e) e.stopPropagation();
            const menu = document.getElementById('view-menu');
            const appleMenu = document.getElementById('apple-menu');
            if (appleMenu) appleMenu.style.display = 'none';
            
            if (menu.style.display === 'block') {
                menu.style.display = 'none';
            } else {
                menu.style.display = 'block';
            }
        }

        // Close menus on click outside
        document.addEventListener('click', function(e) {
            const viewMenu = document.getElementById('view-menu');
            const appleMenu = document.getElementById('apple-menu');
            const viewMenuItem = e.target.closest('.macos-menubar-item');
            const appleLogo = e.target.closest('.macos-apple-logo');
            
            if (viewMenu && viewMenu.style.display === 'block' && !viewMenu.contains(e.target) && (!viewMenuItem || viewMenuItem.textContent.trim() !== 'View')) {
                viewMenu.style.display = 'none';
            }
            
            if (appleMenu && appleMenu.style.display === 'block' && !appleMenu.contains(e.target) && !appleLogo) {
                appleMenu.style.display = 'none';
            }
        });

        // Open page in macOS window
        function openPageWindow(slug, title) {
            const windowId = `page-${slug}`;
            let pageWindow = document.getElementById(windowId);

            if (!pageWindow) {
                // Create new window
                pageWindow = createBaseWindow(windowId, title);

                // Add random offset for new windows
                const offset = Math.floor(Math.random() * 40) - 20;
                pageWindow.style.transform = `translate(calc(-50% + ${offset}px), calc(-50% + ${offset}px))`;

                document.body.appendChild(pageWindow);

                const content = pageWindow.querySelector('.window-content');

                // Load page content in iframe - use pagewindow viewpage for minimal layout
                const pageUrl = `${DOMAIN}?viewpage=pagewindow&slug=${slug}`;
                content.innerHTML = `<iframe src="${pageUrl}" style="width:100%;height:100%;border:none;display:block;"></iframe>`;
                content.style.overflow = 'hidden';

                setupWindowInteractions(pageWindow);
            }

            openWindow(pageWindow);
            
            // Close the view menu
            document.getElementById('view-menu').style.display = 'none';
        }
