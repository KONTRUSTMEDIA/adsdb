<script>
function getQueryParam(name) {
	const params = new URLSearchParams(window.location.search);
	return params.get(name);
}

function ensureMobileDeviceCookie() {
	if (!isAndroidApp()) return;

	const did = getQueryParam('did');
	if (!did) return;

	const hasCookie = document.cookie
		.split('; ')
		.some(row => row.startsWith('mobile_device='));

	if (!hasCookie) {
		document.cookie =
			"mobile_device=" + encodeURIComponent(did) +
			"; path=/" +
			"; domain=.soccernet.ee" +
			"; max-age=" + (60 * 60 * 24 * 365);
	}
}

function isAndroidApp() {
  var ua = navigator.userAgent || "";
  return ua.includes("Android") && ua.includes("SoccernetApp/");
}

function togglePushNotification() {
	const before = $('#push_allowed').prop('checked');

	$.get("/lib/ajax/message.ajax.php?action=toggle_push_notification", function (resp) {
		getPushNotificationStatus();
	}).fail(function(xhr) {
		$('#push_allowed').prop('checked', before);
	});
}


function getPushNotificationStatus() {
	$.get("/lib/ajax/message.ajax.php?action=get_push_notification_status", function (data) {
		if (!data) {
			$('#push_allowed').prop('checked', false);
			return;
		}
		const normalized = String(data).trim();
		const allowed = (normalized === '1');
		$('#push_allowed').prop('checked', allowed);
	});
}

function adjustBannerVisibility() {
    const screenWidth = window.innerWidth;
    let activeClass = "";

    if (screenWidth >= 1700) {
        activeClass = "tower_300";
    } else if (screenWidth >= 1500) {
        activeClass = "tower_200";
    } else if (screenWidth >= 1200) {
        activeClass = "tower_100";
    }

    document.querySelectorAll(".tower").forEach(el => {
        el.style.display = "none";
    });

	if(activeClass) {
		document.querySelectorAll(`.${activeClass}`).forEach(el => {
			el.style.display = "block";
		});
	}

    adjustBannerPositions();

    setTimeout(() => {
        document.querySelectorAll(".ad-banner").forEach(el => {
            el.style.visibility = "visible";
            el.style.transition = "top 0.3s ease-in-out";
        });
    }, 50);
}

function adjustBannerPositions() {
    const leftBanner = document.querySelector(".left-banner");
    const rightBanner = document.querySelector(".right-banner");

    const activeLeftBanner = leftBanner.querySelector("img:not([style='display: none;'])");
    const activeRightBanner = rightBanner.querySelector("img:not([style='display: none;'])");

    if (activeLeftBanner && activeRightBanner) {
        const leftWidth = activeLeftBanner.clientWidth;
        const rightWidth = activeRightBanner.clientWidth;

        leftBanner.style.right = `calc(50% + 555px)`;
        leftBanner.style.left = `auto`;

        rightBanner.style.left = `calc(50% + 555px)`;
    }
}

document.addEventListener("scroll", function () {
    var leftBanner = document.querySelector(".left-banner");
    var rightBanner = document.querySelector(".right-banner");

    if (window.scrollY > 36) {
        leftBanner.classList.add("fixed");
        rightBanner.classList.add("fixed");
    } else {
        leftBanner.classList.remove("fixed");
        rightBanner.classList.remove("fixed");
    }
});

window.addEventListener("load", adjustBannerVisibility);

<script>
    (function($) {
        var monthNames = [
            "JAANUAR", "VEEBRUAR", "MÄRTS", "APRILL", "MAI", "JUUNI",
            "JUULI", "AUGUST", "SEPTEMBER", "OKTOOBER", "NOVEMBER", "DETSEMBER"
        ];

        var today = new Date();
        var todayStr = formatDate(today);

        var snCalState = {
            year: today.getFullYear(),
            month: today.getMonth(),
            days: {},
            selectedDate: null,
            _initialized: false,
            _refreshTimerId: null
        };

        function pad(num) {
            return num < 10 ? "0" + num : "" + num;
        }

        function formatDate(d) {
            return d.getFullYear() + "-" + pad(d.getMonth() + 1) + "-" + pad(d.getDate());
        }

        function gotoPrevMonth() {
            if (snCalState.month === 0) {
                snCalState.month = 11;
                snCalState.year -= 1;
            } else {
                snCalState.month -= 1;
            }
            loadMonth(snCalState.year, snCalState.month, false);
        }

        function gotoNextMonth() {
            if (snCalState.month === 11) {
                snCalState.month = 0;
                snCalState.year += 1;
            } else {
                snCalState.month += 1;
            }
            loadMonth(snCalState.year, snCalState.month, false);
        }

        function loadMonth(year, month, keepSelected) {
            $("#snc-cal-grid").addClass("is-loading");

            $.getJSON("/lib/controller/calendar.php", {
                year: year,
                month: month + 1
            }).done(function(resp) {
                if (!resp) return;

                snCalState.year = resp.year || year;
                snCalState.month = (resp.month ? resp.month - 1 : month);
                snCalState.days = resp.days || {};

                renderMonth();
                renderLegend();

                var selectedDate = snCalState.selectedDate;

                // Kui ei pea vana valikut säilitama, vali täna / esimene sündmustega päev
                if (!keepSelected || !selectedDate) {
                    selectedDate = todayStr;
                    var firstWithEvents = null;

                    Object.keys(snCalState.days).sort().forEach(function(dateKey) {
                        var d = snCalState.days[dateKey];
                        if (!firstWithEvents &&
                            ((d.league_list && d.league_list.length) ||
                             (d.news_list && d.news_list.length))) {
                            firstWithEvents = dateKey;
                        }
                    });

                    if (!isSameMonthString(selectedDate, snCalState.year, snCalState.month) && firstWithEvents) {
                        selectedDate = firstWithEvents;
                    }
                }

                selectDay(selectedDate);
            }).always(function() {
                $("#snc-cal-grid").removeClass("is-loading");
            });
        }

        function isSameMonthString(dateStr, year, month) {
            if (!dateStr) return false;
            var parts = dateStr.split("-");
            if (parts.length !== 3) return false;
            var y = parseInt(parts[0], 10);
            var m = parseInt(parts[1], 10) - 1;
            return y === year && m === month;
        }

        function renderMonth() {
            $("#snc-cal-month-name").text(monthNames[snCalState.month] || "");
            $("#snc-cal-year").text(snCalState.year);

            var firstDay = new Date(snCalState.year, snCalState.month, 1);
            var firstWeekday = (firstDay.getDay() + 6) % 7; // Mon=0
            var daysInMonth = new Date(snCalState.year, snCalState.month + 1, 0).getDate();

            var html = "";

            for (var i = 0; i < firstWeekday; i++) {
                html += '<div class="snc-cal-day snc-cal-day--empty"></div>';
            }

            for (var day = 1; day <= daysInMonth; day++) {
                var dateStr = snCalState.year + "-" + pad(snCalState.month + 1) + "-" + pad(day);
                var dayData = snCalState.days[dateStr];

                var classes = ["snc-cal-day"];
                var styleAttr = "";

                if (dateStr === todayStr) {
                    classes.push("is-today");
                }
                if (dayData && dayData.flags && dayData.flags.length) {
                    classes.push("has-flags");
                    var flag = dayData.flags[0];
                    var color = flag.color || "#ae0039";
                    styleAttr = ' style="--snc-cal-flag-color:' + color + ';"';
                }

                html += '<div class="' + classes.join(" ") + '" data-date="' + dateStr + '"' + styleAttr + '>';
                html += '<span class="snc-cal-day-number">' + day + "</span>";

                if (dayData && dayData.flags && dayData.flags.length) {
                    html += '<div class="snc-cal-indicators">';
                    for (var f = 0; f < Math.min(dayData.flags.length, 3); f++) {
                        var fl = dayData.flags[f];
                        var dotColor = fl.color || "#ae0039";
                        html += '<span class="snc-cal-indicator" style="background:' + dotColor + '"></span>';
                    }
                    html += "</div>";
                }

                html += "</div>";
            }

            $("#snc-cal-grid").html(html);
        }

        function renderLegend() {
            var legends = {};

            Object.keys(snCalState.days).forEach(function(dateStr) {
                var dayData = snCalState.days[dateStr];
                if (!dayData || !dayData.flags) return;
                dayData.flags.forEach(function(flag) {
                    if (!flag || !flag.key) return;
                    if (!legends[flag.key]) {
                        legends[flag.key] = {
                            label: flag.label || flag.key,
                            color: flag.color || "#ae0039"
                        };
                    }
                });
            });

            var keys = Object.keys(legends);
            if (!keys.length) {
                $("#snc-cal-legend").html("");
                return;
            }

            var html = "";
            keys.forEach(function(key) {
                var l = legends[key];
                html += '<div class="snc-cal-legend-item">';
                html += '<span class="snc-cal-legend-dot" style="background:' + l.color + '"></span>';
                html += '<span>' + l.label + "</span>";
                html += "</div>";
            });

            $("#snc-cal-legend").html(html);
        }

        function selectDay(dateStr) {
            snCalState.selectedDate = dateStr || null;

            if (!dateStr) {
                $("#snc-calendar-game-list").html('<p style="text-align:center;">Valitud päeval mänge pole</p>');
                $("#snc-calendar-news-list").html('<p style="text-align:center;">Valitud päeval uudiseid pole</p>');
                return;
            }

            $(".snc-cal-day").removeClass("is-selected");
            var $cell = $('.snc-cal-day[data-date="' + dateStr + '"]');
            if ($cell.length) {
                $cell.addClass("is-selected");
            }

            var dayData = snCalState.days[dateStr] || {};

            renderGameList(dayData.league_list || []);
            renderNewsList(dayData.news_list || []);
        }

        function renderGameList(leagueList) {
            var html = "";

            if (!leagueList || !leagueList.length) {
                html = '<p style="text-align: center;">Valitud päeval mänge pole</p>';
                $("#snc-calendar-game-list").html(html);
                return;
            }

            html += '<div class="calendar_game_list">';
            for (var i = 0; i < leagueList.length; i++) {
                var league = leagueList[i];
                html += '<div class="calendar_topic_container">';

                var season = (new Date()).getFullYear();
                if (league.game_list && league.game_list.length && league.game_list[0].url) {
                    var parts = league.game_list[0].url.split("/");
                    if (parts.length > 3 && /^\d{4}$/.test(parts[2])) {
                        season = parts[2];
                    }
                }

                html += '<a href="/liiga/' + season + '/' + league.alias + '">';
                html += '<div class="calendar_topic_list_header">' + (league.name || "").toUpperCase() + "</div>";
                html += "</a>";

                html += '<div class="calendar_topic_game_list">';
                for (var j = 0; j < league.game_list.length; j++) {
                    var g = league.game_list[j];

                    var lastChar = g.time ? g.time.substr(g.time.length - 1) : "";
                    var color = (lastChar === "'") ? "red" : "";

                    html += '<div class="calendar_topic_game">';
                    html += '<a href="' + g.url + '" class="calendar_topic_game_time" style="color: ' + color + ';">' + (g.time || "") + "</a>";
                    html += '<a href="' + g.url + '" class="calendar_topic_game_teams">';
                    html += '<div style="display: flex; align-items: center;">';
                    html += '<img src="' + g.home_team_logo + '" style="height: 18px;">';
                    html += '<div class="calendar_topic_game_team">' + g.home_team_name + "</div>";
                    html += "</div>";
                    html += '<div style="display: flex; align-items: center;">';
                    html += '<img src="' + g.away_team_logo + '" style="height: 18px;">';
                    html += '<div class="calendar_topic_game_team">' + g.away_team_name + "</div>";
                    html += "</div>";
                    html += "</a>";
                    html += '<div class="calendar_topic_game_score_container">';
                    if (g.score) {
                        html += '<a href="' + g.url + '" class="calendar_topic_game_score">' + g.score + "</a>";
                    }
                    html += "</div>";
                    html += "</div>";
                }
                html += "</div>"; // game_list
                html += "</div>"; // topic_container
            }
            html += "</div>";

            $("#snc-calendar-game-list").html(html);
        }

            if (!newsList || !newsList.length) {
                html = '<p style="text-align: center;">Valitud päeval uudiseid pole</p>';
                $("#snc-calendar-news-list").html(html);
                return;

                html += '<div class="article_topic calendar_news_topic" style="margin-left: 5px;">';
                html += '<a style="color: var(--color-main-accent-font)" href="/teema' + n.topic_alias + '">';
                html += (n.topic_name || "").toUpperCase();
                html += "</a></div>";

                if (n.comment_count && n.comment_count > 0) {
                    html += '<a href="/artikkel/' + n.alias + '/kommentaarid">';
                    html += '<span class="fa-stack fa-3x article_topic" style="padding: 2px 0; margin-left: 3px; font-size: 8px;">';
                    html += '<i class="fa fa-solid fa-message fa-stack-2x" style="color: var(--color-main-accent-font); font-size: 14px; padding: 1px 0;"></i>';
                    html += '<strong class="fa-stack-1x text-primary" style="color: var(--color-main-accent); font-size: 11px;">' + n.comment_count + "</strong>";
                    html += "</span>";
                    html += "</a>";
                }

                html += "</div>"; // header row

                html += "<h3 style='font-size: 14px; line-height: 18px;'>";
                html += '<a href="/artikkel/' + n.alias + '">' + n.headline + "</a>";
                html += "</h3>";

                html += "</div>"; // calendar_news
            }
            html += "</div>";

            $("#snc-calendar-news-list").html(html);
        }

        function buildMonthDropdown() {
            var html = "";
            for (var i = 0; i < monthNames.length; i++) {
                html += '<div class="snc-cal-select-option" data-month="' + i + '">' + monthNames[i] + '</div>';
            }
            $("#snc-cal-month-dropdown").html(html);
        }

        function buildYearDropdown() {
            var startYear = today.getFullYear() + 1;
            var endYear   = 2001;
            var html = "";
            for (var y = startYear; y >= endYear; y--) {
                html += '<div class="snc-cal-select-option" data-year="' + y + '">' + y + '</div>';
            }
            $("#snc-cal-year-dropdown").html(html);
        }

        // ---- LAZY INIT + AUTO-REFRESH FUNKTSIOONID ----

        function startAutoRefresh() {
            if (snCalState._refreshTimerId) return;

            snCalState._refreshTimerId = setInterval(function() {
                var overlayVisible = $('#snc-calendar-overlay').is(':visible');
                var isTodaySelected = (snCalState.selectedDate === todayStr);

                if (overlayVisible && isTodaySelected) {
                    // uuenda sama kuud, hoia valitud kuupäeva
                    loadMonth(snCalState.year, snCalState.month, true);
                }
            }, 60 * 1000);
        }

        // Teeme init-funktsiooni, mida kutsud openSoccernetCalendar() seest
        function initCalendarOnce() {
            if (snCalState._initialized) return;
            snCalState._initialized = true;

            buildMonthDropdown();
            buildYearDropdown();

            // Initial load
            loadMonth(snCalState.year, snCalState.month, false);

            // Auto-refresh taustal – teeb tegeliku päringu ainult siis,
            // kui kalender on lahti JA valitud on tänane kuupäev
            startAutoRefresh();
        }

        // Ekspordi globaalsetesse, et headeri skript saaks kutsuda
        window.sncCalendar = window.sncCalendar || {};
        window.sncCalendar.initOnce = initCalendarOnce;

        // NB! EI tee enam siin $(function(){ ... initial load ... });
        // Lazy-load ainult openSoccernetCalendar() kaudu.
