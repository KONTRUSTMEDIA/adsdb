const sourceText = document.getElementById('source-text');
const targetTextArea = document.getElementById('target-text');

const targetLang = document.getElementById('target-lang');

let typingTimer;
const doneTypingInterval = 500;

const translateText = () => {
const sourceValue = sourceText.value;
const sourceLangValue = sourceLang.value;
const targetLangValue = targetLang.value;

if (sourceValue.trim() === '') {
targetTextArea.value = '';
return;
}

const url = "https://translate-pa.googleapis.com/v1/translateHtml";
const cleanedText = sourceValue.replace(/"/g, '\\"').replace(/\n/g, '\\n');
const body = `[[["${cleanedText}"],"${sourceLangValue}","${targetLangValue}"],"te_lib"]`;

fetch(url, {
method: 'POST',
            headers: {
                "accept": "*/*",
                "accept-language": "fr-FR,fr;q=0.9,en-US;q=0.8,en;q=0.7",
                "content-type": "application/json+protobuf",
                "sec-ch-ua": "\"Not;A=Brand\";v=\"99\", \"Google Chrome\";v=\"139\", \"Chromium\";v=\"139\"",

                "x-goog-api-key": "AIzaSyATBXajvzQLTDHEQbcpq0Ihe0vWDHmO520"
            },
body: body
})
.then(response => {
if (!response.ok) {
throw new Error(`HTTP error! status: ${response.status}`);
}
return response.json();
})
.then(data => {
if (data && data && data) {
const translatedText = data[0];
let parser = new DOMParser();
let doc = parser.parseFromString(translatedText, 'text/html');
targetTextArea.value = doc.documentElement.textContent;
} else {
targetTextArea.value = 'Error: Could not find translated text in response.';
}
})
.catch(error => {
console.error('Error:', error);
targetTextArea.value = `Error: Translation failed. ${error.message}`;
});
};

sourceText.addEventListener('keyup', () => {
clearTimeout(typingTimer);

});

clearTimeout(typingTimer);
});
