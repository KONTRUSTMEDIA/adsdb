		// 계열사 조회
		function loadFamilyCo(){
			var strUrl = "/ko/common/load_family_co_01.jsp";
			var dataString = $("#srcFcForm").serialize();
			
			$.ajax({
				url: strUrl,
				type: "post",
				data: dataString,
				timeout: 5000,
				tryCount: 0,
				retryLimit: 3,
				error: function(xhr, status, e) {
					if (xhr.status == 500) {
						console.log('데이터 조회 중 오류.');						
					} else {
						this.tryCount++;
		
						if (this.tryCount <= this.retryLimit) {
							$.ajax(this);
						} else {
							console.log('데이터 조회 중 오류.');								
						}
					}
				},
				beforeSubmit: function() {						
				},
				success: function(data) {					
					$("#listBodyF01").html(data);				

					$('.linkFamily_a').click(showlinkFamilySub);
					$('.linkFamilySub_calcel_btn').click(hidelinkFamilySub);
					//console.log(data);		
					tryCount = 0;
				}
			});			
		}
