<script>
document.addEventListener('DOMContentLoaded', function () {
  // ルート
  //const URL_COUNT = "/notifications/unread-count";
  //const URL_LIST  = "/notifications";
  //const URL_READ  = "https://solispia.com/notifications/mark-as-read";

    const ORIGIN = 'https://solispia.com';

  const URL_COUNT = ORIGIN + '/notifications/unread-count';
  const URL_LIST  = ORIGIN + '/notifications';
  const URL_READ  = ORIGIN + '/notifications/mark-as-read';

  const CSRF      = document.querySelector('meta[name="csrf-token"]')?.getAttribute('content');

  function attachNotif(toggleId, menuId, badgeId, listId) {
    const toggle   = document.getElementById(toggleId);
    const menu     = document.getElementById(menuId);
    const badge    = document.getElementById(badgeId);
    const list     = document.getElementById(listId);
    const container= toggle ? toggle.closest('.notif, .notif-small') : null;
    const loading  = menu ? menu.querySelector('.notif-loading') : null;

    // ★ small 専用：閉じるボタン＆バックドロップ取得
    const closeBtn = menu ? menu.querySelector('.notif-close') : null;
    const backdrop = container ? container.querySelector('.notif-backdrop') : null;

    if (!toggle || !menu || !badge || !list || !container) return;

    let dataLoaded = false;   // 一覧プリフェッチ済みか
    let hasMarked  = false;   // 既読化済みか
    let openBy     = null;    // 'click' | 'hover' | null
    let hoverCloseTimer = null;

    // マウス環境のみホバー対応（タッチでは無効）
    const canHover = window.matchMedia('(hover: hover)').matches &&
                     window.matchMedia('(pointer: fine)').matches;

    function isOpen() {
      return menu.classList.contains('open');
    }

    function doOpen(reason) {
      // クリックで開いている状態をホバーで格下げしない
      if (isOpen() && openBy === 'click' && reason === 'hover') return;

      if (!isOpen()) {
        menu.classList.add('open');
        toggle.setAttribute('aria-expanded', 'true');
        if (!dataLoaded) fetchNotifications();
        if (!hasMarked)  markAsRead();

        // small（全幅パネル）のときはバックドロップ表示＆スクロールロック
        if (menu.classList.contains('notif-menu-small')) {
          if (backdrop) backdrop.style.display = 'block';
          document.documentElement.style.overflow = 'hidden';
          document.body.style.overflow = 'hidden';
        }
      }
      openBy = reason || openBy || 'click';
    }

    function doClose() {
      // 閉じる瞬間のトランジションを殺す
      menu.classList.add('closing');
      menu.offsetHeight; // reflow

      menu.classList.remove('open');
      toggle.setAttribute('aria-expanded', 'false');
      openBy = null;

      // small 用：バックドロップ & スクロールロック解除
      if (backdrop) backdrop.style.display = 'none';
      document.documentElement.style.overflow = '';
      document.body.style.overflow = '';

      requestAnimationFrame(() => {
        menu.classList.remove('closing');
      });
    }

    // バッジの表示/非表示をまとめて制御
    function setBadgeVisible(visible) {
  if (!badge) return;
  badge.setAttribute('aria-hidden', visible ? 'false' : 'true');

  badge.style.display = visible ? 'inline-block' : 'none';
    }

    function scheduleHoverClose() {

        if (openBy === 'hover' && isOpen()) doClose();

    }

    function cancelHoverClose() {
      if (hoverCloseTimer) {
        clearTimeout(hoverCloseTimer);
        hoverCloseTimer = null;
      }
    }

    // 要素内判定（display:none 等でサイズ0なら false）
    function pointIn(el, x, y) {
      if (!el) return false;
      const r = el.getBoundingClientRect();
      if (r.width === 0 && r.height === 0) return false;
      return (x >= r.left && x <= r.right && y >= r.top && y <= r.bottom);
    }

async function fetchUnreadCount() {
  try {
    const res  = await fetch(URL_COUNT, { credentials: 'include' });
    /*const res = await fetch(URL_COUNT, {
        method: 'GET',
        credentials: 'include'
    });*/

    const data = await res.json();
    const c    = Number(data.unread_count) || 0;
    setBadgeVisible(c > 0);
  } catch (_) {
    /* noop */
  }
}


    async function fetchNotifications() {
      const typeClass = (t) => {
        switch (t) {
          case 'comment':       return 'is-comment';
          case 'commentReply':  return 'is-commentReply';
          case 'review':        return 'is-review';
          case 'followTitle':   return 'is-followTitle';
          case 'followUser':    return 'is-followUser';
          case 'announcement':  return 'is-announcement';
          default:              return '';
        }
      };

      const formatMonthDay = (s) => {
        if (!s) return '';
        const m = s.match(/^(\d{4})-(\d{1,2})-(\d{1,2})/);
        if (m) return `${parseInt(m[2],10)}月${parseInt(m[3],10)}日`;
        const d = new Date(s);
        return isNaN(d.getTime()) ? '' : `${d.getMonth()+1}月${d.getDate()}日`;
      };

      const extractActor = (item) => {
        if (item.actor) return item.actor;
        const t = item.text || '';
        const idx = t.indexOf('：');
        return idx > -1 ? t.slice(0, idx) : '';
      };

      try {
        if (loading) loading.style.display = 'block';

        const res = await fetch(URL_LIST, {
  method: 'GET',
  credentials: 'include'
});

        const data  = await res.json();
        const items = data.items || [];

        list.innerHTML = '';
        if (items.length === 0) {
          list.innerHTML = '<li class="notif-item"><span class="notif-text">通知はありません</span></li>';
          return;
        }

        const frag = document.createDocumentFragment();

        items.forEach(item => {
          const li = document.createElement('li');
          li.className = 'notif-item';

          const a = document.createElement('a');
          a.className = 'notif-link';
          a.href = item.url || '#';

          const head = document.createElement('div');
          head.className = 'message-head';

          const label = document.createElement('span');
          label.className = `message-label ${typeClass(item.type)}`;
          label.textContent = item.label || '通知';

          const actorRaw = extractActor(item);
          const from  = document.createElement('span');
          from.className = 'message-from';
          from.textContent = actorRaw || '';

          const time = document.createElement('time');
          time.className = 'message-date';
          time.textContent = formatMonthDay(item.created_at || '');

          head.appendChild(label);
          if (from.textContent) head.appendChild(from);
          head.appendChild(time);

          const body = document.createElement('div');
          body.className = 'message-body';
          if (item.content && String(item.content).trim().length > 0) {
            const p = document.createElement('div');
            p.className = 'message-text';
            p.textContent = item.content;
            body.appendChild(p);
          }

          a.appendChild(head);
          a.appendChild(body);
          li.appendChild(a);
          frag.appendChild(li);
        });

        list.appendChild(frag);
        dataLoaded = true;
      } catch (e) {
        list.innerHTML = '<li class="notif-item"><span class="notif-text">読み込みに失敗しました</span></li>';
      } finally {
        if (loading) loading.style.display = 'none';
      }
    }

   async function markAsRead() {
  try {
    if (!CSRF) return;
    await fetch(URL_READ, {
      method: 'POST',
      credentials: 'include',
      headers: {
        'Content-Type': 'application/json',
        'X-CSRF-TOKEN': CSRF
      },
      body: JSON.stringify({})
    });
    setBadgeVisible(false);
    hasMarked = true;
  } catch (_) {
    /* noop */
  }
}


    fetchUnreadCount();
    fetchNotifications();

    toggle.addEventListener('click', (ev) => {
      ev.preventDefault();
      ev.stopPropagation();

      if (!isOpen()) {
        doOpen('click');
        return;
      }
      if (openBy === 'hover') {
        openBy = 'click'; // ピン留め
        return;
      }
      doClose(); // openBy === 'click'
    });

    document.addEventListener('click', (e) => {
      if (!menu.contains(e.target) && !toggle.contains(e.target)) {
        if (isOpen()) doClose();
      }
    });

    document.addEventListener('keydown', (e) => {
      if (e.key === 'Escape' && isOpen()) doClose();
    });

    if (closeBtn) {
      closeBtn.addEventListener('click', (e) => {
        e.preventDefault();
        e.stopPropagation();
        if (isOpen()) doClose();
      });
    }
    if (backdrop) {
      backdrop.addEventListener('click', (e) => {
        e.preventDefault();
        if (isOpen()) doClose();
      });
    }

    if (canHover) {
      toggle.addEventListener('mouseenter', () => {
        doOpen('hover');
        cancelHoverClose();
      });

      const onPointerMove = (ev) => {
        if (!isOpen()) return;

        const x = ev.clientX, y = ev.clientY;
        const insideToggle = pointIn(toggle, x, y);
        const insideMenu   = pointIn(menu,   x, y);

        if (insideToggle || insideMenu) {
          cancelHoverClose();
        } else {
          if (openBy === 'hover') scheduleHoverClose(10);
        }
      };

      document.addEventListener('pointermove', onPointerMove, { passive: true });

      document.addEventListener('pointerleave', () => {
        if (openBy === 'hover') scheduleHoverClose(10);
      });

      window.addEventListener('resize', () => {
        if (!isOpen() || openBy !== 'hover') return;
        scheduleHoverClose(10);
      });
    }
  }

  attachNotif('notifToggle',   'notifMenu',   'notifBadge',   'notifItems');
  attachNotif('notifToggleSm', 'notifMenuSm', 'notifBadgeSm', 'notifItemsSm');
});
