        <script>
            (function(){

                const STORAGE_KEY = "bottomAdLastShown";
                const TWELVE_HOURS = 1 * 60 * 60 * 1000;

                function shouldShowAd() {
                    const lastShown = localStorage.getItem(STORAGE_KEY);
                    if (!lastShown) return true;

                    const now = Date.now();
                    return (now - parseInt(lastShown)) > TWELVE_HOURS;
                }

                function showAd() {
                    const ad = document.getElementById("bottomAd");
                    ad.classList.add("showAd");
                    localStorage.setItem(STORAGE_KEY, Date.now().toString());
                }

                window.addEventListener("load", function() {
                    if (shouldShowAd()) {
                        setTimeout(showAd, 800);
                        (adsbygoogle = window.adsbygoogle || []).push({});
                    }
                });

                document.getElementById("closeAd").addEventListener("click", function() {
                    document.getElementById("bottomAd").style.display = "none";
                });

            })();
