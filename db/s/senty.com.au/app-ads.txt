    <script>
      (function () {
        var prefersReduced = window.matchMedia('(prefers-reduced-motion: reduce)').matches;

        /* ---- Scroll Reveal ---- */
        var srEls = document.querySelectorAll('.sr');
        if (!prefersReduced && 'IntersectionObserver' in window) {
          var io = new IntersectionObserver(function (entries) {
            entries.forEach(function (e) {
              if (e.isIntersecting) {
                e.target.classList.add('visible');
                io.unobserve(e.target);
              }
            });
          }, { threshold: 0.12 });
          srEls.forEach(function (el) { io.observe(el); });
        } else {
          srEls.forEach(function (el) { el.classList.add('visible'); });
        }

        /* ---- Current Year ---- */
        var yearEl = document.getElementById('current-year');
        if (yearEl) {
          yearEl.textContent = new Date().getFullYear();
        }

        /* ---- Animated Counter ---- */
        var countEl = document.querySelector('[data-count]');
        if (countEl) {
          try {
            var target = parseInt(countEl.getAttribute('data-count'), 10);
            if (isNaN(target)) throw new Error('Invalid data-count value');
            
            var counted = false;
            var countIO = new IntersectionObserver(function (entries) {
              if (entries[0].isIntersecting && !counted) {
                counted = true;
                var duration = 1200;
                var startTime = null;
                function tick(ts) {
                  try {
                    if (!startTime) startTime = ts;
                    var progress = Math.min((ts - startTime) / duration, 1);
                    var eased = 1 - Math.pow(1 - progress, 3);
                    var current = Math.floor(eased * target);
                    countEl.textContent = current + '+';
                    if (progress < 1) requestAnimationFrame(tick);
                  } catch (err) {
                    console.error('Counter animation tick error:', err);
                    countEl.textContent = target + '+';
                  }
                }
                requestAnimationFrame(tick);
                countIO.unobserve(countEl);
              }
            }, { threshold: 0.5 });
            countIO.observe(countEl);
          } catch (err) {
            console.error('Counter setup error:', err);
            var fallbackCount = countEl.getAttribute('data-count');
            countEl.textContent = fallbackCount ? fallbackCount + '+' : '40+';
          }
        }

        /* ---- Hero Glow Follow ---- */
        var hero = document.querySelector('.hero');
        var heroGlow = document.getElementById('heroGlow');
        if (hero && heroGlow && !prefersReduced) {
          hero.addEventListener('mousemove', function (e) {
            var rect = hero.getBoundingClientRect();
            heroGlow.style.left = (e.clientX - rect.left) + 'px';
            heroGlow.style.top = (e.clientY - rect.top) + 'px';
          });
        }

        /* ---- Card Mouse Glow ---- */
        var cards = document.querySelectorAll('.card');
        cards.forEach(function (card) {
          card.addEventListener('mousemove', function (e) {
            var rect = card.getBoundingClientRect();
            var x = ((e.clientX - rect.left) / rect.width * 100).toFixed(1);
            var y = ((e.clientY - rect.top) / rect.height * 100).toFixed(1);
            card.style.setProperty('--mouse-x', x + '%');
            card.style.setProperty('--mouse-y', y + '%');
          });
        });

        /* ---- Particle Canvas ---- */
        if (prefersReduced) return;

        var canvas = document.getElementById('bg-canvas');
        if (!canvas) return;
        var ctx = canvas.getContext('2d');
        if (!ctx) return;

        var particles = [];
        var mouse = { x: -9999, y: -9999 };
        var rafId;
        var running = false;

        function resize() {
          canvas.width = window.innerWidth;
          canvas.height = window.innerHeight;
          initParticles();
        }

        function initParticles() {
          var count = Math.max(30, Math.floor((canvas.width * canvas.height) / 18000));
          count = Math.min(count, 80);
          particles = [];
          for (var i = 0; i < count; i++) {
            particles.push({
              x: Math.random() * canvas.width,
              y: Math.random() * canvas.height,
              vx: (Math.random() - 0.5) * 0.3,
              vy: (Math.random() - 0.5) * 0.3,
              r: Math.random() * 1.8 + 0.6,
              baseAlpha: Math.random() * 0.5 + 0.3
            });
          }
        }

        function draw() {
          running = true;
          ctx.clearRect(0, 0, canvas.width, canvas.height);
          var threshold = Math.min(180, canvas.width * 0.16);

          for (var i = 0; i < particles.length; i++) {
            var p = particles[i];

            // Mouse attraction
            var mdx = mouse.x - p.x;
            var mdy = mouse.y - p.y;
            var mdist = Math.sqrt(mdx * mdx + mdy * mdy);
            if (mdist < 200 && mdist > 1) {
              var force = (200 - mdist) / 200 * 0.015;
              p.vx += (mdx / mdist) * force;
              p.vy += (mdy / mdist) * force;
            }

            // Damping
            p.vx *= 0.998;
            p.vy *= 0.998;

            p.x += p.vx;
            p.y += p.vy;

            if (p.x < -30) p.x = canvas.width + 30;
            if (p.x > canvas.width + 30) p.x = -30;
            if (p.y < -30) p.y = canvas.height + 30;
            if (p.y > canvas.height + 30) p.y = -30;

            ctx.beginPath();
            ctx.fillStyle = 'rgba(120, 180, 255,' + p.baseAlpha + ')';
            ctx.arc(p.x, p.y, p.r, 0, Math.PI * 2);
            ctx.fill();

            // Connections
            for (var j = i + 1; j < particles.length; j++) {
              var p2 = particles[j];
              var dx = p.x - p2.x;
              var dy = p.y - p2.y;
              var dist = Math.sqrt(dx * dx + dy * dy);
              if (dist < threshold) {
                var alpha = (1 - dist / threshold) * 0.15;
                ctx.strokeStyle = 'rgba(100, 160, 255,' + alpha + ')';
                ctx.lineWidth = 0.8;
                ctx.beginPath();
                ctx.moveTo(p.x, p.y);
                ctx.lineTo(p2.x, p2.y);
                ctx.stroke();
              }
            }
          }

          rafId = requestAnimationFrame(draw);
        }

        resize();
        window.addEventListener('resize', resize, { passive: true });

        document.addEventListener('mousemove', function (e) {
          mouse.x = e.clientX;
          mouse.y = e.clientY;
        }, { passive: true });

        draw();

        document.addEventListener('visibilitychange', function () {
          if (document.hidden && rafId) {
            cancelAnimationFrame(rafId);
            running = false;
          } else if (!document.hidden && !running) {
            draw();
          }
        });
      })();
    </script>
