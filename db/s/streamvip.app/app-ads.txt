    function loginUser  () {
        
        var email = $('#emailLogin').val();
        var password = $('#passwordLogin').val();

        if (email != '' && password != '') {
            $('#loginAlert').text('');
            fetch('https://streamvip.app/beta/api.php?acao=login', {
                method: "post",
                credentials: 'include',
                body: JSON.stringify({
                    email: email, 
                    password: password
                })
            }).then(function(response) {
                return response.json();
            }).then(function(data) {
                if (data.success == 1) {
                    setTimeout("location.href=''", 500);
                } else {
                    if (data.error) {
                        $('#loginAlert').html('<div class="alert alert-danger">'+data.error+'</div>');
                    }
                }
            }).catch((error) => {
                $('#loginAlert').html('<div class="alert alert-danger">Ocorreu um erro. Tente novamente!</div>');
            });
        } else {
            $('#loginAlert').html('<div class="alert alert-danger">Preencha todos os campos!</div>');
        }
    
        return false;
    }
