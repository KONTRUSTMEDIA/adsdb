		}
		function onPlayerStateChange(event) {
			if (event.data == YT.PlayerState.ENDED) {
				nextVideo();
			}
		}
		function nextVideo() {
			kssPlayer.stopVideo();
			var i = 0;
			while (i < kssVideos.length) {
				kssCurrentVideo += 1;
				if (kssCurrentVideo == kssVideos.length) {
					kssCurrentVideo = 0;
				}
				if (isCurrentVideoExpired()) {
					i += 1;
					next;
				}
				else {
					break;
				}
			}
			if (i < kssVideos.length) {
				kssPlayer.loadVideoById(kssVideos[kssCurrentVideo].v);
			}
		}
		function isCurrentVideoExpired() {
			return false;
		}
