        console.warn("Element not found.");
        return;

    element.classList.add("toggleable");

    if (isHidden) {
        element.classList.remove("hidden");
        element.classList.add("visible");
        element.style.display = "block"; 
        setTimeout(() => {
            element.style.height = "auto";
        }, duration);
