                Photos.$list.on('click', '.item', Photos.clickListImage);

                // Bind Prev/Next click
                Photos.$prev.on('click', Photos.prev);
                Photos.$next.on('click', Photos.next);

                // Bind scroll
                Photos.$list.on('scroll', Photos.scroll);
              },

              //
              // Hide Loading
              //
              hideLoading: function() {
                Photos.$frame.find('.loading').hide();
              },

              //
              // Image Click
              //
              clickListImage: function () {
                let $this = $(this);
                let $source = $this.find('figure').clone();

                // Replace Main Element
                Photos.$frame.find('.loading').show();
                $source.find('img').on('load',Photos.hideLoading);
                Photos.$frame.find('.main-photo figure').replaceWith($source);

                // Change "active" class
                Photos.$list.find('.active').removeClass('active');
                $this.addClass('active');
              },

              //
              // Prev
              //
              prev: function (e) {
                let position = Photos.$list.scrollLeft() || 0;
                let margin = parseInt(Photos.$list.find('.thumb-box:last').css('margin-left'));

                Photos.$list.animate({
                  scrollLeft: position - Photos.$list.find('.thumb-box').width() - margin,
                });
                e.preventDefault();
                return false;
              },

              //
              // Next
              //
              next: function (e) {
                let position = Photos.$list.scrollLeft() || 0;
                let margin = parseInt(Photos.$list.find('.thumb-box:last').css('margin-left'));

                Photos.$list.animate({
                  scrollLeft: position + Photos.$list.find('.thumb-box').width() + margin,
                });
                e.preventDefault();
                return false;
              },

              //
              // List Scroll
              //
              scroll: function () {
                let pos = Math.ceil(Photos.$list.width() + Photos.$list.scrollLeft());
                (Photos.$list.scrollLeft() <= 0) ? Photos.$prev.hide() : Photos.$prev.show();
                (pos >= Photos.$list.get(0).scrollWidth) ? Photos.$next.hide() : Photos.$next.show();
              }
            };

            Photos.onReady();
          });
		</script>
