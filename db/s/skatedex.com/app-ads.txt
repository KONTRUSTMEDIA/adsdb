function showNearbyParks(lat,lng){
const out=document.getElementById('nearResults');
const radius=parseInt(document.getElementById('nearRadius').value)||25;
const haversine=(lt1,lg1,lt2,lg2)=>{const R=3959,dLat=(lt2-lt1)*Math.PI/180,dLng=(lg2-lg1)*Math.PI/180,a=Math.sin(dLat/2)**2+Math.cos(lt1*Math.PI/180)*Math.cos(lt2*Math.PI/180)*Math.sin(dLng/2)**2;return R*2*Math.atan2(Math.sqrt(a),Math.sqrt(1-a));};
const nearby=D.p.filter(p=>p.lt&&p.lg).map(p=>({pk:p,dist:haversine(lat,lng,p.lt,p.lg)})).filter(x=>x.dist<=radius).sort((a,b)=>a.dist-b.dist);
if(!nearby.length){out.innerHTML='<p>No skateparks found within '+radius+' miles. Try increasing the radius.</p>';return;}
window._nearbyParks=nearby;window._nearbyShown=20;
renderNearbyBatch(nearby,20,out,radius);
}
function renderNearbyBatch(nearby,show,out,radius){

const hasMore=show<nearby.length;
const items=nearby.slice(0,show).map(({pk,dist})=>{
const thumb=pk.im&&pk.im[0]?'<img class="near-thumb" src="'+pk.im[0]+'" alt="'+pk.n+'" onerror="this.style.background=\'var(--brand)\';this.style.display=\'flex\'">':'<div class="near-thumb" style="background:linear-gradient(135deg,var(--brand),var(--brand-dark));display:flex;align-items:center;justify-content:center;color:white;font-size:18px">🛹</div>';
return '<a href="/'+pk.u+'" class="near-item" style="text-decoration:none;color:inherit">'+thumb+'<div class="near-info"><div class="near-name">'+pk.n+'</div><div class="near-loc">'+(pk.a||gL(pk))+'</div></div><div class="near-dist">'+dist.toFixed(1)+' mi</div></a>';
}).join('');
const footer=hasMore
?'<div style="margin-top:16px;display:flex;flex-direction:column;align-items:center;gap:10px">'
+'<p style="font-size:13px;color:var(--gray-1);margin:0">Showing <strong>'+showing+'</strong> of <strong>'+nearby.length+'</strong> parks</p>'
+'<button id="nearLoadMore" onclick="loadMoreNearby()" style="padding:12px 32px;background:var(--brand);color:white;border:none;border-radius:100px;font-size:14px;font-weight:600;cursor:pointer;font-family:var(--fb);transition:var(--tr)">Show '+Math.min(20,nearby.length-show)+' more parks</button>'
+'</div>'

out.innerHTML='<div style="font-size:14px;font-weight:600;margin-bottom:8px;color:var(--brand)">'+nearby.length+' parks within '+radius+' miles</div><div class="near-results">'+items+'</div>'+footer;
if(hasMore&&showing>20)document.getElementById('nearLoadMore')?.scrollIntoView({behavior:'smooth',block:'nearest'});
}
