            const currentPath = window.location.pathname.replace(/^\/+|\/+$/g, ''); // Clean slashes

            // Remove existing language prefix (check from longest to shortest to avoid partial matches)
            let cleanPath = currentPath;
            for (let code of langCodes) {
                if (currentPath === code || currentPath.startsWith(code + '/')) {
                    cleanPath = currentPath.substring(code.length).replace(/^\/+/, '');

                newUrl = baseUrl + (cleanPath ? cleanPath : '');

                // Other languages - add prefix

            }
            
            // Redirect
