			function showAdScreen() {
                //--Hide start screen
                var x = document.getElementById('startScreen');
                x.style.display = 'none';
                //--Hide death screen
                var x = document.getElementById('deathScreen');
                x.style.display = 'none';
                //--Show ad screen
                x = document.getElementById('adScreen');
                x.style.display = 'block';
            }

			function showAd() {
                showAdScreen();

                //Start ad
                try {
                    aiptag.cmd.player.push(function() { adplayer.startPreRoll(); });
                }
                catch(err) {
                    console.log(err);
                    adFinished();
                }
            }

            function adFinished() {
                //--Hide ad screen
                var x = document.getElementById('adScreen');
                x.style.display = 'none';
                //--Show death screen
                x = document.getElementById('deathScreen');
                x.style.display = 'block';
            }

            var deathsSinceLastAdvert = 0;
			var timeLastAdShown = new Date();
			var showTheAd = false;
            function showDeathScreenCallback() {
                $('.navbar').fadeIn(2000);

                deathsSinceLastAdvert += 1;
                var timeNow = new Date();

                if(deathsSinceLastAdvert >= 3 && timeNow-timeLastAdShown > (2*60*1000)){
                    showTheAd = true;
                    deathsSinceLastAdvert = 0;
                    timeLastAdShown = timeNow;
				}
				else{
                    showTheAd = false;
				}

                
                if(showTheAd){
                    //Show adinplay ad
                    showAdScreen();

                    
                }
                else{
                    

                    

                    
                        if (!window.adsense_shown){
                            (adsbygoogle = window.adsbygoogle || []).push({});
                            window.adsense_shown = true;
                            console.log('Adsense shown');
                        }
                    
				}
                
            }

            function fadeFinishedCallback() {
				//Only play the adinplay ad once the gameover fade has finished
				
                if(showTheAd){
                    showAd();
                }
				
            }

                <script>
                    function calculateLevel(exp) {
                        var lvl = Math.floor(Math.sqrt(exp)/5);
                        return lvl;
                    }

                    function setLeaderboard(values) {

                        //Make sure leaderboard is the correct length
                        generateLeaderboardOfSize(values.length);

                        for(var i=0; i < values.length; i++){

                            var rowSelector = '#leaderboard-'+(i+1)

                            //Check whether this user has an account
                            if(values[i].userid){
                                //Add userid attribute
                                $(rowSelector).attr("userid", values[i].userid);
                                $(rowSelector).attr("title", "Visit profile (opens in new tab)");

                                //Change style
                                $(rowSelector).removeClass('guest-user');
                                $(rowSelector).addClass('signed-in-user');

                                //Calculate users level
                                var expVal = values[i].exp
                                if(values[i].isplayer){expVal = window.exp;}
                                $(rowSelector + ' .leaderboard-lvl-value').text(calculateLevel(expVal));
                                $(rowSelector + ' .leaderboard-lvl').show();
                            }
                            else{
                                //Add normal text, remove level indicator and link
                                $(rowSelector + ' .leaderboard-name').text(values[i].name);
                                $(rowSelector + ' .leaderboard-lvl-value').text("-");
                                $(rowSelector + ' .leaderboard-lvl').hide();
                                $(rowSelector).removeAttr("userid");
                                $(rowSelector).removeAttr("title");
                                $(rowSelector).removeClass('signed-in-user');
                                $(rowSelector).addClass('guest-user');
                            }

                            //Update name and score
                            $(rowSelector + ' .leaderboard-name').text(values[i].name);
                            $(rowSelector + ' .leaderboard-score').text(values[i].score);

                            //Highlight user's position
                            if(values[i].isplayer){
                                $(rowSelector).addClass('currentPlayer');
                            }
                            else{
                                $(rowSelector).removeClass('currentPlayer');
                            }
                        }
                    }

                    function generateLeaderboardOfSize(n){
                        //Check current size
                        var rowCount = $('.leaderboard tr').length;

                        var k = n - rowCount;

                        while(k < 0){
                            //Remove extra rows
                            $('.leaderboard tr:last').remove();
                            k += 1;
                        }

                        while(k > 0){
                            //Add extra rows
							var index = ((n-k)+1);
							var newRow = '<tr id="leaderboard-'+index+'" style="display: flex; width: 100%" onclick="if(this.attributes.userid){window.open(\'profile?user=\'+this.attributes.userid.value, \'_blank\');}">' +
                                '<th scope="row">'+index+'</th>' +
                                '<td class="leaderboard-lvl-col"><span class="leaderboard-lvl">Lvl</span><span class="leaderboard-lvl-value">5</span></td>' +
                                '<td class="leaderboard-name">Name</td>' +
                                '<td class="leaderboard-score">0</td>' +
                                '</tr>';
							if($('.leaderboard tr:last').length == 0){
                                $('.leaderboard table').append(newRow);
							}
							else{
                                $('.leaderboard tr:last').after(newRow);
							}
                            k -= 1;
                        }
                    }
