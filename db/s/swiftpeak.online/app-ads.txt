  <script>
    // Auto-fill if remembered
    window.addEventListener('DOMContentLoaded', () => {
      const saved = localStorage.getItem('hf_remember');
      if (saved) { const d = JSON.parse(saved); document.getElementById('username').value = d.u; document.getElementById('rememberMe').checked = true; }
      // If already logged in, redirect
      if (localStorage.getItem('hf_token')) { window.location.href = 'dashboard.html'; }
    });
    async function handleLogin(e) {
      e.preventDefault();
      const btn = document.getElementById('loginBtn');
      const spinner = document.getElementById('loadingSpinner');
      const errMsg = document.getElementById('errorMsg');
      btn.disabled = true;
      spinner.style.display = 'inline-block';
      errMsg.style.display = 'none';
      const username = document.getElementById('username').value;
      const password = document.getElementById('password').value;
      try {
        const resp = await fetch('/api/auth/login', { method:'POST', headers:{'Content-Type':'application/json'}, body: JSON.stringify({username, password}) });
        const data = await resp.json();
        if (!resp.ok) throw new Error(data.error || 'Login failed');
        localStorage.setItem('hf_token', data.token);
        localStorage.setItem('hf_user', JSON.stringify(data.user));
        if (document.getElementById('rememberMe').checked) { localStorage.setItem('hf_remember', JSON.stringify({u:username})); }
        else { localStorage.removeItem('hf_remember'); }
        window.location.href = 'dashboard.html';
      } catch(err) {
        errMsg.textContent = err.message;
        errMsg.style.display = 'block';
        btn.disabled = false;
        spinner.style.display = 'none';
      }
    }
  </script>
