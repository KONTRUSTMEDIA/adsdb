    <script>
      // Attach touch/click handlers after button is in DOM
      (function() {
        var btn = document.getElementById('html-play-now-btn');
        if (!btn) {
          console.error('[HTML] PLAY NOW button not found!');
          return;
        }

        function handlePlayNow(e) {
          console.log('[HTML] PLAY NOW event:', e.type);
          e.preventDefault();
          e.stopPropagation();

          // Try the JS callback first
          if (window.__crashblockPlayNow) {
            console.log('[HTML] Calling __crashblockPlayNow');
            window.__crashblockPlayNow();
          } else {
            // Fallback: dispatch custom event that JS can catch
            console.log('[HTML] Dispatching play-now-click event');
            window.dispatchEvent(new CustomEvent('play-now-click'));
          }
          return false;
        }

        // Use capture phase to ensure we get the event first
        btn.addEventListener('touchstart', handlePlayNow, { capture: true, passive: false });
        btn.addEventListener('touchend', function(e) { e.preventDefault(); }, { capture: true, passive: false });
        btn.addEventListener('click', handlePlayNow, { capture: true });

        console.log('[HTML] PLAY NOW button handlers attached');
      })();
    </script>
