			<script>
			document.addEventListener('DOMContentLoaded', function() {
				var wrapper = document.querySelector('.whatsapp-action-wrapper');
				var button = document.querySelector('.whatsapp-action-button');
				var menu = document.getElementById('salesMenu');
				
				function isInsideWhatsAppArea(element) {
					if (!element) return false;

					if (button && (button === element || button.contains(element))) {
						return true;
					}
					if (menu && (menu === element || menu.contains(element))) {
						return true;
					}
					if (wrapper && (wrapper === element || wrapper.contains(element))) {
						return true;
					}
					
					return false;
				}

				function toggleSalesMenu(event) {
					if (event) {
						event.preventDefault();
						event.stopPropagation();
						event.stopImmediatePropagation();
					}
					var menu = document.getElementById('salesMenu');
					menu.classList.toggle('show');
				}

				function openWhatsApp(phoneNumber, salesName, event) {
					if (event) {
						event.preventDefault();
						event.stopPropagation();
						event.stopImmediatePropagation();
					}
					
					var message = "Halo, saya tertarik dengan produk Serbaselamat. Bisa bantu saya?";
					var encodedMessage = encodeURIComponent(message);
					var whatsappUrl = "https://wa.me/" + phoneNumber + "?text=" + encodedMessage;
					setTimeout(function() {
						window.open(whatsappUrl, '_blank');
						document.getElementById('salesMenu').classList.remove('show');
					}, 50);
				}

				document.addEventListener('mousedown', function(event) {
					var menu = document.getElementById('salesMenu');
					
					if (!menu || !menu.classList.contains('show')) {
						return;
					}
					if (isInsideWhatsAppArea(event.target)) {
						return;
					}
					menu.classList.remove('show');
				});

				window.toggleSalesMenu = toggleSalesMenu;
				window.openWhatsApp = openWhatsApp;
			});
			</script>
