<script>
// Blog Slider JavaScript
let currentSlide = 0;
let slidesPerView = 4;
let totalSlides = 0;

// Initialize slider
document.addEventListener('DOMContentLoaded', function() {
    const slider = document.getElementById('blogSlider');
    if (!slider) return;
    
    const slides = slider.children;
    totalSlides = slides.length;
    
    // Set slides per view based on screen size
    updateSlidesPerView();
    window.addEventListener('resize', updateSlidesPerView);
    
    // Create dots
    createDots();
    
    // Auto-slide every 5 seconds
    setInterval(autoSlide, 5000);
});

function updateSlidesPerView() {
    const width = window.innerWidth;
    if (width <= 480) {
        slidesPerView = 1;
    } else if (width <= 768) {
        slidesPerView = 2;
    } else if (width <= 1024) {
        slidesPerView = 3;
    } else {
        slidesPerView = 4;
    }
    updateSliderPosition();
}

function slideBlogs(direction) {
    const maxSlide = Math.max(0, totalSlides - slidesPerView);
    
    if (direction === 'next') {
        currentSlide = currentSlide >= maxSlide ? 0 : currentSlide + 1;
    } else {
        currentSlide = currentSlide <= 0 ? maxSlide : currentSlide - 1;
    }
    
    updateSliderPosition();
    updateDots();
}

function updateSliderPosition() {
    const slider = document.getElementById('blogSlider');
    if (slider) {
        const slideWidth = slider.parentElement.offsetWidth / slidesPerView;
        const translateX = -currentSlide * slideWidth;
        slider.style.transform = `translateX(${translateX}px)`;
    }
}

function createDots() {
    const dotsContainer = document.getElementById('sliderDots');
    if (!dotsContainer) return;
    
    const maxSlide = Math.max(0, totalSlides - slidesPerView);
    dotsContainer.innerHTML = '';
    
    for (let i = 0; i <= maxSlide; i++) {
        const dot = document.createElement('div');
        dot.className = i === 0 ? 'dot active' : 'dot';
        dot.onclick = () => goToSlide(i);
        dotsContainer.appendChild(dot);
    }
}

function updateDots() {
    const dots = document.querySelectorAll('.dot');
    dots.forEach((dot, index) => {
        dot.classList.toggle('active', index === currentSlide);
    });
}

function goToSlide(slideIndex) {
    currentSlide = slideIndex;
    updateSliderPosition();
    updateDots();
}

function autoSlide() {
    slideBlogs('next');
}

// Touch/swipe support for mobile
let startX = 0;
let endX = 0;

document.addEventListener('touchstart', e => {
    startX = e.changedTouches[0].screenX;
});

document.addEventListener('touchend', e => {
    endX = e.changedTouches[0].screenX;
    handleSwipe();
});

function handleSwipe() {
    const threshold = 50;
    const diff = startX - endX;
    
    if (Math.abs(diff) > threshold) {
        if (diff > 0) {
            slideBlogs('next');
        } else {
            slideBlogs('prev');
        }
    }
}
</script>

<script>
document.addEventListener('DOMContentLoaded', function() {
  // Get form elements
  const form = document.getElementById('myForm');
  const urlInput = document.getElementById('url');
  const alertError = document.getElementById('alert-error');
  const msgError = document.getElementById('msg-error');
  const pasteButton = document.getElementById('paste');
  const downloadButton = document.getElementById('submiturl');
  const progressBar = document.getElementById('progress-bar');
  
  // Pinterest domain patterns (flexible for future changes)
  const pinterestPatterns = [
    // Current Pinterest domains
    'pinterest.com',
    'pinterest.co.uk',
    'pinterest.fr',
    'pinterest.de',
    'pinterest.jp',
    'pinterest.ca',
    'pinterest.au',
    'pinterest.es',
    'pinterest.it',
    'pinterest.pt',
    'pinterest.mx',
    'pinterest.br',
    'pinterest.in',
    'pinterest.kr',
    'pinterest.ru',
    // Pinterest short URLs
    'pin.it',
    'pinterest.app.link',
    // Pinterest subdomains (flexible)
    '.pinterest.com',
    '.pinterest.',
    // Pinterest image/video CDN
    'pinimg.com',
    'i.pinimg.com',
    'v1.pinimg.com'
  ];

  // Explicitly blocked competitor domains
  const blockedDomains = [
    'pinterestdownloader.com',
    'botdownloader.com',
    'pinterestvideodownloader.com',
    'savepin.app',
    'expertsphp.com',
    'dotsave.app',
    'instagram.com',
    'facebook.com',
    'youtube.com',
    'youtu.be',
    'tiktok.com',
    'likee',
    'google.com',
    'play.google.com',
    'crunchyroll.com',
    'sonyliv.com',
    'onlineconverter.com',
    'vkvideo.ru',
    'xxxbp.tv',
    'erovizor.top',
    't.me',
    'x.com',
    'twitter.com'
  ];

  // Function to extract and clean URLs from input
  function extractAndCleanUrl(input) {
    // Remove extra spaces, newlines, and special characters
    input = input.trim().replace(/\s+/g, ' ').replace(/[^\x00-\x7F]/g, '');
    
    // Handle common user mistakes
    input = input.replace(/^https:\/\/https:\/\//, 'https://'); // Double https
    input = input.replace(/^http:\/\/https:\/\//, 'https://'); // Mixed protocols
    input = input.replace(/^www\.https:\/\//, 'https://www.'); // Wrong order
    
    // Pattern to match URLs (more flexible)
    const urlPattern = /https?:\/\/[^\s<>"{}|\\^`\[\]]+/g;
    const matches = input.match(urlPattern);
    
    if (!matches || matches.length === 0) {
      // Try to detect URLs without protocol
      const partialUrlPattern = /(www\.)?[a-zA-Z0-9][a-zA-Z0-9-]{1,61}[a-zA-Z0-9]\.[a-zA-Z]{2,}[^\s]*/g;
      const partialMatches = input.match(partialUrlPattern);
      
      if (partialMatches && partialMatches.length > 0) {
        // Add https:// to the first match
        return { url: 'https://' + partialMatches[0], multipleUrls: false };
      }
      
      return { url: null, multipleUrls: false };
    }
    
    // Check if multiple URLs were detected
    const multipleUrls = matches.length > 1;
    
    // Get the first URL and clean it
    let firstUrl = matches[0];
    
    // If multiple URLs detected, try to extract just the first complete URL
    if (multipleUrls || input.includes('https://') && input.lastIndexOf('https://') !== input.indexOf('https://')) {
      // Find where the first URL should end by looking for the start of the next URL
      const firstHttpIndex = input.indexOf('http');
      const secondHttpIndex = input.indexOf('http', firstHttpIndex + 4); // Start search after 'http'
      
      if (secondHttpIndex !== -1) {
        // Extract only the first URL
        firstUrl = input.substring(firstHttpIndex, secondHttpIndex).trim();
      }
    }
    
    // Clean common trailing issues
    firstUrl = firstUrl.replace(/[,\s\]}>)]+$/, '');
    firstUrl = firstUrl.replace(/\?.*utm_.*$/, ''); // Remove UTM parameters
    firstUrl = firstUrl.replace(/&.*$/, ''); // Remove additional parameters after &
    
    // Additional cleaning for concatenated URLs
    firstUrl = firstUrl.replace(/https?:\/\/.*$/, function(match) {
      // If the match contains another http, cut it off at that point
      const nextHttp = match.indexOf('http', 7); // Start search after 'http://'
      if (nextHttp !== -1) {
        return match.substring(0, nextHttp);
      }
      return match;
    });
    
    return { url: firstUrl, multipleUrls: multipleUrls };
  }

  // Function to check if domain is Pinterest
  function isPinterestDomain(hostname) {
    hostname = hostname.toLowerCase();
    
    return pinterestPatterns.some(pattern => {
      if (pattern.startsWith('.')) {
        // Subdomain pattern
        return hostname.includes(pattern.substring(1)) || hostname.endsWith(pattern);
      } else {
        // Exact or subdomain match
        return hostname === pattern || hostname.endsWith('.' + pattern);
      }
    });
  }

  // Function to check if domain is blocked
  function isBlockedDomain(hostname) {
    hostname = hostname.toLowerCase();
    
    return blockedDomains.some(blocked => {
      return hostname === blocked || hostname.includes(blocked);
    });
  }

  // Function to validate URL
  function validateUrl(input) {
    // Extract and clean URL
    const extracted = extractAndCleanUrl(input);
    const cleanedUrl = extracted.url;
    const hasMultipleUrls = extracted.multipleUrls;
    
    if (!cleanedUrl) {
      return { 
        isValid: false, 
        cleanedUrl: null, 
        error: 'Please enter a valid URL',
        type: 'invalid',
        multipleUrls: false
      };
    }

    try {
      const urlObj = new URL(cleanedUrl);
      const hostname = urlObj.hostname.toLowerCase();
      
      // Check if it's a blocked domain first
      if (isBlockedDomain(hostname)) {
        return { 
          isValid: false, 
          cleanedUrl: cleanedUrl, 
          error: 'This service is not supported. Please use a Pinterest URL.',
          type: 'blocked',
          multipleUrls: hasMultipleUrls
        };
      }
      
      // Check if it's a Pinterest domain
      if (isPinterestDomain(hostname)) {
        return { 
          isValid: true, 
          cleanedUrl: cleanedUrl, 
          error: null,
          type: 'pinterest',
          multipleUrls: hasMultipleUrls
        };
      }
      
      // Not Pinterest, but valid URL - suggest Pinterest
      return { 
        isValid: false, 
        cleanedUrl: cleanedUrl, 
        error: 'Please enter a Pinterest URL (pinterest.com, pin.it, etc.)',
        type: 'not_pinterest',
        multipleUrls: hasMultipleUrls
      };
      
    } catch (e) {
      return { 
        isValid: false, 
        cleanedUrl: cleanedUrl, 
        error: 'Invalid URL format',
        type: 'invalid',
        multipleUrls: hasMultipleUrls
      };
    }
  }

  // Function to show error message with different styles
  function showError(message, type = 'error') {
    msgError.textContent = message;
    alertError.className = 'alert alert-danger'; // Reset to default
    
    if (type === 'blocked') {
      alertError.className = 'alert alert-warning';
    } else if (type === 'not_pinterest') {
      alertError.className = 'alert alert-info';
    }
    
    alertError.style.display = 'block';
  }

  // Function to hide error message
  function hideError() {
    alertError.style.display = 'none';
  }
  
  // Function to show success message when URL is cleaned
  function showUrlCleaned(originalUrl, cleanedUrl, hasMultipleUrls = false) {
    if (originalUrl !== cleanedUrl || hasMultipleUrls) {
      const tempMessage = document.createElement('div');
      tempMessage.style.cssText = 'background-color: #d4edda; border: 1px solid #c3e6cb; color: #155724; padding: 10px; margin: 10px 0; border-radius: 4px; font-size: 14px;';
      
      let messageText = '✓ URL cleaned and validated successfully.';
      if (hasMultipleUrls) {
        messageText = '✓ Multiple URLs detected. Using the first valid Pinterest URL.';
      }
      
      tempMessage.innerHTML = messageText;
      
      // Insert the message before the form
      form.parentNode.insertBefore(tempMessage, form);
      
      // Remove the message after 4 seconds
      setTimeout(() => {
        if (tempMessage.parentNode) {
          tempMessage.parentNode.removeChild(tempMessage);
        }
      }, 4000);
    }
  }
  
  // Function to start progress bar
  function startProgressBar() {
    let progress = 1;
    const interval = setInterval(() => {
      progressBar.innerHTML = `${progress}%`;
      progress++;
      if (progress === 100) {
        clearInterval(interval);
        progressBar.innerHTML = 'Processing...';
      }
    }, 80);
    
    setTimeout(() => {
      progressBar.style.display = 'block';
    }, 2000);
  }
  
  // Function to validate and process URL
  function validateAndProcessUrl(showMessages = true) {
    const originalUrl = urlInput.value.trim();
    
    if (!originalUrl) {
      if (showMessages) showError('Please enter a Pinterest URL');
      return { isValid: false, cleanedUrl: null };
    }
    
    const validation = validateUrl(originalUrl);
    
    if (!validation.isValid) {
      if (showMessages) {
        showError(validation.error, validation.type);
      }
      return { isValid: false, cleanedUrl: validation.cleanedUrl };
    }
    
    // URL is valid, update input if cleaned
    if (validation.cleanedUrl !== originalUrl || validation.multipleUrls) {
      urlInput.value = validation.cleanedUrl;
      if (showMessages) showUrlCleaned(originalUrl, validation.cleanedUrl, validation.multipleUrls);
    }
    
    return { isValid: true, cleanedUrl: validation.cleanedUrl };
  }
  
  // Form submit event
  form.addEventListener('submit', function(event) {
    const result = validateAndProcessUrl(true);
    
    if (!result.isValid) {
      event.preventDefault();
      return;
    }
    
    hideError();
  });
  
  // Download button click event
  downloadButton.addEventListener('click', (event) => {
    const result = validateAndProcessUrl(true);
    
    if (!result.isValid) {
      event.preventDefault();
      return;
    }
    
    startProgressBar();
  });
  
  // Real-time validation as user types
  let typingTimer;
  urlInput.addEventListener('input', function() {
    clearTimeout(typingTimer);
    hideError();
    
    const url = urlInput.value.trim();
    if (url.length > 5) {
      // Debounce validation - wait 1 second after user stops typing
      typingTimer = setTimeout(() => {
        const result = validateUrl(url);
        if (!result.isValid && url.length > 10) {
          showError(result.error, result.type);
        }
      }, 1000);
    }
  });
  
  // Paste button functionality
  if (navigator.clipboard && navigator.clipboard.readText) {
    pasteButton.classList.remove('disabled');
    
    pasteButton.addEventListener('click', function() {
      navigator.clipboard.readText()
        .then(text => {
          urlInput.value = text.trim();
          hideError();
          
          setTimeout(() => {
            validateAndProcessUrl(true);
          }, 100);
        })
        .catch(err => {
          console.error('Failed to read clipboard: ', err);
          showError('Failed to access clipboard');
        });
    });
  }
  
  // Global function for existing hiddenAlert
  window.hiddenAlert = function() {
    hideError();
    
    const url = urlInput.value.trim();
    if (url) {
      validateAndProcessUrl(true);
    }
  };
  
  // Add helpful placeholder text
  if (urlInput.placeholder === '' || !urlInput.placeholder) {
    urlInput.placeholder = 'Enter Pinterest URL (pinterest.com, pin.it, etc.)';
  }
});
</script>
