  async function generateAndShareNglImage(platform) {
	  // Elements
	  const messageText = jQuery('#ngl-modal-message').text().trim();
	  const replyText = jQuery('#ngl-reply-text').val().trim(); // New reply input
	  const snapshot = document.getElementById('ngl-share-snapshot');
	  const mainReplyBtn = document.getElementById('ngl-main-reply-btn');
	  
	  // Target the specific body container in our hardcoded template
	  // We need to inject the text into .snapshot-card-body
	  const snapshotBody = snapshot.querySelector('.snapshot-card-body');
	  const snapshotReplyContainer = snapshot.querySelector('.snapshot-reply-text');
	  
	  // Show loading state
	  const originalBtnText = mainReplyBtn.innerHTML;
	  mainReplyBtn.innerHTML = '<i class="fas fa-spinner fa-spin"></i> generating...';
	  mainReplyBtn.style.opacity = '0.7';

	  // Inject the message directly into the high-fidelity template
	  // We wrap it in a p tag to ensure it picks up the CSS styles we defined
	  snapshotBody.innerHTML = '<p>' + messageText + '</p>';

	  // Inject the reply text (or hide it if empty)
	  if(replyText) {
		  snapshotReplyContainer.innerHTML = replyText;
		  snapshotReplyContainer.style.display = 'inline-block';
	  } else {
		  snapshotReplyContainer.style.display = 'none';
	  }

	  try {
		  const canvas = await html2canvas(snapshot, {
			  scale: 1, // Full size is already set by CSS
			  useCORS: true,
			  backgroundColor: '#1a0e19ff', // Ensure black bg
			  logging: false,
			  width: 720,
			  height: 1280,
			  scrollY: -window.scrollY
		  });

		  canvas.toBlob(async function(blob) {
			  if (!blob) {
				  alert("Failed to generate image.");
				  resetBtn();
				  return;
			  }

			  const file = new File([blob], "secret-reply.png", { type: "image/png" });
			  
			  const shareData = {
				  files: [file],
				  title: 'My Secret Reply',
				  text: 'Send me anonymous messages!'
			  };

			  if (navigator.canShare && navigator.canShare(shareData)) {
				  try {
					  await navigator.share(shareData);
					  console.log('Shared successfully on ' + platform);
				  } catch (err) {
					  if (err.name !== 'AbortError') {
						  console.error('Share failed:', err);
						  // Fallback: Show Modal if native share crashes (e.g. gesture timeout)
						  const imgUrl = URL.createObjectURL(blob);
						  const fallbackModal = document.getElementById('ngl-fallback-modal');
						  const fallbackImg = document.getElementById('ngl-fallback-image');
						  
						  fallbackImg.src = imgUrl;
						  fallbackModal.style.display = 'flex';
					  }
				  }
			  } else {
				  // Fallback: Show Image in Modal for Long-Press Save
				  const imgUrl = URL.createObjectURL(blob);
				  const fallbackModal = document.getElementById('ngl-fallback-modal');
				  const fallbackImg = document.getElementById('ngl-fallback-image');
				  
				  fallbackImg.src = imgUrl;
				  fallbackModal.style.display = 'flex';
			  }
			  
			  resetBtn();
		  }, 'image/png');

	  } catch (error) {
		  console.error('Snapshot failed:', error);
		  alert("Oops! Something went wrong while creating the image.");
		  resetBtn();
	  }

	  function resetBtn() {
		  mainReplyBtn.innerHTML = originalBtnText;
		  mainReplyBtn.style.opacity = '1';
		  // We don't need to clear snapshotBody per se, but we can if we want to reset state
		  snapshotBody.innerHTML = '';
	  }
  }

function shareImage(surl, fname) {
  if (!navigator.canShare) {
    alert("Your browser doesn't support sharing !");
    gtag('event', 'share_failed', {
      'event_category': 'Share',
      'event_label': 'Unsupported Browser'
    });
    return;
  }

  fetch(surl)
    .then(function(response) {
      return response.blob();
    })
    .then(function(blob) {
      const filesArray = [
        new File(
          [blob],
          fname,
          {
            type: blob.type,
            lastModified: new Date().getTime()
          }
        )
      ];
      const shareData = {
        files: filesArray,
      };
      return navigator.share(shareData);
    })
    .then(function() {
      // Success
      gtag('event', 'insta_share_success', {
        'event_category': 'Share',
        'event_label': 'Image Shared'
      });
    })
    .catch(function(error) {
      // Error
      alert(`error : ${error.message}`);
      gtag('event', 'insta_share_failed', {
        'event_category': 'Share',
        'event_label': `Error: ${error.message}`
      });
    });
}
