    const hpMap = L.map('sx-hp-map', { zoomControl: true, attributionControl: false }).setView([43.88, -79.15], 10);

    async function runHpScan() {
        const status = document.getElementById('status-display');
        const cta = document.getElementById('hp-cta-box');
        cta.innerHTML = "";
        status.innerHTML = "LOCATING SECTOR...";

        navigator.geolocation.getCurrentPosition(async (pos) => {
            const latlng = L.latLng(pos.coords.latitude, pos.coords.longitude);
            if (hpMarker) hpMap.removeLayer(hpMarker);

            const isInside = isWithinPerimeter(latlng, hpPerimeter);
            const style = isInside ? "in-zone" : "out-zone";

            hpMarker = L.marker(latlng, { 
                icon: L.divIcon({ className: 'hp-pulse', html: `<div class="pulse ${style}"></div>`, iconSize: [12, 12], iconAnchor: [6, 6] }) 
            }).addTo(hpMap);

            hpMap.fitBounds(new L.FeatureGroup([hpPerimeter, hpMarker]).getBounds(), { padding: [30, 30], maxZoom: 12 });

            if (isInside) {
                try {
                    const res = await fetch(`https://nominatim.openstreetmap.org/reverse?format=jsonv2&lat=${pos.coords.latitude}&lon=${pos.coords.longitude}`);
                    const data = await res.json();
                    const city = data.address.city || data.address.town || data.address.suburb || "Your Area";
                    const slug = city.toLowerCase().replace(/\s+/g, '-');
                    const link = hpCities.includes(slug) ? `/mobile-tire-service-${slug}` : `/book-online`;

                    status.innerHTML = `<span style="color:var(--sparky-lime)">SCAN COMPLETE: ${city.toUpperCase()} DETECTED</span>`;
                    cta.innerHTML = `<a href="${link}" class="hp-cta">View Rates for ${city}</a>`;
                } catch (e) {
                    status.innerHTML = "<span style='color:var(--sparky-lime)'>SCAN COMPLETE: SECTOR VALIDATED</span>";
                    cta.innerHTML = `<a href="/book-online" class="hp-cta">Book Service Now</a>`;
                }
            } else {
                status.innerHTML = "<span style='color:var(--sparky-red)'>LOCATION OUTSIDE ACTIVE SECTOR</span>";
            }

    function isWithinPerimeter(latlng, poly) {
        const x = latlng.lat, y = latlng.lng;
        const vs = poly.getLatLngs()[0];

            const xi = vs[i].lat, yi = vs[i].lng, xj = vs[j].lat, yj = vs[j].lng;
            const intersect = ((yi > y) !== (yj > y)) && (x < (xj - xi) * (y - yi) / (yj - yi) + xi);
            if (intersect) inside = !inside;
        }
        return inside;
    }
