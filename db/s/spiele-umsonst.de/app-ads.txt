            if (tcData.gdprApplies === false) {
                if (!proceedOnce()) return;
                console.log('GDPR does not apply (TCF v2) – proceeding immediately');
                loadAdsParallel();
                return;
            }

            // EU-Flow
            if (tcData.gdprApplies) {
                if (tcData.eventStatus === 'tcloaded' || tcData.eventStatus === 'useractioncomplete') {
                    if (!proceedOnce()) return;
                    console.log('TCF v2 consent received');
                    loadAdsParallel();
                }
            }
        });
    } else if (hasCMPv1) {
        // Fallback auf CMP v1
        __cmp('addEventListener', ['consent', () => {
            const status = __cmp('consentStatus');
            if (status?.consentExists) {
                if (!proceedOnce()) return;
                console.log('CMP v1 consent received');
                loadAdsParallel();
            }
        }, false], null);
    } else {
        // Kein CMP gefunden - lade nach Delay, aber nur 1x
        console.log('No CMP found - using timeout fallback');
        setTimeout(() => {
            if (!proceedOnce()) return;
            loadAdsParallel();
        }, 2000);
    }
}

// Start bei DOMContentLoaded für optimales Timing
if (document.readyState === 'loading') {
    document.addEventListener('DOMContentLoaded', initConsentManager);
} else {
    initConsentManager();
}
</script>
