document.addEventListener('DOMContentLoaded', function() {
    var notifications = document.querySelectorAll('.push-notification-banner');
    notifications.forEach(function(notification) {
        var notificationId = notification.getAttribute('data-notification-id');
        var dismissedKey = 'push_notification_dismissed_' + notificationId;

        // Check if this notification has been dismissed
        if (!localStorage.getItem(dismissedKey)) {
            notification.style.display = 'block';
        }
    });

function dismissPushNotification(notificationId) {
    var notification = document.getElementById('push-notification-' + notificationId);

    if (notification) {
        // Store dismissal in localStorage
        localStorage.setItem('push_notification_dismissed_' + notificationId, 'true');

        // Hide with fade out effect
        notification.style.transition = 'opacity 0.2s ease-in-out';
        notification.style.opacity = '0';

        setTimeout(function() {
            notification.style.display = 'none';
        }, 200);

        // Also call the server endpoint to record the dismissal

        xhr.open('POST', '/dismiss_push_notification.php', true);

// do this in an independent function so that we can call it again if/when items are added to the page through ajax
function createWishlistLinks($j) {
	// kill existing ones..
	$j('.wl_plus').remove();

	// Find all items on the page
	$j('.wl_item').each(function(wl) {
	    var $this = $j(this);
	    // What is our itemid?
	    var itemid = parseInt($this.attr('data-itemid'));

	    var index = _.indexOf(wish_list, itemid);
	    if (index !== -1) { // We have this on the wishlist
		$this.addClass('wishlist');
		$this.append("<div class='wl_plus wl' data-itemid='"+ itemid +"' data-action='remove'><small>- Wishlist</small></div>");
	    } else {
		$this.addClass('non_wishlist');
		$this.append("<div class='wl_plus wl' data-itemid='"+ itemid +"' data-action='add'><small>+ Wishlist</small></div>");
	    }
	});

	$j('body').on('click', '.wl', function(data) {
	    var $this = $j(this);

	    var item = $this.attr('name');
	    var itemid = $this.attr('data-itemid');
	    var action = $this.attr('data-action');
	    if (action == 'add') {
		$j('.wl_plus[data-itemid="' + itemid +'"]').html("<small>Loading...</small");
		$j.get("/wishlists.php/add/" + itemid, function(rtn) {
		    if (rtn.error) {
			$j('div[data-itemid="' + itemid +'"]').html("<small class='has-tip' title='There are too many items on this wishlist. Change your default wishlist to one with less than 500 items.'><i class='icon exclamation'></i> Error</small");
		    } else {
			$j('.wl_item[data-itemid="' + itemid +'"]').removeClass('non_wishlist');
			$j('.wl_item[data-itemid="' + itemid +'"]').addClass('wishlist');
			$j('.wl_plus[data-itemid="' + itemid +'"]').attr('data-action', 'remove'); // Make it so that it removes it now
			$j('.wl_plus[data-itemid="' + itemid +'"]').html("<small>- Wishlist</small");
			wish_list.push(parseInt(itemid));
		    }
		}, 'JSON');
	    } else {
		$j('.wl_plus[data-itemid="' + itemid +'"]').html("<small>Loading...</small");
		$j.get("/wishlists.php/delete/" + itemid, function(rtn) {
		    $j('.wl_item[data-itemid="' + itemid +'"]').removeClass('wishlist');
		    $j('.wl_item[data-itemid="' + itemid +'"]').addClass('non_wishlist');
		    $j('.wl_plus[data-itemid="' + itemid +'"]').attr('data-action', 'add'); // Make it so that it removes it now
		    $j('.wl_plus[data-itemid="' + itemid +'"]').html("<small>+ Wishlist</small");
		    wish_list = _.without(wish_list, parseInt(itemid));
		});
	    }
	    data.preventDefault();
	});
 }
