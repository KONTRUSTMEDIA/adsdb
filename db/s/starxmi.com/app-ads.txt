    <script>
      new Vue({
        el: "#index",
        data() {
          return {
            lang: lang,
            show_text: showLang,
            topList: [],
            newVideoList: [],
            categoryList: [],
            showList: [],
            likeList: [],
            page: 1,
            op: "",
            pageSize: 5,
            file_token: "",
            currentTagId: 142,
            assets: location.protocol + glo_obj.assets,
          };
        },
        filters: {
          ellipsis(value) {
            if (!value) return "";
            if (value.length > 15) {
              return value.slice(0, 15) + "...";
            }
            return value;
          },
          ellipsis10(value) {
            if (!value) return "";
            if (value.length > 8) {
              return value.slice(0, 8) + "...";
            }
            return value;
          },
          ellipsis30(value) {
            if (!value) return "";
            if (value.length > 30) {
              return value.slice(0, 30) + "...";
            }
            return value;
          },
        },
        mounted() {
          this.init();
          var phone = this.getPhone();
          var op = this.getOp();
          if (op) {
            localStorage.op = op;
          }
          this.op = localStorage.op;
          if (phone) {
            // 临时取消登录
            // this.login(phone)
          }
        },
        watch: {
          lang: function (v) {
            localStorage.lang = v;
          },
        },

        methods: {
          logout: function () {
            localStorage.removeItem("token");
            localStorage.removeItem("phone");
            localStorage.removeItem("vip_expiration");
            localStorage.removeItem("create_time");
            this.$message({
              message: this.show_text.logoutSuccess,
              type: "success",
              duration: 1000,
              onClose: () => {
                location.href = "./index.html";
              },
            });
          },
          getLang() {
            document.querySelector("#select-lang").value = this.lang;
          },
          changeLang() {
            window.location.href = "index.html?lang=" + this.lang;
          },
          init: function () {
            localStorage.lang = this.lang = this.getLang() || localStorage.lang || "pl";
            this.lang = localStorage.lang;
          },
          getOp: function () {
            var op;
            try {
              op = location.href
                .toLowerCase()
                .match(/gj=[\w]+/)[0]
                .split("=")[1];
            } catch (error) {
              op = null;
            }
            return op ? op : null;
          },
          getPhone: function () {
            var phone;
            try {
              phone = location.href
                .toLowerCase()
                .match(/phone=[\d]+/)[0]
                .split("=")[1];
            } catch (error) {
              phone = null;
            }
            return phone ? phone : null;
          },
          login: function (phone) {
            var that = this;
            $.ajax({
              type: "post",
              url: location.protocol + glo_obj.login,
              data: {
                username: phone,
                password: "123456",
              },
              success: function (res) {
                localStorage.token = res.token;
                localStorage.phone = phone;
                localStorage.vip_expiration = res.user.vip_expiration;
                localStorage.create_time = res.user.create_time;
                that.$message({
                  message: that.show_text.loginSuccess,
                  type: "success",
                  duration: 1000,
                });
              },
              error: function () {
                // that.$message.error(that.show_text.loginError);
              },
            });
          },
        },
      });
    </script>
