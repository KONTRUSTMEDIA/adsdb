                : hasSavedPrefs
                    ? config.consent.preferences[cat.key] ?? false
                    : true;

        categoryList.innerHTML += `
            <div class="form-check form-switch cookie-category-card">
                <input class="form-check-input" type="checkbox" id="cookie_${cat.key}"
                       data-key="${cat.key}" ${checked ? "checked" : ""} ${cat.required ? "disabled" : ""}>
                <label class="form-check-label fw-semibold" for="cookie_${cat.key}">
                    ${t?.name ?? cat.key}
                </label>
                <small class="text-muted d-block">${t?.description ?? ""}</small>
            </div>
        `;
    });

    // 🔘 FUNZIONI DI SUPPORTO
    // ================================================================
    const hideBanner = () => {
        banner.classList.remove('show');
        banner.style.opacity = "0";
        setTimeout(() => banner.style.display = "none", 400);
    };

    const showManageButton = () => {
        if (manageButton) {
            manageButton.style.display = 'block';
            manageButton.classList.add('show');
        }
    };

    const saveConsent = (status) => {
        localStorage.setItem("cookie_consent", status);
        localStorage.setItem("cookie_consent_date", new Date().toISOString());
        hideBanner();
        showManageButton();
    };

    // ================================================================
    acceptAllBtn.addEventListener("click", async () => {
        await fetch(`${base}/accept`, { 
            method: "POST", 
            headers: { "X-CSRF-TOKEN": "tmqrfiGmMeSoEETuNCLGFvlnc3NiQqrlqegLGD0Z" }
        });

        gtag('consent', 'update', {
            'analytics_storage': 'granted',
            'ad_storage': 'granted',
            'ad_personalization': 'granted',
            'ad_user_data': 'granted'
        });

        saveConsent("accepted");
    });

    // ❌ RIFIUTA TUTTO
    // ================================================================
    rejectAllBtn.addEventListener("click", async () => {
        await fetch(`${base}/reject`, { 
            method: "POST", 
            headers: { "X-CSRF-TOKEN": "tmqrfiGmMeSoEETuNCLGFvlnc3NiQqrlqegLGD0Z" }
        });

        // Consent Mode V2
        gtag('consent', 'update', {
            'analytics_storage': 'denied',
            'ad_storage': 'denied',
            'ad_personalization': 'denied',
            'ad_user_data': 'denied'
        });

        saveConsent("rejected");
    });

    // ⚙️ SALVA PREFERENZE PERSONALIZZATE
    // ================================================================
    saveBtn.addEventListener("click", async () => {
        const prefs = {};
        document.querySelectorAll("#cookie-categories-list input[type='checkbox']").forEach(input => {
            prefs[input.dataset.key] = input.checked;
        });

        await fetch(`${base}/save`, {
            method: "POST",
            headers: {
                "Content-Type": "application/json",
                "X-CSRF-TOKEN": "tmqrfiGmMeSoEETuNCLGFvlnc3NiQqrlqegLGD0Z"
            },
            body: JSON.stringify({ preferences: prefs })
        });

        localStorage.setItem("cookie_preferences", JSON.stringify(prefs));

        // Consent Mode V2 dinamico
        gtag('consent', 'update', {

        }
    }

document.addEventListener("DOMContentLoaded", function() {
    const lazyEls = [...document.querySelectorAll('[data-bg], img[data-src]')];

    if ('IntersectionObserver' in window) {
        const observer = new IntersectionObserver(entries => {
            entries.forEach(entry => {
                if (entry.isIntersecting) {
                    const el = entry.target;
                    if (el.dataset.bg) {
                        el.style.backgroundImage = `url('${el.dataset.bg}')`;
                        el.removeAttribute('data-bg');
                    }
                    if (el.tagName === 'IMG' && el.dataset.src) {
                        el.src = el.dataset.src;
                        el.removeAttribute('data-src');
                    }
                    observer.unobserve(el);
                }
            });
        }, { rootMargin: '200px' });

        lazyEls.forEach(el => observer.observe(el));
    } else {
        lazyEls.forEach(el => {
            if (el.dataset.bg) el.style.backgroundImage = `url('${el.dataset.bg}')`;
            if (el.dataset.src) el.src = el.dataset.src;
        });
    }
});
