var audio, playbtn, seek_bar, seekslider, volumeslider, seeking, seekto, curtimetext, durtimetext;
// Establish all variables that your Analyser will use
var canvas, ctx, source, context, analyser, fbc_array, bars, bar_x, bar_width, bar_height;
function initAudioPlayer(audicle){
  
 audio = new Audio(); 
 audio.src = "https://scribetemple.com/"+"assets/audicles/"+audicle; 
audio.loop = true; 
document.getElementById("togpp").setAttribute("class", "fa fa-pause fa-2x");
audio.play();

// Set object references
playbtn = document.getElementById("playpausebtn"); 
seekslider = document.getElementById("seekslider"); 
volumeslider = document.getElementById("volumeslider");
curtimetext = document.getElementById("curtimetext");
durtimetext = document.getElementById("durtimetext"); 
// Add Event Handling
playbtn.addEventListener("click",playPause); 
seekslider.addEventListener("mousedown", function(event){ seeking = true; seek(event); }); 
seekslider.addEventListener("mousemove", function(event){ seek(event); }); 
seekslider.addEventListener("mouseup", function(){ seeking = false;}); 
volumeslider.addEventListener("mousemove", setvolume);
audio.addEventListener("timeupdate", seektimeupdate);
context = new AudioContext(); // AudioContext object instance
analyser = context.createAnalyser(); // AnalyserNode method
canvas = document.getElementById('analyser_render'); 
ctx = canvas.getContext('2d'); // Re-route audio playback into the processing graph of the AudioContext
source = context.createMediaElementSource(audio);
source.connect(analyser); 
analyser.connect(context.destination); 
frameLooper();
}
 // Functions
function playPause(){
 if(audio.paused){ 
 audio.play(); 
 document.getElementById("togpp").setAttribute("class", "fa fa-pause fa-2x"); 
 } else { 
 audio.pause();
  document.getElementById("togpp").setAttribute("class", "fa fa-play fa-2x"); 
  } } 
  
   
  function seek(event){ 
   if(seeking){ 
   seekslider.value = event.clientX - seekslider.offsetLeft; 
   seekto = audio.duration * (seekslider.value / 100);
    audio.currentTime = seekto; 
     } 
   }
   
  function setvolume(){ 
  audio.volume = volumeslider.value / 100; 
  }
  
  function frameLooper(){ 
  window.webkitRequestAnimationFrame(frameLooper); 
  fbc_array = new Uint8Array(analyser.frequencyBinCount); 
  analyser.getByteFrequencyData(fbc_array); 
  ctx.clearRect(0, 0, canvas.width, canvas.height);
   // Clear the canvas
ctx.fillStyle = '#337AB7'; 
// Color of the bars
bars = 100; 
for (var i = 0; i < bars; i++) { 
bar_x = i * 3; 
bar_width = 2; 
bar_height = -(fbc_array[i] / 2); 
// fillRect( x, y, width, height ) // Explanation of the parameters below
ctx.fillRect(bar_x, canvas.height, bar_width, bar_height); 
} 
}  
  function seektimeupdate(){ 
  var nt = audio.currentTime * (100 / audio.duration); 
  seekslider.value = nt; 
  var curmins = Math.floor(audio.currentTime / 60); 
  var cursecs = Math.floor(audio.currentTime - curmins * 60); 
  var durmins = Math.floor(audio.duration / 60); 
  var dursecs = Math.floor(audio.duration - durmins * 60); 
  if(cursecs < 10){ cursecs = "0"+cursecs; } 
  if(dursecs < 10){ dursecs = "0"+dursecs; } 
  if(curmins < 10){ curmins = "0"+curmins; }
  if(durmins < 10){ durmins = "0"+durmins; } 
  curtimetext.innerHTML = curmins+":"+cursecs+"&nbsp;/"; 
  durtimetext.innerHTML = "&nbsp;"+durmins+":"+dursecs; 
  }

	function PerformSearch(){
	var article_search_box = document.getElementById("article_search_results");
	var classified_search_box = document.getElementById("classified_search_results");
	var searchtxt = document.getElementById("searchtxt").value;
	article_search_box.innerHTML = "";
	classified_search_box.innerHTML = "";
	 var hr = new XMLHttpRequest();
	 hr.open("POST", "https://scribetemple.com/ajax/search_parser.php", true);
	 hr.setRequestHeader("Content-type", "application/x-www-form-urlencoded");
	 hr.onreadystatechange = function (){
		 if(hr.readyState == 4 && hr.status == 200){
			 var dataArray = hr.responseText.split("<|||>");
			  var ArticleArray = dataArray[0].split("|||");
			  var ClassifiedsArray = dataArray[1].split("|||");
			  var ArticleCount = ArticleArray[0];
			  var ArticleSearchArray = ArticleArray[1].split("||");
			  var ClassifiedsCount = ClassifiedsArray[0];
			  var ClassifiedsSearchArray = ClassifiedsArray[1].split("||");
			  var article_search_output = "";
			  var classifieds_search_output = "";
			  if(ArticleCount == 1){
				    article_search_output +="<h5>"+ArticleCount+" Article matches your search</h5>";
			        }
			  if(ArticleCount > 1){
					article_search_output +="<h5>"+ArticleCount+" Articles match your search</h5>";
					}
			 if(ClassifiedsCount == 1){
				    classifieds_search_output +="<h5>"+ClassifiedsCount+" Item matches your search</h5>";
			        }
			  if(ClassifiedsCount > 1){
					classifieds_search_output +="<h5>"+ClassifiedsCount+" Items match your search</h5>";
					} 
			  
				for(i = 0; i < ArticleSearchArray.length - 1; i++){
					var ArticleItemArray = ArticleSearchArray[i].split('|');
		article_search_output += "<a href='https://scribetemple.com/"+ArticleItemArray[0]+"'style='color:#FFF;'>"+ArticleItemArray[2]+"</a><br/>";
					}
					
					for(i = 0; i < ClassifiedsSearchArray.length - 1; i++){
					var ClassifiedsItemArray = ClassifiedsSearchArray[i].split('|');
		classifieds_search_output += "<a href='https://scribetemple.com/"+ClassifiedsItemArray[0]+"'style='color:#FFF;'>"+ClassifiedsItemArray[1]+"</a><br/>";
					}
					if(ArticleCount >= 1 && ClassifiedsCount == 0){
						var searchfieldResult = document.getElementById("searchfieldResult");
						  searchfieldResult.removeChild(document.getElementById("classifiedSearchfield"));
						 var article_search_results = document.getElementById("article_search_results");
						   article_search_results.style.height = 300+"px";
						}
					if(ClassifiedsCount >= 1 && ArticleCount == 0){
						var searchfieldResult = document.getElementById("searchfieldResult");
						  searchfieldResult.removeChild(document.getElementById("articleSearchfield"));
						 var classified_search_results = document.getElementById("classified_search_results");
						   classified_search_results.style.height = 300+"px";
						}
						
					if(ArticleCount != 0){
				    article_search_box.innerHTML = article_search_output;
					}
					if(ClassifiedsCount != 0){
					classified_search_box.innerHTML = classifieds_search_output;
					}
					if(ArticleCount == 0 && ClassifiedsCount == 0){
						article_search_box.innerHTML = "<h3>Sorry, no articles or classifieds matched your search.</h3>";
						}
			 }
		 }
		 hr.send("searchtxt="+searchtxt);
						   
	}

    var name = cname + "=";
    var ca = document.cookie.split(';');
    for(var i=0; i<ca.length; i++) {

}
