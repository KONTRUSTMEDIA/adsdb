		$(window).ready(function () {
			$(".draw-slide-container").hide()
			$(".list-btn").click(function () {
				$(".draw-slide-container").show()
				$(".draw-slide-container .draw-slide").css("width", "70%")
				$(".draw-slide-container .draw-slide").css("overflow", "hidden")
				$(".draw-slide-container .draw-slide .slide-row").show()
			})

			$("#mask").click(closeSlide)

			$("#close-slide").click(closeSlide)

			function closeSlide() {
				$(".draw-slide-container .draw-slide").css("width", "0%")
				$(".draw-slide-container .draw-slide").css("overflow", "unset")
				$(".draw-slide-container .draw-slide .slide-row").hide()
				$(".draw-slide-container").fadeOut(500)
			}

			$("#mobile-footer").load("template/footer.html", () => {
				$(".footer .footer-list .list-item .item-rows").click(function () {
					if ($(this).children().eq(1).data("traggle") == 0) {
						$(this)
							.children()
							.eq(1)
							.css("transform", "rotate3d(1, 0, 0, 180deg)")
						$(this).children().eq(1).data("traggle", 1)
					} else {
						$(this).children().eq(1).css("transform", "rotate3d(1, 0, 0, 0deg)")
						$(this).children().eq(1).data("traggle", 0)
					}
					$(this).parent().children().eq(1).slideToggle()
				})
			})

			$(".header-content").width(window.innerWidth)
