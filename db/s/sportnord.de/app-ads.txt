    (function ($) {
        $.fn.slideshow = function () {
            if (this.length == 0) return;
            var slider = {};
            var el = this;
            var maxIndex = 3;
            var windowWidth = $(window).width();
            var windowHeight = $(window).height();
            var marginRight = 0;
            var setup = function () {
                slider.children = el.children('li');
                el.wrap('<div class="slideshow-wrapper"><div class="slideshow-viewport"></div></div>');
                slider.viewport = el.parent();
                slider.children.imagesLoaded(start);
            }
            var start = function () {
                slider.viewport.css({'width': '100%', 'overflow': 'hidden', 'position': 'relative', 'height': getViewportHeight() + 'px'});
                el.width(getElWidth());
                slider.children.css({'float': 'left', 'display': 'block', 'margin-right': marginRight + 'px'});
                slider.children.width(slider.viewport.width());
                createMenu();
                $(window).bind('resize', resizeWindow);
                slider.active = {index: 0}
                el.startAuto();
            }
            var getElWidth = function () {
                width = 0;
                slider.children.map(function () {
                    width += $(this).outerWidth(false) + marginRight;
                });
                return width;
            }
            var getViewportHeight = function () {
                height = Math.max.apply(Math, slider.children.map(function () {
                    return $(this).outerHeight(false);
                }).get());
                ;
                return height;
            }
            var getMenuItemWidth = function () {
                width = slider.viewport.width() / slider.children.length;
                return width;
            }
            var getMenuHeight = function () {
                height = Math.max.apply(Math, slider.menu.children.map(function () {
                    return $(this).outerHeight();
                }).get());
                ;
                return height;
            }
            var resizeWindow = function () {
                var windowWidthNew = $(window).width();
                var windowHeightNew = $(window).height();
                if (windowWidth != windowWidthNew || windowHeight != windowHeightNew) {
                    windowWidth = windowWidthNew;
                    windowHeight = windowHeightNew;
                    slider.children.width(slider.viewport.width());
                    slider.viewport.css('height', getViewportHeight());
                    el.width(getElWidth());
                    el.css('margin-left', -slider.active.index * (slider.viewport.width() + marginRight));
                    slider.menu.children.width(getMenuItemWidth());
                    slider.menu.children.children('a').attr('style', '');
                    slider.menu.css('height', getMenuHeight());
                    slider.menu.children.children('a').css('height', (slider.menu.height() - 20) + 'px');
                }
            }
            var createMenu = function () {
                slider.menu = $('<ol class="slideshow-menu"></ol>');
                slider.viewport.parent().append(slider.menu);
                for (i = 0; i < slider.children.length; i++) {
                    item = slider.children.eq(i);
                    imgSrc = item.find('img').attr('src');
                    filename = imgSrc.substring(imgSrc.lastIndexOf('/') + 1);
                    filetype = filename.substring(filename.lastIndexOf('.') + 1);
                    filename = filename.substring(0, filename.lastIndexOf('.'));
                    if (filename.indexOf('-660x385') == filename.length - 8) {
                        filename = filename.substring(0, filename.length - 8);
                    }
                    filepath = imgSrc.substring(0, imgSrc.lastIndexOf('/') + 1);
                    imgSrc = filepath + filename + '.' + filetype;
                    title = item.find('p').html();
                    slider.menu.append('<li><a href="#" data-slide-index="' + i + '"><img src="' + imgSrc + '">' + title + '</a></li>');
                }
                slider.menu.children = slider.menu.children('li');
                slider.menu.children.css({'float': 'left'});
                slider.menu.children.width(getMenuItemWidth());
                slider.menu.children.imagesLoaded(setupMenu);
            }
            var setupMenu = function () {
                slider.menu.css({'width': '10000px', 'overflow': 'hidden', 'height': getMenuHeight() + 'px'});
                slider.menu.children.children('a').css('height', (slider.menu.height() - 20) + 'px');
                slider.menu.children.first().addClass('active');
                slider.menu.children.find('a').click(function (e) {
                    e.preventDefault();
                });
                slider.menu.delegate('a', 'click', clickMenuBind);
            }
            var clickMenuBind = function (e) {
                el.stopAuto();
                var pagerLink = $(e.currentTarget);
                var pagerIndex = parseInt(pagerLink.attr('data-slide-index'));
                if (pagerIndex != slider.active.index) el.goToSlide(pagerIndex);
                e.preventDefault();
            }
            el.goToSlide = function (index) {
                $('.slide_' + slider.active.index).removeClass('active');
                slider.active.index = index;
                // slider.menu.children.find('.active').removeClass('active');
                // slider.menu.children.eq(slider.active.index).addClass('active');
                $('.slide_' + index).addClass('active');
                el.stop().animate({'margin-left': -slider.active.index * (slider.viewport.width() + marginRight)}, 500);
            }
            el.startAuto = function () {
                if (slider.interval) return;
                slider.interval = setInterval(function () {
                    to = (slider.active.index + 1) % slider.children.length;
                    el.goToSlide(to);
                }, 5000);
            }
            el.stopAuto = function () {
                clearInterval(slider.interval);
            }
            setup();
            return this;
        }
    })(jQuery);
