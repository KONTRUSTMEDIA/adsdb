document.addEventListener('DOMContentLoaded', function () {
    const swiperEl = document.querySelector('.homeVideoSwiper');

    if (!swiperEl) return;

    const swiper = new Swiper(swiperEl, {
        loop: true,
        allowTouchMove: false,
    });

    function playVideoInSlide(index) {
        const allVideos = document.querySelectorAll('.video-slide');
        allVideos.forEach(v => {
            v.pause();
            v.currentTime = 0;
        });

        const currentSlide = swiper.slides[swiper.activeIndex];
        const video = currentSlide.querySelector('video');
        if (video) {
            const playPromise = video.play();
            if (playPromise !== undefined) {
                playPromise.catch(error => {
                    console.warn("Autoplay prevented:", error);
                });
            }

            video.onended = () => {
                swiper.slideNext();
                setTimeout(() => playVideoInSlide(swiper.realIndex + 1), 300);
            };
        }
    }

    swiper.on('slideChange', () => {
        playVideoInSlide(swiper.realIndex + 1);
    });

    playVideoInSlide(1); // Start with first real slide
});
</script>

        const header = document.querySelector('.header-fluidcont');
        const headertwo = document.querySelector('.font-resize-controls');

        if (!header) return;

        let lastScrollTop = 0;

        window.addEventListener('scroll', function () {
        const currentScroll = window.pageYOffset || document.documentElement.scrollTop;

        if (currentScroll > 95) {
            // Hide header when scrolling down
            if (currentScroll > lastScrollTop) {
            header.style.transform = 'translateY(-95px)';
            headertwo.style.transform = 'translateY(-25px)';
            } 
            // Show header when scrolling up
            else if (currentScroll < lastScrollTop) {
            header.style.transform = 'translateY(0)';
            headertwo.style.transform = 'translateY(0)';
            }
        } else {
            // Always show at top
            header.style.transform = 'translateY(0)';
            headertwo.style.transform = 'translateY(0)';
        }

        lastScrollTop = currentScroll <= 0 ? 0 : currentScroll;
        });
    });

                if (!$link.data('clicked')) {

                    // Optionally add active class to show submenu

                }
            });

	function resizeFont(change) {
	const oldSize = currentFontSize;
	currentFontSize = Math.max(70, Math.min(200, currentFontSize + change * 10));
	document.documentElement.style.fontSize = currentFontSize + '%';

	const status = document.getElementById('font-size-status');

			status.textContent = `Font size decreased to ${currentFontSize}%.`;
		} else {
			status.textContent = `Font size is already at ${currentFontSize}%.`;
		}
	}

	function toggleContrast() {
		const body = document.body;
		const btn = document.getElementById('contrast-toggle');

		// Toggle high-contrast class
		const isHighContrast = body.classList.toggle('high-contrast');
		btn.classList.toggle('active', isHighContrast);

		// Save preference

	}

		}
