<!-- Contact form handling: submit to server /api/contact -->
<script>
    (function () {
        function showAlert(msg, ok = true) {
            alert(msg);
        }

        // Attach handler if contact form exists
        const contactForm = document.getElementById('contactForm');
        if (contactForm) {
            contactForm.addEventListener('submit', async function (e) {
                e.preventDefault();
                const name = (document.getElementById('contactName') || {}).value || '';
                const email = (document.getElementById('contactEmail') || {}).value || '';
                const phone = (document.getElementById('contactPhone') || {}).value || '';
                const message = (document.getElementById('contactMessage') || {}).value || '';

                try {
                    const res = await fetch('/api/contact', {
                        method: 'POST',
                        headers: { 'Content-Type': 'application/json' },
                        body: JSON.stringify({ name, email, phone, message })
                    });
                    const data = await res.json();
                    if (data && data.status) {
                        showAlert('Thank you — we received your message.');
                        contactForm.reset();
                    } else {
                        showAlert(data.message || 'Failed to send message', false);
                    }
                } catch (err) {
                    console.error(err);
                    showAlert('Unable to send message. Please try again later.', false);
                }
            });
        }
    })();
</script>

<script>
    (function () {
        document.addEventListener('DOMContentLoaded', function () {
            var toggle = document.querySelector('.navbar-toggle');
            var collapse = document.querySelector('.navbar-collapse');

            // Fallback toggle if bootstrap's collapse isn't wired to this markup
            if (toggle && collapse) {
                toggle.addEventListener('click', function (e) {
                    e.preventDefault();
                    collapse.classList.toggle('show');
                });

                // Close collapse when a nav link is clicked (mobile)
                document.querySelectorAll('.navigation a[href^="#"]').forEach(function (el) {
                    el.addEventListener('click', function () {
                        if (window.innerWidth < 992) {
                            collapse.classList.remove('show');
                        }
                    });
                });
            }

            // Clone the Sign Up / Get Started buttons into the mobile menu so they're visible inside the collapsed nav
            // Skip cloning into the old collapse when the slide-panel exists (avoids duplicate menus)
            var menuBtns = document.querySelector('.menu-btns');
            var _hasSlide = !!document.getElementById('siteSlidePanel');
            if (menuBtns && collapse && !_hasSlide) {
                try {
                    var navUl = collapse.querySelector('ul.navigation');
                    if (navUl && !collapse.querySelector('.mobile-action-buttons')) {
                        var li = document.createElement('li');
                        li.className = 'mobile-action-buttons d-lg-none';
                        li.style.padding = '10px 15px';
                        li.innerHTML = menuBtns.innerHTML;
                        // make buttons full width in mobile menu
                        li.querySelectorAll('a').forEach(function (a) { a.classList.add('d-block', 'w-100', 'mb-2'); });
                        navUl.appendChild(li);
                    }
                } catch (err) {
                    console.error('mobile clone error', err);
                }
            }
            
            // Slide-panel: clone main navigation into the slide panel and wire toggle/close
            try {
                var slidePanel = document.getElementById('siteSlidePanel');
                if (slidePanel) {
                    // clone main navigation links into slide panel primary-menu if empty
                    var dest = slidePanel.querySelector('.primary-menu');
                    var source = document.querySelector('.navigation');
                    if (dest && source && dest.children.length === 0) {
                        // Clone only the LI items from the existing navigation into the slide panel
                        try {
                            var nodes = Array.from(source.children || []);
                            nodes.forEach(function (node) {
                                // only clone LI elements
                                if (node.tagName && node.tagName.toLowerCase() === 'li') {
                                    dest.appendChild(node.cloneNode(true));
                                }
                            });

                            // ensure links inside the cloned list close the panel when clicked
                            dest.querySelectorAll('a[href^="#"]').forEach(function (a) {
                                a.addEventListener('click', function (ev) {
                                    // let smooth-scroll handler run first; then close panel
                                    setTimeout(function () {
                                        slidePanel.classList.remove('show-panel');
                                        document.body.classList.remove('no-scroll');
                                    }, 50);
                                });
                            });
                        } catch (err) {
                            console.error('clone items error', err);
                        }
                    }

                    // toggle open when navbar toggle clicked (mobile)
                    if (toggle) {
                        toggle.addEventListener('click', function (e) {
                            if (window.innerWidth < 992) {
                                e.preventDefault();
                                var isOpen = slidePanel.classList.contains('show-panel');
                                if (!isOpen) {
                                    slidePanel.classList.add('show-panel');
                                    document.body.classList.add('no-scroll');
                                } else {
                                    slidePanel.classList.remove('show-panel');
                                    document.body.classList.remove('no-scroll');
                                }
                            }
                        });
                    }

                    // close handlers
                    slidePanel.querySelectorAll('[data-action="close-panel"]').forEach(function (el) {
                        el.addEventListener('click', function () {
                            slidePanel.classList.remove('show-panel');
                            document.body.classList.remove('no-scroll');
                        });
                    });
                }
            } catch (err) {
                console.error('slide panel error', err);
            }
        });
    })();
</script>

<!-- Single-page nav: smooth scroll, active link highlighting, collapse on mobile -->
<script>
    (function () {
        // delegated smooth scroll for internal anchors (works for cloned/dynamic links)
        $(document).on('click', 'a[href^="#"]', function (e) {
            var href = $(this).attr('href');
            if (!href || href === '#') return;
            var $target = $(href);
            if ($target.length) {
                e.preventDefault();
                var offset = Math.max(0, $target.offset().top - 80); // account for header
                $('html, body').animate({ scrollTop: offset }, 600);
                // collapse bootstrap mobile menu if open
                $('.navbar-collapse').collapse('hide');
                // also close our slide-panel if it's open
                var $panel = $('#siteSlidePanel');
                if ($panel.length && $panel.hasClass('show-panel')) {
                    $panel.removeClass('show-panel');
                    $('body').removeClass('no-scroll');
                }
            }
        });

        // update active link on scroll
        var $navLinks = $('.navigation a[href^="#"]');
        var $sections = $('section[id]');

        function updateActive() {
            var scrollPos = $(window).scrollTop() + 100; // slightly below top
            var currentId = null;
            $sections.each(function () {
                var top = $(this).offset().top;
                var bottom = top + $(this).outerHeight();
                if (scrollPos >= top && scrollPos < bottom) {
                    currentId = $(this).attr('id');
                    return false; // break
                }
            });
            $navLinks.removeClass('active');
            if (currentId) {
                $navLinks.filter('[href="#' + currentId + '"]').addClass('active');
            } else {
                $navLinks.first().addClass('active');
            }
        }

        $(window).on('scroll resize', function () {
            updateActive();
        });

        // initial activation
        $(function () { updateActive(); });
    })();
</script>
