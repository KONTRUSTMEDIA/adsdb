// Takım logoları kaydırma fonksiyonu
function scrollTeams(direction) {
    const container = document.querySelector('.teams-scroll-container');
    const scrollElement = document.querySelector('.teams-logos-grid');
    
    if (!container || !scrollElement) return;
    
    const scrollAmount = 200;
    const currentScroll = container.scrollLeft || 0;
    
    let newScroll = currentScroll + (direction * scrollAmount);
    
    const maxScroll = scrollElement.scrollWidth - container.clientWidth;
    newScroll = Math.max(0, Math.min(newScroll, maxScroll));
    
    container.scrollTo({
        left: newScroll,
        behavior: 'smooth'
    });
    
    updateScrollButtons();
}

// Scroll butonlarını güncelle
function updateScrollButtons() {
    const container = document.querySelector('.teams-scroll-container');
    const leftBtn = document.querySelector('.team-scroll-btn.left');
    const rightBtn = document.querySelector('.team-scroll-btn.right');
    
    if (!container || !leftBtn || !rightBtn) return;
    
    const currentScroll = container.scrollLeft;
    const maxScroll = container.scrollWidth - container.clientWidth;
    
    if (currentScroll <= 10) {
        leftBtn.style.opacity = '0.5';
        leftBtn.style.pointerEvents = 'none';
        leftBtn.setAttribute('aria-hidden', 'true');
    } else {
        leftBtn.style.opacity = '1';
        leftBtn.style.pointerEvents = 'all';
        leftBtn.setAttribute('aria-hidden', 'false');
    }
    
    if (currentScroll >= maxScroll - 10) {
        rightBtn.style.opacity = '0.5';
        rightBtn.style.pointerEvents = 'none';
        rightBtn.setAttribute('aria-hidden', 'true');
    } else {
        rightBtn.style.opacity = '1';
        rightBtn.style.pointerEvents = 'all';
        rightBtn.setAttribute('aria-hidden', 'false');
    }
}

// Touch/swipe desteği
let touchStartX = 0;
let touchEndX = 0;

const scrollContainer = document.querySelector('.teams-scroll-container');
if (scrollContainer) {
    scrollContainer.addEventListener('touchstart', function(e) {
        touchStartX = e.touches[0].clientX;
    }, { passive: true });
    
    scrollContainer.addEventListener('touchmove', function(e) {
        touchEndX = e.touches[0].clientX;
    }, { passive: true });
    
    scrollContainer.addEventListener('touchend', function() {
        handleSwipe();
    }, { passive: true });
}

function handleSwipe() {
    const swipeThreshold = 50;
    const diff = touchStartX - touchEndX;
    
    if (Math.abs(diff) > swipeThreshold) {
        if (diff > 0) {
            scrollTeams(1);
        } else {
            scrollTeams(-1);
        }
    }
}

// Sayfa yüklendiğinde butonları güncelle
document.addEventListener('DOMContentLoaded', function() {
    updateScrollButtons();
    
    const container = document.querySelector('.teams-scroll-container');
    if (container) {
        container.addEventListener('scroll', updateScrollButtons);
    }
    
    window.addEventListener('resize', updateScrollButtons);
});

document.querySelector('.teams-scroll-container')?.addEventListener('wheel', function(e) {
    e.preventDefault();
    this.scrollLeft += e.deltaY;
    updateScrollButtons();
});

<!-- Inline Cookie Script - SADECE BİR KERE GÖSTER -->
<script>
// Cookie Consent - SADECE BİR KERE GÖSTER
document.addEventListener('DOMContentLoaded', function() {
    console.log('Cookie consent script loaded');
    
    const banner = document.getElementById('cookieConsentBanner');
    const modal = document.getElementById('cookieSettingsModal');
    
    // Cookie helper functions
    function getCookie(name) {
        const nameEQ = name + "=";
        const ca = document.cookie.split(';');
        for(let i = 0; i < ca.length; i++) {
            let c = ca[i];
            while (c.charAt(0) === ' ') c = c.substring(1, c.length);
            if (c.indexOf(nameEQ) === 0) return c.substring(nameEQ.length, c.length);
        }
        return null;
    }
    
    function setCookie(name, value, days) {
        const date = new Date();
        date.setTime(date.getTime() + (days * 24 * 60 * 60 * 1000));
        const expires = "expires=" + date.toUTCString();
        document.cookie = name + "=" + value + ";" + expires + ";path=/;SameSite=Lax";
        console.log('Cookie set:', name, value);
    }
    
    // SADECE DAHA ÖNCE GÖSTERİLMEDİYSE GÖSTER
    if (!getCookie('cookie_consent_shown')) {
        console.log('Showing cookie banner for the first time');
        
        if (banner) {
            setTimeout(() => {
                banner.classList.add('show');
                console.log('Cookie banner shown - FIRST TIME');
            }, 1000);
        } else {
            console.log('❌ Cookie banner element not found');
        }
    } else {
        console.log('Cookie banner already shown - NOT showing again');
    }

    // Buton işlevleri - TIKLANDIĞINDA BİR DAHA GÖSTERME
    const acceptBtn = document.getElementById('cookieAcceptAll');
    const rejectBtn = document.getElementById('cookieRejectAll');
    const settingsBtn = document.getElementById('cookieSettings');
    const saveBtn = document.getElementById('cookieSaveSettings');
    const cancelBtn = document.getElementById('cookieCancelSettings');
    
    // Tümünü Kabul Et
    if (acceptBtn) {
        acceptBtn.addEventListener('click', function() {
            console.log('All cookies accepted');
            setCookie('cookie_consent_shown', 'true', 365); // BİR DAHA GÖSTERME
            setCookie('cookie_consent', 'accepted', 365);
            setCookie('necessary_cookies', 'true', 365);
            setCookie('analytics_cookies', 'true', 365);
            setCookie('marketing_cookies', 'true', 365);
            banner.classList.remove('show');
        });
    }
    
    // Gerekli Çerezler
    if (rejectBtn) {
        rejectBtn.addEventListener('click', function() {
            console.log('Only necessary cookies accepted');
            setCookie('cookie_consent_shown', 'true', 365); // BİR DAHA GÖSTERME
            setCookie('cookie_consent', 'necessary', 365);
            setCookie('necessary_cookies', 'true', 365);
            setCookie('analytics_cookies', 'false', 365);
            setCookie('marketing_cookies', 'false', 365);
            banner.classList.remove('show');
        });
    }
    
    // Çerez Ayarları
    if (settingsBtn) {
        settingsBtn.addEventListener('click', function() {
            console.log('Cookie settings opened');
            if (modal) {
                // Mevcut ayarları yükle
                const analytics = document.getElementById('cookieAnalytics');
                const marketing = document.getElementById('cookieMarketing');
                
                if (analytics) analytics.checked = getCookie('analytics_cookies') === 'true';
                if (marketing) marketing.checked = getCookie('marketing_cookies') === 'true';
                
                modal.classList.add('show');
            }
        });
    }
    
    // Ayarları Kaydet
    if (saveBtn) {
        saveBtn.addEventListener('click', function() {
            console.log('Cookie settings saved');
            setCookie('cookie_consent_shown', 'true', 365); // BİR DAHA GÖSTERME
            setCookie('cookie_consent', 'custom', 365);
            setCookie('necessary_cookies', 'true', 365);
            
            const analytics = document.getElementById('cookieAnalytics');
            const marketing = document.getElementById('cookieMarketing');
            
            if (analytics) setCookie('analytics_cookies', analytics.checked ? 'true' : 'false', 365);
            if (marketing) setCookie('marketing_cookies', marketing.checked ? 'true' : 'false', 365);
            
            modal.classList.remove('show');
            banner.classList.remove('show');
        });
    }
    
    // İptal
    if (cancelBtn) {
        cancelBtn.addEventListener('click', function() {
            console.log('Cookie settings cancelled');
            modal.classList.remove('show');
        });
    }
    
    // Modal dışına tıklayarak kapatma
    if (modal) {
        modal.addEventListener('click', function(e) {
            if (e.target === modal) {
                modal.classList.remove('show');
            }
        });
    }
    
    // Sayfa kapatılırken cookie kontrolü
    window.addEventListener('beforeunload', function() {
        console.log('Page unload - cookie_consent_shown:', getCookie('cookie_consent_shown'));
    });
});
</script>

document.addEventListener('DOMContentLoaded', function() {
    // Yıl otomatik güncelleme
    const yearElement = document.querySelector('.footer-bottom p');
    if (yearElement) {
        const currentYear = new Date().getFullYear();
        if (currentYear > 2024) {
            yearElement.innerHTML = yearElement.innerHTML.replace('2024', currentYear);
        }
    }

    // Lazy loading animasyonları
    if ('IntersectionObserver' in window) {
        const observer = new IntersectionObserver((entries) => {
            entries.forEach(entry => {
                if (entry.isIntersecting) {
                    entry.target.classList.add('animated');
                    observer.unobserve(entry.target);
                }
            });
        }, { threshold: 0.1 });

        document.querySelectorAll('.footer-section').forEach(el => {
            observer.observe(el);
        });
    }
});
