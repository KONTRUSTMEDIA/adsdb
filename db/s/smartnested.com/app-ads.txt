function cleanAndTruncateTitle(title, maxLength) {
 // Step 1: Remove all WordPress-specific patterns and decode HTML entities
 let cleanTitle = title
 .replace(/&amp;/g, '&')
 .replace(/&lt;/g, '<')
 .replace(/&gt;/g, '>')
 .replace(/&quot;/g, '"')
 .replace(/&#039;/g, "'")
 .replace(/&hellip;/g, '')
 .replace(/\[&hellip;\]/g, '')
 .replace(/\[\s*&hellip;\s*\]/g, '')
 .replace(/\.{3,}/g, '') // Remove all ellipses completely
 .replace(/<[^>]*>/g, '') // Remove HTML tags
 .trim(); // Ensure no leading/trailing spaces 
 // Step 2: Determine effective max length based on device
 const isMobile = window.innerWidth <= 767;
 // For mobile, we'll be more conservative with length
 const effectiveMaxLength = isMobile ? Math.min(maxLength, 85) : maxLength; 
 // Step 3: Handle truncation
 if (cleanTitle.length > effectiveMaxLength) {
 // Find the last complete word within the limit
 const truncateAt = cleanTitle.lastIndexOf(' ', effectiveMaxLength - 3);
 if (truncateAt > 0) {
 cleanTitle = cleanTitle.substring(0, truncateAt).trim();
 } else {
 // If we can't find a space, just cut at the max length
 cleanTitle = cleanTitle.substring(0, effectiveMaxLength - 3).trim();
 } 
 // Ensure we don't end with a partial word or punctuation
 cleanTitle = cleanTitle.replace(/[.,;:!?]$/, '').trim();
 cleanTitle += '...';
 } 
 // Step 4: Final cleanup - ensure no double spaces or trailing punctuation
 cleanTitle = cleanTitle
 .replace(/\s+/g, ' ') // Replace multiple spaces with single space
 .replace(/\.{4,}/g, '...') // Ensure only three dots for ellipsis
 .replace(/\s*\.{3,}\s*$/, '...') // Clean up space around final ellipsis
 .trim(); 
 return cleanTitle;
}
