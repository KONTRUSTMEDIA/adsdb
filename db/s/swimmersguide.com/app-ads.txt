    function searchClick() {
        $("#zoomInfo").css('visibility', 'hidden');
        var geocoder = new google.maps.Geocoder();
        geocoder.geocode({ 'address': $("#txtAutocomplete").val() }, function (results, status) {
            if (status == google.maps.GeocoderStatus.OK) {
                latitude = results[0].geometry.location.lat();
                longitude = results[0].geometry.location.lng();
                var map = $("#map").getKendoMap();
                if (map.zoom() != 12)
                    map.zoom(12);
                //map.center([latitude, longitude]);
                
                locationSearched = true;

                changeCenterLocation(map, latitude, longitude);
                map.layers[1].dataSource.read();
            }
        });
    }

    function onGridFindChange(e) {
        var grid = $("#gridFacilities").data("kendoGrid");
        var selectedValue = e.sender.value();
        var filter1 = { field: "DistanceString", operator: "contains", value: selectedValue }
        var filter2 = { field: "MapPinName", operator: "contains", value: selectedValue }
        var filter3 = { field: "City", operator: "contains", value: selectedValue }
        var filter4 = { field: "Label", operator: "contains", value: selectedValue }
        var filter5 = { field: "PublicAccessType", operator: "contains", value: selectedValue }
        var filter6 = { field: "HotelAccess", operator: "contains", value: selectedValue }
        var filter7 = { field: "Teams", operator: "contains", value: selectedValue }
        var filter8 = { field: "AdditionalTile", operator: "contains", value: selectedValue }
        grid.dataSource.filter({ logic: "or", filters: [filter1, filter2, filter3, filter4, filter5, filter6, filter7, filter8] });
    }
