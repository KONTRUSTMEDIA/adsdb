<script>
    let secondi = 4;
    const total = secondi;
    const timerElement = document.getElementById('timer');
    const barElement = document.getElementById('bar');

    // Avvio immediato della barra per un effetto più fluido
    setTimeout(() => {
        barElement.style.width = "25%"; // Inizio immediato
    }, 100);

    const interval = setInterval(() => {
        secondi--;
        if(secondi >= 0) {
            timerElement.innerText = secondi;
            let progress = ((total - secondi) / total) * 100;
            barElement.style.width = progress + '%';
        }

        if (secondi <= 0) {
            clearInterval(interval);
            window.location.href = "https://www.storiadellarte.com";
        }
    }, 1000);
</script>
