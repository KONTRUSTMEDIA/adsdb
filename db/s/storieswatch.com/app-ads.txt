        function parseInstagramInput(input) {
            // Regex patterns
            const patterns = {
                reel: /instagram\.com\/reel\/([^\/?]+)/,
                story: /instagram\.com\/stories\/([^\/?]+)/,
                username_url: /instagram\.com\/([^\/?]+)/,
                username: /^[A-Za-z0-9._]+$/ // Only match usernames that are not part of a URL
            };

            for (let type in patterns) {
                if (patterns.hasOwnProperty(type)) {
                    let pattern = patterns[type];
                    let matches = input.match(pattern);
                    if (matches) {
                        // For username_url, make sure it's not a reel or story
                        if (type === 'username_url' && matches[1]) {
                            if (!matches[1].includes('reel') && !matches[1].includes('stories')) {
                                return ['username_url', matches[1]];
                            }
                        } else {
                            return [type, matches[1]];
                        }
                    }
                }
            }

            // If no URL pattern matched, check if it's a plain username
            if (patterns.username.test(input)) {
                return ['username', input];
            }

            return ['unknown', null];
        }

        function fixUrl(url) {
            // Fix URL if protocol is missing
            if (!url.startsWith('http://') && !url.startsWith('https://')) {
                url = 'https://' + url;
            }
            return url;
        }

        function extractUsername(parsedInput, originalQuery) {
            if (parsedInput[0] === 'username') {
                return originalQuery;
            } else if (parsedInput[0] === 'username_url') {
                return parsedInput[1];
            } else if (parsedInput[0] === 'story') {
                return parsedInput[1];
            }
            return null;
        }
        
        

function handleSearch(event) {
    event.preventDefault();
    const searchInput = document.getElementById('search-w');
    const query = searchInput.value.trim();
    
    if (!query) {
        return false;
    }

    window.history.pushState(null, null, '#search=' + encodeURIComponent(query));

    // Hide content section when searching
    document.getElementById('contentSection').style.display = 'none';

    // Show loading
    showLoading();
    
    // Parse Instagram input
    const parsedInput = parseInstagramInput(query);
    const username = extractUsername(parsedInput, query);
    
    if (!username) {
        hideLoading();
        showContentSection();

        window.history.replaceState(null, null, window.location.pathname);
        alert('Please enter a valid Instagram username or URL');
        return false;
    }

    // Analytics event
    if (typeof gtag !== 'undefined') {
        gtag('event', 'search', {
            'event_category': 'engagement',
            'event_label': username,
            'value': 1
        });
    }

    // Fetch profile data
    fetchProfile(username);
    return false;
}

        function downloadStory(url, type, button) {
            // Show downloading state
            const originalText = button.innerHTML;
            button.innerHTML = '⬇️ Downloading...';
            button.disabled = true;
            
            // Reset button after 3 seconds
            setTimeout(() => {
                button.innerHTML = originalText;
                button.disabled = false;
            }, 3000);

            // Create download link using mevcut downloadResource fonksiyonu
            return downloadResource(url, getFilename(url));
        }

        function downloadResource(url, filename) {
            if (!filename) filename = url.split('/').pop().split('?')[0];
            
            // Use direct link for better performance
            const a = document.createElement('a');
            a.href = url;
            a.download = filename;
            a.target = '_blank';
            a.rel = 'noopener noreferrer';
            document.body.appendChild(a);
            a.click();
            document.body.removeChild(a);
            
            return false;
        }

        function getFilename(url) {
            const path = new URL(url).pathname;
            const segments = path.split('/');
            return segments[segments.length - 1];
        }

        function startCountdown(resetTime) {
            const timer = document.getElementById('waitTime');
            
            function updateTimer() {
                const now = new Date().getTime();
                const distance = resetTime.getTime() - now;
                
                if (distance > 0) {
                    const minutes = Math.floor((distance % (1000 * 60 * 60)) / (1000 * 60));
                    const seconds = Math.floor((distance % (1000 * 60)) / 1000);
                    
                    timer.textContent = `${minutes.toString().padStart(2, '0')}:${seconds.toString().padStart(2, '0')}`;
                    setTimeout(updateTimer, 1000);
                } else {
                    timer.textContent = "00:00";
                }
            }
            
            updateTimer();
        }

        }

    currentTab = tabName;
}

function createHighlightCard(highlight, index) {
    const card = document.createElement('div');
    card.className = 'highlight-card';

function loadHighlightStories(highlightId, title) {
    showLoading();
    
    fetch(`https://www.storieswatch.com/ajax/highlight_stories/${encodeURIComponent(highlightId)}`)
        .then(response => response.json())
        .then(data => {
            hideLoading();

            }

        .catch(error => {
            hideLoading();
            console.error('Error loading highlight stories:', error);
            alert('Failed to load highlight stories');
        });
}
