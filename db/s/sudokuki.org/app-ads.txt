         /* ---------------- SUDOKU ENGINE ---------------- */
         function isSafe(b,r,c,n){
           for(let i=0;i<9;i++)
             if(b[r][i]==n||b[i][c]==n) return false;
           let br=Math.floor(r/3)*3, bc=Math.floor(c/3)*3;
           for(let i=0;i<3;i++)
             for(let j=0;j<3;j++)
               if(b[br+i][bc+j]==n) return false;
           return true;
         }
         
         function fillBoard(b){
           for(let r=0;r<9;r++){
             for(let c=0;c<9;c++){
               if(b[r][c]==0){
                 let nums=[1,2,3,4,5,6,7,8,9].sort(()=>Math.random()-0.5);
                 for(let n of nums){
                   if(isSafe(b,r,c,n)){
                     b[r][c]=n;
                     if(fillBoard(b)) return true;
                     b[r][c]=0;
                   }
                 }
                 return false;
               }
             }
           }
           return true;
         }
               
         function removeNumbers(b, count){
           const cells=[...Array(81).keys()].sort(()=>Math.random()-0.5);
           for(let i=0;i<count;i++){
             const r=Math.floor(cells[i]/9);
             const c=cells[i]%9;
             b[r][c]=0;
           }
         }
         
         function isSudokuSolved() {
          for (let r = 0; r < 9; r++) {
            for (let c = 0; c < 9; c++) {
              if (puzzle[r][c] === 0) return false;
              if (puzzle[r][c] !== solution[r][c]) return false;
            }
          }
          return true;
         }  
         
         function generateSudoku(diff){
           const base=Array.from({length:9},()=>Array(9).fill(0));

           puzzle=base.map(r=>r.slice());
           removeNumbers(puzzle, difficultyMap[diff]);
           fixed=puzzle.map(row=>row.map(v=>v!==0));
         }

         let adRefreshInterval = null;
         let pageLoadTime = Date.now();
         
         function getRefreshTime() {
           const elapsed = (Date.now() - pageLoadTime) / 1000; // seconds
           if (elapsed < 30) {
             return 10000; // first 30 sec → 10 sec refresh
           } else {
             return 35000; // after 30 sec → 35 sec refresh
           }
         }
         
         function refreshAds() {
           if (document.visibilityState === 'visible') {
             console.log('Refreshing ad...');
             if (window.googletag && googletag.pubads) {
               googletag.pubads().refresh();
         
               function onSlotRender(event) {
                 if (event.slot !== interstitialSlot) return;
                 if (event.isEmpty) {
                   hideOverlay();
                   return;
                 }
                 showOverlay();
               }
             }
           }
         
           // Restart interval with updated timing
           clearInterval(adRefreshInterval);
           adRefreshInterval = setInterval(refreshAds, getRefreshTime());
         }
         
         function startAdRefresh() {
           if (adRefreshInterval) return;
         
           // Show ad on page load immediately
           if (window.googletag && googletag.pubads) {
             googletag.pubads().refresh();
           }
         
           adRefreshInterval = setInterval(refreshAds, getRefreshTime());
         }
         
         function stopAdRefresh() {
           if (adRefreshInterval) {
             clearInterval(adRefreshInterval);
             adRefreshInterval = null;
             console.log('Ad refresh paused (tab inactive)');
           }
         }
         
         // Initial load
         if (document.visibilityState === 'visible') {
           startAdRefresh();
         }
         
         // Listen for tab visibility changes
         document.addEventListener('visibilitychange', () => {
           if (document.visibilityState === 'visible') {
             startAdRefresh();
           } else {
             stopAdRefresh();
           }
         });
      </script>
