    let currentOpen = null; // Track the currently open item

    function fetchDownloadLink(idfile) {
        // Ignore if a fetch request is already ongoing
        if (isFetching) {
            return;
        }

        const resultDiv = document.getElementById(`result-${idfile}`);

        // Close previously open item if different
        if (currentOpen && currentOpen !== idfile) {
            const previousResultDiv = document.getElementById(`result-${currentOpen}`);
            if (previousResultDiv) {
                previousResultDiv.style.display = "none";
                previousResultDiv.innerHTML = "<span class='loading-text'>Loading, please wait...</span>";
            }
            currentOpen = null; // Reset currentOpen immediately
        }

        // Show loading state and lock fetch
        isFetching = true;
        resultDiv.style.display = "block";
        resultDiv.innerHTML = "<img src='https://media.tenor.com/On7kvXhzml4AAAAj/loading-gif.gif' width=25 alt='Loading' style='vertical-align: middle;'> Loading, please wait...";

            .then(data => {
                resultDiv.innerHTML = data;
                currentOpen = idfile; // Update the current open item
            })
            .catch(error => {
                resultDiv.innerHTML = "<p style='color:red;'>Error fetching data!</p>";
                console.error("Error:", error);
            })
            .finally(() => {
                isFetching = false; // Release the fetch lock

    document.addEventListener('DOMContentLoaded', () => {

            if (playIcon) {
                const trackItem = playIcon.closest('.track-item');
                const ytid = trackItem.dataset.ytid;

                // Jika track yang sama diklik, hentikan pemutar
                if (currentTrack === trackItem) {
                    youtubePlayer.src = '';
                    playerBar.classList.add('hidden');
                    playIcon.classList.remove('pause'); // Kembali ke ikon play
                    currentTrack = null;
                    return;
                }

                // Reset ikon pada track sebelumnya
                if (currentTrack) {

                    previousIcon.classList.remove('pause'); // Kembali ke ikon play
