  $('.contact-advertiser').hide();

    $('.contact-advertiser').show();
  });

    $('.contact-publisher').show();

  });

  $('#buttonSend').click(function () {
    $('.error').remove();
    $('.success').remove();
    $('#buttonSend').attr('disabled', true).css('cursor', 'default');
    if ($('.button-advertiser').hasClass('active-button')) {
      $.post('/sendemail-advertiser', {
        params: {
          'name': $('#name').val(),
          'company': $('#company_advertiser').val(),
          'email': $('#email_advertiser').val(),
          'phone_number': $('#phone_number').val(),
          'content': $('#content').val(),
          'recaptcha': $('#g-recaptcha-response').val()
        }
      }, function (data) {
        $('#buttonSend').removeAttr('disabled').css('cursor', 'pointer');
        if (data.status !== undefined && data.status === 'ERROR') {
          for (var prop in data.errors) {
            if (data.errors.hasOwnProperty(prop)) {
              if (typeof (data.errors[prop]) !== 'undefined' && typeof (data.errors[prop]) === 'object') {
                var txt = Object.values(data.errors[prop]);
                data.errors[prop] = txt;
              }
              if (prop === 'recaptcha_response_field' || prop === 'recaptcha') {
                $('<div class="error" style="top:0px;">' + data.errors[prop] + '</div>').insertAfter('.g-recaptcha');
              } else {
                $('<div class="error">' + data.errors[prop] + '</div>').insertAfter('#' + prop);
              }
            }
          }
        } else if (data.status === 'OK') {
          if (data.result.status === 'OK') {
            $('<div class="col-12 alert alert-success my-2">Wiadomość wysłana</div>').insertBefore('.contact-select');
          } else {
            $('<div class="col-12 alert alert-error my-2">Podczas wysyłania wiadomości wystąpił błąd</div>').insertBefore('.contact-select');
          }
        }
      }, 'json');
    } else if ($('.button-publisher').hasClass('active-button')) {
      $.post('/sendemail', {
        params: {
          'website': $('#website').val(),
          'email': $('#email').val(),
          'u-name': $('#u-name').val(),
          'uu': $('#uu').val(),
          'pv': $('#pv').val(),
          'recaptcha': $('#g-recaptcha-response').val()
        }
      }, function (data) {
        $('#buttonSend').removeAttr('disabled').css('cursor', 'pointer');
        if (data.status !== undefined && data.status === 'ERROR') {
          for (var prop in data.errors) {
            if (data.errors.hasOwnProperty(prop)) {
              if (typeof (data.errors[prop]) !== 'undefined' && typeof (data.errors[prop]) === 'object') {
                var txt = Object.values(data.errors[prop]);
                data.errors[prop] = txt;
              }
              if (prop === 'recaptcha_response_field' || prop === 'recaptcha') {
                $('<div class="error" style="top:0px;">' + data.errors[prop] + '</div>').insertAfter('.g-recaptcha');
              } else {
                $('<div class="error">' + data.errors[prop] + '</div>').insertAfter('#' + prop);
              }
            }
          }
        } else if (data.status === 'OK') {
          if (data.result.status === 'OK') {
            $('<div class="col-12 alert alert-success my-2">Wiadomość wysłana</div>').insertBefore('.contact-select');
