    <script>
        document.addEventListener('DOMContentLoaded', function () {
            var heroes = null;

            function renderResults(filteredHeroes, results) {
                results.innerHTML = '';

                if (filteredHeroes.length === 0) {
                    results.style.display = 'none';
                    return;
                }

                filteredHeroes.forEach(hero => {
                    var image = `/images/predecessor/hero-image-data/${hero.display_name.replaceAll(' ', '-')}.webp`;
                    var buildLink = `/predecessor/heroes/${hero.name}/builds`;

                    const item = document.createElement('div');
                    item.className = 'search-result-item';

                    const link = document.createElement('a');
                    link.href = buildLink;

                    const img = document.createElement('img');
                    img.src = image;
                    img.alt = hero.display_name;

                    const heroInfo = document.createElement('div');
                    heroInfo.className = 'search-result-item--info';

                    const heroName = document.createElement('div');
                    heroName.className = 'search-result-item--info---name';
                    heroName.textContent = hero.display_name;

                    const heroRoles = document.createElement('div');
                    heroRoles.className = 'search-result-item--info---roles';
                    heroRoles.textContent = hero.roles.join(', ');

                    heroInfo.appendChild(heroName);
                    heroInfo.appendChild(heroRoles);
                    link.appendChild(img);
                    link.appendChild(heroInfo);
                    item.appendChild(link);

                    results.appendChild(item);
                });

                results.style.display = 'block';
            }

            var currentPath = "/error404";
            var input;
            const results = document.getElementById('search-results')
            if (currentPath == "/") {
                input = document.querySelector('.game-landing-page-search--input');
            } else {
                input = document.querySelector('.header-search-bar--input');
            }
            input.addEventListener('input', function () {
                const query = input.value.toLowerCase();
                const filteredHeroes = heroes.filter(hero => hero.name.toLowerCase().includes(query) || hero.display_name.toLowerCase().includes(query));
                renderResults(filteredHeroes, results);
            });

            document.addEventListener('click', function (event) {
                const isClickInside = input.contains(event.target) || document.getElementById('search-results').contains(event.target);
                if (!isClickInside) {
                    document.getElementById('search-results').style.display = 'none';
                }
            });

            input.addEventListener('click', function (event) {
                event.stopPropagation();
                const query = input.value.toLowerCase();
                if (query) {
                    const filteredHeroes = heroes.filter(hero => hero.name.toLowerCase().includes(query) || hero.display_name.toLowerCase().includes(query));
                    renderResults(filteredHeroes, results);
                }
            });
        });
    </script>
