      function tryLoadStatsig()
      {
        initDevMode();

        // If already loaded, just update and return
        if (g_statsigLoaded){
          tryUpdateStatsig();
          return Promise.resolve();
        }

        // If initialization already in progress, return existing promise
        // This prevents multiple simultaneous initializations
        if (window._statsigInitPromise) {
          return window._statsigInitPromise;
        }

        hydrateIds();

        //console.log("Try load statsig!");

        // Store promise to prevent concurrent initialization attempts
        window._statsigInitPromise = new Promise((resolve)=>{
          var ssTag = document.createElement('script');

              if (!g_isDevMode){
