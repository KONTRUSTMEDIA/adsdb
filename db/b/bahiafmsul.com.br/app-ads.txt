        let urlFetchScheduleList = '/ajax/programacao/mundobahiafm/programacao'
    
    let containerProgramacao = document.querySelector('.ib-programacaoWidget .content')
    document.addEventListener('DOMContentLoaded', () => {
        getScheduleList(urlFetchScheduleList)
    })
    async function getScheduleList(url) {
        const d = new Date();
        const response = await fetch(url)
        const data = await response.json();
        mountScheduleList(data[Object.keys(data)[d.getDay() - 1]], 'on')
    } 
    function getProgramasAtivos(schedule) {
        const now = new Date();
        const filteredProgramas = schedule.filter(programa => {
            const start = new Date(`${now.toDateString()} ${programa.inicio}`);
            const end = new Date(`${now.toDateString()} ${programa.fim}`);
            return start <= now && end >= now;
        });
        const futureProgramas = schedule.filter(programa => {
            const start = new Date(`${now.toDateString()} ${programa.inicio}`);
            const end = new Date(`${now.toDateString()} ${programa.fim}`);
            return start > now;
        }).slice(0, 2);
        return [...filteredProgramas, ...futureProgramas];
    } 

    function mountScheduleList(schedule, live) {
        let programas = getProgramasAtivos(schedule)
        const now = new Date();
        let horas = now.getHours().toString().padStart(2, '0');
        let minutos = now.getMinutes().toString().padStart(2, '0');
        let segundos = now.getSeconds().toString().padStart(2, '0');
        let horarioAtual = `${horas}:${minutos}:${segundos}`;
        const scheduleItems = programas.map(({ programa, descrição, inicio, fim, imagem, locutor }) => {
            const isPlaying = horarioAtual >= inicio && horarioAtual < fim;
            const isLive = live === 'on' && isPlaying;
            return `
      <div class="programa fade-in ${isLive ? 'liveNow' : ''}">
        <figure>
          <img loading="lazy" src="${imagem}" alt="${programa}" />
        </figure>
        <div>
          <h2>${programa}</h2>
          <h3>${(locutor.trim().length > 0 ? locutor : '--')}</h3>
          <p>${descrição}</p>
        </div>
        <button>
          <img loading="lazy" class="plus" src="/themes/default/assets/img/radios/plus.svg" alt="${programa}" />
          <img loading="lazy" class="minus" src="/themes/default/assets/img/radios/minus.svg" alt="${programa}" />
        </button>
        <p>${descrição}</p>
      </div>
    `;
        });
        containerProgramacao.innerHTML = scheduleItems.join('');
        const buttonsList = containerProgramacao.querySelectorAll('.programa');
        buttonsList.forEach((button) =>
            button.addEventListener('click', () => {
                if (button.classList.contains('active')) { return button.classList.remove('active') }
                buttonsList.forEach((button) => button.classList.remove('active'))
                button.classList.add('active')
            }))
    }
