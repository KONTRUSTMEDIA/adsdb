(function () {

    '<button type="button" aria-label="Kapat" style="position:absolute;top:6px;right:8px;background:transparent;border:0;color:#fff;font-size:16px;cursor:pointer;">&times;</button>'+

    '<div class="biv-arrow"></div>';
  document.body.appendChild(tip);

  function place(){
    const r = target.getBoundingClientRect();

    const visible =
      r.width  > 0 &&
      r.height > 0 &&
      r.bottom > 0 &&
      r.right  > 0 &&
      r.top    < window.innerHeight &&
      r.left   < window.innerWidth;

    if (!visible) {
      tip.style.display = 'none';
      tip.classList.remove('visible');
      return;
    }

    if (!tip.classList.contains('visible')) tip.style.display = 'block';

    const tw = tip.offsetWidth, th = tip.offsetHeight, gap = 12;
    const cx = r.left + r.width/2;
    let left = Math.max(8, Math.min(cx - tw*0.6, window.innerWidth - tw - 8));
    let top  = Math.max(8, r.top - th - gap);

    tip.style.left = left+'px';
    tip.style.top  = top +'px';
    tip.style.setProperty('--arrow-x', (cx - left) + 'px');
    tip.classList.add('visible');

  setTimeout(place, 150);

  window.addEventListener('scroll', place,  {passive:true});

  tip.querySelector('button[aria-label="Kapat"]').addEventListener('click', (e)=>{
    e.preventDefault();
    tip.remove();

  });

})();

<script>
(function() {
    'use strict';
    
    var COOKIE_NAME = 'biv_cookie_consent';
    var COOKIE_DAYS = 365;
    var SHOW_DELAY = 2000;
    
    function getCookie(name) {
        var value = '; ' + document.cookie;
        var parts = value.split('; ' + name + '=');
        if (parts.length === 2) return parts.pop().split(';').shift();
        return null;
    }
    
    function setCookie(name, value, days) {
        var date = new Date();
        date.setTime(date.getTime() + (days * 24 * 60 * 60 * 1000));
        var expires = 'expires=' + date.toUTCString();
        document.cookie = name + '=' + value + ';' + expires + ';path=/;SameSite=Lax';
    }
    
    var banner = document.getElementById('bivCookieBanner');
    var acceptBtn = document.getElementById('bivCookieAccept');
    var rejectBtn = document.getElementById('bivCookieReject');
    
    if (!banner || !acceptBtn || !rejectBtn) return;
    
    var consent = getCookie(COOKIE_NAME);
    
    if (!consent) {
        setTimeout(function() {
            banner.classList.add('active');
        }, SHOW_DELAY);
    }
    
    function hideBanner() {
        banner.classList.remove('active');
        setTimeout(function() {
            banner.style.display = 'none';
        }, 400);
    }
    
    acceptBtn.addEventListener('click', function() {
        setCookie(COOKIE_NAME, 'accepted', COOKIE_DAYS);
        hideBanner();
        
        if (typeof gtag === 'function') {
            gtag('consent', 'update', {
                'analytics_storage': 'granted',
                'ad_storage': 'granted'
            });
        }
    });
    
    rejectBtn.addEventListener('click', function() {
        setCookie(COOKIE_NAME, 'rejected', COOKIE_DAYS);
        hideBanner();
        
        if (typeof gtag === 'function') {
            gtag('consent', 'update', {
                'analytics_storage': 'denied',
                'ad_storage': 'denied'
            });
        }
    });
    
})();
</script>

// Cookie yardımcı fonksiyonları
function bivGetClosedNotifications() {
    try {
        const cookie = document.cookie.split('; ').find(row => row.startsWith('biv_closed_notifications='));
        if (cookie) {
            return JSON.parse(decodeURIComponent(cookie.split('=')[1]));
        }
    } catch (e) {}
    return [];
}

function bivSaveClosedNotification(id) {
    let closed = bivGetClosedNotifications();
    if (!closed.includes(id)) {
        closed.push(id);
    }
    // 30 gün geçerli cookie
    const expires = new Date(Date.now() + 30 * 24 * 60 * 60 * 1000).toUTCString();
    document.cookie = 'biv_closed_notifications=' + encodeURIComponent(JSON.stringify(closed)) + '; expires=' + expires + '; path=/; SameSite=Lax';
}

function bivKapat(id, tip, birKez) {
    let el;
    if (tip === 'popup') {
        el = document.querySelector('.biv-popup[data-id="' + id + '"]');
        const overlay = document.querySelector('.biv-popup-overlay[data-for="' + id + '"]');
        if (overlay) overlay.style.display = 'none';
    } else if (tip === 'banner') {
        el = document.querySelector('.biv-banner[data-id="' + id + '"]');
        document.body.classList.remove('biv-has-banner');
    } else if (tip === 'slider') {
        el = document.querySelector('.biv-slider[data-id="' + id + '"]');
        document.body.classList.remove('biv-has-slider');
    } else {
        el = document.querySelector('.biv-toast[data-id="' + id + '"]');
    }
    
    if (el) {
        el.classList.add('biv-kapaniyor');
        setTimeout(() => el.remove(), 300);
    }
    
    // "1 kez göster" aktifse cookie'ye kaydet
    if (birKez == 1 || birKez === true || birKez === '1') {
        bivSaveClosedNotification(id);
    }
    
    // API'ye de bildir
    fetch('/filesystems/bildirim_api.php', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ islem: 'kapat', bildirim_id: id })
    }).catch(() => {});
}

function bivTiklandi(id) {
    fetch('/filesystems/bildirim_api.php', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ islem: 'tiklandi', bildirim_id: id })
    }).catch(() => {});
}

function toggleMenuSection(element) {
    const content = element.nextElementSibling;
    
    element.classList.toggle('active');
    
    if (content.style.display === 'none' || content.style.display === '') {
        content.style.display = 'block';
        content.style.maxHeight = content.scrollHeight + 'px';
    } else {
        content.style.maxHeight = '0px';
        setTimeout(() => {
            content.style.display = 'none';
        }, 300);
    }
}
