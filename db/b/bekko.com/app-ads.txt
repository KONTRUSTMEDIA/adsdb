		<script>
			var language = (navigator.language || navigator.browserLanguage).toLowerCase();
			if (language.indexOf('zh') > -1) window.location.href='zh/';
			else if (language.indexOf('tw') > -1||language.indexOf('hk') > -1) window.location.href='zh/';
			else if (language.indexOf('ja') > -1) window.location.href='ja/';
			else if (language.indexOf('ko') > -1) window.location.href='ko/';
			else window.location.href='zh/';
		</script>
