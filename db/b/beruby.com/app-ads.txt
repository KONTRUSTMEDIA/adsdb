        function updateWord() {
          const rotatingElement = document.getElementById("changing-word");

          // Remove class to restart animation
          if (rotatingElement) {
            if (rotatingElement.classList.contains("swipe-up")){
              rotatingElement.classList.remove("swipe-up");
            }

            // Changes text after short delay to restart animation
            setTimeout(() => {
              rotatingElement.textContent = words[index];
              rotatingElement.classList.add("swipe-up");
              index = (index + 1) % words.length;
            }, 50); // Delay
          }
        }

        setInterval(updateWord, 2500);
        updateWord();
