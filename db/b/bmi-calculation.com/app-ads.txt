document.addEventListener('DOMContentLoaded', function() {

  // GDPR Cookie Consent Management
  const cookieBanner = document.getElementById('cookie-banner');
  const cookieModal = document.getElementById('cookie-modal');
  const acceptBtn = document.getElementById('accept-cookies');
  const rejectBtn = document.getElementById('reject-cookies');
  const settingsBtn = document.getElementById('cookie-settings');
  const saveBtn = document.getElementById('save-preferences');
  const closeBtn = document.getElementById('close-modal');
  const analyticsToggle = document.getElementById('analytics-toggle');

  // Check if user has already made a choice
  function checkCookieConsent() {
    const consentRecorded = getCookie('cookie_consent_recorded');
    if (!consentRecorded) {
      // Show banner if no consent has been recorded yet
      cookieBanner.style.display = 'block';
    } else {
      // Hide banner if consent was already given/rejected
      cookieBanner.style.display = 'none';
      
      // Load analytics only if explicitly accepted
      if (getCookie('analytics_consent') === 'accepted') {
        loadGoogleAnalytics();
      }
    }
  }

  // Accept all cookies
  acceptBtn.addEventListener('click', function() {
    setCookie('cookie_consent_recorded', 'true', 365);
    setCookie('analytics_consent', 'accepted', 365);
    cookieBanner.style.display = 'none';
    loadGoogleAnalytics();
  });

  // Reject all cookies
  rejectBtn.addEventListener('click', function() {
    setCookie('cookie_consent_recorded', 'true', 365);
    setCookie('analytics_consent', 'rejected', 365);
    cookieBanner.style.display = 'none';
    removeAnalyticsCookies();
  });

  // Open settings modal
  settingsBtn.addEventListener('click', function() {
    cookieModal.style.display = 'flex';
    // Always set toggle based on current consent state - default to off if no choice
    if (getCookie('analytics_consent') === 'accepted') {
      analyticsToggle.classList.add('active');
    } else {
      analyticsToggle.classList.remove('active'); // Ensure it's off by default
    }
  });

  // Close modal
  closeBtn.addEventListener('click', function() {
    cookieModal.style.display = 'none';
  });

  // Close modal by clicking outside
  cookieModal.addEventListener('click', function(e) {
    if (e.target === cookieModal) {
      cookieModal.style.display = 'none';
    }
  });

  // Toggle analytics consent
  analyticsToggle.addEventListener('click', function() {
    if (!this.hasAttribute('data-disabled')) {
      this.classList.toggle('active');
    }
  });

  // Save preferences
  saveBtn.addEventListener('click', function() {
    const analyticsEnabled = analyticsToggle.classList.contains('active');
    
    setCookie('cookie_consent_recorded', 'true', 365);
    setCookie('analytics_consent', analyticsEnabled ? 'accepted' : 'rejected', 365);
    
    cookieBanner.style.display = 'none';
    cookieModal.style.display = 'none';
    
    if (analyticsEnabled) {
      loadGoogleAnalytics();
    } else {
      removeAnalyticsCookies();
    }
  });

  // Initialize consent check
  checkCookieConsent();

  // The rest of your BMI calculator code remains unchanged...
  // Height unit toggle functionality
  const heightRadios = document.querySelectorAll('input[name="heightUnit"]');
  const heightCmDiv = document.getElementById('heightCm');
  const heightFtDiv = document.getElementById('heightFt');

  heightRadios.forEach(radio => {
    radio.addEventListener('change', function() {
      // Update toggle button styling
      document.querySelectorAll('.toggle-group label').forEach(label => {
        label.classList.remove('active');
      });
      this.parentElement.classList.add('active');

      // Show/hide height input fields
      if (this.value === 'cm') {
        heightCmDiv.style.display = 'block';
        heightFtDiv.style.display = 'none';
      } else {
        heightCmDiv.style.display = 'none';
        heightFtDiv.style.display = 'block';
      }
    });
  });

  // Weight unit toggle functionality
  const weightRadios = document.querySelectorAll('input[name="weightUnit"]');
  weightRadios.forEach(radio => {
    radio.addEventListener('change', function() {
      // Update toggle button styling
      const weightLabels = document.querySelectorAll('input[name="weightUnit"]').forEach(r => {
        r.parentElement.classList.remove('active');
      });
      this.parentElement.classList.add('active');
    });
  });

  // BMI calculation functionality
  const bmiForm = document.getElementById('bmiForm');
  const resultsSection = document.getElementById('resultsSection');

  bmiForm.addEventListener('submit', function(e) {
    e.preventDefault();
    
    // Get form values
    const heightUnit = document.querySelector('input[name="heightUnit"]:checked').value;
    const weightUnit = document.querySelector('input[name="weightUnit"]:checked').value;
    const weightInput = document.getElementById('weightInput').value;

    let height, weight;

    // Get height in meters
    if (heightUnit === 'cm') {
      const heightCm = document.getElementById('heightCmInput').value;
      if (!heightCm) {
        alert('Please enter your height');
        return;
      }
      height = heightCm / 100; // Convert to meters
    } else {
      const heightFt = document.getElementById('heightFtInput').value;
      const heightIn = document.getElementById('heightInInput').value;
      if (!heightFt || !heightIn) {
        alert('Please enter your height in feet and inches');
        return;
      }
      height = (parseInt(heightFt) * 12 + parseInt(heightIn)) * 0.0254; // Convert to meters
    }

    // Get weight in kg
    if (weightUnit === 'kg') {
      weight = parseFloat(weightInput);
    } else {
      weight = parseFloat(weightInput) * 0.453592; // Convert lbs to kg
    }

    if (!weightInput) {
      alert('Please enter your weight');
      return;
    }

    // Calculate BMI
    const bmi = weight / (height * height);
    const bmiRounded = Math.round(bmi * 10) / 10;

    // Determine category
    let category, categoryClass;
    if (bmi < 18.5) {
      category = 'Underweight';
      categoryClass = 'underweight';
    } else if (bmi < 25) {
      category = 'Normal weight';
      categoryClass = 'normal';
    } else if (bmi < 30) {
      category = 'Overweight';
      categoryClass = 'overweight';
    } else if (bmi < 35) {
      category = 'Obese (Class I)';
      categoryClass = 'obese';
    } else if (bmi < 40) {
      category = 'Obese (Class II)';
      categoryClass = 'obese';
    } else {
      category = 'Obese (Class III)';
      categoryClass = 'obese';
    }

    // Calculate healthy weight range
    const minWeight = 18.5 * height * height;
    const maxWeight = 24.9 * height * height;
    let healthyRange;
    if (weightUnit === 'kg') {
      healthyRange = `${Math.round(minWeight)}-${Math.round(maxWeight)} kg`;
    } else {
      healthyRange = `${Math.round(minWeight * 2.20462)}-${Math.round(maxWeight * 2.20462)} lbs`;
    }

    // Display results
    document.getElementById('bmiValue').textContent = bmiRounded;
    document.getElementById('bmiCategory').textContent = category;

    // Update the animated gauge
    if (bmiGauge) {
      bmiGauge.set(bmiRounded);
    }

    // Display input values
    if (heightUnit === 'cm') {
      document.getElementById('heightValue').textContent = `${document.getElementById('heightCmInput').value} cm`;
    } else {
      document.getElementById('heightValue').textContent = `${document.getElementById('heightFtInput').value}'${document.getElementById('heightInInput').value}"`;
    }
    
    document.getElementById('weightValue').textContent = `${weightInput} ${weightUnit}`;
    document.getElementById('healthyRange').textContent = healthyRange;

    // Show explanation
    let explanation = '';
    if (bmi < 18.5) {
      explanation = `Your BMI indicates you may be underweight. This could be due to genetics, high metabolism, or insufficient nutrition. Consider consulting a healthcare provider to ensure you're meeting your nutritional needs.`;
    } else if (bmi < 25) {
      explanation = `Excellent! Your BMI falls within the healthy weight range. This is associated with lower risk of weight-related health problems. Maintain your current lifestyle with balanced nutrition and regular physical activity.`;
    } else if (bmi < 30) {
      explanation = `Your BMI indicates you may be overweight. This slightly increases your risk for health conditions like heart disease and diabetes. Consider gradual lifestyle changes including balanced nutrition and increased physical activity.`;
    } else {
      explanation = `Your BMI indicates obesity, which significantly increases health risks. We strongly recommend consulting with healthcare professionals for personalized guidance on safe and effective weight management strategies.`;
    }

    document.getElementById('explanationText').innerHTML = `<p>${explanation}</p>`;

    // Show results section FIRST
    resultsSection.classList.add('show');

    // Initialize gauge now that the canvas is visible
    initializeGauge();

    // Update the gauge with the BMI value
    if (bmiGauge) {
      bmiGauge.set(bmiRounded);
    }

    resultsSection.scrollIntoView({ behavior: 'smooth' });
  });
});
