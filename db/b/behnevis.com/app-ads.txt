      // JavaScript to show the alert and set its HTML content
      function recreateAlertIfNeeded() {
        if (!document.getElementById("alertBox")) {
          var alertHTML = `
              <div id="alertBox" class="alert alert-warning alert-dismissible fade fixed-top m-0" role="alert" style="display:none;">
                  <span id="alertText">Initial notification message!</span>
                  <button type="button" class="btn-close" aria-label="Close"></button>
              </div>
          `;
          document.body.insertAdjacentHTML("afterbegin", alertHTML);

          // Re-attach the event listener for the close button
          document.querySelector("#alertBox .btn-close").addEventListener("click", function () {
            var alertDiv = document.getElementById("alertBox");
            alertDiv.style.display = "none";
            alertDiv.classList.remove("show");
          });
        }
      }

      // JavaScript to show the alert and set its HTML content
      function showAlert(htmlContent) {
        recreateAlertIfNeeded();

        var alertDiv = document.getElementById("alertBox");
        var alertTextSpan = document.getElementById("alertText");

        if (alertDiv && alertTextSpan) {
          // Set the alert's HTML content
          alertTextSpan.innerHTML = htmlContent;

          // Make the alert visible
          alertDiv.style.display = "block";
          alertDiv.classList.add("show");
        } else {
          console.error("Alert elements not found!");
        }
      }

      function hideAlert() {
        var alertDiv = document.getElementById("alertBox");
        if (alertDiv) {
          alertDiv.style.display = "none"; // Hide the alert
          alertDiv.classList.remove("show"); // Remove 'show' class to hide it with Bootstrap as well
        }
      }

      function isSafari() {
        var userAgent = navigator.userAgent.toLowerCase();
        var isSafari = /^((?!chrome|android).)*safari/i.test(userAgent);

        return isSafari;
      }

      if (isSafari())
        showAlert(
          "You are using Safari browser which is not supported by the new Behnevis editor. We suggest using <a href='https://www.google.com/chrome/'>Google Chrome</a> instead."
