              const until = Number(localStorage.getItem(LS_KEY) || 0);

                var durum = shouldShow();
                if(durum){

<script>
document.addEventListener('DOMContentLoaded', function () {
  const tabListItems = document.querySelectorAll('.services-list-tab .tabs li');
  const tabPanels    = document.querySelectorAll('.tab_content .tabs_item');

  if (!tabListItems.length || !tabPanels.length) return;

  // İlk tab'ı aktif yap
  function activate(index){
    tabListItems.forEach(li => li.classList.remove('active'));
    tabPanels.forEach(p => p.classList.remove('active'));
    if (tabListItems[index]) tabListItems[index].classList.add('active');
    if (tabPanels[index])    tabPanels[index].classList.add('active');
  }

  // Tıklama
  tabListItems.forEach((li, idx) => {
    li.addEventListener('click', function(e){
      const a = li.querySelector('a');
      if (a) e.preventDefault(); // href="#" scroll etmesin
      activate(idx);
    });
  });

  // Başlangıç
  activate(0);
});
</script>
