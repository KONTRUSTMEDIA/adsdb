<script>
(function(){
  function setLang(lang){
    var nodes=document.querySelectorAll('[data-lang]');
    nodes.forEach(function(n){
      n.classList.remove('show');
      if(n.getAttribute('data-lang')===lang){n.classList.add('show');}
    });
    var toggles=document.querySelectorAll('[data-set-lang]');
    toggles.forEach(function(t){
      if(t.getAttribute('data-set-lang')===lang){t.classList.add('active');}
      else{t.classList.remove('active');}
    });
    document.documentElement.lang=(lang==='ja'?'ja':'en');
  }
  document.addEventListener('click',function(e){
    var target=e.target.closest('[data-set-lang]');
    if(target){setLang(target.getAttribute('data-set-lang'));}
  });
  setLang('ja'); // 初期表示：日本語にしてもいいし 'en' に変えてもOK
})();
</script>
