			function sortTable(table, th) {
				const tbody = table.querySelector('tbody');
				const rows = Array.from(tbody.querySelectorAll('tr:not(.group-child)'));
				const sortType = th.dataset.sort;
				const isAsc = !th.classList.contains('sort-asc');

				table.querySelectorAll('th').forEach(h => h.classList.remove('sort-asc', 'sort-desc'));
				th.classList.add(isAsc ? 'sort-asc' : 'sort-desc');

				// Collect group-child rows before sorting (while DOM is intact)
				const childMap = new Map();
				rows.forEach(row => {
					if (row.classList.contains('group-header')) {
						const children = [];
						let sibling = row.nextElementSibling;
						while (sibling && sibling.classList.contains('group-child')) {
							children.push(sibling);
							sibling = sibling.nextElementSibling;
						}
						childMap.set(row, children);
					}
				});

				let targetCol = 0;
				const allHeaders = Array.from(table.querySelectorAll('thead th'));
				for (let i = 0; i < allHeaders.indexOf(th); i++) {
					targetCol += parseInt(allHeaders[i].getAttribute('colspan') || 1);
				}

				// Find cell at visual column position (accounting for colspan in body rows)
				function getCellAtCol(row, col) {
					let currentCol = 0;
					for (let i = 0; i < row.cells.length; i++) {
						const cell = row.cells[i];
						const span = parseInt(cell.getAttribute('colspan') || 1);
						if (col >= currentCol && col < currentCol + span) {
							return cell;
						}
						currentCol += span;
					}
					return null;
				}

				rows.sort((a, b) => {
					const aCell = getCellAtCol(a, targetCol);
					const bCell = getCellAtCol(b, targetCol);
					if (!aCell || !bCell) return 0;

					let aVal = aCell.dataset.value ?? aCell.textContent.trim();
					let bVal = bCell.dataset.value ?? bCell.textContent.trim();

					let result = 0;
					if (sortType === 'number') {
						aVal = parseFloat(aVal.replace(/[^\d.-]/g, '')) || 0;
						bVal = parseFloat(bVal.replace(/[^\d.-]/g, '')) || 0;
						result = aVal - bVal;
					} else {
						result = aVal.localeCompare(bVal, 'sv');
					}
					return isAsc ? result : -result;
				});

				const fragment = document.createDocumentFragment();
				rows.forEach(row => {
					fragment.appendChild(row);
					if (childMap.has(row)) {
						childMap.get(row).forEach(child => fragment.appendChild(child));
					}
				});
				tbody.appendChild(fragment);
			}

    // Create a ball

        x: carX - 50,
        y: carY + 50,
        vx: 0, // Velocity in x direction
        vy: 0, // Velocity in y direction
        rotation: 0,
        rotationSpeed: 0

    function checkCollision(carX, carY, ball) {
        const dx = carX - ball.x;
        const dy = carY - ball.y;
        const distance = Math.sqrt(dx * dx + dy * dy);
        return distance < 30; // Adjust the collision distance as needed
    }

    function handleCollision(carX, carY, ball) {
        const angle = Math.atan2(ball.y - carY, ball.x - carX);
        const pushSpeed = 20; // Double the push speed
        ball.vx = Math.cos(angle) * pushSpeed;
        ball.vy = Math.sin(angle) * pushSpeed;
        ball.rotationSpeed = -360; // Add rotation speed
        ball.rotation = 0;
    }

    function checkGoal(ball) {
        const goalRect = goal.getBoundingClientRect();
        const ballRect = ball.element.getBoundingClientRect();

        return (
            ballRect.left >= goalRect.left &&
            ballRect.right <= goalRect.right &&
            ballRect.top >= goalRect.top &&
            ballRect.bottom <= goalRect.bottom
        );
    }

    function resetBall(ball) {
        ball.x = carX - 50;
        ball.y = carY + 50;
        ball.vx = 0;
        ball.vy = 0;
        ball.rotationSpeed = 0;
    }

    function updateBallPosition(ball) {
        ball.x += ball.vx;
        ball.y += ball.vy;

        // Apply friction to slow down the ball
        const friction = 0.95; // Adjust the friction coefficient as needed
        ball.vx *= friction;
        ball.vy *= friction;

        // Apply friction to slow down the rotation
        const rotationFriction = 0.90; // Adjust the rotation friction coefficient as needed
        ball.rotationSpeed *= rotationFriction;

        // Bounce off the walls
        if (ball.x < 0 || ball.x > window.innerWidth - 20) { // Assuming ball width is 20px
            ball.vx *= -1;
        }
        if (ball.y < 0 || ball.y > window.innerHeight - 20) { // Assuming ball height is 20px
            ball.vy *= -1;
        }

        // Rotate the ball
        ball.rotation += ball.rotationSpeed;
        ball.element.style.transform = `translate(${ball.x}px, ${ball.y}px) rotate(${ball.rotation}deg)`;

        // Check if the ball goes into the goal
        if (checkGoal(ball)) {
            goals += 1;
            score.textContent = `Goals: ${goals}`;

    function animate() {
        
        now = window.performance.now();
        elapsed = now - then;

        requestAnimationFrame(animate);

        if (elapsed < fpsInterval) return;

        then = now - (elapsed % fpsInterval);

        // Mobile swipe-to-close for modal
        (function() {
            const modal = $('modal');
            let startY = 0;
            let currentY = 0;
            let isDragging = false;

            modal.addEventListener('touchstart', function(e) {
                const content = modal.querySelector('.content');
                if (!content) return;

                const touch = e.touches[0];
                const rect = content.getBoundingClientRect();
                if (touch.clientY - rect.top > 60) return;

                startY = touch.clientY;
                isDragging = true;
                content.style.transition = 'none';
            });

            modal.addEventListener('touchmove', function(e) {
                if (!isDragging) return;
                e.preventDefault();

                const content = modal.querySelector('.content');
                if (!content) return;

                currentY = e.touches[0].clientY;
                const deltaY = currentY - startY;

                if (deltaY > 0) {
                    content.style.transform = `translateY(${deltaY}px)`;
                }
            }, { passive: false });

            modal.addEventListener('touchend', function() {
                if (!isDragging) return;
                isDragging = false;

                const content = modal.querySelector('.content');
                if (!content) return;

                const deltaY = currentY - startY;

                if (deltaY > 100) {
                    // Animate from current position to off-screen
                    content.style.transition = 'transform 200ms ease-out, opacity 200ms ease-out';
                    content.style.transform = 'translateY(100%)';
                    modal.style.transition = 'opacity 200ms ease-out';
                    modal.style.opacity = '0';
                    setTimeout(() => {
                        modal.innerHTML = '';
                        modal.style.transition = '';
                        modal.style.opacity = '';
                    }, 200);
                } else {
                    // Snap back with animation
                    content.style.transition = 'transform 150ms ease';
                    content.style.transform = '';
                    setTimeout(() => {
                        content.style.transition = '';
                    }, 150);
                }

                startY = 0;
                currentY = 0;
            });
        })();

        // ImageSetGallery Viewer
        let isgImages = [], isgIndex = 0;

        function openImageViewer(el) {
            const data = el.closest('.image-set-gallery').nextElementSibling;
            if (data && data.classList.contains('isg-data')) {
                try { isgImages = JSON.parse(data.textContent); } catch(e) { isgImages = []; }
            }
            isgIndex = Array.from(el.parentNode.children).filter(c => c.classList.contains('isg-item')).indexOf(el);
            showIsgImage();
        }

        function showIsgImage() {
            const img = isgImages[isgIndex];
            if (!img) return;
            const hasNav = isgImages.length > 1;
            document.body.insertAdjacentHTML('beforeend', `
                <div class="isg-viewer" onclick="if(event.target===this)closeIsgViewer()">
                    <button class="isg-viewer-close" onclick="closeIsgViewer()"><svg xmlns="http://www.w3.org/2000/svg" viewBox="0 0 24 24" width="20" height="20" fill="none" stroke="currentColor" stroke-width="2.5" stroke-linecap="round"><path d="M18 6L6 18M6 6l12 12"/></svg></button>
                    <img src="${img.url}" alt="${img.label}">
                    <div class="isg-viewer-label">${img.label} (${isgIndex + 1}/${isgImages.length})</div>
                    ${hasNav ? `<div class="isg-viewer-nav">
                        <button onclick="isgPrev()"><svg xmlns="http://www.w3.org/2000/svg" viewBox="0 0 24 24" width="20" height="20" fill="none" stroke="currentColor" stroke-width="2.5" stroke-linecap="round" stroke-linejoin="round"><path d="M15 18l-6-6 6-6"/></svg></button>
                        <button onclick="isgNext()"><svg xmlns="http://www.w3.org/2000/svg" viewBox="0 0 24 24" width="20" height="20" fill="none" stroke="currentColor" stroke-width="2.5" stroke-linecap="round" stroke-linejoin="round"><path d="M9 18l6-6-6-6"/></svg></button>
                    </div>` : ''}
                </div>
            `);
            document.body.style.overflow = 'hidden';
        }

        function closeIsgViewer() {
            document.querySelector('.isg-viewer')?.remove();
            document.body.style.overflow = '';
        }

        function isgPrev() {
            isgIndex = (isgIndex - 1 + isgImages.length) % isgImages.length;
            document.querySelector('.isg-viewer')?.remove();
            showIsgImage();
        }

        function isgNext() {
            isgIndex = (isgIndex + 1) % isgImages.length;
            document.querySelector('.isg-viewer')?.remove();
            showIsgImage();
        }

        document.addEventListener('keydown', e => {
            if (!document.querySelector('.isg-viewer')) return;
            if (e.key === 'ArrowLeft') isgPrev();
            if (e.key === 'ArrowRight') isgNext();
            if (e.key === 'Escape') closeIsgViewer();
        });
