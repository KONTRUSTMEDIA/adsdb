        // scroll kilidi (sayfa zıplamasın)
        document.body.style.setProperty('--scroll-lock-top', `-${__scrollY}px`);
        document.body.classList.add('menu-open');

        icon.classList.remove('fa-bars');
        icon.classList.add('fa-times');
    }

    function closeMenu() {
        const wrapper = document.getElementById('nav-wrapper');
        const icon = document.getElementById('menu-icon');

        wrapper.classList.remove('menu-active');
        document.body.classList.remove('menu-open');
        document.body.style.removeProperty('--scroll-lock-top');

        window.scrollTo(0, __scrollY);

        icon.classList.remove('fa-times');
        icon.classList.add('fa-bars');
    }

    function toggleMenu() {
        const wrapper = document.getElementById('nav-wrapper');
        wrapper.classList.contains('menu-active') ? closeMenu() : openMenu();
    }

    // Menü açıkken herhangi bir link/butona tıklanınca kapat
    document.addEventListener('click', (e) => {
        const wrapper = document.getElementById('nav-wrapper');
        if (!wrapper.classList.contains('menu-active')) return;

        const target = e.target.closest('.nav-link, .btn');
        if (target) closeMenu();
    });

    // ESC ile kapat (opsiyonel)
    document.addEventListener('keydown', (e) => {
        if (e.key === 'Escape') {
            const wrapper = document.getElementById('nav-wrapper');
            if (wrapper.classList.contains('menu-active')) closeMenu();
        }
    });
</script>
