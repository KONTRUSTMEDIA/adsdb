(function() {

    'use strict';

    

    function makeTablesResponsive() {

        // Find all tables on the page

        const tables = document.querySelectorAll('table');

        

        tables.forEach(function(table) {

            // Skip if already processed

            if (table.getAttribute('data-responsive-done')) {

                return;

            }

            

            // Mark as processed

            table.setAttribute('data-responsive-done', 'true');

            

            // Get headers

            const headerRow = table.querySelector('thead tr');

            if (!headerRow) {

                return; // No header row found

            }

            

            const headers = Array.from(headerRow.querySelectorAll('th')).map(function(th) {

                return th.textContent.trim();

            });

            

            // Add data-label to all td elements

            const bodyRows = table.querySelectorAll('tbody tr');

            bodyRows.forEach(function(row) {

                const cells = row.querySelectorAll('td');

                cells.forEach(function(cell, index) {

                    if (headers[index]) {

                        cell.setAttribute('data-label', headers[index]);

                    }

                });

            });

        });

    }

    

    // Run immediately if DOM is ready

    if (document.readyState === 'loading') {

        document.addEventListener('DOMContentLoaded', makeTablesResponsive);

    } else {

        makeTablesResponsive();

    }

    

    // Also run after a short delay to catch any dynamically loaded content

    setTimeout(makeTablesResponsive, 1000);

    setTimeout(makeTablesResponsive, 3000);

})();

(function() {

    // Wait for DOM to be ready

    if (document.readyState === 'loading') {

        document.addEventListener('DOMContentLoaded', initSchemaGenerator);

    } else {

        initSchemaGenerator();

    }



    function initSchemaGenerator() {

        // Find all review cards on the page

        const reviewCards = document.querySelectorAll('.bttr-review-card[data-product-name]');

        

        reviewCards.forEach(card => {

            generateSchema(card);

        });

    }



    function generateSchema(card) {

        // Extract data from card attributes

        const productName = card.getAttribute('data-product-name');

        const brand = card.getAttribute('data-brand');

        const currency = card.getAttribute('data-currency') || 'AUD';

        const description = card.getAttribute('data-product-description');

        

        // Extract price, buy URL, and store name from the visible buy section

        let price = '';

        let buyUrl = '';

        let storeName = '';

        

        const priceElement = card.querySelector('.bttr-price');

        if (priceElement) {

            // Extract just the numbers from the price display (e.g., "$499 AUD" -> "499")

            const priceText = priceElement.textContent.trim();

            const priceMatch = priceText.match(/[\d,.]+/);

            if (priceMatch) {

                price = priceMatch[0].replace(/,/g, '');

            }

        }

        

        const buyButton = card.querySelector('.bttr-buy-button');

        if (buyButton) {

            buyUrl = buyButton.getAttribute('href');

            // Extract store name from button text (e.g., "Buy Now at JB Hi-Fi" -> "JB Hi-Fi")

            const buttonText = buyButton.textContent.trim();

            const storeMatch = buttonText.match(/at (.+)$/i);

            if (storeMatch) {

                storeName = storeMatch[1];

            }

        }

        

        // Auto-pull featured image from Ghost article schema

        let featuredImage = '';

        let authorName = '';

        const articleSchema = document.querySelector('script[type="application/ld+json"]');

        if (articleSchema) {

            try {

                const schemaData = JSON.parse(articleSchema.textContent);

                // Get featured image

                if (schemaData.image && schemaData.image.url) {

                    featuredImage = schemaData.image.url;

                } else if (schemaData.image) {

                    // Sometimes image is just a URL string

                    featuredImage = schemaData.image;

                }

                // Get author name

                if (schemaData.author && schemaData.author.name) {

                    authorName = schemaData.author.name;

                }

            } catch (e) {

                console.log('Could not parse article schema');

            }

        }

        

        // Fallback: try to get from meta tag

        if (!featuredImage) {

            const metaImage = document.querySelector('meta[property="og:image"]');

            if (metaImage) {

                featuredImage = metaImage.getAttribute('content');

            }

        }

        

        // Get the verdict text from the card

        const verdictElement = card.querySelector('.bttr-verdict p');

        const verdict = verdictElement ? verdictElement.textContent.trim() : '';

        

        // Extract pros and cons from the lists

        const prosItems = [];

        const prosElements = card.querySelectorAll('.bttr-pros ul li');

        prosElements.forEach(li => {

            const text = li.textContent.trim();

            if (text) prosItems.push(text);

        });

        

        const consItems = [];

        const consElements = card.querySelectorAll('.bttr-cons ul li');

        consElements.forEach(li => {

            const text = li.textContent.trim();

            if (text) consItems.push(text);

        });

        

        // Build the schema markup (optimized - Ghost already provides author, publisher, date)

        const schema = {

            "@context": "https://schema.org",

            "@type": "Review",

            "itemReviewed": {

                "@type": "Product",

                "name": productName,

                "brand": brand,

                "description": description,

                "image": featuredImage,

                "offers": {

                    "@type": "Offer",

                    "price": price,

                    "priceCurrency": currency,

                    "url": buyUrl

                }

            },

            "description": verdict

        };

        

        // Add author if available from article schema

        if (authorName) {

            schema.author = {

                "@type": "Person",

                "name": authorName

            };

        }

        

        // Add positiveNotes if pros exist

        if (prosItems.length > 0) {

            schema.itemReviewed.positiveNotes = {

                "@type": "ItemList",

                "itemListElement": prosItems.map((pro, index) => ({

                    "@type": "ListItem",

                    "position": index + 1,

                    "name": pro

                }))

            };

        }

        

        // Add negativeNotes if cons exist

        if (consItems.length > 0) {

            schema.itemReviewed.negativeNotes = {

                "@type": "ItemList",

                "itemListElement": consItems.map((con, index) => ({

                    "@type": "ListItem",

                    "position": index + 1,

                    "name": con

                }))

            };

        }

        

        // Create script tag and inject schema

        const schemaScript = document.createElement('script');

        schemaScript.type = 'application/ld+json';

        schemaScript.textContent = JSON.stringify(schema);

        

        // Insert schema into the card

        card.appendChild(schemaScript);

    }

})();

(function() {
    const SUPABASE_URL = 'https://monpljavxkviojygvcyb.supabase.co';
    const SUPABASE_KEY = 'sb_publishable__oa-f3iUDgqS7VBqmoNr9g_zCFEEA0t';

    let _supabase;
    let ALL_DEALS = [];
    let ALL_CATEGORIES = [];
    let currentPage = 0;
    let dealsPerPage = 12;

    const formatPrice = (value) => {
        if (value === null || value === undefined || value === '') return null;
        const cleanValue = typeof value === 'string' ? value.replace(/[^0-9.]/g, '') : value;
        const num = parseFloat(cleanValue);
        if (isNaN(num)) return null;
        return new Intl.NumberFormat('en-AU', {
            style: 'currency',
            currency: 'AUD',
            minimumFractionDigits: 0,
            maximumFractionDigits: 0
        }).format(num);
    };

    async function fetchAllDeals() {
        let allData = [];
        let from = 0;
        const step = 1000;
        let hasMore = true;

        while (hasMore) {
            const { data, error } = await _supabase
                .from('deals')
                .select(`
                    *,
                    merchants:merchant_id!inner(id, name, slug, logo_url, Active),
                    categories:category_id(id, name, slug)
                `)
                .eq('is_active', true)
                .eq('merchants.Active', true)
                .range(from, from + step - 1);

            if (error) {
                console.error("Fetch Error:", error);
                break;
            }

            if (data) {
                allData = [...allData, ...data];
                if (data.length < step) {
                    hasMore = false;
                } else {
                    from += step;
                }
            } else {
                hasMore = false;
            }
        }
        return allData;
    }

    async function init() {
        const container = document.getElementById('merchant-deal-wrapper');
        if (!container) return;
        const customLimit = container.getAttribute('data-limit');
        if (customLimit) dealsPerPage = parseInt(customLimit);

        const rowHeight = 78;
        container.style.minHeight = (dealsPerPage * rowHeight) + 'px';
        
        _supabase = supabase.createClient(SUPABASE_URL, SUPABASE_KEY);

        try {
            const catRes = await _supabase.from('categories').select('*').order('name');
            ALL_CATEGORIES = catRes.data || [];

            const rawDeals = await fetchAllDeals();
            ALL_DEALS = sortAndShuffleDeals(rawDeals);

            setupUI(container);
        } catch (err) { 
            console.error('Engine Error:', err); 
        }

    function sortAndShuffleDeals(deals) {
        const NOW = new Date();
        const URGENT_HOURS = 120; 
        const urgentWindow = NOW.getTime() + (URGENT_HOURS * 60 * 60 * 1000);
        const urgentDeals = [];
        const otherDeals = [];
        deals.forEach(deal => {
            if (deal.expires_at) {
                const expiry = new Date(deal.expires_at).getTime();
                if (expiry > NOW.getTime() && expiry <= urgentWindow) {
                    deal.is_urgent = true; 
                    urgentDeals.push(deal);
                } else { otherDeals.push(deal); }
            } else { otherDeals.push(deal); }
        });
        urgentDeals.sort((a, b) => new Date(a.expires_at) - new Date(b.expires_at));
        for (let i = otherDeals.length - 1; i > 0; i--) {
            const j = Math.floor(Math.random() * (i + 1));
            [otherDeals[i], otherDeals[j]] = [otherDeals[j], otherDeals[i]];
        }
        return [...urgentDeals, ...otherDeals];
    }

    function getCategoryFamilyIds(identifier) {
        if (!identifier) return [];
        const idLower = identifier.toLowerCase();
        const root = ALL_CATEGORIES.find(c => c.slug?.toLowerCase() === idLower || c.name?.toLowerCase() === idLower);
        if (!root) return [];
        const familyIds = [root.id];
        function findChildren(parentId) {
            ALL_CATEGORIES.filter(c => c.parent_id === parentId).forEach(child => {
                familyIds.push(child.id);
                findChildren(child.id);
            });
        }
        findChildren(root.id);
        return familyIds;
    }

                            gOpt.value = grandchild.name;
                            gOpt.textContent = `\u00A0\u00A0\u00A0↳ ${grandchild.name}`;
                            catSelect.appendChild(gOpt);
                        });
                    });
                }

                    const pOpt = document.createElement('option');
                    pOpt.value = parent.name;
                    pOpt.textContent = parent.name.toUpperCase();
                    catSelect.appendChild(pOpt);

        if (merSelect) {

    function setupUI(container) {
        const presetCategory = container.getAttribute('data-category') || container.getAttribute('data-category-slug');
        const dataMerchantSlug = container.getAttribute('data-merchant-slug');
        const dataKeyword = container.getAttribute('data-keyword');
        const countDisplay = document.getElementById('results-count');

        populateDropdowns(presetCategory);

        function applyFilters() {
            const searchInput = document.getElementById('deal-search');
            const keyword = (searchInput?.value || dataKeyword || '').toLowerCase();
            const catVal = document.getElementById('filter-category')?.value || 'all';
            const merVal = document.getElementById('filter-merchant')?.value || 'all';
