  function getTime(){
    var currentDate = new Date();
    var monthNames = ["Jan", "Feb", "Mar", "Apr", "May", "Jun", "Jul", "Aug", "Sep", "Oct", "Nov", "Dec"];
    // 获取当前时间的年、月、日、小时、分钟和秒
    var year = currentDate.getUTCFullYear();
    var month = currentDate.getUTCMonth() + 1; // 月份从 0 开始，所以需要加 1
    var day = currentDate.getUTCDate();
    var hours = currentDate.getUTCHours();
    var minutes = currentDate.getUTCMinutes();
    var seconds = currentDate.getUTCSeconds();

    // 格式化日、小时、分钟和秒，确保它们始终是两位数
    if (day < 10) {
      day = "0" + day;
    }
    if (hours < 10) {
      hours = "0" + hours;
    }
    if (minutes < 10) {
      minutes = "0" + minutes;
    }
    if (seconds < 10) {
      seconds = "0" + seconds;
    }

    // 构建最终的时间字符串，加上 CST 标识
    var cstTime = "#" + day + " " + month + " " + day + " " + hours + ":" + minutes + ":" + seconds + " CST " + year;

  }
