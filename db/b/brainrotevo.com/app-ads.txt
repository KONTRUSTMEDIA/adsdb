    <script>
    // Accessibility Menu Functionality
    (function() {
        const toggle = document.getElementById('accessibilityToggle');
        const menu = document.getElementById('accessibilityMenu');
        const zoomIncrease = document.getElementById('zoomIncrease');
        const zoomDecrease = document.getElementById('zoomDecrease');
        const zoomDisplay = document.getElementById('zoomDisplay');
        const highContrastToggle = document.getElementById('highContrastToggle');
        const reduceMotionToggle = document.getElementById('reduceMotionToggle');
        const resetBtn = document.getElementById('resetAccessibility');

        // 0 = normal (7 cols), 1 = large (6 cols), 2 = x-large (5 cols)
        let zoomLevel = parseInt(localStorage.getItem('accessZoomLevel')) || 0;
        let highContrast = localStorage.getItem('accessHighContrast') === 'true';
        let reduceMotion = localStorage.getItem('accessReduceMotion') === 'true';

        const zoomLabels = ['Normal', 'Large', 'X-Large'];

        // Apply saved settings on load
        function applySettings() {
            // Zoom level
            document.body.classList.remove('zoom-level-1', 'zoom-level-2');
            if (zoomLevel === 1) {
                document.body.classList.add('zoom-level-1');
            } else if (zoomLevel === 2) {
                document.body.classList.add('zoom-level-2');
            }
            zoomDisplay.textContent = zoomLabels[zoomLevel];

            // High contrast
            if (highContrast) {
                document.body.classList.add('high-contrast');
                highContrastToggle.classList.add('active');
                highContrastToggle.setAttribute('aria-checked', 'true');
            } else {
                document.body.classList.remove('high-contrast');
                highContrastToggle.classList.remove('active');
                highContrastToggle.setAttribute('aria-checked', 'false');
            }

            // Reduce motion
            if (reduceMotion) {
                document.body.classList.add('reduce-motion');
                reduceMotionToggle.classList.add('active');
                reduceMotionToggle.setAttribute('aria-checked', 'true');
            } else {
                document.body.classList.remove('reduce-motion');
                reduceMotionToggle.classList.remove('active');
                reduceMotionToggle.setAttribute('aria-checked', 'false');
            }
        }

        applySettings();

        // Toggle menu
        toggle.addEventListener('click', () => {
            const isOpen = menu.classList.toggle('open');
            toggle.setAttribute('aria-expanded', isOpen);
            if (window.AudioManager) window.AudioManager.play('click');
        });

        // Close menu when clicking outside
        document.addEventListener('click', (e) => {
            if (!menu.contains(e.target) && !toggle.contains(e.target)) {
                menu.classList.remove('open');
                toggle.setAttribute('aria-expanded', 'false');
            }
        });

        // Zoom controls
        zoomIncrease.addEventListener('click', () => {
            if (zoomLevel < 2) {
                zoomLevel++;
                localStorage.setItem('accessZoomLevel', zoomLevel);
                applySettings();
                if (window.AudioManager) window.AudioManager.play('click');
            }
        });

        zoomDecrease.addEventListener('click', () => {
            if (zoomLevel > 0) {
                zoomLevel--;
                localStorage.setItem('accessZoomLevel', zoomLevel);
                applySettings();
                if (window.AudioManager) window.AudioManager.play('click');
            }
        });

        // High contrast toggle
        highContrastToggle.addEventListener('click', () => {
            highContrast = !highContrast;
            localStorage.setItem('accessHighContrast', highContrast);
            applySettings();
            if (window.AudioManager) window.AudioManager.play('click');
        });

        // Reduce motion toggle
        reduceMotionToggle.addEventListener('click', () => {
            reduceMotion = !reduceMotion;
            localStorage.setItem('accessReduceMotion', reduceMotion);
            applySettings();
            if (window.AudioManager) window.AudioManager.play('click');
        });

        // Reset all
        resetBtn.addEventListener('click', () => {
            zoomLevel = 0;
            highContrast = false;
            reduceMotion = false;
            localStorage.removeItem('accessZoomLevel');
            localStorage.removeItem('accessHighContrast');
            localStorage.removeItem('accessReduceMotion');
            applySettings();
            if (window.AudioManager) window.AudioManager.play('click');
        });

        // Keyboard support for toggles
        [highContrastToggle, reduceMotionToggle].forEach(el => {
            el.addEventListener('keydown', (e) => {
                if (e.key === 'Enter' || e.key === ' ') {
                    e.preventDefault();
                    el.click();
                }
            });
        });

        toggle.addEventListener('keydown', (e) => {
            if (e.key === 'Enter' || e.key === ' ') {
                e.preventDefault();
                toggle.click();
            }
        });
    })();
    </script>

        function formatNumber(num) {
            if (num === null || num === undefined) return 'N/A';
            const value = typeof num === 'string' ? parseFloat(num) : num;
            if (!Number.isFinite(value)) return '0';
            if (value < 1000) return Math.floor(value).toString();
            
            const abbreviations = [
                { value: 1e39, suffix: 'Dd' }, { value: 1e36, suffix: 'Ud' }, { value: 1e33, suffix: 'Dc' }, { value: 1e30, suffix: 'No' },
                { value: 1e27, suffix: 'Oc' }, { value: 1e24, suffix: 'Sp' }, { value: 1e21, suffix: 'Sx' },
                { value: 1e18, suffix: 'Qi' }, { value: 1e15, suffix: 'Qa' }, { value: 1e12, suffix: 'T' },
                { value: 1e9, suffix: 'B' }, { value: 1e6, suffix: 'M' }, { value: 1e3, suffix: 'K' }
            ];
            
            for (let i = 0; i < abbreviations.length; i++) {
                if (value >= abbreviations[i].value) {
                    const formatted = (value / abbreviations[i].value).toFixed(2);
                    return formatted.replace(/\.?0+$/, '') + abbreviations[i].suffix;
                }
            }
            return value.toString();
        }

        function formatNumberWithCommas(num) {
            if (num === null || num === undefined) return 'N/A';
            if (typeof num === 'string') {
                return num.replace(/\B(?=(\d{3})+(?!\d))/g, ',');
            }
            if (!Number.isFinite(num)) return '0';
            
            let numStr;
            if (num >= 1e21) {
                numStr = num.toExponential().split('e');
                const base = parseFloat(numStr[0]);
                const exp = parseInt(numStr[1]);
                const baseStr = base.toFixed(0);
                numStr = baseStr + '0'.repeat(exp - baseStr.length + 1);
            } else {
                numStr = Math.floor(num).toString();
            }
            return numStr.replace(/\B(?=(\d{3})+(?!\d))/g, ',');
        }

        function showTooltip(element, content) {
            // Check if tooltips are temporarily disabled (e.g., after popup closes)
            if (window.tooltipsDisabled) {
                return;
            }

            // Check if any modal is currently open
            const openModals = document.querySelectorAll('[id$="Modal"][style*="display: flex"], [id$="Modal"][style*="display: block"]');

            if (openModals.length > 0) {
                // A modal is open - check if the element is inside the modal
                let isInsideModal = false;
                openModals.forEach(modal => {
                    if (modal.contains(element)) {
                        isInsideModal = true;
                    }
                });

                // If element is outside the modal, don't show tooltip
                if (!isInsideModal) {
                    return;
                }
            }

            globalTooltip.innerHTML = content;
            globalTooltip.classList.add('visible');

            const rect = element.getBoundingClientRect();
            const tooltipRect = globalTooltip.getBoundingClientRect();
            
            // Calculate initial position (centered below element)
            let left = rect.left + rect.width / 2 - tooltipRect.width / 2;
            let top = rect.bottom + 10;
            
            // Padding from edges
            const padding = 10;
            const arrowHeight = 14; // Height of the arrow (::after pseudo-element)
            
            // Check horizontal bounds
            if (left < padding) {
                left = padding;
            }
            if (left + tooltipRect.width > window.innerWidth - padding) {
                left = window.innerWidth - tooltipRect.width - padding;
            }
            
            // Check vertical bounds - if tooltip goes below screen, position it above the element
            if (top + tooltipRect.height > window.innerHeight - padding) {
                // Position above the element instead
                top = rect.top - tooltipRect.height - arrowHeight - 5;
                
                // If still out of bounds at top, just position it as high as possible
                if (top < padding) {
                    top = padding;
                }
                
                // Flip arrow to point downward when tooltip is above
                globalTooltip.classList.add('tooltip-above');
            } else {
                globalTooltip.classList.remove('tooltip-above');
            }
            
            // Final check - ensure tooltip doesn't go above viewport
            if (top < padding) {
                top = padding;
            }
            
            globalTooltip.style.left = left + 'px';
            globalTooltip.style.top = top + 'px';
        }

        function hideTooltip() {
            globalTooltip.classList.remove('visible', 'tooltip-above');
        }

        // Force-close tooltip safety mechanism
        // This ensures tooltips don't get stuck when mouse moves quickly
        let tooltipSafetyTimer = null;
        let lastHoveredElement = null;

        function forceCloseTooltip() {
            if (globalTooltip.classList.contains('visible')) {
                globalTooltip.classList.remove('visible', 'tooltip-above');
            }
        }

        // Track when mouse enters/leaves tooltip-triggering elements
        document.addEventListener('mouseover', (e) => {
            // Clear any existing safety timer
            if (tooltipSafetyTimer) {
                clearTimeout(tooltipSafetyTimer);
                tooltipSafetyTimer = null;
            }
            lastHoveredElement = e.target;
        });

        document.addEventListener('mouseout', (e) => {
            // Start a safety timer to close tooltip if not hovering anything
            if (tooltipSafetyTimer) {
                clearTimeout(tooltipSafetyTimer);
            }
            tooltipSafetyTimer = setTimeout(() => {
                // Check if tooltip is still visible but mouse isn't over any element that should show it
                if (globalTooltip.classList.contains('visible')) {
                    forceCloseTooltip();
                }
            }, 300); // 300ms delay before force-closing
        });

        // Also close tooltip when clicking anywhere
        document.addEventListener('click', () => {
            if (globalTooltip.classList.contains('visible')) {
                forceCloseTooltip();
            }
        });

        // Close tooltip when scrolling
        document.addEventListener('scroll', () => {
            if (globalTooltip.classList.contains('visible')) {
                forceCloseTooltip();
            }
        }, true);

        // Hash-based routing handler
        function handleHashChange() {
            const hash = window.location.hash.replace('#/', '').toLowerCase() || 'wfl';
            const hashToTab = {
                'wfl': 'wfl',
                'brainrots': 'wikipedia',
                'wikipedia': 'wikipedia',
                'pets': 'pets',
                'relics': 'relics',
                'items': 'items',
                'trinkets': 'trinkets',
                'home': 'home'
            };

            const tabName = hashToTab[hash];

            if (tabName === 'home') {
                // Handle home - close main container
                if (mainContainer) mainContainer.classList.remove('active');
                document.querySelectorAll('.tab-content').forEach(tab => {
                    tab.classList.remove('active');
                });
                resetAllButtonsToBlue();
                const homeImg = document.getElementById('homeBgHome');
                const homeBtn = document.getElementById('homeBtnHome');
                if (homeImg) homeImg.src = addVersion('game-assets/ui/home-yellow.png');
                if (homeBtn) homeBtn.classList.add('active');
                if (homeLanding) homeLanding.classList.remove('hidden');
                currentTab = null;
            } else if (hash === 'tradehub') {
                // Handle Trade Hub — switch to WFL if needed, then open Trade Hub
                if (currentTab !== 'wfl') {
                    isNavigatingFromHash = true;
                    if (mainContainer) mainContainer.classList.add('active');
                    if (homeLanding) homeLanding.classList.add('hidden');
                    switchTab('wfl', false);
                    isNavigatingFromHash = false;
                }
                // Force hash back and open Trade Hub after everything settles
                setTimeout(function() {
                    if (window.location.hash !== '#/tradehub') {
                        history.replaceState(null, '', '#/tradehub');
                    }
                    if (window.TradeHub && !TradeHub.isActive) TradeHub.toggleTradeHub();
                }, 200);
            } else if (tabName && tabConfig[tabName]) {
                isNavigatingFromHash = true;
                if (mainContainer) mainContainer.classList.add('active');
                if (homeLanding) homeLanding.classList.add('hidden');
                switchTab(tabName, false).then(function() {
                    isNavigatingFromHash = false;
                    // Close Trade Hub if navigating away from #/tradehub
                    if (window.TradeHub && TradeHub.isActive) {
                        TradeHub.toggleTradeHub();
                    }
                });
            }
        }

        // WFL Popup functionality
        const wflCheckBtn = document.querySelector('.wfl-check-btn');

        if (wflHelpBtn) {
            wflHelpBtn.addEventListener('click', function() {
                wflPopup.classList.add('active');
            });
        }

        // W/F/L button shows the result popup
        if (wflCheckBtn) {
            wflCheckBtn.addEventListener('click', function() {
                // Get trade values from the WFL system
                const statusText = document.querySelector('.wfl-trade-status');
                if (!statusText) {
                    wflPopup.classList.add('active');
                    return;
                }

                const text = statusText.textContent;
                wflResultContainer.classList.remove('win', 'lose', 'fair', 'bigwin', 'biglose');

                if (text.includes('Great Trade')) {
                    wflResultContainer.classList.add('bigwin');
                    wflResultIcon.textContent = '🏆';
                    wflResultTitle.textContent = 'BIG WIN!';
                    wflResultSubtitle.textContent = text;
                } else if (text.includes('Slightly Good')) {
                    wflResultContainer.classList.add('win');
                    wflResultIcon.textContent = '🎉';
                    wflResultTitle.textContent = 'You Win!';
                    wflResultSubtitle.textContent = text;
                } else if (text.includes('Fair Trade')) {
                    wflResultContainer.classList.add('fair');
                    wflResultIcon.textContent = '🤝';
                    wflResultTitle.textContent = 'Fair Trade!';
                    wflResultSubtitle.textContent = 'This trade is balanced';
                } else if (text.includes('Slightly Bad')) {
                    wflResultContainer.classList.add('lose');
                    wflResultIcon.textContent = '😢';
                    wflResultTitle.textContent = 'You Lose!';
                    wflResultSubtitle.textContent = text;
                } else if (text.includes('Bad Trade')) {
                    wflResultContainer.classList.add('biglose');
                    wflResultIcon.textContent = '💀';
                    wflResultTitle.textContent = 'BIG LOSE!';
                    wflResultSubtitle.textContent = text;
                } else {
                    // No items added, button should be disabled
                    return;
                }

                wflResultPopup.classList.add('active');
            });
        }

        // Close result popup
        if (wflResultClose) {
            wflResultClose.addEventListener('click', function() {
                wflResultPopup.classList.remove('active');
            });
        }

        if (wflResultPopup) {
            wflResultPopup.addEventListener('click', function(e) {
                if (e.target === wflResultPopup) {
                    wflResultPopup.classList.remove('active');
                }
            });
        }

        if (wflPopupClose) {
            wflPopupClose.addEventListener('click', function() {
                wflPopup.classList.remove('active');
            });
        }

        // Close popup when clicking overlay
        if (wflPopup) {
            wflPopup.addEventListener('click', function(e) {
                if (e.target === wflPopup) {
                    wflPopup.classList.remove('active');
                }
            });
        }

        // WFL Vote Popup functionality

    <!-- Height-based scale adjustment - only kicks in if screen is too short -->
    <script>
    (function() {
        const CONTAINER_HEIGHT = 810;
        const NAVBAR_SPACE = 230; // Extra space for navbar and WFL overflow

        // Get what scale the CSS media queries would apply
        function getCSSScale() {
            const w = window.innerWidth;
            if (w <= 1000) return 0.38;
            if (w <= 1150) return 0.44;
            if (w <= 1300) return 0.51;
            if (w <= 1450) return 0.58;
            if (w <= 1600) return 0.65;
            if (w <= 1800) return 0.72;
            if (w < 2000) return 0.765;
            if (w < 2200) return 0.85;
            if (w < 2400) return 0.935;
            if (w < 2600) return 1.02;
            if (w < 2800) return 1.105;
            return 1.19;
        }

        function checkHeight() {
            const cssScale = getCSSScale();
            const neededHeight = (CONTAINER_HEIGHT * cssScale) + NAVBAR_SPACE;
            const availableHeight = window.innerHeight;

            if (neededHeight > availableHeight) {
                // Screen too short - reduce scale to fit
                const maxScale = (availableHeight - NAVBAR_SPACE) / CONTAINER_HEIGHT;
                document.documentElement.style.setProperty('--scale', Math.max(0.3, maxScale));
            } else {
                // Height is fine - remove override, let CSS handle it
                document.documentElement.style.removeProperty('--scale');
            }
        }

        checkHeight();
        window.addEventListener('resize', checkHeight);
    })();
    </script>
