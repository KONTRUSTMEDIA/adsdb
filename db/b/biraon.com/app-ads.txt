<script>
const BASIC = 'basic';
const ADVANCED = 'advanced';
const ENTERPRISE = 'enterprise';

const packageField = document.getElementById('package-field');
const usersField = document.getElementById('users-field');
const billingField = document.getElementById('billing-field');
const schedulingField = document.getElementById('scheduling-field');
const skillsField = document.getElementById('skills-field');
const managementField = document.getElementById('management-field');
const pricingNumber = document.getElementById('pricing-number');

function calcThresholdValue(thresholds, num) {
  let value = num > 0 ? thresholds.initValue : 0;

  for (let info of thresholds.infos) {
    if (num >= info.thresh)
      value = info.value;
  }

  return value;
}

function calcCascThresholdValue(info, num) {
	if (num < info.thresh)
  	return 0;
	
  return ((~~((num - info.thresh) / info.every)) * info.add) + info.value;
}

function getNumeric(field, numPredicate) {
	const num = Number(field.value);
  
  if (isNaN(num))
  	return 0;
    
 	if (numPredicate)
  	return numPredicate(num);
  
  return num;
}

function calcPackageValue(key) {
  switch(key) {
  	case BASIC:
    	return 11;
    case ADVANCED:
    	return 16;
    case ENTERPRISE:
    	return 21;
    default:
    	return 0;
  }
}

const skillsThresholds = {
  initValue: 99,
  infos: [
    { thresh: 100,  value: 249 },
    { thresh: 500,  value: 499 },
    { thresh: 2000, value: 999 }
  ]
}

function calcSkillsValue(num) {
  return calcThresholdValue(skillsThresholds, num);
}

const cascBillingThreshold = { 
	thresh: 3000000, 
  value: 3995, 
  every: 1000000, 
  add: 1000 
};

const billingThresholds = {
  initValue: 49,
  infos: [
    { thresh: 50000,   value: 99 },
    { thresh: 100000,  value: 199 },
    { thresh: 200000,  value: 349 },
    { thresh: 350000,  value: 499 },
    { thresh: 500000,  value: 649 },
    { thresh: 650000,  value: 799 },
    { thresh: 800000,  value: 999 },
    { thresh: 1000000, value: 1249 },
    { thresh: 1250000, value: 1495 },
    { thresh: 1500000, value: 1995 },
    { thresh: 2000000, value: 2495 },
    { thresh: 2500000, value: 2995 },
  ]
}

function calcBillingValue(num) {
  return num >= cascBillingThreshold.thresh ?
  calcCascThresholdValue(cascBillingThreshold, num) :
  calcThresholdValue(billingThresholds, num);
}

function getPackageValue() {
	return calcPackageValue(packageField.value);
}

function getUsersValue() {
	return getNumeric(usersField);
}

function getBillingValue() {
	return getNumeric(billingField, calcBillingValue);
}

function getSchedulingValue() {
	return getNumeric(schedulingField);
}

function getSkillsValue() {
	return getNumeric(skillsField, calcSkillsValue);
}

function getManagementValue() {
	return getNumeric(managementField);
}

function strip(number) {
  return (parseFloat(number).toFixed(2));
}

function fieldUpdate(e) {
	pricingNumber.innerHTML = strip (
    getPackageValue() * getUsersValue() +
    getBillingValue() + 
    11 * getSchedulingValue() + 
    getSkillsValue() +
    37 * getManagementValue()
  ); 
}

fieldUpdate(0);

packageField.addEventListener('input', fieldUpdate);
usersField.addEventListener('input', fieldUpdate);
billingField.addEventListener('input', fieldUpdate);
schedulingField.addEventListener('input', fieldUpdate);
skillsField.addEventListener('input', fieldUpdate);
managementField.addEventListener('input', fieldUpdate);
</script>
