        function get_user_account_details(email)
        {
            $.ajax({

                },
                url: "https://www.businessworld.in/get-user-details",

                    if(result.status == 0) 
                    {
                        $('#user_first_name').val('').attr('readonly', false);
                        $('#user_last_name').val('').attr('readonly', false);
                        $('#user_whatsapp_number').val('').attr('readonly', false);
                        $('#subscribe-btn-div').removeClass('d-none');

                    else if(result.status == 1)
                    {   
                        $('#user_first_name').val(result.userdetails.first_name).attr('readonly', true);;
                        $('#user_last_name').val(result.userdetails.last_name).attr('readonly', true);

                        {
                            $('#user_whatsapp_number').val('').attr('readonly', false);
                        }

                    }
                    else
                    {
                        $('#user_email').val('').attr('placeholder', result.message);
                        $('#subscribe-btn-div').addClass('d-none');
                    }
                }
            })
        }

        function showTime(){
            var date = new Date();
            var h = date.getHours(); // 0 - 23
            var m = date.getMinutes(); // 0 - 59
            var s = date.getSeconds(); // 0 - 59
            var session = "AM";
            
            if(h == 0){
                h = 12;
            }

            if(h == 12){
                h = 12;
                session = "PM";
            }
            
            if(h > 12){
                h = h - 12;
                session = "PM";
            }
            
            h = (h < 10) ? "0" + h : h;
            m = (m < 10) ? "0" + m : m;
            s = (s < 10) ? "0" + s : s;
            
            var time = h + ":" + m + ":" + s + " " + session;
            document.getElementById("MyClockDisplay").innerText = time;
            document.getElementById("MyClockDisplay").textContent = time;            
            setTimeout(showTime, 1000);            
        }
        showTime();
    </script>
