document.addEventListener('DOMContentLoaded', function() {
    // Список доменов, для которых НЕ нужно добавлять nofollow
    const nofollowExceptions = [
        'pincocasinos.com'
    ];

    // Функция проверки: является ли домен исключением
    function isException(host) {
        return nofollowExceptions.some(exception => 
            host === exception || host === 'www.' + exception || host.endsWith('.' + exception)
        );
    }

    // Функция добавления nofollow к внешним ссылкам
    function addNofollowToExternalLinks(container) {
        const links = container.querySelectorAll('a[href^="http"]');
        const siteHost = window.location.hostname;

        links.forEach(link => {
            try {
                const url = new URL(link.href);
                const linkHost = url.hostname;

                // Пропускаем внутренние ссылки и исключения
                if (linkHost === siteHost || isException(linkHost)) {
                    return;
                }

                // Добавляем nofollow, если его ещё нет
                if (!link.rel.split(' ').includes('nofollow')) {
                    link.rel = (link.rel ? link.rel + ' ' : '') + 'nofollow';
                }
            } catch (e) {
                // Игнорируем некорректные URL
                console.warn('Invalid URL in link:', link.href);
            }
        });
    }

    // MutationObserver для отслеживания динамически добавленных ссылок
    const observer = new MutationObserver(function(mutations) {
        mutations.forEach(function(mutation) {
            mutation.addedNodes.forEach(function(node) {
                if (node.nodeType === 1) { // Element node
                    if (node.matches && node.matches('a[href^="http"]')) {
                        try {
                            const url = new URL(node.href);
                            const linkHost = url.hostname;
                            const siteHost = window.location.hostname;

                            if (linkHost !== siteHost && !isException(linkHost)) {
                                if (!node.rel.split(' ').includes('nofollow')) {
                                    node.rel = (node.rel ? node.rel + ' ' : '') + 'nofollow';
                                }
                            }
                        } catch (e) {
                            // skip invalid URLs
                        }
                    }
                    if (node.querySelectorAll) {
                        addNofollowToExternalLinks(node);
                    }
                }
            });
        });
    });

    // Запуск наблюдателя
    observer.observe(document.body, {
        childList: true,
        subtree: true
    });

    // Обработка уже существующих ссылок (на случай, если комментарии загрузились быстро)
    addNofollowToExternalLinks(document.body);
});
