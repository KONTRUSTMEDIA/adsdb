  function encodeToSJISUrl(str) {
    if (!window.Encoding) {
      throw new Error("encoding-japanese ライブラリが読み込まれていません。");
    }
    const unicodeArray = Encoding.stringToCode(str);
    const sjisArray = Encoding.convert(unicodeArray, 'SJIS', 'UNICODE');
    const sjisBytes = new Uint8Array(sjisArray);
    return Array.from(sjisBytes)
      .map(b => '%' + b.toString(16).toUpperCase().padStart(2, '0'))
      .join('');
  }
