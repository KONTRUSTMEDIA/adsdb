            function get_cookie(cname) {
                let name = cname + "=";
                let decodedCookie = decodeURIComponent(document.cookie);
                let ca = decodedCookie.split(';');
                for(let i = 0; i <ca.length; i++) {
                    let c = ca[i];
                    while (c.charAt(0) == ' ') {
                    c = c.substring(1);
                    }
                    if (c.indexOf(name) == 0) {
                    return c.substring(name.length, c.length);
                    }
                }
                return "";
            }

            function sort_builds () {
                let sort = document.getElementsByName("sort_builds")[0].value;
                let builds = [...global_build_list];
                let div = document.getElementById("build_cards");

                if (sort == "new") {
                    builds.sort( (a, b) => {
                        // return the sign
                        return new Date (b.dataset.date) - new Date (a.dataset.date);
                    })
                } else if (sort == "popular") {
                    builds.sort( (a, b) => {
                        let views_a = parseInt(a.getElementsByClassName("build_views")[0].innerText.substring(2));
                        let views_b = parseInt(b.getElementsByClassName("build_views")[0].innerText.substring(2));
                        if (views_a < views_b) {
                            return 1;
                        }
                        if (views_a > views_b) {
                            return -1;
                        }
                        return 0;
                    })
                }
                for (let i = 0; i < builds.length; i++) {
                    div.appendChild(builds[i]);
                }
            }
            
            function filter_builds () {
                let filters = [
                    document.getElementsByName("filter_category")[0].value,
                    document.getElementsByName("filter_hero")[0].value,
                    document.getElementsByName("filter_author")[0].value,
                    document.getElementsByName("filter_tags")[0].value
                ];

                let builds = document.getElementsByClassName("build");

                if (filters.join("") == "allallallall") {
                    // No filters, remove .hide and exit
                    console.log("reset");
                    for (let i = 0; i < builds.length; i++) {
                        // Reset filters
                        builds[i].classList.remove("hide");                    
                    }
                } else {
                    filters = filters.filter(n => n != "all");
                    for (let i = 0; i < builds.length; i++) {
                        let tag_elements = [...builds[i].getElementsByClassName("build_tags")].map(n => n.innerText);
                        let build = [
                            builds[i].getElementsByClassName("build_mode")[0].innerText,
                            builds[i].getElementsByClassName("build_author")[0].innerText,
                            builds[i].dataset.hero,
                            ...tag_elements
                        ];
                        console.log(build);
                        let show = false;
                        let break_loop = false;

                        filters.forEach(filter => {
                            if (build.includes(filter) && !break_loop) {
                                show = true;
                            } else {
                                show = false;
                                break_loop = true;
                            }
                        })

                        if (show) {
                            builds[i].classList.remove("hide");
                        } else {
                            builds[i].classList.add("hide");
                        }
                    }
                }
            }
