function calculateDVLAValue() {
  const plate = document.getElementById("plateInput").value.trim().toUpperCase();
  const age = parseInt(document.getElementById("plateAge").value);
  const rarity = document.getElementById("rarityLevel").value;
  const condition = document.getElementById("condition").value;
  const demand = parseInt(document.getElementById("marketDemand").value);

  let baseValue = 100;

  if (plate.length <= 4 || /^[A-Z]{1,2}[0-9]{1,3}$/.test(plate)) {
    baseValue += 500;
  }

  if (!isNaN(age)) {
    baseValue += age * 10;
  }

  switch (rarity) {
    case "common": baseValue *= 1; break;
    case "uncommon": baseValue *= 1.2; break;
    case "rare": baseValue *= 1.5; break;
    case "ultra-rare": baseValue *= 2; break;
  }

  switch (condition) {
    case "poor": baseValue *= 0.8; break;
    case "fair": baseValue *= 1; break;
    case "good": baseValue *= 1.2; break;
    case "mint": baseValue *= 1.5; break;
  }

  baseValue *= (demand / 100);

  const result = `Estimated Value: £${Math.round(baseValue).toLocaleString()}`;
  document.getElementById("valueResult").innerText = result;
}

<script>
// Corrected function to calculate tire diameter
function parseTireSize(tireStr) {
    const match = tireStr.trim().match(/(d{3})/(d{2})/(d{2})/);
    if (!match) return null;
    
    const width = parseFloat(match[1]); // Section Width in mm
    const aspect = parseFloat(match[2]); // Aspect Ratio
    const rim = parseFloat(match[3]); // Rim Diameter in inches

    // Sidewall height in mm: (Section Width * (Aspect Ratio / 100))
    const sidewallMM = (width * (aspect / 100));

    // Rim diameter in mm: Rim Diameter (inches) * 25.4 mm/inch
    const rimMM = rim * 25.4;

    // Total diameter in mm: (2 * Sidewall height) + Rim diameter
    const totalDiameterMM = (sidewallMM * 2) + rimMM;

    // Total diameter in inches
    const diameterInches = totalDiameterMM / 25.4;

    return { 
        width, 
        aspect, 
        rim, 
        sidewall: sidewallMM, 
        diameter: diameterInches 
    };
}

function calculateFitment() {
    const currentWidth = parseFloat(document.getElementById("currentWidth").value);
    const currentOffset = parseFloat(document.getElementById("currentOffset").value);
    const currentTireStr = document.getElementById("currentTire").value;

    const newWidth = parseFloat(document.getElementById("newWidth").value);
    const newOffset = parseFloat(document.getElementById("newOffset").value);
    const newTireStr = document.getElementById("newTire").value;

    const currentTire = parseTireSize(currentTireStr);
    const newTire = parseTireSize(newTireStr);

    const result = document.getElementById("fitmentResult");
    const details = document.getElementById("fitmentDetails");

    if (!currentTire || !newTire || isNaN(currentWidth) || isNaN(currentOffset) || isNaN(newWidth) || isNaN(newOffset)) {
        result.innerText = "Please fill all fields correctly with proper tire format (e.g., 225/45/17).";
        details.innerHTML = "";
        return;
    }

    // --- Corrected Fitment Calculations ---

    // Calculate the difference in how far the wheel/tire pokes out (poke difference)
    // 12.7mm is half an inch, used for wheel width conversion (1 inch difference in width = 0.5 inches added to the outside)
    // The formula for poke difference (in mm) is: (New Width - Current Width) * 12.7 + (Current Offset - New Offset)
    const pokeDiff = (((newWidth - currentWidth) * 12.7) + (currentOffset - newOffset)).toFixed(2);

    // Calculate the difference in tire diameter (in inches)
    const tireDiameterDiff = (newTire.diameter - currentTire.diameter).toFixed(2);

    // Update Result Text
    let resultText = "";
    if (parseFloat(pokeDiff) > 0) {
        resultText = `New setup will poke out ${pokeDiff}mm more than current.`;
    } else if (parseFloat(pokeDiff) < 0) {
        resultText = `New setup will sit ${Math.abs(pokeDiff)}mm further inside than current.`;
    } else {
        resultText = "The new wheel offset and width result in the same outer position as the current setup.";
    }
    
    result.innerText = resultText;

    // Update Details
    details.innerHTML = `
        <strong>Current Tire Diameter:</strong> ${currentTire.diameter.toFixed(2)} inches<br>
        <strong>New Tire Diameter:</strong> ${newTire.diameter.toFixed(2)} inches<br>
        <strong>Height Difference:</strong> ${tireDiameterDiff} inches<br>
        <strong>Track Width Change (per wheel):</strong> ${pokeDiff} mm<br>
        <strong>Note:</strong> Adjustments over 10mm may affect clearance or handling. Consider fender roll or alignment.
    `;
}
</script>
