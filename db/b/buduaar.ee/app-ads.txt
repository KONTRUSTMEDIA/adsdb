    function setResponsiveSize(event, uniqueName) {
        const parent = document.getElementById(uniqueName);
        const iframe = parent.getElementsByTagName('iframe')[0];

        if (!iframe) {
            return;
        }

        let parentWidth  = parent.parentElement.clientWidth;
        let parentHeight = parent.parentElement.clientHeight;

        if (parentWidth === 0 || parentHeight === 0) {
            parentWidth  = window.innerWidth;
            parentHeight = window.innerHeight;
        }

        if (uniqueName.includes('-fullscreen-')) {
            parentWidth  = window.innerWidth;
            parentHeight = window.innerHeight - 58; // 58px is the height of the header
        }

        if (uniqueName.includes('-billboard-')) {
            parentWidth = parentWidth - 40; // 40px is the padding of the parent element
            parentHeight = parentHeight - 40; // 40px is the padding of the parent element
        }

        const elementWidth  = event.size[0];
        const elementHeight = event.size[1];

        const scaleW = parentWidth / elementWidth;
        const scaleH = parentHeight / elementHeight;
        const scale  = Math.min(scaleW, scaleH);

        if (iframe && iframe.style && scale < 1 && scale > 0) {
            iframe.style.transform = 'scale(' + scale + ')';
            iframe.style.transformOrigin = 'left top';

            if (uniqueName.includes('-fullscreen-')) {
                iframe.style.transformOrigin = 'center top';
            }

            const iframeHeight  = elementHeight * scale;
            const iframeWidth   = elementWidth * scale;
            parent.style.width  = iframeWidth + 'px';
            parent.style.height = iframeHeight + 'px';
        }
    }
