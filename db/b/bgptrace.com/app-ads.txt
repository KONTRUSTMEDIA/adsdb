<script type="743c4843411ea635a3d212fd-text/javascript">
  let withdrawData = [];
  let announceData = [];
  let chartInstance = null;

  async function fetchData() {
    const [withdrawResp, announceResp] = await Promise.all([
      fetch('/data/withdraw_summary.json'),
      fetch('/data/announce_summary.json')
    ]);
    withdrawData = await withdrawResp.json();
    announceData = await announceResp.json();
  }

  function getUniqueASNs() {
    const asnSet = new Set();
    withdrawData.forEach(d => asnSet.add(d.primary_as));
    announceData.forEach(d => asnSet.add(d.primary_as));
    return Array.from(asnSet).sort((a, b) => a - b);
  }

  function buildASDropdown() {
  const select = document.getElementById('asSelect');
  select.innerHTML = ''; // Clear existing options

  const asRowCounts = {};

  // Combine announce and withdraw data
  const combined = [...announceData, ...withdrawData];

  // Count how many times each AS appears
  for (const entry of combined) {
    const asn = entry.primary_as;
    if (!asRowCounts[asn]) {
      asRowCounts[asn] = 0;
    }
    asRowCounts[asn]++;
  }

  // Sort AS numbers by their appearance count (row count)
  const sortedASNs = Object.entries(asRowCounts)
    .sort((a, b) => b[1] - a[1]) // sort by row count descending
    .map(entry => entry[0]);     // extract just the AS numbers

  // Build dropdown options
  for (const asn of sortedASNs) {
    const option = document.createElement('option');
    option.value = asn;
    option.textContent = `${asn}`;
    select.appendChild(option);
  }
}



  function getRandomColor(seed) {
    let hash = 0;
    for (let i = 0; i < seed.length; i++) {
      hash = seed.charCodeAt(i) + ((hash << 5) - hash);
    }
    const c = (hash & 0x00FFFFFF).toString(16).toUpperCase();
    return "#" + "00000".substring(0, 6 - c.length) + c;
  }

  function prepareDatasets(selectedASNs) {
    const datasets = [];

    selectedASNs.forEach(asn => {
      //const withdrawcolor = getRandomColor(asn.toString()+ '_withdraw');
      //const announcecolor = getRandomColor(asn.toString()+ '_announce');
      const withdrawcolor = 'rgba(192, 75, 192, 0.8)'; // Purple
      const announcecolor = 'rgba(75, 192, 192, 0.8)'; // Teal

      const withdrawPoints = withdrawData
        .filter(d => d.primary_as == asn)
        .map(d => ({ x: d.minute, y: d.updates }));

      const announcePoints = announceData
        .filter(d => d.primary_as == asn)
        .map(d => ({ x: d.minute, y: d.updates }));

      datasets.push({
        label: `Withdraw AS${asn}`,
        data: withdrawPoints,
        borderColor: withdrawcolor,
        backgroundColor: withdrawcolor,
        borderDash: [5, 5],
        tension: 0.3,
        fill: false
      });

      datasets.push({
        label: `Announce AS${asn}`,
        data: announcePoints,
        borderColor: announcecolor,
        backgroundColor: announcecolor,
        borderDash: [],
        tension: 0.3,
        fill: false
      });
    });

    return datasets;
  }

  function updateChart() {
  const select = document.getElementById('asSelect');
  let selectedASNs = Array.from(select.selectedOptions).map(opt => parseInt(opt.value));

  // Default to AS14593 if nothing is selected
  if (selectedASNs.length === 0) {
    selectedASNs = [16509];
  }

  const ctx = document.getElementById('updatemessagechart').getContext('2d');
  const datasets = prepareDatasets(selectedASNs);

  if (chartInstance) {
    chartInstance.destroy();
  }

  chartInstance = new Chart(ctx, {
    type: 'bar',
    data: {
      datasets: datasets
    },
    options: {
      responsive: true,
      maintainAspectRatio: false,
      scales: {
        x: {
          type: 'time',
      time: {
        unit: 'minute',
        tooltipFormat: 'yyyy-MM-dd HH:mm',
        displayFormats: {
          minute: 'yyyy-MM-dd HH:mm'  // <-- Force readable labels
        }
      },
      ticks: {
        callback: function(value, index, ticks) {
          const date = new Date(value);
          return date.toISOString().replace('T', ' ').substring(0, 16);
        }
      },
      title: {
        display: true,
        text: 'Time (GMT)'
      }
        },
        y: {
          beginAtZero: true,
          title: {
            display: true,
            text: 'Updates Count'
          }
        }
      },
      plugins: {
        legend: {
          position: 'top'
        },
        tooltip: {
          mode: 'index',
          intersect: false,
          callbacks: {
            // 👇 Fix the tooltip title formatting
            title: function(tooltipItems) {
              const date = new Date(tooltipItems[0].parsed.x);
              // Format: "2025-07-25 18:49"
              return date.toISOString().replace('T', ' ').substring(0, 16);
            },
            // Optional: Clean up label
            label: function(tooltipItem) {
              return `${tooltipItem.dataset.label}: ${tooltipItem.formattedValue}`;
            }
          }
        }
      }
    }
  });
}



  document.addEventListener('DOMContentLoaded', async () => {

    buildASDropdown(asList);
  });

</script>
