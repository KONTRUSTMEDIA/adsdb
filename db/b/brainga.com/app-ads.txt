    <script type='text/javascript'>
      // Global ad refresh timing tracking
      window.lastAdRefreshTime = 0;
      window.lastAnchorRefreshTime = 0;
      
      // Global function to refresh floating/anchor ads with timing control
      window.refreshFloating = function() {
          var currentTime = Date.now();
          var timeSinceLastAnchorRefresh = currentTime - window.lastAnchorRefreshTime;
          
          // Only refresh anchor ads if it's been at least 8 seconds
          if (timeSinceLastAnchorRefresh >= 8000) {
              var toRefresh = [];
              if (window.ezslots) {
                  window.ezslots.forEach(function(name) {
                      var slot = window[name];
                      if (slot && typeof slot.getSlotElementId === 'function' && slot.getSlotElementId().indexOf('Adhesion') > 0) {
                          toRefresh.push(slot);
                      }
                  });
              }
              
              if (toRefresh.length && window.googletag && window.googletag.pubads) {
                  window.googletag.pubads().refresh(toRefresh);
                  window.lastAnchorRefreshTime = currentTime;
              }
          }
      };
      
      // Global function to refresh all ads with timing control
      window.refreshAdsWithTiming = function(adIds) {
          var currentTime = Date.now();
          var timeSinceLastRefresh = currentTime - window.lastAdRefreshTime;
          
          // Refresh regular ads if it's been at least 5 seconds
          if (timeSinceLastRefresh >= 5000 && adIds && adIds.length > 0) {
              if (window.ezstandalone && ezstandalone.cmd) {
                  console.log('Refreshing ads:', adIds);
                  ezstandalone.cmd.push(function() {
                      ezstandalone.refreshAds(adIds);
                  });
                  window.lastAdRefreshTime = currentTime;
              }
          }
          
          // Also check anchor ads
          window.refreshFloating();
      };
    </script>
