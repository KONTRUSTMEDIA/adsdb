$(document).ready(function() {
    $("#contactForm").on("submit", function(event) {
      event.preventDefault();

      $('#form-locker').show();

      $('#sendmessage').slideUp();
      $('#errormessage').slideUp();

      // Recaptcha validation
      if(grecaptcha.getResponse() === "") {
        $('#errormessage').html("Please complete the Recaptcha.").slideDown();
        $('#form-locker').hide();
        return;
      }

      // Serialize form data
      var formData = $(this).serialize();

      // Send POST request to PHP script
      $.post("https://byitholdings.com//php/contact_mail.php", formData, function(response) {
        response = JSON.parse(response);

        if (response.type === 'success') {
          $('#sendmessage').slideDown();
          $('#contactForm').find("input, textarea").val("");
        } else {
          $('#errormessage').slideDown();
          $("#errormessage").html(response.msg);
        }

        $('#form-locker').hide();
      }).fail(function() {
        $('#errormessage').html("An error occurred. Please try again later.").slideDown();
        $('#form-locker').hide();
      });
    });
  });
</script>
