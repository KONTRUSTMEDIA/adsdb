    function toggleMenu() {
      const menu = document.getElementById('dropdownMenu');
      menu.classList.toggle('open');
    }

    document.addEventListener('click', function (e) {
      const menu = document.getElementById('dropdownMenu');
      const button = document.querySelector('.menu-button');
      if (!menu.contains(e.target) && !button.contains(e.target)) {
        menu.classList.remove('open');
      }
    });

    function loadGame() {
      const overlay = document.getElementById('overlay');
      const iframeWrapper = document.getElementById('iframeWrapper');

      overlay.style.display = 'none';

      const iframe = document.createElement('iframe');
      iframe.src = 'https://ubg98.github.io/BallSortPuzzle/';
      iframeWrapper.appendChild(iframe);
    }

    // ===== Modal open/close + accessibility =====
    const aboutBtn = document.getElementById('aboutBtn');
    const aboutModal = document.getElementById('aboutModal');
    const aboutClose = document.getElementById('aboutClose');
    const aboutDialog = aboutModal ? aboutModal.querySelector('.modal-dialog') : null;

    function openAbout() {
      if (!aboutModal) return;
      aboutModal.classList.add('open');
      if (aboutDialog) {
        aboutDialog.setAttribute('tabindex', '-1');
        aboutDialog.focus();
      }
      document.body.style.overflow = 'hidden';
    }

    function closeAbout() {
      if (!aboutModal) return;
      aboutModal.classList.remove('open');
      document.body.style.overflow = '';
      if (aboutBtn) aboutBtn.focus();
    }

    if (aboutBtn) aboutBtn.addEventListener('click', openAbout);
    if (aboutClose) aboutClose.addEventListener('click', closeAbout);

    if (aboutModal) {
      aboutModal.addEventListener('click', (e) => {
        if (e.target === aboutModal) closeAbout();
      });
    }

    document.addEventListener('keydown', (e) => {
      if (e.key === 'Escape' && aboutModal && aboutModal.classList.contains('open')) closeAbout();
    });
