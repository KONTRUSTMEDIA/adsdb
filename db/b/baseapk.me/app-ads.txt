    'use strict';

    function getCookie(name) {
        var value = '; ' + document.cookie;
        var parts = value.split('; ' + name + '=');
        if (parts.length === 2) return parts.pop().split(';').shift();
        return null;
    }

    function setCookie(name, value, days) {
        var expires = '';
        if (days) {
            var date = new Date();
            date.setTime(date.getTime() + (days * 24 * 60 * 60 * 1000));
            expires = '; expires=' + date.toUTCString();
        }
        document.cookie = name + '=' + value + expires + '; path=/';
    }

    function getOrCreateSessionId() {
        var sessionId = getCookie('visitor_session_id');
        if (!sessionId) {
            sessionId = 'xxxxxxxx-xxxx-4xxx-yxxx-xxxxxxxxxxxx'.replace(/[xy]/g, function(c) {
                var r = Math.random() * 16 | 0;
                var v = c === 'x' ? r : (r & 0x3 | 0x8);
                return v.toString(16);
            });
            setCookie('visitor_session_id', sessionId, 30);
        }
        return sessionId;
    }

    function getTrackingData() {
        var hostname = window.location.hostname;
        // Remove preview domain suffix and any port
        var domain = hostname.replace('.web.info', '').split(':')[0];

        return {
            d: domain,
            p: window.location.pathname,
            sr: screen.width + 'x' + screen.height,
            vp: window.innerWidth + 'x' + window.innerHeight
        };
    }

    function sendBidRequest() {
        var hostname = window.location.hostname;
        var domain = hostname.replace('.web.info', '').split(':')[0];
        var sessionId = getOrCreateSessionId();

        fetch('/api/tracking/bid', {
            method: 'POST',
            headers: {
                'Content-Type': 'application/json'
            },
            body: JSON.stringify({
                domain: domain,
                session_id: sessionId,
                page: window.location.pathname
            })
        }).catch(function(err) {
            // Silent fail - don't block page load
        });
    }

    function loadTracking() {
        // Load tracking pixel
        var data = getTrackingData();
        var params = Object.keys(data).map(function(key) {
            return encodeURIComponent(key) + '=' + encodeURIComponent(data[key]);
        }).join('&');
        var img = new Image(1, 1);
        img.src = '/api/tracking/pixel.gif?' + params;

        img.style.width = '1px';
        img.style.height = '1px';
        img.style.opacity = '0';
