function initRevealOnScroll() {
  const items = gsap.utils.toArray("[data-reveal]");

  // ✅ Detectar móvil (puedes ajustar el breakpoint)
  const isMobile = window.matchMedia("(max-width: 767px)").matches;

  items.forEach((el) => {
    const type = el.getAttribute("data-reveal");

    const start = el.dataset.revealStart || "top bottom";
    const end   = el.dataset.revealEnd   || "bottom top";
    const inY   = parseFloat(el.dataset.revealFromY || "24");
    const outY  = parseFloat(el.dataset.revealToY || "-12");

    const inDur  = parseFloat(el.dataset.revealInDur || "0.8");
    const outDur = parseFloat(el.dataset.revealOutDur || "0.35");

    const inEase  = el.dataset.revealInEase || "power3.out";
    const outEase = el.dataset.revealOutEase || "power2.in";

    // Estado inicial: oculto hasta que sea visible
    gsap.set(el, { autoAlpha: 0, y: inY, force3D: true });

    ScrollTrigger.create({
      trigger: el,
      start,
      end,
      // markers: true,

      onEnter: (self) => {
        gsap.to(el, {
          autoAlpha: 1,
          y: 0,
          duration: inDur,
          ease: inEase,
          overwrite: "auto",
          onComplete: () => {
            if (type === "text") {
              // animateTextExtra(el);
            }
          }
        });

        // ✅ En móvil: una vez mostrado, no reverses nunca más
        if (isMobile) self.kill();
      },

      // ✅ Solo en desktop/tablet hacemos “reverse” al salir
      ...(isMobile ? {} : {
        onLeave: () => {
          gsap.to(el, {
            autoAlpha: 0,
            y: outY,
            duration: outDur,
            ease: outEase,
            overwrite: "auto"
          });
        },

        onEnterBack: () => {
          gsap.to(el, {
            autoAlpha: 1,
            y: 0,
            duration: inDur,
            ease: inEase,
            overwrite: "auto"
          });
        },

        onLeaveBack: () => {
          gsap.to(el, {
            autoAlpha: 0,
            y: inY,
            duration: outDur,
            ease: outEase,
            overwrite: "auto"
          });
        }
      })
    });
  });

  ScrollTrigger.refresh();
}
