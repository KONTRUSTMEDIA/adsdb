function quitar_acentos(p_palabra) {
    p_palabra = p_palabra.replace(/á/g, "a");
    p_palabra = p_palabra.replace(/é/g, "e");
    p_palabra = p_palabra.replace(/í/g, "i");
    p_palabra = p_palabra.replace(/ó/g, "o");
    p_palabra = p_palabra.replace(/ú/g, "u");
    p_palabra = p_palabra.replace(/ü/g, "u");
    return p_palabra;
}

// Función para obtener el valor de una cookie por su nombre
function getCookieValue(cookieName) {
    // Separar las cookies individuales
    var cookies = document.cookie.split(';');
    
    // Recorrer cada cookie para buscar la que tenga el nombre especificado
    for(var i = 0; i < cookies.length; i++) {
        var cookie = cookies[i].trim();
        // Verificar si esta cookie tiene el nombre buscado
        if(cookie.indexOf(cookieName + '=') === 0) {
            // Si la encuentra, devolver el valor de la cookie
            return cookie.substring(cookieName.length + 1);
        }
    }
    // Si no se encuentra la cookie, devolver null
    return null;
}

    function convertir_a_12h(hora_24h) {
        //El texto es asi: 21:45 a 23:05
        //Entonces tengo que separar por "a" y tomar la primera parte
        var hora_inicio = hora_24h.split("a")[0].trim();
        //Luego tengo que separar por ":" y tomar la primera parte para obtener la hora
        var hora = parseInt(hora_inicio.split(":")[0]);
        var minutos = hora_inicio.split(":")[1];
        var am_pm = "AM";
        if (hora >= 12) {
            am_pm = "PM";
            if (hora > 12) {
                hora -= 12;
            }
        }
        if (hora == 0) {
            hora = 12;
        }
        //Y tengo que hacer la segunda parte lo mismo
        var hora_fin = hora_24h.split("a")[1].trim();
        var hora_fin_hora = parseInt(hora_fin.split(":")[0]);
        var hora_fin_minutos = hora_fin.split(":")[1];
        var am_pm_fin = "AM";
        if (hora_fin_hora >= 12) {
            am_pm_fin = "PM";
            if (hora_fin_hora > 12) {
                hora_fin_hora -= 12;
            }
        }
        if (hora_fin_hora == 0) {
            hora_fin_hora = 12;
        }
        return hora + ":" + minutos + " " + am_pm + " a " + hora_fin_hora + ":" + hora_fin_minutos + " " + am_pm_fin;
    }
