        async function loadPosts(postType, category, pageNumber, inicial, carregarHighlight, selectedTag, pesquisa) {
            await showLoader()
            
            atualizarPagina(pageNumber)
            jQuery.ajax({
                type: 'GET',
                url: 'https://republica45.com/wp-admin/admin-ajax.php',
                data: {
                    action: 'filter_posts_projetos_recentes',
                    post_type: 'project',
                    category: category,
                    highlight:"no",
                    paged: pageNumber,
                    selectedTag:selectedTag,
                    pesquisa:pesquisa 
                },
                success: async function (response) {
                    
                    jQuery('#post-container-all').html(response);
                    addActiveClass(pageNumber)
                    esconderBotoes(pageNumber)
                    if(!inicial){
                        irParaID('custom-posts-container')
                    }
                    jQuery('#placeholder-div').fadeOut()
                    jQuery('#post-container-all').fadeIn()
                    await hideLoader()
                    
                },
            });
            return

        function loadCategories(postType,selectedTag) {
            
            jQuery.ajax({
                type: 'GET',
                url: 'https://republica45.com/wp-admin/admin-ajax.php',
                data: {
                    action: 'filter_categories_projetos_recentes',
                    post_type: 'project',
                    tag:selectedTag,
                },
                success: function (response) {
                    
                    var categories = response.split('||');
                    var categoryDropdown = jQuery('#category-filter');
                    categoryDropdown.empty();
                    categoryDropdown.append('<option value="all">Service</option>');
                    categories.forEach(function (category) {
                        if(category.trim() != 'Sem Categoria' && category.trim() != 'Uncategorized' && category.trim() != '' ){
                            categoryDropdown.append('<option value="'+ category + '">'+ category + '</option>');

                        }
                    });
                },
            });
        }
        function loadTags(postType,category) {
            
            jQuery.ajax({
                type: 'GET',
                url: 'https://republica45.com/wp-admin/admin-ajax.php',
                data: {
                    action: 'filter_tags_projetos',
                    post_type: 'project',
                    category: category,
                },

                    });
                },
            });
        }
