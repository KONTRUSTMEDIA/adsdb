        let animationFrame;
        let stopTimeout;

        function autoScroll() {
            if (!isDragging && !isAutoScrolling) {
                isAutoScrolling = true; // Marca que o scroll automático está ativo

                function step() {
                    if (!isDragging) {
                        scrollableSection.scrollLeft += scrollSpeed;

                        // Se atingir o final, volta para o começo suavemente
                        if (scrollableSection.scrollLeft >= scrollableSection.scrollWidth - scrollableSection.clientWidth) {
                            scrollableSection.scrollLeft = 0;
                        }

                        animationFrame = requestAnimationFrame(step);
                    } else {
                        isAutoScrolling = false; // Reseta para permitir a retomada posterior
                    }
                }

                animationFrame = requestAnimationFrame(step);
            }
        }

        autoScroll(); // Inicia o scroll automático

        function startDrag(e) {
            isDragging = true;
            startX = (e.pageX || e.touches[0].pageX) - scrollableSection.offsetLeft;
            scrollLeft = scrollableSection.scrollLeft;
            cancelAnimationFrame(animationFrame); // Pausa o scroll automático ao arrastar
            clearTimeout(stopTimeout);
            isAutoScrolling = false;
        }

        function moveDrag(e) {
            if (!isDragging) return;
            e.preventDefault();
            const x = (e.pageX || e.touches[0].pageX) - scrollableSection.offsetLeft;
            const walk = (x - startX) * 1.5; // Multiplicador ajustado
            scrollableSection.scrollLeft = scrollLeft - walk;
        }

        function endDrag() {
            isDragging = false;

            stopTimeout = setTimeout(() => {
                autoScroll(); // Retoma o scroll automático suavemente
            }, 1000);
        }

        // Eventos de mouse
        scrollableSection.addEventListener("mousedown", startDrag);
        scrollableSection.addEventListener("mousemove", moveDrag);
        scrollableSection.addEventListener("mouseup", endDrag);
        scrollableSection.addEventListener("mouseleave", endDrag);

        // Eventos de toque (mobile)
        scrollableSection.addEventListener("touchstart", startDrag);
        scrollableSection.addEventListener("touchmove", moveDrag);
        scrollableSection.addEventListener("touchend", endDrag);
    }

    setTimeout(iniciarScroll, 1000);
