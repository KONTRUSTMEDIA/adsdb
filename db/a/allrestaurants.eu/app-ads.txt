        function voteForAdvert() {
            var modal = $('#voteModal');
            var form = new Form({
                'email': modal.find('.email').val(),
                'vote': modal.find('.vote-value').val(),
                'advert_id': modal.find('.advert-id').val(),

                        icon: 'success',

                        }
                        toast.fire({
                            icon: 'error',
                            title: title,
                        });
                        modal.modal('hide');
