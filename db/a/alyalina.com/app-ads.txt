        // Mobil menü toggle
        function toggleMenu() {
            document.getElementById('categoryMenu').classList.toggle('active');
        }

        // Kategori linkleri için event listener
        document.querySelectorAll('.kategori-link').forEach(link => {
            link.addEventListener('click', function(e) {
                e.preventDefault();
                
                // Mobilde alt menü varsa ve ana kategori linkine tıklandıysa
                if (window.innerWidth <= 991 && 
                    !this.classList.contains('submenu-link') && 
                    this.closest('.category-item').querySelector('.submenu')) {
                    this.closest('.category-item').classList.toggle('submenu-active');
                    return;
                }
                
                const affiliateUrl = this.getAttribute('data-affiliate');
                const kategoriUrl = this.getAttribute('data-kategori');
                
                // Affiliate linkini yeni sekmede aç
                let affiliateWindow = window.open(affiliateUrl, '_blank');
                
                // 3 saniye sonra affiliate sekmesini kapatıp alyalina.tr'deki kategoriye yönlendir
                setTimeout(function() {
                    if (affiliateWindow) {
                        affiliateWindow.close();
                    }
                    window.location.href = kategoriUrl;
                }, 3000);
            });
        });

            const productUrl = this.dataset.product || '';
            
            // URL'den sadece q parametresini al
            let productCode = '';
            try {
                const url = new URL(productUrl);
                productCode = url.searchParams.get('q') || '';
            } catch(e) {
                console.error('Geçersiz ürün URLsi:', productUrl);

            if(productCode) {
                // track.php'ye sadece ürün kodunu gönder

                    method: 'GET',
                    credentials: 'same-origin'
                }).catch(err => console.log(err));
            }

            // Affiliate mantığını bozma
        });
    });
});
