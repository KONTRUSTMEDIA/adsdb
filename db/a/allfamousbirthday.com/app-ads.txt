        const h = m.home_logo ? CDN + m.home_logo : BLANK;
        const a = m.away_logo ? CDN + m.away_logo : BLANK;
        const ll = m.league_logo ? `<img src="${CDN}${m.league_logo}" onerror="this.style.display='none'" alt="">` : '';
        const link = LINK_BASE + m.id + '&lang=en';
        return `
        <a href="${link}" target="_blank" rel="nofollow noopener" class="koora-card${st.cls==='live'?' is-live':''}">
            <div class="koora-league-bar">${ll}🏆 ${m.league_en || m.league || ''}</div>
            <div class="koora-body">
                <div class="koora-team">
                    <img src="${h}" alt="${m.home_en || m.home}" loading="lazy" onerror="this.src='${BLANK}'">
                    <span>${m.home_en || m.home}</span>
                </div>
                <div class="koora-mid">
                    ${showScore
                        ? `<span class="koora-score">${getScore(m)}</span>`
                        : `<span class="koora-time">${m.time}</span>`
                    }
                    <span class="koora-badge ${st.cls}">${st.label}</span>
                </div>
                <div class="koora-team">
                    <img src="${a}" alt="${m.away_en || m.away}" loading="lazy" onerror="this.src='${BLANK}'">
                    <span>${m.away_en || m.away}</span>
                </div>
            </div>
        </a>`;
    }

    function load() {
        fetch(API + 'api/matches/' + today() + '/1?t=' + Date.now())
            .then(r => r.json())
            .then(data => {
                const matches = data.matches || [];
                if (!matches.length) {
                    wrap.innerHTML = '<div class="text-center py-4 text-muted">No matches scheduled today.</div>';
                    return;
                }
                const order = { live: 1, ns: 2, fin: 3 };
                matches.sort((a, b) => (order[getStatus(a).cls]||4) - (order[getStatus(b).cls]||4));
                const live = matches.filter(m => getStatus(m).cls === 'live').length;
                const d = new Date().toLocaleDateString('en-GB', { weekday:'long', day:'numeric', month:'long', year:'numeric' });
                bar.innerHTML = `${d} &nbsp;·&nbsp; ${matches.length} matches${live ? ` &nbsp;·&nbsp; <span style="color:#ef4444;font-weight:700;">🔴 ${live} LIVE</span>` : ''}`;
                wrap.innerHTML = matches.map(buildCard).join('');
            })
            .catch(() => {
                wrap.innerHTML = '<div class="text-center py-4 text-muted">Could not load matches. Please refresh.</div>';
            });
    }

    load();
    setInterval(load, 30000);
})();
</script>
