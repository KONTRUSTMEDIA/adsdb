    // Mettre à jour l'état de la connexion dans localStorage
    function updateOnlineStatus() {
      localStorage.setItem('isOnline', navigator.onLine ? 'true' : 'false');
      document.documentElement.classList.toggle('is-offline', !navigator.onLine);

      // Émettre un événement personnalisé
      window.dispatchEvent(new CustomEvent('network-status-changed', {
        detail: { isOnline: navigator.onLine }
      }));
    }

    // Initialiser et ajouter des écouteurs d'événements
    window.addEventListener('online', updateOnlineStatus);
    window.addEventListener('offline', updateOnlineStatus);
    updateOnlineStatus(); // Initialiser
  </script>
