	function isTouchDevice() {
		return (('ontouchstart' in window) ||
			(navigator.maxTouchPoints > 0) ||
			(navigator.msMaxTouchPoints > 0));
	}

	function countChar(val) {
		var len = val.value.length;
		if (len >= 500) {
			val.value = val.value.substring(0, 500);
		} else {
			jQuery(val).parent().find(".counterValue").text(len + " /");
		}
	};

	function load_reviews(ajax_link, posts_per_page, paged, section_id, post_type) {
		var currentPostCount = jQuery("#" + section_id + " .masonry_list > div.review_item").length;

		paged = paged + 1;
		jQuery("#" + section_id + " .load_more_link .paged").val(paged);
		jsonObj = {
			"action": "filter_data",
			"postperpage": parseInt(posts_per_page),
			"currentPostCount": currentPostCount,
			"paged": paged,
			"post_type": post_type
		};

		jQuery.ajax({
			url: ajax_link,
			data: jsonObj, // form data
			type: "POST", // POST
			async: true,
			beforeSend: function(xhr) {
				console.log("ajax start"); // changing the button label
			},
			success: function(data) {
				// console.log(data.data);

				if (data.data.replace(/(\r\n\t|\n|\r\t)/gm, "").replace(" ", "") == "false" && type != 'loadmore') {
					jQuery('.load_more_img').hide();
					jQuery("#" + section_id + " .masonry_list").html("<div class='no_found'>Result not found!!</div>");
					jQuery("#" + section_id + " .masonry_list").show();
					jQuery("#" + section_id + " .load_more_button").hide();
					return false;
				} else {
					jQuery("#" + section_id + " .masonry_list").show();
					$items = jQuery(data.data);
					jQuery("#" + section_id + " .masonry_list").append($items)
						.isotope('appended', $items);
				}

				setTimeout(function() {
					var total = jQuery("#" + section_id + " #total_post_count").val();
					var current = jQuery("#" + section_id + " #current_post_count").val();

					var c_total = jQuery("#" + section_id + " .masonry_list > div.review_item").length;
					if (total != '' && current != '') {
						if (parseInt(total) == c_total) {
							jQuery("#" + section_id + " .load_more_button").hide();
						} else {
							jQuery("#" + section_id + " .load_more_button").show();
						}
					}
					jQuery("#" + section_id + " #total_post_count").remove();
					jQuery("#" + section_id + " #current_post_count").remove();
				}, 1000);

			},
			error: function(xhr) {
				jQuery('.load_more_img').hide();
				jQuery("#" + section_id + " .masonry_list").show();
				console.log(xhr);
			}
		});
	}
