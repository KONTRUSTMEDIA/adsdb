function addDigits(num) {
    // Base case: if number is single digit, return it
    if (num < 10) return num;
    
    // Calculate sum of digits
    let sum = 0;
    while (num > 0) {
        sum += num % 10;
        num = Math.floor(num / 10);
    }
    
    // Recursive call with sum
    return addDigits(sum);
}
