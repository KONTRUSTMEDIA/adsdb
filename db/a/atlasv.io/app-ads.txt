(function () {

  function updateVimeoIframes() {
    const iframes = document.querySelectorAll('.project-gallery iframe[src*="vimeo.com"]');
    if (!iframes.length) return;

    iframes.forEach(el => {
      // Ne retrait pas deux fois
      if (el.dataset.vimeoProcessed === "1") return;

      let original = el.src;

      // Séparation éventuelle du hash (#t=12s)
      const [urlWithParams, hash] = original.split('#');

      // Base sans params
      const [base, queryString] = urlWithParams.split('?');

      // On récupère les paramètres existants (si le thème en met)
      const params = new URLSearchParams(queryString || "");

      // Écrase uniquement les paramètres liés au background
      params.set("background", "1");
      params.set("autoplay", "1");
      params.set("loop", "1");
      params.set("muted", "1");
      params.set("controls", "0");

      // Construit l’URL finale en conservant un éventuel "#t=xx"
      const finalUrl =
        base + "?" + params.toString() + (hash ? "#" + hash : "");

      el.src = finalUrl;
      el.classList.add("vimeo-bg");
      el.setAttribute("allow", "autoplay; fullscreen");

      el.dataset.vimeoProcessed = "1"; // évite double traitement
    });
  }

  document.addEventListener("DOMContentLoaded", updateVimeoIframes);
  window.addEventListener("load", updateVimeoIframes);
  setTimeout(updateVimeoIframes, 1500);

})();
