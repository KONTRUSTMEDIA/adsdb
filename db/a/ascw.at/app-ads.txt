async function subscribeUser() {
    const nameF = document.getElementById('acy_name');
    const emailF = document.getElementById('acy_email');
    const msg = document.getElementById('acy_msg');
    const hp = document.getElementById('acy_work_email').value;
    const tStart = document.getElementById('acy_t_start').value;

    // 1. Client-seitiger Honeypot Schutz
    if (hp !== "") {
        msg.style.color = "#4CAF50";
        msg.innerText = "Erfolgreich! Bitte prüfe dein Postfach.";
        return; 
    }

    if (!emailF.value.includes('@')) {
        msg.style.color = "#ff5252";
        msg.innerText = "Bitte gib eine gültige E-Mail Adresse an.";
        return;
    }

    msg.style.color = "#fff";
    msg.innerText = "Übermittlung läuft...";

    try {
        const response = await fetch('https://newsletter.ascw.at/subscribe_bridge.php', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({
                name: nameF.value,
                email: emailF.value,
                work_email: hp,
                t_start: tStart
            })
        });

        const result = await response.text();

        if (result.includes('already') || result.includes('Bereits registriert')) {
            msg.style.color = "#ffb300";
            msg.innerText = "Du bist bereits für unseren Newsletter angemeldet!";
        } 
        else if (result.includes('success') || result.includes('1') || response.ok) {
            msg.style.color = "#4CAF50";
            msg.innerText = "Erfolgreich! Bitte prüfe dein Postfach zur Bestätigung.";
            nameF.value = "";
            emailF.value = "";
        } 
        else {
            msg.style.color = "#ff5252";
            msg.innerText = "Ein Fehler ist aufgetreten. Bitte versuche es später erneut.";
        }
    } catch (error) {
        msg.style.color = "#ff5252";
        msg.innerText = "Verbindung zum Server fehlgeschlagen.";
    }
}

(function() {
    const bar = document.getElementById('ascwSocialSidebar');
    const handle = document.getElementById('ascwDragHandle');
    const icon = document.getElementById('ascwToggleIcon');
    let isDragging = false, startY = 0, startTop = 0;

    function updateUI() {
        const isExp = bar.classList.contains('is-expanded');
        icon.innerText = isExp ? '◀' : '▶';
    }

    function toggleSidebar() {
        const expanded = bar.classList.toggle('is-expanded');
        localStorage.setItem('ascwSocialState', expanded ? 'expanded' : 'collapsed');
        updateUI();
    }

    function onMove(clientY) {
        const dy = clientY - startY;
        if (Math.abs(dy) > 20) { 
            if (!isDragging) {
                isDragging = true;
                bar.classList.add('ascw-dragging');
            }
            let nt = startTop + dy;
            nt = Math.max(10, Math.min(window.innerHeight - bar.offsetHeight - 10, nt));
            bar.style.top = nt + "px";
        }
    }

    handle.addEventListener('mousedown', (e) => {
        isDragging = false; startY = e.clientY; startTop = bar.offsetTop;
        const move = (ev) => onMove(ev.clientY);
        const up = () => { 
            bar.classList.remove('ascw-dragging');
            if(!isDragging) toggleSidebar(); 
            else localStorage.setItem('ascwSocialTop', bar.style.top); 
            document.removeEventListener('mousemove', move); 
            document.removeEventListener('mouseup', up); 
        };
        document.addEventListener('mousemove', move); document.addEventListener('mouseup', up);
        e.preventDefault();
    });

    handle.addEventListener('touchstart', (e) => { 
        isDragging = false; startY = e.touches[0].clientY; startTop = bar.offsetTop; 
    }, {passive: true});

    handle.addEventListener('touchmove', (e) => { 
        onMove(e.touches[0].clientY); 
        if (isDragging) e.preventDefault(); 
    }, {passive: false});

    handle.addEventListener('touchend', () => { 
        bar.classList.remove('ascw-dragging');
        if(!isDragging) toggleSidebar();

    });

    updateUI();
})();

<script>
/* Teilnehmerliste laden - Pfad zu ascw_customs angepasst */
document.addEventListener('DOMContentLoaded', function() {
    var elements = document.querySelectorAll('.participantsList');
    
    elements.forEach(function(el) {
        var ticketID = el.getAttribute('data-id');
        if (ticketID) {
            // Der Pfad zeigt nun auf das Unterverzeichnis ascw_customs/
            fetch('https://www.ascw.at/ascw_customs/participants.php?id=' + ticketID)
                .then(response => response.text())
                .then(data => {
                    el.innerHTML = data;
                })
                .catch(err => {
                    console.error('Fetch error:', err);
                    el.innerHTML = ' (Fehler beim Laden) ';
                });
        }
    });
});
</script>
