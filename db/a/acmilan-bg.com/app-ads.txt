<script>
// Close mobile menu when tapping outside
document.addEventListener("click", function (e) {
    var menu = document.querySelector(".navbar-collapse");
    var toggle = document.querySelector(".navbar-toggle");

    if (menu && toggle &&
        $(menu).hasClass("in") &&

        $(toggle).click();
    }
});

<script>
// ๏ฟฝ๏ฟฝ๏ฟฝ๏ฟฝ๏ฟฝ๏ฟฝ๏ฟฝ๏ฟฝ๏ฟฝ ๏ฟฝ๏ฟฝ๏ฟฝ๏ฟฝ๏ฟฝ๏ฟฝ ๏ฟฝ๏ฟฝ๏ฟฝ ๏ฟฝ๏ฟฝ๏ฟฝ๏ฟฝ๏ฟฝ๏ฟฝ๏ฟฝ
(function() {
    const saved = localStorage.getItem("theme");
    if (saved === "dark") {
        document.body.classList.add("theme-dark");
    } else {
        document.body.classList.add("theme-light");
    }
})();

// ๏ฟฝ๏ฟฝ๏ฟฝ๏ฟฝ๏ฟฝ๏ฟฝ๏ฟฝ๏ฟฝ๏ฟฝ๏ฟฝ ๏ฟฝ๏ฟฝ Disqus ๏ฟฝ๏ฟฝ๏ฟฝ๏ฟฝ๏ฟฝ๏ฟฝ
function updateDisqusTheme() {
    const isDark = document.body.classList.contains("theme-dark");
    const theme = isDark ? "dark" : "light";

    if (window.DISQUS) {
        DISQUS.reset({
            reload: true,
            config: function () {
                this.page.url = window.location.href;
                this.page.identifier = window.location.pathname;
                this.page.theme = theme;
            }
        });
    }
}

// ๏ฟฝ๏ฟฝ๏ฟฝ๏ฟฝ๏ฟฝ๏ฟฝ๏ฟฝ๏ฟฝ๏ฟฝ๏ฟฝ๏ฟฝ๏ฟฝ ๏ฟฝ๏ฟฝ ๏ฟฝ๏ฟฝ๏ฟฝ๏ฟฝ๏ฟฝ๏ฟฝ
document.getElementById("theme-toggle").addEventListener("click", function () {
    const body = document.body;

    body.classList.toggle("theme-dark");
    body.classList.toggle("theme-light");

    localStorage.setItem(
        "theme",
        body.classList.contains("theme-dark") ? "dark" : "light"
    );

    setTimeout(updateDisqusTheme, 150);
});
</script>












<script>
document.addEventListener("DOMContentLoaded", function() {

    // Find all blockquote elements inside #newsText
    var quotes = document.querySelectorAll("#newsText blockquote");

    quotes.forEach(function(bq) {

        var parent = bq.parentElement;

        // If blockquote is inside a <p>, move it out
        if (parent && parent.tagName.toLowerCase() === "p") {

            // Insert blockquote before the <p>
            parent.parentNode.insertBefore(bq, parent);

            // If the <p> becomes empty, remove it
            if (parent.textContent.trim() === "") {
                parent.remove();
            }
        }
    });

});
</script>
