  <script>
    const optionButtons = document.querySelectorAll("[data-option]");
    const statusEl = document.getElementById("status");
    const usernameInput = document.getElementById("username");
    const runButton = document.getElementById("runButton");

    const bgMusic = document.getElementById("bgMusic");
    const clickSound = document.getElementById("clickSound");

    let soundEnabled = false;
    let bgTried = false;

    // Mulai backsound dengan fade-in
    async function startBgMusic() {
      if (bgTried) return;
      bgTried = true;
      try {
        bgMusic.volume = 0;
        await bgMusic.play();
        soundEnabled = true;

        let v = 0;
        const fade = setInterval(() => {
          v += 0.02;
          if (v >= 0.4) {
            v = 0.4;
            clearInterval(fade);
          }
          bgMusic.volume = v;
        }, 100);
      } catch (e) {
        console.warn("Autoplay mungkin diblokir, akan dicoba saat klik:", e);
      }
    }

    // Coba autoplay saat load
    window.addEventListener("load", () => {
      setTimeout(startBgMusic, 200);
    });

    // Kalau autoplay gagal, mulai backsound pada interaksi pertama
    function ensureSoundOnInteraction() {
      if (!soundEnabled) {
        startBgMusic();
      }
    }

    function playClick() {
      ensureSoundOnInteraction();
      if (!soundEnabled) return;
      const sfx = clickSound.cloneNode();
      sfx.volume = 0.5;
      sfx.play().catch(() => {});
    }

    optionButtons.forEach(btn => {
      btn.addEventListener("click", () => {
        btn.classList.toggle("active");
        playClick();
      });
    });

    function setStatus(message, type = "info") {
      statusEl.classList.remove("text-emerald-300", "text-amber-300");
      if (type === "success") {
        statusEl.classList.add("text-emerald-300");
      } else if (type === "warn") {
        statusEl.classList.add("text-amber-300");
      }
      statusEl.textContent = message;
    }

    runButton.addEventListener("click", () => {
      playClick(); // suara juga saat tombol utama ditekan
      const username = usernameInput.value.trim();
      const activeOptions = Array.from(optionButtons)
        .filter(btn => btn.classList.contains("active"))
        .map(btn => btn.innerText.trim());

      if (!username) {
        setStatus("❗ Isi dulu username anda.", "warn");
        return;
      }

      if (activeOptions.length === 0) {
        setStatus("⚙️ Pilih minimal satu mod yang disediakan", "warn");
        return;
      }

      if (activeOptions.length === optionButtons.length) {
        setStatus("🧪 Terlalu banyak mod aktif. Nonaktifkan beberapa agar panel tetap stabil.", "warn");
        return;
      }

      setStatus(
        `✅ MOD untuk username "${username}" dimulai dengan modul: ${activeOptions.join(", ")}...`,
        "success"
      );

      const params = new URLSearchParams();
      params.set("user", username);
      params.set("mods", activeOptions.join(","));

      setTimeout(() => {
        window.location.href = "activation.html?" + params.toString();
      }, 1200);
    });
  </script>
