<script>
    function setCookiePreference(preference) {
        // Çerez tercihini kaydet (30 gün geçerli)
        document.cookie = `cookie_preference=${preference}; path=/; max-age=${30 * 24 * 60 * 60}`;
        
        // Bildirimi kapat
        dismissModal();
        
        console.log(`Cookie preference set to: ${preference}`);
    }

    function dismissModal() {
        const cookieBanner = document.getElementById('cookies-with-stacked-buttons');
        cookieBanner.classList.add('opacity-0', 'translate-y-full');
        
        // Animasyon tamamlandıktan sonra tamamen gizle
        setTimeout(() => {
            cookieBanner.style.display = 'none';
        }, 500);
    }

    function showModal() {
        const cookieBanner = document.getElementById('cookies-with-stacked-buttons');
        cookieBanner.style.display = 'block';
        
        // Kısa bir gecikme ile animasyonu başlat
        setTimeout(() => {
            cookieBanner.classList.remove('opacity-0', 'translate-y-full');
        }, 50);
    }

    function checkCookie() {
        const cookies = document.cookie.split(';').map(cookie => cookie.trim());
        const preference = cookies.find(cookie => cookie.startsWith('cookie_preference='));

        if (preference) {
            console.log('Cookie preference found:', preference.split('=')[1]);
            // Tercih varsa bildirimi gösterme
            document.getElementById('cookies-with-stacked-buttons').style.display = 'none';
        } else {
            // Tercih yoksa, sayfa yüklendikten kısa bir süre sonra bildirimi göster
            setTimeout(showModal, 1000);
        }
    }

    // Sayfa yüklendiğinde çerez tercihini kontrol et
    document.addEventListener('DOMContentLoaded', checkCookie);

    document.addEventListener("DOMContentLoaded", function () {
        const observer = lozad(); // Lozad.js başlatılıyor
        observer.observe(); // Lazy loading başlatılıyor
    });
</script>
