		async function getNotice() {
			var noticeContent = document.getElementById('notice-content');

			// 使用 fetch 发送 GET 请求

			fetch(apiurly + '/api/v3/home/getnotice', { credentials: 'include' })
				.then(response => {
					// 检查请求是否成功
					if (!response.ok) {
						throw new Error('请求失败，请重试。');
					}
					return response.json();
				})
				.then(data => {
					// 更新公告内容
					noticeContent.innerHTML = '<h2>' + data.data.text + '</h2>';
				})
				.catch(error => {
					// 显示错误消息
					noticeContent.innerHTML = '<h2>' + error.message + '</h2>';
				});
		}

		async function fetchAndRenderList(parentId) {
			try {
				const response = await fetch(apiurly + '/api/v3/help/getlist', { credentials: 'include' });

				const data = await response.json();

				if (data.code === 0 && data.data && Array.isArray(data.data.list)) {
					const listContainer = document.getElementById(parentId);

					data.data.list.forEach(item => {
						const listItem = document.createElement('md-list-item');
						listItem.setAttribute('type', 'link');
						listItem.setAttribute('href', '/help.html?id=' + item.id);
						// listItem.setAttribute('target', '_blank');

						const headline = document.createElement('div');
						headline.setAttribute('slot', 'headline');
						headline.textContent = item.title;

						const supportingText = document.createElement('div');
						supportingText.setAttribute('slot', 'supporting-text');
						supportingText.textContent = new Date(item.time * 1000).toLocaleDateString();

						listItem.appendChild(headline);
						listItem.appendChild(supportingText);

						listContainer.appendChild(listItem);
					});
				} else {
					console.error('Invalid response data:', data);
				}
			} catch (error) {
				const listContainer = document.getElementById(parentId);
				const listItem = document.createElement('md-list-item');
				const headline = document.createElement('div');
				headline.setAttribute('slot', 'headline');
				// 修正：textContent 应该是字符串
				headline.textContent = '加载失败';

				listItem.appendChild(headline);

				listContainer.appendChild(listItem);
				console.error('Error fetching data:', error);
			}
		}

		const searchHistory = {
			// 初始化历史记录
			init() {
				this.updateDisplay();
			},



			// 更新显示
			updateDisplay() {
				const container = document.getElementById('historyContainer');
				const list = document.getElementById('historyList');
				const history = JSON.parse(localStorage.getItem('searchHistory') || '[]');

				list.innerHTML = '';
				history.forEach(entry => {
					// 分割出name和uid
					const [name, uid] = entry.split(' ');

					const item = document.createElement('div');
					item.className = 'history-item';

					// 显示完整内容
					item.innerHTML = `
      <span style="">${name}</span>
      <span style="margin-left: 3px;
    color: #999;
    font-size: 12px;">${uid}</span>
    `;

					// 点击时只填入uid
					item.onclick = () => {
						document.getElementById('uidInput').value = uid;
					};

					list.appendChild(item);
				});

				container.style.display = history.length ? 'block' : 'none';
			}
		};

		function extractSingleContinuousNumber(input) {
			// 改进：处理 null 或 undefined 输入
			if (typeof input !== 'string') {
				alert("请输入正确的uid");
				return null; // 返回 null 表示无效输入
			}
			// 使用正则表达式匹配连贯的数字
			const matches = input.match(/\d+/g);

			if (!matches) {
				alert("请输入正确的uid");
				return null; // 返回 null 表示未找到数字
			} else if (matches.length > 1) {
				alert("请输入正确的uid (请勿包含多段数字)");
				return null; // 返回 null 表示找到多个数字段
			}

			const textField = document.getElementById('uidInput');
			textField.value = matches[0];
			// 返回匹配到的单个连贯数字
			return matches[0];
		}
