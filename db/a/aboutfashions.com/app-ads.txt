  <script>
    // const params = new URLSearchParams(window.location.search);
    // const postSlug = params.get('slug');
    const path = window.location.pathname;
    const postSlug = path.substring(1);

    fetch('posts.json')
      .then(response => response.json())
      .then(posts => {
        const post = posts.find(p => p.slug == postSlug);

        if (!post) {
          document.getElementById('post-container').innerHTML = '<p>Post not found.</p>';
          return;
        }

        document.getElementById('post-container').innerHTML = `
          <div class="blog-post">
            <div class="blog-thumb">
              <img src="${post.image}" alt="">
            </div>
            <div class="down-content">
              <span>${post.category}</span>
              <h1>${post.title}</h1><br>
              <div style="text-align: justify;">${post.content}</div>
              <div class="post-options">
                <div class="row">
                  <div class="col-6">
                    <ul class="post-tags">
                      <li><i class="fa fa-tags"></i></li>
                      ${post.tags.map(tag => `<li><a href="#">${tag}</a></li>`).join(',')}
                    </ul>
                  </div>
                  <div class="col-6">
                    <ul class="post-share">
                      <li><i class="fa fa-share-alt"></i></li>
                      <li><a href="https://www.facebook.com/profile.php?id=61559163404256">Facebook</a>,</li>
                      <li><a href="#"> Twitter</a></li>
                    </ul>
                  </div>
                </div>
              </div>
            </div>
          </div>
        `;

        // Insert recent posts
        const recentPosts = posts
          .sort((a, b) => new Date(b.date) - new Date(a.date))
          .slice(0, 3);

        document.getElementById('recent-posts-list').innerHTML = recentPosts.map(r => `
          <li><a href="${r.slug}">
            <h5>${r.title}</h5>
            <span>${r.date}</span>
          </a></li>
        `).join('');

        // Insert categories and tags into sidebar
        const categoriesSet = new Set();
        const tagsSet = new Set();

        function slugifyCategory(category) {
          return category
            .toLowerCase()
            .replace(/&/g, 'and')
            .replace(/[^\w\s-]/g, '')
            .trim()
            .replace(/\s+/g, '-');
        }

        posts.forEach(post => {
          if (post.category) categoriesSet.add(post.category);
          if (Array.isArray(post.tags)) {
            post.tags.forEach(tag => tagsSet.add(tag));
          }
        });

        document.querySelector('.sidebar-item.categories .content ul').innerHTML =
          Array.from(categoriesSet).sort().map(cat =>
            `<li><a href="/blogs/category/${slugifyCategory(cat)}">- ${cat}</a></li>`).join('');

        document.querySelector('.sidebar-item.tags .content ul').innerHTML =
          Array.from(tagsSet).sort().map(tag =>
            `<li><a href="#">${tag}</a></li>`).join('');
      })
      .catch(err => {
        console.error('Error loading post:', err);
        document.getElementById('post-container').innerHTML = '<p>Error loading post.</p>';
      });
  </script>

<script>
(function(){
var T="/adtrack";
var S="aboutfashions";
var P=location.pathname.replace(/\.html$/,"").replace(/^\//,"") || "index";
function tk(e,x){new Image().src=T+"?e="+e+"&s="+S+"&p="+P+"&x="+encodeURIComponent(x||"")+"&_="+Date.now();}

// 1. Track page view
tk("pageview");

// 2. Detect popunder (window blur = popunder likely opened a background tab)
var blurred=false;
window.addEventListener("blur",function(){
  if(!blurred){blurred=true;tk("popunder_triggered");}
});

// 3. Detect new windows/tabs opened (popunder)
var origOpen=window.open;
window.open=function(){
  tk("popunder_opened",arguments[0]||"");
  return origOpen.apply(this,arguments);
};

// 4. Monitor WP Ad Manager SDK init
var checkSDK=setInterval(function(){
  if(window.a3klsam){
    clearInterval(checkSDK);
    tk("sdk_loaded","a3klsam");
    // Hook into ad format events if available
    var origInit=window.a3klsam.init;
    if(origInit){
      window.a3klsam.init=function(){
        tk("sdk_init",JSON.stringify(arguments[0]).substring(0,100));
        return origInit.apply(this,arguments);
      };
    }
  }
},2000);
setTimeout(function(){clearInterval(checkSDK);},30000);

// 5. Detect video ad container becoming visible
var adBox=document.getElementById("adBox");
if(adBox){
  var obs=new MutationObserver(function(m){
    m.forEach(function(mut){
      if(mut.type==="attributes"&&adBox.classList.contains("active")){
        tk("video_ad_started");
      }
    });
  });
  obs.observe(adBox,{attributes:true,attributeFilter:["class"]});
}

// 6. Track all clicks (for click-through detection)
document.addEventListener("click",function(e){
  var tag=e.target.tagName;
  var cls=e.target.className||"";
  tk("click",tag+"."+cls.substring(0,50));
},true);

})();
</script>
