(function(){
  document.addEventListener('DOMContentLoaded', function(){
    document.querySelectorAll('.wa-generator').forEach(function(container){
      var phone = container.querySelector('.wa-phone');
      var msg = container.querySelector('.wa-message');
      var gen = container.querySelector('.wa-generate');
      var box = container.querySelector('.wa-resultBox');
      var res = container.querySelector('.wa-resultInput');
      var copy = container.querySelector('.wa-copy');
      var open = container.querySelector('.wa-open');

      gen.addEventListener('click', function(){
        var p = phone.value.trim();
        if(p === ''){ alert('Nomor WA tidak boleh kosong'); return; }
        var m = msg.value.trim();
        var url = 'https://wa.me/' + encodeURIComponent(p) + '?text=' + encodeURIComponent(m);
        res.value = url;
        open.href = url;
        box.style.display = 'block';
      });

      copy.addEventListener('click', function(){
        if(!res.value) return;
        // gunakan Clipboard API jika tersedia dan secure context
        if(navigator.clipboard && window.isSecureContext){
          navigator.clipboard.writeText(res.value).then(function(){ alert('Link berhasil disalin'); }).catch(function(){ alert('Gagal menyalin'); });
        } else {
          // fallback execCommand
          res.select();
          try { document.execCommand('copy'); alert('Link berhasil disalin'); }
          catch(e){ alert('Gagal menyalin. Silakan salin manual.'); }
        }
      });
    });
  });
})();
