<script>


    // encode this in the format expected by chatgpt so we dont' have to mess with that on the back end.
    var conversation = []

    $(document).ready(function() {
        $('#sendButton').click(function() {
            var userInput = $('#userInput').val();
            conversation.push({"role": "user", "content": userInput});

            if (userInput.trim() !== "") {
              $('#messageHistory').append('<div class="alert alert-primary ml-auto" role="alert" style="width:fit-content">' + userInput + '</div>');
              $('#messageHistory').scrollTop($('#messageHistory')[0].scrollHeight);
              $('#userInput').val('');

            $.ajax({
              url: '/get-response',
              method: 'POST',
              contentType: 'application/json',
              data: JSON.stringify({ prompt: conversation }),
              success: function(response) {
                $('#messageHistory').append('<div style="position:relative;"><span style="position: absolute; top:5px"><i class="fa-solid fa-cross"></i></span><div class="alert alert-secondary" style="width:fit-content; max-width:calc(100% - 22px); display: inline-block; margin-left:18px;" role="alert">' + response.message + '</div></div>');
                $('#messageHistory').scrollTop($('#messageHistory')[0].scrollHeight);
                conversation.push({"role": "assistant", "content": response.message});
              },
              error: function() {
                $('#messageHistory').append('<div class="alert alert-danger" role="alert">Error: Could not get a response.</div>');
                $('#messageHistory').scrollTop($('#messageHistory')[0].scrollHeight);
              }
            });
          }
        });

        $('#userInput').keypress(function(event) {
            if (event.which === 13) {
                $('#sendButton').click();
            }
        });
    });
</script>
