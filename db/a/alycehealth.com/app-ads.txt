      const EMAILJS_PUBLIC_KEY  = 'Qi5AyejEnV7txduEH';   
      const EMAILJS_SERVICE_ID  = 'service_cbqwpx4';  
      const EMAILJS_TEMPLATE_ID = 'template_8iyeue7'; 
      
      $('#footer').load('./include/footer.html', function(){
        emailjs.init({ publicKey: EMAILJS_PUBLIC_KEY });
      
        const form = document.querySelector('#footer #contactForm');
        if (!form) return;
      
        // 페이지 URL hidden 값 채우기
        const pageUrl = form.querySelector('[name="page_url"]');
        if (pageUrl) pageUrl.value = location.href;
      
        form.addEventListener('submit', async (e) => {
          e.preventDefault();
      
          // 간단 유효성
          if (!form.user_name.value.trim() || !form.user_email.value.trim() || !form.message.value.trim()) {
            alert('필수 항목을 입력해주세요.');
            return;
          }
      
          const btn = form.querySelector('.btn_submit');
          btn.disabled = true;
          btn.classList.add('loading');
      
          try {
            await emailjs.sendForm(EMAILJS_SERVICE_ID, EMAILJS_TEMPLATE_ID, form);
            alert('문의가 정상 접수되었습니다. 감사합니다!');
            form.reset();
          } catch (err) {
            console.error(err);
            alert('전송 중 오류가 발생했습니다. 잠시 후 다시 시도해주세요.');
          } finally {
            btn.disabled = false;
            btn.classList.remove('loading');
          }
        });
      });
