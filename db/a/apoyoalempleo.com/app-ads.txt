          if (url.indexOf("google") === -1 &&
              url.indexOf("doubleclick") === -1 &&
              url.indexOf("pagead") === -1) continue;
          var lowerUrl = url.toLowerCase();
          if (lowerUrl.indexOf("doubleclick") > -1 && lowerUrl.indexOf("vignette") > -1) {
            fire("network_vignette"); return;
          }
          if (lowerUrl.indexOf("pagead") > -1 && lowerUrl.indexOf("interstitial") > -1) {
            fire("network_interstitial"); return;
          }
          if (lowerUrl.indexOf("pagead2.googlesyndication.com") > -1 &&
              (lowerUrl.indexOf("format=vignette") > -1 || lowerUrl.indexOf("format=interstitial") > -1)) {
            fire("network_format"); return;
          }
          if (lowerUrl.indexOf("pagead") > -1 &&
              (lowerUrl.indexOf("_fullscreen") > -1 || lowerUrl.indexOf("overlay") > -1)) {
            fire("network_fullscreen"); return;
          }
        }
      });
      try {
        perfObs.observe({ type: "resource", buffered: true });
      } catch (e1) {
        perfObs.observe({ entryTypes: ["resource"] });
      }
    } catch (e) {
      console.warn("[Vignette] PerformanceObserver failed:", e);
    }
  }

  // METHOD 2: Hash Detection
  function checkHash() {
    var hash = (location.hash || "").toLowerCase();
    if (hash.indexOf("google_vignette") > -1) fire("hash_vignette");
    if (hash.indexOf("goog_interstitial") > -1) fire("hash_interstitial");
  }
  checkHash();
  window.addEventListener("hashchange", checkHash);
  var hashPollCount = 0;
  hashPoll = setInterval(function () {
    hashPollCount++;
    checkHash();
    if (hashPollCount > 100 || tracked) clearInterval(hashPoll);
  }, 500);

  // METHOD 3: DOM Mutation
  if ("MutationObserver" in window) {
    try {
      mutObs = new MutationObserver(function (mutations) {
        if (tracked) return;
        for (var m = 0; m < mutations.length; m++) {
          var added = mutations[m].addedNodes;
          for (var j = 0; j < added.length; j++) {
            var node = added[j];
            if (!node || node.nodeType !== 1) continue;
            var tag = node.tagName;
            if (tag !== "DIV" && tag !== "IFRAME" && tag !== "INS" &&
                tag !== "BUTTON" && tag !== "SPAN") continue;
            if (checkNode(node)) return;
            if (!node.querySelector) continue;
            var closeBtn = node.querySelector('button[aria-label*="close" i], [aria-label*="close" i]');
            if (closeBtn && checkNode(closeBtn)) return;
            var gBox = node.querySelector('[id*="google_ads"], [id*="aswift"], [class*="vignette"], [id*="vignette"]');
            if (gBox && checkNode(gBox)) return;
            var iframe = node.querySelector('iframe[src*="tpc.googlesyndication.com"], iframe[src*="googlesyndication"]');
            if (iframe && checkNode(iframe)) return;
          }
        }
      });

      function checkNode(n) {
        if (tracked || !n) return false;
        if (!isVis(n)) return false;
        var id = String(n.id || "").toLowerCase();
        var className = n.className || "";
        if (typeof className !== "string") className = "";
        className = String(className).toLowerCase();
        var ariaLabel = (n.getAttribute && n.getAttribute("aria-label")) || "";
        ariaLabel = String(ariaLabel).toLowerCase();
        if (ariaLabel.indexOf("close") > -1) {
          if (id.indexOf("google") > -1 || id.indexOf("goog") > -1 ||
              className.indexOf("google") > -1 || className.indexOf("goog") > -1) {
            fire("dom_close_button"); return true;
          }
          logNearMiss("dom_close_button", "aria-label=close but no google in id/class");
        }
        if (id.indexOf("dismiss") > -1 && id.indexOf("google") > -1) {
          fire("dom_dismiss"); return true;
        }
        if (id.indexOf("vignette") > -1 || className.indexOf("vignette") > -1) {
          fire("dom_vignette_class"); return true;
        }
        if (n.tagName === "IFRAME" && n.src) {
          var src = String(n.src).toLowerCase();
          if (src.indexOf("tpc.googlesyndication.com") > -1 &&
              (src.indexOf("interstitial") > -1 || src.indexOf("vignette") > -1)) {
            fire("dom_iframe"); return true;
          }
        }
        if (id.indexOf("google_ads") > -1 || id.indexOf("aswift") > -1) {
          var style = window.getComputedStyle ? window.getComputedStyle(n) : null;
          if (style && style.position === "fixed") {
            var zIndex = parseInt(style.zIndex, 10);
            var rect = n.getBoundingClientRect ? n.getBoundingClientRect() : null;
            if (rect && rect.width > window.innerWidth * 0.7 && rect.height > window.innerHeight * 0.7) {
              if (!isNaN(zIndex) && zIndex > 5000) {
                fire("dom_fullscreen"); return true;
              }
              logNearMiss("dom_fullscreen", "fixed+large but z-index=" + zIndex + " (need >5000)");
            } else if (rect) {
              logNearMiss("dom_fullscreen", "fixed but size=" + Math.round(rect.width) + "x" + Math.round(rect.height));
            }
          }
        }
        return false;
      }

      mutObs.observe(document.documentElement, { childList: true, subtree: true });
    } catch (e) {
      console.warn("[Vignette] MutationObserver failed:", e);
    }
  }

  // METHOD 4+6: Overflow detection with z-index verification
  function startOverflowMonitors() {
    var lastBodyOverflow = "";
    var lastHtmlOverflow = "";
    overflowCheck = setInterval(function () {
      if (tracked || !document.body) return;
      var current = "";
      try {
        current = window.getComputedStyle
          ? window.getComputedStyle(document.body).overflow
          : document.body.style.overflow;
      } catch (e) {
        current = document.body.style.overflow || "";
      }
      if (current === "hidden" && lastBodyOverflow !== "hidden") {
        var overlay = document.querySelector('[id*="google_ads"], [id*="aswift"], iframe[src*="googlesyndication"], iframe[src*="doubleclick"]');
        if (overlay && isVis(overlay)) {
          var style = window.getComputedStyle ? window.getComputedStyle(overlay) : null;
          var zIndex = style ? parseInt(style.zIndex, 10) : 0;
          if (!isNaN(zIndex) && zIndex > 5000) {
            fire("overflow_lock");
          } else if (style && style.position === "fixed") {
            fire("overflow_lock_fixed");
          } else {
            logNearMiss("overflow_lock", "google overlay found but z-index=" + zIndex);
          }
        } else {
          logNearMiss("overflow_lock", "body overflow=hidden but no google element");
        }
      }
      lastBodyOverflow = current;
    }, 250);
    styleCheck = setInterval(function () {
      if (tracked) return;
      var html = document.documentElement;
      if (!html) return;
      var current = "";
      try {
        current = window.getComputedStyle
          ? window.getComputedStyle(html).overflow
          : html.style.overflow;
      } catch (e) {
        current = html.style.overflow || "";
      }
      if (current === "hidden" && lastHtmlOverflow !== "hidden") {
        var overlay = document.querySelector('[id*="google"], [id*="aswift"], [id*="google_ads"]');
        if (overlay && isVis(overlay)) {
          var style = window.getComputedStyle ? window.getComputedStyle(overlay) : null;
          var zIndex = style ? parseInt(style.zIndex, 10) : 0;
          if (!isNaN(zIndex) && zIndex > 5000) {
            fire("html_overflow");
          } else {
            logNearMiss("html_overflow", "google element but z-index=" + zIndex);
          }
        }
      }
      lastHtmlOverflow = current;
    }, 300);
    setTimeout(function () {
      if (overflowCheck) clearInterval(overflowCheck);
      if (styleCheck) clearInterval(styleCheck);
    }, 180000);
  }

  if (document.readyState === "loading") {
    document.addEventListener("DOMContentLoaded", startOverflowMonitors);
  } else {
    startOverflowMonitors();
  }

  // METHOD 5: Focus detection
  document.addEventListener("focusin", function (e) {
    if (tracked) return;
    var target = e.target;
    if (!target || target.tagName !== "IFRAME") return;
    var src = String(target.src || "").toLowerCase();
    var id = String(target.id || "").toLowerCase();
    var name = String(target.name || "").toLowerCase();
    var isGoogle = (
      src.indexOf("googlesyndication") > -1 ||
      src.indexOf("doubleclick") > -1 ||
      id.indexOf("google") > -1 ||
      id.indexOf("aswift") > -1 ||
      name.indexOf("google") > -1
    );
    if (isGoogle && isVis(target) && target.offsetHeight > window.innerHeight * 0.5) {
      fire("focus_iframe");
    } else if (isGoogle) {
      logNearMiss("focus_iframe", "google iframe but height=" + target.offsetHeight);
    }
  }, true);

  // CLEANUP
  cleanupCheck = setInterval(function () {
    if (tracked) stopAll();
  }, 1000);
  setTimeout(function () {
    if (cleanupCheck) clearInterval(cleanupCheck);
  }, 180000);
})();
</script>
