    function detectAdblock(callback) {
        var bait = document.createElement('div');
        bait.setAttribute('class', 'ad-banner ads-container ad-placement');
        bait.style.cssText = 'height:1px;width:1px;position:absolute;left:-9999px;top:-9999px;';
        document.body.appendChild(bait);
        setTimeout(function() {
            var blocked = false;
            if (!bait.offsetParent || bait.offsetHeight === 0 || bait.offsetWidth === 0) blocked = true;
            try {
                var s = window.getComputedStyle(bait);
                if (s && (s.display === 'none' || s.visibility === 'hidden')) blocked = true;
            } catch(e) {}
            try { document.body.removeChild(bait); } catch(e) {}
            callback(blocked ? 'true' : 'false');
        }, 200);

        payload.append('adblock', window._acapAdblock || 'unknown');
