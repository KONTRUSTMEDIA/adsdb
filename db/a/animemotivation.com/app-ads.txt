document.addEventListener('DOMContentLoaded', function() {
    // Safe Matomo push helper
    function trackPollEvent(action, name, value) {
        if (typeof _paq !== 'undefined') {
            try {
                var label = (value !== undefined && value !== null) ? (name + ' - ' + value) : name;
                _paq.push(['trackEvent', 'Poll', action, label]);
            } catch (e) {
                // fail silently
                console.error('Matomo track error', e);
            }
        }
    }

    // Detect AYS Poll container (adjust if your theme/plugin uses different selectors)
    function findPollContainer(el) {
        return el.closest('[id^="ays-poll"], [class*="ays-poll"]');
    }

    // Get poll name/title in a robust way
    function getPollName(pollEl) {
        if (!pollEl) return 'Unknown Poll';
        var titleEl = pollEl.querySelector('.ays-poll-question, .ays-poll-title, h3, h2');
        return (titleEl && titleEl.textContent) ? titleEl.textContent.trim().substring(0, 120) : (pollEl.id || 'Unknown Poll');
    }

    // Normalize selected value text
    function safeValue(v) {
        if (v === undefined || v === null) return '';
        return String(v).trim().substring(0, 120);
    }

    // Handle answer selection (radio / checkbox / select)
    document.addEventListener('change', function(e) {
        var input = e.target;
        if (!input) return;

        // find the poll container
        var pollEl = findPollContainer(input);
        if (!pollEl) return;

        // determine value: for select, use selected option; for radios/checkbox use value or label text
        var value = '';
        if (input.tagName.toLowerCase() === 'select') {
            value = input.options[input.selectedIndex] ? input.options[input.selectedIndex].text : input.value;
        } else if (input.type === 'radio' || input.type === 'checkbox') {
            value = input.value || (input.nextElementSibling ? input.nextElementSibling.textContent : '');
        } else {
            value = input.value || input.textContent || '';
        }
        value = safeValue(value);

        // avoid duplicate tracking for same poll + same answer
        var last = pollEl.dataset.lastTrackedAnswer || '';
        var name = getPollName(pollEl);
        if (last === value) return; // nothing new

        // store last tracked answer
        pollEl.dataset.lastTrackedAnswer = value;

        // mark that poll has an answer selected (for submit event)
        pollEl.dataset.hasAnswer = '1';

        trackPollEvent('Poll Answer', name, value);
    }, true);

    // Handle clicks to detect submit-like actions inside polls.
    // This covers buttons/labels that trigger a vote, and also handles custom AJAX submit buttons.
    document.addEventListener('click', function(e) {
        var target = e.target;
        if (!target) return;

        var pollEl = findPollContainer(target);
        if (!pollEl) return;

        // If target is a true form submit button (type=submit) or has class/id hinting a submit, treat as submit
        var isSubmit = false;

        // common heuristics: button/input[type=submit], or classes containing 'submit','vote','send','submit-vote'
        var tag = target.tagName.toLowerCase();
        var type = (target.getAttribute('type') || '').toLowerCase();
        var cls = (target.className || '').toLowerCase();

        if (tag === 'button' && (type === 'submit' || cls.indexOf('submit') !== -1 || cls.indexOf('vote') !== -1 || cls.indexOf('send') !== -1)) {
            isSubmit = true;
        } else if ((tag === 'input') && (type === 'submit' || type === 'button')) {
            isSubmit = true;
        } else if (cls.indexOf('ays-poll-submit') !== -1 || cls.indexOf('ays-vote') !== -1) {
            // plugin-specific class heuristics
            isSubmit = true;
        } else {
            // sometimes labels or custom elements trigger the vote - check data-action/data-role
            var dataRole = (target.getAttribute('role') || '').toLowerCase();
            var dataAct = (target.getAttribute('data-action') || '').toLowerCase();
            if (dataRole === 'button' || dataAct.indexOf('vote') !== -1 || dataAct.indexOf('submit') !== -1) {
                isSubmit = true;
            }
        }

        if (!isSubmit) return;

        // prevent duplicates: only allow one submit event per poll per page load
        if (pollEl.dataset.trackedSubmit === '1') return;
        pollEl.dataset.trackedSubmit = '1';

        var name = getPollName(pollEl);
        var lastAnswer = pollEl.dataset.lastTrackedAnswer || '';

        // If no tracked answer, try to infer current selected value
        if (!lastAnswer) {
            try {
                var sel = pollEl.querySelector('input[type="radio"]:checked, input[type="checkbox"]:checked, select');
                if (sel) {
                    if (sel.tagName.toLowerCase() === 'select') {
                        lastAnswer = sel.options[sel.selectedIndex] ? sel.options[sel.selectedIndex].text : sel.value;
                    } else {
                        lastAnswer = sel.value || (sel.nextElementSibling ? sel.nextElementSibling.textContent : '');
                    }
                    lastAnswer = safeValue(lastAnswer);
                }
            } catch (err) {
                lastAnswer = '';
            }
        }

        trackPollEvent('Poll Submit', name, lastAnswer || 'no-answer');
    }, true);

    // Optional: catch keyboard "Enter" submits inside poll areas (for accessibility)
    document.addEventListener('keydown', function(e) {
        if (e.key !== 'Enter') return;
        var target = e.target;
        var pollEl = findPollContainer(target);
        if (!pollEl) return;

        // If Enter pressed on a submitable element, treat as submit (use same de-dup protection)
        var active = document.activeElement;
        if (active && (active.tagName.toLowerCase() === 'button' || active.tagName.toLowerCase() === 'input' || active.getAttribute('role') === 'button')) {
            if (pollEl.dataset.trackedSubmit === '1') return;
            pollEl.dataset.trackedSubmit = '1';
            var name = getPollName(pollEl);
            var lastAnswer = pollEl.dataset.lastTrackedAnswer || 'no-answer';
            trackPollEvent('Poll Submit', name, lastAnswer);
        }
    }, true);

    // Optional: small garbage-collection to avoid memory bloat for long pages with many polls
    setInterval(function() {
        document.querySelectorAll('[id^="ays-poll"], [class*="ays-poll"]').forEach(function(el) {

        });
    }, 600000); // every 10 minutes (no-op by default)
});
