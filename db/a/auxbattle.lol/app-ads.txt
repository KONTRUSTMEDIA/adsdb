      async function createPromptPack() {
    const { value: formValues } = await Swal.fire({
      heightAuto: false, 
      title: "Create Your Prompt Pack",
      html:
      "<p>Inappropriate Packs Containing Offensive Content or Slurs Will Not Be Allowed</p>" +
        '<input id="pack-name" class="swal2-input" placeholder="Pack Name">' +
        '<textarea id="pack-prompts" class="swal2-textarea" placeholder="Please enter your prompts, each on a new line:\nBest song to dance to\nSong that captivates you\netc..."></textarea>',
      focusConfirm: false,
      preConfirm: () => {
        const name = document.getElementById("pack-name").value.trim();
        const prompts = document
          .getElementById("pack-prompts")
          .value.trim()
          .split("\n")
          .filter((prompt) => prompt);

        if (!name || prompts.length === 0) {
          Swal.showValidationMessage("Please provide a pack name and at least one prompt.");
          return null;
        }

        return { [name]: prompts };
      },
    });

    if (formValues) {
  // Send the new prompt pack to the server
  fetch("/update-prompt-pack", {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify(formValues),
  })
    .then((response) => {
      if (response.ok) {
        return response.json().then((data) => {
          Swal.fire({
            title: "Success",
            text: data.message || "Prompt pack uploaded successfully! Please note: custom packs are deleted whenever the game updates, as this is a temporary feature. Additionally, we reserve the right to remove any packs deemed low quality or inappropriate.",
            icon: "success",
            heightAuto: false, // Prevent the popup from adjusting height automatically
          });
        });
      } else {
        return response.json().then((errorData) => {
          if (errorData.error && errorData.reason === "slurs_detected") {
            Swal.fire({
              title: "Inappropriate Content",
              text: "Your prompt pack contains inappropriate language or slurs. Please remove such content and try again.",
              icon: "error",
              heightAuto: false, // Prevent the popup from adjusting height automatically
            });
          } else {
            Swal.fire({
              title: "Error",
              text: errorData.message || "Failed to save your prompt pack.",
              icon: "error",
              heightAuto: false, // Prevent the popup from adjusting height automatically
            });
          }
        });
      }
    })
    .catch(() => {
      Swal.fire({
        title: "Error",
        text: "Could not connect to the server.",
        icon: "error",
        heightAuto: false, // Prevent the popup from adjusting height automatically
      });
    });

}

  }

    const { value: roomName } = await Swal.fire({
      heightAuto: false, 
      title: 'Enter Room Name',
      input: 'text',
      inputPlaceholder: 'Enter room name',
      showCancelButton: true,
      inputValidator: (value) => {
        if (!value || !/\S/.test(value)) {
          return 'Invalid Room Name!';
        }
        if (value.length >= 25) {
          return 'Room name must be less than 25 characters!';
        }
        return null;
      }
    });

    if (roomName) {
      socket.emit("createRoom", roomName, location.pathname);
    }
  }, 250);
}
