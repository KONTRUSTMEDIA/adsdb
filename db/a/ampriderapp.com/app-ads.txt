  // update target on mouse move
  document.addEventListener("mousemove", e => {
    const centerX = window.innerWidth / 2;
    const centerY = window.innerHeight / 2;
    const moveX = (e.clientX - centerX) / centerX;
    const moveY = (e.clientY - centerY) / centerY;
    targetX = moveX * 30;
    targetY = moveY * 30;

  function animate() {
    movers.forEach(obj => {
      // smooth easing

      obj.y += (targetY * obj.strength - obj.y) * 0.1;
      obj.el.style.transform = `translate(${obj.x}px, ${obj.y}px)`;
    });
    requestAnimationFrame(animate);
  }

  animate();

<script>
  const video = document.getElementById('myVideo');
  const overlay = document.getElementById('overlay');
  const startBtn = document.getElementById('startBtn');
  const mobileFrame = document.getElementById('mobileVideo');

  let hasStarted = false;
  let isPausedManually = false;

  // Scroll observer for auto play/pause
  const observer = new IntersectionObserver((entries) => {
    entries.forEach(entry => {
      if (entry.isIntersecting) {
        if (hasStarted && !isPausedManually) {
          video.play().catch(err => console.log('Autoplay blocked:', err));
        }
      } else {
        video.pause();
      }
    });
  }, { threshold: 0.5 });
  observer.observe(mobileFrame);

  // Start or resume video when overlay button pressed
  startBtn.addEventListener('click', () => {
    video.play();
    hasStarted = true;
    isPausedManually = false;
    overlay.classList.add('hidden');
  });

  // Tap video to toggle play/pause
  video.addEventListener('click', () => {
    if (!hasStarted) return; // ignore before first play

    if (video.paused) {
      video.play();
      isPausedManually = false;
      overlay.classList.add('hidden');
    } else {
      video.pause();
      isPausedManually = true;
      overlay.classList.remove('hidden'); // show play overlay again
    }
  });

  // Automatically show overlay when paused
  video.addEventListener('pause', () => {
    if (hasStarted && isPausedManually) {
      overlay.classList.remove('hidden');
    }
  });

  // Hide overlay when playing
  video.addEventListener('play', () => {
