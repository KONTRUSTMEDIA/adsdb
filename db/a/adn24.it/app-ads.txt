            document.addEventListener('DOMContentLoaded', function () {
                var wrapper = document.querySelector('.breaking-scroll-wrapper');
                if (!wrapper) return;
                var track = wrapper.querySelector('.breaking-scroll');
                var prev = wrapper.querySelector('.breaking-prev');
                var next = wrapper.querySelector('.breaking-next');
                if (!track || !prev || !next) return;

                function getItems() {
                    return Array.prototype.slice.call(track.querySelectorAll('.breaking-item'));
                }

                var items = getItems();
                var currentIndex = 0;

                function clampIndex(i) {
                    if (!items.length) return 0;
                    return Math.max(0, Math.min(i, items.length - 1));
                }

                function setIndex(i) {
                    currentIndex = clampIndex(i);
                    var el = items[currentIndex];
                    if (!el) return;
                    try {
                        el.scrollIntoView({ behavior: 'smooth', inline: 'start', block: 'nearest' });
                    } catch (e) {
                        // Fallback
                        var rectT = track.getBoundingClientRect();
                        var rectE = el.getBoundingClientRect();
                        var left = rectE.left - rectT.left + track.scrollLeft - 4; // offset
                        track.scrollLeft = Math.max(0, left);
                    }
                }

                function updateIndexFromScroll() {
                    var minDelta = Infinity;
                    var idx = currentIndex;
                    var tLeft = track.getBoundingClientRect().left;
                    for (var i = 0; i < items.length; i++) {
                        var d = Math.abs(items[i].getBoundingClientRect().left - tLeft);
                        if (d < minDelta) { minDelta = d; idx = i; }
                    }
                    currentIndex = idx;
                }

                track.addEventListener('scroll', function () {
                    // Debounced-ish update
                    window.clearTimeout(track.__snapTimer);
                    track.__snapTimer = window.setTimeout(updateIndexFromScroll, 80);
                }, { passive: true });

                prev.addEventListener('click', function () {
                    setIndex(currentIndex - 1);
                });
                next.addEventListener('click', function () {
                    setIndex(currentIndex + 1);
                });
            });
