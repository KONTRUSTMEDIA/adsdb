    <script>
      // X-Ray Particles - Optimized & Enhanced
      (function () {
        let canvas;
        let c = init("canvas"),
          w = (canvas.width = window.innerWidth-5),
          h = (canvas.height = window.innerHeight);

        let mouse = {x: w/2, y: h/2};
        let time = 0;
        const MAX_PARTICLES = 250; // Maximum total particles allowed
        let lastClickTime = 0;
        const CLICK_COOLDOWN = 150; // Milliseconds between clicks

        class Particle {
          constructor(x, y, isBurst = false){
            this.x = x || Math.random() * w;
            this.y = y || Math.random() * h;
            this.vx = isBurst ? 0 : (Math.random() - 0.5) * 0.5;
            this.vy = isBurst ? 0 : (Math.random() - 0.5) * 0.5;
            this.size = Math.random() * 2 + 1;
            this.maxSize = this.size;
            this.opacity = 0;
            this.targetOpacity = Math.random() * 0.4 + 0.3;
            this.life = isBurst ? 1 : 0;
            this.isBurst = isBurst;
            this.pulseOffset = Math.random() * Math.PI * 2;
          }

          update(){
            // Velocity with slight drift
            this.vx += (Math.random() - 0.5) * 0.05;
            this.vy += (Math.random() - 0.5) * 0.05;

            // Mouse interaction - subtle attraction
            if(!this.isBurst){
              let dx = mouse.x - this.x;
              let dy = mouse.y - this.y;
              let dist = Math.sqrt(dx*dx + dy*dy);
              if(dist < 150){
                let force = (150 - dist) / 150 * 0.3;
                this.vx += (dx / dist) * force;
                this.vy += (dy / dist) * force;
              }
            }

            // Apply velocity with damping
            this.x += this.vx;
            this.y += this.vy;
            this.vx *= 0.98;
            this.vy *= 0.98;

            // Fade in/out
            if(this.isBurst){
              this.life -= 0.01;
              this.opacity = this.life * 0.6;
            } else {
              this.opacity += (this.targetOpacity - this.opacity) * 0.05;
            }

            // Edge bounce
            if(this.x < 0 || this.x > w) this.vx *= -0.5;
            if(this.y < 0 || this.y > h) this.vy *= -0.5;
            this.x = Math.max(0, Math.min(w, this.x));
            this.y = Math.max(0, Math.min(h, this.y));

            // Pulsing effect
            let pulse = Math.sin(time * 0.02 + this.pulseOffset) * 0.3 + 1;
            this.currentSize = this.maxSize * pulse;
          }

          draw(){
            if(this.opacity <= 0.01) return;

            // Use shadowBlur for glow (more performant than gradients)
            c.shadowBlur = 15;
            c.shadowColor = `rgba(255, 255, 255, ${this.opacity * 0.6})`;

            c.beginPath();
            c.arc(this.x, this.y, this.currentSize, 0, Math.PI * 2);
            c.fillStyle = `rgba(255, 255, 255, ${this.opacity})`;
            c.fill();

            c.shadowBlur = 0;
          }

          isDead(){
            return this.isBurst && this.life <= 0;
          }
        }

        let particles = [];

        function makeCluster(x, y){
          // Check particle limit before creating burst
          if (particles.length >= MAX_PARTICLES - 40) {
            console.log('Max particles reached, skipping burst');
            return;
          }

          // Check click cooldown
          const now = Date.now();
          if (now - lastClickTime < CLICK_COOLDOWN) {
            return;
          }
          lastClickTime = now;

          // Dramatic burst with more particles
          let count = 40;
          for(let i = 0; i < count; i++){
            let p = new Particle(x, y, true);
            let angle = (Math.PI * 2 * i) / count;
            let speed = Math.random() * 4 + 2;
            p.vx = Math.cos(angle) * speed;
            p.vy = Math.sin(angle) * speed;
            p.size = Math.random() * 3 + 1;
            p.maxSize = p.size;
            p.currentSize = p.size;
            particles.push(p);
          }
        }

        function drawConnections(){
          // Spatial optimization: only check nearby particles
          let gridSize = 150;
          let grid = {};

          particles.forEach(p => {
            let key = Math.floor(p.x/gridSize) + ',' + Math.floor(p.y/gridSize);
            if(!grid[key]) grid[key] = [];
            grid[key].push(p);
          });

          c.globalAlpha = 1;
          for(let key in grid){
            let cell = grid[key];
            let [gx, gy] = key.split(',').map(Number);

            // Check current and adjacent cells
            for(let dx = -1; dx <= 1; dx++){
              for(let dy = -1; dy <= 1; dy++){
                let adjKey = (gx+dx) + ',' + (gy+dy);
                if(!grid[adjKey]) continue;

                for(let p1 of cell){
                  for(let p2 of grid[adjKey]){
                    if(p1 === p2) continue;
                    let dx = p1.x - p2.x;
                    let dy = p1.y - p2.y;
                    let dist = Math.sqrt(dx*dx + dy*dy);

                    if(dist < 120){
                      let alpha = (1 - dist/120) * 0.2 * Math.min(p1.opacity, p2.opacity);
                      c.beginPath();
                      c.strokeStyle = `rgba(255, 255, 255, ${alpha})`;
                      c.lineWidth = 0.5;
                      c.moveTo(p1.x, p1.y);
                      c.lineTo(p2.x, p2.y);
                      c.stroke();
                    }
                  }
                }
              }
            }
          }
        }

        function draw() {
          time++;

          // Add new particles slowly (respecting max limit)
          while(particles.length < 120 && particles.length < MAX_PARTICLES){
            particles.push(new Particle());
          }

          // Update and clean
          particles = particles.filter(p => {
            p.update();
            return !p.isDead();
          });

          // Draw connections first (behind particles)
          drawConnections();

          // Draw particles
          particles.forEach(p => p.draw());
        }

        canvas.addEventListener("mousemove", function(e) {
          mouse.x = e.clientX;
          mouse.y = e.clientY;
        }, false);
        function init(elemid) {
          canvas = document.createElement("canvas");
          canvas.style.position='absolute';
          canvas.style.zIndex=0;
          canvas.style.pointerEvents='none'; // Allow clicks to pass through

          window.addEventListener("click",
        function(e){
          makeCluster(e.clientX, e.clientY);
        });

          //let canvas = document.getElementById(elemid),
          let c = canvas.getContext("2d"),
          w = (canvas.width = window.innerWidth),
          h = (canvas.height = window.innerHeight);
          c.fillStyle = "rgba(255,255,255,0)"; // Transparent for white background
          c.fillRect(0, 0, w, h);
          document.body.prepend(canvas);
          return c;
        }
        function loop() {
          requestAnimationFrame(loop);
          c.clearRect(0, 0, w, h);
          draw();
        }

        window.addEventListener("resize", function() {
          w = canvas.width = window.innerWidth;
          h = canvas.height = window.innerHeight;
        });

        loop(); // Start animation
      }());
    </script>
