  <div class="w3-container w3-card w3-white w3-round w3-margin"><h3 class='w3-center  rheader'>What is Arts Intercultural Amusement and Movie Makers Association</h3>
  <p class="text-justify">The Arts Intercultural Amusement and Movie Makers Association is a social service organization affiliated with <strong>AIAMMA</strong>.  It is dedicated to promoting arts, fostering cultural exchange, and supporting the public. Through creativity-driven initiatives and community-focused programs, the association enhances collaboration and contributes to social welfare.</p>
<p class="text-justify">The association operates through a broad network of area-wise, district-wise, and state-wise chapters, providing a space for individuals in various artistic and cultural fields to connect and collaborate. With over 2,152+ registered members and more than 22,521+ supporting members, the association unites professionals and enthusiasts alike, each benefiting from Arts Intercultural Amusement and Movie Makers Association’s mission to nurture and support the arts.</p>
<p class="text-justify">The association's journey began with the visionary aspirations of Varghese (Angel) and Anila (Aswathy), who sought to establish a shared space for collaborative opportunities and the collective welfare of its members across various artistic disciplines.         <span onclick="this.parentElement.style.display='none'" class="w3-button w3-text-white w3-display-topright"><i class="fa fa-remove"></i></span>
        <div class="icon">${icon}</div>
        <div class="title">${title}</div>
        <div class="message">${message}</div>
        <div class="date">${date}</div>`;
    container.appendChild(notification);

    // Remove notification after 14 seconds
    setTimeout(() => {
        notification.remove();
    }, 14000);
}

// Function to fetch notifications and show them one by one
function fetchNotifications() {
    const today = new Date().toISOString().split('T')[0]; // Get today's date
    const lastShownDate = localStorage.getItem('notificationShownDate');

    if (lastShownDate === today) {
        console.log('Notifications already shown today.');
        return;
    }

    fetch('https://www.aiamma.in/admin/notifications.php')
        .then(response => response.json())
        .then(data => {
            let index = 0;

            function showNextNotification() {
                if (index < data.length) {
                    const { title, message, icon, date } = data[index];
                    createNotification(title, message, icon, date);
                    index++;
                    setTimeout(showNextNotification, 8000); // Show the next notification after 8 seconds
                } else {
                    localStorage.setItem('notificationShownDate', today); // Mark as shown for today
                }
            }

            showNextNotification();
        })
        .catch(error => console.error('Error fetching notifications:', error));
}
// Event listener for the button to fetch notifications
document.getElementById('fetchNotifications').addEventListener('click', function () {
    window.location.href = 'https://www.aiamma.in/notification.php'; // Redirect to the notification page
});
// Fetch notifications on page load
window.onload = fetchNotifications;
