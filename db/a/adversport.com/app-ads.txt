        document.addEventListener('DOMContentLoaded', function() {
            const burgerMenu = document.querySelector('.burger-menu');
            const mobileMenu = document.querySelector('.mobile-menu');
            const mockups = document.querySelectorAll('.mockup');
            const accordionToggles = document.querySelectorAll('.brand-card-header');
            
            // Ajout des contrôleurs de slider
            const prevButton = document.querySelector('.slider-control.prev');
            const nextButton = document.querySelector('.slider-control.next');
            const datesSlider = document.querySelector('.dates-slider');
            
            // Fonction pour faire défiler le slider
            function scrollSlider(direction) {
                // Augmenter la quantité de défilement pour montrer plus de contenu
                const scrollAmount = 756; // Quantité de défilement en pixels
                if (direction === 'left') {
                    datesSlider.scrollBy({
                        left: -scrollAmount,
                        behavior: 'smooth'
                    });
                } else {
                    datesSlider.scrollBy({
                        left: scrollAmount,
                        behavior: 'smooth'
                    });
                }
            }
            
            // Ajouter les écouteurs d'événements
            if (prevButton && nextButton && datesSlider) {
                prevButton.addEventListener('click', function() {
                    scrollSlider('left');
                });
                
                nextButton.addEventListener('click', function() {
                    scrollSlider('right');
                });
            }

            burgerMenu.addEventListener('click', function() {
                burgerMenu.classList.toggle('active');
                mobileMenu.classList.toggle('active');
            });

            // Effet de parallaxe au scroll
            window.addEventListener('scroll', function() {
                const scrolled = window.pageYOffset;
                mockups.forEach((mockup, index) => {
                    const speed = 0.05 * (index + 1);
                    const yPos = -(scrolled * speed);
                    mockup.style.transform = `translateY(${yPos}px) translateX(-50%)`;
                });
            });

            accordionToggles.forEach(toggle => {
                toggle.addEventListener('click', function() {
                    const card = this.closest('.brand-card');
                    const toggleIcon = this.querySelector('.toggle-icon');
                    const contentId = toggleIcon.getAttribute('aria-controls');
                    const content = document.getElementById(contentId);

                    if (!card || !content) return;

                    const isExpanded = toggleIcon.getAttribute('aria-expanded') === 'true';
                    console.log(isExpanded);
                    toggleIcon.setAttribute('aria-expanded', !isExpanded);
                    card.classList.toggle('active', !isExpanded);

                    if (!isExpanded) {
                        content.style.display = 'flex';
                        void content.offsetWidth;
                        const scrollHeight = content.scrollHeight;
                        content.style.maxHeight = scrollHeight + "px";
                        content.style.opacity = "1";
                        content.addEventListener('transitionend', function handler() {
                            if (card.classList.contains('active')) {
                                content.style.overflow = 'visible';
                            }
                            content.removeEventListener('transitionend', handler);
                        });
                    } else {
                        content.style.overflow = 'hidden';
                        content.style.maxHeight = "0";
                        content.style.opacity = "0";
                        content.addEventListener('transitionend', function handler() {
                            if (!card.classList.contains('active')) {
                                content.style.display = 'none';
                            }
                            content.removeEventListener('transitionend', handler);
                        });
                    }
                });
            });

            // --- Open first accordion item by default ---
            const firstAccordionCard = document.querySelector('.brands-accordion .brand-card');
            if (firstAccordionCard) {
                const firstToggle = firstAccordionCard.querySelector('.toggle-icon');
                const firstContentId = firstToggle.getAttribute('aria-controls');
                const firstContent = document.getElementById(firstContentId);

                if (firstToggle && firstContent) {
                    firstAccordionCard.classList.add('active');
                    firstToggle.setAttribute('aria-expanded', 'true');
                    // Make content visible before measuring scrollHeight
                    firstContent.style.display = 'flex';
                    firstContent.style.maxHeight = firstContent.scrollHeight + 'px';
                    firstContent.style.opacity = "1";
                    firstContent.style.overflow = 'visible'; // Ensure overflow is visible when opened by default
                     // No need to set icon opacity explicitly, CSS .active handles it
                }
            }
            // ---------------------------------------------

            const teamSlider = document.querySelector('.teams-slider');
            const teamPrevButton = document.querySelector('.teams .slider-control.prev');
            const teamNextButton = document.querySelector('.teams .slider-control.next');
            const teamScrollAmount = 756; // Adjust this value as needed

            if (teamPrevButton) {
                teamPrevButton.addEventListener('click', function() {
                    teamSlider.scrollBy({
                        left: -teamScrollAmount,
                        behavior: 'smooth'
                    });
                });
            }

            if (teamNextButton) {
                teamNextButton.addEventListener('click', function() {
                        teamSlider.scrollBy({
                            left: teamScrollAmount,
                        behavior: 'smooth'
                    });
                });
            }
        });
