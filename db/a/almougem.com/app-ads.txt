        // (اختياري) توليد Favicon بحرف "ع"
        function createDynamicFavicon() {
            const size = 64;
            const canvas = document.createElement('canvas');
            canvas.width = size; canvas.height = size;
            const ctx = canvas.getContext('2d');
            if (ctx.roundRect) { ctx.fillStyle='#a52a2a'; ctx.beginPath(); ctx.roundRect(0,0,size,size,8); ctx.fill(); }
            else { ctx.fillStyle='#a52a2a'; ctx.fillRect(0,0,size,size); }
            ctx.fillStyle = '#ffffff';
            ctx.font = '900 55px "Amiri","Times New Roman",serif';
            ctx.textAlign = 'center'; ctx.textBaseline = 'middle';
            ctx.fillText('ع', size/2, size/2 + 5);
            let link = document.querySelector("link[rel~='icon']");
            if (!link) { link = document.createElement('link'); link.rel = 'icon'; document.head.appendChild(link); }
            link.href = canvas.toDataURL('image/png');
        }

            const params = new URLSearchParams();
            if (mdatabase) params.set('mdatabase', mdatabase);
            if (query !== undefined) params.set('query', query);
            return '/search.php?' + params.toString();
        }

        // جلب النتائج جزئيًا وتحديث الصفحة بـ JSON (متوافق مع /search.php المعدّل)
        async function fetchAndRender(url, push = true) {
            try {
                const res = await fetch(url, {
                    headers: {
                        'X-Requested-With': 'XMLHttpRequest',
                        'Accept': 'application/json'
                    }
                });
                if (!res.ok) throw new Error('HTTP ' + res.status);

                const data = await res.json();

                // عنوان الصفحة
                if (data.title) {
                    document.title = data.title;
                    dynamicTitle = data.title; // تحديث قيمة العنوان الديناميكي
                }

                // <h1>
                const h1 = document.querySelector('h1.word-h');
                if (h1 && typeof data.h1_text === 'string') {
                    h1.textContent = data.h1_text;
                }

                // النتائج
                const contentBody = document.querySelector('#contentBody');
                if (contentBody && typeof data.content_html === 'string') {
                    contentBody.innerHTML = data.content_html;
                }

                // القاموس النشط (النص)
                const activeLexText = document.getElementById('activeLexText');
                if (activeLexText && typeof data.activeLexText === 'string') {
                    activeLexText.textContent = data.activeLexText;
                }

                // تفعيل التبويب بصريًا + تحديث الحقل الخفي
                if (data.mdatabase && allowedLex.includes(data.mdatabase)) {
                    setActiveTab(data.mdatabase);
                    const mdInput = document.getElementById('mdatabaseInput');
                    if (mdInput) mdInput.value = data.mdatabase;
                }

                // حدّث حقل البحث بالقيمة الجديدة
                const searchInput = document.getElementById('lexSearch');
                if (searchInput && typeof data.query === 'string') {
                    searchInput.value = data.query;
                }

                // إدارة العنوان/التاريخ
                if (push) {
                    const urlObj = new URL(url, location.origin);
                    history.pushState({ url: urlObj.pathname + '?' + urlObj.searchParams.toString() }, '', urlObj.pathname + '?' + urlObj.searchParams.toString());
                }
            } catch (err) {
                // في حال الفشل—إرسال عادي للنموذج (تجربة سليمة للمستخدم)
                console.error('Partial fetch failed:', err);
                const form = document.getElementById('searchForm');
                if (form) form.submit();
            }
        }

        function updateLex(val, submit=true){
            if (!allowedLex.includes(val)) return;
            document.getElementById('mdatabaseInput').value = val;
            setActiveTab(val);

            // طلب النتائج جزئيًا بدل إعادة تحميل
            if (submit) {
                const q = (document.getElementById('lexSearch')?.value ?? '').trim();
                const url = buildSearchUrl(q, val);
                fetchAndRender(url, true);
            }
        }

        // العنوان الديناميكي (من PHP) لكي نعيده عند focus
        let dynamicTitle = document.title;

        // تفاعل العنوان عند التركيز/الفقدان
        window.addEventListener('blur',  () => { document.title = "عد إلينا في المعجم.. 🔎"; });
        window.addEventListener('focus', () => { document.title = dynamicTitle; });

        // تحديث ارتفاع الهيدر الفعلي وحقنه في --header-height (لأجل padding-top في الجوال)
        function updateHeaderHeightVar() {
            const nav = document.querySelector('.navbar-full');
            if (nav) {
                const h = nav.offsetHeight;
                document.documentElement.style.setProperty('--header-height', h + 'px');
            }
        }

        // تهيئة
        window.addEventListener('load', ()=>{
            // createDynamicFavicon(); // فعّل إذا رغبت

            // حساب ارتفاع الهيدر (لمنع اختفاء المحتوى خلفه في الجوال)
            updateHeaderHeightVar();
            window.addEventListener('resize', updateHeaderHeightVar);

            const savedTheme = localStorage.getItem('userTheme');
            if(savedTheme) setTheme(savedTheme);

            const params = new URLSearchParams(location.search);
            const current = params.get('mdatabase') || 'smougem';

            // فعّل حالة التبويب الحالي
            setActiveTab(current);

            // فعل أزرار التبويب (ديسكتوب)
            document.querySelectorAll('.tab-btn').forEach(btn=>{
                btn.addEventListener('click', ()=> updateLex(btn.dataset.lex, true));
            });

            // فعل المنسدلة (جوال)
            const mobile = document.getElementById('mobileLex');
            if(mobile){
                mobile.value = current;
                mobile.addEventListener('change', e => updateLex(e.target.value, true));
            }

            // اعتراض إرسال نموذج البحث للقيام بجلب جزئي
            const form = document.getElementById('searchForm');
            if (form) {
                form.addEventListener('submit', (ev)=>{
                    ev.preventDefault();
                    const q = (document.getElementById('lexSearch')?.value ?? '').trim();
                    const md = (document.getElementById('mdatabaseInput')?.value ?? 'smougem').trim();
                    const url = buildSearchUrl(q, md);
                    fetchAndRender(url, true);
                });
            }

            // دعم الرجوع/التقدم في المتصفح
            window.addEventListener('popstate', (e)=>{
                const url = (e.state && e.state.url) ? e.state.url : location.href;
                fetchAndRender(url, false);
            });
        });

    const JSON_URL = 'quraan.json'; // مسار ملف الـ JSON الجديد
    const DISPLAY_DURATION = 15000;
    let records = [];
    let currentIndex = 0;

    async function initApp() {
        try {
            // جلب البيانات بصيغة JSON مباشرة دون الحاجة لمكتبات XLSX
            const response = await fetch(JSON_URL);
            if (!response.ok) throw new Error("فشل الوصول للملف");

            // خلط البيانات عشوائياً

            <div class="fade-in">
                <h2 style="margin:0">${item.word}</h2>
                <p style="color:gray">${item.mean}</p>
            </div>
        `;
        currentIndex++;
    }

    initApp();
