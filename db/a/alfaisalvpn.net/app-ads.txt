<script>
/* تحكم بالترحيبات */
const greetings = [
  "مرحبا", "Hello", "你好", "नमस्ते", "Hola", "Bonjour", "Здравствуйте", "こんにちは",
  "Guten Tag", "Olá", "স্বাগতম", "خوش آمدید", "Bienvenue", "హలో", "ہیلو", "Ciao",
  "שלום", "Γειά σου", "Xin chào", "Witam", "Hallo", "Selamat datang", "ਸਤ ਸ੍ਰੀ ਅਕਾਲ", "สวัสดี",
  "안녕하세요", "Merhaba", "Sawubona", "Habari", "ਸਤ ਸ੍ਰੀ ਅਕਾਲ", "Salam", "হ্যালো"
];

const container = document.querySelector('.greetings');
let idx = 0;
let autoPlay = true;
let animPaused = false;
let rotTimer = null;

function makeGreetNode(text){
  const el = document.createElement('div');
  el.className = 'greet';
  // we will use small font if long
  if (text.length > 12) el.classList.add('small');
  // typed span
  const span = document.createElement('span');
  span.className = 'typed';
  span.textContent = text;
  el.appendChild(span);
  return el;
}

function showGreeting(i){
  container.innerHTML = '';
  const el = makeGreetNode(greetings[i]);
  container.appendChild(el);
  // force reflow then show
  requestAnimationFrame(()=> {
    el.classList.add('show');
  });
}

// rotate
function nextGreeting(){
  idx = (idx + 1) % greetings.length;
  showGreeting(idx);
}

function prevGreeting(){
  idx = (idx - 1 + greetings.length) % greetings.length;
  showGreeting(idx);
}

// autoplay loop
function startRotate(){
  stopRotate();
  rotTimer = setInterval(()=> {
    if (!animPaused) nextGreeting();
  }, 3500);
}

function stopRotate(){
  if (rotTimer) { clearInterval(rotTimer); rotTimer = null; }
}

/* Clock: عرض حي بمنطقة Asia/Dubai */
function updateClock(){
  // use Intl to format in Asia/Dubai
  const now = new Date();
  const optsTime = { hour:'2-digit', minute:'2-digit', second:'2-digit', hour12:false, timeZone:'Asia/Dubai' };
  const optsDate = { weekday:'long', year:'numeric', month:'long', day:'numeric', timeZone:'Asia/Dubai' };
  const timeStr = new Intl.DateTimeFormat(undefined, optsTime).format(now);
  const dateStr = new Intl.DateTimeFormat(undefined, optsDate).format(now);
  document.getElementById('liveClock').textContent = timeStr;
  document.getElementById('liveDate').textContent = dateStr;
}

// wire buttons
document.getElementById('nextBtn').addEventListener('click', ()=> { nextGreeting(); resetTimer(); });
document.getElementById('pauseBtn').addEventListener('click', ()=>{
  animPaused = true;
  document.getElementById('pauseBtn').style.display = 'none';
  document.getElementById('playBtn').style.display = 'inline-block';
});
document.getElementById('playBtn').addEventListener('click', ()=>{
  animPaused = false;
  document.getElementById('pauseBtn').style.display = 'inline-block';
  document.getElementById('playBtn').style.display = 'none';
});

function resetTimer(){
  if (autoPlay) {
    stopRotate();
    startRotate();
  }
}

// init
showGreeting(idx);
startRotate();
setInterval(updateClock, 1000);
updateClock();

// keyboard navigation (optional)
document.addEventListener('keydown', (e)=>{
  if (e.key === 'ArrowRight') { nextGreeting(); resetTimer(); }
  if (e.key === 'ArrowLeft') { prevGreeting(); resetTimer(); }
});
</script>
