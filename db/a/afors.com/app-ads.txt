<script>
	document.addEventListener("DOMContentLoaded", () => {
		const searchBtn = document.getElementById("searchBtn");

		document.addEventListener("change", (e) => {
			const target = e.target;
			const val = target.value;

			switch (target.id) {
				case 'search-countries':
					searchBtn.setAttribute("data-country", val === 'aa' ? '' : val);
					break;
				case 'search-regions':
					searchBtn.setAttribute("data-region", val === 'aa' ? '' : val);
					break;
				case 'search-sections':
					searchBtn.setAttribute("data-section", val);
					break;
				case 'search-categories':
					searchBtn.setAttribute("data-category", val);
					break;
				case 'search-daysOld':
					searchBtn.setAttribute("data-daysold", val);
					break;
				case 'search-perPage':
					searchBtn.setAttribute("data-perpage", val);
					break;
				case 'search-sold':
					searchBtn.setAttribute("data-sold", target.checked ? '1' : '0');
					break;
				case 'search-archived':
					searchBtn.setAttribute("data-archived", target.checked ? '1' : '0');
					break;
				case 'search-expired':
					searchBtn.setAttribute("data-expired", target.checked ? '1' : '0');
					break;
			}
		});

		document.addEventListener("input", (e) => {
			const target = e.target;
			const val = target.value.trim();

			switch (target.id) {
				case 'search-manu':
					searchBtn.setAttribute("data-manu", val);
					break;
				case 'search-keyword':
					searchBtn.setAttribute("data-keyword", val);
					break;
				case 'search-minAge':
					searchBtn.setAttribute("data-minage", val);
					break;
				case 'search-maxAge':
					searchBtn.setAttribute("data-maxage", val);
					break;
				case 'search-minPrice':
					searchBtn.setAttribute("data-minprice", val);
					break;
				case 'search-maxPrice':
					searchBtn.setAttribute("data-maxprice", val);
					break;
			}
		});

		const el = {
			country: document.getElementById('search-countries'),
			region: document.getElementById('search-regions'),
			section: document.getElementById('search-sections'),
			category: document.getElementById('search-categories'),
		};

		window.searchSession = {
			region: '',
			category: ''
		};

		// ── Register cascade listeners FIRST ──

		if (el.country && el.region) {
			el.country.addEventListener('change', () => {
				el.region.length = 0;
				el.region.options.add(new Option('Loading regions...', ''));

				fetch('/get-region-options/?xhr=1', {
					method: 'POST',
					headers: { 'Content-Type': 'application/json', 'X-Requested-With': 'XMLHttpRequest' },
					body: JSON.stringify({ countryCode: el.country.value })
				})
						.then(res => res.json())
						.then(data => {
							el.region.length = 0;
							const regions = data.regions || [];
							el.region.options.add(
									new Option(regions.length
											? 'Select a region...'
											: 'No regions available', ''
									)
							);
							regions.forEach(r => el.region.options.add(new Option(
									r.name || r.Name,
									r.code || r.Code
							)));
							if (window.searchSession?.region) {
								el.region.value = window.searchSession.region;
								delete window.searchSession.region;
							}
						})
						.catch(err => console.error('Region Error:', err));
			});
		}

		if (el.section && el.category) {
			el.section.addEventListener('change', () => {
				el.category.length = 0;
				el.category.options.add(new Option('Loading categories...', ''));

				fetch('/get-category-options/?xhr=1', {
					method: 'POST',
					headers: { 'Content-Type': 'application/json', 'X-Requested-With': 'XMLHttpRequest' },
					body: JSON.stringify({ sectionId: el.section.value })
				})
						.then(res => res.json())
						.then(data => {
							el.category.length = 0;
							const categories = data.categories || [];
							el.category.options.add(
									new Option(categories.length
											? 'Select a category...'
											: 'No categories available', ''
									)
							);
							categories.forEach(c => el.category.options.add(new Option(c.name || c.Name, c.url || c.Url)));
							if (window.searchSession?.category) {
								el.category.value = window.searchSession.category;
								delete window.searchSession.category;
							}
						})
						.catch(err => console.error('Category Error:', err));
			});
		}

		// ── THEN trigger pre-population ──

		if (el.country && el.country.value) {
			el.country.dispatchEvent(new Event('change'));
		}
		if (el.section && el.section.value) {
			el.section.dispatchEvent(new Event('change'));
		}

		const searchForm = document.querySelector('.search-form-container');
		if (searchForm) {
			searchForm.addEventListener('keydown', (e) => {
				if (e.key === 'Enter') {
					e.preventDefault();
					searchBtn.click();
				}
			});
		}

		const clearBtn = document.getElementById('clearSearchBtn');
		if (clearBtn) {
			clearBtn.addEventListener('click', () => {
				// Reset dropdowns to first option
				['search-countries', 'search-sections', 'search-daysOld', 'search-perPage'].forEach(id => {
					const select = document.getElementById(id);
					if (select) select.selectedIndex = 0;
				});

				// Reset cascading dropdowns
				el.region.length = 0;
				el.region.options.add(new Option('All Regions', ''));
				el.category.length = 0;
				el.category.options.add(new Option('All Categories', ''));

				// Reset text and number inputs
				['search-manu', 'search-keyword', 'search-minAge', 'search-maxAge', 'search-minPrice', 'search-maxPrice'].forEach(id => {
					const input = document.getElementById(id);
					if (input) input.value = '';
				});

				// Reset check boxes
				['search-sold', 'search-archived', 'search-expired'].forEach(id => {
					const cb = document.getElementById(id);
					if (cb) cb.checked = false;
				});

				// Reset all data-attributes on the search button
				const defaults = {
					country: '', region: '', section: '', category: '',
					manu: '', keyword: '', minage: '', maxage: '',
					minprice: '', maxprice: '', daysold: '90', perpage: '25',
					sold: '0', archived: '0', expired: '0'
				};
				Object.entries(defaults).forEach(([key, val]) => {
					searchBtn.setAttribute('data-' + key, val);
				});
			});
		}

	});
