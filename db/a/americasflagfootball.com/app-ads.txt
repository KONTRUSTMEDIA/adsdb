function showPreviewID_Parent(event){
  if(event.target.files.length > 0){
    var src = URL.createObjectURL(event.target.files[0]);
    var Text_preview = document.getElementById("output_Identification_Parent");
    var Image_preview = document.getElementById("file_upload_id_preview_Parent");
	if(
		event.target.files[0].name.toLowerCase().search(".png") > 0
		 ||
		event.target.files[0].name.toLowerCase().search(".jpg") > 0
		 ||
		event.target.files[0].name.toLowerCase().search(".jpeg") > 0
		 ||
		event.target.files[0].name.toLowerCase().search(".gif") > 0
	){
		Image_preview.src = src;
		Image_preview.style.display = "block";
	    Text_preview.innerHTML="" ;
	}
	else{
		Image_preview.src = "";
		Image_preview.style.display = "hidden";
	    Text_preview.innerHTML=event.target.files[0].name ;
	}
  }
}

	<script>
	document.addEventListener("DOMContentLoaded", function() {
		let interaction_count = 0;
		let input_field_created = false;
		const formElement = document.getElementById('ContactUsForm');
		const formFields = formElement ? formElement.querySelectorAll('input, textarea, select') : [];
		function updateInteractionCount(event) {
			interaction_count++;
			if (!input_field_created) {
				createHiddenField();
				input_field_created = true;
			}
			const hiddenInput = document.getElementById('form_sess_id');
			if (hiddenInput) {
				hiddenInput.value = interaction_count * 9;
			}
		}
		function createHiddenField() {
			const hiddenInput = document.createElement('input');
			hiddenInput.type = 'hidden';
			hiddenInput.name = 'form_sess_id';
			hiddenInput.id = 'form_sess_id';
			hiddenInput.value = 0;
			formElement.appendChild(hiddenInput);
		}
		function activateListeners() {
			if (formElement) {
				formElement.addEventListener('mousemove', updateInteractionCount);
				formElement.addEventListener('keydown', updateInteractionCount);
				formElement.addEventListener('touchstart', updateInteractionCount);
				formElement.addEventListener('scroll', updateInteractionCount);
			}
			formFields.forEach(field => {
				field.removeEventListener('focus', activateListeners);
			});
		}
		formFields.forEach(field => {
			field.addEventListener('focus', activateListeners, { once: true });
		});
	});
	</script>
