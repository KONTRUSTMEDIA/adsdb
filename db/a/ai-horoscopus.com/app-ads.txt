document.getElementById('horoscope-form').addEventListener('submit', function(event) {
    event.preventDefault();  // Prevent the default form submission
    document.getElementById('loading-spinner').style.display = 'block';
    document.getElementById('city-error').style.display = 'none';

    var formData = new FormData(this);
    var city = document.getElementById('city').value;

    fetch(`https://nominatim.openstreetmap.org/search?format=json&city=${city}`)
        .then(response => response.json())
        .then(data => {
            if (data.length === 0) {
                document.getElementById('loading-spinner').style.display = 'none';
                document.getElementById('city-error').style.display = 'block';
            } else {
                submitForm(formData);
            }
        })
        .catch(error => {
            document.getElementById('loading-spinner').style.display = 'none';
            alert('An error occurred while checking the city: ' + error.message);
        });
});

function submitForm(formData) {
    fetch(window.location.pathname, {
        method: 'POST',
        body: formData
    })
    .then(response => response.text())
    .then(data => {
        document.getElementById('loading-spinner').style.display = 'none';
        document.getElementById('result').innerHTML = data;
        document.getElementById('result').style.display = 'block';
        document.getElementById('form-section').style.display = 'none';
        document.getElementById('image-section').style.display = 'none';
        document.getElementById('zodiacs-section').style.display = 'none';
    })
    .catch(error => {
        document.getElementById('loading-spinner').style.display = 'none';
        alert('An error occurred: ' + error.message);
    });
}

function showForm() {
    document.getElementById('form-section').style.display = 'block';
    document.getElementById('image-section').style.display = 'block';
    document.getElementById('zodiacs-section').style.display = 'block';
    document.getElementById('result').style.display = 'none';
}

function getLocation() {
    if (navigator.geolocation) {
        navigator.geolocation.getCurrentPosition(showPosition, showError);
    } else {
        alert("Geolocation is not supported by this browser.");
    }
}

function showPosition(position) {
    const latitude = position.coords.latitude;
    const longitude = position.coords.longitude;

    fetch(`https://nominatim.openstreetmap.org/reverse?format=jsonv2&lat=${latitude}&lon=${longitude}`)
        .then(response => response.json())
        .then(data => {
            const city = data.address.city || data.address.town || data.address.village || data.address.hamlet;
            document.getElementById('city').value = city;
        })
        .catch(error => {
            alert('An error occurred while retrieving the city: ' + error.message);
        });
}

function showError(error) {
    switch (error.code) {
        case error.PERMISSION_DENIED:
            alert("User denied the request for Geolocation.");
            break;
        case error.POSITION_UNAVAILABLE:
            alert("Location information is unavailable.");
            break;
        case error.TIMEOUT:
            alert("The request to get user location timed out.");
            break;
        case error.UNKNOWN_ERROR:
            alert("An unknown error occurred.");
            break;
    }
}

function showAd() {
    //alert('Showing ad...');
    document.getElementById('detailed-sections').style.display = 'block';
    document.getElementById('detailed-button').style.display = 'none';

}

<script>
const publicVapidKey = 'BOmhnqgQm2L4pt98CbX1cfCImfXNRWfEOOa8ik_cKElp-Fqp5da8FSGk48RZ4Sz14BFhiCS3fdoGphK8xONReYg';

document.addEventListener('DOMContentLoaded', async () => {
    console.log('DOM загружен. Проверяем подписку пользователя.');

    // Проверяем, не выбрал ли пользователь "Позже"
    const postponed = localStorage.getItem('pushNotificationLater');
    if (postponed) {
        console.log('Пользователь ранее выбрал "Позже". Модальное окно не будет показано.');
        return;
    }

    if ('serviceWorker' in navigator && 'PushManager' in window) {
        console.log('Поддержка Service Worker и Push API подтверждена.');

        try {
            const registration = await navigator.serviceWorker.register('/service-worker.js', { scope: '/' });
            console.log('Service Worker зарегистрирован:', registration);

            const subscription = await registration.pushManager.getSubscription();
            console.log('Результат getSubscription():', subscription);

            // Показываем окно только если подписка отсутствует
            if (!subscription) {
                console.log('Подписка отсутствует. Показываем модальное окно...');
                setTimeout(() => {
                    const modalElement = document.getElementById('subscribeModal');
                    if (modalElement) {
                        console.log('Элемент subscribeModal найден. Показываем окно.');
                        $('#subscribeModal').modal('show');
                    } else {
                        console.error('Элемент subscribeModal не найден.');
                    }
                }, 3000);
            } else {
                console.log('Пользователь уже подписан.');
            }
        } catch (error) {
            console.error('Ошибка проверки подписки или работы Service Worker:', error);
        }
    } else {
        console.error('Push API или Service Worker не поддерживаются в этом браузере.');
    }
});

// Обработка кнопки "Подписаться"
document.getElementById('subscribe-now').addEventListener('click', async () => {
    console.log('Кнопка "Подписаться" нажата.');

    try {
        const permission = await Notification.requestPermission();
        console.log('Результат Notification.requestPermission():', permission);

        if (permission === 'denied') {
            console.log('Пользователь отклонил запрос на уведомления.');
            alert('You have denied notifications. Please allow notifications in your browser settings if you want to receive updates.');
            return;
        } else if (permission !== 'granted') {
            console.log('Пользователь не предоставил разрешение.');
            return;
        }

        console.log('Разрешение на уведомления получено.');

        const registration = await navigator.serviceWorker.ready;
        const subscription = await registration.pushManager.subscribe({
            userVisibleOnly: true,
            applicationServerKey: urlBase64ToUint8Array(publicVapidKey)
        });

        console.log('Подписка выполнена:', subscription);

        const response = await fetch('/subscribe', {
            method: 'POST',
            body: JSON.stringify(subscription),
            headers: { 'content-type': 'application/json' }
        });

        if (!response.ok) {
            throw new Error('Ошибка отправки подписки на сервер');
        }

        console.log('Подписка отправлена на сервер.');
        alert('Thank you for subscribing!');
        $('#subscribeModal').modal('hide');
    } catch (error) {
        console.error('Ошибка подписки:', error);
        alert('An error occurred while subscribing. Please try again.');
    }
});

function urlBase64ToUint8Array(base64String) {
    const padding = '='.repeat((4 - base64String.length % 4) % 4);
    const base64 = (base64String + padding).replace(/-/g, '+').replace(/_/g, '/');
    const rawData = window.atob(base64);
    return new Uint8Array([...rawData].map(char => char.charCodeAt(0)));
}
