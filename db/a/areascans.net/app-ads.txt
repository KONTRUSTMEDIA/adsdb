document.addEventListener('DOMContentLoaded', function() {
    // --- SIDEBAR TOGGLE ---
    const toggle = document.getElementById('sidebarToggle');
    const close = document.getElementById('sidebarClose');
    const menu = document.getElementById('sidebarMenu');
    const backdrop = document.getElementById('sidebarBackdrop');

    function openMenu() {
        menu.classList.add('active');
        backdrop.classList.add('active');
        document.body.style.overflow = 'hidden'; // Prevent scrolling
    }

    function closeMenu() {
        menu.classList.remove('active');
        backdrop.classList.remove('active');
        document.body.style.overflow = '';
    }

    if(toggle) toggle.addEventListener('click', openMenu);
    if(close) close.addEventListener('click', closeMenu);
    if(backdrop) backdrop.addEventListener('click', closeMenu);

    // --- SMART HEADER HIDE/SHOW ON SCROLL ---
    var lastScrollTop = 0;
    var delta = 5;
    var navbarHeight = document.querySelector('header').offsetHeight;
    var header = document.querySelector('header');

    window.addEventListener('scroll', function() {
        var didScroll = true;
        
        // Use requestAnimationFrame for performance
        if (didScroll) {
             hasScrolled();
             didScroll = false;
        }
    });

    function hasScrolled() {
        var st = window.pageYOffset || document.documentElement.scrollTop;
        if(Math.abs(lastScrollTop - st) <= delta) return;
        
        if (st > lastScrollTop && st > navbarHeight){
            header.classList.add('nav-up');
        } else {
            if(st + window.innerHeight < document.documentElement.scrollHeight) {
                header.classList.remove('nav-up');
            }
        }
        lastScrollTop = st;
    }

    // --- MOBILE SEARCH OVERLAY ---
    const searchTrigger = document.querySelector('.search-trigger.mobile-only');
    const searchOverlay = document.querySelector('.mobile-search-overlay');
    const searchClose = document.querySelector('.mobile-search-close');

    if (searchTrigger && searchOverlay) {
        searchTrigger.addEventListener('click', function(e) {
            e.preventDefault();
            searchOverlay.classList.add('active');
        });
    }

    if (searchClose && searchOverlay) {
        searchClose.addEventListener('click', function() {
            searchOverlay.classList.remove('active');
        });
    }
});
</script>

<script>
jQuery(document).ready(function($) {
    var $btn = $('#totop');
    var lastScrollTop = 0;
    var delta = 5;

    $(window).on('scroll', function() {
        var st = $(this).scrollTop();
        
        // If simply at top, always hide
        if (st < 300) {
            $btn.removeClass('visible');
            lastScrollTop = st;
            return;
        }

        // Make sure they scrolled more than delta
        if(Math.abs(lastScrollTop - st) <= delta)
            return;
        
        if (st > lastScrollTop){
            // Scroll Down - Hide
            $btn.removeClass('visible');
        } else {
            // Scroll Up - Show
            $btn.addClass('visible');
        }
        lastScrollTop = st;
    });

    $btn.on('click', function(e) {
        e.preventDefault();
        $('html, body').animate({scrollTop:0}, 600);
    });
});
</script>

/* Dedicated Reader Scroll Logic */
jQuery(document).ready(function($) {
    var lastScroll = 0;
    var $readerBottom = $('.reader-bottom-nav, .readingnav');
    var $readerTop = $('.reader-top-bar');

    $(window).on('scroll', function() {
        var currentScroll = $(this).scrollTop();
        
        // Ignore small scroll movements
        if (Math.abs(lastScroll - currentScroll) <= 5) return;

        if (currentScroll > lastScroll && currentScroll > 100) {
            // Scroll Down -> Hide
            $readerBottom.addClass('bar-hidden');
            $readerTop.addClass('bar-hidden');
            
            // Legacy support
            if($readerBottom.hasClass('readingnav')) $readerBottom.fadeOut();

            // Scroll Up -> Show
            $readerBottom.removeClass('bar-hidden');
            $readerTop.removeClass('bar-hidden');
            
            // Legacy support
            if($readerBottom.hasClass('readingnav')) $readerBottom.fadeIn();
        }
        lastScroll = currentScroll;
    });
});
