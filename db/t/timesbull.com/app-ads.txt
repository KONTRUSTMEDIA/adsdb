            function updateScrollIndicators() {
                var $scrollContainer = $('.trending-scroll-container');
                var scrollLeft = $scrollContainer.scrollLeft();
                var scrollWidth = $scrollContainer[0].scrollWidth;
                var clientWidth = $scrollContainer.width();
                
                // Remove existing classes
                $scrollContainer.removeClass('has-left has-right');
                
                // Add classes based on scroll position
                if (scrollLeft > 10) {
                    $scrollContainer.addClass('has-left');
                }
                
                if (scrollLeft + clientWidth < scrollWidth - 10) {
                    $scrollContainer.addClass('has-right');
                }
            }
            
            // Initial check
            function checkTrendingScroll() {
                var $scrollContainer = $('.trending-scroll-container');
                var $scrollContent = $('.trending-scroll-content');
                var containerWidth = $scrollContainer.width();
                var contentWidth = $scrollContent.width();
                
                if (contentWidth > containerWidth) {
                    $scrollContainer.addClass('scrollable');
                    updateScrollIndicators();
                } else {
                    $scrollContainer.removeClass('scrollable has-left has-right');
                }
