    <!-- Script for page navigation -->
    <script>
        // Close menu is no longer needed - menu is always visible
        
        // Function to show the selected page
        function showPage(pageId) {
            // Hide all pages
            document.querySelectorAll('.page').forEach(page => {
                page.classList.remove('active');
            });
            
            // Show the selected page
            document.getElementById(pageId).classList.add('active');
            
            // Update active menu item
            document.querySelectorAll('.nav-menu a').forEach(link => {
                link.classList.remove('active');
            });
            
            // Find the clicked menu item and add active class
            document.querySelector(`.nav-menu a[href="#${pageId}"]`).classList.add('active');
            
            // Hide or show the top logo based on the page
            const topLogo = document.getElementById('top-logo');
            if (pageId === 'live-audio') {
                topLogo.classList.add('hidden');
            } else {
                topLogo.classList.remove('hidden');
            }
            
            // Update URL with hash
            history.pushState(null, null, `#${pageId}`);
        }
                      
        // Smart Banner Detection and Display
        document.addEventListener('DOMContentLoaded', function() {
            // Detect device type for hero image
            const ua = navigator.userAgent || navigator.vendor || window.opera;
            const heroImage = document.getElementById('hero-image');
            
            if (heroImage) {
                if (/android/i.test(ua)) {
                    // Android device
                    heroImage.src = 'https://filedn.com/lYdCEOaQt7RzelEFMlfs2dh/Website/android.png';
                    heroImage.alt = 'The Yellow Station on Android';
                } else if (/iPad|iPhone|iPod/.test(ua) && !window.MSStream) {
                    // iOS device
                    heroImage.src = 'https://filedn.com/lYdCEOaQt7RzelEFMlfs2dh/Website/iphone.png';
                    heroImage.alt = 'The Yellow Station on iPhone';
                }
                // Otherwise, keep the default phonesnew.png image
            }
            
            // Check if user has previously closed the banner
            const bannerClosed = localStorage.getItem('appBannerClosed');
            
            if (!bannerClosed) {
                // Only show Android custom banner
                // Remove iOS detection since Apple Smart Banner handles that
                if (/android/i.test(ua)) {
                    showBanner('android-banner');
                }
            }
            
            // Add event listeners for close buttons
            document.querySelectorAll('.app-banner-close').forEach(function(button) {
                button.addEventListener('click', function() {
                    const banner = this.closest('.app-banner');
                    hideBanner(banner.id);
                    
                    // Remember user preference
                    localStorage.setItem('appBannerClosed', 'true');
                });
            });
        });
        
        function showBanner(bannerId) {
            const banner = document.getElementById(bannerId);
            if (banner) {
                banner.style.display = 'flex';
                document.body.classList.add('has-app-banner');
            }
        }
        
        function hideBanner(bannerId) {
            const banner = document.getElementById(bannerId);
            if (banner) {
                banner.style.display = 'none';
                document.body.classList.remove('has-app-banner');
            }
        }
        
        // Event listener for when the page loads
        document.addEventListener('DOMContentLoaded', function() {
            // Check URL hash and show appropriate page
            const hash = window.location.hash;
            
            if (hash) {
                const pageId = hash.substring(1); // Remove the # symbol
                showPage(pageId);
            }
        });
        
        // Handle browser navigation (back/forward buttons)
        window.addEventListener('popstate', function() {
            const hash = window.location.hash;
            if (hash) {
                const pageId = hash.substring(1);
                showPage(pageId);
            } else {
                showPage('home');
            }
        });
    </script>
