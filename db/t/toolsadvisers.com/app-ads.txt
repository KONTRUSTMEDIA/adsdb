				localStorage.setItem(STORAGE_TIME_KEY, Date.now().toString());
			} catch(e) {}
		}

		function cameFromInternalLink() {
			try {
				const referrer = document.referrer;
				if (!referrer) return false;
				const currentHost = window.location.hostname;
				const referrerUrl = new URL(referrer);
				return referrerUrl.hostname === currentHost;
			} catch(e) {
				return false;
			}
		}

		// =====================================================
		// Geo Detection (Hybrid VPN Check)
		// =====================================================
		function getCloudflareCountry() {
			const cfMeta = document.querySelector('meta[name="cf-country"]');
			if (cfMeta) return cfMeta.getAttribute('content');
			return null;
		}

		function getBDTimezoneOffset() {
			const offset = new Date().getTimezoneOffset();
			return offset === -360;
		}

		function detectCountry() {
			const cfCountry = getCloudflareCountry();
			const isBDTime = getBDTimezoneOffset();

			if (cfCountry && ['US', 'GB', 'UK', 'CA'].includes(cfCountry.toUpperCase()) && isBDTime) {
				console.log('PBS: VPN Detected (IP: ' + cfCountry + ', Timezone: BD)');
				return 'BD';
			}

			if (cfCountry) return cfCountry.toUpperCase();
			if (isBDTime) return 'BD';

			return 'UNKNOWN';
		}

		function shouldBlockBD(country) {
			if (!CONFIG.blockBD) return false;
			if (country === 'BD') return true;
			if (country === 'UNKNOWN' && getBDTimezoneOffset()) return true;
			return false;
		}

		function isAllowedCountry(country) {
			const allowed = [];
			if (CONFIG.allowUSA) allowed.push('US');
			if (CONFIG.allowUK) allowed.push('GB', 'UK');
			if (CONFIG.allowCA) allowed.push('CA');
			if (allowed.length === 0) return true;
			return allowed.includes(country);
		}

		// =====================================================
		// Link Finders
		// =====================================================
		function getRandomInternalLink() {
			const articleSelectors = [
				'article .entry-content', 'article .post-content', '.entry-content', '.post-content', 
				'.content', 'main article', '.single-post article', '.article-content', 'body'
			];
			let articleBody = null;
			for (let selector of articleSelectors) {
				articleBody = document.querySelector(selector);
				if (articleBody) break;
			}
			if (!articleBody) articleBody = document.querySelector('article');
			if (!articleBody) return null;
			
			const links = Array.from(articleBody.querySelectorAll('a[href]')).filter(a => {
				const href = a.getAttribute('href');
				return href && !href.startsWith('#') && !href.startsWith('mailto:') && !href.startsWith('tel:') &&
				   (href.startsWith('/') || href.includes(window.location.hostname)) && href !== window.location.href;
			});
			if (links.length === 0) return null;
			return links[Math.floor(Math.random() * links.length)].href;
		}

		function getTargetLink() {
			if (CONFIG.customPopupLink) return CONFIG.customPopupLink;

			if (CONFIG.targetDomain) {
				const normalizedDomain = CONFIG.targetDomain.toLowerCase().replace(/^www\./, '');

				const links = Array.from(document.querySelectorAll('a[href]')).filter(link => {
					const href = link.href.toLowerCase();
					return href.includes(normalizedDomain) || 
						   (normalizedDomain.includes('amazon') && href.includes('amzn.to')) ||
						   (normalizedDomain.includes('amazon') && href.includes('/dp/')) ||
						   (normalizedDomain.includes('amazon') && href.includes('/gp/'));
				});

				if (links.length > 0) {
					if (CONFIG.targetLinkMode === 'random') {
						return links[Math.floor(Math.random() * links.length)].href;
					} else {
						return links[0].href;
					}
				}
			}
			
			const buyBtn = document.querySelector('.buy_btn');
			if (buyBtn && buyBtn.href) return buyBtn.href;
			
			const home = CONFIG.homeUrl.replace(/^https?:\/\//, '');
			const allLinks = document.querySelectorAll('a[href]');
			for (let j = 0; j < allLinks.length; j++) {
				const href = allLinks[j].href;
				if (!href || href.startsWith('javascript') || href.startsWith('mailto') || href.startsWith('tel')) continue;
				if (href.indexOf(home) === -1) return href;
			}
			return null;
		}

		// =====================================================
		// CORE LOGIC
		// =====================================================
		
		if (CONFIG.hideForLoggedIn && document.body.classList.contains('logged-in')) {
			console.log('PBS: User is logged in, popup suppressed');
			return;
		}

		const userCountry = detectCountry();
		
		if (shouldBlockBD(userCountry)) {
			console.log('PBS: Visitor blocked (Bangladesh/VPN). No elements will show.');
			return;
		}

		if (hasUserTakenAction()) {
			console.log('PBS: User already took action, showing float bar directly');
			const floatBarDirect = document.getElementById('page-bottom-notice');
			if (floatBarDirect) {
				setTimeout(() => {
					floatBarDirect.classList.add('is-visible');
				}, 3000);
			}
			return;
		}
		
		if (cameFromInternalLink() && CONFIG.internalBoost) {
			console.log('PBS: Came from internal link, popup suppressed');
			return;
		}

		let shouldShowPopup = true;
		let useInternalLink = false;
		let targetUrl = '';

		if (CONFIG.internalBoost) {
			if (isAllowedCountry(userCountry)) {
				useInternalLink = false;
				targetUrl = getTargetLink();
			} else {
				useInternalLink = true;
				const internalLink = getRandomInternalLink();
				if (internalLink) {
					targetUrl = internalLink;
				} else {
					shouldShowPopup = false;
				}
			}
		} else {
			useInternalLink = false;
			targetUrl = getTargetLink();
		}

		if (!shouldShowPopup || !targetUrl) return;

		// =====================================================
		// Tracking Helper
		// =====================================================
		let popupWasBlocked = false;

		function trackEvent(eventType) {
			// Analytics disabled হলে কোনো DB call হবে না
			if (!CONFIG.analyticsEnabled) return;
			const data = new URLSearchParams();
			data.append('action', 'pbs_track_event');
			data.append('event_type', eventType);
			data.append('country', userCountry);
			data.append('is_internal', useInternalLink ? '1' : '0');
			data.append('popup_blocked', popupWasBlocked ? '1' : '0');
			fetch(CONFIG.ajaxUrl, {
				method: 'POST',
				headers: { 'Content-Type': 'application/x-www-form-urlencoded' },
				body: data
			}).catch(err => console.error(err));
		}

		// =====================================================
		// Device Detection
		// =====================================================
		function isMobile() { 
			return /Android|webOS|iPhone|iPad|iPod|BlackBerry|IEMobile|Opera Mini/i.test(navigator.userAgent); 
		}
		
		function isIOS() { 
			return /iPhone|iPad|iPod/i.test(navigator.userAgent) ||
				   (navigator.platform === 'MacIntel' && navigator.maxTouchPoints > 1);
		}

		function isSafari() { 
			const ua = navigator.userAgent.toLowerCase();
			return ua.includes('safari') && !ua.includes('chrome') && !ua.includes('android');
		}

		// =====================================================
		// Tab-Under & New Tab Actions
		// =====================================================

		// TAB-UNDER:
		// → Current article URL → new (background) tab এ খোলে
		// → Affiliate/target URL → current tab এ load হয়
		function doTabUnder(affiliateUrl) {
			const currentPageUrl = window.location.href;
			try { window.open(currentPageUrl, '_blank'); } catch(e) {}
			try { window.location.href = affiliateUrl; } catch(e) {}
		}

		// NEW TAB:
		// → Affiliate/target URL → new tab এ খোলে
		// → Current tab যেখানে আছে সেখানেই থাকে
		function doNewTab(affiliateUrl) {
			const newWindow = window.open(affiliateUrl, '_blank');
			if (!newWindow || newWindow.closed || typeof newWindow.closed === 'undefined') {
				// Popup blocked হলে fallback: current tab এ redirect
				popupWasBlocked = true;
				console.log('PBS: New tab blocked. Fallback to current tab redirect.');
				window.location.href = affiliateUrl;
			}
		}

		function determineCloseBehavior() {
			if (isIOS() && isSafari()) {
				if (CONFIG.iosBehavior === 'global') return CONFIG.closeBehavior;
				return CONFIG.iosBehavior;
			}
			return CONFIG.closeBehavior;
		}

		// =====================================================
		// DOM & Actions
		// =====================================================
		const overlay = document.getElementById('site-page-layer');
		const popup = document.getElementById('site-interaction-wrapper');
		const closeBtn = document.getElementById('action-close-trigger');
		const btn1 = document.getElementById('btn-action-primary');
		const btn2 = document.getElementById('btn-action-secondary');
		const floatBar = document.getElementById('page-bottom-notice');
		const floatCloseBtn = document.getElementById('notice-close-trigger');
		const floatActionBtn = document.getElementById('notice-action-btn');

		if (!popup) return;

		if (CONFIG.autoTagEnabled && CONFIG.targetDomain && !useInternalLink) {
			document.querySelectorAll('a[href*="' + CONFIG.targetDomain + '"]').forEach(link => {
				link.classList.add('buy_btn');
			});
		}

		let popupShown = false;
		function showPopup() {
			if (popupShown) return;
			popupShown = true;
			overlay.style.display = 'block';
			setTimeout(() => {
				popup.classList.add('active-state');
				document.body.classList.add('interaction-active');
			}, 50);
			trackEvent('popup_shown');
		}

		function hidePopup() {
			popup.classList.remove('active-state');
			setTimeout(() => {
				overlay.style.display = 'none';
				document.body.classList.remove('interaction-active');

				if (floatBar) {
					setTimeout(() => {
						floatBar.classList.add('is-visible');
						trackEvent('float_bar_shown');
					}, 3000);
				}
			}, 300);
		}

		const handleAction = (type) => {
			trackEvent(type);
			markActionTaken();
			hidePopup();
			
			const behavior = determineCloseBehavior();
			if (behavior === 'new_tab') {
				doNewTab(targetUrl);
			} else {
				doTabUnder(targetUrl);
			}
		};

		btn1.addEventListener('click', () => handleAction('popup_primary_click'));
		btn2.addEventListener('click', () => handleAction('popup_secondary_click'));
		closeBtn.addEventListener('click', () => handleAction('popup_close_click'));

		if (CONFIG.overlayClick !== 'disabled') {
			overlay.addEventListener('click', (e) => {
				if (e.target !== overlay) return;
				const shouldTrigger = CONFIG.overlayClick === 'all' ||
					(CONFIG.overlayClick === 'desktop_only' && !isMobile()) ||
					(CONFIG.overlayClick === 'mobile_only' && isMobile());
				if (shouldTrigger) handleAction('overlay_click');
			});
		}

		if (floatBar) {
			const floatUrl = CONFIG.customFloatLink || targetUrl;
			floatActionBtn.addEventListener('click', () => {
				trackEvent('float_bar_click');
				window.open(floatUrl, '_blank');
			});
			floatCloseBtn.addEventListener('click', () => {
				trackEvent('float_bar_close');
				floatBar.classList.remove('is-visible');
			});
		}

		let delay = CONFIG.fixedDelay;
		if (CONFIG.delayMode === 'random_delay') {
			delay = Math.floor(Math.random() * (CONFIG.randomMax - CONFIG.randomMin + 1)) + CONFIG.randomMin;
		}
		setTimeout(showPopup, delay * 1000);

	})();
	</script>
	</body>
