    }
    
    function getSelectedRadioValueEn(name) {
        const radios = formEn.querySelectorAll(`input[name="${name}"]:checked`); // Scope to the form
        if (radios.length > 0) {
            return radios[0].value;
        }
        return 'general'; // Default
    }

    function calculateTaxLiveEn() {
        if (!formEn || !liveResultsDivEn) return; // Ensure elements exist

        const basicSalary = parseFloat(document.getElementById('basicSalaryEn').value) || 0;
        const houseRent = parseFloat(document.getElementById('houseRentEn').value) || 0;
        const conveyance = parseFloat(document.getElementById('conveyanceEn').value) || 0;
        const medical = parseFloat(document.getElementById('medicalEn').value) || 0;
        const bonuses = parseFloat(document.getElementById('bonusesEn').value) || 0;
        const overtime = parseFloat(document.getElementById('overtimeEn').value) || 0;
        const otherIncome = parseFloat(document.getElementById('otherIncomeEn').value) || 0;
        const advanceTax = parseFloat(document.getElementById('advanceTaxEn').value) || 0;

        const shanchayPatra = parseFloat(document.getElementById('shanchayPatraEn').value) || 0;
        const dps = parseFloat(document.getElementById('dpsEn').value) || 0;
        const mutualFund = parseFloat(document.getElementById('mutualFundEn').value) || 0;
        const treasuryBond = parseFloat(document.getElementById('treasuryBondEn').value) || 0;
        const stock = parseFloat(document.getElementById('stockEn').value) || 0;
        const providentFundEmployee = parseFloat(document.getElementById('providentFundEmployeeEn').value) || 0;
        const providentFundEmployer = parseFloat(document.getElementById('providentFundEmployerEn').value) || 0;
        const providentFundInterest = parseFloat(document.getElementById('providentFundInterestEn').value) || 0;

        const category = getSelectedRadioValueEn('category');
        const children = parseInt(document.getElementById('childrenEn').value) || 0;

        const totalEarnings = basicSalary + houseRent + conveyance + medical + bonuses + overtime + otherIncome;
        const taxfreeincome = Math.min(totalEarnings / 3, 450000);
        
        let categoryTaxFreeLimit = 350000;
        switch (category) {
            case "woman": case "senior": categoryTaxFreeLimit = 400000; break;
            case "disabled": case "thirdGender": categoryTaxFreeLimit = 475000; break;
            case "freedomFighter": categoryTaxFreeLimit = 500000; break;
        }
        categoryTaxFreeLimit += children * 50000;

        const taxableIncome = Math.max(0, totalEarnings - taxfreeincome);
        const incomeForSlabCalculation = Math.max(0, taxableIncome - categoryTaxFreeLimit);
        
        let tax = 0;
        let remainingTaxable = incomeForSlabCalculation;

        if (remainingTaxable > 0) { const slab = Math.min(remainingTaxable, 100000); tax += slab * 0.05; remainingTaxable -= slab; }
        if (remainingTaxable > 0) { const slab = Math.min(remainingTaxable, 400000); tax += slab * 0.10; remainingTaxable -= slab; }
        if (remainingTaxable > 0) { const slab = Math.min(remainingTaxable, 500000); tax += slab * 0.15; remainingTaxable -= slab; }
        if (remainingTaxable > 0) { const slab = Math.min(remainingTaxable, 500000); tax += slab * 0.20; remainingTaxable -= slab; }
        if (remainingTaxable > 0) { const slab = Math.min(remainingTaxable, 2000000); tax += slab * 0.25; remainingTaxable -= slab; }
        if (remainingTaxable > 0) { tax += remainingTaxable * 0.30; }

        const totalInvestment = Math.min(shanchayPatra, 500000) + Math.min(dps, 120000) + mutualFund + treasuryBond + stock + providentFundEmployee + providentFundEmployer + providentFundInterest;
        const rebate = Math.min(totalInvestment * 0.15, taxableIncome * 0.03, 1000000);
        let netTax = Math.max(0, tax - rebate - advanceTax);
        
        const minimumTax = 3000;
        if (taxableIncome > categoryTaxFreeLimit) {
            netTax = Math.max(netTax, minimumTax);
        } else {
            netTax = 0;
        }

        liveResultsDivEn.innerHTML = `
            <h2 class="text-2xl md:text-3xl font-bold text-blue-700 mb-8 text-center">Live Results</h2>
            <div class="space-y-3 text-sm md:text-base">
                <div class="flex justify-between items-center py-2 border-b border-gray-200">
                    <span class="font-medium text-gray-600">Total Earnings:</span>
                    <span class="font-semibold text-blue-600">BDT ${totalEarnings.toLocaleString('en-IN', { minimumFractionDigits: 2, maximumFractionDigits: 2 })}</span>
                </div>
                <div class="flex justify-between items-center py-2 border-b border-gray-200">
                    <span class="font-medium text-gray-600">Tax-Free Income (General Exemption):</span>
                    <span class="font-semibold text-blue-600">BDT ${taxfreeincome.toLocaleString('en-IN', { minimumFractionDigits: 2, maximumFractionDigits: 2 })}</span>
                </div>
                <div class="flex justify-between items-center py-2 border-b border-gray-200">
                    <span class="font-medium text-gray-600">Net Taxable Income (After General Exemption):</span>
                    <span class="font-semibold text-blue-600">BDT ${taxableIncome.toLocaleString('en-IN', { minimumFractionDigits: 2, maximumFractionDigits: 2 })}</span>
                </div>
                <div class="flex justify-between items-center py-2 border-b border-gray-200">
                    <span class="font-medium text-gray-600">0% Tax Limit (Category Based):</span>
                    <span class="font-semibold text-blue-600">BDT ${categoryTaxFreeLimit.toLocaleString('en-IN', { minimumFractionDigits: 2, maximumFractionDigits: 2 })}</span>
                </div>
                <div class="flex justify-between items-center py-2 border-b border-gray-200">
                    <span class="font-medium text-gray-600">Income Subject to Slab Calculation:</span>
                    <span class="font-semibold text-blue-600">BDT ${incomeForSlabCalculation.toLocaleString('en-IN', { minimumFractionDigits: 2, maximumFractionDigits: 2 })}</span>
                </div>
                <div class="flex justify-between items-center py-2 border-b border-gray-200">
                    <span class="font-medium text-gray-600">Total Tax Before Rebate:</span>
                    <span class="font-semibold text-blue-600">BDT ${tax.toLocaleString('en-IN', { minimumFractionDigits: 2, maximumFractionDigits: 2 })}</span>
                </div>
                <div class="flex justify-between items-center py-2 border-b border-gray-200">
                    <span class="font-medium text-gray-600">Eligible Investment for Rebate:</span>
                    <span class="font-semibold text-blue-600">BDT ${totalInvestment.toLocaleString('en-IN', { minimumFractionDigits: 2, maximumFractionDigits: 2 })}</span>
                </div>
                <div class="flex justify-between items-center py-2 border-b border-gray-200">
                    <span class="font-medium text-gray-600">Calculated Tax Rebate:</span>
                    <span class="font-semibold text-blue-600">BDT ${rebate.toLocaleString('en-IN', { minimumFractionDigits: 2, maximumFractionDigits: 2 })}</span>
                </div>
                <div class="flex justify-between items-center py-2 border-b border-gray-200">
                    <span class="font-medium text-gray-600">Advance Income Tax Paid (AIT):</span>
                    <span class="font-semibold text-blue-600">BDT ${advanceTax.toLocaleString('en-IN', { minimumFractionDigits: 2, maximumFractionDigits: 2 })}</span>
                </div>
                <div class="flex justify-between items-center py-3 mt-4 border-t-2 border-blue-500">
                    <h3 class="text-lg md:text-xl font-bold text-red-600">Net Tax Payable:</h3>
                    <span class="text-lg md:text-xl font-bold text-red-600">BDT ${netTax.toLocaleString('en-IN', { minimumFractionDigits: 2, maximumFractionDigits: 2 })}</span>
                </div>
            </div>`;
    }
});
