function adjustCardHeights() {
  if (window.innerWidth >= 992) { // uniquement desktop (≥992px)
    document.querySelectorAll('.row').forEach(row => {
      const cards = row.querySelectorAll('.card');
      let maxHeight = 0;

      cards.forEach(card => {
        card.style.height = 'auto'; // Réinitialise la hauteur
        maxHeight = Math.max(maxHeight, card.offsetHeight);
      });

      cards.forEach(card => {
        card.style.height = `${maxHeight}px`;
      });
    });
  } else {
    // Sur mobile/tablette : reset les hauteurs
    document.querySelectorAll('.row .card').forEach(card => {
      card.style.height = 'auto';
    });
  }
}

// Exécution initiale + sur redimensionnement
window.addEventListener('load', adjustCardHeights);
window.addEventListener('resize', debounce(adjustCardHeights, 100));

// Helper pour limiter les appels lors du resize
function debounce(fn, delay) {
  let timer;
  return () => {
    clearTimeout(timer);
    timer = setTimeout(fn, delay);
  };
}
</script>
