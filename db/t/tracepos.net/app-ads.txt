                    icon.classList.add('fa-chevron-up');
                }
            });
        });
        
        // Pricing toggle functionality
        const monthlyButton = document.getElementById('monthly-pricing-button');
        const annualButton = document.getElementById('annual-pricing-button');
        
        monthlyButton.addEventListener('click', () => {
            monthlyButton.classList.add('bg-primary', 'text-white');
            annualButton.classList.remove('bg-primary', 'text-white');
            
            // Update pricing displays
            document.querySelectorAll('[data-monthly]').forEach(element => {
                element.textContent = element.getAttribute('data-monthly');
            });
        });
        
        annualButton.addEventListener('click', () => {
            annualButton.classList.add('bg-primary', 'text-white');
            monthlyButton.classList.remove('bg-primary', 'text-white');
            
            // Update pricing displays
            document.querySelectorAll('[data-annual]').forEach(element => {
                element.textContent = element.getAttribute('data-annual');
            });
        });
        
        // Animation on scroll
        const animateOnScroll = () => {
            const elements = document.querySelectorAll('.slide-up, .fade-in');
            
            elements.forEach(element => {
                const elementPosition = element.getBoundingClientRect().top;
                const windowHeight = window.innerHeight;
                
                if (elementPosition < windowHeight - 100) {
                    element.style.opacity = '1';
                    element.style.transform = 'translateY(0)';
                }
            });
        };
        
        window.addEventListener('scroll', animateOnScroll);
        window.addEventListener('load', animateOnScroll);
    </script>
