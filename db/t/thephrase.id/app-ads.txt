    function asteriskCard(string)
        {
        var lastfour = string.length > 6 ? string.substr(string.length - (string.length - 7)) : "";
        var firstfour = string.substr(0, 3);
        var dots = '*****'.substring(0,string.length - 3);
        var total = (firstfour) + (dots.length > 0 ? ' ' + dots + ' ' : dots) + (lastfour);
        return total;
    }

    function LoadFooter() 
    {
        const newsSection = document.querySelector('#site__footer');
        const observer = new IntersectionObserver(entries => {
        if (entries[0].isIntersecting) 
        {
            let dS = 'page=footer';
            $.ajax({type:"GET",url: endpoint+"pages",dataType: 'json',data:dS,
                    success: function(res){$.each(res.data, function(k){$("#"+res.data[k].section).html(res.data[k].content);});},
                    error: function(xhr,ajaxOptions,thrownError){},
                    complete: function(xhr,status){}
                    });
                    observer.disconnect(); // stop observing once the AJAX call is made
            }
        });
        observer.observe(newsSection);
    }

    // Ads impression tracking
    function adsGetCookie(name) {
        var match = document.cookie.match(new RegExp('(^| )' + name + '=([^;]+)'));
        return match ? match[2] : null;
    }

    function adsSetCookie(name, value, days) {
        var expires = '';
        if (days) {
            var date = new Date();
            date.setTime(date.getTime() + (days*24*60*60*1000));
            expires = '; expires=' + date.toUTCString();
        }
        document.cookie = name + '=' + value + expires + '; path=/';
    }

    function adsGenerateId() {
        return 'av_' + Math.random().toString(36).slice(2) + Date.now().toString(36);
    }

    function adsGetVisitorId() {
        var id = null;
        try {
            id = localStorage.getItem('ads_visitor_id');
            if (!id) {
                id = adsGenerateId();
                localStorage.setItem('ads_visitor_id', id);
            }
        } catch (e) {
            id = adsGetCookie('ads_visitor_id');
            if (!id) {
                id = adsGenerateId();
                adsSetCookie('ads_visitor_id', id, 365);
            }
        }
        return id;
    }


    function adsGetOSBrowser() {
        var ua = (navigator.userAgent || '').toLowerCase();
        var os = 'Other';
        if (ua.indexOf('windows') !== -1) { os = 'Windows'; }
        else if (ua.indexOf('android') !== -1) { os = 'Android'; }
        else if (ua.indexOf('iphone') !== -1 || ua.indexOf('ipad') !== -1 || ua.indexOf('ipod') !== -1) { os = 'iOS'; }
        else if (ua.indexOf('mac os') !== -1 || ua.indexOf('macintosh') !== -1) { os = 'MacOS'; }
        else if (ua.indexOf('linux') !== -1) { os = 'Linux'; }

        var browser = 'Other';
        if (ua.indexOf('edg/') !== -1 || ua.indexOf('edge') !== -1) { browser = 'Edge'; }
        else if (ua.indexOf('chrome') !== -1 && ua.indexOf('safari') !== -1) { browser = 'Chrome'; }
        else if (ua.indexOf('safari') !== -1 && ua.indexOf('chrome') === -1) { browser = 'Safari'; }
        else if (ua.indexOf('firefox') !== -1) { browser = 'Firefox'; }
        else if (ua.indexOf('opera') !== -1 || ua.indexOf('opr/') !== -1) { browser = 'Opera'; }

        return { os: os, browser: browser };
    }

    function adsBuildPayload(slideId, extra) {
        var payload = {
            slide_id: slideId,
            visitor_id: adsGetVisitorId(),
            page_url: window.location.href,
            referrer: document.referrer || '',
            timezone: (window.Intl && Intl.DateTimeFormat) ? Intl.DateTimeFormat().resolvedOptions().timeZone : '',
            language: navigator.language || '',
            screen: (window.screen ? (screen.width + 'x' + screen.height) : ''),
            viewport: (window.innerWidth + 'x' + window.innerHeight),
            device: (window.innerWidth <= 768 ? 'mobile' : 'desktop'),
            os: adsGetOSBrowser().os,
            browser: adsGetOSBrowser().browser
        };
        if (extra) {
            Object.keys(extra).forEach(function(key) {
                if (extra[key] !== undefined && extra[key] !== null && extra[key] !== '') {
                    payload[key] = extra[key];
                }
            });
        }
        return payload;
    }

    function logAdsImpression(slideId, extra) {
        if (!slideId || !endpoint) { return; }
        var payload = adsBuildPayload(slideId, extra);
        $.ajax({
            type: 'POST',
            url: endpoint + 'log_ads_impression',
            dataType: 'json',
            data: payload
        });
    }

    function logAdsImpressionsOnPage(context) {
        var scope = context || document;
        $(scope).find('[data-ads-id]').each(function() {
            var $el = $(this);
            if ($el.data('ads-logged')) { return; }
            var slideId = parseInt($el.attr('data-ads-id'), 10);
            if (!slideId) { return; }
            $el.data('ads-logged', 1);
            logAdsImpression(slideId, {
                ads_format: $el.attr('data-ads-format') || '',
                for_page: $el.attr('data-ads-page') || ''
            });
        });
    }

    $(document).ready(function() {
        logAdsImpressionsOnPage();
    });
