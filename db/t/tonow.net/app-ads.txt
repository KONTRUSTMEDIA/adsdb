//button animation
function createButton(buttonElement) {
  const el = gsap.utils.selector(buttonElement);

  const DOM = {
    button: buttonElement,
    flair: el(".button-flair")
  };

  const xSet = gsap.quickSetter(DOM.flair, "xPercent");
  const ySet = gsap.quickSetter(DOM.flair, "yPercent");

  function getXY(e) {
    const { left, top, width, height } = DOM.button.getBoundingClientRect();

    const xTransformer = gsap.utils.pipe(
      gsap.utils.mapRange(0, width, 0, 100),
      gsap.utils.clamp(0, 100)
    );

    const yTransformer = gsap.utils.pipe(
      gsap.utils.mapRange(0, height, 0, 100),
      gsap.utils.clamp(0, 100)
    );

    return {
      x: xTransformer(e.clientX - left),
      y: yTransformer(e.clientY - top)
    };
  }

  function onMouseEnter(e) {
    const { x, y } = getXY(e);
    xSet(x);
    ySet(y);

    gsap.to(DOM.flair, {
      scale: 1,
      duration: 0.4,
      ease: "power2.out"
    });
  }

  function onMouseLeave(e) {
    const { x, y } = getXY(e);
    gsap.killTweensOf(DOM.flair);

    gsap.to(DOM.flair, {
      xPercent: x > 90 ? x + 20 : x < 10 ? x - 20 : x,
      yPercent: y > 90 ? y + 20 : y < 10 ? y - 20 : y,
      scale: 0,
      duration: 0.3,
      ease: "power2.out"
    });
  }

  function onMouseMove(e) {
    const { x, y } = getXY(e);

    gsap.to(DOM.flair, {
      xPercent: x,
      yPercent: y,
      duration: 0.4,
      ease: "power2"
    });
  }

  // Attach events
  DOM.button.addEventListener("mouseenter", onMouseEnter);
  DOM.button.addEventListener("mouseleave", onMouseLeave);
  DOM.button.addEventListener("mousemove", onMouseMove);
}

// Apply to all matching buttons
document.querySelectorAll('[data-block="button"]').forEach(createButton);
