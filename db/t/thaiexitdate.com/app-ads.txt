  <script>
    function calculateExit() {
      const arrivalInput = document.getElementById("arrivalDate").value;
      const stayDays = parseInt(document.getElementById("stayLength").value);
      const resultDiv = document.getElementById("result");

      if (!arrivalInput) {
        resultDiv.innerHTML = '<p class="error">⚠ Please select an arrival date.</p>';
        resultDiv.style.display = 'block';
        return;
      }

      const arrivalDate = new Date(arrivalInput);
      const exitDate = new Date(arrivalDate);
      exitDate.setDate(exitDate.getDate() + stayDays - 1);

      const options = { weekday: 'long', year: 'numeric', month: 'long', day: 'numeric' };
      const formattedDate = exitDate.toLocaleDateString('en-US', options);

      resultDiv.innerHTML = `
        <div class="success">
          <p><strong>✔ Your Last Legal Day in Thailand:</strong></p>
          <p class="exit-date">${formattedDate}</p>
          <p class="exit-date-iso">${exitDate.toISOString().split("T")[0]}</p>
          <p class="tip">Book your departure flight for this date or earlier to avoid overstay penalties.</p>
        </div>
      `;
      resultDiv.style.display = 'block';
      
      // Smooth scroll to result on mobile
      if (window.innerWidth < 768) {
        resultDiv.scrollIntoView({ behavior: 'smooth', block: 'nearest' });
      }
    }

    // Set today's date as default
    document.addEventListener('DOMContentLoaded', function() {
      const today = new Date().toISOString().split('T')[0];
      document.getElementById('arrivalDate').value = today;
    });
  </script>
