        // Sound effect generators
        function generateChime(isYes) {
            if (!audioContext) return;

            const oscillator = audioContext.createOscillator();
            const gainNode = audioContext.createGain();

            oscillator.connect(gainNode);
            gainNode.connect(audioContext.destination);

            if (isYes) {
                // Positive sound
                oscillator.frequency.setValueAtTime(440, audioContext.currentTime); // A4 note
                oscillator.frequency.linearRampToValueAtTime(880, audioContext.currentTime + 0.2); // A5 note
                gainNode.gain.setValueAtTime(0.3, audioContext.currentTime);
                gainNode.gain.linearRampToValueAtTime(0, audioContext.currentTime + 0.3);
                oscillator.start();
                oscillator.stop(audioContext.currentTime + 0.3);

                { pitch: 1.2, rate: 1.2 }    // Excited
            ],
            no: [
                { pitch: 0.9, rate: 0.9 },   // Firm
                { pitch: 0.8, rate: 1 },     // Deep
                { pitch: 1, rate: 0.8 },     // Calm
                { pitch: 0.9, rate: 1.1 }    // Clear
            ]
        };
        class DecisionCounter {
            constructor() {
                this.baseCount = 248;
                this.currentCount = this.loadCount();
                this.lastResetDate = this.loadLastResetDate();

                this.checkReset();
                setInterval(() => this.checkReset(), 24 * 60 * 60 * 1000);
            }

            loadCount() {
                return parseInt(localStorage.getItem('decisionCount')) || this.baseCount;
            }

            loadLastResetDate() {
                const savedDate = localStorage.getItem('lastResetDate');
                return savedDate ? new Date(savedDate) : new Date();
            }

            saveState() {
                localStorage.setItem('decisionCount', this.currentCount);
                localStorage.setItem('lastResetDate', this.lastResetDate.toISOString());
            }

            increment() {
                this.currentCount++;
                this.updateDisplay();
                this.saveState();
            }

            checkReset() {
                const now = new Date();
                const daysSinceReset = (now - this.lastResetDate) / (1000 * 60 * 60 * 24);

                if (daysSinceReset >= 30) {
                    this.currentCount = this.baseCount;
                    this.lastResetDate = now;
                    this.updateDisplay();
                    this.saveState();
                }
            }

            updateDisplay() {
                const counterElement = document.querySelector('.counter-number');
                counterElement.textContent = this.currentCount;
            }
        }

        const decisionCounter = new DecisionCounter();

        function showResult(decision) {
            // Background effect
            const backgroundEffect = document.querySelector('.background-effect');
            backgroundEffect.className = 'background-effect';
            backgroundEffect.classList.add(decision ? 'yes' : 'no');
            backgroundEffect.classList.add('active');

            // Create and show result panel
            const existingPanel = document.querySelector('.result-panel');
            if (existingPanel) {

            }

            resultPanel.className = 'result-panel';
            resultPanel.innerHTML = `
                <div class="result-content ${decision ? 'yes-result' : 'no-result'}">
                    ${decision ? 'YES' : 'NO'}
                </div>
            `;
            document.body.appendChild(resultPanel);

            // Add this right after creating the result panel
            const combo = decision 
                ? voiceCombinations.yes[Math.floor(Math.random() * voiceCombinations.yes.length)]
                : voiceCombinations.no[Math.floor(Math.random() * voiceCombinations.no.length)];

            generateChime(decision);
            setTimeout(() => {
                speakResult(decision ? 'YES' : 'NO', combo.pitch, combo.rate);
            }, 300);

            // Trigger animations
            setTimeout(() => {
                resultPanel.classList.add('active');
            }, 100);

                resultPanel.classList.remove('active');
                backgroundEffect.classList.remove('active');
                setTimeout(() => {
                    resultPanel.remove();
                }, 500);
            }, 3000);

            // Clear question input
            document.querySelector('.question-input').value = '';

            // Increment decision counter
            decisionCounter.increment();
        }
