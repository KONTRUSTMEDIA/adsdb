        $('.social-icon.search').click(function(e) {
            e.stopPropagation(); // ngăn click lan ra ngoài
            $('.header-search').fadeToggle();
        });

        // Click ra ngoài vùng header-search => ẩn
        $(document).click(function(e) {
            if (!$(e.target).closest('.header-search').length &&
                !$(e.target).closest('.social-icon.search').length) {
                $('.header-search').fadeOut();
            }
        });

    });

    document.addEventListener('DOMContentLoaded', function() {
        const form = document.querySelector('.search-vne')
        const btn = document.querySelector('.btn-search-vne')
        const input = document.getElementById('keywordHeader')

        btn.addEventListener('click', function(e) {
            e.preventDefault() // chặn submit ngay lập tức để kiểm tra

            // nếu chưa có class active thì thêm
            if (!form.classList.contains('active')) {
                form.classList.add('active')
                return
            }

            // đã active: kiểm tra nội dung input
            if (input.value.trim() === '') {
                // rỗng -> bỏ active
                form.classList.remove('active')
            } else {
                // có dữ liệu -> giữ nguyên và cho phép submit
                form.submit()
            }
        })
    })
