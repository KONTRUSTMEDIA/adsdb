        // Add smooth interactions
        document.addEventListener('DOMContentLoaded', function() {
            const buttons = document.querySelectorAll('.action-button');
            
            buttons.forEach(button => {
                button.addEventListener('mouseenter', function() {
                    this.style.transform = 'translateY(-2px)';
                });
                
                button.addEventListener('mouseleave', function() {
                    this.style.transform = 'translateY(0)';
                });
                
                // Add click handlers for demo
                button.addEventListener('click', function(e) {
                    if (this.textContent.includes('Contact Support')) {
                        e.preventDefault();
                        alert('This would open your email client to send a message to support@company.com');
                    } else if (this.textContent.includes('Closure FAQ')) {
                        e.preventDefault();
                        alert('This would open a FAQ page with detailed information about the closure.');
                    }
                });
            });
            
            // Add subtle animation to the icon
            const iconWrapper = document.querySelector('.icon-wrapper');
            setInterval(() => {
                iconWrapper.style.animation = 'none';
                setTimeout(() => {
                    iconWrapper.style.animation = 'gentlePulse 3s infinite ease-in-out';
                }, 10);
            }, 12000);
        });
