  <script>
  document.addEventListener('DOMContentLoaded', function() {
    const scrollContainer = document.getElementById('navScrollContainer');
    const navBtn = document.getElementById('navScrollBtn');
    const navIcon = document.getElementById('navScrollIcon');
    const fadeLeft = document.querySelector('.nav-fade-left');
    const fadeRight = document.querySelector('.nav-fade-right');

    if (!scrollContainer || !navBtn || !navIcon) return;

    // Configuration
    const scrollAmount = 300;
    const scrollSpeed = 600; // Tăng thời gian animation để mượt hơn

    // Update button state and fade indicators
    function updateUI() {
      const maxScroll = scrollContainer.scrollWidth - scrollContainer.clientWidth;
      const needsScroll = maxScroll > 5; // Content is wider than container
      const isAtStart = scrollContainer.scrollLeft <= 5;
      const isAtEnd = scrollContainer.scrollLeft >= (maxScroll - 5);

      // Show/hide scroll buttons based on whether scrolling is needed
      if (needsScroll) {
        navBtn.parentElement.classList.add('show');

        // Update button icon and action
        if (isAtStart) {
          // Show next button
          navBtn.setAttribute('data-action', 'next');
          navBtn.setAttribute('aria-label', 'Next');
          navIcon.innerHTML =
            '<path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M9 5l7 7-7 7"></path>';
        } else if (isAtEnd) {
          // Show previous button (scroll to start)
          navBtn.setAttribute('data-action', 'prev');
          navBtn.setAttribute('aria-label', 'Previous');
          navIcon.innerHTML =
            '<path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M15 19l-7-7 7-7"></path>';
        } else {
          // Show next button (default)
          navBtn.setAttribute('data-action', 'next');
          navBtn.setAttribute('aria-label', 'Next');
          navIcon.innerHTML =
            '<path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M9 5l7 7-7 7"></path>';
        }

        // Update fade indicators
        if (isAtStart) {
          fadeLeft.classList.remove('show');
        } else {
          fadeLeft.classList.add('show');
        }

        if (isAtEnd) {
          fadeRight.classList.add('hide');
        } else {
          fadeRight.classList.remove('hide');
        }
      } else {
        // Hide scroll buttons and fade indicators when not needed
        navBtn.parentElement.classList.remove('show');
        fadeLeft.classList.remove('show');
        fadeRight.classList.remove('show');
      }
    }

    // Smooth scroll function with improved easing
    function smoothScrollTo(targetScrollLeft) {
      const startScrollLeft = scrollContainer.scrollLeft;
      const distance = targetScrollLeft - startScrollLeft;
      const duration = scrollSpeed;
      let startTime = null;

      function animation(currentTime) {
        if (startTime === null) startTime = currentTime;
        const timeElapsed = currentTime - startTime;
        const progress = Math.min(timeElapsed / duration, 1);

        // Improved easing function (ease-in-out cubic)
        const easeInOutCubic = progress < 0.5 ?
          4 * progress * progress * progress :
          1 - Math.pow(-2 * progress + 2, 3) / 2;

        scrollContainer.scrollLeft = startScrollLeft + (distance * easeInOutCubic);

        if (progress < 1) {
          requestAnimationFrame(animation);
        } else {
          // Đảm bảo scroll đến đúng vị trí cuối cùng
          scrollContainer.scrollLeft = targetScrollLeft;
          updateUI();
        }
      }

      requestAnimationFrame(animation);
    }

    // Single button click handler with debouncing
    let isScrolling = false;

    navBtn.addEventListener('click', function() {
      if (isScrolling) return; // Prevent multiple clicks during animation

      const action = navBtn.getAttribute('data-action');
      isScrolling = true;

      if (action === 'next') {
        // Scroll right
        const maxScroll = scrollContainer.scrollWidth - scrollContainer.clientWidth;
        const newScrollLeft = Math.min(maxScroll, scrollContainer.scrollLeft + scrollAmount);
        smoothScrollTo(newScrollLeft);
      } else if (action === 'prev') {
        // Scroll to start (về vị trí đầu tiên)
        smoothScrollTo(0);
      }

      // Reset scrolling flag after animation
      setTimeout(() => {
        isScrolling = false;
      }, scrollSpeed + 100);
    });

    // Touch/swipe support
    let startX = 0;
    let startScrollLeft = 0;
    let isDragging = false;

    scrollContainer.addEventListener('touchstart', function(e) {
      startX = e.touches[0].clientX;
      startScrollLeft = scrollContainer.scrollLeft;
      isDragging = true;
    }, {
      passive: true
    });

    scrollContainer.addEventListener('touchmove', function(e) {
      if (!isDragging) return;

      const currentX = e.touches[0].clientX;
      const diffX = startX - currentX;
      const newScrollLeft = startScrollLeft + diffX;

      scrollContainer.scrollLeft = newScrollLeft;
      updateUI();
    }, {
      passive: true
    });

    scrollContainer.addEventListener('touchend', function() {
      isDragging = false;
    }, {
      passive: true
    });

    // Mouse drag support
    scrollContainer.addEventListener('mousedown', function(e) {
      startX = e.clientX;
      startScrollLeft = scrollContainer.scrollLeft;
      isDragging = true;
      scrollContainer.style.cursor = 'grabbing';
      e.preventDefault();
    });

    scrollContainer.addEventListener('mousemove', function(e) {
      if (!isDragging) return;

      const currentX = e.clientX;
      const diffX = startX - currentX;
      const newScrollLeft = startScrollLeft + diffX;

      scrollContainer.scrollLeft = newScrollLeft;
      updateUI();
    });

    scrollContainer.addEventListener('mouseup', function() {
      isDragging = false;
      scrollContainer.style.cursor = 'grab';
    });

    scrollContainer.addEventListener('mouseleave', function() {
      isDragging = false;
      scrollContainer.style.cursor = 'grab';
    });

    // Update on scroll
    scrollContainer.addEventListener('scroll', updateUI);

    // Update on resize
    window.addEventListener('resize', function() {
      setTimeout(updateUI, 100);
    });

    // Initial update
    updateUI();

    // Navigation works the same on all screen sizes
    // Scroll buttons and fade indicators are always visible
    // No need for screen size checking
  });

  // Search dropdown functionality is handled in main.js

  // Desktop search functionality (backup)
  document.addEventListener('DOMContentLoaded', function() {
    const searchToggleDesktop = document.getElementById('searchToggleDesktop');
    const searchInput = document.getElementById('searchInput');

    console.log('Desktop search elements:', {
      searchToggleDesktop,
      searchInput
    });

    if (searchToggleDesktop && searchInput) {
      console.log('Desktop search initialized successfully');
      let isExpanded = false;

      searchToggleDesktop.addEventListener('click', function(e) {
        e.preventDefault();

        if (!isExpanded) {
          // Expand input
          searchInput.style.width = '240px';
          searchInput.style.paddingLeft = '16px';
          searchInput.style.paddingRight = '40px';
          searchInput.focus();
          isExpanded = true;
        } else {
          // Collapse input
          searchInput.style.width = '0px';
          searchInput.style.paddingLeft = '0px';
          searchInput.style.paddingRight = '40px';
          searchInput.value = '';
          isExpanded = false;
        }
      });

      // Collapse when clicking outside
      document.addEventListener('click', function(e) {
        if (!searchToggleDesktop.contains(e.target) && !searchInput.contains(e.target)) {
          if (isExpanded) {
            searchInput.style.width = '0px';
            searchInput.style.paddingLeft = '0px';
            searchInput.style.paddingRight = '40px';
            searchInput.value = '';
            isExpanded = false;
          }
        }
      });

      // Submit form on Enter key
      searchInput.addEventListener('keypress', function(e) {
        if (e.key === 'Enter') {
          if (searchInput.value.trim()) {
            searchInput.closest('form').submit();
          }
        }
      });
    }
  });
