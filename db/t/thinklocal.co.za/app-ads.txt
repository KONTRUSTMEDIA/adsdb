                        <script>

                            function slugify(text)
                            {
                                return text.toString().toLowerCase()
                                        .replace(/\s+/g, '-')           // Replace spaces with -
                                        .replace(/[^\w\-]+/g, '')       // Remove all non-word chars
                                        .replace(/\-\-+/g, '-')         // Replace multiple - with single -
                                        .replace(/^-+/, '')             // Trim - from start of text
                                        .replace(/-+$/, '');            // Trim - from end of text
                            }

                            $('#l').focus(function() {
                                $ ( "#tipWhere" ).tooltip("hide");
                                $ ( "#tipWhere" ).tooltip("disable");
                            });

                            function submitSearch()
                            {
                                if ($('#q_type').val() == 'business' && $('#q_id').val() && $('#q_id').val()!='0' && $('#q_keyword_url').val()) {
                                    window.location=$('#q_keyword_url').val();
                                    return false;
                                }

                                var what = encodeURIComponent($.trim($('#q').val())).replace(/%20/g, "+").toLowerCase();

                                var where = $.trim($('#l').val());

                                if (!where) {
                                    $ ( "#tipWhere" ).tooltip("enable");
                                    $ ( "#tipWhere" ).tooltip("show");
                                    return false;
                                } else {
                                    var arWhere = where.split(',');
                                    var searchPrefix = '';
                                    if (arWhere.length == 1) {
                                        where = slugify($.trim(where));
                                    } else {
                                        for (counter = arWhere.length - 1; counter >= 0; counter--) {
                                            if (counter == arWhere.length - 1) {
                                                where = slugify($.trim(arWhere[counter]));
                                            } else {
                                                where += '/' + slugify($.trim(arWhere[counter]));
                                            }
                                        }
                                    }

                                    if ($('#last_search_type').val() == 'classifieds') {
                                        searchPrefix = 'search/classifieds/';
                                    } else {
                                        if  ($('#last_search_type').val() == 'all') {
                                            searchPrefix = 'search/all/';
                                        }
                                    }

                                    if (!what && (searchPrefix=='' || searchPrefix=='search/business')) {
                                        window.location = 'https://www.thinklocal.co.za/directory/' + where ;
                                    } else {
                                        window.location = 'https://www.thinklocal.co.za/' + searchPrefix + where + '/' + what;
                                    }
                                    return false;
                                }
                            }

    var emailRegex = /^[a-zA-Z0-9.!#$%&'*+\/=?^_`{|}~-]+@[a-zA-Z0-9](?:[a-zA-Z0-9-]{0,61}[a-zA-Z0-9])?(?:\.[a-zA-Z0-9](?:[a-zA-Z0-9-]{0,61}[a-zA-Z0-9])?)*$/,
            sender_name = $( "#sender_name" ),

            tips = $( "#validateTips");

    function updateTips( t ) {
        tips
                .text( t )
                .addClass( "ui-state-highlight" );
        setTimeout(function() {
            tips.removeClass( "ui-state-highlight", 500 );
        }, 2500 );
    }

    function checkLength( o, n, min, max ) {
        if ( o.val().length > max || o.val().length < min ) {
            o.addClass( "ui-state-error" );
            updateTips( "Please enter " + n + "." );
            /*updateTips( "Length of " + n + " must be between " +
             min + " and " + max + "." );*/
            return false;
        } else {
            return true;
        }
    }

    function checkRegexp( o, regexp, n ) {
        if ( !( regexp.test( o.val() ) ) ) {
            o.addClass( "ui-state-error" );
            updateTips( n );
            return false;
        } else {
            return true;
        }
    }

    function checkValid () {
        var valid = true;

        allFields.removeClass( "ui-state-error" );

        valid = valid && checkLength( sender_name, "your name", 2, 100 );
        if (sender_mobile.val().length>0) {
            valid = valid && checkLength(sender_mobile, "a valid phone number", 10, 15);
            valid = valid && checkRegexp( sender_mobile, /^[\+]?([0-9\s])+$/, "a valid phone number (numbers only)" );

        }

        valid = valid && checkLength( sender_email, "your email", 6, 80 );
        valid = valid && checkRegexp( sender_email, emailRegex, "a valid email address, eg. els@abc.co.za" );

        valid = valid && checkLength( sender_message, "a message", 5, 2000 );

        return valid;
    }

    var emailRegex_errors = /^[a-zA-Z0-9.!#$%&'*+\/=?^_`{|}~-]+@[a-zA-Z0-9](?:[a-zA-Z0-9-]{0,61}[a-zA-Z0-9])?(?:\.[a-zA-Z0-9](?:[a-zA-Z0-9-]{0,61}[a-zA-Z0-9])?)*$/,
            sender_name_errors = $( "#sender_name_errors" ),
            sender_mobile_errors = $( "#sender_mobile_errors" ),
            sender_email_errors = $( "#sender_email_errors" ),
            sender_message_errors = $( "#sender_message_errors" ),
            allFields_errors = $( [] ).add( sender_name_errors ).add( sender_mobile_errors ).add( sender_email_errors ).add( sender_message_errors ),
            tips_errors = $( "#validateTips_errors");

    function updateTips_errors( t ) {
        tips_errors
                .text( t )
                .addClass( "ui-state-highlight" );
        setTimeout(function() {
            tips_errors.removeClass( "ui-state-highlight", 500 );
        }, 2500 );
    }

    function checkLength_errors( o, n, min, max ) {
        if ( o.val().length > max || o.val().length < min ) {
            o.addClass( "ui-state-error" );
            updateTips_errors( "Please enter " + n + "." );
            return false;
        } else {
            return true;
        }
    }

    function checkRegexp_errors( o, regexp, n ) {
        if ( !( regexp.test( o.val() ) ) ) {
            o.addClass( "ui-state-error" );
            updateTips_errors( n );
            return false;
        } else {
            return true;
        }
    }

    function checkValid_errors () {
        var valid = true;

        allFields_errors.removeClass( "ui-state-error" );

        valid = valid && checkLength( sender_name_errors, "your name", 2, 60 );
        if (sender_mobile_errors.val().length>0) {
            valid = valid && checkLength(sender_mobile_errors, "a valid phone number", 10, 15);
            valid = valid && checkRegexp( sender_mobile_errors, /^[\+]?([0-9\s])+$/, "a valid phone number (numbers only)" );

        }

        valid = valid && checkLength( sender_email_errors, "your email", 6, 80 );
        valid = valid && checkRegexp( sender_email_errors, emailRegex_errors, "a valid email address, eg. els@abc.co.za" );

        valid = valid && checkLength( sender_message_errors, "a message", 5, 2000 );

        return valid;
    }

    function setBusinessID_errors(business_id_errors, business_name_errors) {
        $('#business_id_errors').val(business_id_errors);
        $('#business_name_errors').val(business_name_errors);
        $('#ReportErrorsLabel').html('Report Errors on ' + business_name_errors);
        $('#sender_message_errors').val('');
    };

    function SendMessageEmail_errors() {
        if(checkValid_errors()) {

            $('#btnSendQuery_errors').html('Sending...');
            $.ajaxSetup({
                headers: {
                    'X-CSRF-TOKEN': $('input[name="_token"]').val()
                }
            })

            var formData = {
                                    g_recaptcha_response: $('#ReportErrorsModal').find('.g-recaptcha-response').val(),
                                business_id: $('#business_id_errors').val(),
                business_name_errors: $('#business_name_errors').val(),
                sender_name_errors: $('#sender_name_errors').val(),
                sender_mobile_errors: $('#sender_mobile_errors').val(),
                sender_email_errors: $('#sender_email_errors').val(),
                sender_message_errors: $('#sender_message_errors').val()
            }

            $.ajax({
                type: 'POST',
                url: '/client/report',
                data: formData,
                dataType: 'json',
                success: function (response) {
                    $('#btnSendQuery_errors').html("Send Report");
                    updateTips_errors( response.message );
                    if (response.message == 'Your report has been sent.') {
                        $( "#report_submit_success").show();
                        $( "#report_submit_buttons").hide();
                    }
                }
            });
        }
        return false;
    }
