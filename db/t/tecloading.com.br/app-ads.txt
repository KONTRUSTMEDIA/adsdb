    <script>
        document.addEventListener('DOMContentLoaded', function() {
            
            // MENU MOBILE
            const menuBtn = document.getElementById('mobile-menu-btn');
            const menuOverlay = document.getElementById('mobile-menu-overlay');
            const closeMenuBtn = document.getElementById('close-mobile-menu');
            const menuLinks = menuOverlay ? menuOverlay.querySelectorAll('a') : [];

            function openMenu() {
                if(!menuOverlay) return;
                menuOverlay.classList.remove('hidden');
                menuOverlay.classList.add('flex');
                setTimeout(() => { menuOverlay.classList.remove('opacity-0'); }, 10);
                document.body.style.overflow = 'hidden';
            }

            function closeMenu() {
                if(!menuOverlay) return;
                menuOverlay.classList.add('opacity-0');
                setTimeout(() => {
                    menuOverlay.classList.add('hidden');
                    menuOverlay.classList.remove('flex');
                }, 300);
                document.body.style.overflow = '';
            }

            if(menuBtn) menuBtn.addEventListener('click', openMenu);
            if(closeMenuBtn) closeMenuBtn.addEventListener('click', closeMenu);
            menuLinks.forEach(link => link.addEventListener('click', closeMenu));

            // BUSCA
            const searchModal = document.getElementById('search-modal');
            const searchInput = document.getElementById('search-input');

            window.toggleSearch = function() {
                if (!searchModal) return;
                if (searchModal.classList.contains('hidden')) {
                    searchModal.classList.remove('hidden');
                    searchModal.classList.add('flex');
                    setTimeout(() => { 
                        searchModal.classList.remove('opacity-0');
                        if(searchInput) searchInput.focus();
                    }, 10);
                    document.body.style.overflow = 'hidden';
                } else {
                    searchModal.classList.add('opacity-0');
                    setTimeout(() => {
                        searchModal.classList.add('hidden');
                        searchModal.classList.remove('flex');
                    }, 300);
                    document.body.style.overflow = '';
                }
            };

            // MODAL LOGIN
            const authModal = document.getElementById('auth-modal');
            const authPanel = document.getElementById('auth-panel');
            
            window.toggleLoginModal = function() {
                if (!authModal) return;
                if(menuOverlay && !menuOverlay.classList.contains('hidden')) closeMenu();

                if (authModal.classList.contains('hidden')) {
                    authModal.classList.remove('hidden');
                    authModal.classList.add('flex');
                    setTimeout(() => {
                        authModal.classList.remove('opacity-0');
                        if(authPanel) {
                            authPanel.classList.remove('scale-95');
                            authPanel.classList.add('scale-100');
                        }
                    }, 10);
                    document.body.style.overflow = 'hidden';
                } else {
                    authModal.classList.add('opacity-0');
                    if(authPanel) {
                        authPanel.classList.add('scale-95');
                        authPanel.classList.remove('scale-100');
                    }
                    setTimeout(() => {
                        authModal.classList.add('hidden');
                        authModal.classList.remove('flex');
                    }, 300);
                    document.body.style.overflow = '';
                }
            };

            // ABAS LOGIN/REGISTER
            window.switchAuthTab = function(tab) {
                document.querySelectorAll('.auth-view-modal').forEach(el => el.classList.add('hidden'));
                document.getElementById('modal-view-' + tab).classList.remove('hidden');
                
                const btnLogin = document.getElementById('modal-tab-login');
                const btnRegister = document.getElementById('modal-tab-register');

                if (tab === 'login') {
                    btnLogin.className = 'flex-1 pb-3 text-center font-bold text-sm uppercase tracking-wider text-brand-primary border-b-2 border-brand-primary transition';
                    btnRegister.className = 'flex-1 pb-3 text-center font-bold text-sm uppercase tracking-wider text-gray-500 hover:text-brand-text transition';
                } else {
                    btnRegister.className = 'flex-1 pb-3 text-center font-bold text-sm uppercase tracking-wider text-brand-primary border-b-2 border-brand-primary transition';
                    btnLogin.className = 'flex-1 pb-3 text-center font-bold text-sm uppercase tracking-wider text-gray-500 hover:text-brand-text transition';
                }
            };

            // MENU USUÁRIO DROPDOWN (Lógica simples de display)
            const userBtn = document.getElementById('user-menu-btn');
            const userDropdown = document.getElementById('user-dropdown');
            
            if(userBtn && userDropdown) {
                userBtn.addEventListener('click', function(e) {
                    e.preventDefault();
                    e.stopPropagation();
                    userDropdown.classList.toggle('show');
                });

                document.addEventListener('click', function(e) {
                    if (!userDropdown.contains(e.target) && !userBtn.contains(e.target)) {
                        userDropdown.classList.remove('show');
                    }
                });
            }

            // DARK MODE
            const themeToggleBtn = document.getElementById('theme-toggle');
            const darkIcon = document.getElementById('theme-toggle-dark-icon');
            const lightIcon = document.getElementById('theme-toggle-light-icon');

            function applyTheme() {
                if (localStorage.getItem('color-theme') === 'light') {
                    document.body.classList.add('light-mode');
                    if(lightIcon) lightIcon.classList.add('hidden');
                    if(darkIcon) darkIcon.classList.remove('hidden');
                } else {
                    document.body.classList.remove('light-mode');
                    if(darkIcon) darkIcon.classList.add('hidden');
                    if(lightIcon) lightIcon.classList.remove('hidden');
                }
            }
            applyTheme();

            if(themeToggleBtn) {
                themeToggleBtn.addEventListener('click', function() {
                    if (document.body.classList.contains('light-mode')) {
                        document.body.classList.remove('light-mode');
                        localStorage.setItem('color-theme', 'dark');
                    } else {
                        document.body.classList.add('light-mode');
                        localStorage.setItem('color-theme', 'light');
                    }
                    applyTheme();
                });
            }

            // ATALHO ESC
            document.addEventListener('keydown', function(event) {
                if (event.key === "Escape") {
                    if (searchModal && !searchModal.classList.contains('hidden')) toggleSearch();
                    if (authModal && !authModal.classList.contains('hidden')) toggleLoginModal();
                    if (menuOverlay && !menuOverlay.classList.contains('hidden')) closeMenu();
                }
            });

            // ERRO URL PARAMS
            const urlParams = new URLSearchParams(window.location.search);
            if(urlParams.has('login_error')) toggleLoginModal();
            if(urlParams.has('register_error')) { toggleLoginModal(); switchAuthTab('register'); }
        });
    </script>
