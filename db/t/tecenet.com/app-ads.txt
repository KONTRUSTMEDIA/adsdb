<script>
  // 定义全局变量
  let formData = null;
  let captchaToken = null;

  // 滑块验证全局对象
  const SliderVerify = {
    // 初始化滑块验证
    init: function (containerId, options = {}) {
      const container = document.getElementById(containerId);
      if (!container) {
        console.error('Slider container not found:', containerId);
        return null;
      }

      // 默认配置
      const config = Object.assign({
        onSuccess: null,
        onFail: null,
        successText: '验证成功',
        dragText: '正在拖动...',
        defaultText: '向右滑动验证'
      }, options);

      // 创建滑块元素
      if (!container.querySelector('#slider')) {
        container.innerHTML = `
          <div id="slider-bg"></div>
          <div id="slider" class="flex items-center justify-center">
            <i class="fa fa-arrow-right text-primary text-lg"></i>
          </div>
          <div id="slider-text" class="select-none">
            <span class="text-gray-500">${config.defaultText}</span>
          </div>
        `;
      }

      const slider = container.querySelector('#slider');
      const sliderBg = container.querySelector('#slider-bg');
      const sliderText = container.querySelector('#slider-text');

      let isDragging = false;
      let startX = 0;
      let currentLeft = 0;
      let isVerified = false;

      // 获取滑块容器的尺寸
      function getDimensions() {
        const containerRect = container.getBoundingClientRect();
        const sliderRect = slider.getBoundingClientRect();
        return {
          containerWidth: containerRect.width,
          sliderWidth: sliderRect.width,
          maxDistance: containerRect.width - sliderRect.width
        };
      }

      // 鼠标按下事件
      slider.addEventListener('mousedown', function (e) {
        e.preventDefault();
        startDrag(e.clientX);
      });

      // 触摸开始事件
      slider.addEventListener('touchstart', function (e) {
        e.preventDefault();
        startDrag(e.touches[0].clientX);
      });

      // 开始拖动
      function startDrag(clientX) {
        if (isVerified) return;

        isDragging = true;
        startX = clientX;

        // 移除过渡效果
        slider.style.transition = 'none';
        sliderBg.style.transition = 'none';

        // 添加活动状态
        slider.classList.add('active');
        sliderText.innerHTML = `<span class="text-gray-500">${config.dragText}</span>`;

        // 添加移动和结束事件监听器
        document.addEventListener('mousemove', onMove);
        document.addEventListener('touchmove', onTouchMove);
        document.addEventListener('mouseup', onEnd);
        document.addEventListener('touchend', onEnd);
      }

      // 鼠标移动事件
      function onMove(e) {
        if (!isDragging) return;
        e.preventDefault();
        moveSlider(e.clientX);
      }

      // 触摸移动事件
      function onTouchMove(e) {
        if (!isDragging) return;
        e.preventDefault();
        moveSlider(e.touches[0].clientX);
      }

      // 移动滑块
      function moveSlider(clientX) {
        const dimensions = getDimensions();
        const deltaX = clientX - startX;
        const newLeft = Math.max(0, Math.min(currentLeft + deltaX, dimensions.maxDistance));

        // 计算完成百分比
        const percent = (newLeft / dimensions.maxDistance) * 100;

        // 更新滑块位置
        slider.style.transform = `translateX(${newLeft}px)`;
        sliderBg.style.width = `${percent}%`;

        currentLeft = newLeft;
        startX = clientX;
      }

      // 结束拖动
      function onEnd(e) {
        if (!isDragging) return;

        isDragging = false;

        // 移除事件监听器
        document.removeEventListener('mousemove', onMove);
        document.removeEventListener('touchmove', onTouchMove);
        document.removeEventListener('mouseup', onEnd);
        document.removeEventListener('touchend', onEnd);

        // 恢复过渡效果
        slider.style.transition = 'transform 0.3s ease';
        sliderBg.style.transition = 'width 0.3s ease';

        // 移除活动状态
        slider.classList.remove('active');

        const dimensions = getDimensions();

        // 判断是否完成验证
        if (currentLeft >= dimensions.maxDistance - 10) {
          // 验证成功
          validateSuccess();
        } else {
          // 验证失败
          validateFailure();
        }
      }

      // 验证成功
      function validateSuccess() {
        isVerified = true;
        slider.classList.add('success');
        slider.innerHTML = '<i class="fa fa-check text-success text-lg"></i>';
        sliderText.innerHTML = `<span class="text-success">${config.successText}</span>`;

        const dimensions = getDimensions();
        slider.style.transform = `translateX(${dimensions.maxDistance}px)`;
        sliderBg.style.width = '100%';

        // 生成验证令牌
        generateCaptchaToken().then(token => {
          captchaToken = token;

          // 调用成功回调
          if (typeof config.onSuccess === 'function') {
            config.onSuccess(token);
          }
        });
      }

      // 验证失败
      function validateFailure() {
        currentLeft = 0;
        isVerified = false;

        // 添加晃动效果
        slider.classList.add('shake');

        // 重置滑块位置和样式
        slider.style.transform = 'translateX(0)';
        sliderBg.style.width = '0';
        slider.innerHTML = '<i class="fa fa-arrow-right text-primary text-lg"></i>';
        sliderText.innerHTML = `<span class="text-gray-500">${config.defaultText}</span>`;

        // 移除成功状态
        slider.classList.remove('success', 'active');

        // 移除晃动效果
        setTimeout(() => {
          slider.classList.remove('shake');
        }, 1000);

        // 调用失败回调
        if (typeof config.onFail === 'function') {
          config.onFail();
        }
      }

      // 重置滑块
      function reset() {
        currentLeft = 0;
        isVerified = false;

        // 重置滑块位置和样式
        slider.style.transition = 'transform 0.3s ease';
        sliderBg.style.transition = 'width 0.3s ease';
        slider.style.transform = 'translateX(0)';
        sliderBg.style.width = '0';
        slider.innerHTML = '<i class="fa fa-arrow-right text-primary text-lg"></i>';
        sliderText.innerHTML = `<span class="text-gray-500">${config.defaultText}</span>`;

        // 移除所有状态类
        slider.classList.remove('success', 'active', 'shake');
      }

      // 检查是否已验证
      function isValid() {
        return isVerified;
      }

      // 返回公共方法
      return {
        reset,
        isValid
      };
    }
  };

  // 生成验证令牌
  async function generateCaptchaToken() {
    // 生成随机令牌
    const randomBytes = new Uint8Array(16);
    window.crypto.getRandomValues(randomBytes);
    const token = Array.from(randomBytes).map(b => b.toString(16).padStart(2, '0')).join('');

    try {
      // 发送令牌到服务器进行验证
      const response = await fetch('/api/v1.Captcha/verify', {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
          'server': true
        },
        body: JSON.stringify({
          token: token,
          timestamp: Date.now()
        })
      });

      const result = await response.json();
      if (result.code === 1) {
        return result.data.token || token;
      } else {
        console.error('Captcha verification failed:', result.msg);
        return null;
      }
    } catch (error) {
      console.error('Error generating captcha token:', error);
      return token; // 如果服务器验证失败，仍返回本地生成的令牌
    }
  }

  // 等待DOM加载完成后初始化
  document.addEventListener('DOMContentLoaded', function () {
    initializeForm();
  });

  // 初始化函数
  function initializeForm() {
    console.log('Initializing form...');

    // 获取表单元素
    const form = document.getElementById('contactForm');
    if (!form) {
      console.error('Contact form not found');
      return;
    }

    // 绑定表单提交事件
    form.addEventListener('submit', handleFormSubmit);

    // 绑定成功对话框关闭按钮
    const successCloseBtn = document.getElementById('success-close-btn');
    if (successCloseBtn) {
      successCloseBtn.addEventListener('click', function () {
        const successDialog = document.getElementById('success-dialog');
        if (successDialog) {
          successDialog.classList.add('hidden');
        }
      });
    }

    // 点击遮罩层关闭验证弹窗
    const captchaModal = document.getElementById('captcha-modal');
    if (captchaModal) {
      captchaModal.addEventListener('click', function (e) {
        if (e.target === captchaModal || e.target === captchaModal.firstElementChild) {
          captchaModal.classList.add('hidden');
          captchaModal.classList.remove('flex');
          if (window.sliderInstance) {
            window.sliderInstance.reset();
          }
        }
      });
    }

    console.log('Form initialized successfully');
  }

  // 表单验证和提交函数
  function handleFormSubmit(event) {
    event.preventDefault();
    console.log('Form submission started');

    const form = event.target;
    const nameInput = form.querySelector('#name');
    const phoneInput = form.querySelector('#phone');
    const messageInput = form.querySelector('#message');

    if (!nameInput || !phoneInput || !messageInput) {
      console.error('Form fields not found');
      showToast('表单字段获取失败', 'error');
      return false;
    }

    formData = {
      name: nameInput.value.trim(),
      phone: phoneInput.value.trim(),
      message: messageInput.value.trim()
    };

    // 表单验证
    if (!formData.name) {
      showToast('请输入姓名', 'error');
      return false;
    }
    if (!formData.phone) {
      showToast('请输入联系电话', 'error');
      return false;
    }
    if (!formData.message) {
      showToast('请输入留言内容', 'error');
      return false;
    }

    // 显示滑块验证弹出层
    const captchaModal = document.getElementById('captcha-modal');
    captchaModal.classList.remove('hidden');
    captchaModal.classList.add('flex');

    // 初始化滑块验证
    if (!window.sliderInstance) {
      window.sliderInstance = SliderVerify.init('captcha-container', {
        onSuccess: function (token) {
          // 验证成功后延迟关闭弹窗并提交表单
          setTimeout(() => {
            captchaModal.classList.add('hidden');
            captchaModal.classList.remove('flex');
            submitForm(formData, token);
          }, 500);
        },
        onFail: function () {
          showToast('请完成滑块验证', 'error');
        }
      });
    } else {
      window.sliderInstance.reset();
    }

    return false;
  }

  // 提交表单数据到服务器
  async function submitForm(data, token) {
    try {
      // 添加验证令牌到请求数据
      const submitData = {
        ...data,
        captcha_token: token || captchaToken
      };

      const response = await fetch('/api/v1.Contact/add', {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
          'server': true
        },
        body: JSON.stringify(submitData)
      });

      console.log('API response status:', response.status);

      if (!response.ok) {
        throw new Error('提交失败，请稍后重试');
      }

      const result = await response.json();
      console.log('API response:', result);

      if (result.code === 1) {
        // 显示成功对话框
        const successDialog = document.getElementById('success-dialog');
        if (successDialog) {
          successDialog.classList.remove('hidden');
        }

        // 重置表单
        const form = document.getElementById('contactForm');
        if (form) {
          form.reset();
        }

        // 重置验证令牌
        captchaToken = null;
      } else {
        throw new Error(result.msg || '提交失败，请稍后重试');
      }
    } catch (error) {
      console.error('Form submission error:', error);
      showToast(error.message, 'error');
    }
  }

  // 显示提示消息
  function showToast(message, type = 'success') {
    console.log('Showing toast:', message, type);

    const toast = document.getElementById('toast-notification');
    const toastMessage = document.getElementById('toast-message');
    const toastIcon = document.getElementById('toast-icon');

    if (!toast || !toastMessage || !toastIcon) {
      console.error('Toast elements not found');
      return;
    }

    toastMessage.textContent = message;
    toastIcon.className = type === 'success' ? 'fa fa-check-circle text-green-500 text-xl mr-3' : 'fa fa-times-circle text-red-500 text-xl mr-3';

    toast.classList.remove('translate-x-full', 'opacity-0');

    setTimeout(() => {
      toast.classList.add('translate-x-full', 'opacity-0');
    }, 3000);
  }

  function hideToast() {
    const toast = document.getElementById('toast-notification');
    if (toast) {
      toast.classList.add('translate-x-full', 'opacity-0');
    }
  }

            // 生成随机令牌

            const token = Array.from(randomBytes).map(b => b.toString(16).padStart(2, '0')).join('');

                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json',
                        'server': true
                    },
                    body: JSON.stringify({
                        token: token,
                        timestamp: Date.now()
                    })
                });

                const result = await response.json();
                if (result.code === 1) {
                    return result.data.token || token;
                } else {
                    console.error('Captcha verification failed:', result.msg);
                    return null;
                }
            } catch (error) {
                console.error('Error generating captcha token:', error);
                return token; // 如果服务器验证失败，仍返回本地生成的令牌

            const captchaModal = document.getElementById('captcha-modal');
            captchaModal.classList.remove('hidden');
            captchaModal.classList.add('flex');

            // 保存回调函数
            formSubmitCallback = successCallback;

            // 初始化滑块验证
            if (!window.sliderInstance) {
                window.sliderInstance = SliderVerify.init('captcha-container', {
                    onSuccess: function (token) {
                        // 验证成功后延迟关闭弹窗并提交表单
                        setTimeout(() => {
                            captchaModal.classList.add('hidden');
                            captchaModal.classList.remove('flex');
                            submitFormWithToken(formData, token);
                        }, 500);
                    },
                    onFail: function () {
                        showToast('请完成滑块验证', 'error');
                    }
                });
            } else {
                window.sliderInstance.reset();
            }
        }

        // 添加带令牌的表单提交函数
        function submitFormWithToken(formData, token) {
            // 添加验证令牌到请求数据
            const submitData = {
                ...formData,
                captcha_token: token
            };

            return fetch('/api/v1.Contact/add', {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json',
                    'server': true
                },
                body: JSON.stringify(submitData),
                signal: AbortSignal.timeout(10000) // 10 second timeout
            })
                .then(response => {
                    if (!response.ok) {
                        throw new Error('网络错误，请稍后再试');
                    }
                    return response.json();
                })
                .then(data => {
                    if (formSubmitCallback) {
                        formSubmitCallback();
                    } else {
                        // Default success behavior
                        showSuccessDialog();
                    }
                    return data;
                })
                .catch(error => {
                    // Provide specific error messages based on error type
                    let errorMessage = '提交失败';

                    if (error.name === 'TypeError' && error.message === 'Failed to fetch') {
                        errorMessage = '无法连接到服务器，请检查网络连接或稍后再试';
                    } else if (error.name === 'AbortError') {
                        errorMessage = '请求超时，请稍后再试';
                    } else if (error.message) {
                        errorMessage = '提交失败：' + error.message;
                    }

                    showToast(errorMessage, 'error');
                    throw error; // Re-throw to handle in the caller
                });
        }

        // 点击遮罩层关闭验证弹窗
        const captchaModal = document.getElementById('captcha-modal');
        if (captchaModal) {
            captchaModal.addEventListener('click', function (e) {
                if (e.target === captchaModal || e.target === captchaModal.firstElementChild) {
                    captchaModal.classList.add('hidden');
                    captchaModal.classList.remove('flex');
                    if (window.sliderInstance) {
                        window.sliderInstance.reset();
                    }
                }
            });
        }

                let nameValue = '';
                let phoneValue = '';
                let messageValue = '';

                formInputs.forEach(input => {

                    formValues[input.name] = value;

                    if (input.name === 'name') nameValue = value;
                    if (input.name === 'phone') phoneValue = value;
                    if (input.name === 'message') messageValue = value;

                    formInputs.forEach(input => {
                        if (input.name === 'phone') input.focus();
                    });
                    return;
                }

                // Get title information - try to find from page or use generic

                if (titleElement) {

                // Get the area from the page if available

                const area = areaElement ? areaElement.textContent.trim() : '';

                // Prepare form data based on form type
                let formTitle = '';
                let formattedContent = '';

                switch (form.dataset.formType) {
                    case 'repair':
                        formTitle = '【有同款设备要修】';

                    case 'engineer':

                        let serviceContent = '';

                        formInputs.forEach(input => {
                            if (input.name === 'service_area') serviceArea = input.value.trim();
                            if (input.name === 'service_content') serviceContent = input.value.trim();
                        });

                        formattedContent = `${formTitle}\n设备: ${title}\n称呼: ${nameValue}\n联系方式: ${phoneValue}\n服务区域: ${serviceArea}\n服务内容: ${serviceContent}`;
                        break;

                    case 'trade':
                        // Check if it's buy or sell
                        const typeRadios = form.querySelectorAll('input[name="type"]');
                        const isBuy = typeRadios.length ? typeRadios[0].checked : true;
                        formTitle = isBuy ? '【我想买设备】' : '【我想卖设备】';
                        formattedContent = `${formTitle}\n设备: ${title}\n需求关系: ${isBuy ? '我想买' : '我想卖'}\n联系人: ${nameValue}\n联系方式: ${phoneValue}\n留言备注: ${messageValue || '无'}`;
                        break;

                        formattedContent = `${formTitle}\n设备: ${title}\n联系人: ${nameValue}\n联系方式: ${phoneValue}\n服务区域: ${area}\n${messageValue ? ('内容: ' + messageValue) : ''}`;
                }

                // Format content - standardize for API
                const apiFormData = {
                    name: nameValue,
                    phone: phoneValue,
                    content: formattedContent
                };

                // Disable form while submitting
                const formElements = form.querySelectorAll('input, textarea, button, select');
                formElements.forEach(el => el.disabled = true);

                submitButton.innerHTML = '<span class="flex items-center justify-center"><svg class="animate-spin -ml-1 mr-2 h-4 w-4 text-white" xmlns="http://www.w3.org/2000/svg" fill="none" viewBox="0 0 24 24"><circle class="opacity-25" cx="12" cy="12" r="10" stroke="currentColor" stroke-width="4"></circle><path class="opacity-75" fill="currentColor" d="M4 12a8 8 0 018-8V0C5.373 0 0 5.373 0 12h4zm2 5.291A7.962 7.962 0 014 12H0c0 3.042 1.135 5.824 3 7.938l3-2.647z"></path></svg>提交中...</span>';

            dialog.appendChild(modalTitle);

            dialog.appendChild(form);
            modal.appendChild(dialog);

            modal.addEventListener("click", (e) => {
                if (e.target === modal) {
                    document.body.removeChild(modal);
                }
            });
