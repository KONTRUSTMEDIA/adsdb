    function onHeaderSearchClick() {
        if ($("#headerSearchTerm").val() == "") {
            alert("Unesite pojam pretrage");
            return false;
        }
        var keyword = $("#headerSearchTerm").val();
        window.searchEntity = (window.searchEntity) ? window.searchEntity : "";
        location.href = "/Search/SearchView/?keyword=" + keyword + "&entity=" + window.searchEntity;
        return false;
        }

        }

        }

        function onSearchActionClick(e) {
            var t = $(e.target);
            var searchInput = $("#headerSearchTerm");
            var container = $(".header-search-container");
            var search = $(".header-search-control");

            if (t.hasClass('header-search-container')) {
                container.removeClass('show');
                search.toggleClass('focus');
                searchInput.blur();
                return false;
            } else if (t.hasClass('search-indicator')) {
                search.toggleClass('focus');
                if (search.hasClass('focus')) {
                    container.addClass('show');
                    searchInput.focus();
                } else {
                    container.removeClass('show');
                    searchInput.blur();
                }
            }





        }
