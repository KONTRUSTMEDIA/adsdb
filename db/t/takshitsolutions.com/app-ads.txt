            const level2Panels = document.querySelectorAll('.menu-level-2');


            /* ===============================
               STOP submenu bubbling
            =============================== */
            submenus.forEach(menu => {
                menu.addEventListener('click', e => e.stopPropagation());
            });


            /* ===============================
               MOBILE MAIN MENU OPEN
            =============================== */
            if (openBtn) {
                openBtn.addEventListener('click', e => {
                    e.stopPropagation();
                    navList.classList.add('open');
                    overlay.style.display = 'block';
                });
            }


            /* ===============================
               MAIN NAV CLICK
            =============================== */
            navItems.forEach(item => {

                const link = item.querySelector('.nav-link');
                const submenu = item.querySelector('.submenu');

                if (!link) return;

                link.addEventListener('click', e => {

                    e.stopPropagation();

                    const isOpen = item.classList.contains('is-open');

                    // reset everything
                    navItems.forEach(el => el.classList.remove('is-open'));
                    submenus.forEach(s => s.classList.remove('active'));
                    level2Panels.forEach(p => p.classList.remove('active'));

                    if (!isOpen) {

                        item.classList.add('is-open');
                        overlay.style.display = 'block';

                        if (submenu && window.innerWidth <= 1200) {
                            submenu.classList.add('active');
                        }

                    } else {
                        smartClose();
                    }

                });

            });


            /* ===============================
               LEVEL 2 PANEL OPEN (ID BASED)
            =============================== */
            subLinks.forEach(link => {

                const targetId = link.dataset.target;
                if (!targetId) return;

                const panel = document.getElementById(targetId);

                link.addEventListener('click', e => {

                    e.stopPropagation();

                    level2Panels.forEach(p => p.classList.remove('active'));

                    if (panel) panel.classList.add('active');

                });

            });


            /* ===============================
               LEVEL 2 BACK BUTTON
            =============================== */
            document.querySelectorAll('.menu-level-2 .back-btn')
                .forEach(btn => {

                    btn.addEventListener('click', e => {

                        e.stopPropagation();

                        const panel = btn.closest('.menu-level-2');
                        if (panel) panel.classList.remove('active');

                    });

                });


            /* ===============================
               SUBMENU BACK BUTTON
            =============================== */
            document.querySelectorAll('.submenu > .container .back-btn')
                .forEach(btn => {

                    btn.addEventListener('click', e => {

                        e.stopPropagation();

                        const submenu = btn.closest('.submenu');
                        const navItem = btn.closest('.nav-item');

                        // close only submenu level
                        if (submenu) submenu.classList.remove('active');
                        if (navItem) navItem.classList.remove('is-open');

                        // close level 2 panels
                        level2Panels.forEach(p => p.classList.remove('active'));

                        // ✅ DO NOT call smartClose
                        // keep main nav open

                    });

                });

            /* ===============================
               SMART CLOSE SYSTEM
            =============================== */
            function smartClose() {

                const level2 = document.querySelector('.menu-level-2.active');
                if (level2) {
                    level2.classList.remove('active');
                    return;
                }

                const submenu = document.querySelector('.submenu.active');
                if (submenu) {
                    submenu.classList.remove('active');
                    return;
                }

                const openItem = document.querySelector('.nav-item.is-open');
                if (openItem) {
                    openItem.classList.remove('is-open');
                }

                const anythingOpen =
                    document.querySelector('.menu-level-2.active') ||
                    document.querySelector('.submenu.active') ||
                    document.querySelector('.nav-item.is-open');

                if (!anythingOpen) {
                    navList.classList.remove('open');
                    overlay.style.display = 'none';
                }

                            dropdown.style.display = "none";

                        };

                        list.appendChild(div);

                    });
            }

            /* search */
            search.oninput = () => {

                let v = search.value.toLowerCase();

                render(v);

                reset.style.display = v ? "block" : "none";

            };

            /* reset */
            reset.onclick = () => {

                search.value = "";
                render();
                reset.style.display = "none";

            };

            /* outside click */
            document.addEventListener("click", e => {

                if (!document.querySelector(".product-select")
                    .contains(e.target))
                    dropdown.style.display = "none";

            });

        });

    </script>

    <script>
        document.addEventListener("DOMContentLoaded", function () {

            const mobileNav = document.getElementById("nav-list");
            const mobileNavOpenBtn = document.getElementById("open-menu");
            const mobileNavCloseBtn = document.querySelector(".close-btn-menu");
            const overlay = document.getElementById("overlay");

            function isMobile() {
                return window.innerWidth <= 1200;
            }

            function openMobileNav() {
                if (!mobileNav) return;

                mobileNav.classList.add("open");

                if (isMobile()) {
                    document.body.classList.add("menu-open");
                }
            }

            function closeMobileNav() {
                if (!mobileNav) return;

                mobileNav.classList.remove("open");
                document.body.classList.remove("menu-open");
            }

            if (mobileNavOpenBtn)
                mobileNavOpenBtn.addEventListener("click", openMobileNav);

            if (mobileNavCloseBtn)
                mobileNavCloseBtn.addEventListener("click", closeMobileNav);

            /* 🔥 overlay click → menu close */
            if (overlay)
                overlay.addEventListener("click", closeMobileNav);

            window.addEventListener("resize", () => {
                if (!isMobile()) {
                    document.body.classList.remove("menu-open");
                }
            });

            /* modal detector */
            document.addEventListener("show.bs.modal", closeMobileNav);
            document.addEventListener("shown.bs.modal", closeMobileNav);

            if (window.jQuery) {
                $(document).on("show.bs.modal", closeMobileNav);
            }

            document.addEventListener("click", function (e) {
                if (e.target.closest('[data-toggle="modal"], [data-bs-toggle="modal"]')) {
                    closeMobileNav();
                }
            });

        });
    </script>
