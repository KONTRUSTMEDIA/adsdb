        // Set the countdown to 2 days from when the page loads
        let countdownDate = new Date().getTime() + 2 * 24 * 60 * 60 * 1000;

        const updateCountdown = () => {
            const now = new Date().getTime();
            const distance = countdownDate - now;

            // If the countdown is over, reset it for another 2 days
            if (distance < 0) {
                countdownDate = new Date().getTime() + 2 * 24 * 60 * 60 * 1000;
                // No need to continue this execution, the next interval will calculate correctly
                return;
            }

            const days = Math.floor(distance / (1000 * 60 * 60 * 24));
            const hours = Math.floor((distance % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));
            const minutes = Math.floor((distance % (1000 * 60 * 60)) / (1000 * 60));
            const seconds = Math.floor((distance % (1000 * 60)) / 1000);

            document.getElementById("days").innerHTML = String(days).padStart(2, '0');
            document.getElementById("hours").innerHTML = String(hours).padStart(2, '0');
            document.getElementById("minutes").innerHTML = String(minutes).padStart(2, '0');
            document.getElementById("seconds").innerHTML = String(seconds).padStart(2, '0');
        };

        // Update the countdown immediately on page load
        updateCountdown();
        // Then update it every second
