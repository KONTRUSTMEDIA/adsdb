            return "light";

        return defaultDarkMode;
    }

        document.documentElement.dataset.appliedMode = mode;
    }
    function rotatePreferences(userPreference) {
        if (userPreference === "light") {
            return "dark";
        }
        if (userPreference === "dark") {
            return "light";
        }
        return defaultDarkMode;
    }
