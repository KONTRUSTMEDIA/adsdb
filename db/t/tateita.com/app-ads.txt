        document.addEventListener('DOMContentLoaded', function() {
            const bannerClosed = localStorage.getItem('languageBannerClosed');
            if (bannerClosed === 'true') {
                return;
            }

            const userLang = navigator.language || navigator.userLanguage;
            const pageLang = document.documentElement.lang;
            const currentPath = window.location.pathname;
            const filename = currentPath.substring(currentPath.lastIndexOf('/') + 1);

            const languageDirs = {
                'ja': '/',
                'en': '/en/',
                'es': '/es/',
                'fr': '/fr/',
                'id': '/id/',
                'ko': '/ko/',
                'ru': '/ru/',
                'th': '/th/',
                'zh-Hans': '/zh_hans/',
                'zh-Hant': '/zh_hant/'
            };

            const languageMessages = {
                'ja': 'このページは{pageLang}で表示されています。{userLang}版も利用可能です。 <a href="{link}">切り替える</a>',
                'en': 'This page is displayed in {pageLang}. It is also available in {userLang}. <a href="{link}">Switch</a>',
                'es': 'Esta página se muestra en {pageLang}. También está disponible en {userLang}. <a href="{link}">Cambiar</a>',
                'fr': 'Cette page est affichée en {pageLang}. Elle est également disponible en {userLang}. <a href="{link}">Changer</a>',
                'id': 'Halaman ini ditampilkan dalam bahasa {pageLang}. Tersedia juga dalam bahasa {userLang}. <a href="{link}">Ganti</a>',
                'ko': '이 페이지는 {pageLang}로 표시되고 있습니다. {userLang} 버전도 이용 가능합니다. <a href="{link}">전환</a>',
                'ru': 'Эта страница отображается на {pageLang}. Она также доступна на {userLang}. <a href="{link}">Переключить</a>',
                'th': 'หน้านี้แสดงในภาษา{pageLang} มีเวอร์ชันภาษา{userLang}ด้วย <a href="{link}">เปลี่ยน</a>',
                'zh-Hans': '此页面显示为{pageLang}。{userLang}版本也可用。<a href="{link}">切换</a>',
                'zh-Hant': '此頁面顯示為{pageLang}。{userLang}版本也可用。<a href="{link}">切換</a>'
            };

            const banner = document.getElementById('language-banner');
            const message = document.getElementById('language-message');
            let userLangCode = userLang.substring(0, 2);

            if (userLangCode in languageDirs) {
                if (userLangCode !== pageLang) {
                    const link = languageDirs[userLangCode] + filename;
                    const messageTemplate = languageMessages[userLangCode] || languageMessages['en'];
                    message.innerHTML = messageTemplate
                        .replace('{pageLang}', getLanguageName(pageLang))
                        .replace('{userLang}', getLanguageName(userLang))
                        .replace('{link}', link);
                    banner.style.display = 'block';
                }
            } else if (pageLang !== 'en') {
                const link = languageDirs['en'] + filename;
                const messageTemplate = "Sorry, we don't have a version in your language 😭 Would you like to switch to English? <a href='{link}'>Switch to English</a>";
                message.innerHTML = messageTemplate.replace('{link}', link);
                banner.style.display = 'block';
            }

            document.querySelector('#language-banner .close-btn').addEventListener('click', function() {
                banner.style.display = 'none';
                localStorage.setItem('languageBannerClosed', 'true');
            });

            function getLanguageName(langCode) {
                const languages = {
                    'ja': '日本語',
                    'en': 'English',
                    'es': 'Español',
                    'fr': 'Français',
                    'id': 'Bahasa Indonesia',
                    'ko': '한국어',
                    'ru': 'Русский',
                    'th': 'ไทย',
                    'zh-Hans': '简体中文',
                    'zh-Hant': '繁體中文'
                };
                return languages[langCode.substring(0, 2)] || langCode;
            }
        });
