            var wait = document.getElementById(submitId);
            
            if (wait.tagName.toLowerCase() === 'input') {
              // Handle <input>
              if (wait.value.length > 9) 
                wait.value = "Loading ";
              else 
                wait.value += ".";
            } else {
              // Handle <button> and <a>
