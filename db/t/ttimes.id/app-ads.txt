        function onScrap(id, elem) {
            if (!isWorking) { //중복 클릭 방지
                isWorking = true;

                var clsAttr = $(elem).attr("class").split(' ');

                if (clsAttr.length != 1) {

                            if (osName == "" || isMobileApp) {

                                if (valConfirm) {
                                    selectCountSet("minus", id, $("#count_" + id));
                                    $(elem).removeClass(clsAttr[1]);
                                } else {
                                    isWorking = false;
                                }
                            } else {
                                selectCountSet("minus", id, $("#count_" + id));
                                $(elem).removeClass(clsAttr[1]);
                            }
                        } else {
                            selectCountSet("minus", id, $("#count_" + id));
                            $(elem).removeClass(clsAttr[1]);
                        }
                    }
                } else {
                    $(elem).addClass('on');

                    selectCountSet("plus", id, $("#count_" + id));
                }
            } else {
                console.log("block");
            }
        }

        function selectCountSet(counttype, articleId, elem) {
            callApp('indicatorState', 'state=show');
            var url = 'countSave.html';

            $.ajax({
                url: url,
                data: {
                    countype: counttype,
                    gisa_id: articleId
                },
                success: function (data) {
                    elem.text(data);

                    if (counttype == "minus") {
                        for (var i = 0; i < favoriteArr.length; i++) {
                            if (favoriteArr[i] == articleId) {
                                favoriteArr.splice(i, 1);
                                break;
                            }
                        }

                        alert("Tidak Dipilih");

                        if (category == "favorite") {
                            $("list_p1").UiBuilder('removeFromView', articleId);
                        }
                    } else {
                        if (favoriteArr == null) {
                            favoriteArr = new Array();
                        }

                        /* 기사가 중복으로 localStorage에 들어가는 것을 방지 */
                        var isDuplicated = false;
                        for (var i = 0; i < favoriteArr.length; i++) {
                            if (favoriteArr[i] == articleId) {
                                isDuplicated = true;
                            }
                        }

                        alert("Dipilih");

                        if (!isDuplicated) {
                            favoriteArr.unshift(articleId);
                        }
                    }

                    localStorage.setItem("TTimes_favorite", JSON.stringify(favoriteArr));
                    callApp('indicatorState', 'state=hide');
                    isWorking = false;
                }
            });

		var prevOffsetY = 0;
		$(window).scroll(function(){
			var offsetY = window.scrollY;
			
			var isScrollStart = offsetY < 200;
			if(isScrollStart){
				if($(".hdbtop").css("display") != 'none')
					$(".hdbtop").hide();
			}else{
				if($(".hdbtop").css("display") == 'none')
					$(".hdbtop").show();
			}

			if(!isSearchOpen) $("#content").css('margin-top', $("#header").height());
			
			if(cPage == "view"){
				var mHeaderObj = $("#header").eq(0);

				var direction = prevOffsetY - offsetY < 0;
				if(direction){
					if(offsetY >= 53 && $('.top_head').css('display') != "none"){
						$(".top_head").hide();
					}
				}else{
                    if(offsetY <= 53 && $('.top_head').css('display') == "none"){
                        $(".top_head").show();
                    }
				}

				prevOffsetY = offsetY;
			}
		});
		

        function getQueryStringObject(url){
            var obj = [];
            var queryStrings = url.slice(url.indexOf('?')+1).split('&');
            for(var i in queryStrings){
                var _o = queryStrings[i].split('=');
                obj.push(_o[0]);

                obj[_o[0]] = _o[1];
            }

            return obj;
        }

        function markLikeSign(datas){

            if(isMobileApp || (!isMobileApp && osName == '')){
                if(favoriteArr != null){
                    for(var i=0; i<datas.length; i++){
                        for(var j=0; j<favoriteArr.length; j++){
                            if(datas[i].id == favoriteArr[j]){
                                datas[i].favorite = " on";
                                break;
                            }
                        }
                    }
                }
            }
        }
