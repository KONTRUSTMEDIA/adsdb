    function shuffle(array) {
      for (let i = array.length - 1; i > 0; i--) {
        const j = Math.floor(Math.random() * (i + 1));
        [array[i], array[j]] = [array[j], array[i]];
      }

    // --- Improved Simulation Functions ---

    // Generates a random number within a human-like range
    function getRandomDelay(min, max) {
      return Math.random() * (max - min) + min;
    }

    // Generates a random point within the iframe's visible area
    function getRandomIframePoint() {
      const iframeRect = iframe.getBoundingClientRect();
      const padding = 20; // Keep cursor away from edges
      const x = getRandomDelay(padding, iframeRect.width - padding);
      const y = getRandomDelay(padding, iframeRect.height - padding);
      return { x, y };
    }

    // Simulates natural mouse movement to a target (x, y) with bezier curves and jitters
    function simulateNaturalMouseMovement(targetX, targetY, duration = 500) {
      return new Promise(resolve => {
        const startX = parseFloat(cursor.style.left) || getRandomIframePoint().x;
        const startY = parseFloat(cursor.style.top) || getRandomIframePoint().y;
        const startTime = performance.now();

        // Control points for a quadratic bezier curve
        const cp1x = getRandomDelay(Math.min(startX, targetX), Math.max(startX, targetX));
        const cp1y = getRandomDelay(Math.min(startY, targetY), Math.max(startY, targetY));

        function animateMouse(currentTime) {
          const elapsedTime = currentTime - startTime;
          let progress = Math.min(elapsedTime / duration, 1);

          // Quadratic Bezier curve calculation
          const x = (1 - progress) * (1 - progress) * startX +
                    2 * (1 - progress) * progress * cp1x +
                    progress * progress * targetX;
          const y = (1 - progress) * (1 - progress) * startY +
                    2 * (1 - progress) * progress * cp1y +
                    progress * progress * targetY;

          // Add subtle jitter

          cursor.style.left = (x + jitterX) + 'px';
          cursor.style.top = (y + jitterY) + 'px';

          if (progress < 1) {
            originalRequestAnimationFrame(animateMouse);
          } else {
            resolve();
          }
        }
        originalRequestAnimationFrame(animateMouse);
      });
    }

    async function simulateMouse() {

      cursor.classList.add('active');
      await simulateNaturalMouseMovement(x, y, getRandomDelay(400, 1200)); // Vary movement duration

      cursor.classList.remove('active');

    async function simulateScroll() {

    async function runSimulation() {
      if (!isLoading) {
        const actions = [simulateMouse, simulateClick, simulateScroll];
        const weights = [60, 20, 20]; // Adjusted weights for more mouse movement
        const total = weights.reduce((a, b) => a + b, 0);
        let random = Math.random() * total;

        for (let i = 0; i < actions.length; i++) {
          if (random < weights[i]) {
            await actions[i](); // Use await for sequential actions
            break;
          }
          random -= weights[i];
        }
      }
    }

    function startSimulation() {

          runSimulation();
