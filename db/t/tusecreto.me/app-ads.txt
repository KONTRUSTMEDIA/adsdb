    <script>
        function openAuthModal(tab) {
            document.getElementById('auth-modal').classList.remove('hidden');
            
            // Ocultar pestañas si es login puro
            const tabs = document.getElementById('auth-tabs');
            if (tab === 'login') {
                tabs.classList.add('hidden');
            } else {
                tabs.classList.remove('hidden');
            }
            
            switchTab(tab);
        }
        
        function closeAuthModal() {
            document.getElementById('auth-modal').classList.add('hidden');
        }
        
        function switchTab(tab) {
            const l = document.getElementById('login-form'), r = document.getElementById('register-form');
            const bl = document.getElementById('tab-login'), br = document.getElementById('tab-register');
            const t = document.getElementById('modal-title');
            
            if(tab === 'login') { 
                l.classList.remove('hidden'); r.classList.add('hidden'); 
                bl.classList.add('active'); br.classList.remove('active'); 
                t.textContent='Iniciar Sesión'; 
            } else { 
                l.classList.add('hidden'); r.classList.remove('hidden'); 
                bl.classList.remove('active'); br.classList.add('active'); 
                t.textContent='Crear Cuenta'; 
            }
        }
        
        function showFeedback(id, msg, type) {
            const el = document.getElementById(id); el.textContent = msg; el.classList.remove('hidden');
            el.className = `p-3 rounded-xl text-sm font-medium text-center mb-4 animate-slide-up block ${type === 'error' ? 'bg-red-50 text-red-600' : 'bg-green-50 text-green-600'}`;
        }

        const menuBtn = document.getElementById('menu-btn');
        const mobileMenu = document.getElementById('mobile-menu');
        if(menuBtn && mobileMenu) {
            menuBtn.addEventListener('click', (e) => { e.stopPropagation(); mobileMenu.classList.toggle('hidden'); });
            document.addEventListener('click', (e) => { if(!menuBtn.contains(e.target) && !mobileMenu.contains(e.target)) mobileMenu.classList.add('hidden'); });
        }

        // LOGIN
        document.getElementById('login-form').addEventListener('submit', async function(e) {
            e.preventDefault();
            const btn = this.querySelector('button');
            const country = this.querySelector('[name="country"]').value;
            const phone = this.querySelector('[name="phone"]').value;
            const password = this.querySelector('[name="password"]').value;

            btn.disabled = true; btn.innerText = 'Verificando...';
            document.getElementById('login-feedback').classList.add('hidden');

            try {
                const res = await fetch('auth_action.php', { method: 'POST', headers: {'Content-Type': 'application/json'}, body: JSON.stringify({
                    action: 'login', phone: country + phone, password: password
                })});
                const text = await res.text();
                let data; try { data = JSON.parse(text); } catch(e) { throw new Error("Error servidor"); }
                
                if(data.status==='success') { showFeedback('login-feedback', '¡Hola!', 'success'); setTimeout(() => window.location.href='dashboard.php', 1000); }
                else { showFeedback('login-feedback', data.message, 'error'); btn.disabled=false; btn.innerText='Entrar'; }
            } catch(e) { showFeedback('login-feedback', 'Error conexión', 'error'); btn.disabled=false; btn.innerText='Entrar'; }
        });

        // REGISTER HERO
        document.getElementById('hero-register-form').addEventListener('submit', async function(e) {
            e.preventDefault(); 
            const btn = this.querySelector('button[type="submit"]');
            const pass = this.querySelector('[name="password"]').value;
            const confirm = this.querySelector('[name="confirm_password"]').value;
            
            if(pass !== confirm) { showFeedback('hero-reg-feedback', 'Contraseñas no coinciden', 'error'); return; }

            const countrySelect = document.getElementById('hero-country');
            const countryCode = countrySelect.value;
            const countryIso = countrySelect.options[countrySelect.selectedIndex].getAttribute('data-iso') || 'OT';
            const phone = this.querySelector('[name="phone"]').value;
            
            btn.disabled = true; btn.innerText = 'Creando...';
            document.getElementById('hero-reg-feedback').classList.add('hidden');

            try {
                const res = await fetch('auth_action.php', {
                    method: 'POST', headers: {'Content-Type': 'application/json'},
                    body: JSON.stringify({ action: 'register', phone: countryCode + phone, password: pass, country: countryIso })
                });
                const data = await res.json();
                if (data.status === 'success') {
                    showFeedback('hero-reg-feedback', '¡Cuenta creada!', 'success');
                    setTimeout(() => window.location.href='dashboard.php', 1000);
                } else {
                    showFeedback('hero-reg-feedback', data.message, 'error');
                    btn.disabled = false; btn.innerText = 'Registrarme';
                }
            } catch(e) { showFeedback('hero-reg-feedback', 'Error conexión', 'error'); btn.disabled = false; btn.innerText = 'Registrarme'; }
        });
