<script>
    (function () {
        const API_URL = "https://theartofwar.eu/api/ts-stats.php";

        function formatNumber(value) {
            const n = Number(value);
            if (!Number.isFinite(n)) return "-";
            return n.toLocaleString("de-DE");
        }

        function setText(id, text) {
            const el = document.getElementById(id);
            if (!el) return;
            el.textContent = text;
        }

        function setStatus(isOnline) {
            const el = document.getElementById("tsStatServerStatus");
            if (!el) return;

            el.classList.remove("is-online", "is-offline");

            if (isOnline) {
                el.textContent = "Online";
                el.classList.add("is-online");
            } else {
                el.textContent = "Offline";
                el.classList.add("is-offline");
            }
        }

        async function loadStats() {
            try {
                const res = await fetch(API_URL, { method: "GET", cache: "no-store" });
                if (!res.ok) throw new Error("HTTP " + res.status);

                const json = await res.json();
                if (!json || json.ok !== true || !json.data) throw new Error("Bad JSON");

                const server = json.data.server || {};
                const totals = json.data.totals || {};

                setStatus(!!server.online);

                setText("tsStatTotalUsers", formatNumber(totals.users));
                setText("tsStatChannels", formatNumber(server.channels));
                setText(
                    "tsStatOnlineClients",
                    formatNumber(server.online_clients) + " / " + formatNumber(server.max_clients)
                );
                setText("tsStatPing", formatNumber(server.ping_ms) + " ms");

                const packetloss = Number(server.packetloss);
                const packetlossPercent = Number.isFinite(packetloss) ? (packetloss * 100) : NaN;
                setText(
                    "tsStatPacketloss",
                    Number.isFinite(packetlossPercent) ? packetlossPercent.toFixed(4) + "%" : "-"
                );
            } catch (e) {
                setStatus(false);
                setText("tsStatTotalUsers", "-");
                setText("tsStatChannels", "-");
                setText("tsStatOnlineClients", "-");
                setText("tsStatPing", "-");
                setText("tsStatPacketloss", "API offline");
            }
        }

        loadStats();
        setInterval(loadStats, 30000);
    })();
