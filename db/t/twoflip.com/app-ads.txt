    <script>
      // Apply Splash Theme Settings immediately
      (function() {
        try {
          const cached = localStorage.getItem('theme_settings_cache');
          if (cached) {
            const settings = JSON.parse(cached);
            
            // 1. Check if disabled
            if (settings.theme_splash_enabled === 'false') {
              const splash = document.getElementById('splash-screen');
              if (splash) splash.style.display = 'none';
              return;
            }

            // 2. Update Logo
            if (settings.theme_splash_logo) {
              const logo = document.getElementById('splash-logo');
              if (logo) logo.src = settings.theme_splash_logo;
            }

            // 3. Update Title
            if (settings.theme_splash_title) {
              const title = document.getElementById('splash-name');
              if (title) title.textContent = settings.theme_splash_title;
            }
          }
        } catch (e) {
          console.error("Error applying splash settings:", e);
        }
      })();

      // Fallback: Hide splash screen if it's still visible after a timeout (in case React fails)
      setTimeout(function() {
        const splash = document.getElementById('splash-screen');
        if (splash && document.body.contains(splash)) {
          splash.style.opacity = '0';
          setTimeout(function() {
            if (document.body.contains(splash)) splash.remove();
          }, 500);
        }
      }, 8000); // 8 seconds safety timeout
    </script>
