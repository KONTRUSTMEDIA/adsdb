async function cacheGet(key) {
  try {
    const cache = await caches.open(CACHE_NAME);
    const res = await cache.match(key);
    if (res) return await res.json();
  } catch (e) { console.warn("Cache read failed:", e); }

}

  try {

function updateMetaTags({ title, description, image, url, published }) {
  document.title = `${title} | Arabian Post WIRE`;

  // Remove previous SEO tags
  document.querySelectorAll('meta[name="description"], meta[property^="og:"], meta[name^="twitter:"], script[type="application/ld+json"]').forEach(el => el.remove());

  const head = document.head;
  const metaList = [
    { name: "description", content: description },
    { property: "og:title", content: title },
    { property: "og:description", content: description },
    { property: "og:image", content: image },
    { property: "og:type", content: "article" },
    { property: "og:url", content: url },
    { name: "twitter:card", content: "summary_large_image" },
    { name: "twitter:title", content: title },
    { name: "twitter:description", content: description },
    { name: "twitter:image", content: image },
  ];
  metaList.forEach(meta => {
    const m = document.createElement("meta");
    if (meta.name) m.setAttribute("name", meta.name);
    if (meta.property) m.setAttribute("property", meta.property);
    m.setAttribute("content", meta.content);
    head.appendChild(m);
  });

  // Canonical tag
  let canonical = document.querySelector("link[rel='canonical']");
  if (!canonical) {
    canonical = document.createElement("link");
    canonical.rel = "canonical";
    document.head.appendChild(canonical);
  }
  canonical.href = url;

  // JSON-LD structured data
  const schema = {
    "@context": "https://schema.org",
    "@type": "NewsArticle",
    "headline": title,
    "image": [image],
    "datePublished": published,
    "dateModified": published,
    "author": { "@type": "Organization", "name": "Arabian Post WIRE" },

    },
    "description": description,
    "mainEntityOfPage": { "@type": "WebPage", "@id": url }
  };
  const script = document.createElement("script");
  script.type = "application/ld+json";
  script.textContent = JSON.stringify(schema, null, 2);
  head.appendChild(script);
}

function extractImageMeta(html){
  const d = document.createElement("div");
  d.innerHTML = html;
  const img = d.querySelector("img");
  if (!img) return { src: "", w: 0, h: 0, alt: "" };
  const w = parseInt(img.getAttribute("width")) || 0;
  const h = parseInt(img.getAttribute("height")) || 0;
  const alt = img.getAttribute("alt") || "";
  return { src: img.src, w, h, alt };
}


// Guess width/height from common URL patterns if not present in HTML
function guessDimsFromURL(src){
  if (!src) return { w: 0, h: 0 };
  const m = src.match(/(\d{3,5})x(\d{3,5})/); // e.g. ...1200x800...
  if (m) return { w: +m[1], h: +m[2] };
  try {
    const u = new URL(src, location.href);
    const w = parseInt(u.searchParams.get("w") || u.searchParams.get("width") || "0");
    const h = parseInt(u.searchParams.get("h") || u.searchParams.get("height") || "0");
    return { w: w||0, h: h||0 };
  } catch { return { w: 0, h: 0 }; }
}

// Handles cache-first post loading, then live fetch with graceful fallback.
// Works with the top-3 ordering logic inside renderPosts() — do NOT modify sequence.
// Safe to use with infinite scroll, cache refresh, and modal article loading.
async function fetchPosts() {

  const cacheKey = "posts-home";
  showLoadingBar();

  if (slugFromURL) {
    modal.style.display = "flex";
    postContent.innerHTML = "<div style='color:#aaa;text-align:center;'><div class='loader'><div class='dot'></div><div class='dot'></div><div class='dot'></div><div class='dot'></div><div class='dot'></div><div class='dot'></div><div class='dot'></div><div class='dot'></div></div></div>";

  } else {
    createSkeletons(9);
  }

  // ---- Step 1: Try browser cache first ----
  // ---- Step 1: Try browser cache first ----
  const cachedData = await cacheGet(cacheKey);
  if (cachedData && cachedData.items && cachedData.items.length > 0) {
    console.log("✅ Loaded posts from local cache");

    // 🌙 NEW: Preload top images in background for instant smoothness
    requestIdleCallback(() => {
      cachedData.items.slice(0, 10).forEach(p => {
        const img = new Image();
        img.src = extractImage(p.content) || "https://cms.1arabia.com/assets/ap-img-arab-news-post.jpg?bust=1";
      });
    });

    clearSkeletons();
    renderPosts(cachedData.items);
    if (slugFromURL) setTimeout(() => openBySlug(slugFromURL), 50);

    // If not forcing, stop here (don’t refetch from server)
    if (!forceFetch) {
      finishLoadingBar();
      return;
    }
  }


  // ---- Step 2: Try fetching from server ----
  try {
    console.log("🌐 Fetching from server...");
    const res = await fetch(API_BASE + (forceFetch ? "?forceFetch=1" : ""));
    const data = await res.json();
    finishLoadingBar();

    if (data && data.items && data.items.length > 0) {
      clearSkeletons();
      renderPosts(data.items);
      await cacheSet(cacheKey, data);
      console.log("💾 Cache updated from server data");
    } else if (cachedData && cachedData.items) {
      console.warn("⚠️ Server empty, using cached posts");
      clearSkeletons();
      renderPosts(cachedData.items);
    } else {
      console.warn("⚠️ No live posts, trying archive...");
      await loadFromArchive();
    }

    if (slugFromURL) setTimeout(() => openBySlug(slugFromURL), 50);
  } catch (e) {
    console.error("❌ Network fetch failed:", e);
    finishLoadingBar();

    if (cachedData && cachedData.items && cachedData.items.length > 0) {
      clearSkeletons();
      renderPosts(cachedData.items);
      console.log("📦 Served from cache due to fetch error");
    } else {
      console.warn("⚠️ Network failed and no cache, trying archive...");
      await loadFromArchive(true);
    }
  }
}

async function loadFromArchive(isOffline = false) {
  try {
    const res = await fetch(`${API_BASE}?source=archived&page=1`);

  }
}





// === WIRE CUSTOM LOGIC ===
// Ensures the top 3 posts are always the latest (sorted by published date DESC)
// Everything after the first 3 remains randomised to keep layout variety.
// Safe modification: does NOT affect caching, lazy-load, or infinite scroll.

// ===== RENDER POSTS =====
// ===== RENDER POSTS =====
function renderPosts(posts) {
  const FALLBACK_IMG = "https://cms.1arabia.com/assets/ap-img-arab-news-post.jpg?bust=1";
  window.displayedPostIds = window.displayedPostIds || new Set();

  if (!posts || !Array.isArray(posts) || !posts.length) return;

  // --- Sort newest first ---
  posts.sort((a, b) => new Date(b.published) - new Date(a.published));

  const top3 = posts.slice(0, 3);
  const rest = posts.slice(3);

  // Shuffle the rest
  for (let i = rest.length - 1; i > 0; i--) {
    const j = Math.floor(Math.random() * (i + 1));
    [rest[i], rest[j]] = [rest[j], rest[i]];
  }

  const finalList = [...top3, ...rest];
  const BATCH = 6;
  let i = 0;

  function createCard(p, index) {
    if (window.displayedPostIds.has(p.id)) return null;
    window.displayedPostIds.add(p.id);

    const slug = slugify(p.title);
    slugMap[slug] = p.id;
    idMap[p.id] = slug;

    const card = document.createElement("div");
    card.className = "card";
    randomizeCardSize(card);

    const meta = extractImageMeta(p.content);
    let imgSrc = meta.src || FALLBACK_IMG;
    const { w, h } = meta.w && meta.h ? meta : guessDimsFromURL(imgSrc);

    card.innerHTML = `
      <div class="media" style="--w:${w || 16};--h:${h || 9}"></div>
      <div class="content">
        <h3>${p.title}</h3>
        <p>${truncate(stripHTML(p.content), 120)}</p>
      </div>`;

    const media = card.querySelector(".media");
    const imgEl = new Image();
    imgEl.loading = "lazy";
    imgEl.decoding = "async";
    imgEl.alt = esc(meta.alt || p.title || "");
    imgEl.src = imgSrc;

    imgEl.addEventListener("load", () => media.classList.add("loaded"));
    imgEl.addEventListener("error", () => {
      if (imgEl.src !== FALLBACK_IMG) imgEl.src = FALLBACK_IMG;
      media.classList.add("loaded");
    });

    media.appendChild(imgEl);
    card.addEventListener("click", () => openPost(p.id, slug));
    return card;
  }

  function renderBatch() {
    const slice = finalList.slice(i, i + BATCH);
    slice.forEach((p, idx) => {
      const card = createCard(p, i + idx);
      if (card) {
        grid.appendChild(card);
        requestAnimationFrame(() => card.classList.add("visible"));
      }
    });

    i += BATCH;
    if (i < finalList.length) requestIdleCallback(renderBatch, { timeout: 100 });
  }

  requestIdleCallback(renderBatch, { timeout: 100 });
}

async function openPost(id, slug) {
  const skipAdFeature = false; // ⬅️ Set to false to disable skip message & timer

  showLoadingBar();
  modal.style.display = "flex";
  history.pushState({ slug }, "", slug);

  if (!skipAdFeature) {
    // === Direct fast load (no ad delay) ===
    postContent.innerHTML = "<p style='text-align:center;color:#aaa;'>Loading article...</p>";
    try {
      const res = await fetch(`${API_BASE}?id=${id}`);
      const data = await res.json();
      finishLoadingBar();

      const img =
        extractImage(data.content) ||
        "https://cms.1arabia.com/assets/ap-img-arab-news-post.jpg?bust=1";
      const description = truncate(stripHTML(data.content), 160);
      const url = `https://thearabianpost.com/wirenews/${slug}`;

      postContent.innerHTML = `
        <h2>${data.title}</h2>
        <p><small>${new Date(data.published).toLocaleString()}</small></p>
        <div>${data.content}</div>`;

      updateMetaTags({
        title: data.title,
        description,
        image: img,
        url,
        published: data.published,
      });
    } catch {
      finishLoadingBar();
      postContent.innerHTML = "<p style='text-align:center;color:#aaa;'>Error loading post.</p>";
    }
    return; // exit here
  }

  // === Ad-delay version (with skip option) ===
  let countdown = 3000; // 3 seconds
  const intervalTime = 50; // update every 50ms
  let timer;
  let articleData = null;

  // Create countdown display
  const countdownEl = document.createElement("p");
  countdownEl.style = "text-align:center;color:#aaa;font-size:1rem;margin-top:20px;";
  postContent.innerHTML = "";
  postContent.appendChild(countdownEl);

  // Create Skip button
  const skipBtn = document.createElement("button");
  skipBtn.textContent = "⏩ Skip Ad";
  skipBtn.style = `
    margin-top: 10px;
    padding: 8px 14px;
    border: 2px solid var(--accent);
    background: transparent;
    color: var(--accent);
    font-weight: bold;
    border-radius: 6px;
    cursor: pointer;
    display: block;
    margin-left: auto;
    margin-right: auto;
  `;
  skipBtn.style.display = "none"; // hidden until timer is running
  postContent.appendChild(skipBtn);

  const updateCountdown = () => {
    countdown -= intervalTime;
    if (countdown < 0) countdown = 0;
    countdownEl.textContent = `Please wait, showing ad… ${(countdown / 1000).toFixed(2)}s`;
    if (countdown > 0) {
      skipBtn.style.display = "block"; // show skip button while ticking
    }
  };

  updateCountdown();
  timer = setInterval(updateCountdown, intervalTime);

  try {
    const res = await fetch(`${API_BASE}?id=${id}`);
    articleData = await res.json();
    finishLoadingBar();

    const renderArticle = () => {
      clearInterval(timer);
      skipBtn.style.display = "none";
      const img =
        extractImage(articleData.content) ||
        "https://cms.1arabia.com/assets/ap-img-arab-news-post.jpg?bust=1";
      const description = truncate(stripHTML(articleData.content), 160);
      const url = `https://thearabianpost.com/wirenews/${slug}`;

      postContent.innerHTML = `
        <h2>${articleData.title}</h2>
        <p><small>${new Date(articleData.published).toLocaleString()}</small></p>
        <div>${articleData.content}</div>`;

      updateMetaTags({
        title: articleData.title,
        description,
        image: img,
        url,
        published: articleData.published,
      });
    };

    // Auto load after 3s
    setTimeout(() => {
      if (articleData) renderArticle();
    }, 3000);

    // Skip button click → load instantly
    skipBtn.addEventListener("click", renderArticle);

  } catch {
    finishLoadingBar();
    clearInterval(timer);
    postContent.innerHTML = "<p style='text-align:center;color:#aaa;'>Error loading post.</p>";
  }
}




async function openBySlug(slug) {
  // wait until posts are loaded or cached properly
  let retries = 0;
  while (!window.displayedPostIds && retries < 10) {
    await new Promise(r => setTimeout(r, 200)); // wait 200ms
    retries++;
  }

  try {
    // 1️⃣ Try cache first
    const cache = await cacheGet("posts-home");
    if (cache && cache.items) {
      const found = cache.items.find(p => slugify(p.title) === slug);
      if (found) return openPost(found.id, slug);
    }

    // 2️⃣ Try live feed (page=1)
    const res = await fetch(`${API_BASE}?page=1`);
    const data = await res.json();

    if (data && data.items) {
      const found = data.items.find(p => slugify(p.title) === slug);
      if (found) return openPost(found.id, slug);
    }

    // 3️⃣ Final fallback: ask CMS directly for slug
    const directRes = await fetch(`${API_BASE}?slug=${slug}`);
    if (directRes.ok) {
      const article = await directRes.json();
      if (article && (article.id || article.title)) return openPost(article.id, slug);
    }

    postContent.innerHTML = `<p style='color:#999;text-align:center;'>Article not found.</p>`;
  } catch (e) {
    console.error("Slug fetch error:", e);
    postContent.innerHTML = `<p style='color:#999;text-align:center;'>Failed to load article.</p>`;
  }
}
