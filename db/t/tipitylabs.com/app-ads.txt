    }
   }
  });
 })();

(function(){
  var BANNER_UNIT = '1271228320251021T180759727EB14ED91826A4C5284C53E06B1962816'; // ChristianAdNet 4.12 unit
  var BANNER_SRC  = 'https://secureaddisplay.com/au/bmd/';
  var injected = false;

  function ensureDock(){
    var d = document.getElementById('ad-dock');
    if (!d){
      d = document.createElement('div');
      d.id = 'ad-dock';
      document.body.appendChild(d);
    }
    return d;
  }

  function injectBanner(){
    if (injected) return; injected = true;
    var dock = ensureDock();
    dock.innerHTML = '';
    var ins = document.createElement('ins');
    ins.className = 'CANBMDDisplayAD';
    ins.setAttribute('data-bmd-ad-unit', BANNER_UNIT);
    ins.style.display = 'block';
    dock.appendChild(ins);
    var s = document.createElement('script');
    s.src = BANNER_SRC;
    s.async = true;
    dock.appendChild(s);
    dock.style.display = 'block';
  }

  // Helper: determine if an element looks like the splash container
  function isSplash(el){
    if (!el) return false;
    var id = (el.id || '').toLowerCase();
    var cls = (el.className || '').toString().toLowerCase();
    return /splash|intro|landing|start|welcome/.test(id + ' ' + cls);
  }

  // 1) Attach to common "Play/Start" buttons
  var selectors = ['#play','#start','.btn-start','.start','.play','[data-start]','[data-action="start"]'];
  selectors.forEach(function(sel){
    var el = document.querySelector(sel);
    if (el){
      el.addEventListener('click', function(){ setTimeout(injectBanner, 0); }, { once:true });
    }
  });

  // 2) Global click capture: if button/link text contains Play/Start, trigger
  document.addEventListener('click', function(e){
    var t = e.target.closest && e.target.closest('button,a,.btn,.button');
    var s = (t ? t.textContent : e.target.textContent) || '';
    s = s.trim().toLowerCase();
    if (/play|start|begin/.test(s)){
      setTimeout(injectBanner, 0);
    }
  }, true);

  // 3) MutationObserver: watch for splash element going hidden/removed
  var obs = new MutationObserver(function(){
    var splash = document.querySelector('#splash, .splash, #splash-screen, .splash-screen, .intro, #intro');
    if (splash){
      var style = window.getComputedStyle(splash);
      if (style.display === 'none' || style.visibility === 'hidden' || style.opacity === '0'){
        setTimeout(injectBanner, 0);
        obs.disconnect();
      }
    } else {
      // If splash node removed from DOM entirely, assume main page visible
      setTimeout(injectBanner, 0);
      obs.disconnect();
    }
  });
  obs.observe(document.documentElement, { childList: true, subtree: true, attributes: true, attributeFilter: ['style','class'] });

  // Safety: if main content becomes interactive after 2.5s and no banner injected, try once
  setTimeout(function(){ if (!injected) injectBanner(); }, 2500);
})();
