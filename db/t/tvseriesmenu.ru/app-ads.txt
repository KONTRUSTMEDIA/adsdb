        <!DOCTYPE html>
        <html>
        <head>
            <meta charset="utf-8">
            <meta name="robots" content="noindex, nofollow">
            <script>
                // 1. Ставим куку на 7 дней
                document.cookie = "wp_human_check=1; max-age=604800; path=/";

                // 2. Если кука записалась — перезагружаем страницу чисто
                if (document.cookie.indexOf("wp_human_check") !== -1) {
                    window.location.replace(window.location.protocol + '//' + window.location.host + window.location.pathname + window.location.search);
                } else {
                    // Если куки отключены
                    document.write('<div style="text-align:center;padding-top:20%;font-family:sans-serif;">Пожалуйста, включите Cookies в браузере.</div>');
                }
            </script>
        </head>
        <body style="background:#fff;"></body>
        </html>
