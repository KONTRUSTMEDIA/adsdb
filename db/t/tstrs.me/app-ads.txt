  function recordResult(methodName, status, details) {
    const result = {
      method: methodName,
      status: status,
      timestamp: Date.now(),
      duration: Date.now() - detectionResults.startTime,
      details: details || {},
      adBlockDetected: status === 'blocked'
    };
    
    detectionResults.results.push(result);
    detectionResults.completedTests++;
    
    if (result.adBlockDetected) {
      detectionResults.hasAdBlock = true;
      detectionResults.blockedMethods.push(methodName);
    }
    
    if (detectionResults.completedTests === detectionResults.totalTests) {
      setTimeout(showDetectionSummary, 100);
    }
  }
  
  function detectByBaitElement() {
    const baits = [
      createBait('ad ads adsbox ad-placement banner-ad', 'bait1'),
      createBait('adsbygoogle ad-slot google-ad', 'bait2'),
      createBait('advertisement pub_300x250 pub_728x90', 'bait3')

    function createBait(className, id) {
      const bait = document.createElement('div');
      bait.className = className;
      bait.id = id;
      bait.style.cssText = 'position:absolute;top:-1px;left:-1px;width:1px;height:1px;';
      document.body.appendChild(bait);
      return bait;
    }
    
    setTimeout(function() {
      let blockedCount = 0;
      let baitDetails = [];
      
      baits.forEach(function(bait) {
        const isHidden = bait.offsetHeight === 0 || 
                        bait.offsetWidth === 0 || 
                        window.getComputedStyle(bait).display === 'none' ||
                        window.getComputedStyle(bait).visibility === 'hidden';
        
        baitDetails.push({
          id: bait.id,
          className: bait.className,
          被屏蔽: isHidden,
          高度: bait.offsetHeight,
          宽度: bait.offsetWidth,
          display: window.getComputedStyle(bait).display
        });
        
        if (isHidden) {
          blockedCount++;
        }
        
        document.body.removeChild(bait);
      });
      
      if (blockedCount > 0) {
        recordResult('诱饵元素检测', 'blocked', {
          blockedCount: blockedCount,
          totalBaits: baits.length,
          details: baitDetails
        });

          blockedCount: 0,

  }

  function detectByFetch() {
    const startTime = Date.now();
    
    const testUrl = '/static/ad-pixel.png?' + Date.now();
    
    fetch(testUrl, {
      method: 'GET'
    })
    .then(response => {

        const duration = Date.now() - startTime;
        recordResult('Fetch请求检测', 'success', {
          duration: duration

        throw new Error('Request failed');

      const duration = Date.now() - startTime;
      
      recordResult('Fetch请求检测', 'blocked', {
        error: error.message,
        errorType: error.name,
        duration: duration
      });
    });
  }
  
  function detectByExternalDomain() {
    const cacheBuster = Date.now() + '_' + Math.random().toString(36).substring(2, 9);
    
    const adDomains = [
      {

        fallback: null

    let completedTests = 0;
    let failedDomains = [];
    let networkErrors = [];
    
    adDomains.forEach(function(domain, index) {
      const testUrl = domain.url + '?_cb=' + cacheBuster + '&_t=' + Date.now();
      
      const startTime = Date.now();
      const timeout = 5000;
      
      const timeoutId = setTimeout(function() {
        const duration = Date.now() - startTime;
        
        failedDomains.push(domain.name);
        networkErrors.push({
          domain: domain.name,
          error: 'timeout',
          duration: duration,
          reason: 'DNS解析超时或网络连接失败'
        });
        
        completedTests++;
        checkComplete();
      }, timeout);
      
      const script = document.createElement('script');
      script.src = testUrl;
      script.async = true;
      
      script.onload = function() {
        clearTimeout(timeoutId);
        completedTests++;
        checkComplete();
      };
      
      script.onerror = function(event) {
        clearTimeout(timeoutId);
        const duration = Date.now() - startTime;
        
        let errorReason = '未知错误';
        if (duration < 100) {
          errorReason = 'DNS解析失败或Hosts文件屏蔽';
        } else if (duration < 1000) {
          errorReason = '网关层连接被拒绝';
        } else {
          errorReason = '网络超时或服务器不可达';
        }
        
        failedDomains.push(domain.name);
        networkErrors.push({
          domain: domain.name,
          url: domain.url,
          error: 'network_error',
          duration: duration,
          reason: errorReason
        });
        
        completedTests++;
        checkComplete();
      };
      
      document.head.appendChild(script);
    });
    
    function checkComplete() {
      if (completedTests === adDomains.length) {
        if (failedDomains.length > 0) {
          recordResult('外部域名检测', 'blocked', {
            blockedDomains: failedDomains,
            totalTested: adDomains.length,
            errors: networkErrors,
            detectionType: 'DNS/Gateway/Hosts'
          });
        } else {
          recordResult('外部域名检测', 'success', {
            totalTested: adDomains.length,
            allAccessible: true
          });
        }
      }
    }
  }

<script>
  function changeSearchType(text, iconClass, searchType) {
    document.getElementById('dropdownText').innerText = text;
    document.getElementById('searchType').value = searchType;
    const button = document.querySelector('.dropdown-toggle i');
    button.className = `bi ${iconClass}`;
  }

  function debounce(func, delay) {
    let timer;
    return function () {
      const context = this;
      const args = arguments;
      clearTimeout(timer);
      timer = setTimeout(() => func.apply(context, args), delay);
    };
  }

  const handleSearch = debounce(function () {
    const query = document.getElementById('searchQuery').value.trim();
    if (query.length >= 3) {
      fetch(`/rts?q=${encodeURIComponent(query)}`)
        .then(response => response.json())
        .then(data => {
          const suggestionsBox = document.getElementById('suggestionsBox');
          suggestionsBox.innerHTML = '';

          if (data.length > 0) {
            data.forEach(suggestion => {
              const suggestionItem = document.createElement('button');
              suggestionItem.type = 'button';
              suggestionItem.className = 'list-group-item list-group-item-action';
              suggestionItem.addEventListener('click', function () {
                // 直接填入并提交，后端会记录搜索热词（本地自增）
                document.getElementById('searchQuery').value = suggestion;
                document.getElementById('searchForm').submit();
              });

              suggestionItem.innerText = suggestion;
              suggestionsBox.appendChild(suggestionItem);
            });

            suggestionsBox.style.display = 'block';
          } else {
            suggestionsBox.style.display = 'none';
          }
        })
        .catch(error => {
          console.error('Error fetching search suggestions:', error);
        });
    } else {
      document.getElementById('suggestionsBox').style.display = 'none';
    }
  }, 500);

  document.getElementById('searchQuery').addEventListener('input', handleSearch);

  document.getElementById('searchForm').addEventListener('submit', function (event) {
    const query = document.getElementById('searchQuery').value.trim();
    if (!query) {
      event.preventDefault();
      alert("请输入搜索关键词！");
      return;
    }
    // 不再前端打点；提交后由后端在 /search 路由本地自增记录搜索词
  });

  document.querySelectorAll('.hot-keyword-btn').forEach(button => {
    button.addEventListener('click', function () {
      const keyword = this.innerText.trim();
      document.getElementById('searchQuery').value = keyword;
      document.getElementById('searchForm').submit();
    });
  });
</script>

<script>
document.addEventListener('DOMContentLoaded', function () {
  function getVisibleCount() {
    if (window.matchMedia('(max-width: 480px)').matches) return 2;
    if (window.matchMedia('(min-width: 481px) and (max-width: 640px)').matches) return 3;
    if (window.matchMedia('(min-width: 641px) and (max-width: 1024px)').matches) return 4;
    return 5;
  }

  function setupCarousel(root) {
    const viewport = root.querySelector('.carousel-viewport');
    const track    = root.querySelector('.carousel-track');
    const prevBtn  = root.querySelector('.js-prev');
    const nextBtn  = root.querySelector('.js-next');

    // 明确：不支持拖动/滑动
    if (viewport) {
      viewport.style.touchAction = 'none';
      viewport.onpointerdown = null;
      viewport.ontouchstart = null;
      viewport.onmousedown = null;
    }

    const originals = Array.from(track.children);
    const TOTAL = originals.length;

    let visible = getVisibleCount();
    let itemPercent = 100 / visible;
    let current = 0;
    let isMoving = false;

    const CLICK_STEP = 2; // 每次点击移动的卡片数

    function build() {
      const wasIndex = current; // 记录当前位置（可按需复用）
      visible = getVisibleCount();
      itemPercent = 100 / visible;

      track.innerHTML = '';
      const head = originals.map(n => n.cloneNode(true));
      const tail = originals.map(n => n.cloneNode(true));
      const nodes = [...head, ...originals, ...tail];
      nodes.forEach(n => track.appendChild(n));

      Array.from(track.children).forEach(col => {
        col.style.flexBasis = itemPercent + '%';
        col.style.maxWidth  = itemPercent + '%';
      });

      current = TOTAL; // 居中到原始列表的起始
      setTranslate(-current * itemPercent, false);
    }

    function setTranslate(percent, withTransition = true) {
      track.style.transition = withTransition ? 'transform .45s ease' : 'none';
      track.style.transform = `translateX(${percent}%)`;
      if (!withTransition) {
        // 立刻生效后恢复过渡
        void track.offsetHeight;
        track.style.transition = 'transform .45s ease';
      }
    }

    function step(dir, magnitude = 1) {
      if (isMoving) return;
      isMoving = true;
      current += dir * magnitude;
      setTranslate(-current * itemPercent, true);
    }

    track.addEventListener('transitionend', () => {
      // 无缝循环
      if (current < TOTAL) {
        current += TOTAL;
        setTranslate(-current * itemPercent, false);
      } else if (current >= TOTAL + TOTAL) {
        current -= TOTAL;
        setTranslate(-current * itemPercent, false);
      }
      isMoving = false;
    });

    // 仅保留按钮点击
    prevBtn.addEventListener('click', () => step(-1, CLICK_STEP));
    nextBtn.addEventListener('click', () => step(+1, CLICK_STEP));

    // 可选：键盘无障碍（左右方向键控制）
    root.addEventListener('keydown', (e) => {
      if (e.key === 'ArrowLeft') { step(-1, CLICK_STEP); }
      else if (e.key === 'ArrowRight') { step(+1, CLICK_STEP); }
    });

    // 页面尺寸变化时重建
    let resizeTimer = null;
    window.addEventListener('resize', () => {
      clearTimeout(resizeTimer);
      resizeTimer = setTimeout(build, 120);
    });

    build();
  }

  document.querySelectorAll('.seauthor-carousel').forEach(setupCarousel);
});
</script>

(function() {
  // 主题切换功能
  var themeToggle = document.getElementById('theme-toggle');
  
  function getTheme() {
    var saved = localStorage.getItem('theme');
    if (saved) return saved;
    // 默认使用日间模式
    return 'light';
  }
  
  function setTheme(theme) {
    document.documentElement.setAttribute('data-theme', theme);
    try {
      localStorage.setItem('theme', theme);
    } catch (e) {
      // localStorage 可能在隐私模式下不可用
    }
  }
  
  // 点击切换
  if (themeToggle) {
    themeToggle.addEventListener('click', function() {
      var currentTheme = document.documentElement.getAttribute('data-theme') || 'light';
      var newTheme = currentTheme === 'dark' ? 'light' : 'dark';
      setTheme(newTheme);
    });
  }
  
  // 监听系统主题变化（仅当用户未手动设置时）
  window.matchMedia('(prefers-color-scheme: dark)').addEventListener('change', function(e) {
    if (!localStorage.getItem('theme')) {
      setTheme(e.matches ? 'dark' : 'light');
    }
  });
  
  // 初始化时确保版本戳同步
  var THEME_VERSION = '20260131';
  var savedVersion = localStorage.getItem('theme_version');
  if (savedVersion !== THEME_VERSION) {
    try {
      localStorage.removeItem('theme');
      localStorage.setItem('theme_version', THEME_VERSION);
      setTheme('light');
    } catch (e) {}
  }
})();
</script>
