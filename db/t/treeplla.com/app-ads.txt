    <script>
        // 페이지 로딩 상태 관리
        window.TreepllaLoader = {
            resourcesLoaded: 0,
            totalResources: 0,
            minLoadTime: 1000, // 최소 로딩 시간 (2초)
            maxLoadTime: 2000, // 최대 로딩 시간 (10초)
            startTime: Date.now(),
            
            init: function() {
                this.countResources();
                this.addResourceListeners();
                this.startFallbackTimer();
                
                // DOM이 완전히 로드된 후 확인
                if (document.readyState === 'loading') {
                    document.addEventListener('DOMContentLoaded', this.checkComplete.bind(this));
                } else {
                    this.checkComplete();
                }
            },
            
            countResources: function() {
                // CSS 파일 수 계산
                var cssLinks = document.querySelectorAll('link[rel="stylesheet"]');
                this.totalResources += cssLinks.length;
                
                // JavaScript 파일 수 계산
                var scripts = document.querySelectorAll('script[src]');
                this.totalResources += scripts.length;
                
                console.log('Total resources to load:', this.totalResources);
            },
            
            addResourceListeners: function() {
                var self = this;
                
                // CSS 로드 확인
                var cssLinks = document.querySelectorAll('link[rel="stylesheet"]');
                cssLinks.forEach(function(link) {
                    if (link.sheet || link.styleSheet) {
                        self.onResourceLoaded();
                    } else {
                        link.addEventListener('load', function() {
                            self.onResourceLoaded();
                        });
                        link.addEventListener('error', function() {
                            console.warn('Failed to load CSS:', link.href);
                            self.onResourceLoaded();
                        });
                    }
                });
                
                // JavaScript 로드는 별도로 확인 (스크립트가 순차적으로 로드되므로)
                var scripts = document.querySelectorAll('script[src]');
                var loadedScripts = 0;
                scripts.forEach(function(script) {
                    if (script.readyState === 'complete' || script.readyState === 'loaded') {
                        self.onResourceLoaded();
                    } else {
                        script.addEventListener('load', function() {
                            self.onResourceLoaded();
                        });
                        script.addEventListener('error', function() {
                            console.warn('Failed to load JS:', script.src);
                            self.onResourceLoaded();
                        });
                    }
                });
            },
            
            onResourceLoaded: function() {
                this.resourcesLoaded++;
                console.log('Resource loaded:', this.resourcesLoaded + '/' + this.totalResources);
                this.checkComplete();
            },
            
            checkComplete: function() {
                var elapsed = Date.now() - this.startTime;
                var allResourcesLoaded = this.resourcesLoaded >= this.totalResources;
                var minTimeElapsed = elapsed >= this.minLoadTime;
                
                if (allResourcesLoaded && minTimeElapsed) {
                    this.showPage();
                }
            },
            
            startFallbackTimer: function() {
                var self = this;
                // 최대 로딩 시간 후 강제로 페이지 표시
                setTimeout(function() {
                    console.log('Fallback timer triggered - showing page');
                    self.showPage();
                }, this.maxLoadTime);
            },
            
            showPage: function() {
                if (this.pageShown) return;
                this.pageShown = true;
                
                console.log('Showing page - Resources loaded:', this.resourcesLoaded + '/' + this.totalResources);
                
                // 로딩 화면 제거
                setTimeout(function() {
                    var loadingCover = document.querySelector('.PageTransitionCover_cover__pYL_k');
                    if (loadingCover) {
                        loadingCover.classList.add('off');
                    }
                    document.body.classList.remove('screen_load');
                }, 100);
            }
        };
        
        // 즉시 초기화
        window.TreepllaLoader.init();
    </script>
