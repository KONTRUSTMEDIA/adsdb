    const moreToggle = document.getElementById('navMoreToggle');

    const recalc = () => {
        const containerWidth = nav.parentElement?.clientWidth || nav.clientWidth;
        const items = Array.from(nav.querySelectorAll(':scope > li.menu-item'));
        // Reset first
        items.forEach(i => i.classList.remove('is-overflow-hidden'));

        let used = 0;
        let hiddenCount = 0;
        for (const item of items) {
            const w = item.offsetWidth + 30; // include margin-right heuristic
            const reserved = (moreToggle ? moreToggle.offsetWidth + 30 : 0);
            if (used + w + reserved > containerWidth) {
                item.classList.add('is-overflow-hidden');
                hiddenCount++;
            } else {
                used += w;
            }
        }

        if (moreToggle) {
            moreToggle.style.display = hiddenCount > 0 ? '' : 'none';
        }
    };

    recalc();
    window.addEventListener('resize', () => {
        window.requestAnimationFrame(recalc);
    });

    // Open desktop mega menu when clicking the ellipsis (fallback to mobile off-canvas)
    moreToggle?.addEventListener('click', () => {
        const mega = document.getElementById('megaMenuOverlay');
        if (mega) {
            const container = mega.querySelector('.mega-menu-panel');
            mega.classList.remove('d-none');
            requestAnimationFrame(() => {
                container?.classList.add('show');
            });
            return;
        }
        const opener = document.querySelector('.menu-toggle, #menu-toggle');
        opener?.dispatchEvent(new Event('click'));
    });
});

document.addEventListener('DOMContentLoaded', function () {
    const nav = document.getElementById('mainNavigation');
    if (!nav) return;

    const moreToggle = document.getElementById('navMoreToggle');

    const recalc = () => {
        const containerWidth = nav.parentElement?.clientWidth || nav.clientWidth;
        const items = Array.from(nav.querySelectorAll(':scope > li.menu-item'));
        // Reset first
        items.forEach(i => i.classList.remove('is-overflow-hidden'));

        let used = 0;
        let hiddenCount = 0;
        for (const item of items) {
            const w = item.offsetWidth + 30; // include margin-right heuristic
            const reserved = (moreToggle ? moreToggle.offsetWidth + 30 : 0);
            if (used + w + reserved > containerWidth) {
                item.classList.add('is-overflow-hidden');
                hiddenCount++;
            } else {
                used += w;
            }
        }

        if (moreToggle) {
            moreToggle.style.display = hiddenCount > 0 ? '' : 'none';
        }
    };

    recalc();
    window.addEventListener('resize', () => {
        window.requestAnimationFrame(recalc);
    });

    // Open desktop mega menu when clicking the ellipsis (fallback to mobile off-canvas)
    moreToggle?.addEventListener('click', () => {
        const mega = document.getElementById('megaMenuOverlay');
        if (mega) {
            const container = mega.querySelector('.mega-menu-panel');
            mega.classList.remove('d-none');
            requestAnimationFrame(() => {
                container?.classList.add('show');
            });
            return;
        }
        const opener = document.querySelector('.menu-toggle, #menu-toggle');
        opener?.dispatchEvent(new Event('click'));
    });
});

    if (!window.flyBannericonLoaded) {

        function loadFlyBannericonBanners() {
            fetch('/banners/position/flyBannericon')
                .then(response => {
                    if (!response.ok) {
                        throw new Error('Network response was not ok');
                    }
                    return response.json();
                })
                .then(data => {
                    if (data.success && data.data && data.data.length > 0) {
                        // Only process the first active flyBannericon banner to avoid duplicates
                        const activeBanner = data.data[0];
                        createFlyBannericonContainer(activeBanner);
                    }
                })
                .catch(error => {
                    console.error('Error loading flyBannericon banners:', error);
                });
        }

        function createFlyBannericonContainer(banner) {
            // Get banner image URL using global function
            const imageUrl = getBannerImageURL(banner, '', false);
            // Create a single container that flyBannericon.js will use
            const container = document.createElement('div');
            container.id = 'flyBannericon-container-' + banner.id;
            container.setAttribute('data-banner-id', banner.id);

            // Inject banner data for flyBannericon.js to use
            window.flyBannericonData = [{
                id: banner.id,
                img: banner.image ? imageUrl : '',
                url: banner.link || '#',
                type_web: 1,
                title: banner.title || banner.name || '',
                open_new_tab: banner.open_new_tab || false
            }];

            // Add the script content directly if it's a script type banner
            if (banner.content && banner.content.trim()) {
                const scriptElement = document.createElement('script');
                scriptElement.innerHTML = banner.content;
                container.appendChild(scriptElement);
            } else {
                // Use default flyBannericon.js with injected data
                const scriptElement = document.createElement('script');
                scriptElement.src = '/assets/js/flyBannericon.js';
                scriptElement.onload = function() {
                    // Track banner view after script loads
                    if (typeof trackBannerView === 'function') {
                        trackBannerView(banner.id);
                    }
                };
                container.appendChild(scriptElement);
            }

            document.body.appendChild(container);
        }
    }
