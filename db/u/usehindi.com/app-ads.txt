// Check if there are items in the cart stored in localStorage
function getCart() {
const cart = JSON.parse(localStorage.getItem(`cart`)) || [];
return cart;
}

// Save the cart to localStorage
function saveCart(cart) {
localStorage.setItem(`cart`, JSON.stringify(cart));
}

// Add an item to the cart
function addToCart(productName, price) {
const cart = getCart(); // Get the current cart

// Check if the product is already in the cart
const productIndex = cart.findIndex(item => item.name === productName);

if (productIndex > -1) {
// If item exists, update the quantity
cart[productIndex].quantity = 1; // +=
} else {
// If item doesn`t exist, add new item to cart
cart.push({
name: productName,
price: price,
quantity: 1
});
}

saveCart(cart); // Save the updated cart to localStorage
renderCart(); // Re-render the cart UI
}

// Remove an item from the cart
function removeFromCart(productName) {
let cart = getCart();
cart = cart.filter(item => item.name !== productName); // Remove item from the cart
saveCart(cart); // Save the updated cart to localStorage
renderCart(); // Re-render the cart UI
}

// Calculate total quantity and total price
function calculateTotals(cart) {
let totalQuantity = 0;
let totalPrice = 0;

cart.forEach(item => {
totalQuantity += item.quantity;
totalPrice += item.price * item.quantity;
});

return { totalQuantity, totalPrice };
}
