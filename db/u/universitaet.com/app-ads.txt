            let searchWrapper = $('.search-wrapper');
            
            $(window).scroll(function(){
                let ScrollTop = parseInt($(window).scrollTop());
                if (x.matches) {
                    let searchHeight = searchWrapper.height();
                    if (ScrollTop >= searchHeight + 80) {
                        $('#main-heading').css('display', 'none');
                        searchWrapper.insertBefore($('.menu-items')).addClass('searchMove');
                        $('.search-fields-wrapper .bootstrap-select .dropdown-toggle .filter-option').css({
                            'font-size': '1vw',
                            'margin': '7px 0px 0 10px'
                        });
                        $('.fa-search').css('font-size', '1.3vw');
                        $('#interest_test').css('display', 'none');
                        $('.more-details').css('display', 'none');
                        $('#advanced-search-div').css('display', 'none');
                        $('.fa-plus').removeClass('rotated-item');

                    } else {
                        searchWrapper.insertBefore($('.statistics')).removeClass('searchMove');
                        $('.search-fields-wrapper .bootstrap-select .dropdown-toggle .filter-option').css({
                            'font-size': '2vw',
                            'margin': '0'
                        });
                        $('.fa-search').css('font-size', '2vw');

                        $('.more-details').css('display', 'block');
                        searchWrapper.css({'position': 'inherit', 'width': '100%', 'margin-top': '0'});
                        $('#main-heading').css('display', 'block');

                }
            });
