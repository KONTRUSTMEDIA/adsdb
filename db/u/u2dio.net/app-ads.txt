    <script>
      (function () {
        function translatePage(lang) {
          const table = (window.Translations && window.Translations[lang])
            ? window.Translations[lang]
            : (window.Translations ? window.Translations['en'] : null);

          if (!table) return;

          document.querySelectorAll("[data-i18n]").forEach((el) => {
            const key = el.getAttribute("data-i18n");
            if (table[key]) {
              if (el.tagName === "INPUT" || el.tagName === "TEXTAREA") {
                el.placeholder = table[key];
              } else {
                el.innerHTML = table[key];
              }
            }
          });

          if (table.logo_sub) {
            const titleEl = document.querySelector("title");
            if (titleEl) titleEl.innerText = `Future Code - ${table.logo_sub}`;
          }
        }

        function applyActiveLang(lang) {
          document.documentElement.setAttribute("lang", lang);
          document.querySelectorAll(".lang-switcher .lang-link").forEach((a) => {
            const on = a.dataset.lang === lang;
            a.classList.toggle("active", on);
            if (on) a.setAttribute("aria-current", "true");
            else a.removeAttribute("aria-current");
          });
          translatePage(lang);
        }

        window.addEventListener("DOMContentLoaded", () => {
          // URL 경로를 최우선으로 언어 감지 (미들웨어 fallback 상황 대비 위함)
          let lang = document.documentElement.getAttribute("lang");
          const path = window.location.pathname;
          const parts = path.split('/').filter(p => p);
          const SUPPORTED = ['ko', 'en', 'ja', 'es', 'pt', 'de'];

          // 1. URL에 명시된 언어가 있으면 최우선 (미들웨어가 lang 속성을 못 바꿨을 경우를 대비)
          if (parts.length > 0 && SUPPORTED.includes(parts[0])) {
            lang = parts[0];
          }
          // 2. 아니면 기존 lang 속성 또는 기본값 ko
          if (!lang) lang = "ko";

          applyActiveLang(lang);

          if (window.App && typeof window.App.changeLanguage === "function") {
            const orig = window.App.changeLanguage.bind(window.App);
            window.App.changeLanguage = function (l) {
              applyActiveLang(l);
              return orig(l);
            };
          }
        });
      })();
    </script>

      (function () {
        function resumeAutoAds() {
          try {
            if (window.adsbygoogle && typeof window.adsbygoogle === "object") {
              // 동의 로직에서 pauseAdRequests=1로 멈춰둔 경우를 대비해 재개
              if ("pauseAdRequests" in window.adsbygoogle) {
                window.adsbygoogle.pauseAdRequests = 0;
              }
              // 광고 푸시 (자동 광고 단위는 이미 존재할 경우 무시됨)
              (window.adsbygoogle = window.adsbygoogle || []).push({});
            }
          } catch (e) {
            console.error("AdSense resume error:", e);
          }
        }

        function patchAppOnce() {
          if (!window.App) return;
          const methods = ["generateCodeFromSelections", "decodeExistingCode", "generateNewCode", "reset", "goHome", "goToGuide", "changeLanguage"];
          methods.forEach(function (m) {
            const fn = window.App[m];
            if (typeof fn !== "function" || fn.__u2dio_ads_patched) return;

            window.App[m] = function () {
              const r = fn.apply(this, arguments);
              setTimeout(resumeAutoAds, 100);
              return r;
            };
            window.App[m].__u2dio_ads_patched = true;
          });
        }

        window.addEventListener("DOMContentLoaded", function () {
          patchAppOnce();
          setTimeout(patchAppOnce, 500);
        });

        // 커스텀 이벤트 연동
        ["u2dio:consent", "consent:accepted", "cmp:consent", "u2dio:viewchange"].forEach(evName => {
          window.addEventListener(evName, resumeAutoAds);
        });
      })();
