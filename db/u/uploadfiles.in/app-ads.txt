<script>
(function () {
  // ===== CONFIG =====
  var AD_UNIT_PATH = "/23200296342/Uploadfiles";
  var SLOT_DIV_ID = "gpt-passback";
  var CONTAINER_ID = "gpt-sticky-footer";

  // Lazy load without scroll
  var LOAD_AFTER_MS = 1500;        // delay before first load
  var LOAD_ON_FIRST_INTERACTION = true;

  // Refresh rules
  var REFRESH_EVERY_MS = 30000;
  var MIN_INVIEW_RATIO = 0.50;
  var MAX_REFRESHES = 8;

  // ===== ELEMENTS =====
  var container = document.getElementById(CONTAINER_ID);

  function isTabVisible() {
    return document.visibilityState === "visible";
  }

  function showSticky() {
    if (!document.getElementById(CONTAINER_ID)) return;
    container.classList.add("is-visible");
    document.body.classList.add("has-sticky-footer-ad");
  }

  function hideSticky() {
    if (!document.getElementById(CONTAINER_ID)) return;
    container.classList.remove("is-visible");
    document.body.classList.remove("has-sticky-footer-ad");
  }

  // ===== VIEWABILITY TRACKING =====
  var inViewRatio = 0;
  var viewObs = new IntersectionObserver(function (entries) {
    entries.forEach(function (e) {
      if (e.target && e.target.id === CONTAINER_ID) {
        inViewRatio = e.intersectionRatio || 0;
      }
    });
  }, { threshold: [0, 0.25, 0.5, 0.75, 1] });
  viewObs.observe(container);

  // ===== GPT SETUP =====
  window.googletag = window.googletag || { cmd: [] };
  var slot = null;
  var slotDefined = false;
  var renderedOnce = false;

  var refreshCount = 0;
  var lastRefreshTs = 0;

  function defineAndDisplaySlot() {
    if (slotDefined) return;
    slotDefined = true;

    googletag.cmd.push(function () {
      var mapping = googletag.sizeMapping()
        .addSize([1024, 0], [[970, 90], [728, 90]])
        .addSize([768, 0],  [[728, 90]])
        .addSize([0, 0],    [[320, 100], [320, 50], [300, 50]])
        .build();

      slot = googletag.defineSlot(
        AD_UNIT_PATH,
        [[970, 90], [728, 90], [320, 100], [320, 50], [300, 50]],
        SLOT_DIV_ID
      )
      .defineSizeMapping(mapping)
      .addService(googletag.pubads());

      googletag.pubads().enableSingleRequest();
      googletag.pubads().collapseEmptyDivs(true);

      // Hide sticky if no ad is returned
      googletag.pubads().addEventListener("slotRenderEnded", function (event) {
        if (event.slot === slot) {
          renderedOnce = true;
          lastRefreshTs = Date.now();
          if (event.isEmpty) {
            hideSticky();
          } else {
            showSticky();
          }
        }
      });

      googletag.enableServices();
      googletag.display(SLOT_DIV_ID);
    });
  }

  // ===== LAZY LOAD (NO SCROLL) =====
  var loadTriggered = false;

  function triggerLoadOnce() {
    if (loadTriggered) return;
    loadTriggered = true;
    defineAndDisplaySlot();
  }

  // Load after delay
  setTimeout(triggerLoadOnce, LOAD_AFTER_MS);

  // Or load earlier on first interaction
  if (LOAD_ON_FIRST_INTERACTION) {
    ["pointerdown", "keydown", "touchstart", "mousemove"].forEach(function (evt) {
      window.addEventListener(evt, triggerLoadOnce, { passive: true, once: true });
    });
  }

  // ===== REFRESH LOOP =====
  function tryRefresh() {
    if (!slot || !renderedOnce) return;
    if (!document.getElementById(CONTAINER_ID)) return;
    if (!isTabVisible()) return;
    if (refreshCount >= MAX_REFRESHES) return;
    if (inViewRatio < MIN_INVIEW_RATIO) return;

    var now = Date.now();
    if (now - lastRefreshTs < REFRESH_EVERY_MS) return;

    googletag.cmd.push(function () {
      googletag.pubads().refresh([slot]);
      refreshCount += 1;
      lastRefreshTs = now;
    });
  }

  setInterval(tryRefresh, 1000);

})();
</script>
