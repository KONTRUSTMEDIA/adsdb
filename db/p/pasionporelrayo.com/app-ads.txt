<!-- Validación de formulario para evitar enlaces y correos falsos -->
<script>
document.addEventListener('DOMContentLoaded', function () {
  const form = document.querySelector('#fluentform_4');
  if (!form) return;

  const emailField = form.querySelector('input[type="email"]');
  const messageField = form.querySelector('textarea[name="description"]');
  const blockedDomains = ['fake.com', 'test.com', 'noexiste.com', 'correo.com', 'mailinator.com'];
  const urlRegex = /(https?:\/\/|www\.)[^\s]+/i;

  form.addEventListener('submit', function (e) {
    const email = emailField.value.trim();
    const domain = email.split('@')[1];
    const message = messageField.value.trim();

    if (blockedDomains.includes(domain)) {
      e.preventDefault();
      alert('Por favor, introduce un correo electrónico real. El dominio que has usado está bloqueado.');
      emailField.focus();
      return;
    }

    if (urlRegex.test(message)) {
      e.preventDefault();
      alert('Por seguridad, no está permitido incluir enlaces en el mensaje.');
      messageField.focus();
      return;
    }
  });
});
</script>

<!-- Script para flechas en barra de escudos -->
<script>
document.addEventListener('DOMContentLoaded', function() {

    const barra = document.querySelector('.liga-escudos-barra');
    const arrowLeft = document.querySelector('.left-arrow');
    const arrowRight = document.querySelector('.right-arrow');

    if (!barra) return;

    const updateArrows = () => {
        const scrollLeft = barra.scrollLeft;
        const maxScroll = barra.scrollWidth - barra.clientWidth;

        if (scrollLeft <= 5) {
            arrowLeft.style.opacity = "0";
            arrowLeft.style.pointerEvents = "none";
        } else {
            arrowLeft.style.opacity = "1";
            arrowLeft.style.pointerEvents = "auto";
        }

        if (scrollLeft >= maxScroll - 5) {
            arrowRight.style.opacity = "0";
            arrowRight.style.pointerEvents = "none";
        } else {
            arrowRight.style.opacity = "1";
            arrowRight.style.pointerEvents = "auto";
        }
    };

    arrowLeft.addEventListener('click', () => {
        barra.scrollBy({ left: -150, behavior: 'smooth' });
    });

    arrowRight.addEventListener('click', () => {
        barra.scrollBy({ left: 150, behavior: 'smooth' });
    });

    barra.addEventListener('scroll', updateArrows);

    updateArrows();
});
</script>
