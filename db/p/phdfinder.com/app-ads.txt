<script>
let selectedTerms = [];

function addTerm(term) {
  if (!selectedTerms.includes(term)) {
    selectedTerms.push(term);
    updateSelectedDisplay();
  }
}

function removeTerm(term) {
  selectedTerms = selectedTerms.filter(t => t !== term);
  updateSelectedDisplay();
}

function updateSelectedDisplay() {
  const container = document.getElementById('selectedTerms');
  container.innerHTML = '';
  selectedTerms.forEach(term => {
    const el = document.createElement('span');
    el.className = 'selected-term';
    el.innerHTML = term + '<span onclick="removeTerm(\'' + term + '\')">✕</span>';
    container.appendChild(el);
  });
}

function runSearch() {
  const keyword = document.getElementById('searchInput').value.trim();
  const terms = [...selectedTerms];
  if (keyword) terms.push(keyword);
  const query = terms.join('+');
  if (query) {
    window.location.href = '/?s=' + encodeURIComponent(query);
  }
}
