    // 날짜에 따른 버튼 표시 제어
    function checkButtonVisibility() {
      const today = new Date();
      const currentYear = today.getFullYear();
      
      // 버튼 표시 설정 객체
      const buttonConfigs = {
        'modal1-btn': {
          showDate: new Date(currentYear, 9, 21), // 10월 21일
          hideDate: new Date(currentYear, 11, 1)  // 12월 1일
        },
        // 'modal2-btn': {
        //   showDate: new Date(currentYear, 9, 21), // 10월 21일
        //   hideDate: new Date(currentYear, 11, 1)  // 12월 1일
        // }
      };

      // 각 버튼에 대해 표시/숨김 처리
      Object.entries(buttonConfigs).forEach(([buttonId, config]) => {
        const button = document.getElementById(buttonId);
        if (!button) return; // 버튼이 존재하지 않으면 건너뛰기
        
        const isInDisplayPeriod = today >= config.showDate && today < config.hideDate;
        button.style.display = isInDisplayPeriod ? 'inline-block' : 'none';
      });
    }

    // 페이지 로드 시 버튼 표시 상태 확인
    document.addEventListener('DOMContentLoaded', function() {
      checkButtonVisibility();
    });

    // 매일 자정에 버튼 표시 상태 재확인 (선택사항)
    setInterval(function() {
      checkButtonVisibility();
    }, 60000 * 60 * 24); // 24시간마다 확인
