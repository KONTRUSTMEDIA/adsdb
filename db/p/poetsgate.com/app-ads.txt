<script>
  (function($) {
    $(document).ready(function() {
      loadFriendlySites();

      function normalizeUrl(url) {
        if (!/^https?:\/\//i.test(url)) {
          return 'https://' + url;
        }
        return url;
      }

      function loadFriendlySites() {
        $('#friendly-sites-error').addClass('d-none');
        $('#friendly-sites-empty').addClass('d-none');
        $('#friendly-sites-loading').removeClass('d-none');

        $.ajax({
          url: 'GetFriendlySites_jQury.php',
          type: 'POST',
          dataType: 'json',
          cache: false,
          data: { SiteStatus: 1 }
        }).done(function(response) {
          const $list = $('#friendly-sites').empty();

          if (!Array.isArray(response) || response.length === 0) {
            $('#friendly-sites-empty').removeClass('d-none');
            return;
          }

          response.forEach(function(item) {
            // حماية خفيفة ضد قيم null/undefined
            const name = (item.SiteName || '').toString().trim();
            const addr = (item.SiteAddress || '').toString().trim();
            if (!name || !addr) return;

            const href = normalizeUrl(addr);

            // عنصر Bootstrap 5 أنيق وقابل للنقر
            const $li = $(`
              <li class="list-group-item d-flex align-items-center justify-content-between">
                <span class="text-truncate"></span>
                <a class="stretched-link ms-2" target="_blank" rel="noopener noreferrer">
                  <i class="bi bi-box-arrow-up-right"></i>
                </a>
              </li>
            `);

            // استخدم text() لتجنب إدخال HTML
            $li.find('span').text(name);
            $li.find('a').attr('href', href).attr('title', name);

            $list.append($li);
          });

          // إذا بعد الفلترة لا يوجد عناصر
          if ($list.children().length === 0) {
            $('#friendly-sites-empty').removeClass('d-none');
          }
        }).fail(function() {
          $('#friendly-sites-error').removeClass('d-none');
        }).always(function() {
          $('#friendly-sites-loading').addClass('d-none');
        });
      }
    });
  })(jQuery);
</script>

<script>
  (function($){
    $(document).ready(function () {
      loadFriendlySitesSide();

      function normalizeUrl(url){
        if(!/^https?:\/\//i.test(url)) return 'https://' + url;
        return url;
      }

      function loadFriendlySitesSide(){
        $('#friendly-side-error').addClass('d-none');
        $('#friendly-side-empty').addClass('d-none');
        $('#friendly-side-loading').removeClass('d-none');

        $.ajax({
          url: 'GetFriendlySites_jQury.php',
          type: 'POST',
          dataType: 'json',
          cache: false,
          data: { SiteStatus: 1 }
        })
        .done(function(response){
          const $list = $('.SiteList').empty();

          if(!Array.isArray(response) || response.length === 0){
            $('#friendly-side-empty').removeClass('d-none');
            return;
          }

          response.forEach(function(item){
            const name = (item.SiteName || '').toString().trim();
            const addr = (item.SiteAddress || '').toString().trim();
            if(!name || !addr) return;

            const href = normalizeUrl(addr);

            const $li = $(`
              <li class="list-group-item py-2 px-3 text-start">
                <a class="link-primary text-decoration-none text-truncate d-inline-block w-100"
                   target="_blank" rel="noopener noreferrer"></a>
              </li>
            `);

            $li.find('a').text(name).attr('href', href).attr('title', name);
            $list.append($li);
          });

          if($list.children().length === 0){
            $('#friendly-side-empty').removeClass('d-none');
          }
        })
        .fail(function(){
          $('#friendly-side-error').removeClass('d-none');
        })
        .always(function(){
          $('#friendly-side-loading').addClass('d-none');
        });
      }
    });
  })(jQuery);
</script>

function compactNumberSmart(value) {
  const num = Number(String(value).replace(/[^\d.-]/g, ''));
  if (!Number.isFinite(num)) return '0';

  const abs = Math.abs(num);
  if (abs >= 1_000_000_000) return (num / 1_000_000_000).toFixed(2) + ' B ';
  if (abs >= 1_000_000)     return (num / 1_000_000).toFixed(2)     + ' M ';
  if (abs >= 1_000)         return (num / 1_000).toFixed(2)         + ' K ';

  // أقل من 1000 → بدون كسور
  return new Intl.NumberFormat('en-US', { maximumFractionDigits: 0 }).format(num);
}

// للـ title (القيمة الكاملة) بدون كسور للقيم الصغيرة، ومع فواصل ألوف
function formatFullIntlSmart(value) {
  const num = Number(String(value).replace(/[^\d.-]/g, ''));
  if (!Number.isFinite(num)) return '0';
  return new Intl.NumberFormat('en-US', {
    minimumFractionDigits: 0,
    maximumFractionDigits: 2
  }).format(num);
}

  if (!el) return;
  const full  = formatFullIntlSmart(rawValue ?? 0);
  const short = compactNumberSmart(rawValue ?? 0);
  el.textContent = short;
  el.title = full;        
  el.dataset.full = full;
}
