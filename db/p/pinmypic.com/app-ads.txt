                                        const rating = Math.max(0, Math.min(5, Number(val) || 0));
                                        const full = Math.round(rating); // simple rounding
                                        return `
        <span class="stars" aria-label="Rating ${rating} dari 5">
          ${Array.from({ length: 5 }).map((_, i) => `<span class="star ${i < full ? '' : 'off'}"></span>`).join('')}
        </span>`;
                                    };

                                    const fmtDate = (iso) => {
                                        if (!iso) return "";
                                        const d = new Date(iso);
                                        if (isNaN(d)) return iso;
                                        return d.toLocaleDateString("id-ID", { year: "numeric", month: "short", day: "2-digit" });
                                    };

                                    const cardsHTML = reviews.map(r => {
                                        const author = r?.author?.name ?? "Anonim";
                                        const title = r?.name ?? "Review";
                                        const body = r?.reviewBody ?? "";
                                        const date = fmtDate(r?.datePublished);
                                        const rating = r?.reviewRating?.ratingValue ?? 5;

                                        return `
        <article class="review-card">
          <div class="review-card__top">
            <p class="review-card__name">${author}</p>
            <div class="review-card__meta">
              ${makeStars(rating)}
              <span aria-label="Tanggal">${date}</span>
            </div>
          </div>
          <h3 class="review-card__title">${title}</h3>
