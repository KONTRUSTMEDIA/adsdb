      // Запоминаем выбор
      localStorage.setItem('pyp-lang', lang);

      // Сверху страницы
      window.scrollTo({top:0, behavior:'smooth'});
    }

    const saved = localStorage.getItem('pyp-lang');
    if(saved){
      setLang(saved);
    }else{
      const n = (navigator.language||'').toLowerCase();
      setLang(n.startsWith('es') ? 'es' : n.startsWith('en') ? 'en' : 'ru');
    }
