<script>
  (function () {
    const toggleBtn = document.getElementById('navToggle');
    const mobileMenu = document.getElementById('mobileMenu');
    const menuGroup = document.querySelector('.menu-group');
    const submenu = menuGroup ? menuGroup.querySelector('.submenu') : null;
    const notesGroup = document.querySelector('.notes-group');
    const notesMenu = notesGroup ? notesGroup.querySelector('.notes-menu') : null;
    const practiceGroup = document.querySelector('.practice-group');
    const practiceMenu = practiceGroup ? practiceGroup.querySelector('.practice-menu') : null;
    const icon = document.getElementById('navToggleIcon');

    if (!toggleBtn || !mobileMenu || !icon) return;

    toggleBtn.addEventListener('click', () => {
      mobileMenu.classList.toggle('hidden');
      const open = !mobileMenu.classList.contains('hidden');
      icon.innerHTML = open
        ? '<path stroke-linecap=\"round\" stroke-linejoin=\"round\" d=\"M6 18L18 6M6 6l12 12\" />'
        : '<path stroke-linecap=\"round\" stroke-linejoin=\"round\" d=\"M4 6h16M4 12h16M4 18h16\" />';
    });

    const setExpanded = (button, state) => {
      if (button) {
        button.setAttribute('aria-expanded', state ? 'true' : 'false');
      }
    };

    if (menuGroup && submenu) {
      let closeTimeout = null;
      const triggerButton = menuGroup.querySelector('button');

      const openMenu = () => {
        if (closeTimeout) {
          clearTimeout(closeTimeout);
          closeTimeout = null;
        }
        submenu.classList.remove('hidden');
        setExpanded(triggerButton, true);
      };

      const scheduleClose = () => {
        if (closeTimeout) {
          clearTimeout(closeTimeout);
        }
        closeTimeout = setTimeout(() => {
          submenu.classList.add('hidden');
          setExpanded(triggerButton, false);
        }, 150);
      };

      menuGroup.addEventListener('mouseenter', openMenu);
      menuGroup.addEventListener('mouseleave', scheduleClose);
      submenu.addEventListener('mouseenter', openMenu);
      submenu.addEventListener('mouseleave', scheduleClose);

      if (triggerButton) {
        triggerButton.addEventListener('click', (event) => {
          event.preventDefault();
          if (submenu.classList.contains('hidden')) {
            openMenu();
          } else {
            submenu.classList.add('hidden');
            setExpanded(triggerButton, false);
          }
        });
      }

      menuGroup.addEventListener('focusin', openMenu);
      menuGroup.addEventListener('focusout', (event) => {
        if (!menuGroup.contains(event.relatedTarget)) {
          scheduleClose();
        }
      });
    }

    if (notesGroup && notesMenu) {
      let notesCloseTimeout = null;
      const notesButton = notesGroup.querySelector('button');

      const openNotes = () => {
        if (notesCloseTimeout) {
          clearTimeout(notesCloseTimeout);
          notesCloseTimeout = null;
        }
        notesMenu.classList.remove('hidden');
        setExpanded(notesButton, true);
      };

      const scheduleNotesClose = () => {
        if (notesCloseTimeout) {
          clearTimeout(notesCloseTimeout);
        }
        notesCloseTimeout = setTimeout(() => {
          notesMenu.classList.add('hidden');
          setExpanded(notesButton, false);
        }, 150);
      };

      notesGroup.addEventListener('mouseenter', openNotes);
      notesGroup.addEventListener('mouseleave', scheduleNotesClose);
      notesMenu.addEventListener('mouseenter', openNotes);
      notesMenu.addEventListener('mouseleave', scheduleNotesClose);

      if (notesButton) {
        notesButton.addEventListener('click', (event) => {
          event.preventDefault();
          if (notesMenu.classList.contains('hidden')) {
            openNotes();
          } else {
            notesMenu.classList.add('hidden');
            setExpanded(notesButton, false);
          }
        });
      }
    }

    if (practiceGroup && practiceMenu) {
      let practiceCloseTimeout = null;
      const practiceButton = practiceGroup.querySelector('button');

      const openPractice = () => {
        if (practiceCloseTimeout) {
          clearTimeout(practiceCloseTimeout);
          practiceCloseTimeout = null;
        }
        practiceMenu.classList.remove('hidden');
        setExpanded(practiceButton, true);
      };

      const schedulePracticeClose = () => {
        if (practiceCloseTimeout) {
          clearTimeout(practiceCloseTimeout);
        }
        practiceCloseTimeout = setTimeout(() => {
          practiceMenu.classList.add('hidden');
          setExpanded(practiceButton, false);
        }, 150);
      };

      practiceGroup.addEventListener('mouseenter', openPractice);
      practiceGroup.addEventListener('mouseleave', schedulePracticeClose);
      practiceMenu.addEventListener('mouseenter', openPractice);
      practiceMenu.addEventListener('mouseleave', schedulePracticeClose);

      if (practiceButton) {
        practiceButton.addEventListener('click', (event) => {
          event.preventDefault();
          if (practiceMenu.classList.contains('hidden')) {
            openPractice();
          } else {
            practiceMenu.classList.add('hidden');
            setExpanded(practiceButton, false);
          }
        });
      }
    }
  })();
</script>

  <script>
    document.addEventListener('DOMContentLoaded', function () {
      const container = document.getElementById('headerSearchContainer');
      const navBar = document.getElementById('mainNav');
      const toggle = document.getElementById('globalSearchToggle');
      const input = document.getElementById('headerSearchInput');
      const dropdown = document.getElementById('headerSearchResults');
      const list = document.getElementById('headerSearchList');
      const emptyState = document.getElementById('headerSearchEmpty');
      const mobileOverlay = document.getElementById('mobileSearchOverlay');
      const mobileOverlayInput = document.getElementById('mobileSearchInput');
      const mobileChips = document.querySelector('.mobile-search-panel__chips');
      const mobileOverlaySuggestions = document.getElementById('mobileSearchSuggestions');
      const mobileOverlayClose = document.getElementById('mobileSearchOverlayClose');
      const endpoint = '/Pathanto/search-suggestions.php';
      let debounce;

      if (!container || !toggle || !input || !dropdown || !list || !emptyState) {
        return;
      }

      const helper = document.getElementById('headerSearchHelper');
      const isMobile = () => window.matchMedia('(max-width: 1023px)').matches;
      const showHelper = () => helper?.classList.remove('hidden');
      const hideHelper = () => helper?.classList.add('hidden');

      const open = () => {
        container.classList.add('active');
        document.body.classList.add('search-open');
        navBar?.classList.add('search-active');
        input.focus();
        if (input.value.trim()) {
          fetchSuggestions(input.value.trim());
        }
      };

      const close = () => {
        container.classList.remove('active');
        document.body.classList.remove('search-open');
        navBar?.classList.remove('search-active');
        dropdown.classList.add('hidden');
      };

      const renderResults = (items) => {
        if (!items.length) {
          emptyState.classList.remove('hidden');
          list.innerHTML = '';
          dropdown.classList.remove('hidden');
          if (isMobile()) {
            showHelper();
          }
          return;
        }
        emptyState.classList.add('hidden');
        list.innerHTML = items.map((item) => `
          <a href="${item.url}" class="search-dropdown__card">
            <span class="search-dropdown__pill">${item.label}</span>
            <span class="search-dropdown__title">${item.title}</span>
            <span class="search-dropdown__description">${item.description}</span>
          </a>
        `).join('');
        dropdown.classList.remove('hidden');
        hideHelper();
      };

      const fetchSuggestions = (term) => {
        if (!term) {
          dropdown.classList.add('hidden');
          return;
        }
        fetch(endpoint + '?q=' + encodeURIComponent(term))
          .then((res) => res.json())
          .then((data) => renderResults(Array.isArray(data) ? data : []))
          .catch(() => {
            emptyState.classList.remove('hidden');
            emptyState.textContent = 'Unable to load suggestions.';
            dropdown.classList.remove('hidden');
          });
      };

      const renderMobileSuggestions = (items) => {
        mobileOverlaySuggestions.innerHTML = items.map((item) => `
          <div class="mobile-search-panel__result">
            <a href="${item.url}" class="mobile-search-panel__result-link">${item.title}</a>
            <p class="mobile-search-panel__result-description">${item.description}</p>
          </div>
        `).join('');
      };

      const fetchMobileSuggestions = (term) => {
        if (!term) {
          mobileOverlaySuggestions.innerHTML = '';
          return;
        }
        fetch(endpoint + '?q=' + encodeURIComponent(term))
          .then((res) => res.json())
          .then((data) => renderMobileSuggestions(Array.isArray(data) ? data : []))
          .catch(() => {
            mobileOverlaySuggestions.innerHTML = '<div class="mobile-search-panel__result">Unable to load suggestions.</div>';
          });
      };

      const openMobileOverlay = () => {
        mobileOverlay?.classList.add('active');
        mobileOverlay?.setAttribute('aria-hidden', 'false');
        mobileOverlayInput?.focus();
      };

      const closeMobileOverlay = () => {
        mobileOverlay?.classList.remove('active');
        mobileOverlay?.setAttribute('aria-hidden', 'true');
        mobileOverlaySuggestions.innerHTML = '';
        toggleMobileChips('');
        if (mobileOverlayInput) {
          mobileOverlayInput.value = '';
        }
      };

      toggle?.addEventListener('click', (event) => {
        event.preventDefault();
        if (container.classList.contains('active')) {
          close();
        } else {
          open();
        }
      });
      const mobileToggle = document.getElementById('mobileSearchToggle');
      mobileToggle?.addEventListener('click', (event) => {
        event.preventDefault();
        openMobileOverlay();
      });
      mobileOverlayClose?.addEventListener('click', (event) => {
        event.preventDefault();
        closeMobileOverlay();
      });
      mobileOverlay?.addEventListener('click', (event) => {
        if (event.target === mobileOverlay) {
          closeMobileOverlay();
        }
      });
      document.querySelectorAll('.mobile-search-panel__chips .chip').forEach((chip) => {
        chip.addEventListener('click', () => {
          const query = chip.getAttribute('data-query') || '';
          if (mobileOverlayInput) {
            mobileOverlayInput.value = query;
          }
          toggleMobileChips(query);
          fetchMobileSuggestions(query);
        });
      });

      input?.addEventListener('input', (event) => {
        clearTimeout(debounce);
        debounce = setTimeout(() => fetchSuggestions(event.target.value.trim()), 250);
      });
      const toggleMobileChips = (value) => {
        if (!mobileChips) {
          return;
        }
        if (value.trim().length) {
          mobileChips.classList.add('hidden');
        } else {
          mobileChips.classList.remove('hidden');
        }
      };
      mobileOverlayInput?.addEventListener('input', (event) => {
        const value = event.target.value;
        toggleMobileChips(value);
        clearTimeout(debounce);
        debounce = setTimeout(() => fetchMobileSuggestions(value.trim()), 250);
      });

      document.addEventListener('click', (event) => {
        if (!container.contains(event.target)) {
          close();
        }
      });

      document.addEventListener('keydown', (event) => {
        if (event.key === 'Escape') {
          close();
        }
      });
    });

    document.addEventListener('DOMContentLoaded', function () {
      const toggle = document.getElementById('mobileNavDrawerToggle');
      const drawer = document.getElementById('mobileNavDrawer');
      const mobileWrapper = document.querySelector('.mobile-header');
      const legacyButton = document.getElementById('legacyChapterToggle');

      const updateMobileOffset = () => {
        if (!mobileWrapper) {
          document.body.style.removeProperty('--mobile-nav-offset');
          document.body.style.removeProperty('--legacy-panel-offset');
          return;
        }
        if (window.innerWidth > 990) {
          document.body.style.setProperty('--mobile-nav-offset', '0px');
          document.body.style.removeProperty('--legacy-panel-offset');
          return;
        }
        const offset = mobileWrapper.offsetHeight;
        document.body.style.setProperty('--mobile-nav-offset', offset + 'px');
        document.body.style.setProperty('--legacy-panel-offset', offset + 'px');
      };

      const toggleLegacyButtonVisibility = () => {
        if (!legacyButton) {
          return;
        }
        const legacyPanel = document.querySelector('.panel');
        legacyButton.style.display = legacyPanel ? 'inline-flex' : 'none';
      };

      updateMobileOffset();
      toggleLegacyButtonVisibility();
      window.addEventListener('resize', updateMobileOffset);

      if (!toggle || !drawer) {
        return;
      }

      const closeDrawer = () => {
        drawer.classList.remove('is-open');
        toggle.classList.remove('is-open');
        toggle.setAttribute('aria-expanded', 'false');
      };

      toggle.addEventListener('click', function (event) {
        event.stopPropagation();
        drawer.classList.toggle('is-open');
        const expanded = drawer.classList.contains('is-open');
        toggle.setAttribute('aria-expanded', expanded ? 'true' : 'false');
        toggle.classList.toggle('is-open', expanded);
        updateMobileOffset();
      });

      document.addEventListener('click', function (event) {
        if (!drawer.classList.contains('is-open')) {
          return;
        }
        if (event.target.closest('.mobile-nav__drawer') || event.target.closest('#mobileNavDrawerToggle')) {
          return;
        }
        closeDrawer();
      });
    });
  </script>
