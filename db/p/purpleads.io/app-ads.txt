function getSourceFromQuery(query) {
  if(query.purple_source) {
    return query.purple_source;
  } else if (query.utm_source) {
    return query.utm_source;
  } else if (query.gclid) {
    return 'adwords';
  } else if (query.fbclid) {
    return 'facebook';
  } else if (query.ref) {
    return query.ref;
  }
  return null;
}

function getQueryParams() {
	return window.location.search.substring(1);

function getQueryAsObject() {
	if (window.location.search.substring(1).length > 0) {

			return prev;
		}, {});
	}
	return {};
}
