	function carouselPro(thisBtn,type){
		var carouselScroll = thisBtn.parents('.pro-list-scroll');
		var carouselBox = thisBtn.siblings('.pro-list-box');
		var carouselContent = carouselBox.find('.pro-list');
		var carouselItem = carouselContent.find('.pro-item');
		var carouselBoxW = carouselBox.width();
		var carouselContentW = carouselContent.width();
		var carouselItemW = carouselItem.outerWidth();
		var carouselLeft = carouselContent.css('margin-left').replace('px','') * 1;
		var carouselRight = carouselContent.css('margin-right').replace('px','') * 1;
		var carouselDistance = carouselBoxW - carouselItemW;
		var carouselLeftMin = carouselBoxW - carouselContentW;
		var carouselRightMin = carouselBoxW - carouselContentW;
		var prevBtn = carouselScroll.find('.carousel-btn.prev');
		var nextBtn = carouselScroll.find('.carousel-btn.next');
		if(type == 'next'){
			var newLeft = carouselLeft - carouselDistance;
			var newRight = carouselRight - carouselDistance;
			if(newLeft < carouselLeftMin){
				newLeft = carouselLeftMin;
			}
			if(newRight < carouselRightMin){
				newRight = carouselRightMin;
			}
		}else if(type == 'prev'){
			var newLeft = carouselLeft + carouselDistance;
			var newRight = carouselRight + carouselDistance;
			if(newLeft > 0){
				newLeft = 0;
			}
			if(newRight > 0){
				newRight = 0;
			}
		}
		if("ID" == 'AR'){
			carouselContent.css('margin-right', newRight+'px');
			if(newRight == carouselRightMin){
				prevBtn.attr('disabled','disabled');
			}else{
				prevBtn.removeAttr('disabled');
			}
			if(newRight == 0){
				nextBtn.attr('disabled','disabled');
			}else{
				nextBtn.removeAttr('disabled');
			}
		}else{
			carouselContent.css('margin-left', newLeft+'px');
			if(newLeft == 0){
				prevBtn.attr('disabled','disabled');
			}else{
				prevBtn.removeAttr('disabled');
			}
			if(newLeft == carouselLeftMin){
				nextBtn.attr('disabled','disabled');
			}else{
				nextBtn.removeAttr('disabled');
			}
		}
	}
	function scrollbarInt(thisBox){
		var proBoxW = thisBox.find('.pro-list-box').width();
		var proContentW = thisBox.find('.pro-list').width();
		var proItemW = thisBox.find('.pro-item').outerWidth();
		var scrollbarThumbW = proItemW / proContentW * 100 + '%';
		thisBox.find('.scrollbar-thumb').css('width',scrollbarThumbW);
	}
	function scrollbarImitate(thisBox){
		var proBoxW = thisBox.find('.pro-list-box').width();
		var proContentW = thisBox.find('.pro-list').width();
		var proItemW = thisBox.find('.pro-item').outerWidth();
		var scrollW = proContentW - proBoxW;
		var proContentL = thisBox.find('.pro-list-box').scrollLeft();
		var proContentR = 0 - proContentL;
		var scrollbarThumbL = (1 - proItemW / proContentW) * (proContentL / scrollW) * 100 + '%';
		var scrollbarThumbR = (1 - proItemW / proContentW) * (proContentR / scrollW) * 100 + '%';
		if("ID" == 'AR'){
			thisBox.find('.scrollbar-thumb').css('margin-right',scrollbarThumbR);
		}else{
			thisBox.find('.scrollbar-thumb').css('margin-left',scrollbarThumbL);
		}
	}
