<script>
    function updateCountdownSubtle() {
        const now = new Date();
        const tomorrow = new Date();
        tomorrow.setHours(24, 0, 0, 0); 

        const diff = tomorrow - now;

        const hours = Math.floor((diff / (1000 * 60 * 60)) % 24);
        const minutes = Math.floor((diff / 1000 / 60) % 60);
        const seconds = Math.floor((diff / 1000) % 60);

        document.getElementById('hours-s').innerText = hours.toString().padStart(2, '0');
        document.getElementById('minutes-s').innerText = minutes.toString().padStart(2, '0');
        document.getElementById('seconds-s').innerText = seconds.toString().padStart(2, '0');
    }

    // Lógica del botón de recordatorio
    document.getElementById('pwa-remind-btn').addEventListener('click', function() {
        if ("Notification" in window) {
            Notification.requestPermission().then(permission => {
                if (permission === "granted") {
                    this.style.display = 'none';
                    document.getElementById('remind-msg').style.display = 'inline-block';
                } else {
                    alert("Please enable notifications in your browser settings to receive alerts.");
                }
            });
        }
    });

    setInterval(updateCountdownSubtle, 1000);
    updateCountdownSubtle();
</script>
