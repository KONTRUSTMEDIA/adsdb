String.prototype.subcharstr = function() {
	var iStart = (arguments.length > 0 ? arguments[0] : 0);
	var iEnd = (arguments.length > 1 ? arguments[1] : this.length);
	var maxlen = (iEnd-iStart) * 2;

	var tmp = "";
	var chridx = 0;
	for (i = iStart; i < this.length; i++) {
		if (this.charCodeAt(i) > 127) chridx++;
		chridx++;

		if (chridx < maxlen) tmp += this.charAt(i);
		else break;
	}

	if (this != tmp) tmp += "..";
	return tmp;
}
