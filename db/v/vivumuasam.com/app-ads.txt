		function fulltime(){
			var now = new Date();
			var num = now.getDay();
			var day;
			switch(num) {
				case 1:
					day = 'Thứ Hai';break;
				case 2:
					day = 'Thứ Ba';break;
				case 3:
					day = 'Thứ Tư';break;
				case 4:
					day = 'Thứ Năm';break;
				case 5:
					day = 'Thứ Sáu';break;
				case 6:
					day = 'Thứ Bảy';break;
				default:
					day = 'Chủ Nhật';
			}
			var d = now.getDate();
			var m = now.getMonth() + 1;
			var y = now.getFullYear();
			var h = now.getHours();
			var i = now.getMinutes();
			var s = now.getSeconds();
