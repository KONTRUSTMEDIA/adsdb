                                const aIsJapan = startsWithKeyword(a?.value, "asia/japan/");

                                if (aIsJapan && !bIsJapan) return -1;
                                if (!aIsJapan && bIsJapan) return 1;

                                return a.label.localeCompare(b.label);
                            });
                    }
