XF.ready(() =>
{

	let isLoading = false;
	let loader = document.querySelector('.porta-article-loader');
	let loadMoreBtn = loader?.querySelector('.porta-article-button');
	let currentPage = 1;
	const autoLoadPages = 3;
	const maxPages = 14; // Số trang cuối cùng, nên cập nhật nếu thay đổi
		
	function showLoadingIndicator() {
		let loadingDiv = document.createElement('div');
		loadingDiv.className = 'porta-loading';
		loadingDiv.innerHTML = '<button class="button" disabled><span class="button-text">Đang tải...</span></button>';
		loader.parentNode.insertBefore(loadingDiv, loader);
		return loadingDiv;
	}
		
	function getLastFiveIds() {
		const allItems = Array.from(document.querySelectorAll('.porta-article-item'));
		const lastFiveIds = allItems.slice(-5).map(item => item.getAttribute('data-id'));
		console.log(lastFiveIds);
		return lastFiveIds;
	}

	function loadNextPage() {
		if (isLoading || currentPage >= maxPages) return;
		isLoading = true;
		currentPage++;
		loader.classList.add('porta-loading');
    	const lastFiveIds = getLastFiveIds();
		
		let linkinf = '/blogs/page-2';
		linkinf = linkinf.replace('2', '');
		fetch(linkinf + currentPage)
			.then(res => res.text())
			.then(html => {
				let parser = new DOMParser();
				let doc = parser.parseFromString(html, 'text/html');
				let newItems = doc.querySelectorAll('.porta-article-item');
		
				newItems.forEach(item => {
					const id = item.getAttribute('data-id');
					if (!lastFiveIds.includes(id)) {
						loader.parentNode.insertBefore(item, loader);
					}
				});
		
				loader.classList.remove('porta-loading');
				isLoading = false;

				// Sau khi auto load đủ số trang quy định → hiện nút
				if (currentPage === autoLoadPages) {
					loadMoreBtn.style.display = 'block';
				}

				// Nếu đã đến trang cuối thì ẩn luôn loader
				if (currentPage >= maxPages) {
					loader.remove();
				}
			})
			.catch(() => {
				isLoading = false;
			});
	}

	if (loader) {
		const observer = new IntersectionObserver(entries => {
			entries.forEach(entry => {
				if (entry.isIntersecting && currentPage < autoLoadPages) {
					loadNextPage();
				}
			});
		}, {
			rootMargin: '0px',
			threshold: 0.1
		});

		observer.observe(loader);

		// Bắt sự kiện click nút Load More
		loadMoreBtn?.addEventListener('click', function () {
			loadNextPage();
		});
	}
	
})
</script>

function timeAgo(unixTime) {
  const now = new Date();
  const then = new Date(unixTime * 1000);
  const diffSec = Math.floor((now - then) / 1000);

  const phrase = (key, count) => XF.phrase(`b_stat_${key}_ago`, { count });

  if (diffSec < 60) return b_stat_phrase('{count} giây trước', diffSec);
  if (diffSec < 3600) return b_stat_phrase('{count} phút trước', Math.floor(diffSec / 60));
  if (diffSec < 86400) return b_stat_phrase('{count} giờ trước', Math.floor(diffSec / 3600));
  if (diffSec < 2592000) return b_stat_phrase('{count} ngày trước', Math.floor(diffSec / 86400));

  const yearDiff = now.getFullYear() - then.getFullYear();
  const monthDiff = now.getMonth() - then.getMonth();
  let totalMonths = yearDiff * 12 + monthDiff;

  if (totalMonths < 12) return b_stat_phrase('{count} tháng trước', totalMonths);

  const years = Math.floor(totalMonths / 12);
  const months = totalMonths % 12;

  if (months === 0) return b_stat_phrase('{count} năm trước', years);
  return b_stat_phrase('{count} năm trước', years);
}
function b_stat_phrase(templatex, count) {
    count = Math.max(1, count);
    return templatex
        .replace('{count}', count)
        .replace("{s}", count == 1 ? '' : 's');
}
