  class MyStorage {
    static has(key) {
      const reg = new RegExp(`${key}=([^;]+)`);
      return reg.test(document.cookie) || localStorage.getItem('verified') === 'true'
    }

    static set(key, value, expireDay) {
      const expire = new Date();
      expire.setTime(expire.getTime() + 1000 * 60 * 60 * 24 * expireDay);
      document.cookie = `${key}=${value}; expires=${expire.toUTCString()}`;
      localStorage.setItem(key, value);
    }

    static remove(key) {
      const expire = new Date();
      expire.setTime(expire.getTime() - 1);
      document.cookie = `${key}=; expires=${expire.toUTCString()}`;
      localStorage.removeItem(key);
    }
  }

  document.addEventListener('DOMContentLoaded', () => {
    const ua = navigator.userAgent.toLowerCase()
    const key = 'verified'
    const value = 'true'
    const className = 'is-unverified'
    const yesBtn = document.querySelector('#js-verification__nav__link--yes')
    const noBtn = document.querySelector('#js-verification__nav__link--no')

    if (ua.indexOf('googlebot') !== -1 || MyStorage.has(key) || document.getElementById("js-verification") == null) {
      return
    }

    document.body.classList.add(className)

    yesBtn.addEventListener('click', (e) => {
      e.preventDefault()
      MyStorage.set(key, value, 3)
      document.body.classList.remove(className)
    })

    noBtn.addEventListener('click', (e) => {
      e.preventDefault()
      MyStorage.remove(key)

    })
