<script>
    // Toggle Menu Thông báo
    function toggleNotify(event) {
        const wrapper = event.currentTarget;
        const wasActive = wrapper.classList.contains('active');
        
        // Đóng tất cả dropdown khác trước
        document.querySelectorAll('.notify-wrapper').forEach(el => el.classList.remove('active'));
        
        // Toggle cái hiện tại
        if (!wasActive) {
            wrapper.classList.add('active');
        }
        event.stopPropagation();
    }

    // Đóng dropdown khi click ra ngoài
    document.addEventListener('click', function(event) {
        const notifyWrapper = document.querySelector('.notify-wrapper');
        if (notifyWrapper && !notifyWrapper.contains(event.target)) {
            notifyWrapper.classList.remove('active');
        }
    });
</script>
<script>
    // Hàm chuyển đổi Theme
    function toggleTheme() {
        const html = document.documentElement;
        const icon = document.getElementById('theme-icon');
        
        if (html.getAttribute('data-theme') === 'dark') {
            html.setAttribute('data-theme', 'light');
            icon.classList.remove('fa-sun');
            icon.classList.add('fa-moon');
            localStorage.setItem('theme', 'light');
        } else {
            html.setAttribute('data-theme', 'dark');
            icon.classList.remove('fa-moon');
            icon.classList.add('fa-sun');
            localStorage.setItem('theme', 'dark');
        }
    }

    // Hàm khởi tạo icon khi load trang (để icon khớp với theme đang lưu)
    document.addEventListener("DOMContentLoaded", function() {
        const savedTheme = localStorage.getItem('theme') || 'light';
        const icon = document.getElementById('theme-icon');
        if (savedTheme === 'dark' && icon) {
            icon.classList.remove('fa-moon');
            icon.classList.add('fa-sun');
        }
    });
</script>

/**
 * Toggle password visibility
 */
function togglePassword() {
    const passwordInput = document.getElementById('password');
    const toggleIcon = document.getElementById('passwordToggleIcon');
    
    if (passwordInput.type === 'password') {
        passwordInput.type = 'text';
        toggleIcon.classList.remove('fa-eye');
        toggleIcon.classList.add('fa-eye-slash');
    } else {
        passwordInput.type = 'password';
        toggleIcon.classList.remove('fa-eye-slash');
        toggleIcon.classList.add('fa-eye');
    }
}

        });
    }

    <script>
        // Sidebar Toggle
        function toggleSidebar() {
            document.getElementById('sidebar').classList.toggle('toggled');
            document.getElementById('content-wrapper').classList.toggle('toggled');
        }

        // Chat Widget
        var chatInterval = null;
        var isChatOpen = false;

        function toggleChat() {
            var box = document.getElementById('chat-widget');
            if (box.style.display === 'none' || box.style.display === '') {
                box.style.display = 'flex';
                isChatOpen = true;
                loadMessages();
                chatInterval = setInterval(loadMessages, 3000);
                $('#chatBadge').hide();
            } else {
                box.style.display = 'none';
                isChatOpen = false;
                if(chatInterval) clearInterval(chatInterval);
            }
        }

        function sendChatMessage(e) {
            e.preventDefault();
            var msg = $('#chatInput').val();
            if(!msg.trim()) return;

            appendMessage(msg, 'user', 'Đang gửi...');
            $('#chatInput').val('');
            scrollToBottom();

            $.post('/ticket/ajax-send', { message: msg }, function(res) {
                if(res.status === 'success') {
                    loadMessages();
                }
            }, 'json');
        }

        function loadMessages() {
            if(!isChatOpen) return;

            $.get('/ticket/ajax-get', function(data) {
                var html = '';
                if(data.length === 0) {
                    html = '<div class="text-center text-muted mt-5 pt-5"><i class="fas fa-comments fa-3x mb-3" style="opacity: 0.3;"></i><p class="fw-medium">Xin chào! 👋</p><p class="small">Chúng tôi có thể giúp gì cho bạn?</p></div>';
                } else {
                    data.forEach(function(item) {
                        var cls = item.type === 'admin' ? 'msg-admin' : 'msg-user';
                        html += `<div class="msg ${cls}">
                                    ${item.msg}
                                    <span class="msg-time">${item.time}</span>
                                 </div>`;
                    });
                }
                
                var currentHtml = $('#chatBody').html();
                if(currentHtml.length !== html.length + 300) {
                    $('#chatBody').html(html);
                    scrollToBottom();
                }
            }, 'json');
        }

        function appendMessage(msg, type, time) {
            var cls = type === 'admin' ? 'msg-admin' : 'msg-user';
            var html = `<div class="msg ${cls}">
                            ${msg}
                            <span class="msg-time">${time}</span>
                         </div>`;
            $('#chatBody').append(html);
        }

        function scrollToBottom() {
            var d = document.getElementById("chatBody");
            d.scrollTop = d.scrollHeight;
        }

        // Haptic Feedback for Mobile Nav (if supported)
        document.querySelectorAll('.nav-item-mobile').forEach(item => {
            item.addEventListener('click', function() {
                if (navigator.vibrate) {
                    // Light haptic cho nav thường, medium cho dashboard
                    if (this.classList.contains('highlight')) {
                        navigator.vibrate(15); // Rung mạnh hơn cho dashboard
                    } else {
                        navigator.vibrate(8); // Rung nhẹ cho các nav khác
                    }
                }
            });
        });

        // Touch feedback cho dashboard button
        const dashboardBtn = document.querySelector('.nav-item-mobile.highlight');
        if (dashboardBtn) {
            dashboardBtn.addEventListener('touchstart', function(e) {
                this.style.transform = 'translateY(-2px)';
            });
            
            dashboardBtn.addEventListener('touchend', function(e) {
                this.style.transform = '';
            });
        }

        // Prevent double tap zoom on nav items
        let lastTouchEnd = 0;
        document.querySelectorAll('.nav-item-mobile').forEach(item => {
            item.addEventListener('touchend', function(e) {
                const now = Date.now();
                if (now - lastTouchEnd <= 300) {
                    e.preventDefault();
                }
                lastTouchEnd = now;
            }, false);
        });
    </script>
