    var allLogin = false;
    function login(obj) {
        if(allLogin){
            var url = '/member/login/';
            var username = $(obj).find("#username").val();
            var password = $(obj).find("#password").val();
            var checkcode = $(obj).find("#checkcode").val();
            $.ajax({
                type: 'POST',
                url: url,
                dataType: 'json',
                data: {
                    username: username,
                    password: password,
                    checkcode: checkcode
                },
                success: function (response, status) {
                    if (response.code) {
                        alert(response.data)
                        location.reload();

                    } else {
                        $('#codeimg1').click();
                        alert(response.data)
                    }
                },
                error: function (xhr, status, error) {
                    alert('Return data exception!');
                }
            });
            return false;
        }else{
            return false;
        }


    }

    function LoginEmail() {
        var url = '0';
        var username = $("#username").val();
        var reg = /^([a-zA-Z0-9_-])+@([a-zA-Z0-9_-])+(.[a-zA-Z0-9_-])+/;
        var flag = reg.test(username);
        if (!username) {
            $("#username").addClass("is-invalid");
            $("#usernote").addClass("invalid-feedback");
            $("#username").removeClass("is-valid");
            $("#usernote").removeClass("valid-feedback");
            $("#usernote").text("Please enter your email address！");
            allLogin = false;
        }
        if (flag) {
            $("#username").addClass("is-valid");
            $("#usernote").addClass("valid-feedback");
            $("#username").removeClass("is-invalid");
            $("#usernote").removeClass("invalid-feedback");
            $("#usernote").text("The email you entered can be used！");
            allLogin = true;
            return true;
        } else {
            $("#username").addClass("is-invalid");
            $("#usernote").addClass("invalid-feedback");
            $("#username").removeClass("is-valid");
            $("#usernote").removeClass("valid-feedback");
            $("#usernote").text("Please enter the correct email address！");
            allLogin = false;
            return false;
        }
        $.ajax({
            type: 'POST',
            url: url,
            dataType: 'json',
            data: {
                username: username
            },
            success: function (response, status) {
                if (response.code) {
                    $("#username").addClass("is-invalid");
                    $("#usernote").addClass("invalid-feedback");
                    $("#username").removeClass("is-valid");
                    $("#usernote").removeClass("valid-feedback");

                } else {
                    $("#username").addClass("is-valid");
                    $("#usernote").addClass("valid-feedback");
                    $("#username").removeClass("is-invalid");
                    $("#usernote").removeClass("invalid-feedback");
                }
                $("#usernote").text(response.data);
            },
            error: function (xhr, status, error) {
                alert('Return data exception！');
            }
        });
        return false;
    }

    function LoginPwd() {
        var password = $("#password").val();
        var rpassword = $("#rpassword").val();
        var reg = /[A-Za-z].*[0-9]|[0-9].*[A-Za-z]/;
        var flag = reg.test(password)
        if (!password) {
            $("#password").addClass("is-invalid");
            $("#passwordnote").addClass("invalid-feedback");
            $("#password").removeClass("is-valid");
            $("#passwordnote").removeClass("valid-feedback");
            $("#passwordnote").text("Please enter the password！");
            allLogin = false;
        }
        if (flag) {
            $("#password").addClass("is-valid");
            $("#passwordnote").addClass("valid-feedback");
            $("#password").removeClass("is-invalid");
            $("#passwordnote").removeClass("invalid-feedback");
            $("#passwordnote").text("");

            $("#passwordnote").addClass("invalid-feedback");

            allLogin = false;
            return false;
        }

        if (password.length >= 6 && password.length <= 18) {

            $("#passwordnote").addClass("valid-feedback");

            $("#passwordnote").removeClass("invalid-feedback");

            allLogin = true;

            $("#passwordnote").removeClass("valid-feedback");
            $("#passwordnote").text("Please enter a password of 6-18 words！");
            allLogin = false;
            return false;
        }
    }
