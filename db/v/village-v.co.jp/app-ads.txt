$(function () {

console.log("default");
  function onScroll() {
    var winTop = $win.scrollTop();
    var winH   = $win.height();

    var sectionTop = $section.offset().top;
    var footerTop  = $footer.offset().top;

    // 追従対象をすべて処理
    $('.js-camp').each(function () {

    var $el = $(this);
      var elH = $el.outerHeight();

      // 1) セクション以降で表示
      if (winTop + winH > sectionTop) {
          console.log("付与処理");
        $el.addClass('is-visible');
      } else {
          console.log("りむ処理");
        $el.removeClass('is-visible');
      }

      // 2) フッター手前で止める
      var stopTop = footerTop - elH - marginBottom;

      if (winTop + winH > stopTop) {
        // body（もしくは親ラッパー）基準で固定
          console.log("ストップ");
        $el
          .addClass('is-stop')
          .css({
            top: stopTop + 'px',
            bottom: 'auto'
            // left / right はCSSに任せるので触らない
          });
      } else {
                console.log("ストップ");
        // 通常の追従（fixed）
        $el
          .removeClass('is-stop')
          .css({
            top: '',
            bottom: bottomFixed + 'px'
          });
      }
    });
  }

  $win.on('scroll resize', onScroll);
  onScroll();
