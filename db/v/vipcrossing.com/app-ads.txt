     function ValidateEmail(mail)
     {
         var expressionReguliere = /^(([^<>()[]\.,;:s@]+(.[^<>()[]\.,;:s@]+)*)|(.+))@(([[0-9]{1,3}.[0-9]{1,3}.[0-9]{1,3}.[0-9]{1,3}])|(([a-zA-Z-0-9]+.)+[a-zA-Z]{2,}))$/;
           if (expressionReguliere.test(mail))
         {
             return true;
         }
         else
         {
             return false;
         }
     }
