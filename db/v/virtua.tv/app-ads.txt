    <script>
        function toggleMobileMenu(force) {
            const menu = document.getElementById('mobile-menu');
            const menuIcon = document.getElementById('menu-icon');
            const closeIcon = document.getElementById('close-icon');
            if (!menu) return;
            if (force !== undefined) {
                if (!force) {
                    menu.classList.add('translate-x-full');
                    menuIcon.classList.remove('hidden');
                    closeIcon.classList.add('hidden');
                } else {
                    menu.classList.remove('translate-x-full');
                    menuIcon.classList.add('hidden');
                    closeIcon.classList.remove('hidden');
                }
            } else {
                menu.classList.toggle('translate-x-full');
                menuIcon.classList.toggle('hidden');
                closeIcon.classList.toggle('hidden');
            }
        }

        // Navbar scroll effect
        window.addEventListener('scroll', () => {
            const nav = document.getElementById('navbar');
            if (!nav) return;
            if (window.scrollY > 10) {
                nav.classList.add('bg-black/80', 'backdrop-blur-md', 'shadow-lg', 'py-3');
                nav.classList.remove('bg-transparent', 'py-5');
            } else {
                nav.classList.remove('bg-black/80', 'backdrop-blur-md', 'shadow-lg', 'py-3');
                nav.classList.add('bg-transparent', 'py-5');
            }
        });
    </script>

async function loadCarousel(url, containerId, isMovie = true) {
    const container = document.getElementById(containerId);
    if (!container) return;
    
    try {
        const response = await fetch(url);
        const data = await response.json();
        const items = data.results.slice(0, 10);
        
        container.innerHTML = items.map(item => `
            <div class="movie-card min-w-[240px] w-[240px] flex-shrink-0">
                <img src="https://image.tmdb.org/t/p/w500${item.poster_path}" alt="${isMovie ? item.title : item.name}" class="w-full h-[360px] object-cover rounded-lg" loading="lazy">
                <div class="movie-card-overlay">
                    <h4 class="font-medium text-white mb-1">${isMovie ? item.title : item.name}</h4>
                    <p class="text-xs text-gray-300 line-clamp-2">${item.overview}</p>
                </div>
            </div>
        `).join('');
    } catch (e) {
        console.error("Carousel Error", e);
    }
}
