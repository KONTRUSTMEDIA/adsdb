$(document).ready(function() {
    $('#searchform').hide();

    $('#btn-show-search').on('click', function() {
        var $icon = $(this).find('i');
        $('#searchform').toggle();


        if ($icon.hasClass('fa-magnifying-glass')) {
            $icon.removeClass('fa-magnifying-glass').addClass('fa-times');
            $('#btn-show-search').addClass("show");
        } else {
            $icon.removeClass('fa-times').addClass('fa-magnifying-glass');
            $('#btn-show-search').removeClass("show");
        }
    });

     $(document).ready(function() {
	    $('#top-btn-show-search').click(function() {
	        $('.btn-x-search-top').removeClass('d-none');
	        $('#searchsubmittop').removeClass('d-none');
	        $('.fa-magnifying-glass').addClass('d-none');
	        $('.field').removeClass('w-0');
	        $('.field').removeClass('p-0');
	        $('.field').removeClass('b-0');

	    });

	    $('.btn-x-search-top').click(function() {
	        $(this).addClass('d-none');
	        $('#searchsubmittop').addClass('d-none');
	        $('.fa-magnifying-glass').removeClass('d-none');
	        $('.field').addClass('w-0');
	        $('.field').addClass('p-0');
	        $('.field').addClass('b-0');
	    });

	});
});

let lastScrollTop = 0;
        
$(window).on('scroll', function() {
    let scrollTop = $(this).scrollTop();

    if (scrollTop > lastScrollTop) {
        $('#wrapper-navbar').addClass('menu-top-fixed'); // Esconde o menu
        $('#logo-top-menu').removeClass('d-none');
        $('.search-menu-top').removeClass('d-none');

    } 
    if (scrollTop < 100){
        $('#wrapper-navbar').removeClass('menu-top-fixed'); // Mostra o menu
        $('#logo-top-menu').addClass('d-none');
        $('.search-menu-top').addClass('d-none');
		$('#searchform').hide();
        $('#btn-show-search').removeClass("show");
        $('#btn-show-search').find('i').removeClass('fa-times').addClass('fa-magnifying-glass');

    }
    lastScrollTop = scrollTop;
});

$(document).ready(function(){
    const date = new Date();
    const day = date.getDate();
    const year = date.getFullYear();
    const months = ["JAN", "FEB", "MAR", "APR", "MAY", "JUN", 
                    "JUL", "AUG", "SEP", "OCT", "NOV", "DEC"];
    const month = months[date.getMonth()];
    
    const formattedDate = `${day} ${month} ${year}`;
    $('#dateElement').text(formattedDate);
});
