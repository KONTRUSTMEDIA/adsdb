    // 仅在明确禁用SW的情况下，才注销并清理缓存（灰度/回滚使用）
    function shouldDisableSW() {
      try {
        const params = new URLSearchParams(window.location.search);
        if (params.get('sw') === 'off') return true;
        const stored = localStorage.getItem('sw:enabled');
        if (stored === '0' || stored === 'false') return true;
      } catch (e) {}
      return false;
    }

    if ('serviceWorker' in navigator && shouldDisableSW()) {
      navigator.serviceWorker.getRegistrations().then(function(registrations) {
        const unregisterPromises = registrations.map(function(registration) {
          console.log('正在卸载Service Worker:', registration.scope);
          return registration.unregister().then(function(success) {
            console.log('Service Worker卸载' + (success ? '成功' : '失败'));
            return success;
          });
        });
        
        Promise.all(unregisterPromises).then(function() {
          console.log('所有Service Worker已卸载');
          
          if ('caches' in window) {
            caches.keys().then(function(cacheNames) {
              if (cacheNames.length === 0) {
                console.log('没有找到缓存');
              }
              
              const deletePromises = cacheNames.map(function(cacheName) {
                console.log('正在删除缓存:', cacheName);
                return caches.delete(cacheName).then(function(success) {
                  console.log('缓存删除' + (success ? '成功' : '失败') + ':', cacheName);
                  return success;
                });
              });
              
              Promise.all(deletePromises).then(function() {
                console.log('所有缓存已清理');
                
                if (sessionStorage.getItem('sw_fixed') !== 'true') {
                  console.log('设置修复标记并刷新页面');
                  sessionStorage.setItem('sw_fixed', 'true');
                  window.location.href = window.location.href.split('?')[0] + '?cache_bust=' + Date.now();
                }
              });
            });
          }
        });
      }).catch(function(error) {
        console.error('Service Worker卸载过程中出错:', error);
      });
    } else {
      console.log('SW 保持启用（未执行注销/清理）');
    }
  </script>
