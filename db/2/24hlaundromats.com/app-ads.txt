  document.addEventListener('DOMContentLoaded', () => {
    const loadMoreButton = document.getElementById('load-more-cities');
    const citySearchInput = document.getElementById('city-search');
    const allCityCards = Array.from(document.querySelectorAll('.city-card')); // Store all cards initially
    const initialCount = 9;
    const batchSize = 6;

    let currentFilteredCities = [];
    let currentVisibleCount = 0;

    const updateDisplay = () => {
      const searchTerm = citySearchInput ? citySearchInput.value.toLowerCase() : '';

      // Determine the set of cities that match the search term
      currentFilteredCities = allCityCards.filter(card => {
        const cityName = card.querySelector('.city-name a').textContent.toLowerCase();
        return cityName.includes(searchTerm);
      });

      // Hide all cards first
      allCityCards.forEach(card => {
        card.style.display = 'none';
      });

      // If search term is empty, or if it's a new search, reset visible count to initialCount
      if (searchTerm === '') {
        // If no search term, and we're just loading more, keep previous visible count
        // Otherwise, reset to initialCount
        if (currentVisibleCount === 0 || currentVisibleCount > initialCount) {
          currentVisibleCount = Math.min(currentVisibleCount === 0 ? initialCount : currentVisibleCount, currentFilteredCities.length);
        } else {
          currentVisibleCount = Math.min(initialCount, currentFilteredCities.length);
        }
      } else {
        // If there's a search term, always reset to initialCount for the filtered set
        currentVisibleCount = Math.min(initialCount, currentFilteredCities.length);
      }
      
      // Display the visible cards
      for (let i = 0; i < currentVisibleCount; i++) {
        if (currentFilteredCities[i]) {
          currentFilteredCities[i].style.display = 'flex';
        }
      }

      updateLoadMoreButton();
    };

    const updateLoadMoreButton = () => {
      if (loadMoreButton) {
        if (currentVisibleCount < currentFilteredCities.length) {
          loadMoreButton.style.display = 'block';
        } else {
          loadMoreButton.style.display = 'none';
        }
      }
    };

    if (loadMoreButton) {
      loadMoreButton.addEventListener('click', () => {
        const nextVisibleCount = Math.min(currentVisibleCount + batchSize, currentFilteredCities.length);
        
        // Display the newly visible cards
        for (let i = currentVisibleCount; i < nextVisibleCount; i++) {
          if (currentFilteredCities[i]) {
            currentFilteredCities[i].style.display = 'flex';
          }
        }
        currentVisibleCount = nextVisibleCount;
        updateLoadMoreButton();
      });
    }

    if (citySearchInput) {
      citySearchInput.addEventListener('keyup', updateDisplay);
    }

    // Initial display
    updateDisplay();
  });
