        function openLeaderboard(quizId) {

                .then(data => {
                    list.innerHTML = "";

                        data.data.forEach((entry, index) => {
                            list.innerHTML += `<li><strong>${index + 1}.</strong> Score: <span class="score">${entry.user_score}</span></li>`;

                        list.innerHTML = "<li>No scores available.</li>";
