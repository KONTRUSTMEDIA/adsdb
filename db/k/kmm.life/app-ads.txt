    function saLoadedLinkEvents() {
      document
        .querySelectorAll("a[data-sa-link-event]")
        .forEach(function (element) {
          var href = element.getAttribute("href");
          var eventName = element.getAttribute("data-sa-link-event");
          if (!href || !window.sa_event || !window.sa_loaded) return;

          element.addEventListener("click", function (event) {
            var target = element.getAttribute("target");
            if (target === "_blank") {
              event.preventDefault();
              window.sa_event(eventName, function () {
                window.location.href = href;
              });
              return false;
            } else {
              window.sa_event(eventName);
              return true;
            }
          });
        });
    }

    if (document.readyState === "ready" || document.readyState === "complete") {
      saLoadedLinkEvents();
    } else {
      document.addEventListener("readystatechange", function (event) {
        if (event.target.readyState === "complete") saLoadedLinkEvents();
      });
    }
  })();

document.addEventListener('DOMContentLoaded', function() {
    // Scope everything to the main container only
    const container = document.querySelector('.main-library-filter');
    if (!container) return; // Safety: exit if not found (e.g. on other pages)
    const titleElement = container.querySelector('#active-category-title');
    const descElement = container.querySelector('#active-category-description');
    const filterControls = container.querySelectorAll('[uk-filter-control]');
    if (!titleElement || !descElement || filterControls.length === 0) return;
    function updateContent(element) {
        const title = element.dataset.title || 'All Resources';
        const description = element.dataset.description || 'Browse all learning resources...';
        // [titleElement.style](http://titleElement.style).opacity = '0';
       // [descElement.style](http://descElement.style).opacity = '0';
        setTimeout(() => {
            titleElement.textContent = title;
            descElement.textContent = description;
           // [titleElement.style](http://titleElement.style).opacity = '1';
           // [descElement.style](http://descElement.style).opacity = '1';
        }, 150);
    }
    filterControls.forEach(control => {
        control.addEventListener('click', function() {
            updateContent(this);
        });
    });
    // Set initial content
    const initialActive = container.querySelector('[uk-filter-control].uk-active');
    if (initialActive) {
        updateContent(initialActive);
    }
});
</script>
