  <script>
    const form = document.getElementById('subscribeForm');
    const status = document.getElementById('status'); 

    form.addEventListener('submit', async (e) => {
      e.preventDefault();
      const email = document.getElementById('email').value.trim();
 
      if (!email) {
        status.textContent = 'Lütfen geçerli bir e-posta girin.';
        status.style.color = '#ff3f00';
        return;
      }

      try {
        const res = await fetch("https://kirayaortak.com/api/subscribe", {
          method: "POST",
          headers: {
            "Content-Type": "application/json"
          },
          body: JSON.stringify({ email })
        });
	
	console.log(res.text());
        if (res.ok) {
          status.textContent = "Kaydınız başarıyla alındı. Teşekkürler!";
          status.style.color = '#00ff88';
          form.reset();
        } else {
          status.textContent = "Bir hata oluştu, lütfen tekrar deneyin.";
          status.style.color = '#ff3f00';
        }
      } catch (err) {
        status.textContent = "Sunucuya bağlanılamadı.";
        status.style.color = '#ff3f00';
      }
    });
  
    // Countdown timer
    const targetDate = new Date("2026-01-31T00:00:00");
    const countdownEl = document.getElementById('countdown');
    function updateCountdown() {
      const now = new Date();
      const diff = targetDate - now;
      if (diff <= 0) {
        countdownEl.textContent = "Yayındayız!";
        return;
      }
      const days = Math.floor(diff / (1000*60*60*24));
      const hours = Math.floor((diff / (1000*60*60)) % 24);
      const mins = Math.floor((diff / (1000*60)) % 60);
      const secs = Math.floor((diff / 1000) % 60);
      countdownEl.textContent = `${days}g ${hours}s ${mins}d ${secs}sn`;
    }
    setInterval(updateCountdown, 1000);
    updateCountdown();
  </script>
