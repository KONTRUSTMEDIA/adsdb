    $(document).ready(function() {
        let isSpinning = false;
        const wheel = $('#wheel');
        // const spinButton = $('.clickToPinButton');
        // const centerButton = $('#spinButton');
        const resultModal = $('#resultModal');
        const resultText = $('#resultText');

        // CSRF token setup for Laravel
        $.ajaxSetup({
            headers: {
                'X-CSRF-TOKEN': $('meta[name="csrf-token"]').attr('content')
            }
        });

        function spinWheel() {
            if (isSpinning) return;

            //disable click
            $('.clickToPinButton').css('pointer-events', 'none');

            isSpinning = true;
            $('.clickToPinButton').prop('disabled', true);
            $('.clickToPinButton').addClass('spinning');

            // Make AJAX call to Laravel backend
            $.ajax({
                url: '/spin-wheel',
                method: 'POST',
                success: function(response) {
                    // Calculate rotation
                    const prizeIndex = response.prize_index;
                    const segmentAngle = 360 / 8; // 8 segments
                    $('#prizeIndex').val(prizeIndex);

                    $('#spinRemainder').text(response.raffle);

                    if (response.raffle == 0) {
                        //change .clickToPinButton id from spinButton to notEligible
                        $('.clickToPinButton').removeClass('clickToPinButton').addClass('notEligible');
                    }

                    // Calculate the angle to the CENTER of the winning segment
                    // Each segment starts at index * segmentAngle
                    // We need to point to the middle, so add half a segment
                    const segmentCenter = (prizeIndex * segmentAngle) + (segmentAngle / 2);

                    // The pointer is at the top (0 degrees), so we need to rotate
                    // the wheel so the segment center aligns with top
                    // Subtract from 360 because wheel rotates clockwise
                    const targetAngle = 360 - segmentCenter;

                    // Add multiple full rotations for effect (5 full spins)
                    const totalRotation = 360 * 5 + targetAngle;

                    // Apply rotation
                    wheel.css('transform', `rotate(${totalRotation}deg)`);

                    // Show appropriate result based on prize index
                    setTimeout(function() {
                        const prizeResult = $('#prizeResult');
                        const accountIdForm = $('#accountIdForm');

                        // Reset and hide form
                        accountIdForm.hide();
                        accountIdForm[0].reset();

                        // Clear previous result
                        prizeResult.empty();

                        // Create result content based on prize index
                        if (response.prize_index === 3) { // TRY AGAIN
                            prizeResult.html(`
                                <h2>😕 Try Again</h2>
                                <p style="margin-bottom: 3px;">${response.prize_text}</p>
                                <p style="margin-bottom: 7px;">Better luck next time!</p>
                                <button class="btn btn-primary mt-3" id="closeModal">Close</button>
                            `);
                        }
                        else if (response.prize_index === 6) { // ONE ADDITIONAL FREE SPIN
                            prizeResult.html(`
                                <h2>🎉 Congratulations! 🎉</h2>
                                <p style="margin-bottom: 3px;">You won: <strong>${response.prize_text}</strong></p>
                                <p style="margin-bottom: 7px;">Your free spin has been added to your account!</p>
                                <button class="btn btn-primary mt-3" id="closeModal">Close</button>
                            `);
                        }
                        else if (response.prize_index === 5 || response.prize_index === 7) { // 1xbet or Sportybet
                            const is1xbet = response.prize_index === 5;
                            const platform = is1xbet ? '1xbet' : 'Sportybet';
                            const fieldLabel = is1xbet ? '1xbet Account ID' : 'Sportybet Account ID';

                            prizeResult.html(`
                                <h2>🎉 Congratulations! 🎉</h2>
                                <p style="margin-bottom: 3px;">You won: <strong style="font-weight: bold;">${response.prize_text}</strong></p>
                                <p style="margin-bottom: 7px;" class="mb-2">Please enter your ${platform} account ID below to claim your prize.</p>
                            `);

                            // Set up the form
                            $('#prizeType').val(platform.toLowerCase());
                            $('#accountIdLabel').text(fieldLabel);
                            accountIdForm.show().insertAfter(prizeResult);

                            // Add close button after form
                            $('<button class="btn btn-secondary mt-3 ml-2" id="closeModal">Close</button>')
                                .insertAfter(accountIdForm.find('button[type="submit"]'));
                        }
                        else if (response.prize_index === 1) { // DSTV Subscription
                            prizeResult.html(`
                                <h2>🎉 Congratulations! 🎉</h2>
                                <p style="margin-bottom: 3px;">You won: <strong style="font-weight: bold;">${response.prize_text}</strong></p>
                                <p style="margin-bottom: 7px;" class="mb-2">Please enter your DSTV account number below to claim your prize.</p>
                            `);

                            // Set up the form
                            $('#prizeType').val('dstv');
                            $('#accountIdLabel').text('DSTV Account Number');
                            accountIdForm.show().insertAfter(prizeResult);

                            // Add close button after form
                            $('<button class="btn btn-secondary mt-3 ml-2" id="closeModal">Close</button>')
                                .insertAfter(accountIdForm.find('button[type="submit"]'));
                        }
                        else if (response.prize_index === 4) { // Premium Access
                            prizeResult.html(`
                                <h2>🎉 Congratulations! 🎉</h2>
                                <p style="margin-bottom: 3px;">You won: <strong style="font-weight: bold;">${response.prize_text}</strong></p>
                                <p style="margin-bottom: 7px;" class="mb-2">Please enter your Kingpredict email address to claim your premium access.</p>
                            `);

                            // Set up the form
                            $('#prizeType').val('premium');
                            $('#accountIdLabel').text('Kingpredict Email');
                            accountIdForm.show().insertAfter(prizeResult);

                            // Add close button after form
                            $('<button class="btn btn-secondary mt-3 ml-2" id="closeModal">Close</button>')
                                .insertAfter(accountIdForm.find('button[type="submit"]'));
                        }
                        else { // Other prizes
                            prizeResult.html(`
                                <h2>🎉 Congratulations! 🎉</h2>
                                <p style="margin-bottom: 3px;">You won: <strong>${response.prize_text}</strong></p>
                                <p>Your prize will be credited to your account shortly.</p>
                                <button class="btn btn-primary mt-3" id="closeModal">Close</button>
                            `);
                        }

                        // Show the modal
                        resultModal.css('display', 'flex');
                        isSpinning = false;
                        $('.clickToPinButton').prop('disabled', false);
                        $('.clickToPinButton').removeClass('spinning');

                        // Reset wheel rotation for next spin
                        wheel.css('transition', 'none');
                        wheel.css('transform', 'rotate(0deg)');
                        setTimeout(() => {
                            wheel.css('transition', 'transform 4s cubic-bezier(0.17, 0.67, 0.12, 0.99)');
                        }, 50);

                        // Re-enable pointer events
                        $('.clickToPinButton').css('pointer-events', 'auto');
                    }, 2000);
                },
                error: function(xhr) {
                    alert('Error spinning wheel. Please try again.');
                    isSpinning = false;
                    $('.clickToPinButton').prop('disabled', false);
                    $('.clickToPinButton').removeClass('spinning');
                }
            });
        }

        // Bind click events
        $('.clickToPinButton').on('click', spinWheel);
        // centerButton.on('click', spinWheel);

        // Close modals
        function closeAllModals() {
            $('.result-modal').css('display', 'none');
        }

        // Close modals when clicking the X or Close button
        $(document).on('click', '#closeModal, #closeResultModal, #closeEligibleModal, #closeRuleModal', closeAllModals);

        // Close modal when clicking outside the content
        $(document).on('click', '.result-modal', function(e) {
            if (e.target === this) {
                closeAllModals();
            }
        });

        // Not eligible button
        $('.notEligible').on('click', function() {
            $('#eligibleModal').css('display', 'flex');
        });

        // View rules
        $('#viewRule').on('click', function() {
            $('#ruleModal').css('display', 'flex');
        });

        // Account ID form submission
        $('#accountIdForm').on('submit', function(e) {
            e.preventDefault();

            const formData = {
                _token: $('meta[name="csrf-token"]').attr('content'),
                prize_type: $('#prizeType').val(),
                account_id: $('#accountId').val(),
                prize_index: $('#prizeIndex').val(),
            };

            // Disable form during submission
            const submitBtn = $(this).find('button[type="submit"]');
            submitBtn.prop('disabled', true).html('<i class="fa fa-spinner fa-spin"></i> Submitting...');

            // Send the data to the server
            $.ajax({
                url: '/submit-account-id',
                method: 'POST',
                data: formData,
                success: function(response) {
                    if (response.success) {
                        $('#prizeResult').html(`
                            <h2>✅ Success!</h2>
                            <p style="margin-bottom: 3px;">Your ${formData.prize_type} account ID has been received.</p>
                            <p>Your prize will be credited to your account shortly.</p>
                            <button class="btn btn-primary mt-3" id="closeModal">Close</button>
                        `);
                        $('#accountIdForm').hide();
                    } else {
                        alert(response.message || 'An error occurred. Please try again.');
                        submitBtn.prop('disabled', false).text('Submit');
                    }
                },
                error: function(xhr) {
                    const errorMsg = xhr.responseJSON?.message || 'An error occurred. Please try again.';
                    alert(errorMsg);
                    submitBtn.prop('disabled', false).text('Submit');
                }
            });
        });
    });
</script>

function investmentTimer(endTime, showProgress, market) {
    function updateTimer() {
        const now = new Date().getTime();
        const distance = new Date(endTime).getTime() - now;

        // Time calculations for hours, minutes and seconds
        const hours = Math.floor((distance % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));
        const minutes = Math.floor((distance % (1000 * 60 * 60)) / (1000 * 60));
        const seconds = Math.floor((distance % (1000 * 60)) / 1000);

        // Display the result
        const hoursElement = document.getElementById(`hours_${market}`);
        const minutesElement = document.getElementById(`minutes_${market}`);
        const secondsElement = document.getElementById(`seconds_${market}`);

        if (hoursElement) hoursElement.innerHTML = hours.toString().padStart(2, '0');
        if (minutesElement) minutesElement.innerHTML = minutes.toString().padStart(2, '0');
        if (secondsElement) secondsElement.innerHTML = seconds.toString().padStart(2, '0');

        // If the countdown is finished, redirect or show message
        if (distance < 0) {
            clearInterval(timer);
            const timerElement = document.querySelector(`.countdown-widget[data-market="${market}"] .countdown-timer`);
            if (timerElement) {
                timerElement.innerHTML = `
                    <div class="match-started text-center py-2">
                        <div class="badge badge-success p-2">
                            <i class="fa fa-play-circle mr-1"></i> Match In Progress
                        </div>
                        <p class="small text-muted mt-2 mb-0">
                            Stay tuned for live updates!
                        </p>
                    </div>
                `;
            }
        }
    }

    // Update the countdown every 1 second
    const timer = setInterval(updateTimer, 1000);
    updateTimer(); // Initial call to avoid delay

    // Initialize progress bar if enabled
    if (showProgress) {
        const startTime = new Date().getTime();
        const endTimeMs = new Date(endTime).getTime();
        if (endTimeMs > startTime) {
            updateProgressBar(market, startTime, endTimeMs);
        }
    }
}
// Update the progress bar
function updateProgressBar(market, startTime, endTime) {
    const now = new Date().getTime();
    const total = endTime - startTime;
    const progress = Math.min(100, Math.max(0, ((now - startTime) / total) * 100));
    document.getElementById(`progress_${market}`).style.width = `${progress}%`;

    if (progress < 100) {
        requestAnimationFrame(() => updateProgressBar(market, startTime, endTime));
    }
}

// Initialize progress bar for each countdown
function initProgressBars() {
    document.querySelectorAll('.countdown-timer').forEach(timer => {
        const market = 'investment_tip';
        const startTime = new Date().getTime();
        const endTime = new Date('2026-02-24 11:30:28').getTime();

        if (endTime > startTime) {
            updateProgressBar(market, startTime, endTime);
        }
    });
}

// Call this when the component is loaded
document.addEventListener('DOMContentLoaded', function() {
    if (document.querySelector('.countdown-timer')) {
        initProgressBars();
    }
});
