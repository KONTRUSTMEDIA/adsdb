document.addEventListener("DOMContentLoaded", function() {
    const featuresPerPage = 2;
    let currentPage = 0;
    const features = document.querySelectorAll('.feature-box');
    const totalPages = Math.ceil(features.length / featuresPerPage);
    let autoMoveInterval;
    let userInteractionTimeout;

    function showPage(page) {
        // Hide all features
        features.forEach(feature => {
            feature.style.display = 'none';
        });

        // Calculate start and end indices
        const start = page * featuresPerPage;
        const end = start + featuresPerPage;

        // Show the features for the current page
        features.forEach((feature, index) => {
            if (index >= start && index < end) {
                feature.style.display = 'block';
            }
        });

        // Update button visibility
        document.getElementById('prev').disabled = page === 0;
        document.getElementById('next').disabled = page >= totalPages - 1;
    }

    function nextPage() {
        if (currentPage < totalPages - 1) {
            currentPage++;
        } else {
            currentPage = 0; // Loop back to the first page
        }
        showPage(currentPage);
    }

    function restartAutoMove() {
        clearInterval(autoMoveInterval);
        autoMoveInterval = setInterval(nextPage, 5000);
    }

    function pauseAutoMove() {
        clearInterval(autoMoveInterval);
        clearTimeout(userInteractionTimeout);

        // Restart auto-move after 10 seconds if no interaction
        userInteractionTimeout = setTimeout(() => {
            restartAutoMove();
        }, 10000);
    }

    // Add event listeners to buttons
    document.getElementById('prev').addEventListener('click', () => {
        if (currentPage > 0) {
            currentPage--;
            showPage(currentPage);
            pauseAutoMove();
        }
    });

    document.getElementById('next').addEventListener('click', () => {
        if (currentPage < totalPages - 1) {
            currentPage++;
            showPage(currentPage);
            pauseAutoMove();
        }
    });

    // Initially display the first page
    showPage(currentPage);

    // Start auto-move
    restartAutoMove();
});

document.addEventListener("DOMContentLoaded", function () {
    const orderButtons = document.querySelectorAll(".order-btn");
    const modal = document.getElementById("order-modal");
    const modalTitle = document.getElementById("modal-title");
    const modalImage = document.getElementById("modal-image");
    const modalPrice = document.getElementById("modal-price");
    const totalPriceField = document.getElementById("total-price");
    const hiddenProductName = document.getElementById("hidden-product-name");
    const hiddenTotalPrice = document.getElementById("hidden-total-price");
    const quantityInput = document.getElementById("quantity");
    const hiddenQuantity = document.getElementById("hidden-quantity");
    const cardTypeSelect = document.getElementById('card_type');
    const hiddenInput = document.getElementById('hidden-card-type');

    const increaseBtn = document.getElementById("increase-quantity");
    const decreaseBtn = document.getElementById("decrease-quantity");
    
if (cardTypeSelect && hiddenInput) {
    cardTypeSelect.addEventListener('change', function () {
        hiddenInput.value = this.value;

        // Update modal price display when "multi" is selected
        let extra = (this.value === 'multi') ? 100 : 0;
        let updatedPrice = baseProductPrice + extra;

        modalPrice.textContent = updatedPrice.toFixed(0); // ✅ Show updated unit price in modal
        
        updateTotalPrice(); // Recalculate total based on new unit price
    });
}

    orderButtons.forEach(button => {
        button.addEventListener("click", function () {
            const productName = this.getAttribute("data-product");
            const productPrice = parseFloat(this.getAttribute("data-price"));
            const productImage = this.getAttribute("data-image");

            const orderLabel = "Order";
            modalTitle.textContent = `${orderLabel} - ${productName}`;
            modalImage.src = productImage;
            baseProductPrice = productPrice; // ✅ Store the original price for use in type switch
            modalPrice.textContent = baseProductPrice.toFixed(0);
            totalPriceField.textContent = productPrice.toFixed(0);
            hiddenProductName.value = productName;
            hiddenTotalPrice.value = productPrice.toFixed(0);

            quantityInput.value = "1"; // Reset quantity to 1 when opening modal
            hiddenQuantity.value = "1"; // Ensure hidden field also resets

            modal.style.display = "flex";
            document.body.style.overflowY = "hidden";
            
        });
    });

    // Function to update total price
    function updateTotalPrice() {
        let quantity = parseInt(quantityInput.value);
        let basePrice = parseFloat(modalPrice.textContent);

        if (isNaN(quantity) || quantity < 1) {
            quantity = 1; // Prevent invalid values
        }

        let newTotal = basePrice * quantity;
        totalPriceField.textContent = newTotal.toFixed(0);
        hiddenTotalPrice.value = newTotal.toFixed(0);
        hiddenQuantity.value = quantity; // Sync hidden quantity
    }

    // Increase Quantity
    increaseBtn.addEventListener("click", function () {
        let quantity = parseInt(quantityInput.value);
        quantity++;
        quantityInput.value = quantity;
        updateTotalPrice();
    });

    // Decrease Quantity (Ensure Minimum is 1)
    decreaseBtn.addEventListener("click", function () {
        let quantity = parseInt(quantityInput.value);
        if (quantity > 1) {
            quantity--;
            quantityInput.value = quantity;
            updateTotalPrice();
        }
    });

    // Update Quantity when user manually types a value
    quantityInput.addEventListener("input", updateTotalPrice);

    // Ensure Quantity is Updated Before Form Submission
    document.getElementById("order-form").addEventListener("submit", function (event) {
        hiddenQuantity.value = quantityInput.value; // Sync before submitting
    });

    // Close modal when clicking the "X" button
    document.querySelector(".close").addEventListener("click", function () {
        modal.style.display = "none";
        document.body.style.overflowY = "unset"
    });

    // Close modal when clicking outside the modal
    window.addEventListener("click", function (event) {
        if (event.target === modal) {
            modal.style.display = "none";
            document.body.style.overflowY = "unset"
        }
    });

    // **Form Validation Before Submission**
    document.getElementById("order-form").addEventListener("submit", function (event) {
        const name = document.getElementById("name").value.trim();
        const phone = document.getElementById("phone").value.trim();
        const city = document.getElementById("city").value.trim();
        const address = document.getElementById("address").value.trim();
        const quantity = parseInt(quantityInput.value);
        const totalPrice = parseFloat(hiddenTotalPrice.value);

        // Basic Validation
        if (name === "" || phone === "" || city === "" || address === "") {
            alert("Please fill out all fields.");
            event.preventDefault();
            return;
        }

        // Phone Number Validation
        const phoneRegex = /^[0-9+\-\s]{8,15}$/;
        if (!phoneRegex.test(phone)) {
            alert("Please enter a valid phone number.");
            event.preventDefault();
            return;
        }

        // **Fixed Quantity Validation**
        if (isNaN(quantity) || quantity < 1 || quantity > 100) {
            alert("Quantity must be between 1 and 100.");
            event.preventDefault();
            return;
        }

        // Price Validation
        if (totalPrice <= 0 || isNaN(totalPrice)) {
            alert("Invalid total price. Please check your order.");
            event.preventDefault();
            return;
        }
        
    });
});

document.addEventListener('DOMContentLoaded', function() {
    // Get all video containers
    const videoContainers = document.querySelectorAll('.video-container');
    
    videoContainers.forEach(container => {
        const video = container.querySelector('video');
        const playButton = container.querySelector('.play-button');
        
        // Click to play/pause
        container.addEventListener('click', function() {
            if (video.paused) {
                // Pause all other videos first
                document.querySelectorAll('.feature-video').forEach(v => {
                    if (v !== video) {
                        v.pause();
                        v.parentElement.classList.remove('playing');
                    }
                });
                
                video.play();
                container.classList.add('playing');
            } else {
                video.pause();
                container.classList.remove('playing');
            }
        });
        
        // Show play button when video ends
        video.addEventListener('ended', function() {
            container.classList.remove('playing');
        });
        
        // Keyboard accessibility
        playButton.addEventListener('keydown', function(e) {
            if (e.key === 'Enter' || e.key === ' ') {
                e.preventDefault();
                video.play();
                container.classList.add('playing');
            }
        });
    });
    
    // Intersection Observer to pause videos when not visible
    const observer = new IntersectionObserver((entries) => {
        entries.forEach(entry => {
            const video = entry.target.querySelector('video');
            if (!entry.isIntersecting && !video.paused) {
                video.pause();
                entry.target.classList.remove('playing');
            }
        });
    }, { threshold: 0.5 });
    
    document.querySelectorAll('.video-container').forEach(container => {
        observer.observe(container);
    });
});
