  <script>
    function aktualizujStatus() {
      fetch('twitch.php')
          .then(response => response.json())
          .then(data => {
              const statusElement = document.getElementById('twitch-status');
              if (data.status === 'online') {
                  // Vytvoří iframe přehrávače
                  statusElement.innerHTML = `
                      <iframe
                          src="https://player.twitch.tv/?channel=kudriscz&parent=${window.location.hostname}"
                          height="300"
                          width="500"
                          frameborder="0"
                          scrolling="no"
                          allowfullscreen>
                      </iframe>
                  `;
              }
            })
          .catch(error => {
              console.error('Chyba při načítání dat:', error);
              document.getElementById('twitch-status').innerText = 'Chyba při načítání stavu.';
          });
  }
    aktualizujStatus();
    setInterval(aktualizujStatus, 60000);
  </script>
