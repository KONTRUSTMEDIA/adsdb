<script>
    document.addEventListener('DOMContentLoaded', () => {
        // --- 1. THEME LOGIC (Body Only) ---
        const toggleBtn = document.getElementById('theme-toggle');
        const htmlEl = document.documentElement;
        const icon = toggleBtn.querySelector('i');

        // Check LocalStorage or Default to Light
        const currentTheme = localStorage.getItem('theme') || 'light';
        applyTheme(currentTheme);

        // Click Event Listener
        toggleBtn.addEventListener('click', () => {
            const newTheme = htmlEl.getAttribute('data-theme') === 'light' ? 'dark' : 'light';
            applyTheme(newTheme);
            localStorage.setItem('theme', newTheme);
        });

        // Apply Theme Function
        function applyTheme(theme) {
            htmlEl.setAttribute('data-theme', theme);
            if (theme === 'dark') {
                icon.classList.remove('fa-moon');
                icon.classList.add('fa-sun');
                toggleBtn.classList.replace('btn-outline-light', 'btn-outline-warning'); 
            } else {
                icon.classList.remove('fa-sun');
                icon.classList.add('fa-moon');
                toggleBtn.classList.replace('btn-outline-warning', 'btn-outline-light');
            }
        }

        // --- 2. MOBILE MENU AUTO-CLOSE (Fixing the Click Outside Issue) ---
        const navMenu = document.getElementById('mainNav');
        const navToggler = document.querySelector('.navbar-toggler');

        document.addEventListener('click', function(event) {
            const isClickInside = navMenu.contains(event.target) || navToggler.contains(event.target);
            const isMenuOpen = navMenu.classList.contains('show');

            if (!isClickInside && isMenuOpen) {
                // If clicked outside and menu is open, close it programmatically
                const bsCollapse = new bootstrap.Collapse(navMenu, { toggle: true });
            }
        });

        // Also close menu when a link is clicked
        const navLinks = document.querySelectorAll('.nav-link-custom');
        navLinks.forEach(link => {
            link.addEventListener('click', () => {
                if (navMenu.classList.contains('show')) {
                    const bsCollapse = new bootstrap.Collapse(navMenu, { toggle: true });
                }
            });
        });
    });
</script>

<script>
document.addEventListener('DOMContentLoaded', () => {
    const track = document.getElementById('testiTrack');
    const viewport = document.getElementById('testiViewport');
    const originalCards = Array.from(document.querySelectorAll('.t-card'));
    
    // 1. CLONE CARDS FOR INFINITE LOOP
    // We append copies to ensure the scroll never "runs out" visually
    originalCards.forEach(card => {
        let clone = card.cloneNode(true);
        track.appendChild(clone);
    });
    originalCards.forEach(card => {
        let clone = card.cloneNode(true);
        track.appendChild(clone);
    });

    // 2. VARIABLES
    let allCards = Array.from(document.querySelectorAll('.t-card'));
    let cardWidth = 300; // Match CSS width
    let gap = 40; // Match CSS gap
    let itemWidth = cardWidth + gap;
    let totalOriginalWidth = originalCards.length * itemWidth;
    
    // Start in the middle set of clones to allow scrolling both ways immediately
    let scrollPos = totalOriginalWidth; 
    let speed = 0.8; // Speed of auto-scroll (Higher = Faster)
    let isPaused = false;

    // 3. ANIMATION LOOP
    function animate() {
        if (!isPaused) {
            scrollPos += speed;
        }

        // INFINITE RESET LOGIC
        // If we've scrolled past the first set, jump back instantly
        if (scrollPos >= totalOriginalWidth * 2) {
            scrollPos = totalOriginalWidth;
        } 
        // If we scrolled backwards past the start
        else if (scrollPos <= 0) {
            scrollPos = totalOriginalWidth;
        }

        // Apply Movement
        track.style.transform = `translateX(-${scrollPos}px)`;

        // HIGHLIGHT CENTER LOGIC
        highlightCenter();

        requestAnimationFrame(animate);
    }

    function highlightCenter() {
        // Calculate the center of the viewport relative to the track movement
        const viewportCenter = viewport.offsetWidth / 2;
        
        // Convert that into "Track Coordinates"
        const trackCenterPos = scrollPos + viewportCenter;

        let closestCard = null;
        let minDistance = Infinity;

        allCards.forEach((card, index) => {
            // Calculate where this specific card is on the track
            // (Index * ItemWidth) + (Half Item Width to get center of card)
            const cardCenter = (index * itemWidth) + (cardWidth / 2);
            
            const distance = Math.abs(trackCenterPos - cardCenter);

            if (distance < minDistance) {
                minDistance = distance;
                closestCard = card;
            }
        });

        // Apply Classes
        allCards.forEach(c => c.classList.remove('active'));
        if (closestCard) closestCard.classList.add('active');
    }

    // 4. START
    animate();

    // 5. PAUSE ON HOVER
    viewport.addEventListener('mouseenter', () => isPaused = true);
    viewport.addEventListener('mouseleave', () => isPaused = false);
    
    // Mobile Touch Pause
    viewport.addEventListener('touchstart', () => isPaused = true);
    viewport.addEventListener('touchend', () => isPaused = false);
});
</script>
