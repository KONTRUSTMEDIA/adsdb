                try {
                    await fetch(`${CONFIG.backendUrl}/api/web-click-track`, {
                        method: 'POST',
                        headers: { 'Content-Type': 'application/json' },
                        keepalive: true, 
                        body: JSON.stringify({ fbclid, ttclid, version: '9.0', user_agent: navigator.userAgent })
                    });
