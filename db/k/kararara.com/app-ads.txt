<script>
document.addEventListener('DOMContentLoaded', () => {
  const form = document.getElementById('searchForm');
  const qInput = document.getElementById('qInput');
  const box = document.getElementById('suggestionsBox');
  const micBtn = document.getElementById('micBtn');
  const clearBtn = document.getElementById('clearBtn');

  let controller = null;
  let activeIndex = -1;

  function showBox(){ box.classList.remove('d-none'); }
  function hideBox(){
    box.classList.add('d-none');
    box.innerHTML = '';
    activeIndex = -1;
  }
  function submitSearch(){
    if (form.requestSubmit) form.requestSubmit();
    else form.submit();
  }

  function toggleClear(){
    if (!clearBtn) return;
    const hasText = qInput.value.trim().length > 0;
    clearBtn.classList.toggle('d-none', !hasText);
  }

  async function fetchSuggest(query){
    if (controller) controller.abort();
    controller = new AbortController();

    const res = await fetch('/suggest.php?q=' + encodeURIComponent(query), {
      signal: controller.signal
    });
    if (!res.ok) return [];
    const data = await res.json();
    return Array.isArray(data) ? data : [];
  }

  function render(list){
    box.innerHTML = '';
    activeIndex = -1;

    list.slice(0, 7).forEach((text) => {
      const item = document.createElement('div');
      item.className = 'suggestion-item';
      item.dataset.value = text;
      item.innerHTML = `<span class="suggestion-type">ÖNERİ</span><span>${text}</span>`;

// --- Mobilde scroll vs tap ayrımı ---
let moved = false;

item.addEventListener('pointerdown', (e) => {
  // pointerdown'da seçim YAPMA! sadece "başladı" de.
  moved = false;
}, { passive: true });

item.addEventListener('pointermove', () => {
  moved = true; // kullanıcı sürükledi/scroll yaptı
}, { passive: true });

item.addEventListener('pointerup', (e) => {
  // Scroll yapıldıysa seçim yok
  if (moved) return;

  // Gerçek bir "tap/click" ise seç
  e.preventDefault();
  qInput.value = text;
  toggleClear();
  hideBox();
  submitSearch();
}, { passive: false });

// Desktop fallback: bazı eski tarayıcılarda pointer olayları sorun çıkarırsa
item.addEventListener('click', (e) => {
  e.preventDefault();
  qInput.value = text;
  toggleClear();
  hideBox();
  submitSearch();
});

      box.appendChild(item);
    });

    if (box.children.length) showBox();
    else hideBox();
  }

  function highlight(){
    const items = box.querySelectorAll('.suggestion-item');
    items.forEach((el, i) => el.classList.toggle('active', i === activeIndex));
    if (activeIndex >= 0 && items[activeIndex]) {
      items[activeIndex].scrollIntoView({ block: 'nearest' });
    }
  }

  qInput.addEventListener('input', async () => {
    toggleClear();

    const q = qInput.value.trim();
    if (q.length < 3) { hideBox(); return; }

    const list = await fetchSuggest(q);
    if (!list.length) { hideBox(); return; }
    render(list);
  });

  qInput.addEventListener('keydown', (e) => {
    const items = box.querySelectorAll('.suggestion-item');
    const open = items.length > 0 && !box.classList.contains('d-none');
    if (!open) return;

    if (e.key === 'ArrowDown') {
      e.preventDefault();
      activeIndex = (activeIndex + 1) % items.length;
      highlight();
      return;
    }
    if (e.key === 'ArrowUp') {
      e.preventDefault();
      activeIndex = (activeIndex - 1 + items.length) % items.length;
      highlight();
      return;
    }
    if (e.key === 'Enter' && activeIndex >= 0) {
      e.preventDefault();
      qInput.value = items[activeIndex].dataset.value;
      toggleClear();
      hideBox();
      submitSearch();
      return;
    }
    if (e.key === 'Escape') hideBox();
  });

  document.addEventListener('click', (e) => {
    if (e.target !== qInput && !box.contains(e.target)) hideBox();
  });

  /* ✅ Clear butonu */
  if (clearBtn){
    const clear = (e) => {
      e.preventDefault();
      qInput.value = '';
      hideBox();
      toggleClear();
      qInput.focus();
      qInput.dispatchEvent(new Event('input', { bubbles: true }));
    };

    clearBtn.addEventListener('mousedown', clear);
    clearBtn.addEventListener('touchstart', clear, { passive: false });
    clearBtn.addEventListener('click', clear);
  }

  /* Mikrofon */
  const SpeechRecognition = window.SpeechRecognition || window.webkitSpeechRecognition;
  let recognition = null;
  let listening = false;

  function setMicState(on){
    listening = on;
    micBtn.classList.toggle('listening', on);
    const icon = micBtn.querySelector('i');
    if (icon) icon.className = on ? 'bi bi-mic-fill' : 'bi bi-mic';
    micBtn.title = on ? 'Dinleniyor... Durdurmak için tıkla' : 'Sesle ara';
  }

  if (!SpeechRecognition) {
    micBtn.disabled = true;
    micBtn.title = 'Tarayıcı sesli aramayı desteklemiyor';
  } else {
    recognition = new SpeechRecognition();
    recognition.lang = 'tr-TR';
    recognition.interimResults = true;
    recognition.continuous = false;

    recognition.onstart = () => setMicState(true);
    recognition.onend = () => setMicState(false);
    recognition.onerror = () => setMicState(false);

    recognition.onresult = (event) => {
      let transcript = '';
      for (let i = event.resultIndex; i < event.results.length; i++) {
        transcript += event.results[i][0].transcript;
      }
      transcript = transcript.trim();
      if (transcript) {
        qInput.value = transcript;
        toggleClear();
        qInput.dispatchEvent(new Event('input', { bubbles: true }));
      }
    };

    micBtn.addEventListener('click', () => {
      try{
        if (listening) recognition.stop();
        else recognition.start();
      }catch(e){}
    });
  }
  /* Sayfa açılışında (q parametresi varsa) clear'ı göster */
  toggleClear();
});
</script>
