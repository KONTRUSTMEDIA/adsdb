                        loaderSpinner =

                        $(document).find("#search-result").html('');

                        let debounceSearchTimer;
                        let lastSearchRequest = null;

                        $(document).on("keyup keydown", "#searchParam", function(e) {
                            const $input = $(this);
                            const keyCode = e.keyCode || e.which;

                            if (keyCode === 13) {
                                e.preventDefault();
                                window.location.href = '/search?param=' + $input.val().trim();
                                return false;
                            }

                            const value = $input.val().trim();

                            if (value === "") {
                                $("#search-result").html('').hide();
                                return;
                            }

                            // Clear previous debounce timer
                            clearTimeout(debounceSearchTimer);

                            // Debounce execution by 500ms
                            debounceSearchTimer = setTimeout(() => {
                                removeLoad($input);
                                $input.after(loaderSpinner);

                                // Abort the previous request if it's still pending
                                if (lastSearchRequest && lastSearchRequest.readyState !== 4) {
                                    lastSearchRequest.abort();
                                }


                                // New AJAX request
                                lastSearchRequest = $.get("https://kalkinemedia.com/search-suggestions", {
                                    query: value
                                }, function(data) {
                                    if (value === $("#searchParam").val()) {
                                        $("#search-result").html(data).show();
                                    }
                                    removeLoad($input);
                                });
                            }, 500);
                        });

                        function removeLoad(val) {
                            val.parent().find("div.spinner-border").remove();
                        }

                        $(document).on("click", function (e) {
                            if (
                                !$(e.target).closest("#search-result").length
                            ) {
                                $("#search-result").hide();
                                $("#searchParam").val('');
                            }
                        });

    function formatZeroCount(number) {
        if (typeof number === 'string') {
            number = number.replace(/,/g, '');
        }

        number = String(number).trim();

        if (number === '') {
            return null;
        }

        let parsedNumber = parseFloat(number);
        if (isNaN(parsedNumber)) {
            return null;
        }

        let parts = parsedNumber.toString().split('.');
        let integerPart = parts[0];
        let decimalPart = parts[1] || '';

        let decimalTrimmed = decimalPart.replace(/0+$/, '');

        if (integerPart === '0') {
            let leadingZeros = decimalPart.match(/^0+/);
            let leadingZerosCount = leadingZeros ? leadingZeros[0].length : 0;
            // let remainingPart = decimalPart.replace(/^0+/, '');
            let remainingPart = decimalTrimmed.replace(/^0+/, '').slice(0, 8 - leadingZerosCount);

            if (leadingZerosCount >= 2) {
                return `$0.0<sub style="font-size: 60%;">${leadingZerosCount}</sub>${remainingPart}`;
            }
        }

        return "$" + parsedNumber.toFixed(3);
    }
