function newsletter_subscription(mode) {
    let type = '';
    let email = '';
    let name = '';
    let way = '';
    let agree = false;
    let ad = false;
    let form = null;

    if (mode === 'popup') {
        form = $("#subscribeNlForm");
    } else if (mode === 'body') {
        form = $("#subscribeNlForm2");
    } else {
        return false;
    }

    type = form.find('input[name=type]').val();
    email = form.find('input[name=email]').val();
    name = form.find('input[name=name]').val();
    way = form.find('input[name=way]').val();
    agree = form.find('input[name=agreeNlSubscribe1]').is(":checked");
    ad = form.find('input[name=agreeNlSubscribe2]').is(":checked");

    if (email === "") {
        alert('이메일 주소를 입력해주세요.');
        return false;
    }

    if (!/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email)) {
        alert('이메일 주소 형식이 아닙니다. 다시 확인해주세요.');
        return false;
    }

    if (name === "") {
        alert('닉네임을 입력해주세요.');
        return false;
    }

    if (/[`~!@#$%^&*|\\'" ;:\/?=+_()<>]/.test(name)) {
        alert('닉네임에 특수문자는 입력할 수 없습니다.');
        return false;
    }


    if (agree === false) {
        alert('개인정보 수집 및 이용에 동의해주세요.');
        return false;
    }

    let data = {
        type : type,
        email : email,
        name : name,
        ad : ad
    }

    if (type === 'cube') {
        data.way = way

        if (/[`~!@#$%^&*|\\'" ;:\/?=+_()<>]/.test(way)) {
            alert('점선면을 알게 된 경로에 특수문자는 입력할 수 없습니다.');
            return false;
        }

    }

    $.ajax({
        url : 'https://api.khan.co.kr/newsletter/index.html',
        method : 'POST',
        data : data,
        success : function(response) {
            if (response.status === '200') {
                if (type === 'flat' && response.is_update === false) {
                    closePopup_newsletter('subscribeNewsletterPopup');
                    openPopup_newsletter('subscribeNewsletterEmailPopup', type, email);
                } else {
                    closePopup_newsletter('subscribeNewsletterPopup');
                    openPopup_newsletter('subscribeNewsletterCompletePopup', null, null);
                    sessionStorage.setItem('newsletter_subscribed', sessionStorage.getItem('newsletter_subscribed') + "/" +  type);
                    $(".already-done").show();
                    if ($("#subscribeNlForm2").length) {
                        $("#subscribeNlForm1 .input-group").addClass('done');
                    }
                }
            } else {
                alert('뉴스레터 시스템의 응답이 지연되고 있습니다. 잠시 뒤에 다시 시도해주세요.');
            }

            $("#step1").removeClass('hidden');
            $("#step2").addClass('hidden');
            $("#step1 .fade-out").removeClass("fade-out");

            $("#step1 input[type=text]").val('');
            $("#step2 input[type=text]").val('');
            $("#subscribeNlForm input[type=text]").val('');
            $("#subscribeNlFormArticle input[type=text]").val('');

            if ($("input[name=agreeNlSubscribeAll]").prop("checked") === true) {
                $("#agreeNlSubscribeAll").trigger("click");
            } else {
                $("#agreeNlSubscribeAll").trigger("click");
                $("#agreeNlSubscribeAll").trigger("click");
            }
        },
        error : function(xhr) {
            alert('뉴스레터 시스템의 응답이 지연되고 있습니다. 잠시 뒤에 다시 시도해주세요.');
        }
    });
}

function openPopup_newsletter(id, type, email, event_label, area) {
    let selector = $("#"+id);

    selector.parent().addClass('active');

    if (id === 'subscribeNewsletterEmailPopup') {
        selector.find('.conts').html('<p>입력하신 이메일 주소(' + email + ')로 확인 메일을 보내드렸습니다.<br>메일 인증하시고 구독 신청을 완료해주세요. </p>')
    } else if (id === 'subscribeNewsletterPopup') {
        let subscribed_newsletter = sessionStorage.getItem('newsletter_subscribed');
        if (subscribed_newsletter !== null) {
            if (subscribed_newsletter.includes(type)) {
                $(".already-done").show();
            }
        }

        if (type === 'cube') {
            selector.parent().find('input[name=way]').parent().show();
        } else {
            selector.parent().find('input[name=way]').parent().hide();
        }

        selector.parent().find('input[name=type]').val(type);
        selector.parent().find('input[name=email]').val(email);
        
        if (area === 'main') {
            selector.find('.btn-subsc').attr('ep_event_area', '뉴스레터_프로모션_팝업_레터받기');
            selector.find('.btn-subsc').attr('ep_event_label', event_label);

            selector.find('.btn-close').attr('ep_event_area', '뉴스레터_프로모션_팝업_취소');
            selector.find('.btn-close').attr('ep_event_label', event_label);
        } else if (area === 'section') {
            selector.find('.btn-subsc').attr('ep_event_area', '섹션_뉴스레터_팝업_레터받기');
            selector.find('.btn-subsc').attr('ep_event_label', event_label);
            selector.find('.btn-subsc').attr('ep_event_action', '전환_' + event_label + '_뉴스레터_상세');

            selector.find('.btn-close').attr('ep_event_area', '섹션_뉴스레터_팝업_취소');
            selector.find('.btn-close').attr('ep_event_label', event_label);
            selector.find('.btn-close').attr('ep_event_action', '전환_' + event_label + '_뉴스레터_상세');
        } else {
            selector.find('.btn-subsc').attr('ep_event_area', 'article_bottom_팝업_레터받기');
            selector.find('.btn-subsc').attr('ep_event_label', event_label);

            selector.find('.btn-close').attr('ep_event_area', 'article_bottom_팝업_취소');
            selector.find('.btn-close').attr('ep_event_label', event_label);
        }
    }

    $(".dimm_area").show();
    $("body").addClass('not_scroll');
    if ($('.btn-top').length > 0) $('.btn-top').hide();
}

function closePopup_newsletter(obj) {
    let popupType;

    if (typeof obj === 'object') {
        popupType = obj.data('popup');
    } else {
        popupType = obj;
    }

    let selector = $("#"+popupType);
    selector.parent().removeClass('active');

    $(".dimm_area").hide();
    $('body').removeClass('not_scroll');
    if ($('.btn-top').length > 0) $('.btn-top').show();
}

$(document).ready(function() {
    let all = false;
    let agree1 = false;
    let agree2 = false;

    let subscribed_newsletter = sessionStorage.getItem('newsletter_subscribed');

    if (subscribed_newsletter !== null) {
        if ($("#subscribeNlForm2").length) {
            let type = $("#subscribeNlForm2 input[name=type]").val();
            if (subscribed_newsletter.includes(type)) {
                $("#subscribeNlForm1 .input-group").addClass('done');
                $("#subscribeNlForm1 .already-done").show();
            }
        }

        if ($("#subscribeNlFormArticle").length) {
            let type = $("#subscribeNlFormArticle input[name=type]").val();
            if (subscribed_newsletter.includes(type)) {
                $("#subscribeNlFormArticle .already-done").show();
            }
        }
    }

    $("input[name=agreeNlSubscribeAll]").click(function(e) {
        if (agree1 === true && agree2 === true) {
            all = false;
            agree1 = false;
            agree2 = false;

            $("input[name=agreeNlSubscribe1]").prop("checked", false);
            $("input[name=agreeNlSubscribe2]").prop("checked", false);
            $("input[name=agreeNlSubscribeAll]").prop("checked", false);
        } else {
            all = true;
            agree1 = true;
            agree2 = true;

            $("input[name=agreeNlSubscribeAll]").prop("checked", true);
            $("input[name=agreeNlSubscribe1]").prop("checked", true);
            $("input[name=agreeNlSubscribe2]").prop("checked", true);
        }
    });

    $("input[name=agreeNlSubscribe1]").click(function(e) {
        if (agree1 === true) {
            all = false;
            agree1 = false;
            $("input[name=agreeNlSubscribeAll]").prop("checked", false);
            $("input[name=agreeNlSubscribe1]").prop("checked", false);
        } else {
            agree1 = true;
            $("input[name=agreeNlSubscribe1").prop("checked", true);
            if (agree2 === true) {
                all = true;
                $("input[name=agreeNlSubscribeAll]").prop("checked", true);
            }
        }
    });

    $("input[name=agreeNlSubscribe2]").click(function(e) {
        if (agree2 === true) {
            all = false;
            agree2 = false;
            $("input[name=agreeNlSubscribeAll]").prop("checked", false);
            $("input[name=agreeNlSubscribe2]").prop("checked", false);
        } else {
            agree2 = true;
            $("input[name=agreeNlSubscribe2]").prop("checked", true);
            if (agree1 === true) {
                all = true;
                $("input[name=agreeNlSubscribeAll]").prop("checked", true);
            }
        }
    });

    $(".btn-policy-layer.txt").click(function(e) {
        let index = $('.btn-policy-layer.txt').index(this);
        if ($(this).hasClass("active")) {
            $(".btn-policy-layer.txt").eq(index).removeClass('active');
            $(".policy-box").eq(index).removeClass('active');
        } else {
            $(".btn-policy-layer.txt").eq(index).addClass('active');
            $(".policy-box").eq(index).addClass('active');
        }

        //let index = $('.btn-policy-layer.txt').index(this);
        //$(".btn-policy-layer.txt").removeClass("active");
        //$(".policy-box").removeClass('active');
        //$(".btn-policy-layer.txt").eq(index).addClass('active');
        //$(".policy-box").eq(index).addClass('active');
    });

    $(".dimm_area").click(function(e) {
        closePopup_newsletter('subscribeNewsletterPopup');
        closePopup_newsletter('subscribeNewsletterEmailPopup');
        closePopup_newsletter('subscribeNewsletterCompletePopup');
        closePopup_newsletter('subscribeNewsletterAgreePopup1');
        closePopup_newsletter('subscribeNewsletterAgreePopup2');
        $(".dimm_area").hide();
        $('body').removeClass('not_scroll');
        if ($('.btn-top').length > 0) $('.btn-top').show();
    });
});
