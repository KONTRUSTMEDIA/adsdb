  <script src="https://code.jquery.com/jquery-3.6.0.min.js"></script>
  <script>
  $(function () {
      const $popup = $('#popup');
      const $closeBtn = $popup.find('.close-btn');
      const $reopenTab = $('#reopenTab');

      function closePopup() {
        $popup.removeClass('visible');
        $reopenTab.show();
      }

      function openPopup() {
        $popup.addClass('visible');
        $reopenTab.hide();
      }

      $closeBtn.on('click', closePopup);
      $reopenTab.on('click keypress', function(e) {
        if (e.type === 'click' || e.key === 'Enter' || e.key === ' ') {
          openPopup();
        }
      });

      $reopenTab.on('keydown', function(e) {
        if(e.key === 'Enter' || e.key === ' ') {
          e.preventDefault();
          openPopup();
        }
      });

      // Démarrer avec volet fermé
      closePopup();

	  $(window).on('message', function(event) {
			const originalEvent = event.originalEvent;

			if(originalEvent.data && originalEvent.data.action === 'openRadio') {
				// Affiche la popup (webradio)
				$('#popup').addClass('visible');
				$('#reopenTab').hide();

				// Si un URL à charger est précisé, on charge dans la 2ème iframe
				if(originalEvent.data.pageToLoad) {
					// Assuming your 2nd iframe has id 'pageWeb'
					$('#pageWeb').attr('src', originalEvent.data.pageToLoad);
				}
			}
		});

	  // Si paramètre openRadio=1 dans l'URL du frameset, on ouvre aussi la webradio
	  // Et en plus on charge la page dans pageWeb si fourni dans pageToLoad param
	  const params = new URLSearchParams(window.location.search);
	  if(params.get('openRadio') === '1') {
  		  $('#popup').addClass('visible');
		  $('#reopenTab').hide();

		  // On regarde si pageToLoad est dans l'URL et charge la 2ème iframe
		  const pageToLoad = params.get('pageToLoad');
		  if(pageToLoad) {
  			  $('#pageWeb').attr('src', pageToLoad);
		  }
	}
  });
  </script>
