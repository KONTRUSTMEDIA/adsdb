  <script>
    const { createApp } = Vue
    const csrfCookieName = 'csrftoken'
    const statsDisabled = false

    function getCookie(name) {
      const value = `; ${document.cookie}`
      const parts = value.split(`; ${name}=`)
      if (parts.length === 2) {
        return parts.pop().split(';').shift()
      }
      return ''
    }

    function sendStatsEvent(name, value) {
      if (statsDisabled) return
      if (!name) return
      const payload = JSON.stringify({ name, value })
      if (navigator.sendBeacon) {
        const blob = new Blob([payload], { type: 'application/json' })
        navigator.sendBeacon('/internal-stats/events/', blob)
        return
      }
      fetch('/internal-stats/events/', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: payload,
        keepalive: true
      }).catch(() => {})
    }

    function sendStatsEvents(events) {
      if (statsDisabled) return
      if (!Array.isArray(events) || !events.length) return
      const payload = JSON.stringify({ events })
      if (navigator.sendBeacon) {
        const blob = new Blob([payload], { type: 'application/json' })
        navigator.sendBeacon('/internal-stats/events/', blob)
        return
      }
      fetch('/internal-stats/events/', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: payload,
        keepalive: true
      }).catch(() => {})
    }

    async function refreshCsrfTokens() {
      let token = getCookie(csrfCookieName)
      if (!token) {
        try {
          await fetch('/csrf/', { credentials: 'same-origin' })
        } catch (error) {
          return
        }
        token = getCookie(csrfCookieName)
      }
      if (!token) return
      document.querySelectorAll('input[name=csrfmiddlewaretoken]').forEach(input => {
        input.value = token
      })
    }

    if (document.readyState === 'loading') {
      document.addEventListener('DOMContentLoaded', refreshCsrfTokens)
    } else {
      refreshCsrfTokens()
    }

    const app = createApp({
      delimiters: ['[[', ']]'],
      data() {
        return {
          isLoading: false
        }
      },
      methods: {
        search() {
          this.isLoading = true
          const form = document.getElementById('form')
          const submitForm = () => {
            refreshCsrfTokens()
              .finally(() => {
                this.isLoading = false
                form.submit()
              })
          }
          navigator.geolocation.getCurrentPosition(position => {
            form.latitude.value = position.coords.latitude
            form.longitude.value = position.coords.longitude
            submitForm()
          }, err => {
            submitForm()
          })
        },
        trackEvent(name, value) {
          sendStatsEvent(name, value)
        },
        onProvinceChange(event) {
          const provinceId = event?.target?.value
          if (!provinceId) return
          const params = new URLSearchParams(window.location.search)
          params.set('province', provinceId)
          params.delete('city')
          const themeSelect = document.getElementById('id_theme')
          if (themeSelect && themeSelect.value) {
            params.set('theme', themeSelect.value)
          }
          window.location.search = params.toString()
        }
      }
    })
    app.mount('#main')
    window.faguTrackEvent = sendStatsEvent
    window.faguTrackEvents = sendStatsEvents
  </script>
