  <script>
    // Suppress known warnings (Flutter handles these internally)
    const originalConsoleWarn = console.warn;
    console.warn = function(...args) {
      // Filter out known warnings
      const message = args.join(' ');
      if (message.includes('prepareServiceWorker') && message.includes('4000ms')) {
        // Suppress service worker timeout warning - it's handled by Flutter
        return;
      }
      if (message.includes('google_sign_in plugin') && message.includes('signIn') && message.includes('deprecated')) {
        // Suppress deprecation warning - functionality still works, migration planned for later
        return;
      }
      originalConsoleWarn.apply(console, args);
    };

    // Hide loading indicator when Flutter app loads
    let loadingHidden = false;
    let appReady = false;
    
    function hideLoading() {
      if (loadingHidden) return;
      loadingHidden = true;
      const loading = document.getElementById('loading');
      if (loading) {
        // Immediately hide and remove - no fade animation to prevent blocking
        loading.style.display = 'none';
        loading.remove();
        document.body.style.overflow = '';
        console.log('✅ Loading screen removed');
      }
    }
    
    function showError(message) {
      const loading = document.getElementById('loading');
      if (loading && !loadingHidden) {
        loading.innerHTML = `
          <div style="text-align: center; color: #fff;">
            <div style="font-size: 48px; margin-bottom: 16px;">⚠️</div>
            <p style="font-size: 18px; font-weight: 500; margin-bottom: 8px;">${message}</p>
            <p style="font-size: 14px; opacity: 0.9; margin-bottom: 24px;">Please refresh the page</p>
            <button onclick="window.location.reload()" style="padding: 12px 24px; background: #fff; color: #9C27B0; border: none; border-radius: 6px; font-size: 16px; font-weight: 500; cursor: pointer;">Refresh Page</button>
          </div>
        `;
      }
    }

    // Hide on flutter-first-frame event
    window.addEventListener('flutter-first-frame', function() {
      appReady = true;
      console.log('✅ Flutter first frame rendered');
      // Hide loading immediately when first frame renders
      hideLoading();
    });
    
    // IMMEDIATE: Hide loading screen right away - don't wait for anything
    // This ensures the Flutter app is visible immediately
    setTimeout(function() {
      if (!loadingHidden) {
        console.log('⚠️ IMMEDIATE: Force hiding loading screen');
        appReady = true;
        hideLoading();
      }
    }, 0); // 0ms - immediate
    
    // Also check for Flutter app container immediately
    setTimeout(function() {
      if (!loadingHidden) {
        const flutterContainer = document.querySelector('flt-glass-pane, flt-scene-host, body > *:not(#loading)');
        if (flutterContainer) {
          console.log('✅ Flutter container found, hiding loading');
          appReady = true;
          hideLoading();
        } else {
          // Even if container not found, hide loading
          console.log('⚠️ Container not found but hiding loading anyway');
          hideLoading();
        }
      }
    }, 0); // 0ms - immediate
    
    // Check immediately on DOM ready
    if (document.readyState === 'loading') {
      document.addEventListener('DOMContentLoaded', function() {
        if (!loadingHidden) {
          console.log('⚠️ DOM ready - force hiding loading');
          hideLoading();
        }
      });
    } else {
      // DOM already ready - hide immediately
      if (!loadingHidden) {
        console.log('⚠️ DOM already ready - force hiding loading');
        hideLoading();
      }
    }
    
    // Check for Flutter errors
    window.addEventListener('error', function(e) {
      console.error('Global error:', e.error);
      if (!appReady && e.error && e.error.toString().includes('flutter')) {
        showError('Failed to load application');
      }
    });
    
    // Check for unhandled promise rejections
    window.addEventListener('unhandledrejection', function(e) {
      console.error('Unhandled promise rejection:', e.reason);
    });
    
    // Fallback: Show error if app doesn't load in 15 seconds
    setTimeout(() => {
      if (!appReady && !loadingHidden) {
        console.warn('⚠️ App loading timeout - checking if Flutter initialized');
        // Check if Flutter bootstrap loaded
        if (window.flutter_bootstrap) {
          console.log('Flutter bootstrap exists, waiting longer...');
          setTimeout(() => {
            if (!appReady) {
              showError('Application is taking longer than expected to load');
            }
          }, 5000);
        } else {
          showError('Failed to load application. Please check your internet connection.');
        }
      }
    }, 15000);
    
    // Monitor for Flutter app initialization (fallback) - very frequent checks
    let checkInterval = setInterval(function() {
      if (appReady || loadingHidden) {
        clearInterval(checkInterval);
        return;
      }
      // Check if Flutter app container exists and has content
      const flutterContainer = document.querySelector('flt-glass-pane, flt-scene-host, body > *:not(#loading)');
      if (flutterContainer && flutterContainer.children.length > 0) {
        console.log('✅ Flutter container detected with content');
        appReady = true;
        clearInterval(checkInterval);
        hideLoading();
      }
      // Also check if any script has loaded
      if (window.flutter_bootstrap || window.flutter) {
        console.log('✅ Flutter bootstrap detected');
        appReady = true;
        clearInterval(checkInterval);
        hideLoading();
      }
      // Force hide after 100ms regardless
      if (Date.now() - window.pageLoadTime > 100) {
        console.log('⚠️ 100ms passed - force hiding loading');
        appReady = true;
        clearInterval(checkInterval);
        hideLoading();
      }
    }, 10); // Check every 10ms - very aggressive
    
    // Track page load time
    window.pageLoadTime = Date.now();
    
    // Clear interval after 200ms (loading will be force-hidden by then)
    setTimeout(() => clearInterval(checkInterval), 200);
