        function searchGames() {
            var input = document.activeElement.id === 'searchInput' ?
                document.getElementById('searchInput') :
                document.getElementById('searchInput-mobile');
            var filter = input.value.toLowerCase().trim();

            var popular = document.querySelectorAll('.popular-section');
            var newest = document.querySelectorAll('.new-section');
            var moreHeader = document.querySelector('h2.section-heading.more-section');
            var moreGames = document.querySelectorAll('.more-section .game');
            var moreAd = document.getElementById('more-ad');
            var gameCardAds = document.querySelectorAll('.game-card-ad, .game-card-ad + .ad-label');
            var afterPopularAd = document.querySelector('.after-popular');

            if (filter.length > 0) {
                popular.forEach(el => el.style.display = 'none');
                newest.forEach(el => el.style.display = 'none');
                if (moreHeader) {
                    moreHeader.style.display = 'none';
                }
                if (moreAd) {
                    moreAd.style.display = 'none';
                }
                if (afterPopularAd) {
                    afterPopularAd.parentElement.style.display = 'none';
                }
                gameCardAds.forEach(ad => {
                    ad.parentElement.style.display = 'none';
                });
            } else {
                popular.forEach(el => el.style.display = '');
                newest.forEach(el => el.style.display = '');
                if (moreHeader) {
                    moreHeader.style.display = '';
                }
                if (moreAd) {
                    moreAd.style.display = '';
                }
                if (afterPopularAd) {
                    afterPopularAd.parentElement.style.display = '';
                }
                gameCardAds.forEach(ad => {
                    ad.parentElement.style.display = '';
                });
            }

            var found = false;
            moreGames.forEach(game => {
                var txtValue = (game.textContent || game.innerText).toLowerCase();
                if (txtValue.indexOf(filter) > -1) {
                    game.style.display = '';
                    found = true;
                } else {
                    game.style.display = 'none';
                }
            });
            document.getElementById('noResults').style.display = found ? 'none' : 'block';
        }

<script>
document.addEventListener('DOMContentLoaded', function() {
    const switcher = document.getElementById('language-switcher');
    if (!switcher) return;

    const button = switcher.querySelector('.language-switcher-button');

    button.addEventListener('click', function(event) {
        event.stopPropagation();
        switcher.classList.toggle('open');
    });

    document.addEventListener('click', function(event) {
        if (switcher.classList.contains('open') && !switcher.contains(event.target)) {
            switcher.classList.remove('open');
        }
    });

    document.addEventListener('keydown', function(event) {
        if (event.key === 'Escape' && switcher.classList.contains('open')) {
            switcher.classList.remove('open');
        }
    });
});
</script>

// Modern Mobile Navigation Implementation
class MobileNavigation {
    constructor() {
        this.overlay = document.getElementById('mobileNavOverlay');
        this.toggleButton = document.querySelector('.mobile-menu-toggle');
        this.closeButton = document.querySelector('.mobile-nav-close');
        this.isOpen = false;
        
        this.init();
    }
    
    init() {
        if (!this.overlay || !this.toggleButton || !this.closeButton) return;
        
        // Event listeners
        this.toggleButton.addEventListener('click', () => this.toggle());
        this.closeButton.addEventListener('click', () => this.close());
        
        // Close when clicking outside content area
        this.overlay.addEventListener('click', (e) => {
            if (e.target === this.overlay) {
                this.close();
            }
        });
        
        // Close on escape key
        document.addEventListener('keydown', (e) => {
            if (e.key === 'Escape' && this.isOpen) {
                this.close();
            }
        });
        
        // Prevent body scroll when menu is open
        this.overlay.addEventListener('transitionend', () => {
            if (!this.isOpen) {
                document.body.style.overflow = '';
            }
        });
    }
    
    toggle() {
        this.isOpen ? this.close() : this.open();
    }
    
    open() {
        this.isOpen = true;
        this.overlay.classList.add('active');
        this.toggleButton.classList.add('active');
        document.body.style.overflow = 'hidden';
        
        // Focus management for accessibility
        this.closeButton.focus();
    }
    
    close() {
        this.isOpen = false;
        this.overlay.classList.remove('active');
        this.toggleButton.classList.remove('active');
        document.body.style.overflow = '';
        
        // Return focus to toggle button
        this.toggleButton.focus();
    }
}

// Initialize when DOM is ready
document.addEventListener('DOMContentLoaded', function() {
    // Initialize mobile navigation
    new MobileNavigation();

    const desktopSearch = document.getElementById('searchInput');
    const mobileSearch = document.getElementById('searchInput-mobile');
    if (typeof window.searchGames === 'function') {
        if (desktopSearch) desktopSearch.addEventListener('input', window.searchGames);
        if (mobileSearch) mobileSearch.addEventListener('input', window.searchGames);
    }
});
