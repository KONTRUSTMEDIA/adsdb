    window.navigator.standalone === true ||
    (window.matchMedia && window.matchMedia('(display-mode: standalone)').matches);

  // In-app browsers: herkenbare tokens (FB/IG vrijwel altijd)

  const isInApp = /FBAN|FBAV|Instagram|LinkedInApp|Line|Snapchat|TikTok|Pinterest/i.test(ua);

  // Echte Safari = Safari én NIET in-app
  const isSafariReal = isSafari && !isInApp;

  const KEY = 'fw_install_hidden_until';
  const now = Date.now();
  const hiddenUntil = parseInt(localStorage.getItem(KEY) || '0', 10);

  function hideForDays(el, days) {
