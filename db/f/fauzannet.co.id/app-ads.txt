                        }
                        let joinDate = new Date(m.created_at).toLocaleDateString('id-ID', { year: 'numeric', month: 'short' });

                        container.innerHTML += `

<script>
    // --- 1. LOGIKA LOGIN ---
    
    function toggleLoginModal() {
        const modal = document.getElementById('modalLogin');
        modal.classList.toggle('hidden');
    }

    function handleLogin(e) {
        e.preventDefault(); 
        
        const btn = document.getElementById('btnLogin');
        const originalText = btn.innerHTML;
        const formData = new FormData(e.target);

        document.getElementById('loginError').classList.add('hidden');

        btn.innerHTML = '<i class="ti ti-loader animate-spin mr-2"></i> Memproses...'; 
        btn.disabled = true;

        // [PERBAIKAN] Tambahkan 'credentials: include' agar Cookie Session terbawa di Safari/Edge
        // [PERBAIKAN] Pastikan BASEURL di Config.php menggunakan HTTPS jika website sudah SSL
        fetch('https://www.fauzannet.co.id/authajax/login', { 
            method: 'POST', 
            body: formData,
            credentials: 'include' // PENTING UNTUK SAFARI/IPHONE
        })
        .then(res => {
            if (!res.ok) throw new Error("HTTP Status: " + res.status);
            return res.text().then(text => {
                try {
                    return JSON.parse(text); 
                } catch (err) {
                    console.error("Raw Response:", text); 
                    throw new Error("Respon server tidak valid (Bukan JSON).");
                }
            });
        })
        .then(data => {
            if(data.status === 'success') { 
                location.reload(); 
            } else { 
                document.getElementById('loginErrorMsg').innerText = data.message;
                document.getElementById('loginError').classList.remove('hidden');
                
                btn.innerHTML = originalText; 
                btn.disabled = false;
                if(typeof grecaptcha !== 'undefined') grecaptcha.reset(); 
            }
        })
        .catch(err => { 
            console.error(err); 
            // Jika error karena sesi hilang, beri pesan yang jelas
            let msg = 'Gagal terhubung ke server. Coba refresh halaman.';
            if(err.message.includes('JSON')) msg = 'Terjadi kesalahan sistem (Session expired). Silakan refresh.';

            Swal.fire({
                icon: 'error',
                title: 'Gagal Masuk',
                text: msg
            });
            btn.innerHTML = originalText; 
            btn.disabled = false; 
        });
    }

    // --- 2. LOGIKA LUPA PASSWORD ---
    
    let userData = {}; // Simpan data sementara user

    function openForgot() {
        document.getElementById('modalLogin').classList.add('hidden');
        document.getElementById('modalForgot').classList.remove('hidden');
        showStep(1);
    }
    
    function closeForgot() {
        document.getElementById('modalForgot').classList.add('hidden');
    }
    
    function showStep(n) {
        for(let i=1; i<=4; i++) document.getElementById('forgotStep'+i).classList.add('hidden');
        document.getElementById('forgotStep'+n).classList.remove('hidden');
    }

    // Step 1: Cari Akun
    function cekAkun() {
        const id = document.getElementById('f_identity').value;
        if(!id) return Swal.fire('Error', 'Masukan identitas!', 'error');

        const btn = document.getElementById('btnCekAkun');
        btn.innerHTML = 'Mencari...'; btn.disabled = true;

        const fd = new FormData(); fd.append('identity', id);
        fetch('https://www.fauzannet.co.id/authajax/check_account', { method: 'POST', body: fd })
        .then(res => res.json())
        .then(res => {
            btn.innerHTML = 'Cari Akun'; btn.disabled = false;
            if(res.status === 'success') {
                userData = res.data;
                renderStep2();
            } else {
                Swal.fire('Gagal', res.message, 'error');
            }
        })
        .catch(err => {
            btn.innerHTML = 'Cari Akun'; btn.disabled = false;
            Swal.fire('Error', 'Gagal koneksi ke server', 'error');
        });
    }

    // Step 2: Render Pilihan Metode OTP
    function renderStep2() {
        document.getElementById('f_nama').innerText = userData.nama;
        document.getElementById('f_avatar').innerText = userData.nama.charAt(0);
        document.getElementById('f_id').innerText = 'ID Member : ' + userData.id_member;
        
        const cont = document.getElementById('methodContainer');
        cont.innerHTML = '';

        if(userData.methods.includes('wa')) {
            cont.innerHTML += `<button onclick="sendOTP('wa')" class="w-full flex items-center justify-between p-3 border border-green-200 bg-green-50 text-green-700 rounded-lg hover:bg-green-100 font-bold text-sm">
                <span><i class="ti ti-brand-whatsapp mr-2"></i> WhatsApp</span>
                <span>${userData.mask_wa}</span>
            </button>`;
        }
        if(userData.methods.includes('email')) {
            cont.innerHTML += `<button onclick="sendOTP('email')" class="w-full flex items-center justify-between p-3 border border-blue-200 bg-blue-50 text-blue-700 rounded-lg hover:bg-blue-100 font-bold text-sm">
                <span><i class="ti ti-mail mr-2"></i> Email</span>
                <span>${userData.mask_email}</span>
            </button>`;
        }
        showStep(2);
    }

    // Step 3: Kirim OTP
    function sendOTP(method) {
        const target = (method === 'email') ? userData.real_email : userData.real_wa;
        userData.target = target; // Simpan target aktif untuk verifikasi

        Swal.fire({title: 'Mengirim OTP...', didOpen: () => Swal.showLoading()});

        const fd = new FormData();
        fd.append('method', method);
        fd.append('target', target);

        fetch('https://www.fauzannet.co.id/authajax/send_otp', { method: 'POST', body: fd })
        .then(res => res.json())
        .then(res => {
            Swal.close();
            if(res.status === 'success') {
                document.getElementById('targetSent').innerText = (method=='email' ? userData.mask_email : userData.mask_wa);
                showStep(3);
                startTimer();
            } else {
                Swal.fire('Gagal Kirim', res.message, 'error');
            }
        });
    }

    // Timer Mundur OTP
    let timer;
    function startTimer() {
        let sec = 60;
        clearInterval(timer);
        timer = setInterval(() => {
            document.getElementById('countdown').innerText = sec;
            sec--;
            if(sec < 0) clearInterval(timer);
        }, 1000);
    }

    // Step 4: Verifikasi OTP
    function verifyOTP() {
        const code = document.getElementById('otpCode').value;
        const btn = document.getElementById('btnVerify');
        btn.innerHTML = 'Verifikasi...';

        const fd = new FormData();
        fd.append('target', userData.target);
        fd.append('token', code);

        fetch('https://www.fauzannet.co.id/authajax/verify_otp', { method: 'POST', body: fd })
        .then(res => res.json())
        .then(res => {
            btn.innerHTML = 'Verifikasi';
            if(res.status === 'success') {
                showStep(4);
            } else {
                Swal.fire('Gagal', res.message, 'error');
            }
        });
    }

    // Cek Kekuatan Password
    function checkStrength(pass) {
        const bar = document.getElementById('strengthBar');
        const txt = document.getElementById('strengthText');
        const btn = document.getElementById('btnResetFinal');
        let score = 0;

        if(pass.length >= 8) score++;
        if(/[A-Z]/.test(pass)) score++;
        if(/[a-z]/.test(pass)) score++;
        if(/[0-9]/.test(pass)) score++;
        if(/[^A-Za-z0-9]/.test(pass)) score++;

        if(pass.length < 8) {
            bar.style.width = '10%'; bar.className = 'h-full bg-red-500 transition-all';
            txt.innerText = 'Terlalu Pendek'; txt.className = 'text-[10px] mt-1 text-right text-red-500';
            btn.disabled = true; btn.className = 'w-full bg-gray-300 text-white font-bold py-3 rounded-lg cursor-not-allowed';
        } else if(score < 3) { 
            bar.style.width = '50%'; bar.className = 'h-full bg-yellow-500 transition-all';
            txt.innerText = 'Cukup'; txt.className = 'text-[10px] mt-1 text-right text-yellow-600';
            btn.disabled = false; btn.className = 'w-full bg-sky-600 text-white font-bold py-3 rounded-lg';
        } else if(score >= 4) { 
            bar.style.width = '100%'; bar.className = 'h-full bg-green-500 transition-all';
            txt.innerText = 'Sangat Kuat'; txt.className = 'text-[10px] mt-1 text-right text-green-600';
            btn.disabled = false; btn.className = 'w-full bg-sky-600 text-white font-bold py-3 rounded-lg';
        }
    }

    // Step 5: Reset Password Final
    function resetPassFinal() {
        const p1 = document.getElementById('newPass').value;
        const p2 = document.getElementById('confirmPass').value;

        if(p1 !== p2) return Swal.fire('Error', 'Konfirmasi password tidak cocok!', 'error');

        const fd = new FormData();
        fd.append('id_user', userData.id_user);
        fd.append('new_pass', p1);

        fetch('https://www.fauzannet.co.id/authajax/reset_password', { method: 'POST', body: fd })
        .then(res => res.json())
        .then(res => {
            if(res.status === 'success') {
                Swal.fire('Sukses', 'Password berhasil diubah. Silakan login.', 'success')
                .then(() => location.reload());
            } else {
                Swal.fire('Error', 'Gagal mereset password.', 'error');
            }
        });
    }

    function acceptCookie() {
        const banner = document.getElementById('cookie-banner');
        
        // 1. Simpan status ke LocalStorage browser user
        localStorage.setItem('fauzannet_cookie_consent', 'true');
        
        // 2. Animasi slide down (sembunyi)
