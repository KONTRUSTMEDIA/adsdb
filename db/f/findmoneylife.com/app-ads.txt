<script>
function setPopupCookie(name, days) {
  const d = new Date();
  d.setTime(d.getTime() + (days * 24 * 60 * 60 * 1000));
  document.cookie = name + "=Y; expires=" + d.toUTCString() + "; path=/";
}

function hasPopupCookie(name) {
  return document.cookie.indexOf(name + "=Y") > -1;
}

document.addEventListener("DOMContentLoaded", function () {
  const overlay = document.getElementById("eventPopup");
  const popups = document.querySelectorAll(".popup-item");

  let visibleCount = 0;

  popups.forEach(popup => {
    const key = popup.dataset.popup;
    if (hasPopupCookie(key)) {
      popup.style.display = "none";
    } else {
      visibleCount++;
    }
  });

  if (visibleCount > 0) {
    overlay.style.display = "block";
  }

  document.querySelectorAll(".popupCloseBtn").forEach(btn => {
    btn.addEventListener("click", function () {
      const popup = btn.closest(".popup-item");
      const key = popup.dataset.popup;
      const checkbox = popup.querySelector(".popupTodayClose");

      if (checkbox.checked) {
        setPopupCookie(key, 1);   // ⭐ 팝업별 쿠키
      }

      popup.style.display = "none";

      // 남은 팝업 없으면 배경 닫기
      const remain = Array.from(popups).some(p => p.style.display !== "none");
      if (!remain) {
        overlay.style.display = "none";
      }
    });
  });
});
</script>
