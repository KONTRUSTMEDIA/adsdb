    const expiryTimestamp = 0 * 1000; // JS uses ms
    const countdownEl = document.getElementById('countdown');
    const linkEl = document.getElementById('forgotLink');

    if (countdownEl && expiryTimestamp > Date.now()) {
        const timer = setInterval(() => {
            const now = new Date().getTime();
            const distance = expiryTimestamp - now;

            if (distance <= 0) {
                clearInterval(timer);
                countdownEl.innerHTML = '';
                linkEl.removeAttribute('style');
                linkEl.removeAttribute('onclick');

            const minutes = Math.floor((distance % (1000 * 60 * 60)) / (1000 * 60));

            countdownEl.innerHTML = `(${minutes}:${seconds < 10 ? '0' + seconds : seconds} left)`;
        }, 1000);
    }
