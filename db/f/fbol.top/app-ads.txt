<script>
  // Função para definir a frase de acordo com o idioma do navegador
  function setMessage() {
    var language = navigator.language || navigator.userLanguage;
    var messageElement = document.getElementById("message");
    var adTextElement = document.getElementById("adText");
    var message;
    var adText;
    var buttonText;

    switch (language) {
      case 'en': // Inglês
      case 'en-US':
        message = "Your link is being generated!";
        buttonText = "GET LINK";
        break;
      case 'es': // Espanhol
      case 'es-ES':
        message = "¡Su enlace se está generando!";
        buttonText = "OBTENER ENLACE";
        break;
      case 'fr': // Francês
      case 'fr-FR':
        message = "Votre lien est en cours de génération !";
        buttonText = "OBTENIR LE LIEN";
        break;
      case 'de': // Alemão
      case 'de-DE':
        message = "Ihr Link wird generiert!";
        buttonText = "LINK ERHALTEN";
        break;
      case 'it': // Italiano
      case 'it-IT':
        message = "Il tuo link è in fase di generazione!";
        buttonText = "OTTENERE IL LINK";
        break;
      case 'pt': // Português
      case 'pt-BR':
      default:   // Padrão para português
        message = "Seu link está sendo gerado!";
        buttonText = "OBTER LINK";
        break;
    }

    messageElement.textContent = message;
    adTextElement.textContent = adText;
    return buttonText;
  }

  // Chama a função para definir a mensagem e o texto do botão
  var buttonText = setMessage();

  // Código do cronômetro
  var targetTime = 7; // 10 seconds
  var currentTime = 0;
  
  var canvas = document.getElementById("timerCircle");
  var context = canvas.getContext("2d");
  var centerX = canvas.width / 2;
  var centerY = canvas.height / 2;
  var radius = 70;
  var startingAngle = 1.5 * Math.PI;
  var endingAngle = 1.5 * Math.PI;
  var counterClockwise = false;
  
  var interval = setInterval(function() {
    currentTime += 0.01; // increment time by 10 milliseconds
    endingAngle = 1.5 * Math.PI + (2 * Math.PI * currentTime / targetTime);
    
    context.clearRect(0, 0, canvas.width, canvas.height);
    context.beginPath();
    context.arc(centerX, centerY, radius, startingAngle, endingAngle, counterClockwise);
    context.lineWidth = 10;
    context.strokeStyle = "#000000";
    context.stroke();
    
    context.font = "20px 'Poppins'";
    context.fillStyle = "#000000";
    context.textAlign = "center";
    context.textBaseline = "middle";
    context.fillText(Math.floor(currentTime) + "s", centerX, centerY);
    
    if (currentTime >= targetTime) {
      clearInterval(interval);
      var button = document.createElement("button");
      button.id = "getLinkButton";
      button.innerHTML = buttonText; // Definir o texto do botão traduzido
      button.onclick = function() {
        window.location.href = "https://z1.linkmonetizado.com";
      };
      document.getElementById("linkContainer").appendChild(button);
    }
  }, 10);
</script>
