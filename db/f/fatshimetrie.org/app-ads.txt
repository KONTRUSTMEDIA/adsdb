  function getQuotesForLang(lang) {
    if (QUOTES[lang] && QUOTES[lang].length) return QUOTES[lang];
    if (QUOTES.en && QUOTES.en.length) return QUOTES.en;
    return QUOTES.fr || [];
  }

  const quotes = getQuotesForLang(LANG);
  let idx = Math.floor(Math.random() * (quotes.length || 1));

  let quoteTimer = null;
  let showDelayTimer = null;
  let slowTimer = null;

  let loaderShown = false;
  let alreadyHidden = false;

  function showQuote() {
    if (!quoteEl || !quotes.length) return;
    quoteEl.classList.remove("hidden");
    quoteEl.textContent = quotes[idx % quotes.length];
    idx += 1;
  }

  function cleanupTimers() {
    if (showDelayTimer) clearTimeout(showDelayTimer);
    if (slowTimer) clearTimeout(slowTimer);
    if (quoteTimer) clearInterval(quoteTimer);
    showDelayTimer = null;
    slowTimer = null;
    quoteTimer = null;
  }

  // --- API: afficher le loader maintenant (pour clics / submits) ---
  function showLoaderImmediate() {
    // ✅ IMPORTANT: on ne bloque PAS si alreadyHidden = true
    // parce qu’après le load, ton code supprime le loader du DOM
    // => on ne pourra plus l’afficher. On gère ça plus bas.
    if (!loader) return;

    loader.classList.remove("hidden");
    loaderShown = true;

    // si c’est long, quote après 2.5s
    if (!slowTimer) {
      slowTimer = setTimeout(() => {
        showQuote();
        quoteTimer = setInterval(showQuote, 3200);
      }, 2500);
    }
  }

  // --- Mode "page load": afficher uniquement si la page traîne (150ms) ---
  function initLoadTimers() {
    showDelayTimer = setTimeout(() => {
      loader.classList.remove("hidden");
      loaderShown = true;
    }, 150);

    slowTimer = setTimeout(() => {
      if (!loaderShown) {
        loader.classList.remove("hidden");
        loaderShown = true;
      }
      showQuote();
      quoteTimer = setInterval(showQuote, 3200);
    }, 2500);
  }

  function hideLoader() {
    if (alreadyHidden) return;
    alreadyHidden = true;

    cleanupTimers();

    // si le loader n'a même pas été affiché, rien à faire
    if (!loaderShown) return;

    loader.style.opacity = "0";
    loader.style.transition = "opacity .25s ease";
    setTimeout(() => {
      if (loader && loader.parentNode) loader.parentNode.removeChild(loader);
    }, 260);
  }

  // Expose pour usage ailleurs si tu veux
  window.showPageLoader = showLoaderImmediate;

  // ✅ Déclencheur au clic sur liens + submit (NOUVEAU)
  function shouldHandleLink(a, e) {
    if (!a) return false;

    const hrefAttr = (a.getAttribute("href") || "").trim();
    if (!hrefAttr || hrefAttr === "#") return false;
    if (hrefAttr.startsWith("#")) return false;
    if (hrefAttr.toLowerCase().startsWith("javascript:")) return false;
    if (hrefAttr.startsWith("mailto:") || hrefAttr.startsWith("tel:")) return false;

    if (a.hasAttribute("download")) return false;
    if ((a.getAttribute("target") || "").toLowerCase() === "_blank") return false;

    if (e.metaKey || e.ctrlKey || e.shiftKey || e.altKey) return false;

    // Only internal links (same origin)
    let url;
    try { url = new URL(a.href, window.location.href); }
    catch (err) { return false; }

    if (url.origin !== window.location.origin) return false;

    // same page => pas de loader
    if (url.href === window.location.href) return false;

    return true;
  }

  document.addEventListener("click", function (e) {
    const a = e.target.closest("a");
    if (!a) return;
    if (!shouldHandleLink(a, e)) return;

    // ✅ NOUVEAU: afficher loader au clic
    // MAIS ton hideLoader() supprime le loader du DOM après load,
    // donc après le premier chargement, il n’existe plus.
    // => solution: recréer un loader minimal si on ne le trouve plus.
    const existing = document.getElementById("page-loader");

      tmp.id = "page-loader";
      tmp.className = "fixed inset-0 z-[99999] bg-white flex items-center justify-center";
      tmp.innerHTML = `

          <div class="mt-4 text-sm text-slate-700 font-semibold">Chargement…</div>
          <div class="mt-4 w-56 h-1.5 bg-slate-100 rounded-full overflow-hidden mx-auto">
            <div class="h-full w-1/2 bg-slate-900 rounded-full animate-[loaderbar_1.2s_ease-in-out_infinite]"></div>
          </div>
        </div>
      `;
      document.body.appendChild(tmp);
    }

    const href = a.href;
    e.preventDefault();
    setTimeout(() => {
      window.location.href = href;
    }, 50);
  }, { capture: true });

  document.addEventListener("submit", function (e) {
    const form = e.target;
    if (!form) return;
    if ((form.getAttribute("target") || "").toLowerCase() === "_blank") return;

    const existing = document.getElementById("page-loader");
    if (existing) showLoaderImmediate();
  }, { capture: true });

  // ✅ Comportement d’origine CONSERVÉ
  initLoadTimers();
  window.addEventListener("load", hideLoader);

})();

<script>
(function () {
  // Règles stockage
  const KEY_SHOWN_AT = "global_invite_shown_at";
  const KEY_DISMISSED_UNTIL = "global_invite_dismissed_until";

  const ONE_DAY = 24 * 60 * 60 * 1000;

  function canShowNow() {
    // 1) ne pas spammer si dismissed
    const dismissedUntil = parseInt(localStorage.getItem(KEY_DISMISSED_UNTIL) || "0", 10);
    if (dismissedUntil && Date.now() < dismissedUntil) return false;

    // 2) max 1 fois / jour
    const shownAt = parseInt(localStorage.getItem(KEY_SHOWN_AT) || "0", 10);
    if (shownAt && (Date.now() - shownAt) < ONE_DAY) return false;

    return true;
  }

  function markShown() {
    localStorage.setItem(KEY_SHOWN_AT, String(Date.now()));
  }

  function dismiss(days) {
    const until = Date.now() + (days * ONE_DAY);
    localStorage.setItem(KEY_DISMISSED_UNTIL, String(until));
  }

  function isPushAlreadyGranted() {
    if (!("Notification" in window)) return false;
    return Notification.permission === "granted";
  }

  function openInvite(mode) {
    if (typeof window.openGlobalInvite !== "function") return;
    window.openGlobalInvite(mode);
    markShown();
  }

  // ---- AUTO RULES ----
  window.addEventListener("load", function () {
    // Si push déjà activé => ne pas montrer
    if (isPushAlreadyGranted()) return;

    if (!canShowNow()) return;

    // 1) après 8 secondes
    setTimeout(() => {
      // re-check (user may have enabled in the meantime)
      if (isPushAlreadyGranted()) return;
      if (!canShowNow()) return;

      openInvite("both");
    }, 8000);

    // 2) ou après scroll 60% (si pas déjà montré)
    let opened = false;
    function onScroll() {
      if (opened) return;
      if (isPushAlreadyGranted()) return;
      if (!canShowNow()) return;

      const doc = document.documentElement;
      const scrollTop = doc.scrollTop || document.body.scrollTop || 0;
      const height = (doc.scrollHeight || 1) - (doc.clientHeight || 1);
      if (height <= 0) return;
      const ratio = scrollTop / height;
      if (ratio >= 0.60) {
        opened = true;
        openInvite("both");
        window.removeEventListener("scroll", onScroll);
      }
    }
    window.addEventListener("scroll", onScroll, { passive: true });

    // 3) quand l’utilisateur ferme => “Plus tard” 1 jour
    document.addEventListener("click", (e) => {
      const closer = e.target.closest("[data-modal-close='globalPromptModal']");
      if (!closer) return;

      // simple : fermer => 1 jour
      dismiss(1);
    });
  });
