              console.log(data);
              const obj = JSON.parse(data);
              var amount = [0,0,0,0];

              var whole = 0;

              //Sort object amount
              for (var i = 0; i < 4; i++) {
                  if (obj[i] !== undefined) {
                      amount[obj[i].answer-1] = obj[i].amount;
                  }
              }

              console.log(amount);

              for (var i = 0; i < 4; i++) {
                  //if (obj[i] !== undefined) {
                  if (amount[i] != 0) {
                      whole += parseInt(amount[i]);
                  }
              }

              var percentage = [0,0,0,0];

              for (var i = 0; i < 4; i++) {
                  //if (obj[i] !== undefined) {
                  if (amount[i] != 0) {
                      percentage[i] = amount[i] / whole;
                  }
              }

              // Identify winner (if exists)
              var winner = 0;
              var winnerAmount = 0;

              for (var i = 0; i < 4; i++) {
                  //if (obj[i] !== undefined) {
                  if (amount[i] != 0) {
                      if (parseInt(amount[i]) > winnerAmount) {
                          winnerAmount = parseInt(amount[i]);
                          winner = i;
                      }
                  }
              }

              $("#surveyResult").css("display","block");
              $("#surveyOptions").css("display","none");
              $("#amount").html(' ('+whole+')');

              for (var i = 0; i < 4; i++) {
                  if (amount[i] != 0) {
                      $("#progress-bar-"+(i+1)).css("width", (percentage[i]*100)+"%");
                      if (i == winner) {
                          console.log("Winner = "+i)
                          $("#progress-bar-"+(i+1)).removeClass("bg-secondary");
                          $("#progress-bar-"+(i+1)).css("background-color", "#9C83FF");
                      }
                  }
              }

              setSurveyCookie();
          });
      }
