        $(document).ready(function() {
            $('#lang-select').val('uk').trigger('change');

            $('#lang-select').on('change', function() {
                var lang = $(this).select2('data')[0]['id'];
                if(lang!='uk'){
                    var currentURL = window.location.href.replace(window.location.origin, '');
                    var urlParts = currentURL.split('/').filter(function(part) {
                        return part !== '';
                    });
                    urlParts[0] = lang;
                    window.location.href = '/'+urlParts.join('/')+'/';    
                } 
            });
        });

        document.getElementById("change-lang-button").addEventListener("click", function() {
            var element = document.getElementById("lang-select-div");
            if (element.style.display === "none") {
                element.style.display = "initial";
                $('#lang-select').select2('open');
            } else {
                element.style.display = "none";
            }
        });
    </script>
