    // Configuration pour éviter les erreurs JavaScript
    window.addEventListener('load', function(ev) {
      // Gestion améliorée des erreurs de polices et ressources
      const originalWarn = console.warn;
      const originalError = console.error;

      // Filtrer les warnings de polices et ressources
      console.warn = function(...args) {
        const message = args[0];
        if (typeof message === 'string') {
          // Ignorer les warnings de polices
          if (message.includes('MaterialIcons') ||
              message.includes('font') ||
              message.includes('Noto') ||
              message.includes('Police gérée automatiquement')) {
            console.log('Police gérée automatiquement:', ...args);
            return;
          }
          // Ignorer les warnings de ressources préchargées
          if (message.includes('was preloaded using link preload but not used') ||
              message.includes('preload')) {
            console.log('Ressource préchargée gérée automatiquement');
            return;
          }
        }
        originalWarn.apply(console, args);
      };

      // Gestion des erreurs JavaScript avec filtrage
      window.addEventListener('error', function(e) {
        const message = e.message || '';

        // Ignorer les erreurs de polices
        if (message.includes('MaterialIcons') ||
            message.includes('font') ||
            message.includes('Failed to load font') ||
            message.includes('Could not find a set of Noto fonts')) {
          console.log('Erreur de police gérée automatiquement:', message);
          e.preventDefault();
          return false;
        }

        // Ignorer les erreurs d'icônes du manifest
        if (message.includes('Icon-192.png') ||
            message.includes('Download error or resource isn\'t a valid image')) {
          console.log('Erreur d\'icône gérée automatiquement:', message);
          e.preventDefault();
          return false;
        }

        // Log des autres erreurs
        console.error('Erreur JavaScript:', message, e.filename, e.lineno);
      });

      // Gestion des erreurs de promesse non gérées
      window.addEventListener('unhandledrejection', function(e) {
        const reason = e.reason;
        if (reason && reason.message) {
          const message = reason.message;

          // Ignorer les erreurs de polices dans les promesses
          if (message.includes('font') ||
              message.includes('MaterialIcons') ||
              message.includes('Failed to load font')) {
            console.log('Erreur de police dans promesse gérée automatiquement');
            e.preventDefault();
            return false;
          }
        }

        // Log des autres erreurs de promesse
        console.error('Promesse rejetée non gérée:', reason);
      });

      // Prévention des erreurs de caractères spéciaux
      String.prototype.normalize = String.prototype.normalize || function() {
        return this;
      };

      // Fonction pour vérifier la disponibilité des polices
      window.checkFontAvailability = function() {
        const testElement = document.createElement('span');
        testElement.style.fontFamily = 'MaterialIcons';
        testElement.style.position = 'absolute';
        testElement.style.left = '-9999px';
        testElement.textContent = ''; // Code point pour une icône Material

        document.body.appendChild(testElement);

        setTimeout(() => {
          const hasMaterialIcons = testElement.offsetWidth > 0;
          console.log('Police MaterialIcons disponible:', hasMaterialIcons);

          if (!hasMaterialIcons) {
            console.log('Utilisation de la police de fallback pour les icônes');
            // Appliquer une classe CSS pour utiliser les fallbacks
            document.body.classList.add('material-icons-fallback');
          }

          document.body.removeChild(testElement);
        }, 100);
      };

      // Vérifier les polices après un court délai
      setTimeout(window.checkFontAvailability, 500);
    });

    // Fonction de fallback pour les caractères manquants
    window.getFontFallback = function(char) {
      // Retourner une police système appropriée
      if (char && char.length > 0) {
        const code = char.charCodeAt(0);
        if (code >= 0x1F600 && code <= 0x1F64F) { // Emojis
          return 'Apple Color Emoji, Segoe UI Emoji, Noto Color Emoji';
        }
      }
      return 'Roboto, Arial, sans-serif';
    };
