    <script>
    (function(){
        const menu = document.getElementById('offcanvasNav');
        const backdrop = document.getElementById('menuBackdrop');
        const openBtn = document.getElementById('menuToggle');
        const closeBtn = document.getElementById('menuClose');

        let lastFocus = null;
        const focusable = 'a[href],button:not([disabled]),[tabindex]:not([tabindex="-1"])';

        function lockScroll(lock){
            document.documentElement.style.overflowY = lock ? 'hidden' : '';
            document.body.style.overflowY = lock ? 'hidden' : '';
            document.body.style.touchAction = lock ? 'none' : '';
        }
        function trapFocus(e){
            if(!menu.classList.contains('open')) return;
            if(e.key!=='Tab' && e.key!=='Escape') return;
            const nodes = menu.querySelectorAll(focusable);
            if(!nodes.length) return;
            const first = nodes[0], last = nodes[nodes.length-1];
            if(e.key==='Tab'){
                if(e.shiftKey && document.activeElement===first){ e.preventDefault(); last.focus(); }
                else if(!e.shiftKey && document.activeElement===last){ e.preventDefault(); first.focus(); }
            } else if(e.key==='Escape'){ closeMenu(); }
        }
        function openMenu(){
            lastFocus = document.activeElement;
            menu.hidden = false; backdrop.hidden = false;
            requestAnimationFrame(()=>{
                menu.classList.add('open');
                backdrop.classList.add('show');
                lockScroll(true);
                openBtn.setAttribute('aria-expanded','true');
                // фокус на перший пункт меню
                const firstLink = menu.querySelector('.menu-list a');
                (firstLink || menu).focus();
            });
            document.addEventListener('keydown', trapFocus);
        }
        function closeMenu(){
            menu.classList.remove('open');
            backdrop.classList.remove('show');
            lockScroll(false);
            openBtn.setAttribute('aria-expanded','false');
            document.removeEventListener('keydown', trapFocus);
            setTimeout(()=>{ menu.hidden = true; backdrop.hidden = true; lastFocus?.focus?.(); }, 280);
        }

        openBtn.addEventListener('click', openMenu);
        closeBtn?.addEventListener('click', closeMenu);
        backdrop.addEventListener('click', closeMenu);
        document.addEventListener('keydown', e => { if(e.key==='Escape' && menu.classList.contains('open')) closeMenu(); });

        menu.addEventListener('click', e => {
            const a = e.target.closest('a[href]');
            if(a) closeMenu();
        });

        const BP = 1024; let wasDesktop = window.innerWidth >= BP;
        window.addEventListener('resize', () => {
            const isDesktop = window.innerWidth >= BP;
            if (isDesktop && !wasDesktop) { closeMenu(); }
            wasDesktop = isDesktop;
        });
    })();
</script>
