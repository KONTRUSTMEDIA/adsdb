// Marathi weekday map
const marathiDays = {
  "Sunday":"रविवार","Monday":"सोमवार","Tuesday":"मंगळवार",
  "Wednesday":"बुधवार","Thursday":"गुरुवार","Friday":"शुक्रवार","Saturday":"शनिवार"
};

document.getElementById('today').value = new Date().toISOString().split('T')[0];
document.getElementById('dob').value = "1990-01-01";

document.getElementById('clearBtn').addEventListener('click', () => {
  document.getElementById('today').value = new Date().toISOString().split('T')[0];
  document.getElementById('dob').value = "1990-01-01";
  document.getElementById('result').innerHTML = "";
});

document.getElementById('ageForm').addEventListener('submit', e => {
  e.preventDefault();

  let todayVal = document.getElementById('today').value;
  let dobVal   = document.getElementById('dob').value;
  if (!todayVal || !dobVal) return;

  let today = new Date(todayVal);
  let dob   = new Date(dobVal);

  if (dob > today) {
    document.getElementById('result').innerHTML = `<div class="card error">जन्मतारीख भविष्यात असू शकत नाही!</div>`;
    return;
  }

  // age calc
  let ageY = today.getFullYear() - dob.getFullYear();
  let ageM = today.getMonth() - dob.getMonth();
  let ageD = today.getDate() - dob.getDate();

  if (ageD < 0) {
    ageM--; 
    let prevMonth = new Date(today.getFullYear(), today.getMonth(), 0);
    ageD += prevMonth.getDate();
  }
  if (ageM < 0) {
    ageY--;
    ageM += 12;
  }

  // next birthday
  let nextBday = new Date(today.getFullYear(), dob.getMonth(), dob.getDate());
  if (nextBday <= today) nextBday.setFullYear(today.getFullYear()+1);

  let nbDiff = nextBday - today;
  let nbDays = Math.floor(nbDiff / (1000*60*60*24));
  let nbM = Math.floor(nbDays/30); // approx months
  let nbD = nbDays % 30;

  // weekdays
  let todayDay = marathiDays[today.toLocaleDateString("en-US",{weekday:"long"})];
  let dobDay   = marathiDays[dob.toLocaleDateString("en-US",{weekday:"long"})];
  let nbDay    = marathiDays[nextBday.toLocaleDateString("en-US",{weekday:"long"})];

  // date formatting dd-mm-yyyy
  let dobFormatted = dob.toLocaleDateString("en-GB"); 
  let nextBdayFormatted = nextBday.toLocaleDateString("en-GB");

  // show result


  document.getElementById('result').innerHTML = `
    <div class="card">
      <h2>आज: ${todayDay}</h2>
      <div class="pill">
        <div class="mini"><h3>YEARS</h3><div class="val">${ageY}</div></div>
        <div class="mini"><h3>MONTHS</h3><div class="val">${ageM}</div></div>
        <div class="mini"><h3>DAYS</h3><div class="val">${ageD}</div></div>
      </div>
      <div class="muted">जन्म तारीख : (${dobFormatted}) ${dobDay}</div>
    </div>

    <div class="card">
      <h2>पुढील वाढदिवस: ${nextBdayFormatted} (${nbDay})</h2>
      <div class="grid2">
        <div class="mini"><h3>MONTHS</h3><div class="val">${nbM}</div></div>
        <div class="mini"><h3>DAYS</h3><div class="val">${nbD}</div></div>
      </div>
    </div>
  `;
});
