  <script>
    /**
     * Christmas logo swapper (client clock, Dec 1–31).
     * - Swaps normal logo -> christmas logo in:
     *   • <img src="...">
     *   • <link rel="icon" href="..."> (and apple-touch-icon if present)
     *   • <script type="application/ld+json"> image field
     * - Observes DOM for late-added elements (Angular, lazy content).
     */
    (function christmasLogoSwap() {
      // ---- configuration -------------------------------------------------------
      const NORMAL_LOGO_FILENAME = "fast_logo_voe.png";
      const XMAS_LOGO_FILENAME = "fast_logo_christmas.png";

      // absolute URLs (used for JSON-LD or when you prefer to set absolute)
      const NORMAL_LOGO_URL = "https://fast.farming-community.eu/assets/logos/" + NORMAL_LOGO_FILENAME;
      const XMAS_LOGO_URL = "https://fast.farming-community.eu/assets/logos/" + XMAS_LOGO_FILENAME;

      // ---- helpers -------------------------------------------------------------
      function isChristmasSeason(now = new Date()) {
        // 0=Jan ... 11=Dec
        return now.getMonth() === 11; // Dec 1–31
      }

      function swapFilename(pathOrUrl, oldName, newName) {
        return typeof pathOrUrl === "string" && pathOrUrl.includes(oldName)
          ? pathOrUrl.replace(oldName, newName)
          : pathOrUrl;
      }

      function updateImgElement(img) {
        if (!img || typeof img.getAttribute !== "function") return;
        const current = img.getAttribute("src"); // preserves relative URLs
        const swapped = swapFilename(current, NORMAL_LOGO_FILENAME, XMAS_LOGO_FILENAME);
        if (swapped && swapped !== current) img.setAttribute("src", swapped);
      }

      function updateFavicons() {
        // rel~="icon" matches "icon", "shortcut icon", etc.
        const iconSelectors = [
          'link[rel~="icon"]',
          'link[rel="apple-touch-icon"]',
          'link[rel="apple-touch-icon-precomposed"]',
        ];
        document.querySelectorAll(iconSelectors.join(",")).forEach(link => {
          const href = link.getAttribute("href");
          const swapped = swapFilename(href, NORMAL_LOGO_FILENAME, XMAS_LOGO_FILENAME);
          if (swapped && swapped !== href) link.setAttribute("href", swapped);
        });
      }

      function updateJsonLdImage() {
        const node = document.querySelector('script[type="application/ld+json"]');
        if (!node || !node.textContent) return;

        try {
          const data = JSON.parse(node.textContent);

          // Mutates "image" if it contains the normal logo (string or array of strings)
          if (typeof data === "object" && data) {
            if (typeof data.image === "string" && data.image.includes(NORMAL_LOGO_URL)) {
              data.image = XMAS_LOGO_URL;
            } else if (Array.isArray(data.image)) {
              data.image = data.image.map(v =>
                (typeof v === "string" && v.includes(NORMAL_LOGO_URL)) ? XMAS_LOGO_URL : v
              );
            }
          }

          node.textContent = JSON.stringify(data);
        } catch {
        }
      }

      function initialSwap() {
        updateFavicons();
        updateJsonLdImage();
        document.querySelectorAll("img").forEach(updateImgElement);
      }

      function observeDomForLateElements() {
        const observer = new MutationObserver(mutations => {
          for (const m of mutations) {
            // handle direct <img> additions
            m.addedNodes.forEach(node => {
              if (!(node instanceof Element)) return;

              if (node.tagName === "IMG") {
                updateImgElement(node);
              } else {
                // handle <img> nodes in added subtrees
                node.querySelectorAll?.("img").forEach(updateImgElement);
              }

              // if favicon or json-ld are reinserted/changed
              if (node.matches?.('link[rel~="icon"], link[rel="apple-touch-icon"], link[rel="apple-touch-icon-precomposed"]')) {
                updateFavicons();
              }
              if (node.matches?.('script[type="application/ld+json"]')) {
                updateJsonLdImage();
              }
            });
          }
        });

        observer.observe(document.documentElement, { childList: true, subtree: true });
      }

      // ---- run (only in December) ----
      if (!isChristmasSeason()) return;
      initialSwap();
      observeDomForLateElements();
    })();
  </script>
